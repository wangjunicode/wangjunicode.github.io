## 一、冷却管理器

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 技能冷却状态
/// </summary>
public class CooldownState
{
    public string SkillId;
    public float TotalDuration;    // 总冷却时间（应用减冷后）
    public float Remaining;        // 剩余冷却时间
    public int MaxCharges;         // 最大充能数（1=普通技能）
    public int CurrentCharges;     // 当前充能数
    public float ChargeProgress;   // 当前充能进度（0-1）
    
    public bool IsReady => CurrentCharges > 0;
    public float Progress => TotalDuration > 0 ? 1f - Remaining / TotalDuration : 1f;
}

/// <summary>
/// 技能冷却管理器
/// </summary>
public class CooldownManager : MonoBehaviour
{
    // 全局冷却（GCD）：使用任意技能后触发，所有技能短暂无法使用
    private float globalCooldownDuration = 1f;
    private float globalCooldownRemaining;
    
    public bool IsGCDActive => globalCooldownRemaining > 0;
    
    // 技能独立冷却
    private Dictionary<string, CooldownState> cooldowns 
        = new Dictionary<string, CooldownState>();
    
    // 冷却减少属性（百分比，如 0.2 = 减少20%冷却时间）
    private float cooldownReduction;
    
    // 事件
    public event Action<string, CooldownState> OnCooldownStarted;
    public event Action<string> OnCooldownFinished;
    public event Action OnGCDStarted;
    public event Action OnGCDFinished;

    void Update()
    {
        float dt = Time.deltaTime;
        
        // 更新GCD
        if (globalCooldownRemaining > 0)
        {
            globalCooldownRemaining -= dt;
            if (globalCooldownRemaining <= 0)
            {
                globalCooldownRemaining = 0;
                OnGCDFinished?.Invoke();
            }
        }
        
        // 更新各技能冷却
        var toRemove = new List<string>();
        
        foreach (var kv in cooldowns)
        {
            var state = kv.Value;
            
            if (state.CurrentCharges < state.MaxCharges)
            {
                state.Remaining -= dt;
                state.ChargeProgress = 1f - state.Remaining / state.TotalDuration;
                
                if (state.Remaining <= 0)
                {
                    state.CurrentCharges++;
                    
                    if (state.CurrentCharges < state.MaxCharges)
                    {
                        // 还有充能需要恢复，重置冷却
                        state.Remaining = state.TotalDuration;
                        state.ChargeProgress = 0;
                    }
                    else
                    {
                        // 充能已满
                        state.Remaining = 0;
                        state.ChargeProgress = 1f;
                        OnCooldownFinished?.Invoke(kv.Key);
                    }
                }
            }
        }
    }

    /// <summary>
    /// 使用技能（检查并消耗冷却）
    /// </summary>
    public bool UseSkill(string skillId, float baseCooldown, 
        bool triggersGCD = true, int maxCharges = 1)
    {
        // 检查GCD
        if (IsGCDActive)
        {
            Debug.Log("[Cooldown] 全局冷却中");
            return false;
        }
        
        // 检查技能冷却
        if (cooldowns.TryGetValue(skillId, out var state))
        {
            if (state.CurrentCharges <= 0)
            {
                Debug.Log($"[Cooldown] {skillId} 冷却中，剩余 {state.Remaining:F1}s");
                return false;
            }
            
            // 消耗充能
            state.CurrentCharges--;
            
            // 如果是从满充能变为非满，开始计时
            if (state.CurrentCharges < state.MaxCharges && state.Remaining <= 0)
            {
                state.Remaining = state.TotalDuration;
                state.ChargeProgress = 0;
            }
        }
        else
        {
            // 首次使用，创建冷却状态
            float adjustedCD = ApplyCDReduction(baseCooldown);
            var newState = new CooldownState
            {
                SkillId = skillId,
                TotalDuration = adjustedCD,
                Remaining = adjustedCD,
                MaxCharges = maxCharges,
                CurrentCharges = maxCharges - 1, // 使用后减1
                ChargeProgress = 0
            };
            cooldowns[skillId] = newState;
            OnCooldownStarted?.Invoke(skillId, newState);
        }
        
        // 触发GCD
        if (triggersGCD)
        {
            globalCooldownRemaining = ApplyCDReduction(globalCooldownDuration);
            OnGCDStarted?.Invoke();
        }
        
        return true;
    }

    /// <summary>
    /// 强制重置技能冷却（GM命令/特殊BUFF）
    /// </summary>
    public void ResetCooldown(string skillId)
    {
        if (cooldowns.TryGetValue(skillId, out var state))
        {
            state.CurrentCharges = state.MaxCharges;
            state.Remaining = 0;
            state.ChargeProgress = 1f;
        }
    }

    public void ResetAllCooldowns()
    {
        foreach (var state in cooldowns.Values)
        {
            state.CurrentCharges = state.MaxCharges;
            state.Remaining = 0;
        }
        globalCooldownRemaining = 0;
    }

    /// <summary>
    /// 应用冷却减少
    /// </summary>
    float ApplyCDReduction(float baseCooldown)
    {
        return baseCooldown * (1f - Mathf.Clamp(cooldownReduction, 0f, 0.8f)); // 上限80%减冷
    }

    public void SetCooldownReduction(float reduction)
    {
        cooldownReduction = reduction;
    }

    public CooldownState GetCooldownState(string skillId)
    {
        return cooldowns.TryGetValue(skillId, out var state) ? state : null;
    }
    
    public float GetGCDProgress() =>
        globalCooldownDuration > 0 ? 
            1f - globalCooldownRemaining / globalCooldownDuration : 1f;
}
```

## 二、冷却UI组件

```csharp
/// <summary>
/// 技能冷却UI（弧形遮罩显示冷却进度）
/// </summary>
public class SkillCooldownUI : MonoBehaviour
{
    [SerializeField] private string skillId;
    [SerializeField] private UnityEngine.UI.Image iconImage;
    [SerializeField] private UnityEngine.UI.Image cooldownMask;  // 弧形遮罩（Filled类型）
    [SerializeField] private UnityEngine.UI.Text cooldownText;
    [SerializeField] private GameObject readyIndicator;          // 就绪闪光效果
    
    private CooldownManager cooldownManager;
    private bool wasOnCooldown;

    void Start()
    {
        cooldownManager = GetComponentInParent<CooldownManager>() 
            ?? FindObjectOfType<CooldownManager>();
    }

    void Update()
    {
        var state = cooldownManager?.GetCooldownState(skillId);
        
        if (state == null || state.IsReady)
        {
            // 技能就绪
            if (cooldownMask != null) cooldownMask.fillAmount = 0;
            if (iconImage != null) iconImage.color = Color.white;
            if (cooldownText != null) cooldownText.text = "";
            
            // 如果刚从冷却变就绪，播放就绪动画
            if (wasOnCooldown)
            {
                wasOnCooldown = false;
                PlayReadyEffect();
            }
        }
        else
        {
            // 技能冷却中
            float remaining = state.Remaining;
            float progress = state.Progress;
            
            // 更新弧形遮罩（从1到0）
            if (cooldownMask != null)
                cooldownMask.fillAmount = 1f - progress;
            
            // 图标变暗
            if (iconImage != null)
                iconImage.color = new Color(0.5f, 0.5f, 0.5f, 1f);
            
            // 剩余时间文本
            if (cooldownText != null)
                cooldownText.text = remaining > 1f ? 
                    $"{Mathf.CeilToInt(remaining)}" : 
                    $"{remaining:F1}";
            
            wasOnCooldown = true;
        }
    }

    void PlayReadyEffect()
    {
        // 播放就绪特效（闪光/音效）
        if (readyIndicator != null)
        {
            readyIndicator.SetActive(true);
            Invoke(nameof(HideReadyIndicator), 0.5f);
        }
    }

    void HideReadyIndicator()
    {
        if (readyIndicator != null)
            readyIndicator.SetActive(false);
    }
}
```

## 三、离线冷却计算

```csharp
/// <summary>
/// 离线冷却（游戏关闭期间冷却仍然计算）
/// </summary>
public class OfflineCooldownCalculator
{
    public static void ApplyOfflineCooldownReduction(CooldownManager manager)
    {
        long lastSaveTime = PlayerPrefs.GetInt("last_session_end_timestamp", 0);
        if (lastSaveTime == 0) return;
        
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        float offlineSeconds = Mathf.Min(now - lastSaveTime, 3600f); // 最多计算1小时离线
        
        if (offlineSeconds <= 0) return;
        
        Debug.Log($"[OfflineCD] 离线 {offlineSeconds:F0} 秒，应用冷却减少");
        
        // 对所有技能应用离线时间
        foreach (var state in GetAllCooldownStates(manager))
        {
            if (!state.IsReady)
            {
                state.Remaining = Mathf.Max(0, state.Remaining - offlineSeconds);
                if (state.Remaining <= 0)
                {
                    state.CurrentCharges = state.MaxCharges;
                    state.Remaining = 0;
                }
            }
        }
    }
    
    static IEnumerable<CooldownState> GetAllCooldownStates(CooldownManager manager)
    {
        yield break; // 需要暴露 CooldownManager 的内部状态
    }

    public static void RecordSessionEnd()
    {
        PlayerPrefs.SetInt("last_session_end_timestamp",
            (int)DateTimeOffset.UtcNow.ToUnixTimeSeconds());
        PlayerPrefs.Save();
    }
}
```

## 四、冷却系统设计要点

| 要点 | 方案 |
|------|------|
| GCD + 独立CD | 并行管理，互不干扰 |
| 充能技能 | 独立充能计数器，充能满前持续计时 |
| 冷却减少上限 | 限制80%，防止0秒CD |
| 离线冷却 | 记录下线时间，上线时减去离线时长 |
| 网络同步 | 服务端权威CD，客户端做预测显示 |
| UI闪光反馈 | 冷却结束时给明显的视觉+声音提示 |
