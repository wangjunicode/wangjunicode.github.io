using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 物品配置（ScriptableObject，所有类型物品共用）

[CreateAssetMenu(fileName = "Item", menuName = "Game/Item")]
public class ItemData : ScriptableObject
{
    [Header("基础信息")]
    public string ItemId;
    public string DisplayName;
    [TextArea(2, 4)] public string Description;
    public Sprite Icon;
    public ItemType Type;
    public ItemRarity Rarity;

    public bool IsStackable;

    public EquipmentSlotType EquipSlot;
    public List<StatModifier> StatModifiers; // 装备时的属性加成
    
    [Header("消耗品配置（Type=Consumable时）")]
    public List<ConsumableEffect> Effects;
    
    [Header("价值")]
    public long BuyPrice;       // NPC购买价
    public long SellPrice;      // 出售给NPC价格
    public bool CanSell = true;
    public bool CanTrade = true;
    public bool IsBound;        // 绑定物品
}

public enum ItemType { Consumable, Equipment, Material, QuestItem, Currency, Special }
public enum ItemRarity { Common, Uncommon, Rare, Epic, Legendary }
public enum EquipmentSlotType { Weapon, OffHand, Head, Chest, Legs, Feet, Neck, Ring1, Ring2 }

[Serializable]
public class StatModifier
{
    public string StatId;
    public StatModifierOrder Order;
    public float Value;
}

[Serializable]
public class ConsumableEffect
{
    public string EffectType;  // "heal_hp", "buff_atk", etc.
    public float Value;
    public float Duration;     // 0=即时，>0=持续时间

    public int EnhanceLevel;   // 强化等级
    public List<string> Enchantments; // 附魔
}

## 二、背包系统核心

```csharp
/// <summary>
/// 背包系统
/// </summary>
public class InventorySystem : MonoBehaviour
{
    [Header("容量")]
    [SerializeField] private int initialCapacity = 30;
    [SerializeField] private int expandPerPurchase = 5;

    private List<InventoryItem?> slots; // null表示空格子
    private int capacity;
    
    private Dictionary<string, ItemData> itemDatabase;
    
    public event Action<InventoryItem, int> OnItemAdded;        // item, slotIndex
    public event Action<int, int> OnItemRemoved;                 // slotIndex, remainCount
    public event Action<int, int> OnItemMoved;                   // fromSlot, toSlot

    void Awake()
    {
        capacity = initialCapacity;
        slots = new List<InventoryItem?>(new InventoryItem?[capacity]);
    }

    /// <summary>
    /// 检查是否有空间存放物品
    /// </summary>
    public bool HasSpace(ItemData item, int count = 1)
    {
        if (item.IsStackable)
        {
            // 检查现有堆叠是否还有空间
            int remaining = count;
            foreach (var slot in slots)
            {
                if (slot?.ItemId == item.ItemId)
                    remaining -= (item.MaxStackSize - slot.Count);
                if (remaining <= 0) return true;
            }
            // 还需要新格子
            return remaining <= 0 || GetEmptySlotCount() > 0;
        }
        return GetEmptySlotCount() >= count;
    }

    /// <summary>
    /// 添加物品
    /// </summary>
    public bool AddItem(ItemData item, int count = 1)
    {
        if (!HasSpace(item, count)) return false;
        
        int remaining = count;
        
        // 先尝试堆叠到已有格子
        if (item.IsStackable)
        {
            for (int i = 0; i < slots.Count && remaining > 0; i++)
            {
                if (slots[i]?.ItemId == item.ItemId)
                {
                    int canAdd = item.MaxStackSize - slots[i].Count;
                    int adding = Mathf.Min(remaining, canAdd);
                    slots[i].Count += adding;
                    remaining -= adding;
                    OnItemAdded?.Invoke(slots[i], i);
                }
            }
        }
        
        // 放入空格子
        while (remaining > 0)
        {
            int emptySlot = GetFirstEmptySlot();
            if (emptySlot < 0) break;
            
            int stackCount = item.IsStackable 
                ? Mathf.Min(remaining, item.MaxStackSize) 
                : 1;
            
            var newItem = new InventoryItem
            {
                InstanceId = Guid.NewGuid().ToString("N").Substring(0, 8),
                ItemId = item.ItemId,
                Count = stackCount,
                SlotIndex = emptySlot
            };
            
            slots[emptySlot] = newItem;
            remaining -= stackCount;
            
            OnItemAdded?.Invoke(newItem, emptySlot);
        }
        
        return remaining <= 0;
    }

    /// <summary>
    /// 移除物品
    /// </summary>
    public bool RemoveItem(string itemId, int count = 1)
    {
        int remaining = count;
        
        for (int i = 0; i < slots.Count && remaining > 0; i++)
        {
            if (slots[i]?.ItemId == itemId)
            {
                int removing = Mathf.Min(remaining, slots[i].Count);
                slots[i].Count -= removing;
                remaining -= removing;
                
                if (slots[i].Count <= 0)
                {
                    slots[i] = null;
                    OnItemRemoved?.Invoke(i, 0);
                }
                else
                {
                    OnItemRemoved?.Invoke(i, slots[i].Count);
                }
            }
        }
        
        return remaining <= 0;
    }

    /// <summary>
    /// 移动物品（格子拖拽）
    /// </summary>
    public bool MoveItem(int fromSlot, int toSlot)
    {
        if (fromSlot < 0 || fromSlot >= slots.Count) return false;
        if (toSlot < 0 || toSlot >= slots.Count) return false;
        
        var from = slots[fromSlot];
        var to = slots[toSlot];
        
        // 如果目标槽有同类可堆叠物品，则堆叠
        if (from != null && to != null && from.ItemId == to.ItemId)
        {
            var itemData = GetItemData(from.ItemId);
            if (itemData?.IsStackable == true)
            {
                int canAdd = itemData.MaxStackSize - to.Count;
                int moving = Mathf.Min(from.Count, canAdd);
                to.Count += moving;
                from.Count -= moving;
                
                if (from.Count <= 0)
                    slots[fromSlot] = null;
                
                OnItemMoved?.Invoke(fromSlot, toSlot);
                return true;
            }
        }
        
        // 交换两个格子
        slots[fromSlot] = to;
        slots[toSlot] = from;
        
        if (from != null) from.SlotIndex = toSlot;
        if (to != null) to.SlotIndex = fromSlot;
        
        OnItemMoved?.Invoke(fromSlot, toSlot);
        return true;
    }

    public bool HasItem(string itemId, int count = 1)
    {
        int total = 0;
        foreach (var slot in slots)
            if (slot?.ItemId == itemId) total += slot.Count;
        return total >= count;
    }

    public InventoryItem GetItem(int slotIndex) =>
        slotIndex >= 0 && slotIndex < slots.Count ? slots[slotIndex] : null;

    int GetFirstEmptySlot()
    {
        for (int i = 0; i < slots.Count; i++)
            if (slots[i] == null) return i;
        return -1;
    }

    int GetEmptySlotCount()
    {
        int count = 0;
        foreach (var slot in slots)
            if (slot == null) count++;
        return count;
    }

    public void ExpandCapacity()
    {
        capacity += expandPerPurchase;
        for (int i = 0; i < expandPerPurchase; i++)
            slots.Add(null);
    }

    ItemData GetItemData(string itemId) =>
        itemDatabase.TryGetValue(itemId, out var data) ? data : null;
}
```

## 三、装备槽系统

```csharp
/// <summary>
/// 装备槽管理器
/// </summary>
public class EquipmentSystem : MonoBehaviour
{
    private Dictionary<EquipmentSlotType, InventoryItem> equipped 
        = new Dictionary<EquipmentSlotType, InventoryItem>();
    
    [SerializeField] private CharacterAttributes characterAttributes;
    
    public event Action<EquipmentSlotType, InventoryItem> OnEquipped;
    public event Action<EquipmentSlotType> OnUnequipped;

    public bool Equip(InventoryItem item, ItemData data)
    {
        if (data.Type != ItemType.Equipment) return false;
        
        var slot = data.EquipSlot;
        
        // 卸下原有装备
        if (equipped.ContainsKey(slot))
            Unequip(slot);
        
        equipped[slot] = item;
        
        // 应用属性
        if (characterAttributes != null && data.StatModifiers != null)
        {
            characterAttributes.ApplyEquipment(item.InstanceId, data.StatModifiers);
        }
        
        OnEquipped?.Invoke(slot, item);
        return true;
    }

    public bool Unequip(EquipmentSlotType slot)
    {
        if (!equipped.TryGetValue(slot, out var item)) return false;
        
        equipped.Remove(slot);
        
        // 移除属性
        characterAttributes?.RemoveEquipment(item.InstanceId);
        
        OnUnequipped?.Invoke(slot);
        return true;
    }

    public InventoryItem GetEquipped(EquipmentSlotType slot) =>
        equipped.TryGetValue(slot, out var item) ? item : null;
}
```

## 四、背包设计要点

| 要点 | 方案 |
|------|------|
| 堆叠规则 | 消耗品/材料可堆叠，装备/任务物品不堆叠 |
| 客户端乐观更新 | 操作立即反映在UI，服务端失败时回滚 |
| 格子拖拽 | IBeginDragHandler+IDropHandler接口实现 |
| 物品筛选 | 按类型/稀有度/名称快速筛选 |
| 背包排序 | 按稀有度/类型/时间自动整理 |
| 绑定物品 | 拾取绑定/装备绑定，不可交易/出售 |
