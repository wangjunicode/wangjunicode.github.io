## 一、战令数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class BattlePassLevel
{
    public int Level;
    public int RequiredExp;      // 到达该级需要的累计经验
    public BattlePassReward FreeReward;
    public BattlePassReward PremiumReward;  // 付费轨道奖励
}

[Serializable]
public class BattlePassReward
{
    public string ItemId;
    public int Count;
    public RewardDisplayType DisplayType;  // 普通/高亮/限定
}

public enum RewardDisplayType { Normal, Highlighted, Limited }

[Serializable]
public class BattlePassData
{
    public string SeasonId;
    public string SeasonName;
    public long SeasonEndTime;
    public int MaxLevel;
    public BattlePassLevel[] Levels;
    public int PremiumPrice;  // 购买付费战令的硬通货价格
    
    public int GetLevelForExp(int totalExp)
    {
        for (int i = Levels.Length - 1; i >= 0; i--)
            if (totalExp >= Levels[i].RequiredExp) return Levels[i].Level;
        return 0;
    }
}

[Serializable]
public class BattlePassPlayerState
{
    public bool HasPremium;
    public int CurrentExp;
    public int CurrentLevel;
    public List<int> ClaimedFreeLevels = new List<int>();
    public List<int> ClaimedPremiumLevels = new List<int>();
}
```

## 二、战令管理器

```csharp
public class BattlePassManager : MonoBehaviour
{
    private static BattlePassManager instance;
    public static BattlePassManager Instance => instance;
    
    private BattlePassData seasonData;
    private BattlePassPlayerState playerState;
    
    public event Action OnStateUpdated;
    public event Action<int> OnLevelUp;   // 升级事件
    public event Action<int, bool> OnRewardClaimed; // 等级, 是否是付费

    void Awake() { instance = this; }

    public int GetCurrentLevel() => playerState?.CurrentLevel ?? 0;
    public int GetCurrentExp() => playerState?.CurrentExp ?? 0;
    public bool HasPremium() => playerState?.HasPremium ?? false;

    public float GetLevelProgress()
    {
        if (playerState == null || seasonData == null) return 0;
        
        int curLevel = playerState.CurrentLevel;
        if (curLevel >= seasonData.MaxLevel) return 1f;
        
        int curLevelExp = seasonData.Levels[curLevel].RequiredExp;
        int nextLevelExp = curLevel + 1 < seasonData.Levels.Length ? 
            seasonData.Levels[curLevel + 1].RequiredExp : curLevelExp;
        
        return (float)(playerState.CurrentExp - curLevelExp) / (nextLevelExp - curLevelExp);
    }

    public void AddExp(int exp, string source)
    {
        if (playerState == null) return;
        
        int oldLevel = playerState.CurrentLevel;
        playerState.CurrentExp += exp;
        playerState.CurrentLevel = seasonData.GetLevelForExp(playerState.CurrentExp);
        
        if (playerState.CurrentLevel > oldLevel)
        {
            OnLevelUp?.Invoke(playerState.CurrentLevel);
            // 检查是否有新的可领奖励
            UIManager.Instance?.ShowBattlePassLevelUp(playerState.CurrentLevel);
        }
        
        OnStateUpdated?.Invoke();
        
        // 同步到服务端
        NetworkService.UpdateBattlePassExp(seasonData.SeasonId, playerState.CurrentExp);
    }

    public bool CanClaimReward(int level, bool isPremium)
    {
        if (isPremium && !HasPremium()) return false;
        if (level > playerState.CurrentLevel) return false;
        
        var claimed = isPremium ? playerState.ClaimedPremiumLevels 
                                : playerState.ClaimedFreeLevels;
        return !claimed.Contains(level);
    }

    public async System.Threading.Tasks.Task<bool> ClaimReward(int level, bool isPremium)
    {
        if (!CanClaimReward(level, isPremium)) return false;
        
        bool success = await NetworkService.ClaimBattlePassReward(
            seasonData.SeasonId, level, isPremium);
        
        if (success)
        {
            var claimed = isPremium ? playerState.ClaimedPremiumLevels 
                                    : playerState.ClaimedFreeLevels;
            claimed.Add(level);
            
            OnRewardClaimed?.Invoke(level, isPremium);
            OnStateUpdated?.Invoke();
        }
        
        return success;
    }

    public async System.Threading.Tasks.Task ClaimAllAvailableRewards()
    {
        for (int i = 0; i <= playerState.CurrentLevel; i++)
        {
            if (CanClaimReward(i, false))
                await ClaimReward(i, false);
            if (HasPremium() && CanClaimReward(i, true))
                await ClaimReward(i, true);
        }
    }

    public int GetUnclaimedCount()
    {
        int count = 0;
        for (int i = 0; i <= playerState.CurrentLevel; i++)
        {
            if (CanClaimReward(i, false)) count++;
            if (HasPremium() && CanClaimReward(i, true)) count++;
        }
        return count;
    }

    public TimeSpan GetTimeRemaining()
    {
        var endTime = DateTimeOffset.FromUnixTimeSeconds(seasonData.SeasonEndTime);
        return endTime - DateTimeOffset.UtcNow;
    }
}
```

## 三、战令设计规范

| 要素 | 设计建议 |
|------|----------|
| 赛季周期 | 6-8周（手游）/ 3个月（主机）|
| 等级数量 | 50-100级（提供足够进度感）|
| 经验来源 | 每日任务/挑战/签到/对战 |
| 免费奖励 | 吸引力足够，让玩家有动力参与 |
| 付费奖励 | 外观为主，避免付费破坏平衡 |
| 限时感 | 倒计时显示，最后一周提醒冲级 |
| 买级 | 补级上限（如最多买50级）|
| 每日上限 | 经验每日上限防止全勤玩家和外挂 |
