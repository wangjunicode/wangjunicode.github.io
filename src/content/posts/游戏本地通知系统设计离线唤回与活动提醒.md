## 一、Unity Mobile Notifications 集成

```csharp
using System;
using UnityEngine;
#if UNITY_IOS
using Unity.Notifications.iOS;
#endif
#if UNITY_ANDROID
using Unity.Notifications.Android;
#endif

/// <summary>
/// 本地通知管理器
/// </summary>
public class LocalNotificationManager : MonoBehaviour
{
    private static LocalNotificationManager instance;
    public static LocalNotificationManager Instance => instance;

    private const string CHANNEL_ID = "game_notifications";
    private const string CHANNEL_NAME = "游戏通知";

    void Awake()
    {
        instance = this;
        InitializeNotifications();
    }

    void InitializeNotifications()
    {
#if UNITY_ANDROID
        // 创建通知渠道（Android 8.0+必须）
        var channel = new AndroidNotificationChannel
        {
            Id = CHANNEL_ID,
            Name = CHANNEL_NAME,
            Importance = Importance.Default,
            Description = "游戏提醒通知",
            EnableVibration = true,
            LockScreenVisibility = LockScreenVisibility.Public
        };
        AndroidNotificationCenter.RegisterNotificationChannel(channel);
#endif
    }

    /// <summary>
    /// 请求通知权限（选择最佳时机调用）
    /// </summary>
    public async System.Threading.Tasks.Task<bool> RequestPermission()
    {
#if UNITY_IOS
        var authRequest = new AuthorizationRequest(
            AuthorizationOption.Alert | AuthorizationOption.Badge | AuthorizationOption.Sound,
            true);
        
        using (authRequest)
        {
            await authRequest;
            return authRequest.IsError ? false : authRequest.Granted;
        }
#elif UNITY_ANDROID
        // Android 13+ 需要明确请求权限
        if (UnityEngine.Android.Permission.HasUserAuthorizedPermission(
            "android.permission.POST_NOTIFICATIONS"))
            return true;
        
        UnityEngine.Android.Permission.RequestUserPermission(
            "android.permission.POST_NOTIFICATIONS");
        return true; // Android 13以下默认允许
#else
        return true;
#endif
    }

    /// <summary>
    /// 发送延迟本地通知
    /// </summary>
    public int ScheduleNotification(string title, string body, 
        TimeSpan delay, string data = "")
    {
        return ScheduleNotificationAt(title, body, DateTime.Now + delay, data);
    }

    public int ScheduleNotificationAt(string title, string body, 
        DateTime triggerTime, string data = "")
    {
        if (triggerTime <= DateTime.Now) return -1;

#if UNITY_ANDROID
        var notification = new AndroidNotification
        {
            Title = title,
            Text = body,
            FireTime = triggerTime,
            IntentData = data,
            SmallIcon = "icon_small",
            LargeIcon = "icon_large",
            Style = NotificationStyle.BigTextStyle,
            Color = new Color(0.2f, 0.6f, 1f, 1f)
        };
        return AndroidNotificationCenter.SendNotification(notification, CHANNEL_ID);
        
#elif UNITY_IOS
        var notification = new iOSNotification
        {
            Title = title,
            Body = body,
            Subtitle = "《你的游戏》",
            Data = data,
            Badge = 1,
            ShowInForeground = false,
            Trigger = new iOSNotificationTimeIntervalTrigger
            {
                TimeInterval = triggerTime - DateTime.Now,
                Repeats = false
            }
        };
        iOSNotificationCenter.ScheduleNotification(notification);
        return 0;
#else
        return -1;
#endif
    }

    public void CancelAllNotifications()
    {
#if UNITY_ANDROID
        AndroidNotificationCenter.CancelAllNotifications();
#elif UNITY_IOS
        iOSNotificationCenter.RemoveAllScheduledNotifications();
#endif
    }

    public void CancelNotification(int id)
    {
#if UNITY_ANDROID
        AndroidNotificationCenter.CancelNotification(id);
#endif
    }
}
```

## 二、游戏场景通知策略

```csharp
/// <summary>
/// 游戏通知场景管理器
/// </summary>
public class GameNotificationScheduler : MonoBehaviour
{
    [Header("通知配置")]
    [SerializeField] private bool enableOfflineReward = true;    // 离线奖励通知
    [SerializeField] private int offlineRewardHours = 4;         // 离线多久后提醒
    [SerializeField] private bool enableEventReminder = true;    // 活动提醒
    [SerializeField] private bool enableEnergyFull = true;       // 体力满提醒

    void OnApplicationPause(bool pauseStatus)
    {
        if (pauseStatus)
        {
            // 游戏切到后台时，安排通知
            ScheduleOfflineNotifications();
        }
        else
        {
            // 游戏回到前台时，取消所有未触发的通知（重新计算）
            LocalNotificationManager.Instance?.CancelAllNotifications();
        }
    }

    void ScheduleOfflineNotifications()
    {
        var notif = LocalNotificationManager.Instance;
        if (notif == null) return;
        
        // 1. 离线奖励通知
        if (enableOfflineReward)
        {
            notif.ScheduleNotification(
                "离线奖励等你领取！",
                $"您已离线{offlineRewardHours}小时，金币正在等您收取！",
                TimeSpan.FromHours(offlineRewardHours),
                "offline_reward");
        }
        
        // 2. 体力恢复满提醒
        if (enableEnergyFull)
        {
            int currentEnergy = PlayerDataService.GetLocalPlayerData()?.Energy ?? 0;
            int maxEnergy = 100;
            float recoveryRate = 5f; // 每分钟恢复5点
            
            if (currentEnergy < maxEnergy)
            {
                float minutesToFull = (maxEnergy - currentEnergy) / recoveryRate;
                notif.ScheduleNotification(
                    "体力已回满！",
                    "冒险者，您的体力已完全恢复，快来继续冒险吧！",
                    TimeSpan.FromMinutes(minutesToFull));
            }
        }
        
        // 3. 限时活动即将结束
        if (enableEventReminder)
        {
            ScheduleEventReminders();
        }
    }

    void ScheduleEventReminders()
    {
        // 从本地配置读取当前活动列表
        var events = EventManager.Instance?.GetActiveEvents();
        if (events == null) return;
        
        foreach (var evt in events)
        {
            TimeSpan timeLeft = evt.EndTime - DateTime.Now;
            
            if (timeLeft.TotalHours > 2 && timeLeft.TotalDays < 2)
            {
                // 活动2小时前提醒
                LocalNotificationManager.Instance?.ScheduleNotificationAt(
                    $"限时活动即将结束！",
                    $"【{evt.Name}】还有2小时结束，快去参与吧！",
                    evt.EndTime.AddHours(-2));
            }
        }
    }
}
```

## 三、通知规范与节制

```csharp
/// <summary>
/// 通知频率控制（防止过度打扰）
/// </summary>
public class NotificationRateLimiter
{
    private const int MAX_DAILY_NOTIFICATIONS = 3;  // 每日最多3条通知
    private const string KEY_LAST_NOTIFICATION_DATE = "notif_last_date";
    private const string KEY_DAILY_COUNT = "notif_daily_count";

    public bool CanSendNotification()
    {
        string today = DateTime.Now.ToString("yyyyMMdd");
        string lastDate = PlayerPrefs.GetString(KEY_LAST_NOTIFICATION_DATE, "");
        
        if (lastDate != today)
        {
            // 新的一天，重置计数
            PlayerPrefs.SetString(KEY_LAST_NOTIFICATION_DATE, today);
            PlayerPrefs.SetInt(KEY_DAILY_COUNT, 0);
        }
        
        int count = PlayerPrefs.GetInt(KEY_DAILY_COUNT, 0);
        return count < MAX_DAILY_NOTIFICATIONS;
    }

    public void RecordNotificationSent()
    {
        int count = PlayerPrefs.GetInt(KEY_DAILY_COUNT, 0);
        PlayerPrefs.SetInt(KEY_DAILY_COUNT, count + 1);
    }
}
```

## 四、通知最佳实践

| 原则 | 说明 |
|------|------|
| 价值优先 | 只推送对玩家有价值的通知（奖励/限时）|
| 频率节制 | 每天不超过3条，避免被关掉通知权限 |
| 最佳时机 | 请求权限在玩家完成首次游戏后（有了游戏认知再问）|
| 个性化 | 根据玩家行为定制通知内容（称玩家昵称）|
| 回前台取消 | 玩家回到游戏后立即取消所有已安排通知 |
| A/B测试 | 测试不同文案的打开率 |
