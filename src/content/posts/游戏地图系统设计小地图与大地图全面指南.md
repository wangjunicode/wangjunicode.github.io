## 一、RenderTexture小地图

```csharp
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 小地图控制器（基于 RenderTexture 俯视摄像机）
/// </summary>
public class MinimapController : MonoBehaviour
{
    [Header("俯视摄像机")]
    [SerializeField] private Camera minimapCamera;
    [SerializeField] private float cameraHeight = 50f;
    [SerializeField] private float orthographicSize = 30f;   // 视野范围
    [SerializeField] private Transform followTarget;          // 跟随目标（玩家）
    
    [Header("UI 配置")]
    [SerializeField] private RawImage minimapDisplay;         // 显示 RenderTexture 的 RawImage
    [SerializeField] private int renderTextureSize = 256;
    [SerializeField] private float rotateWithPlayer = 1f;    // 1=地图跟随旋转，0=北方朝上
    
    [Header("图标")]
    [SerializeField] private Transform iconContainer;        // 地图图标父节点
    [SerializeField] private GameObject playerIconPrefab;
    [SerializeField] private GameObject enemyIconPrefab;
    [SerializeField] private GameObject questIconPrefab;
    
    private RenderTexture renderTexture;
    private Dictionary<string, MinimapIcon> activeIcons = new Dictionary<string, MinimapIcon>();

    void Awake()
    {
        SetupRenderTexture();
        SetupCamera();
    }

    void SetupRenderTexture()
    {
        renderTexture = new RenderTexture(renderTextureSize, renderTextureSize, 16);
        renderTexture.antiAliasing = 1;
        minimapCamera.targetTexture = renderTexture;
        minimapDisplay.texture = renderTexture;
    }

    void SetupCamera()
    {
        minimapCamera.orthographic = true;
        minimapCamera.orthographicSize = orthographicSize;
        minimapCamera.clearFlags = CameraClearFlags.SolidColor;
        minimapCamera.backgroundColor = new Color(0.1f, 0.15f, 0.1f, 1f);
        
        // 设置小地图摄像机 CullingMask（只渲染地形和标记层）
        minimapCamera.cullingMask = LayerMask.GetMask("Terrain", "MinimapIcon");
    }

    void LateUpdate()
    {
        if (followTarget == null) return;
        
        // 摄像机跟随玩家
        minimapCamera.transform.position = new Vector3(
            followTarget.position.x, 
            followTarget.position.y + cameraHeight, 
            followTarget.position.z);
        
        // 地图旋转（跟随玩家朝向）
        if (rotateWithPlayer > 0.5f)
        {
            float angle = followTarget.eulerAngles.y;
            minimapCamera.transform.rotation = Quaternion.Euler(90f, angle, 0f);
        }
        else
        {
            minimapCamera.transform.rotation = Quaternion.Euler(90f, 0f, 0f);
        }
        
        UpdateIcons();
    }

    void UpdateIcons()
    {
        foreach (var kv in activeIcons)
            kv.Value.UpdatePosition(minimapCamera);
    }

    /// <summary>
    /// 注册小地图图标
    /// </summary>
    public MinimapIcon RegisterIcon(string id, Transform worldTarget, 
        MinimapIconType type, Color color = default)
    {
        if (activeIcons.ContainsKey(id))
            return activeIcons[id];
        
        GameObject prefab = type switch
        {
            MinimapIconType.Player => playerIconPrefab,
            MinimapIconType.Enemy  => enemyIconPrefab,
            MinimapIconType.Quest  => questIconPrefab,
            _ => playerIconPrefab
        };
        
        var iconGo = Instantiate(prefab, iconContainer);
        var icon = iconGo.AddComponent<MinimapIcon>();
        icon.Setup(worldTarget, minimapCamera, color != default ? color : Color.white);
        
        activeIcons[id] = icon;
        return icon;
    }

    public void UnregisterIcon(string id)
    {
        if (activeIcons.TryGetValue(id, out var icon))
        {
            Destroy(icon.gameObject);
            activeIcons.Remove(id);
        }
    }
}

public enum MinimapIconType { Player, Enemy, Quest, NPC, Objective, Custom }

/// <summary>
/// 小地图图标（将世界坐标映射到RenderTexture UV）
/// </summary>
public class MinimapIcon : MonoBehaviour
{
    private Transform worldTarget;
    private Camera minimapCamera;
    private RectTransform rectTransform;
    private Image icon;
    
    [SerializeField] private float iconSize = 12f;

    public void Setup(Transform target, Camera cam, Color color)
    {
        worldTarget = target;
        minimapCamera = cam;
        rectTransform = GetComponent<RectTransform>();
        icon = GetComponent<Image>();
        if (icon != null) icon.color = color;
        rectTransform.sizeDelta = Vector2.one * iconSize;
    }

    public void UpdatePosition(Camera cam)
    {
        if (worldTarget == null)
        {
            Destroy(gameObject);
            return;
        }
        
        // 世界坐标 → 小地图摄像机视口 → UI坐标
        Vector3 viewportPos = cam.WorldToViewportPoint(worldTarget.position);
        
        // 如果在摄像机视野外，将图标夹在边缘
        bool inRange = viewportPos.x >= 0 && viewportPos.x <= 1 &&
                       viewportPos.y >= 0 && viewportPos.y <= 1 &&
                       viewportPos.z > 0;
        
        if (inRange)
        {
            gameObject.SetActive(true);
            
            // 将 0-1 视口坐标转为 RawImage 上的 UI 位置
            var parentRect = transform.parent.GetComponent<RectTransform>();
            float x = (viewportPos.x - 0.5f) * parentRect.sizeDelta.x;
            float y = (viewportPos.y - 0.5f) * parentRect.sizeDelta.y;
            rectTransform.anchoredPosition = new Vector2(x, y);
        }
        else
        {
            // 超出范围，在边缘显示方向图标
            // 可以改为显示一个指向目标的箭头（边缘提示）
            gameObject.SetActive(false);
        }
    }
}
```

## 二、大地图系统

```csharp
/// <summary>
/// 大地图控制器（支持缩放/平移/标记）
/// </summary>
public class WorldMapController : MonoBehaviour
{
    [Header("地图配置")]
    [SerializeField] private RectTransform mapContent;    // 地图图片容器
    [SerializeField] private ScrollRect scrollRect;
    [SerializeField] private RectTransform viewport;
    
    [Header("缩放配置")]
    [SerializeField] private float minScale = 0.5f;
    [SerializeField] private float maxScale = 3f;
    [SerializeField] private float scrollZoomSpeed = 0.1f;
    
    [Header("玩家标记")]
    [SerializeField] private RectTransform playerMarker;
    [SerializeField] private Vector2 worldSize = new Vector2(1000, 1000); // 游戏世界尺寸
    [SerializeField] private Transform playerTransform;

    private float currentScale = 1f;

    void Update()
    {
        // 鼠标滚轮缩放
        float scroll = Input.mouseScrollDelta.y;
        if (scroll != 0)
        {
            float newScale = Mathf.Clamp(currentScale + scroll * scrollZoomSpeed, 
                minScale, maxScale);
            ZoomMap(newScale);
        }
        
        // 更新玩家标记位置
        UpdatePlayerMarker();
    }

    void ZoomMap(float newScale)
    {
        // 以鼠标位置为中心缩放
        Vector2 mousePos;
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            viewport, Input.mousePosition, null, out mousePos);
        
        float scaleDelta = newScale / currentScale;
        currentScale = newScale;
        
        mapContent.localScale = Vector3.one * currentScale;
        
        // 调整位置确保以鼠标为中心
        Vector2 newPos = mapContent.anchoredPosition;
        newPos = newPos * scaleDelta - mousePos * (scaleDelta - 1);
        mapContent.anchoredPosition = newPos;
    }

    void UpdatePlayerMarker()
    {
        if (playerTransform == null || playerMarker == null) return;
        
        // 玩家世界坐标 → 地图UV → 地图UI坐标
        float normalizedX = (playerTransform.position.x + worldSize.x / 2) / worldSize.x;
        float normalizedY = (playerTransform.position.z + worldSize.y / 2) / worldSize.y;
        
        Vector2 mapSize = mapContent.GetComponent<RectTransform>().sizeDelta;
        playerMarker.anchoredPosition = new Vector2(
            (normalizedX - 0.5f) * mapSize.x,
            (normalizedY - 0.5f) * mapSize.y);
        
        // 玩家图标朝向
        playerMarker.localEulerAngles = new Vector3(0, 0, -playerTransform.eulerAngles.y);
    }

    /// <summary>
    /// 聚焦到地图上的指定世界位置
    /// </summary>
    public void FocusOnWorldPos(Vector3 worldPos)
    {
        float normalizedX = (worldPos.x + worldSize.x / 2) / worldSize.x;
        float normalizedY = (worldPos.z + worldSize.y / 2) / worldSize.y;
        
        Vector2 mapSize = mapContent.sizeDelta;
        Vector2 targetContentPos = new Vector2(
            -(normalizedX - 0.5f) * mapSize.x * currentScale,
            -(normalizedY - 0.5f) * mapSize.y * currentScale);
        
        // 平滑移动
        DG.Tweening.DOTween.To(
            () => mapContent.anchoredPosition,
            p => mapContent.anchoredPosition = p,
            targetContentPos, 0.5f)
            .SetEase(DG.Tweening.Ease.OutCubic);
    }
}
```

## 三、地图标记管理

```csharp
/// <summary>
/// 地图标记（玩家自定义标记）
/// </summary>
public class MapMarkerManager : MonoBehaviour
{
    [SerializeField] private GameObject markerPrefab;
    [SerializeField] private Transform markerContainer;
    [SerializeField] private int maxCustomMarkers = 10;
    
    private List<MapMarker> customMarkers = new List<MapMarker>();

    public bool AddCustomMarker(Vector3 worldPos, string label, Color color)
    {
        if (customMarkers.Count >= maxCustomMarkers)
        {
            // 移除最旧的标记
            RemoveMarker(customMarkers[0]);
        }
        
        var markerGo = Instantiate(markerPrefab, markerContainer);
        var marker = markerGo.GetComponent<MapMarker>();
        marker.Setup(worldPos, label, color);
        customMarkers.Add(marker);
        
        return true;
    }

    public void RemoveMarker(MapMarker marker)
    {
        customMarkers.Remove(marker);
        Destroy(marker.gameObject);
    }
}
```

## 四、小地图性能优化

| 优化项 | 方案 | 效果 |
|--------|------|------|
| RenderTexture 尺寸 | 256x256 足够 | 内存省 |
| 更新频率 | 小地图摄像机每2帧更新一次 | 节省GPU |
| 图标数量 | 只显示范围内图标 | 减少渲染 |
| 旋转模式 | 可选「北方朝上」（无旋转开销）| 稳定性 |
| Culling Mask | 只渲染必要层 | 大幅减少Draw Call |
