## 一、体力系统核心

```csharp
using System;
using UnityEngine;

/// <summary>
/// 体力数据
/// </summary>
[Serializable]
public class StaminaData
{
    public int Current;
    public int Max;
    public int RecoveryPerMinute = 1; // 每分钟恢复量
    public long LastRecoveryTime;      // 最后恢复计算时间（Unix毫秒）
    
    // 购买恢复次数（每日限购）
    public int DailyBuyCount;
    public int MaxDailyBuyCount = 5;
    public long LastBuyResetDate; // 每日重置时间
}

/// <summary>
/// 体力管理器（客户端，需服务端同步）
/// </summary>
public class StaminaManager : MonoBehaviour
{
    private static StaminaManager instance;
    public static StaminaManager Instance => instance;

    private StaminaData data;
    
    public int Current => data?.Current ?? 0;
    public int Max => data?.Max ?? 0;
    public float FillRatio => Max > 0 ? (float)Current / Max : 0;
    
    public bool IsFull => Current >= Max;
    
    // 恢复倒计时（下次恢复还需多少秒）
    public float NextRecoverySeconds
    {
        get
        {
            if (IsFull) return -1;
            
            long now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
            float secPerStamina = 60f / data.RecoveryPerMinute;
            float elapsed = (now - data.LastRecoveryTime) / 1000f;
            
            return Mathf.Max(0, secPerStamina - elapsed % secPerStamina);
        }
    }
    
    public event Action<int, int> OnStaminaChanged; // current, max
    public event Action OnStaminaFull;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    void Start()
    {
        InvokeRepeating(nameof(TickRecovery), 1f, 1f);
    }

    /// <summary>
    /// 从服务端同步体力数据
    /// </summary>
    public void SyncFromServer(StaminaData serverData)
    {
        data = serverData;
        
        // 立即计算一次离线恢复（服务端也会计算，这里用于即时显示）
        CalculateOfflineRecovery();
        
        OnStaminaChanged?.Invoke(data.Current, data.Max);
    }

    /// <summary>
    /// 计算离线期间恢复的体力
    /// </summary>
    void CalculateOfflineRecovery()
    {
        if (data == null || IsFull) return;
        
        long now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        long elapsed = now - data.LastRecoveryTime;
        
        // 可恢复数量
        int recoverable = (int)(elapsed / 60000f * data.RecoveryPerMinute);
        
        if (recoverable > 0)
        {
            int oldCurrent = data.Current;
            data.Current = Math.Min(data.Current + recoverable, data.Max);
            data.LastRecoveryTime += (long)(recoverable / (float)data.RecoveryPerMinute * 60000);
            
            Debug.Log($"[Stamina] 离线恢复: +{data.Current - oldCurrent}");
        }
    }

    /// <summary>
    /// 每秒检查一次恢复
    /// </summary>
    void TickRecovery()
    {
        if (data == null || IsFull) return;
        
        long now = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        float elapsed = (now - data.LastRecoveryTime) / 1000f;
        float secPerStamina = 60f / data.RecoveryPerMinute;
        
        int recovered = (int)(elapsed / secPerStamina);
        
        if (recovered > 0)
        {
            bool wasFull = IsFull;
            int before = data.Current;
            
            data.Current = Math.Min(data.Current + recovered, data.Max);
            data.LastRecoveryTime += (long)(recovered * secPerStamina * 1000);
            
            if (data.Current != before)
            {
                OnStaminaChanged?.Invoke(data.Current, data.Max);
                
                if (IsFull && !wasFull)
                    OnStaminaFull?.Invoke();
            }
        }
    }

    /// <summary>
    /// 消耗体力（客户端预检查，服务端验证）
    /// </summary>
    public bool ConsumeStamina(int amount)
    {
        if (data == null || data.Current < amount)
        {
            UIManager.Instance?.ShowStaminaShortageUI(amount, data?.Current ?? 0);
            return false;
        }
        
        // 乐观更新（服务端失败时回滚）
        data.Current -= amount;
        OnStaminaChanged?.Invoke(data.Current, data.Max);
        
        return true;
    }

    /// <summary>
    /// 购买体力（用宝石恢复）
    /// </summary>
    public async System.Threading.Tasks.Task<bool> BuyStamina(int amount, int gemCost)
    {
        if (!CurrencyManager.Instance.HasEnough(CurrencyType.Gem, gemCost))
        {
            UIManager.Instance?.ShowMessage("宝石不足");
            return false;
        }
        
        if (data.DailyBuyCount >= data.MaxDailyBuyCount)
        {
            UIManager.Instance?.ShowMessage($"今日购买体力次数已达上限（{data.MaxDailyBuyCount}次）");
            return false;
        }
        
        try
        {
            var result = await NetworkService.BuyStamina(amount, gemCost);
            
            if (result.Success)
            {
                data.Current = Math.Min(data.Current + amount, data.Max);
                data.DailyBuyCount++;
                OnStaminaChanged?.Invoke(data.Current, data.Max);
                return true;
            }
            return false;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Stamina] 购买失败: {e.Message}");
            return false;
        }
    }

    /// <summary>
    /// 满体力时的过量奖励（活动/特殊情况）
    /// </summary>
    public void AddOverflow(int amount)
    {
        // 满体力时用"体力瓶"存储，不立即使用
        // PlayerDataService.GetLocalPlayerData().StaminaBottles += amount;
        Debug.Log($"[Stamina] 已满，{amount}体力存入体力瓶");
    }
}
```

## 二、体力UI组件

```csharp
/// <summary>
/// 体力显示UI
/// </summary>
public class StaminaUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Slider bar;
    [SerializeField] private TMPro.TextMeshProUGUI countText;
    [SerializeField] private TMPro.TextMeshProUGUI timerText;
    [SerializeField] private UnityEngine.UI.Button addButton;

    void Start()
    {
        StaminaManager.Instance.OnStaminaChanged += UpdateUI;
        addButton?.onClick.AddListener(OnAddClicked);
        
        UpdateUI(StaminaManager.Instance.Current, StaminaManager.Instance.Max);
    }

    void OnDestroy()
    {
        if (StaminaManager.Instance != null)
            StaminaManager.Instance.OnStaminaChanged -= UpdateUI;
    }

    void Update()
    {
        // 更新恢复倒计时
        float nextRecovery = StaminaManager.Instance.NextRecoverySeconds;
        
        if (nextRecovery < 0 || StaminaManager.Instance.IsFull)
        {
            timerText.text = "已满";
        }
        else
        {
            int minutes = (int)nextRecovery / 60;
            int seconds = (int)nextRecovery % 60;
            timerText.text = $"{minutes:D2}:{seconds:D2}";
        }
    }

    void UpdateUI(int current, int max)
    {
        if (bar != null) bar.value = max > 0 ? (float)current / max : 0;
        if (countText != null) countText.text = $"{current}/{max}";
    }

    void OnAddClicked()
    {
        // 打开购买体力面板
        UIManager.Instance?.OpenPanel("BuyStaminaPanel");
    }
}
```

## 三、体力系统设计要点

| 要点 | 方案 |
|------|------|
| 服务端计算 | 真实体力值由服务端计算，防止客户端篡改 |
| 时间精确 | 用Unix毫秒时间戳，避免时区/夏令时问题 |
| 离线补偿 | 登录时一次补偿离线期间恢复的体力 |
| 溢出设计 | 满体力后收到的体力存入体力瓶 |
| 每日重置 | 每日购买次数在服务端清零，防止客户端欺骗 |
| 倒计时精准 | 根据时间差实时计算，不用本地计时器 |
