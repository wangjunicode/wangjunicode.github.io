## 一、货币体系设计

```
多层货币体系：
├── 付费硬通货（Premium Currency）
│   ├── 真实付费获得（IAP）
│   ├── 不可通过游戏获得（或极少量）
│   └── 用途：高价值道具/月卡/礼包
│
├── 游戏软通货（Soft Currency）
│   ├── 通过游戏行为自然获得（金币/积分）
│   ├── 相对容易获得，可大量积累
│   └── 用途：基础消耗/装备维修/普通购买
│
└── 绑定货币（Bound Currency）
    ├── 完成活动/成就/竞技获得
    ├── 不可交易，防止代练/代拍
    └── 用途：限定内容/荣誉奖励
```

## 二、货币系统实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 货币类型枚举
/// </summary>
public enum CurrencyType
{
    Gold        = 1,  // 金币（软通货）
    Gem         = 2,  // 宝石（硬通货）
    Honor       = 3,  // 荣誉值（绑定）
    ActivityPoint = 4, // 活动积分（限时绑定）
}

/// <summary>
/// 货币变动原因（审计日志关键）
/// </summary>
public enum CurrencyReason
{
    Unknown             = 0,
    QuestReward         = 101,
    MonsterDrop         = 102,
    DailyLogin          = 103,
    AchievementReward   = 104,
    PVPReward           = 105,
    MarketplaceSell     = 106,
    IAP                 = 201,  // 付费充值
    BuyItem             = 301,
    RepairEquipment     = 302,
    CraftItem           = 303,
    MarketplaceBuy      = 304,
    GM                  = 999,  // GM操作（可疑，需标记）
}

/// <summary>
/// 货币变动记录（服务端存储，客户端镜像）
/// </summary>
public class CurrencyTransaction
{
    public string TransactionId;
    public string PlayerId;
    public CurrencyType Currency;
    public long Delta;          // 正数=增加，负数=减少
    public long BalanceAfter;   // 变动后余额
    public CurrencyReason Reason;
    public string DetailNote;   // 详细备注（如"购买了XX物品"）
    public long Timestamp;
}

/// <summary>
/// 客户端货币管理器（镜像服务端状态）
/// </summary>
public class CurrencyManager : MonoBehaviour
{
    private static CurrencyManager instance;
    public static CurrencyManager Instance => instance;

    // 本地缓存的余额（仅用于显示，实际以服务端为准）
    private Dictionary<CurrencyType, long> balances = new Dictionary<CurrencyType, long>();
    
    public event Action<CurrencyType, long, long> OnBalanceChanged; // type, oldBalance, newBalance

    void Awake() { instance = this; }

    /// <summary>
    /// 初始化余额（从服务端同步）
    /// </summary>
    public void Initialize(Dictionary<CurrencyType, long> serverBalances)
    {
        balances = new Dictionary<CurrencyType, long>(serverBalances);
    }

    /// <summary>
    /// 获取当前余额
    /// </summary>
    public long GetBalance(CurrencyType type)
    {
        return balances.TryGetValue(type, out long balance) ? balance : 0;
    }

    /// <summary>
    /// 检查是否有足够余额（客户端预检查，实际验证在服务端）
    /// </summary>
    public bool HasEnough(CurrencyType type, long amount)
    {
        return GetBalance(type) >= amount;
    }

    /// <summary>
    /// 服务端确认后更新本地余额
    /// </summary>
    public void OnServerConfirmed(CurrencyType type, long newBalance)
    {
        long oldBalance = GetBalance(type);
        balances[type] = newBalance;
        OnBalanceChanged?.Invoke(type, oldBalance, newBalance);
    }

    /// <summary>
    /// 购买请求（发送到服务端验证）
    /// </summary>
    public async System.Threading.Tasks.Task<PurchaseResult> RequestPurchase(
        string itemId, CurrencyType currency, long price)
    {
        // 客户端预检查
        if (!HasEnough(currency, price))
        {
            return new PurchaseResult 
            { 
                Success = false, 
                ErrorCode = "INSUFFICIENT_BALANCE",
                Message = $"余额不足，需要{price}，当前{GetBalance(currency)}" 
            };
        }
        
        // 发送到服务端
        try
        {
            var result = await NetworkService.Purchase(itemId, currency, price);
            
            if (result.Success)
            {
                // 更新本地余额
                OnServerConfirmed(currency, result.NewBalance);
            }
            
            return result;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Currency] Purchase failed: {e.Message}");
            return new PurchaseResult { Success = false, ErrorCode = "NETWORK_ERROR" };
        }
    }
}

public class PurchaseResult
{
    public bool Success;
    public string ErrorCode;
    public string Message;
    public long NewBalance;
    public string TransactionId;
}
```

## 三、货币经济监控

```csharp
/// <summary>
/// 经济监控指标（服务端计算，客户端展示）
/// </summary>
public class EconomyMetrics
{
    // 货币流通指标
    public float DailyCurrencyInflow;    // 每日流入（所有来源）
    public float DailyCurrencyOutflow;   // 每日流出（所有消耗）
    public float InflationRate;          // 通货膨胀率 = (流入-流出)/流通量
    
    // 玩家货币分布
    public float MedianBalance;          // 中位数余额
    public float TopPlayerSharePercent; // 前10%玩家持有比例（越高越不健康）
    
    // 市场价格指数
    public float CommonItemPriceIndex;  // 普通物品价格指数（相对基准）
    public float RareItemPriceIndex;    // 稀有物品价格指数
}

/// <summary>
/// 通货膨胀控制策略
/// </summary>
public static class InflationControlStrategy
{
    // 当通胀率过高时的干预手段

    // 1. 货币汇率（充值获得更少软通货）
    // 2. 物品价格上调（NPC商店）
    // 3. 增加货币消耗途径（限时活动/节日消耗）
    // 4. 减少货币奖励（活动奖励调低）
    // 5. 货币回收（稀有配方需大量材料+货币）
    // 6. 财富税（超大量持有征收维护费）
    
    public static List<string> GetRecommendations(EconomyMetrics metrics)
    {
        var recommendations = new List<string>();
        
        if (metrics.InflationRate > 0.05f) // 通胀超过5%
        {
            recommendations.Add("建议增加限时货币消耗活动");
            recommendations.Add("建议降低常规任务货币奖励10%");
        }
        
        if (metrics.TopPlayerSharePercent > 0.6f) // 马太效应过强
        {
            recommendations.Add("贫富分化严重，建议新手引导奖励翻倍");
            recommendations.Add("建议为低余额玩家设置补贴机制");
        }
        
        if (metrics.RareItemPriceIndex > 3.0f) // 稀有物品价格过高
        {
            recommendations.Add("稀有物品价格过高，建议增加稀有物品供给");
        }
        
        return recommendations;
    }
}
```

## 四、拍卖行价格系统

```csharp
/// <summary>
/// 物品历史价格追踪（帮助玩家定价参考）
/// </summary>
public class AuctionPriceOracle
{
    private Dictionary<string, PriceHistory> priceHistories 
        = new Dictionary<string, PriceHistory>();

    public class PriceHistory
    {
        public string ItemId;
        public List<long> RecentPrices = new List<long>(); // 最近50次成交价
        
        public long GetSuggestedPrice()
        {
            if (RecentPrices.Count == 0) return 0;
            
            // 去掉最高/最低10%，取均值（防止异常值）
            var sorted = new List<long>(RecentPrices);
            sorted.Sort();
            
            int trimCount = Mathf.Max(1, sorted.Count / 10);
            long sum = 0;
            int count = 0;
            
            for (int i = trimCount; i < sorted.Count - trimCount; i++)
            {
                sum += sorted[i];
                count++;
            }
            
            return count > 0 ? sum / count : sorted[sorted.Count / 2];
        }
        
        public long GetMedianPrice()
        {
            if (RecentPrices.Count == 0) return 0;
            var sorted = new List<long>(RecentPrices);
            sorted.Sort();
            return sorted[sorted.Count / 2];
        }
    }

    public void RecordSale(string itemId, long price)
    {
        if (!priceHistories.ContainsKey(itemId))
            priceHistories[itemId] = new PriceHistory { ItemId = itemId };
        
        var history = priceHistories[itemId];
        history.RecentPrices.Add(price);
        
        // 只保留最近50条
        if (history.RecentPrices.Count > 50)
            history.RecentPrices.RemoveAt(0);
    }

    public long GetSuggestedPrice(string itemId)
    {
        return priceHistories.TryGetValue(itemId, out var h) ?
            h.GetSuggestedPrice() : 0;
    }
}
```

## 五、经济设计要点

| 原则 | 说明 |
|------|------|
| 流入 = 流出 | 长期应趋于平衡，否则通胀/通缩 |
| 货币分层 | 不同货币对应不同价值密度 |
| 货币汇率 | 软通货↔硬通货不应直接兑换 |
| 交易税 | 拍卖行征税2-5%，减少货币流通 |
| 货币沉没 | 装备损耗/强化失败消耗货币 |
| 服务端权威 | 所有货币变动必须经服务端验证 |
| 实时监控 | 每日统计流入流出，发现异常即介入 |
