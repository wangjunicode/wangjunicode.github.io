## 二、飘字系统实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using DG.Tweening;

/// <summary>
/// 飘字类型
/// </summary>
public enum FloatTextType
{
    NormalDamage,
    CriticalDamage,
    MagicDamage,
    Heal,
    Debuff,
    Immune,
    Absorb,
    Miss,
    Exp,
    Gold,
    Custom
}

/// <summary>
/// 飘字配置
/// </summary>
[CreateAssetMenu(fileName = "FloatTextConfig", menuName = "Game/FloatText Config")]
public class FloatTextConfig : ScriptableObject
{
    [Serializable]
    public class TypeConfig
    {
        public FloatTextType Type;
        public Color Color = Color.white;
        public float Scale = 1f;
        public string Prefix = "";
        public string Suffix = "";
        public FloatTextAnimation Animation = FloatTextAnimation.Linear;
        public float Duration = 1.5f;
        public float RiseDistance = 100f;
        public float FontSize = 36f;
        public bool IsBold;
        public bool HasOutline = true;
    }

    public List<TypeConfig> Configs = new List<TypeConfig>();

    public TypeConfig Get(FloatTextType type)
    {
        return Configs.Find(c => c.Type == type) ?? Configs[0];
    }
}

public enum FloatTextAnimation { Linear, Arc, Bounce, Spiral, Popup }

/// <summary>
/// 单个飘字实例
/// </summary>
public class FloatTextInstance : MonoBehaviour
{
    [SerializeField] private Text textComponent;
    [SerializeField] private Outline outline;
    [SerializeField] private Shadow shadow;

    private RectTransform rectTransform;
    private CanvasGroup canvasGroup;
    private Action onComplete;

    void Awake()
    {
        rectTransform = GetComponent<RectTransform>();
        canvasGroup = GetOrAdd<CanvasGroup>();
    }

    public void Play(string text, FloatTextConfig.TypeConfig config,
        Vector2 startPos, Action onComplete = null)
    {
        this.onComplete = onComplete;
        
        // 设置文本外观
        textComponent.text = $"{config.Prefix}{text}{config.Suffix}";
        textComponent.color = config.Color;
        textComponent.fontSize = (int)config.FontSize;
        textComponent.fontStyle = config.IsBold ? FontStyle.Bold : FontStyle.Normal;
        
        if (outline != null)
        {
            outline.enabled = config.HasOutline;
            outline.effectColor = new Color(0, 0, 0, 0.8f);
        }
        
        // 初始化状态
        rectTransform.anchoredPosition = startPos;
        rectTransform.localScale = Vector3.one * config.Scale;
        canvasGroup.alpha = 1f;
        
        // 播放动画
        DOTween.Kill(rectTransform);
        DOTween.Kill(canvasGroup);
        
        switch (config.Animation)
        {
            case FloatTextAnimation.Linear:
                PlayLinearAnim(config);
                break;
            case FloatTextAnimation.Arc:
                PlayArcAnim(config, startPos);
                break;
            case FloatTextAnimation.Bounce:
                PlayBounceAnim(config);
                break;
            case FloatTextAnimation.Popup:
                PlayPopupAnim(config);
                break;
        }
    }

    void PlayLinearAnim(FloatTextConfig.TypeConfig config)
    {
        var seq = DOTween.Sequence();
        
        // 上飘
        seq.Append(rectTransform.DOAnchorPosY(
            rectTransform.anchoredPosition.y + config.RiseDistance, 
            config.Duration).SetEase(Ease.OutQuad));
        
        // 后半段淡出
        seq.Insert(config.Duration * 0.6f, 
            canvasGroup.DOFade(0f, config.Duration * 0.4f));
        
        seq.OnComplete(ReturnToPool);
    }

    void PlayArcAnim(FloatTextConfig.TypeConfig config, Vector2 startPos)
    {
        // 随机偏移方向（不同飘字分散显示）
        float xOffset = UnityEngine.Random.Range(-60f, 60f);
        
        var seq = DOTween.Sequence();
        seq.Append(rectTransform.DOAnchorPos(
            startPos + new Vector2(xOffset, config.RiseDistance), 
            config.Duration).SetEase(Ease.OutQuad));
        seq.Insert(config.Duration * 0.5f, 
            canvasGroup.DOFade(0f, config.Duration * 0.5f));
        seq.OnComplete(ReturnToPool);
    }

    void PlayBounceAnim(FloatTextConfig.TypeConfig config)
    {
        var seq = DOTween.Sequence();
        
        // 缩放弹出
        rectTransform.localScale = Vector3.zero;
        seq.Append(rectTransform.DOScale(config.Scale * 1.3f, 0.15f).SetEase(Ease.OutBack));
        seq.Append(rectTransform.DOScale(config.Scale, 0.1f));
        seq.Append(rectTransform.DOAnchorPosY(
            rectTransform.anchoredPosition.y + config.RiseDistance, 
            config.Duration - 0.25f).SetEase(Ease.OutCubic));
        seq.Insert(config.Duration * 0.6f, 
            canvasGroup.DOFade(0f, config.Duration * 0.4f));
        seq.OnComplete(ReturnToPool);
    }

    void PlayPopupAnim(FloatTextConfig.TypeConfig config)
    {
        var seq = DOTween.Sequence();
        
        // 弹出
        rectTransform.localScale = Vector3.zero;
        seq.Append(rectTransform.DOScale(config.Scale * 1.5f, 0.2f).SetEase(Ease.OutBack));
        seq.Append(rectTransform.DOScale(config.Scale, 0.1f));
        
        // 停留后淡出
        seq.AppendInterval(0.5f);
        seq.Append(canvasGroup.DOFade(0f, 0.3f));
        seq.OnComplete(ReturnToPool);
    }

    void ReturnToPool()
    {
        onComplete?.Invoke();
    }

    T GetOrAdd<T>() where T : Component => GetComponent<T>() ?? gameObject.AddComponent<T>();
}

/// <summary>
/// 飘字管理器（对象池 + 位置分散）
/// </summary>
public class FloatTextManager : MonoBehaviour
{
    private static FloatTextManager instance;
    public static FloatTextManager Instance => instance;

    [SerializeField] private GameObject floatTextPrefab;
    [SerializeField] private FloatTextConfig config;
    [SerializeField] private Canvas canvas;
    [SerializeField] private int poolSize = 30;

    private Queue<FloatTextInstance> pool = new Queue<FloatTextInstance>();
    
    // 位置分散：防止多个飘字重叠
    private Dictionary<string, List<Vector2>> recentPositions 
        = new Dictionary<string, List<Vector2>>();

    void Awake()
    {
        instance = this;
        
        // 预热对象池
        for (int i = 0; i < poolSize; i++)
        {
            var inst = CreateInstance();
            inst.gameObject.SetActive(false);
            pool.Enqueue(inst);
        }
    }

    /// <summary>
    /// 显示飘字（主接口）
    /// </summary>
    public void Show(FloatTextType type, float value, Vector3 worldPos)
    {
        Show(type, FormatNumber(value, type), worldPos);
    }

    public void Show(FloatTextType type, string text, Vector3 worldPos)
    {
        var typeConfig = config.Get(type);
        
        // 世界坐标 → 屏幕坐标 → Canvas坐标
        Vector2 screenPos = RectTransformUtility.WorldToScreenPoint(
            Camera.main, worldPos);
        
        if (!RectTransformUtility.ScreenPointToLocalPointInRectangle(
            canvas.GetComponent<RectTransform>(), screenPos, 
            canvas.worldCamera, out Vector2 canvasPos))
            return;
        
        // 分散位置（避免多个飘字堆叠）
        canvasPos = GetScatteredPosition(canvasPos);
        
        // 获取实例
        var inst = GetFromPool();
        inst.gameObject.SetActive(true);
        inst.transform.SetParent(canvas.transform, false);
        
        inst.Play(text, typeConfig, canvasPos, () =>
        {
            inst.gameObject.SetActive(false);
            pool.Enqueue(inst);
        });
    }

    Vector2 GetScatteredPosition(Vector2 basePos)
    {
        // 将屏幕分成格子，避免同一格子内飘字重叠
        string key = $"{(int)(basePos.x / 80)},{(int)(basePos.y / 80)}";
        
        if (!recentPositions.ContainsKey(key))
            recentPositions[key] = new List<Vector2>();
        
        var positions = recentPositions[key];
        
        // 找到不与现有飘字重叠的位置
        Vector2 finalPos = basePos;
        for (int attempt = 0; attempt < 5; attempt++)
        {
            bool overlaps = false;
            foreach (var pos in positions)
            {
                if (Vector2.Distance(finalPos, pos) < 40f)
                {
                    overlaps = true;
                    finalPos.x += UnityEngine.Random.Range(-30f, 30f);
                    finalPos.y += 40f;
                    break;
                }
            }
            if (!overlaps) break;
        }
        
        positions.Add(finalPos);
        
        // 限制记录数量
        if (positions.Count > 5) positions.RemoveAt(0);
        
        return finalPos;
    }

    string FormatNumber(float value, FloatTextType type)
    {
        // 大数字简化：1500 → 1.5K
        int intVal = Mathf.Abs(Mathf.RoundToInt(value));
        
        if (intVal >= 10000)
            return $"{intVal / 1000f:F1}K";
        
        return intVal.ToString();
    }

    FloatTextInstance GetFromPool()
    {
        if (pool.Count > 0) return pool.Dequeue();
        return CreateInstance();
    }

    FloatTextInstance CreateInstance()
    {
        var go = Instantiate(floatTextPrefab, canvas.transform);
        return go.GetComponent<FloatTextInstance>() ?? 
               go.AddComponent<FloatTextInstance>();
    }
}
```

## 三、伤害数字聚合（DPS场景优化）

```csharp
/// <summary>
/// 伤害数字聚合器（短时间内多次伤害合并为一个飘字）
/// </summary>
public class DamageAggregator : MonoBehaviour
{
    [SerializeField] private float aggregateWindow = 0.1f;  // 聚合时间窗口（秒）
    [SerializeField] private int maxAggregateCount = 10;    // 最多聚合次数
    
    private float accumulatedDamage;
    private int hitCount;
    private float timer;
    private Vector3 lastPosition;
    private FloatTextType lastType;
    private bool isDirty;

    public void AddDamage(float damage, Vector3 position, FloatTextType type)
    {
        accumulatedDamage += damage;
        hitCount++;
        lastPosition = position;
        lastType = type;
        isDirty = true;
        timer = 0f;
    }

    void Update()
    {
        if (!isDirty) return;
        
        timer += Time.deltaTime;
        
        if (timer >= aggregateWindow || hitCount >= maxAggregateCount)
        {
            FlushDamage();
        }
    }

    void FlushDamage()
    {
        if (accumulatedDamage <= 0) return;
        
        string text = hitCount > 1 ? 
            $"{Mathf.RoundToInt(accumulatedDamage)}(x{hitCount})" : 
            Mathf.RoundToInt(accumulatedDamage).ToString();
        
        FloatTextManager.Instance?.Show(lastType, text, lastPosition);
        
        accumulatedDamage = 0;
        hitCount = 0;
        isDirty = false;
    }
}
```

## 四、最佳实践总结

1. **对象池化**：飘字数量在大规模战斗中极多，必须池化
2. **位置分散**：通过格子分散算法防止堆叠
3. **大数字简化**：10000+ 显示为 "10K"，保持画面整洁
4. **暴击强调**：暴击飘字明显更大、颜色不同，强化反馈感
5. **聚合机制**：AOE/DoT 场景下短时间合并多次伤害，避免屏幕被刷屏
6. **移动端**：飘字字体需要比桌面端更大（手指宽度 ~44px）
