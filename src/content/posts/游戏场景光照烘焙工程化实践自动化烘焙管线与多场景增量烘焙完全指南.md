# 游戏场景光照烘焙工程化实践：自动化烘焙管线与多场景增量烘焙完全指南

## 一、烘焙工程化的必要性

Unity 的 Lightmapper（Enlighten/Progressive）提供了强大的离线 GI 计算能力，但随着项目规模扩大，烘焙面临的工程挑战远超技术挑战：

```
工程挑战：
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
├── 效率：30个场景，每场景 2小时烘焙 = 60小时，美术无法接受
├── 协作：多人修改场景后 lightmap 文件 Git 冲突
├── 质量：烘焙完成但 UV 重叠/漏光/接缝 无人发现
├── 增量：全量重烘太慢，只改了一个灯光怎么办
└── CI/CD：夜间自动烘焙，第二天来就有新结果
```

本文将构建一套完整的**烘焙工程化**体系，把烘焙从"一次次手动点按钮"变成可维护、可追踪、可自动化的流水线。

## 三、烘焙内容哈希系统（增量核心）

```csharp
/// <summary>
/// 场景烘焙内容哈希计算器
/// 通过对影响烘焙结果的场景元素计算哈希，判断是否需要重新烘焙
/// </summary>
public static class BakeContentHasher
{
    /// <summary>
    /// 计算影响烘焙结果的场景内容哈希
    /// 包括：Static Mesh Transform、光源参数、材质、天空盒
    /// </summary>
    public static string ComputeSceneHash(string scenePath)
    {
        using var scene = UnityEditor.SceneManagement.EditorSceneManager
            .OpenScene(scenePath, UnityEditor.SceneManagement.OpenSceneMode.Additive);
        
        var sb = new System.Text.StringBuilder();
        
        // 1. 所有 Static Mesh 的 Transform + Mesh Hash
        var meshRenderers = UnityEngine.Object.FindObjectsOfType<MeshRenderer>();
        foreach (var mr in meshRenderers)
        {
            if (!GameObjectUtility.AreStaticEditorFlagsSet(mr.gameObject, 
                StaticEditorFlags.ContributeGI))
                continue;
            
            var t = mr.transform;
            sb.Append($"MR:{t.position:F3}:{t.rotation.eulerAngles:F3}:{t.localScale:F3}");
            
            var mf = mr.GetComponent<MeshFilter>();
            if (mf?.sharedMesh != null)
                sb.Append($":mesh{mf.sharedMesh.GetInstanceID()}");
            
            // 材质的 Emission 设置影响烘焙
            foreach (var mat in mr.sharedMaterials)
            {
                if (mat != null && mat.IsKeywordEnabled("_EMISSION"))
                    sb.Append($":emit{mat.GetInstanceID()}");
            }
        }
        
        // 2. 所有光源参数
        var lights = UnityEngine.Object.FindObjectsOfType<Light>();
        foreach (var light in lights)
        {
            if (light.lightmapBakeType == LightmapBakeType.Realtime)
                continue;
            
            sb.Append($"L:{light.type}:{light.color}:{light.intensity:F3}:" +
                $"{light.transform.position:F3}:{light.transform.rotation.eulerAngles:F3}:" +
                $"{light.shadows}:{light.shadowStrength:F2}");
        }
        
        // 3. RenderSettings（天空盒、环境光）
        var rs = RenderSettings.ambientLight;
        sb.Append($"RS:{rs.r:F3}:{rs.g:F3}:{rs.b:F3}");
        if (RenderSettings.skybox != null)
            sb.Append($":sky{RenderSettings.skybox.GetInstanceID()}");
        
        // 4. LightProbes 位置
        var probeGroup = UnityEngine.Object.FindObjectsOfType<LightProbeGroup>();
        foreach (var pg in probeGroup)
        {
            foreach (var pos in pg.probePositions)
                sb.Append($":lp{pos:F2}");
        }
        
        // 计算 MD5 哈希
        using var md5 = System.Security.Cryptography.MD5.Create();
        var bytes = System.Text.Encoding.UTF8.GetBytes(sb.ToString());
        var hash = md5.ComputeHash(bytes);
        
        UnityEditor.SceneManagement.EditorSceneManager.CloseScene(scene, false);
        
        return BitConverter.ToString(hash).Replace("-", "").ToLower();
    }
    
    /// <summary>
    /// 读取上次烘焙时保存的哈希
    /// </summary>
    public static string LoadSavedHash(string scenePath)
    {
        string hashFile = GetHashFilePath(scenePath);
        return System.IO.File.Exists(hashFile) 
            ? System.IO.File.ReadAllText(hashFile).Trim() 
            : string.Empty;
    }
    
    /// <summary>
    /// 保存本次烘焙的哈希（烘焙完成后调用）
    /// </summary>
    public static void SaveHash(string scenePath, string hash)
    {
        string hashFile = GetHashFilePath(scenePath);
        System.IO.Directory.CreateDirectory(System.IO.Path.GetDirectoryName(hashFile));
        System.IO.File.WriteAllText(hashFile, hash);
    }
    
    public static bool NeedsRebake(string scenePath, float changeThreshold = 0f)
    {
        string current = ComputeSceneHash(scenePath);
        string saved = LoadSavedHash(scenePath);
        
        if (string.IsNullOrEmpty(saved)) return true;
        
        // 完全相同不需要重烘
        if (current == saved) return false;
        
        // 如果设置了变化阈值（模糊比较），可以忽略微小变化
        // 简单实现：完整哈希不同则重烘（实际可做更细粒度比较）
        return true;
    }
    
    private static string GetHashFilePath(string scenePath)
    {
        string sceneName = System.IO.Path.GetFileNameWithoutExtension(scenePath);
        return $"Library/LightmapHashes/{sceneName}.hash";
    }
}
```

## 四、自动化烘焙管线

### 4.1 烘焙任务调度器

```csharp
/// <summary>
/// 多场景批量烘焙任务调度器
/// 支持优先级排序、增量跳过、错误重试
/// </summary>
public class BakePipelineScheduler
{
    private readonly List<BakeTask> _taskQueue = new();
    private int _completedCount;
    private int _skippedCount;
    private int _failedCount;
    private readonly BakePipelineConfig _config;
    
    public BakePipelineScheduler(BakePipelineConfig config)
    {
        _config = config;
    }
    
    /// <summary>
    /// 添加烘焙任务（按优先级自动排序）
    /// </summary>
    public void AddTask(SceneBakeConfig bakeConfig)
    {
        foreach (var scene in bakeConfig.TargetScenes)
        {
            if (scene == null) continue;
            
            string scenePath = UnityEditor.AssetDatabase.GetAssetPath(scene);
            
            // 增量检查
            if (bakeConfig.EnableIncrementalBake && 
                !BakeContentHasher.NeedsRebake(scenePath, bakeConfig.IncrementalChangeThreshold))
            {
                UnityEngine.Debug.Log($"[BakePipeline] 跳过未变更场景: {scene.name}");
                _skippedCount++;
                continue;
            }
            
            _taskQueue.Add(new BakeTask
            {
                Config = bakeConfig,
                ScenePath = scenePath,
                SceneName = scene.name,
                Priority = GetPriority(bakeConfig.Category),
                Status = BakeTaskStatus.Pending
            });
        }
        
        _taskQueue.Sort((a, b) => b.Priority.CompareTo(a.Priority));
    }
    
    /// <summary>
    /// 执行所有烘焙任务（Editor Batch Mode 下调用）
    /// </summary>
    public void ExecuteAll()
    {
        UnityEngine.Debug.Log($"[BakePipeline] 开始执行 {_taskQueue.Count} 个烘焙任务...");
        var startTime = DateTime.Now;
        
        foreach (var task in _taskQueue)
        {
            ExecuteTask(task);
        }
        
        var elapsed = DateTime.Now - startTime;
        GeneratePipelineReport(elapsed);
    }
    
    private void ExecuteTask(BakeTask task)
    {
        UnityEngine.Debug.Log($"[BakePipeline] 开始烘焙场景: {task.SceneName}");
        task.Status = BakeTaskStatus.Running;
        task.StartTime = DateTime.Now;
        
        try
        {
            // 1. 打开场景
            var scene = UnityEditor.SceneManagement.EditorSceneManager
                .OpenScene(task.ScenePath, UnityEditor.SceneManagement.OpenSceneMode.Single);
            
            // 2. 应用烘焙配置
            ApplyBakeSettings(task.Config);
            
            // 3. 执行烘焙（同步模式，用于批处理）
            Lightmapping.Bake();
            
            // 4. 质量验证
            if (task.Config.RunQualityCheck)
            {
                var issues = RunQualityCheck(task);
                if (issues.Count > 0)
                {
                    task.Status = BakeTaskStatus.CompletedWithWarnings;
                    task.QualityIssues = issues;
                }
            }
            
            // 5. 保存场景和哈希
            UnityEditor.SceneManagement.EditorSceneManager.SaveScene(scene);
            BakeContentHasher.SaveHash(task.ScenePath, 
                BakeContentHasher.ComputeSceneHash(task.ScenePath));
            
            // 6. 导出 Lightmap 到指定目录
            ExportLightmaps(task);
            
            if (task.Status == BakeTaskStatus.Running)
                task.Status = BakeTaskStatus.Completed;
            
            _completedCount++;
        }
        catch (Exception e)
        {
            task.Status = BakeTaskStatus.Failed;
            task.ErrorMessage = e.Message;
            _failedCount++;
            UnityEngine.Debug.LogError($"[BakePipeline] 烘焙失败: {task.SceneName}\n{e}");
        }
        finally
        {
            task.EndTime = DateTime.Now;
            task.Duration = task.EndTime - task.StartTime;
        }
    }
    
    private void ApplyBakeSettings(SceneBakeConfig config)
    {
        // 应用 Lightmap 尺寸和密度
        LightmapEditorSettings.maxAtlasSize = config.LightmapSize;
        LightmapEditorSettings.bakeResolution = config.LightmapTexelsPerUnit;
        
        // 应用 AO 设置
        LightmapEditorSettings.ao = config.UseAO;
        LightmapEditorSettings.aoMaxDistance = config.AORadius;
        
        // 应用 Progressive Lightmapper 采样数
        LightmapEditorSettings.directSampleCount = config.DirectSamples;
        LightmapEditorSettings.indirectSampleCount = config.IndirectSamples;
        LightmapEditorSettings.environmentSampleCount = config.EnvironmentSamples;
        
        // 质量预设覆盖
        ApplyQualityPreset(config.QualityPreset);
    }
    
    private void ApplyQualityPreset(LightingQualityPreset preset)
    {
        switch (preset)
        {
            case LightingQualityPreset.Preview:
                LightmapEditorSettings.directSampleCount = 8;
                LightmapEditorSettings.indirectSampleCount = 32;
                LightmapEditorSettings.environmentSampleCount = 32;
                LightmapEditorSettings.maxAtlasSize = 512;
                break;
            
            case LightingQualityPreset.Medium:
                LightmapEditorSettings.directSampleCount = 32;
                LightmapEditorSettings.indirectSampleCount = 256;
                LightmapEditorSettings.environmentSampleCount = 128;
                LightmapEditorSettings.maxAtlasSize = 1024;
                break;
            
            case LightingQualityPreset.Release:
                LightmapEditorSettings.directSampleCount = 64;
                LightmapEditorSettings.indirectSampleCount = 1024;
                LightmapEditorSettings.environmentSampleCount = 512;
                LightmapEditorSettings.maxAtlasSize = 2048;
                break;
        }
    }
    
    private List<BakeQualityIssue> RunQualityCheck(BakeTask task)
    {
        var issues = new List<BakeQualityIssue>();
        
        // 检查1：Lightmap 文件是否生成
        string lightmapDir = $"Assets/{task.SceneName}";
        if (!System.IO.Directory.Exists(lightmapDir))
        {
            issues.Add(new BakeQualityIssue
            {
                Type = QualityIssueType.Missing,
                Message = $"Lightmap 目录不存在: {lightmapDir}",
                Severity = IssueSeverity.Error
            });
            return issues;
        }
        
        // 检查2：UV 重叠（通过检查 Mesh 的 UV2 通道）
        var meshRenderers = UnityEngine.Object.FindObjectsOfType<MeshRenderer>();
        int overlapCount = 0;
        int totalMeshes = 0;
        
        foreach (var mr in meshRenderers)
        {
            if (!GameObjectUtility.AreStaticEditorFlagsSet(mr.gameObject, StaticEditorFlags.ContributeGI))
                continue;
            
            totalMeshes++;
            var mf = mr.GetComponent<MeshFilter>();
            if (mf?.sharedMesh != null && mf.sharedMesh.uv2.Length == 0)
            {
                overlapCount++;
                issues.Add(new BakeQualityIssue
                {
                    Type = QualityIssueType.MissingUV2,
                    Message = $"缺少 UV2（光照贴图UV）: {mr.name}",
                    AffectedObject = mr.gameObject.name,
                    Severity = IssueSeverity.Warning
                });
            }
        }
        
        // 检查3：Lightmap 文件大小合理性
        var lightmapFiles = System.IO.Directory.GetFiles(lightmapDir, "*.exr");
        foreach (var file in lightmapFiles)
        {
            var info = new System.IO.FileInfo(file);
            if (info.Length < 1024) // 小于 1KB 说明烘焙可能失败
            {
                issues.Add(new BakeQualityIssue
                {
                    Type = QualityIssueType.InvalidLightmap,
                    Message = $"Lightmap 文件过小（可能烘焙失败）: {file}",
                    Severity = IssueSeverity.Error
                });
            }
        }
        
        return issues;
    }
    
    private void ExportLightmaps(BakeTask task)
    {
        string srcDir = $"Assets/{task.SceneName}";
        string dstDir = System.IO.Path.Combine(task.Config.LightmapOutputPath, task.SceneName);
        
        if (!System.IO.Directory.Exists(srcDir)) return;
        
        System.IO.Directory.CreateDirectory(dstDir);
        
        foreach (var file in System.IO.Directory.GetFiles(srcDir))
        {
            string fileName = System.IO.Path.GetFileName(file);
            string dstFile = System.IO.Path.Combine(dstDir, fileName);
            System.IO.File.Copy(file, dstFile, overwrite: true);
        }
        
        UnityEngine.Debug.Log($"[BakePipeline] Lightmap 已导出至: {dstDir}");
    }
    
    private int GetPriority(BakeSceneCategory category)
    {
        return category switch
        {
            BakeSceneCategory.AssetPack => 100,
            BakeSceneCategory.MainCity => 80,
            BakeSceneCategory.Cinematic => 60,
            BakeSceneCategory.Development => 40,
            _ => 50
        };
    }
    
    private void GeneratePipelineReport(TimeSpan elapsed)
    {
        var report = new System.Text.StringBuilder();
        report.AppendLine("========================================");
        report.AppendLine("烘焙管线执行报告");
        report.AppendLine("========================================");
        report.AppendLine($"总耗时: {elapsed:hh\\:mm\\:ss}");
        report.AppendLine($"完成: {_completedCount}  跳过: {_skippedCount}  失败: {_failedCount}");
        report.AppendLine();
        
        foreach (var task in _taskQueue)
        {
            string statusIcon = task.Status switch
            {
                BakeTaskStatus.Completed => "✓",
                BakeTaskStatus.CompletedWithWarnings => "⚠",
                BakeTaskStatus.Failed => "✗",
                BakeTaskStatus.Skipped => "⊘",
                _ => "?"
            };
            
            report.AppendLine($"  {statusIcon} {task.SceneName} [{task.Duration:mm\\:ss}]");
            
            if (task.QualityIssues?.Count > 0)
            {
                foreach (var issue in task.QualityIssues)
                    report.AppendLine($"      [{issue.Severity}] {issue.Message}");
            }
            
            if (!string.IsNullOrEmpty(task.ErrorMessage))
                report.AppendLine($"      错误: {task.ErrorMessage}");
        }
        
        report.AppendLine("========================================");
        
        string reportPath = $"Library/BakeReport_{DateTime.Now:yyyyMMdd_HHmmss}.txt";
        System.IO.File.WriteAllText(reportPath, report.ToString());
        UnityEngine.Debug.Log(report.ToString());
        UnityEngine.Debug.Log($"[BakePipeline] 报告已保存至: {reportPath}");
    }
    
    private sealed class BakeTask
    {
        public SceneBakeConfig Config;
        public string ScenePath;
        public string SceneName;
        public int Priority;
        public BakeTaskStatus Status;
        public DateTime StartTime;
        public DateTime EndTime;
        public TimeSpan Duration;
        public string ErrorMessage;
        public List<BakeQualityIssue> QualityIssues;
    }
    
    private enum BakeTaskStatus
    {
        Pending, Running, Completed, CompletedWithWarnings, Failed, Skipped
    }
}

public class BakePipelineConfig
{
    public int MaxRetryCount = 1;
    public bool ContinueOnError = true;
}

public class BakeQualityIssue
{
    public QualityIssueType Type;
    public string Message;
    public string AffectedObject;
    public IssueSeverity Severity;
}

public enum QualityIssueType
{
    Missing, MissingUV2, InvalidLightmap, UVOverlap, DarkPatch
}

public enum IssueSeverity { Info, Warning, Error }
```

## 五、Editor 工具窗口

```csharp
/// <summary>
/// 烘焙管线 Editor 窗口
/// 可视化管理所有烘焙配置和执行状态
/// </summary>
public class BakePipelineWindow : EditorWindow
{
    private List<SceneBakeConfig> _bakeConfigs = new();
    private Vector2 _scrollPos;
    private bool _isRunning;
    private string _statusMessage;
    
    [MenuItem("Tools/Lighting/烘焙管线管理器")]
    public static void ShowWindow()
    {
        GetWindow<BakePipelineWindow>("烘焙管线管理器");
    }
    
    private void OnGUI()
    {
        EditorGUILayout.LabelField("烘焙管线管理器", EditorStyles.boldLabel);
        EditorGUILayout.Space();
        
        // 配置列表
        EditorGUILayout.LabelField("烘焙配置列表:", EditorStyles.boldLabel);
        
        _scrollPos = EditorGUILayout.BeginScrollView(_scrollPos, GUILayout.Height(300));
        
        for (int i = 0; i < _bakeConfigs.Count; i++)
        {
            EditorGUILayout.BeginHorizontal("box");
            
            _bakeConfigs[i] = (SceneBakeConfig)EditorGUILayout.ObjectField(
                _bakeConfigs[i], typeof(SceneBakeConfig), false);
            
            if (_bakeConfigs[i] != null)
            {
                // 显示增量状态
                bool needsBake = _bakeConfigs[i].TargetScenes.Length > 0 &&
                    _bakeConfigs[i].TargetScenes[0] != null &&
                    BakeContentHasher.NeedsRebake(
                        AssetDatabase.GetAssetPath(_bakeConfigs[i].TargetScenes[0]));
                
                GUI.color = needsBake ? Color.yellow : Color.green;
                EditorGUILayout.LabelField(needsBake ? "需要重烘" : "已是最新", 
                    GUILayout.Width(70));
                GUI.color = Color.white;
            }
            
            if (GUILayout.Button("X", GUILayout.Width(20)))
                _bakeConfigs.RemoveAt(i--);
            
            EditorGUILayout.EndHorizontal();
        }
        
        EditorGUILayout.EndScrollView();
        
        if (GUILayout.Button("+ 添加烘焙配置"))
            _bakeConfigs.Add(null);
        
        EditorGUILayout.Space();
        
        // 执行按钮
        EditorGUI.BeginDisabledGroup(_isRunning);
        
        EditorGUILayout.BeginHorizontal();
        
        if (GUILayout.Button("▶ 执行增量烘焙", GUILayout.Height(40)))
            ExecutePipeline(incremental: true);
        
        if (GUILayout.Button("▶▶ 执行全量烘焙", GUILayout.Height(40)))
            ExecutePipeline(incremental: false);
        
        EditorGUILayout.EndHorizontal();
        
        // 工具按钮
        EditorGUILayout.BeginHorizontal();
        
        if (GUILayout.Button("清除所有缓存哈希"))
            ClearAllHashes();
        
        if (GUILayout.Button("验证所有场景 UV"))
            ValidateAllUVs();
        
        if (GUILayout.Button("查看最新报告"))
            OpenLatestReport();
        
        EditorGUILayout.EndHorizontal();
        
        EditorGUI.EndDisabledGroup();
        
        // 状态信息
        if (!string.IsNullOrEmpty(_statusMessage))
        {
            EditorGUILayout.HelpBox(_statusMessage, MessageType.Info);
        }
    }
    
    private void ExecutePipeline(bool incremental)
    {
        _isRunning = true;
        
        var config = new BakePipelineConfig { ContinueOnError = true };
        var scheduler = new BakePipelineScheduler(config);
        
        foreach (var bakeConfig in _bakeConfigs)
        {
            if (bakeConfig == null) continue;
            
            if (!incremental)
            {
                // 全量烘焙：清除哈希，强制重烘
                foreach (var scene in bakeConfig.TargetScenes)
                {
                    if (scene != null)
                    {
                        string path = AssetDatabase.GetAssetPath(scene);
                        string hashFile = $"Library/LightmapHashes/{scene.name}.hash";
                        if (System.IO.File.Exists(hashFile))
                            System.IO.File.Delete(hashFile);
                    }
                }
            }
            
            scheduler.AddTask(bakeConfig);
        }
        
        // 注意：实际项目中应使用异步方式，避免阻塞 UI
        scheduler.ExecuteAll();
        
        _isRunning = false;
        _statusMessage = $"烘焙完成！{DateTime.Now:HH:mm:ss}";
        Repaint();
    }
    
    private void ClearAllHashes()
    {
        string hashDir = "Library/LightmapHashes";
        if (System.IO.Directory.Exists(hashDir))
        {
            System.IO.Directory.Delete(hashDir, recursive: true);
            _statusMessage = "已清除所有烘焙缓存哈希";
        }
    }
    
    private void ValidateAllUVs()
    {
        var meshRenderers = FindObjectsOfType<MeshRenderer>();
        int issueCount = 0;
        
        foreach (var mr in meshRenderers)
        {
            if (!GameObjectUtility.AreStaticEditorFlagsSet(mr.gameObject, StaticEditorFlags.ContributeGI))
                continue;
            
            var mf = mr.GetComponent<MeshFilter>();
            if (mf?.sharedMesh != null && mf.sharedMesh.uv2.Length == 0)
            {
                Debug.LogWarning($"缺少 UV2: {mr.name}", mr.gameObject);
                issueCount++;
            }
        }
        
        _statusMessage = issueCount == 0 ? "UV 验证通过！" : $"发现 {issueCount} 个 UV 问题，请查看 Console";
    }
    
    private void OpenLatestReport()
    {
        var reports = System.IO.Directory.GetFiles("Library", "BakeReport_*.txt");
        if (reports.Length == 0) { _statusMessage = "未找到烘焙报告"; return; }
        
        Array.Sort(reports);
        EditorUtility.RevealInFinder(reports[^1]);
    }
}
```

## 六、CI/CD 集成（批处理模式）

### 6.1 命令行烘焙入口

```csharp
/// <summary>
/// Unity Batch Mode 下的烘焙入口
/// 通过 -executeMethod 参数在 CI/CD 流水线中调用
/// </summary>
public static class BatchModeBakeEntryPoint
{
    /// <summary>
    /// CI 全量烘焙入口
    /// 命令: Unity -batchmode -executeMethod BatchModeBakeEntryPoint.BakeAll -quit
    /// </summary>
    [MenuItem("Tools/CI/执行全量烘焙")]
    public static void BakeAll()
    {
        UnityEngine.Debug.Log("[CI-Bake] 开始全量烘焙...");
        
        // 从命令行参数读取配置路径
        string configPath = GetArgValue("-bakeConfigPath", "Assets/LightingConfigs");
        
        var configs = AssetDatabase.FindAssets("t:SceneBakeConfig", new[] { configPath });
        var scheduler = new BakePipelineScheduler(new BakePipelineConfig());
        
        foreach (var guid in configs)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            var config = AssetDatabase.LoadAssetAtPath<SceneBakeConfig>(path);
            if (config != null)
                scheduler.AddTask(config);
        }
        
        scheduler.ExecuteAll();
        
        // CI 环境：将结果写入文件供流水线读取
        WriteCIResult();
    }
    
    /// <summary>
    /// CI 增量烘焙入口
    /// </summary>
    public static void BakeIncremental()
    {
        // 与 BakeAll 相同，但依赖 SceneBakeConfig 的 EnableIncrementalBake 设置
        BakeAll();
    }
    
    private static void WriteCIResult()
    {
        // 输出 JSON 格式的结果供 CI 系统解析
        var result = new
        {
            timestamp = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss"),
            status = "completed",
            unityVersion = Application.unityVersion
        };
        
        string json = UnityEngine.JsonUtility.ToJson(result);
        System.IO.File.WriteAllText("Library/bake_result.json", json);
    }
    
    private static string GetArgValue(string argName, string defaultValue = "")
    {
        var args = System.Environment.GetCommandLineArgs();
        for (int i = 0; i < args.Length - 1; i++)
        {
            if (args[i] == argName)
                return args[i + 1];
        }
        return defaultValue;
    }
}
```

### 6.2 Jenkins/GitHub Actions 配置示例

```yaml
# .github/workflows/lightmap-bake.yml
name: Nightly Lightmap Bake

on:
  schedule:
    - cron: '0 2 * * *'  # 每天凌晨2点执行
  workflow_dispatch:       # 支持手动触发

jobs:
  bake:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
        with:
          lfs: true
          
      - name: 执行增量烘焙
        run: |
          /opt/unity/Editor/Unity \
            -batchmode \
            -projectPath ${{ github.workspace }} \
            -executeMethod BatchModeBakeEntryPoint.BakeIncremental \
            -bakeConfigPath Assets/LightingConfigs \
            -logFile bake_log.txt \
            -quit
          
      - name: 检查烘焙结果
        run: |
          if grep -q "烘焙失败" bake_log.txt; then
            echo "❌ 存在烘焙失败场景"
            exit 1
          fi
          echo "✅ 烘焙完成"
          
      - name: 提交 Lightmap 到 Git LFS
        run: |
          git config user.name "CI Bot"
          git add Assets/LightmapData/
          git add Library/LightmapHashes/
          git diff --staged --quiet || git commit -m "chore: 自动更新 Lightmap [skip ci]"
          git push
          
      - name: 上传烘焙报告
        uses: actions/upload-artifact@v3
        with:
          name: bake-reports
          path: Library/BakeReport_*.txt
```

## 七、团队协作规范

### 7.1 Git LFS 配置（必须）

```bash
# .gitattributes 中添加 Lightmap 文件追踪
*.exr filter=lfs diff=lfs merge=lfs -text
*.hdr filter=lfs diff=lfs merge=lfs -text
LightingData.asset filter=lfs diff=lfs merge=lfs -text

# 场景文件正常 git 追踪（不用 LFS，需要 merge）
*.unity -text merge=unityyamlmerge diff=.unity

# Lightmap 哈希文件也要提交（用于 CI 增量判断）
Library/LightmapHashes/*.hash -text
```

## 八、常见问题与解决

| 问题 | 原因 | 解决方案 |
|------|------|---------|
| 增量烘焙未触发 | 哈希未变化 | 手动删除 `Library/LightmapHashes/*.hash` |
| Lightmap 接缝 | Texels Per Unit 不一致 | 统一所有场景 MeshRenderer 的 Scale In Lightmap |
| 烘焙很久没进度 | GPU 驱动崩溃 | 切换到 CPU Lightmapper，添加 -force-gfx-direct 参数 |
| CI 烘焙比本地慢10倍 | CI 机器无 GPU | 使用 CPU Progressive，降低采样数 |
| 场景切换后黑屏 | LightmapData 未随场景正确加载 | 检查 LightingDataAsset 绑定，使用 LightingSettings.lightingDataAsset |

## 九、总结

烘焙工程化的核心价值链：

```
内容哈希 → 增量判断 → 跳过未变更 → 节省 70%+ 烘焙时间
     ↓
质量验证 → 提前发现 UV/漏光问题 → 减少返工
     ↓
自动化管线 → 夜间 CI 烘焙 → 第二天直接可用结果
     ↓
Git LFS + 统一规范 → 消灭团队协作冲突
```

把烘焙从"艺术流程"变成"工程流水线"，是大型游戏项目降本增效、保证品质一致性的关键工程实践。
