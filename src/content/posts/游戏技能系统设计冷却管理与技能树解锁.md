## 一、技能数据配置

```csharp
using System;
using UnityEngine;

[CreateAssetMenu(menuName = "Game/Skill")]
public class SkillData : ScriptableObject
{
    [Header("基础")]
    public string SkillId;
    public string SkillName;
    [TextArea] public string Description;
    public Sprite Icon;
    public SkillType Type;
    
    [Header("消耗")]
    public float ManaCost;
    public float Cooldown;          // 冷却时间（秒）
    public float CastTime;          // 吟唱时间（0=即时）
    
    [Header("效果")]
    public float Damage;
    public float Range;
    public float AreaRadius;
    public float Duration;
    
    [Header("技能树")]
    public SkillData[] Prerequisites;  // 前置技能
    public int SkillPointCost = 1;
    public int MaxLevel = 5;
    
    [Header("视觉")]
    public GameObject CastVFX;
    public GameObject HitVFX;
    public AudioClip CastSFX;
}

public enum SkillType { Active, Passive, Toggle, Channel }
```

## 二、冷却管理器

```csharp
public class SkillCooldownManager : MonoBehaviour
{
    private Dictionary<string, float> cooldownEndTimes = new Dictionary<string, float>();
    
    public bool IsOnCooldown(string skillId)
    {
        return cooldownEndTimes.TryGetValue(skillId, out float endTime) 
            && Time.time < endTime;
    }
    
    public float GetRemainingCooldown(string skillId)
    {
        if (!cooldownEndTimes.TryGetValue(skillId, out float endTime)) return 0;
        return Mathf.Max(0, endTime - Time.time);
    }
    
    public float GetCooldownProgress(string skillId, float totalCooldown)
    {
        if (totalCooldown <= 0) return 1f;
        float remaining = GetRemainingCooldown(skillId);
        return 1f - (remaining / totalCooldown);
    }
    
    public void StartCooldown(string skillId, float duration)
    {
        cooldownEndTimes[skillId] = Time.time + duration;
    }
    
    public void ResetCooldown(string skillId)
    {
        cooldownEndTimes.Remove(skillId);
    }
    
    public void ReduceAllCooldowns(float seconds)
    {
        var keys = new List<string>(cooldownEndTimes.Keys);
        foreach (var key in keys)
            cooldownEndTimes[key] -= seconds;
    }
}
```

## 三、技能执行器

```csharp
public class SkillExecutor : MonoBehaviour
{
    [SerializeField] private SkillCooldownManager cooldownMgr;
    [SerializeField] private CharacterAttributes attributes;
    
    public event Action<SkillData> OnSkillCast;
    public event Action<string> OnSkillFailed;

    public bool TryCastSkill(SkillData skill, Vector3 targetPos, GameObject target = null)
    {
        // 检查冷却
        if (cooldownMgr.IsOnCooldown(skill.SkillId))
        {
            OnSkillFailed?.Invoke("冷却中");
            return false;
        }
        
        // 检查魔力
        float currentMP = attributes?.GetValue(StatId.MaxMP) ?? 0;
        if (currentMP < skill.ManaCost)
        {
            OnSkillFailed?.Invoke("魔力不足");
            return false;
        }
        
        // 执行技能
        ExecuteSkill(skill, targetPos, target);
        cooldownMgr.StartCooldown(skill.SkillId, skill.Cooldown);
        OnSkillCast?.Invoke(skill);
        
        return true;
    }

    void ExecuteSkill(SkillData skill, Vector3 targetPos, GameObject target)
    {
        // 播放施法特效
        if (skill.CastVFX != null)
            Instantiate(skill.CastVFX, transform.position, Quaternion.identity);
        
        if (skill.CastSFX != null)
            AudioManager.Instance?.PlaySFX(skill.CastSFX);
        
        // 根据类型执行不同效果
        switch (skill.Type)
        {
            case SkillType.Active:
                if (skill.AreaRadius > 0)
                    ApplyAreaEffect(skill, targetPos);
                else if (target != null)
                    ApplySingleTarget(skill, target);
                break;
        }
    }

    void ApplyAreaEffect(SkillData skill, Vector3 center)
    {
        var hits = Physics.OverlapSphere(center, skill.AreaRadius);
        foreach (var hit in hits)
        {
            var health = hit.GetComponent<HealthComponent>();
            if (health != null)
                health.TakeDamage((int)skill.Damage, DamageType.Magic, gameObject);
        }
        
        if (skill.HitVFX != null)
            Instantiate(skill.HitVFX, center, Quaternion.identity);
    }

    void ApplySingleTarget(SkillData skill, GameObject target)
    {
        var health = target.GetComponent<HealthComponent>();
        health?.TakeDamage((int)skill.Damage, DamageType.Magic, gameObject);
    }
}
```
