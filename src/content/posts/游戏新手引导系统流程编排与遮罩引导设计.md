## 一、引导步骤定义

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 引导步骤
/// </summary>
[Serializable]
public class TutorialStep
{
    public string StepId;
    
    [Header("触发条件")]
    public string TriggerEvent;       // 触发此步骤的事件名
    public string[] RequiredPrevSteps; // 必须完成的前置步骤
    
    [Header("引导内容")]
    public TutorialStepType StepType;
    public string TargetUIPath;        // 高亮目标的UI路径（如"Canvas/HUD/AttackButton"）
    public Vector2 TargetOffset;       // 偏移
    public string InstructionText;     // 引导文字
    public Sprite InstructionIcon;
    
    [Header("引导交互")]
    public bool IsSkippable;           // 是否可跳过
    public bool WaitForAction;         // 是否等待玩家完成指定操作
    public string CompletionEvent;     // 完成该步骤的事件
    public float AutoCompleteDelay;    // 自动完成延迟（0=不自动）
    
    [Header("遮罩配置")]
    public bool ShowMask = true;       // 是否显示遮罩
    public float HighlightPadding = 10f;
    
    [Header("手势动画")]
    public GestureType Gesture;
    public Vector2 GestureTarget;
}

public enum TutorialStepType
{
    HighlightUI,         // 高亮UI元素
    ShowMessage,         // 显示消息气泡
    ForceClick,          // 强制点击某按钮
    WaitForGameEvent,    // 等待游戏事件
    PlayCutscene,        // 播放过场动画
    FreePlay,            // 自由操作期
}

public enum GestureType { None, Tap, SwipeLeft, SwipeRight, SwipeUp, Hold, Pinch }

/// <summary>
/// 引导流程（一组步骤）
/// </summary>
[CreateAssetMenu(fileName = "Tutorial", menuName = "Game/Tutorial Flow")]
public class TutorialFlow : ScriptableObject
{
    public string FlowId;
    public string FlowName;
    public int Priority;              // 优先级（多个引导同时触发时，高优先级先执行）
    public TutorialStep[] Steps;
    
    public TutorialStep GetStep(string stepId) =>
        System.Array.Find(Steps, s => s.StepId == stepId);
}
```

## 二、引导管理器

```csharp
/// <summary>
/// 新手引导管理器
/// </summary>
public class TutorialManager : MonoBehaviour
{
    private static TutorialManager instance;
    public static TutorialManager Instance => instance;

    [SerializeField] private TutorialFlow[] allFlows;
    [SerializeField] private TutorialMaskUI maskUI;
    [SerializeField] private TutorialTooltipUI tooltipUI;

    private HashSet<string> completedSteps = new HashSet<string>();
    private TutorialFlow activeFlow;
    private int currentStepIndex;
    private bool isTutorialActive;
    
    public event Action<TutorialStep> OnStepStarted;
    public event Action<string> OnStepCompleted;  // stepId
    public event Action<string> OnFlowCompleted;  // flowId

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
        LoadProgress();
    }

    /// <summary>
    /// 触发引导事件（游戏各处调用）
    /// </summary>
    public void TriggerEvent(string eventName)
    {
        if (isTutorialActive) return;
        
        // 查找匹配的引导步骤
        foreach (var flow in allFlows)
        {
            foreach (var step in flow.Steps)
            {
                if (step.TriggerEvent == eventName && 
                    !completedSteps.Contains(step.StepId) &&
                    ArePrerequisitesMet(step))
                {
                    StartFlow(flow, step.StepId);
                    return;
                }
            }
        }
    }

    bool ArePrerequisitesMet(TutorialStep step)
    {
        if (step.RequiredPrevSteps == null) return true;
        
        foreach (var req in step.RequiredPrevSteps)
            if (!completedSteps.Contains(req)) return false;
        
        return true;
    }

    void StartFlow(TutorialFlow flow, string startStepId)
    {
        activeFlow = flow;
        isTutorialActive = true;
        
        currentStepIndex = System.Array.FindIndex(
            flow.Steps, s => s.StepId == startStepId);
        
        if (currentStepIndex < 0) currentStepIndex = 0;
        
        StartStep(flow.Steps[currentStepIndex]);
    }

    void StartStep(TutorialStep step)
    {
        OnStepStarted?.Invoke(step);
        
        // 显示遮罩
        if (step.ShowMask)
        {
            var targetTransform = FindUITarget(step.TargetUIPath);
            maskUI?.ShowMask(targetTransform, step.HighlightPadding);
        }
        
        // 显示提示文字
        if (!string.IsNullOrEmpty(step.InstructionText))
        {
            tooltipUI?.Show(step.InstructionText, step.InstructionIcon, 
                FindUITarget(step.TargetUIPath));
        }
        
        // 手势动画
        if (step.Gesture != GestureType.None)
            maskUI?.PlayGesture(step.Gesture, step.GestureTarget);
        
        // 自动完成
        if (step.AutoCompleteDelay > 0)
            Invoke(nameof(AutoCompleteCurrentStep), step.AutoCompleteDelay);
    }

    void AutoCompleteCurrentStep()
    {
        CompleteCurrentStep();
    }

    /// <summary>
    /// 完成当前步骤（外部调用，如玩家点击了目标按钮）
    /// </summary>
    public void CompleteCurrentStep()
    {
        if (!isTutorialActive || activeFlow == null) return;
        
        var step = activeFlow.Steps[currentStepIndex];
        completedSteps.Add(step.StepId);
        
        OnStepCompleted?.Invoke(step.StepId);
        
        // 隐藏当前UI
        maskUI?.HideMask();
        tooltipUI?.Hide();
        
        // 前进到下一步
        currentStepIndex++;
        
        if (currentStepIndex >= activeFlow.Steps.Length)
        {
            // 引导流程完成
            isTutorialActive = false;
            OnFlowCompleted?.Invoke(activeFlow.FlowId);
            SaveProgress();
        }
        else
        {
            StartStep(activeFlow.Steps[currentStepIndex]);
        }
    }

    RectTransform FindUITarget(string path)
    {
        if (string.IsNullOrEmpty(path)) return null;
        
        var go = GameObject.Find(path);
        return go?.GetComponent<RectTransform>();
    }

    void SaveProgress()
    {
        var ids = new List<string>(completedSteps);
        PlayerPrefs.SetString("tutorial_completed_steps", 
            string.Join(",", ids));
        PlayerPrefs.Save();
    }

    void LoadProgress()
    {
        string saved = PlayerPrefs.GetString("tutorial_completed_steps", "");
        if (!string.IsNullOrEmpty(saved))
        {
            foreach (var id in saved.Split(','))
                if (!string.IsNullOrEmpty(id))
                    completedSteps.Add(id);
        }
    }
}
```

## 三、遮罩UI组件

```csharp
/// <summary>
/// 引导遮罩UI（高亮目标区域，屏蔽其他区域点击）
/// </summary>
public class TutorialMaskUI : MonoBehaviour
{
    [SerializeField] private Canvas maskCanvas;
    [SerializeField] private UnityEngine.UI.Image fullMask;     // 全屏半透明遮罩
    [SerializeField] private RectTransform highlightHole;       // 高亮区域（镂空）
    [SerializeField] private UnityEngine.UI.Image gesturePrefab; // 手势动画预制体
    
    private UnityEngine.UI.Image gestureInstance;

    public void ShowMask(RectTransform targetUI, float padding = 10f)
    {
        if (targetUI == null)
        {
            fullMask.gameObject.SetActive(true);
            highlightHole.gameObject.SetActive(false);
            return;
        }
        
        // 计算高亮区域大小
        Vector2 size = targetUI.rect.size + Vector2.one * padding * 2;
        Vector2 pos;
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            maskCanvas.GetComponent<RectTransform>(),
            RectTransformUtility.WorldToScreenPoint(null, targetUI.position),
            null, out pos);
        
        highlightHole.sizeDelta = size;
        highlightHole.anchoredPosition = pos;
        highlightHole.gameObject.SetActive(true);
        fullMask.gameObject.SetActive(true);
    }

    public void HideMask()
    {
        fullMask.gameObject.SetActive(false);
        highlightHole.gameObject.SetActive(false);
        HideGesture();
    }

    public void PlayGesture(GestureType type, Vector2 position)
    {
        // 播放对应手势动画
        if (gestureInstance == null)
            gestureInstance = Instantiate(gesturePrefab, maskCanvas.transform);
        
        gestureInstance.gameObject.SetActive(true);
        gestureInstance.rectTransform.anchoredPosition = position;
        
        string animName = type.ToString();
        gestureInstance.GetComponent<Animator>()?.Play(animName);
    }

    public void HideGesture()
    {
        if (gestureInstance != null)
            gestureInstance.gameObject.SetActive(false);
    }
}
```

## 四、引导系统设计要点

| 要点 | 方案 |
|------|------|
| 数据驱动 | ScriptableObject配置引导流程，策划可修改 |
| 前置依赖 | 支持引导步骤间的依赖关系 |
| 存档断点续接 | 保存已完成步骤，续接未完成的引导 |
| 跳过机制 | 老玩家可跳过，新玩家不可跳关键步骤 |
| 遮罩镂空 | 只有高亮区域可点击，其他区域遮挡 |
| 事件驱动 | 引导在游戏事件发生时触发，而非固定时序 |
