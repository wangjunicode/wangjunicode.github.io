```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 任务配置（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "Quest", menuName = "Game/Quest")]
public class QuestData : ScriptableObject
{
    [Header("基础信息")]
    public string QuestId;
    public string Title;
    [TextArea(3, 6)] public string Description;
    public QuestType Type;         // 主线/支线/日常/限时
    public int RecommendedLevel;
    
    [Header("前置条件")]
    public string[] RequiredQuestIds;    // 前置任务（需完成这些才能开始）
    public int RequiredPlayerLevel;
    
    [Header("任务目标")]
    public QuestObjective[] Objectives;
    
    [Header("奖励")]
    public QuestReward[] Rewards;
    
    [Header("时限")]
    public float TimeLimit;              // 时间限制（秒），0=无限制
    
    [Header("NPC")]
    public string GiverNPCId;            // 接任务NPC
    public string TurnInNPCId;           // 交任务NPC
    
    [Header("追踪")]
    public bool AutoTrack = true;        // 自动追踪到任务追踪器
    public QuestData[] NextQuests;       // 完成后触发的后续任务（任务链）
}

public enum QuestType { MainStory, SideQuest, Daily, Weekly, Timed, Hidden }

/// <summary>
/// 任务目标
/// </summary>
[Serializable]
public class QuestObjective
{
    public string ObjectiveId;
    [TextArea(1, 2)] public string Description;
    public ObjectiveType Type;
    public string TargetId;        // 击杀/收集/到达的目标ID
    public int RequiredCount;      // 需要数量
    public bool IsOptional;        // 可选目标
    public bool IsHidden;          // 隐藏目标（完成后才显示）
}

public enum ObjectiveType
{
    KillEnemy,      // 击杀怪物
    CollectItem,    // 收集物品
    ReachLocation,  // 到达地点
    TalkToNPC,      // 与NPC对话
    DefendLocation, // 守护地点
    EscortNPC,      // 护送NPC
    UseItem,        // 使用物品
    Custom          // 自定义
}

/// <summary>
/// 任务奖励
/// </summary>
[Serializable]
public class QuestReward
{
    public RewardType Type;
    public string ItemId;          // 物品ID（Type=Item时）
    public int Amount;             // 数量/经验/货币
}

public enum RewardType { Experience, Currency, Item, TitleUnlock }
```

## 二、任务管理器

```csharp
/// <summary>
/// 任务状态
/// </summary>
[Serializable]
public class QuestState
{
    public string QuestId;
    public QuestStatus Status;
    public Dictionary<string, int> ObjectiveProgress = new Dictionary<string, int>();
    public float StartTime;
    public float ElapsedTime;
    
    public bool IsObjectiveComplete(QuestObjective obj)
    {
        int progress = ObjectiveProgress.TryGetValue(obj.ObjectiveId, out int p) ? p : 0;
        return progress >= obj.RequiredCount;
    }
}

public enum QuestStatus { NotStarted, InProgress, Completed, Failed, Abandoned }

/// <summary>
/// 任务管理器
/// </summary>
public class QuestManager : MonoBehaviour
{
    private static QuestManager instance;
    public static QuestManager Instance => instance;

    [SerializeField] private QuestData[] allQuests;
    
    private Dictionary<string, QuestData> questDataMap 
        = new Dictionary<string, QuestData>();
    
    private Dictionary<string, QuestState> questStates 
        = new Dictionary<string, QuestState>();
    
    // 事件
    public event Action<QuestData> OnQuestStarted;
    public event Action<QuestData, QuestObjective, int> OnObjectiveProgress; // quest, obj, newCount
    public event Action<QuestData, QuestObjective> OnObjectiveComplete;
    public event Action<QuestData> OnQuestCompleted;
    public event Action<QuestData> OnQuestFailed;

    void Awake()
    {
        instance = this;
        foreach (var quest in allQuests)
            questDataMap[quest.QuestId] = quest;
    }

    void Update()
    {
        // 检查时限任务
        foreach (var state in questStates.Values)
        {
            if (state.Status != QuestStatus.InProgress) continue;
            
            var quest = questDataMap[state.QuestId];
            if (quest.TimeLimit <= 0) continue;
            
            state.ElapsedTime += Time.deltaTime;
            
            if (state.ElapsedTime >= quest.TimeLimit)
            {
                FailQuest(state.QuestId);
            }
        }
    }

    /// <summary>
    /// 检查玩家是否可以接受任务
    /// </summary>
    public bool CanAcceptQuest(string questId)
    {
        if (!questDataMap.TryGetValue(questId, out var quest)) return false;
        
        // 已经接了/完成了
        if (questStates.TryGetValue(questId, out var state))
            return state.Status == QuestStatus.Failed; // 失败可重接（某些任务）
        
        // 检查前置任务
        if (quest.RequiredQuestIds != null)
        {
            foreach (var reqId in quest.RequiredQuestIds)
            {
                if (!IsQuestCompleted(reqId)) return false;
            }
        }
        
        // 检查等级
        if (PlayerDataService.GetLocalPlayerData()?.Level < quest.RequiredPlayerLevel)
            return false;
        
        return true;
    }

    /// <summary>
    /// 接受任务
    /// </summary>
    public bool AcceptQuest(string questId)
    {
        if (!CanAcceptQuest(questId)) return false;
        
        var state = new QuestState
        {
            QuestId = questId,
            Status = QuestStatus.InProgress,
            StartTime = Time.time,
            ElapsedTime = 0
        };
        
        questStates[questId] = state;
        
        OnQuestStarted?.Invoke(questDataMap[questId]);
        
        // 自动追踪
        var quest = questDataMap[questId];
        if (quest.AutoTrack)
            QuestTracker.Instance?.TrackQuest(questId);
        
        return true;
    }

    /// <summary>
    /// 更新任务目标进度
    /// </summary>
    public void UpdateObjectiveProgress(ObjectiveType type, string targetId, int count = 1)
    {
        foreach (var state in questStates.Values)
        {
            if (state.Status != QuestStatus.InProgress) continue;
            
            var quest = questDataMap[state.QuestId];
            
            foreach (var obj in quest.Objectives)
            {
                if (obj.Type != type || obj.TargetId != targetId) continue;
                if (state.IsObjectiveComplete(obj)) continue;
                
                // 更新进度
                if (!state.ObjectiveProgress.ContainsKey(obj.ObjectiveId))
                    state.ObjectiveProgress[obj.ObjectiveId] = 0;
                
                int oldProgress = state.ObjectiveProgress[obj.ObjectiveId];
                int newProgress = Mathf.Min(oldProgress + count, obj.RequiredCount);
                state.ObjectiveProgress[obj.ObjectiveId] = newProgress;
                
                OnObjectiveProgress?.Invoke(quest, obj, newProgress);
                
                if (newProgress >= obj.RequiredCount)
                {
                    OnObjectiveComplete?.Invoke(quest, obj);
                    CheckQuestCompletion(state, quest);
                }
            }
        }
    }

    void CheckQuestCompletion(QuestState state, QuestData quest)
    {
        // 检查所有非可选目标是否完成
        bool allDone = true;
        foreach (var obj in quest.Objectives)
        {
            if (obj.IsOptional) continue;
            if (!state.IsObjectiveComplete(obj))
            {
                allDone = false;
                break;
            }
        }
        
        if (allDone)
            CompleteQuest(state.QuestId);
    }

    /// <summary>
    /// 完成任务（玩家手动交任务）
    /// </summary>
    public bool CompleteQuest(string questId)
    {
        if (!questStates.TryGetValue(questId, out var state)) return false;
        if (state.Status != QuestStatus.InProgress) return false;
        
        state.Status = QuestStatus.Completed;
        
        var quest = questDataMap[questId];
        
        // 发放奖励
        GrantRewards(quest);
        
        OnQuestCompleted?.Invoke(quest);
        
        // 触发后续任务
        if (quest.NextQuests != null)
        {
            foreach (var nextQuest in quest.NextQuests)
                AcceptQuest(nextQuest.QuestId);
        }
        
        return true;
    }

    public void FailQuest(string questId)
    {
        if (!questStates.TryGetValue(questId, out var state)) return;
        state.Status = QuestStatus.Failed;
        OnQuestFailed?.Invoke(questDataMap[questId]);
    }

    void GrantRewards(QuestData quest)
    {
        if (quest.Rewards == null) return;
        
        foreach (var reward in quest.Rewards)
        {
            switch (reward.Type)
            {
                case RewardType.Experience:
                    ExperienceManager.Instance?.AddExp(reward.Amount);
                    break;
                case RewardType.Currency:
                    CurrencyManager.Instance?.OnServerConfirmed(CurrencyType.Gold, 
                        CurrencyManager.Instance.GetBalance(CurrencyType.Gold) + reward.Amount);
                    break;
                case RewardType.Item:
                    // 开奖励物品
                    break;
            }
        }
        
        // 显示奖励UI
        UIManager.Instance?.ShowQuestRewardPanel(quest.Rewards);
    }

    public bool IsQuestCompleted(string questId) =>
        questStates.TryGetValue(questId, out var s) && s.Status == QuestStatus.Completed;

    public QuestState GetQuestState(string questId) =>
        questStates.TryGetValue(questId, out var s) ? s : null;

    public List<QuestState> GetActiveQuests() =>
        new List<QuestState>(questStates.Values)
            .FindAll(s => s.Status == QuestStatus.InProgress);
}
```

## 三、任务日志UI

```csharp
/// <summary>
/// 任务追踪器（屏幕上显示当前追踪任务的进度）
/// </summary>
public class QuestTracker : MonoBehaviour
{
    private static QuestTracker instance;
    public static QuestTracker Instance => instance;

    [SerializeField] private Transform trackingContainer;
    [SerializeField] private GameObject trackingItemPrefab;
    [SerializeField] private int maxTrackedQuests = 3;

    private List<string> trackedQuestIds = new List<string>();
    private Dictionary<string, GameObject> trackingItems = new Dictionary<string, GameObject>();

    void Awake()
    {
        instance = this;
        
        QuestManager.Instance.OnObjectiveProgress += OnObjectiveProgress;
        QuestManager.Instance.OnQuestCompleted += OnQuestCompleted;
    }

    public void TrackQuest(string questId)
    {
        if (trackedQuestIds.Contains(questId)) return;
        
        if (trackedQuestIds.Count >= maxTrackedQuests)
        {
            // 移除最旧的追踪
            string oldest = trackedQuestIds[0];
            UntrackQuest(oldest);
        }
        
        trackedQuestIds.Add(questId);
        AddTrackingUI(questId);
    }

    public void UntrackQuest(string questId)
    {
        trackedQuestIds.Remove(questId);
        if (trackingItems.TryGetValue(questId, out var item))
        {
            Destroy(item);
            trackingItems.Remove(questId);
        }
    }

    void AddTrackingUI(string questId)
    {
        var go = Instantiate(trackingItemPrefab, trackingContainer);
        trackingItems[questId] = go;
        RefreshTrackingItem(questId);
    }

    void RefreshTrackingItem(string questId)
    {
        if (!trackingItems.TryGetValue(questId, out var item)) return;
        
        // 更新UI
        // item.GetComponentInChildren<Text>().text = ...
    }

    void OnObjectiveProgress(QuestData quest, QuestObjective obj, int progress)
    {
        if (trackedQuestIds.Contains(quest.QuestId))
            RefreshTrackingItem(quest.QuestId);
    }

    void OnQuestCompleted(QuestData quest)
    {
        UntrackQuest(quest.QuestId);
    }
}
```

## 四、任务系统设计规范

| 要点 | 方案 |
|------|------|
| 任务前置 | 有向无环图，避免循环依赖 |
| 目标类型 | 枚举扩展点，方便添加新类型 |
| 时限任务 | 独立计时，与服务器时间同步 |
| 奖励分配 | 服务端确认，防止客户端伪造 |
| 任务存档 | 仅保存状态+进度，不保存配置 |
| NPC状态 | 任务完成后NPC对话内容变化 |
