## 一、碰撞矩阵优化

### 问题：默认情况下所有层都互相检测碰撞，浪费大量性能

```
推荐的层（Layer）划分：

Layer 0:  Default
Layer 6:  Player
Layer 7:  Enemy
Layer 8:  PlayerProjectile    ← 玩家子弹
Layer 9:  EnemyProjectile     ← 敌人子弹
Layer 10: Ground
Layer 11: Trigger             ← 触发器（不需要物理碰撞）
Layer 12: UI3D
Layer 13: Interactable        ← 可交互物品
Layer 14: DestructibleProp    ← 可破坏道具
Layer 15: InvisibleWall       ← 隐形墙

碰撞矩阵（在 Physics Settings 中配置）：
PlayerProjectile ↔ Enemy         ✓
PlayerProjectile ↔ Ground        ✓
PlayerProjectile ↔ DestructibleProp ✓
PlayerProjectile ↔ Player        ✗  （不打自己）
PlayerProjectile ↔ PlayerProjectile ✗  （子弹不碰子弹）
EnemyProjectile  ↔ Player        ✓
EnemyProjectile  ↔ Ground        ✓
EnemyProjectile  ↔ Enemy         ✗  （不打自己）
```

## 二、Non-Alloc 物理查询

```csharp
using UnityEngine;

/// <summary>
/// 物理查询优化示例
/// </summary>
public class PhysicsQueryOptimizedV2 : MonoBehaviour
{
    // ============ 预分配缓冲区（在 Awake 中分配，复用）============
    
    private const int MAX_HITS = 32;
    private readonly Collider[] overlapBuffer = new Collider[MAX_HITS];
    private readonly RaycastHit[] raycastBuffer = new RaycastHit[MAX_HITS];
    private readonly RaycastHit2D[] raycast2DBuffer = new RaycastHit2D[MAX_HITS];

    [Header("查询配置")]
    [SerializeField] private LayerMask enemyLayer;
    [SerializeField] private LayerMask groundLayer;
    [SerializeField] private float updateInterval = 0.1f; // 降低查询频率
    
    private float queryTimer;

    void Update()
    {
        queryTimer += Time.deltaTime;
        if (queryTimer < updateInterval) return;
        queryTimer = 0;
        
        // 范围内敌人检测
        int count = Physics.OverlapSphereNonAlloc(
            transform.position, 10f, 
            overlapBuffer, enemyLayer);
        
        for (int i = 0; i < count; i++)
        {
            ProcessEnemy(overlapBuffer[i]);
        }
    }

    // ============ 各种 NonAlloc 查询对比 ============

    int QueryEnemiesInSphere(Vector3 center, float radius)
    {
        // OverlapSphere：球形范围检测，最常用
        return Physics.OverlapSphereNonAlloc(
            center, radius, overlapBuffer, enemyLayer);
    }

    int QueryEnemiesInBox(Vector3 center, Vector3 halfExtents, Quaternion rotation)
    {
        // OverlapBox：矩形范围检测，适合扇形攻击等
        return Physics.OverlapBoxNonAlloc(
            center, halfExtents, overlapBuffer, rotation, enemyLayer);
    }

    int RaycastToGround(Vector3 origin, Vector3 direction, float maxDistance)
    {
        // RaycastAll NonAlloc：获取所有命中点
        return Physics.RaycastNonAlloc(
            origin, direction, raycastBuffer, maxDistance, groundLayer);
    }

    bool CheckGroundBelow(Vector3 origin, out RaycastHit hit, float maxDistance = 2f)
    {
        // 单次 Raycast：最快，只返回第一个碰撞
        return Physics.Raycast(origin, Vector3.down, out hit, maxDistance, groundLayer);
    }

    int SphereCastToEnemy(Vector3 origin, float radius, Vector3 direction, float distance)
    {
        // SphereCast：比 Raycast 更宽，适合子弹/攻击判定
        return Physics.SphereCastNonAlloc(
            origin, radius, direction, raycastBuffer, distance, enemyLayer);
    }

    void ProcessEnemy(Collider col) { }
}
```

## 三、碰撞检测模式选择

```csharp
/// <summary>
/// 刚体碰撞检测模式配置
/// </summary>
public class RigidbodyCollisionSetup : MonoBehaviour
{
    [SerializeField] private Rigidbody rb;

    void Start()
    {
        // 速度很快的子弹：需要 CCD 防穿墙
        // rb.collisionDetectionMode = CollisionDetectionMode.ContinuousDynamic;
        
        // 速度中等的角色：默认即可
        // rb.collisionDetectionMode = CollisionDetectionMode.Discrete;
        
        // 静态障碍物：Static 性能最好
        // rb.collisionDetectionMode = CollisionDetectionMode.Continuous;
    }

    // 性能说明：
    // Discrete:          最快，可能穿墙
    // Continuous:        中等，动态vs静态 CCD
    // ContinuousDynamic: 最慢，动态vs动态 CCD，高速物体必用
    // Speculative:       中等，预测性 CCD，性能较好的替代方案
}

/// <summary>
/// 刚体休眠优化（静止物体停止物理计算）
/// </summary>
public class RigidbodySleepOptimizer : MonoBehaviour
{
    void Start()
    {
        var rb = GetComponent<Rigidbody>();
        if (rb != null)
        {
            // 调整休眠阈值（速度低于此值时进入休眠）
            rb.sleepThreshold = 0.005f;  // 默认0.005
            
            // 强制立即休眠（适合场景中大量不动的刚体）
            rb.Sleep();
        }
    }
}
```

## 四、物理 LOD 系统

```csharp
/// <summary>
/// 物理 LOD：根据与玩家的距离动态调整物理更新频率
/// </summary>
public class PhysicsLOD : MonoBehaviour
{
    [System.Serializable]
    public class LODConfig
    {
        public float Distance;              // 切换到此 LOD 的距离
        public float UpdateInterval;        // 物理更新间隔（0=每帧）
        public bool DisableRigidbody;       // 是否完全禁用刚体
    }

    [SerializeField] private LODConfig[] lodLevels;
    [SerializeField] private float checkInterval = 1f;

    private Rigidbody rb;
    private Transform playerTransform;
    private float checkTimer;
    private float physicsTimer;
    private int currentLOD = 0;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
        playerTransform = GameObject.FindGameObjectWithTag("Player")?.transform;
        
        // 排序（距离从小到大）
        System.Array.Sort(lodLevels, (a, b) => a.Distance.CompareTo(b.Distance));
    }

    void Update()
    {
        checkTimer += Time.deltaTime;
        if (checkTimer >= checkInterval)
        {
            checkTimer = 0;
            UpdateLOD();
        }
        
        // 按 LOD 间隔进行物理更新
        if (currentLOD < lodLevels.Length)
        {
            float interval = lodLevels[currentLOD].UpdateInterval;
            if (interval > 0)
            {
                physicsTimer += Time.deltaTime;
                if (physicsTimer >= interval)
                {
                    physicsTimer = 0;
                    rb?.WakeUp();
                }
            }
        }
    }

    void UpdateLOD()
    {
        if (playerTransform == null || rb == null) return;
        
        float dist = Vector3.Distance(transform.position, playerTransform.position);
        
        int newLOD = lodLevels.Length - 1;
        for (int i = 0; i < lodLevels.Length; i++)
        {
            if (dist <= lodLevels[i].Distance)
            {
                newLOD = i;
                break;
            }
        }
        
        if (newLOD == currentLOD) return;
        currentLOD = newLOD;
        
        if (currentLOD < lodLevels.Length)
        {
            bool disableRb = lodLevels[currentLOD].DisableRigidbody;
            rb.detectCollisions = !disableRb;
            if (disableRb) rb.Sleep();
        }
    }
}
```

## 五、物理性能调优清单

| 优化项 | 效果 | 实施难度 |
|--------|------|----------|
| 配置碰撞矩阵 | 减少不必要的碰撞对检测 | 低 |
| 使用 NonAlloc 查询 | 消除 GC 分配 | 低 |
| 降低查询频率 | 减少 CPU 开销 | 低 |
| 静止刚体休眠 | 大量减少静止物理计算 | 低 |
| 物理 LOD | 远处对象减少更新 | 中 |
| 减少 Fixed Timestep | 默认0.02s，可改0.033s（30Hz）| 低 |
| 合并碰撞体 | 减少碰撞体数量 | 中 |
| 避免 SendMessage | 改用接口/直接引用 | 低 |

**Physics.fixedDeltaTime 优化建议：**
- 默认：0.02s（50Hz物理更新）
- 移动端建议：0.033s（30Hz）省30%物理开销
- 高帧率（120fps游戏）：考虑0.02s保持精确性
