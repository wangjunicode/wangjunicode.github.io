## 一、伤害计算公式设计

```csharp
using UnityEngine;

/// <summary>
/// 战斗伤害计算器
/// </summary>
public static class DamageFormulas
{
    /// <summary>
    /// 线性防御减伤（简单，但高防御时减伤过高）
    /// damage = atk - def
    /// </summary>
    public static float LinearDefense(float atk, float def)
    {
        return Mathf.Max(1f, atk - def); // 最低1点伤害
    }
    
    /// <summary>
    /// 百分比防御减伤（不会减到0，上限有收益递减）
    /// damage = atk × (1 - def/(def + K))
    /// K是常数，决定防御曲线斜率
    /// </summary>
    public static float PercentageDefense(float atk, float def, float k = 100f)
    {
        float damageReduction = def / (def + k); // 0到1之间
        return atk * (1f - damageReduction);
    }
    
    /// <summary>
    /// 对数防御减伤（防御成长曲线更平滑）
    /// </summary>
    public static float LogarithmicDefense(float atk, float def)
    {
        float reduction = Mathf.Log(1f + def / 100f, 2f) * 0.1f;
        reduction = Mathf.Clamp01(reduction);
        return atk * (1f - reduction);
    }
    
    /// <summary>
    /// 暴击伤害计算
    /// </summary>
    public static float ApplyCritical(float baseDamage, float critRate, float critMultiplier)
    {
        bool isCrit = IsCritical(critRate);
        return isCrit ? baseDamage * critMultiplier : baseDamage;
    }
    
    /// <summary>
    /// 伪随机暴击（PRNG - 防止连续不暴击的挫败感）
    /// 随着连续未暴击次数增加，下次暴击的概率递增
    /// </summary>
    private static int noCritStreak; // 未暴击连续次数
    
    public static bool IsCritical(float baseCritRate)
    {
        // 伪随机概率：每次未暴击后，实际暴击率提高
        float actualCritRate = baseCritRate + noCritStreak * baseCritRate * 0.5f;
        actualCritRate = Mathf.Min(actualCritRate, 1f);
        
        if (Random.value < actualCritRate)
        {
            noCritStreak = 0;
            return true;
        }
        else
        {
            noCritStreak++;
            return false;
        }
    }
    
    /// <summary>
    /// 元素反应伤害（类似原神的元素系统）
    /// </summary>
    public static float ElementalReaction(float baseDamage, ElementType attacker, 
        ElementType target)
    {
        float multiplier = GetReactionMultiplier(attacker, target);
        return baseDamage * multiplier;
    }
    
    static float GetReactionMultiplier(ElementType atk, ElementType def)
    {
        // 相克关系表
        if (atk == ElementType.Fire && def == ElementType.Ice)   return 2.0f;
        if (atk == ElementType.Water && def == ElementType.Fire) return 1.5f;
        if (atk == ElementType.Thunder && def == ElementType.Water) return 2.0f;
        return 1.0f; // 无反应
    }
}

public enum ElementType { None, Fire, Water, Thunder, Ice, Wind, Rock }
```

## 二、成长曲线设计

```csharp
/// <summary>
/// 角色成长曲线（等级→属性值）
/// </summary>
public static class GrowthCurves
{
    /// <summary>
    /// 线性成长：每级固定增加
    /// </summary>
    public static float Linear(float baseValue, float perLevelGain, int level)
    {
        return baseValue + perLevelGain * (level - 1);
    }
    
    /// <summary>
    /// 指数成长：后期成长加速（高风险，容易过强）
    /// </summary>
    public static float Exponential(float baseValue, float growthRate, int level)
    {
        return baseValue * Mathf.Pow(growthRate, level - 1);
    }
    
    /// <summary>
    /// S型成长（早期慢，中期快，后期趋于平缓）
    /// 模拟真实成长曲线
    /// </summary>
    public static float SCurve(float minValue, float maxValue, int level, 
        int maxLevel, float steepness = 0.5f)
    {
        float t = (float)(level - 1) / (maxLevel - 1);
        float sigmoid = 1f / (1f + Mathf.Exp(-steepness * 10f * (t - 0.5f)));
        return Mathf.Lerp(minValue, maxValue, sigmoid);
    }
    
    /// <summary>
    /// 分段成长（前期快，后期慢——常见于手游平衡）
    /// </summary>
    public static float Piecewise(float baseValue, int level, 
        AnimationCurve growthCurve, float maxBonus = 500f)
    {
        float t = (float)level / 100f; // 假设最大等级100
        return baseValue + growthCurve.Evaluate(t) * maxBonus;
    }
}

/// <summary>
/// 数值平衡工具（编辑器中快速验证TTK）
/// </summary>
public class CombatSimulator
{
    /// <summary>
    /// 模拟战斗，计算TTK（Time To Kill，击杀所需时间）
    /// </summary>
    public static float SimulateTTK(
        float attackerAtk, float attackerCritRate, float attackerCritMult,
        float targetHp, float targetDef,
        float attacksPerSecond)
    {
        float totalDamage = 0;
        float time = 0;
        float attackInterval = 1f / attacksPerSecond;
        int maxSimFrames = 10000;
        
        for (int i = 0; i < maxSimFrames; i++)
        {
            // 计算单次攻击伤害
            float rawDamage = DamageFormulas.PercentageDefense(attackerAtk, targetDef);
            float finalDamage = DamageFormulas.ApplyCritical(
                rawDamage, attackerCritRate, attackerCritMult);
            
            totalDamage += finalDamage;
            time += attackInterval;
            
            if (totalDamage >= targetHp)
                break;
        }
        
        return time;
    }
    
    /// <summary>
    /// 蒙特卡罗法：模拟N次战斗，统计平均TTK和方差（反映数值稳定性）
    /// </summary>
    public static (float avgTTK, float stdDev) MonteCarloTTK(
        float atk, float critRate, float critMult,
        float targetHp, float targetDef,
        float aps, int iterations = 1000)
    {
        float sumTTK = 0;
        float sumTTKSquared = 0;
        
        for (int i = 0; i < iterations; i++)
        {
            float ttk = SimulateTTK(atk, critRate, critMult, targetHp, targetDef, aps);
            sumTTK += ttk;
            sumTTKSquared += ttk * ttk;
        }
        
        float avgTTK = sumTTK / iterations;
        float variance = sumTTKSquared / iterations - avgTTK * avgTTK;
        float stdDev = Mathf.Sqrt(variance);
        
        return (avgTTK, stdDev);
    }
}
```

## 三、防御减伤曲线对比

```
防御值 vs 伤害减免（K=100）：

防御    | 百分比减伤
--------|----------
0       | 0%
50      | 33%
100     | 50%
200     | 67%
300     | 75%
400     | 80%
900     | 90%
∞       | 100%（渐近线）

结论：
- 收益递减型防御：不会有无敌坦克，但堆防御始终有价值
- 每增加相同防御值，减伤越来越少（良好的平衡曲线）
```
