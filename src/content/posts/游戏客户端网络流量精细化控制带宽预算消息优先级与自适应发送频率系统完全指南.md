# 游戏客户端网络流量精细化控制：带宽预算、消息优先级与自适应发送频率系统完全指南

## 一、为什么需要精细化流量控制

移动游戏面临的网络环境极为复杂：4G/5G 信号波动、Wi-Fi 干扰、隧道/地铁断连……粗暴地"有数据就发"往往导致三类问题：

| 问题 | 表象 | 根因 |
|------|------|------|
| 流量爆炸 | 月流量超标，玩家投诉 | 高频无差别广播 |
| 延迟跳变 | 操作响应迟钝、橡皮带 | 瞬时发送量超过链路容量 |
| 丢包雪崩 | 战斗卡死、位置回跳 | 拥塞后 TCP 队列堆积/UDP 被丢弃 |

精细化流量控制的核心目标：**在有限带宽内，以最低延迟传递最重要的信息**。

## 三、消息优先级分类系统

### 3.1 优先级定义

```csharp
/// <summary>
/// 网络消息优先级枚举
/// 数字越小优先级越高，调度时严格按优先级顺序发送
/// </summary>
public enum MessagePriority : byte
{
    /// <summary>关键控制消息：登录/心跳/断线重连，不可延迟，不可丢弃</summary>
    Critical = 0,
    
    /// <summary>高优先级：战斗指令/技能释放/碰撞事件，延迟 < 50ms</summary>
    High = 1,
    
    /// <summary>普通：位置同步/状态更新，延迟 < 200ms</summary>
    Normal = 2,
    
    /// <summary>低优先级：聊天/UI事件/非战斗数据，可延迟 < 1s</summary>
    Low = 3,
    
    /// <summary>批量传输：资源预加载/日志上报，带宽空闲时发送</summary>
    Bulk = 4
}

/// <summary>
/// 消息优先级配置元数据，通过 Attribute 标记协议
/// </summary>
[AttributeUsage(AttributeTargets.Class | AttributeTargets.Struct)]
public sealed class MessagePriorityAttribute : Attribute
{
    public MessagePriority Priority { get; }
    public bool Reliable { get; }          // 是否需要可靠传输
    public bool AllowMerge { get; }        // 是否允许与相邻消息合并
    public int MaxDelayMs { get; }         // 最大允许延迟（毫秒）
    
    public MessagePriorityAttribute(
        MessagePriority priority,
        bool reliable = true,
        bool allowMerge = false,
        int maxDelayMs = 0)
    {
        Priority = priority;
        Reliable = reliable;
        AllowMerge = allowMerge;
        MaxDelayMs = maxDelayMs;
    }
}

// 使用示例
[MessagePriority(MessagePriority.Critical, reliable: true)]
public struct HeartbeatMessage { public long Timestamp; }

[MessagePriority(MessagePriority.High, reliable: true, maxDelayMs: 50)]
public struct BattleCommandMessage { public int SkillId; public long TargetId; }

[MessagePriority(MessagePriority.Normal, reliable: false, allowMerge: true, maxDelayMs: 100)]
public struct PositionSyncMessage { public long EntityId; public float X, Y, Z; }

[MessagePriority(MessagePriority.Low, reliable: true, maxDelayMs: 1000)]
public struct ChatMessage { public string Content; }

[MessagePriority(MessagePriority.Bulk, reliable: true)]
public struct LogUploadMessage { public byte[] Data; }
```

### 3.2 优先级队列实现

```csharp
using System;
using System.Collections.Generic;
using System.Threading;

/// <summary>
/// 多级优先级发送队列
/// 支持抢占式调度：高优先级消息可插队到低优先级消息之前
/// </summary>
public sealed class PriorityMessageQueue
{
    // 每个优先级一个独立队列
    private readonly Queue<PendingMessage>[] _queues;
    private readonly int _priorityCount;
    private int _totalCount;
    
    // 饥饿防护：低优先级消息等待超时后强制提升
    private readonly long[] _starvationThresholdTicks;
    
    public int Count => _totalCount;
    
    public PriorityMessageQueue(int priorityLevels = 5)
    {
        _priorityCount = priorityLevels;
        _queues = new Queue<PendingMessage>[priorityLevels];
        _starvationThresholdTicks = new long[priorityLevels];
        
        for (int i = 0; i < priorityLevels; i++)
        {
            _queues[i] = new Queue<PendingMessage>();
            // 优先级越低，饥饿阈值越小（更快被提升）
            _starvationThresholdTicks[i] = TimeSpan.FromMilliseconds(200 * (i + 1)).Ticks;
        }
    }
    
    public void Enqueue(PendingMessage message)
    {
        int priority = Mathf.Clamp((int)message.Priority, 0, _priorityCount - 1);
        _queues[priority].Enqueue(message);
        Interlocked.Increment(ref _totalCount);
    }
    
    /// <summary>
    /// 按优先级取出消息，带饥饿防护
    /// </summary>
    public bool TryDequeue(out PendingMessage message)
    {
        long now = DateTime.UtcNow.Ticks;
        
        // 第一步：检查低优先级是否有超时消息（饥饿防护）
        for (int p = _priorityCount - 1; p > 0; p--)
        {
            if (_queues[p].Count > 0)
            {
                var peek = _queues[p].Peek();
                if (now - peek.EnqueueTicks > _starvationThresholdTicks[p])
                {
                    message = _queues[p].Dequeue();
                    Interlocked.Decrement(ref _totalCount);
                    return true;
                }
            }
        }
        
        // 第二步：按优先级顺序取消息
        for (int p = 0; p < _priorityCount; p++)
        {
            if (_queues[p].Count > 0)
            {
                message = _queues[p].Dequeue();
                Interlocked.Decrement(ref _totalCount);
                return true;
            }
        }
        
        message = default;
        return false;
    }
    
    /// <summary>
    /// 按优先级批量取出，受带宽预算限制
    /// </summary>
    public List<PendingMessage> DequeueBatch(int maxBytes)
    {
        var batch = new List<PendingMessage>(16);
        int consumed = 0;
        long now = DateTime.UtcNow.Ticks;
        
        for (int p = 0; p < _priorityCount && consumed < maxBytes; p++)
        {
            while (_queues[p].Count > 0 && consumed < maxBytes)
            {
                var peek = _queues[p].Peek();
                
                // 检查是否超过最大延迟（即使没预算也必须发）
                bool mustSend = peek.MaxDelayMs > 0 &&
                    now - peek.EnqueueTicks > TimeSpan.FromMilliseconds(peek.MaxDelayMs).Ticks;
                
                if (consumed + peek.EstimatedBytes > maxBytes && !mustSend)
                    break;
                
                batch.Add(_queues[p].Dequeue());
                Interlocked.Decrement(ref _totalCount);
                consumed += peek.EstimatedBytes;
            }
        }
        
        return batch;
    }
}

public struct PendingMessage
{
    public MessagePriority Priority;
    public byte[] Data;
    public int EstimatedBytes;
    public int MaxDelayMs;
    public long EnqueueTicks;
    public bool Reliable;
}
```

## 四、带宽预算管理器

### 4.1 分层带宽预算

```csharp
/// <summary>
/// 带宽预算管理器
/// 为每个优先级分配固定带宽配额，高优先级可借用低优先级配额
/// </summary>
public sealed class BandwidthBudgetManager
{
    private readonly BandwidthConfig _config;
    private readonly BandwidthMeasurement _measurement;
    
    // 每个优先级的预算（bytes/tick）
    private readonly float[] _budgetPerPriority;
    // 当前 tick 已消耗
    private readonly float[] _consumedPerPriority;
    
    private float _totalBudgetBytesPerTick;
    private long _lastResetTicks;
    
    public BandwidthBudgetManager(BandwidthConfig config)
    {
        _config = config;
        _measurement = new BandwidthMeasurement();
        _budgetPerPriority = new float[5];
        _consumedPerPriority = new float[5];
        RecalculateBudgets();
    }
    
    private void RecalculateBudgets()
    {
        // 默认带宽分配比例（总和为1.0）
        // Critical: 10% | High: 40% | Normal: 30% | Low: 15% | Bulk: 5%
        float[] ratios = { 0.10f, 0.40f, 0.30f, 0.15f, 0.05f };
        
        float totalBytesPerSecond = _config.TargetBandwidthBytesPerSecond;
        float tickDuration = _config.SendTickIntervalSeconds;
        _totalBudgetBytesPerTick = totalBytesPerSecond * tickDuration;
        
        for (int i = 0; i < 5; i++)
        {
            _budgetPerPriority[i] = _totalBudgetBytesPerTick * ratios[i];
        }
    }
    
    /// <summary>
    /// 查询指定优先级本 tick 可用字节数
    /// 支持"向上借用"：低优先级未用完的预算可借给高优先级
    /// </summary>
    public int GetAvailableBytes(MessagePriority priority)
    {
        int p = (int)priority;
        float available = _budgetPerPriority[p] - _consumedPerPriority[p];
        
        // Critical 和 High 可以借用低优先级剩余预算
        if (p <= 1)
        {
            for (int lower = p + 1; lower < 5; lower++)
            {
                float lowerAvailable = _budgetPerPriority[lower] - _consumedPerPriority[lower];
                if (lowerAvailable > 0)
                    available += lowerAvailable * 0.5f; // 最多借用 50%
            }
        }
        
        return Mathf.Max(0, (int)available);
    }
    
    /// <summary>
    /// 消耗预算
    /// </summary>
    public bool TryConsume(MessagePriority priority, int bytes)
    {
        int p = (int)priority;
        int available = GetAvailableBytes(priority);
        
        // Critical 消息即使超预算也必须发送
        if (priority == MessagePriority.Critical || bytes <= available)
        {
            _consumedPerPriority[p] += bytes;
            _measurement.Record(bytes);
            return true;
        }
        
        return false;
    }
    
    /// <summary>
    /// 每个发送 Tick 重置预算（通常每帧或每50ms调用一次）
    /// </summary>
    public void ResetTick()
    {
        Array.Clear(_consumedPerPriority, 0, _consumedPerPriority.Length);
        _lastResetTicks = DateTime.UtcNow.Ticks;
    }
    
    /// <summary>
    /// 根据实测带宽动态调整目标带宽
    /// 例如：当 RTT 持续升高时，认为链路拥塞，主动降低发送量
    /// </summary>
    public void AdaptToCongestion(NetworkMetrics metrics)
    {
        float congestionFactor = CalculateCongestionFactor(metrics);
        float adaptedBandwidth = _config.TargetBandwidthBytesPerSecond * congestionFactor;
        
        float tickDuration = _config.SendTickIntervalSeconds;
        _totalBudgetBytesPerTick = adaptedBandwidth * tickDuration;
        
        // 等比例缩放各优先级预算
        float scale = adaptedBandwidth / _config.TargetBandwidthBytesPerSecond;
        for (int i = 0; i < 5; i++)
        {
            _budgetPerPriority[i] *= scale;
        }
    }
    
    private float CalculateCongestionFactor(NetworkMetrics metrics)
    {
        // 基于 RTT 变化率和丢包率计算拥塞系数 [0.2, 1.0]
        float rttFactor = 1.0f;
        if (metrics.RttMs > _config.BaselineRttMs)
        {
            float rttRatio = metrics.RttMs / (float)_config.BaselineRttMs;
            rttFactor = 1.0f / rttRatio; // RTT 翻倍则降低 50% 发包量
        }
        
        float lossFactor = 1.0f - metrics.PacketLossRate * 2.0f; // 10% 丢包降低 20%
        
        return Mathf.Clamp(rttFactor * lossFactor, 0.2f, 1.0f);
    }
    
    public BandwidthStats GetStats()
    {
        return new BandwidthStats
        {
            TargetBytesPerSecond = _config.TargetBandwidthBytesPerSecond,
            ActualBytesPerSecond = _measurement.GetBytesPerSecond(),
            UtilizationRate = _measurement.GetBytesPerSecond() / _config.TargetBandwidthBytesPerSecond
        };
    }
}

[Serializable]
public class BandwidthConfig
{
    /// <summary>目标上行带宽（字节/秒）。移动端建议 30-60 KB/s</summary>
    public float TargetBandwidthBytesPerSecond = 51200; // 50 KB/s
    
    /// <summary>发送 Tick 间隔（秒）</summary>
    public float SendTickIntervalSeconds = 0.05f; // 50ms
    
    /// <summary>基线 RTT（毫秒），用于拥塞判断基准</summary>
    public int BaselineRttMs = 50;
    
    /// <summary>最大允许 RTT（超过此值触发紧急降流）</summary>
    public int MaxRttMs = 300;
}
```

## 五、令牌桶自适应发送速率控制

### 5.1 令牌桶算法实现

```csharp
/// <summary>
/// 令牌桶算法实现
/// 平滑突发流量，防止瞬时大量发包导致网络拥塞
/// </summary>
public sealed class TokenBucketRateController
{
    private double _tokens;
    private readonly double _capacity;        // 桶容量（字节）
    private readonly double _refillRatePerMs; // 补充速率（字节/毫秒）
    private long _lastRefillTimestamp;
    
    // AIMD (Additive Increase Multiplicative Decrease) 拥塞控制
    private double _currentRate;
    private double _minRate;
    private double _maxRate;
    private double _ssthresh; // 慢启动阈值
    private bool _inSlowStart;
    
    public double CurrentRateBytesPerSecond => _currentRate;
    
    public TokenBucketRateController(BandwidthConfig config)
    {
        _maxRate = config.TargetBandwidthBytesPerSecond;
        _minRate = _maxRate * 0.1f;
        _currentRate = _maxRate * 0.5f; // 从 50% 开始慢启动
        _ssthresh = _maxRate * 0.75f;
        _inSlowStart = true;
        
        _capacity = _maxRate * 0.2f;   // 容纳 200ms 的数据作为突发容量
        _tokens = _capacity * 0.5f;
        _refillRatePerMs = _currentRate / 1000.0;
        _lastRefillTimestamp = GetTimestampMs();
    }
    
    /// <summary>
    /// 尝试获取令牌以发送指定字节数
    /// </summary>
    public bool TryAcquire(int bytes)
    {
        RefillTokens();
        
        if (_tokens >= bytes)
        {
            _tokens -= bytes;
            return true;
        }
        return false;
    }
    
    /// <summary>
    /// 强制获取令牌（用于关键消息，允许透支）
    /// </summary>
    public void ForceAcquire(int bytes)
    {
        RefillTokens();
        _tokens -= bytes; // 允许为负，下一次补充时恢复
    }
    
    private void RefillTokens()
    {
        long now = GetTimestampMs();
        long elapsed = now - _lastRefillTimestamp;
        
        if (elapsed > 0)
        {
            double newTokens = elapsed * _refillRatePerMs;
            _tokens = Math.Min(_capacity, _tokens + newTokens);
            _lastRefillTimestamp = now;
        }
    }
    
    /// <summary>
    /// 收到 ACK 时调用：增加发送速率（AIMD 的加法递增）
    /// </summary>
    public void OnAck(int ackedBytes)
    {
        if (_inSlowStart)
        {
            // 慢启动：每个 ACK 增加一个 MSS
            _currentRate += ackedBytes;
            if (_currentRate >= _ssthresh)
            {
                _inSlowStart = false;
            }
        }
        else
        {
            // 拥塞避免：每个 RTT 增加一个 MSS
            _currentRate += (ackedBytes * ackedBytes) / _currentRate;
        }
        
        _currentRate = Math.Min(_currentRate, _maxRate);
        _refillRatePerMs = _currentRate / 1000.0;
    }
    
    /// <summary>
    /// 检测到丢包时调用：乘法减少发送速率（AIMD 的乘法递减）
    /// </summary>
    public void OnPacketLoss()
    {
        _ssthresh = _currentRate * 0.5;
        _currentRate = _ssthresh; // 不回到慢启动，直接设为 ssthresh
        _refillRatePerMs = _currentRate / 1000.0;
        
        // 触发严重拥塞时进入慢启动
        if (_currentRate < _minRate * 2)
        {
            _currentRate = _minRate;
            _inSlowStart = true;
        }
    }
    
    /// <summary>
    /// 根据 RTT 变化率调整速率（BBR 风格）
    /// </summary>
    public void OnRttUpdate(float currentRttMs, float baselineRttMs)
    {
        float rttRatio = currentRttMs / baselineRttMs;
        
        if (rttRatio > 1.5f)
        {
            // RTT 升高 50% 以上，主动降速
            float reductionFactor = 1.0f / rttRatio;
            _currentRate = Math.Max(_minRate, _currentRate * reductionFactor);
            _refillRatePerMs = _currentRate / 1000.0;
        }
    }
    
    private static long GetTimestampMs() =>
        DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
}
```

## 六、消息合并与批量发送

### 6.1 消息合并器

```csharp
/// <summary>
/// 消息合并器：将多个小消息打包成一个大数据包发送
/// 减少包头开销，提升有效带宽利用率
/// </summary>
public sealed class MessageCoalescer
{
    private readonly List<PendingMessage> _pendingMessages = new(32);
    private int _pendingBytes;
    private long _coalescingDeadlineTicks;
    
    // 合并配置
    private const int MaxCoalescingDelayMs = 10;  // 最长等待合并时间
    private const int MaxPacketBytes = 1400;       // 接近 MTU，避免 IP 分片
    private const int MinCoalescingBytes = 200;    // 最小合并收益阈值
    
    public bool HasPending => _pendingMessages.Count > 0;
    
    public void Add(PendingMessage message)
    {
        if (_pendingMessages.Count == 0)
        {
            _coalescingDeadlineTicks = DateTime.UtcNow.Ticks +
                TimeSpan.FromMilliseconds(MaxCoalescingDelayMs).Ticks;
        }
        
        _pendingMessages.Add(message);
        _pendingBytes += message.EstimatedBytes;
    }
    
    /// <summary>
    /// 检查是否应该立即刷新（不再等待合并）
    /// </summary>
    public bool ShouldFlush(bool hasCriticalMessage)
    {
        if (hasCriticalMessage) return true;
        if (_pendingBytes >= MaxPacketBytes) return true;
        if (_pendingMessages.Count > 0 && DateTime.UtcNow.Ticks >= _coalescingDeadlineTicks) return true;
        return false;
    }
    
    /// <summary>
    /// 将待发送消息序列化为合并数据包
    /// 格式：[包头4字节][消息数量2字节][消息1长度2字节+消息1数据][消息2...]
    /// </summary>
    public byte[] Flush()
    {
        if (_pendingMessages.Count == 0) return null;
        
        // 计算总大小
        int totalSize = 6; // 包头(4) + 消息数量(2)
        foreach (var msg in _pendingMessages)
            totalSize += 2 + msg.Data.Length; // 长度前缀(2) + 数据
        
        byte[] packet = new byte[totalSize];
        int offset = 0;
        
        // 写入包头魔数
        packet[offset++] = 0xAB;
        packet[offset++] = 0xCD;
        // 写入总长度
        packet[offset++] = (byte)(totalSize >> 8);
        packet[offset++] = (byte)(totalSize & 0xFF);
        // 写入消息数量
        packet[offset++] = (byte)(_pendingMessages.Count >> 8);
        packet[offset++] = (byte)(_pendingMessages.Count & 0xFF);
        
        // 写入各消息
        foreach (var msg in _pendingMessages)
        {
            packet[offset++] = (byte)(msg.Data.Length >> 8);
            packet[offset++] = (byte)(msg.Data.Length & 0xFF);
            Buffer.BlockCopy(msg.Data, 0, packet, offset, msg.Data.Length);
            offset += msg.Data.Length;
        }
        
        _pendingMessages.Clear();
        _pendingBytes = 0;
        return packet;
    }
}
```

## 七、位置同步专项优化：增量发送与死区过滤

```csharp
/// <summary>
/// 位置同步优化器
/// 通过死区过滤、增量编码减少位置同步流量
/// </summary>
public sealed class PositionSyncOptimizer
{
    // 每个实体的上次同步数据
    private readonly Dictionary<long, EntitySyncState> _lastSyncStates = new();
    
    // 死区配置
    private const float PositionDeadZone = 0.05f;  // 移动 < 5cm 不同步
    private const float RotationDeadZone = 1.0f;   // 旋转 < 1度 不同步
    private const long MaxSyncIntervalMs = 200;     // 最长 200ms 必须同步一次（防止漂移）
    
    /// <summary>
    /// 判断实体是否需要同步，并返回增量数据
    /// </summary>
    public bool TryGetSyncData(long entityId, Vector3 position, float rotation, 
        out PositionDelta delta)
    {
        long nowMs = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        
        if (!_lastSyncStates.TryGetValue(entityId, out var lastState))
        {
            // 新实体，首次必须全量同步
            delta = new PositionDelta
            {
                EntityId = entityId,
                DeltaX = position.x,
                DeltaY = position.y,
                DeltaZ = position.z,
                DeltaRotation = rotation,
                IsFullSync = true
            };
            _lastSyncStates[entityId] = new EntitySyncState(position, rotation, nowMs);
            return true;
        }
        
        // 计算增量
        float dx = position.x - lastState.Position.x;
        float dy = position.y - lastState.Position.y;
        float dz = position.z - lastState.Position.z;
        float dr = Mathf.DeltaAngle(lastState.Rotation, rotation);
        
        bool posChanged = Mathf.Sqrt(dx*dx + dy*dy + dz*dz) > PositionDeadZone;
        bool rotChanged = Mathf.Abs(dr) > RotationDeadZone;
        bool timeout = nowMs - lastState.LastSyncMs > MaxSyncIntervalMs;
        
        if (!posChanged && !rotChanged && !timeout)
        {
            delta = default;
            return false; // 变化太小，跳过本次同步
        }
        
        delta = new PositionDelta
        {
            EntityId = entityId,
            DeltaX = posChanged ? dx : 0,
            DeltaY = posChanged ? dy : 0,
            DeltaZ = posChanged ? dz : 0,
            DeltaRotation = rotChanged ? dr : 0,
            HasPosition = posChanged,
            HasRotation = rotChanged,
            IsFullSync = false
        };
        
        _lastSyncStates[entityId] = new EntitySyncState(position, rotation, nowMs);
        return true;
    }
    
    /// <summary>
    /// 量化压缩：将浮点坐标量化为 16 位整数，节省 50% 字节
    /// 精度：1/100 米（1厘米）
    /// 范围：±327 米
    /// </summary>
    public static short QuantizePosition(float worldPos)
    {
        return (short)Mathf.Clamp(worldPos * 100f, short.MinValue, short.MaxValue);
    }
    
    public static float DequantizePosition(short quantized)
    {
        return quantized / 100f;
    }
}

public struct PositionDelta
{
    public long EntityId;
    public float DeltaX, DeltaY, DeltaZ;
    public float DeltaRotation;
    public bool HasPosition;
    public bool HasRotation;
    public bool IsFullSync;
}

public struct EntitySyncState
{
    public Vector3 Position;
    public float Rotation;
    public long LastSyncMs;
    
    public EntitySyncState(Vector3 pos, float rot, long ts)
    {
        Position = pos;
        Rotation = rot;
        LastSyncMs = ts;
    }
}
```

```csharp
/// <summary>
/// 网络发送调度器：整合优先级队列、带宽预算、令牌桶的完整调度逻辑
/// 建议每 50ms (20Hz) 调用一次 Tick
/// </summary>
public sealed class NetworkSendScheduler : IDisposable
{
    private readonly PriorityMessageQueue _queue;
    private readonly BandwidthBudgetManager _budgetManager;
    private readonly TokenBucketRateController _rateController;
    private readonly MessageCoalescer _coalescer;
    private readonly INetworkTransport _transport;
    
    private NetworkMetrics _metrics;
    private bool _disposed;
    
    public NetworkSendScheduler(BandwidthConfig config, INetworkTransport transport)
    {
        _queue = new PriorityMessageQueue();
        _budgetManager = new BandwidthBudgetManager(config);
        _rateController = new TokenBucketRateController(config);
        _coalescer = new MessageCoalescer();
        _transport = transport;
    }
    
    /// <summary>
    /// 提交消息到发送队列
    /// </summary>
    public void Send<T>(T message) where T : struct
    {
        var attr = typeof(T).GetCustomAttribute<MessagePriorityAttribute>();
        var priority = attr?.Priority ?? MessagePriority.Normal;
        
        byte[] data = SerializeMessage(message);
        
        _queue.Enqueue(new PendingMessage
        {
            Priority = priority,
            Data = data,
            EstimatedBytes = data.Length + 6, // 数据 + 包头开销
            MaxDelayMs = attr?.MaxDelayMs ?? 0,
            Reliable = attr?.Reliable ?? true,
            EnqueueTicks = DateTime.UtcNow.Ticks
        });
    }
    
    /// <summary>
    /// 发送 Tick，由外部定时驱动（建议 50ms/次）
    /// </summary>
    public void Tick()
    {
        if (_disposed) return;
        
        _budgetManager.ResetTick();
        _budgetManager.AdaptToCongestion(_metrics);
        
        // 先处理关键消息（无视预算限制）
        FlushCriticalMessages();
        
        // 按带宽预算发送其他消息
        int availableBytes = _budgetManager.GetAvailableBytes(MessagePriority.High);
        var batch = _queue.DequeueBatch(availableBytes);
        
        foreach (var msg in batch)
        {
            if (!_rateController.TryAcquire(msg.EstimatedBytes))
                continue;
            
            if (msg.Priority <= MessagePriority.High || !CanCoalesce(msg))
            {
                // 高优先级直接发送，不合并
                _transport.Send(msg.Data, msg.Reliable);
                _budgetManager.TryConsume(msg.Priority, msg.EstimatedBytes);
            }
            else
            {
                // 低优先级加入合并队列
                _coalescer.Add(msg);
                _budgetManager.TryConsume(msg.Priority, msg.EstimatedBytes);
            }
        }
        
        // 刷新合并队列
        if (_coalescer.ShouldFlush(false))
        {
            byte[] coalesced = _coalescer.Flush();
            if (coalesced != null)
                _transport.Send(coalesced, reliable: false);
        }
    }
    
    private void FlushCriticalMessages()
    {
        // 扫描队列中的 Critical 消息，强制立即发送
        // 实际实现中可维护一个单独的 Critical 队列
    }
    
    private bool CanCoalesce(PendingMessage msg)
    {
        return msg.Priority >= MessagePriority.Normal &&
               msg.EstimatedBytes < 400; // 小包才合并
    }
    
    public void UpdateMetrics(NetworkMetrics metrics)
    {
        _metrics = metrics;
        _rateController.OnRttUpdate(metrics.RttMs, 50f);
        
        if (metrics.PacketLossRate > 0.05f) // 5% 以上丢包
            _rateController.OnPacketLoss();
    }
    
    private byte[] SerializeMessage<T>(T message)
    {
        // 集成 MessagePack 或 FlatBuffers 序列化
        return MemoryPackSerializer.Serialize(message);
    }
    
    public void Dispose()
    {
        _disposed = true;
    }
}

public struct NetworkMetrics
{
    public float RttMs;
    public float PacketLossRate;
    public float JitterMs;
    public float BandwidthUtilization;
}

public interface INetworkTransport
{
    void Send(byte[] data, bool reliable);
}
```

```csharp
    }

        _scheduler?.Dispose();
    }
```
