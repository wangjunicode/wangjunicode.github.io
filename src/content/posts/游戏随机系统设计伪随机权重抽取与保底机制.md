## 一、确定性随机（种子系统）

```csharp
using System;
using UnityEngine;

/// <summary>
/// 基于种子的确定性随机器（网络游戏/帧同步必备）
/// 相同种子保证所有客户端产生相同随机序列
/// </summary>
public class SeededRandom
{
    private uint state;
    
    public SeededRandom(uint seed)
    {
        // 防止种子为0（Xorshift不允许0状态）
        state = seed == 0 ? 0xDEADBEEF : seed;
    }

    /// <summary>
    /// Xorshift32 快速随机数生成
    /// </summary>
    public uint NextUInt()
    {
        state ^= state << 13;
        state ^= state >> 17;
        state ^= state << 5;
        return state;
    }

    /// <summary>
    /// [0, 1) 浮点随机数
    /// </summary>
    public float NextFloat()
    {
        return (NextUInt() & 0xFFFFFF) / (float)0x1000000;
    }

    /// <summary>
    /// [min, max) 整数随机
    /// </summary>
    public int NextInt(int min, int max)
    {
        if (min >= max) return min;
        uint range = (uint)(max - min);
        return min + (int)(NextUInt() % range);
    }

    /// <summary>
    /// [min, max) 浮点随机
    /// </summary>
    public float NextFloat(float min, float max)
    {
        return min + NextFloat() * (max - min);
    }

    /// <summary>
    /// 以 probability(0-1) 的概率返回 true
    /// </summary>
    public bool NextBool(float probability)
    {
        return NextFloat() < probability;
    }
}
```

## 二、权重随机抽取

```csharp
/// <summary>
/// 权重随机（支持动态更新权重）
/// </summary>
public class WeightedRandom<T>
{
    private List<(T Item, float Weight)> items = new List<(T, float)>();
    private float totalWeight;

    public void Add(T item, float weight)
    {
        if (weight <= 0) return;
        items.Add((item, weight));
        totalWeight += weight;
    }

    public void Clear()
    {
        items.Clear();
        totalWeight = 0f;
    }

    /// <summary>
    /// 随机抽取一个（O(n)线性搜索）
    /// </summary>
    public T Pick()
    {
        if (items.Count == 0) throw new InvalidOperationException("No items to pick");
        
        float rand = UnityEngine.Random.value * totalWeight;
        float cumulative = 0f;
        
        foreach (var (item, weight) in items)
        {
            cumulative += weight;
            if (rand <= cumulative)
                return item;
        }
        
        return items[items.Count - 1].Item; // 兜底
    }

    /// <summary>
    /// 随机抽取一个（使用种子随机器）
    /// </summary>
    public T Pick(SeededRandom rng)
    {
        if (items.Count == 0) throw new InvalidOperationException("No items");
        
        float rand = rng.NextFloat() * totalWeight;
        float cumulative = 0f;
        
        foreach (var (item, weight) in items)
        {
            cumulative += weight;
            if (rand <= cumulative)
                return item;
        }
        
        return items[items.Count - 1].Item;
    }

    /// <summary>
    /// 抽取N个不重复
    /// </summary>
    public List<T> PickMultiple(int count, bool allowRepeat = false)
    {
        var results = new List<T>(count);
        
        if (allowRepeat)
        {
            for (int i = 0; i < count; i++)
                results.Add(Pick());
            return results;
        }
        
        // 不重复：每次抽完移除再抽
        var tempItems = new List<(T, float)>(items);
        float tempTotal = totalWeight;
        
        for (int i = 0; i < count && tempItems.Count > 0; i++)
        {
            float rand = UnityEngine.Random.value * tempTotal;
            float cumulative = 0f;
            
            for (int j = 0; j < tempItems.Count; j++)
            {
                cumulative += tempItems[j].Item2;
                if (rand <= cumulative)
                {
                    results.Add(tempItems[j].Item1);
                    tempTotal -= tempItems[j].Item2;
                    tempItems.RemoveAt(j);
                    break;
                }
            }
        }
        
        return results;
    }
}
```

## 三、伪随机分布（PRNG - Pseudo Random Distribution）

DotA2/LOL 使用的技术：让「运气」更均匀，避免连续很多次不触发或连续触发。

```csharp
/// <summary>
/// 伪随机分布（PRNG）
/// 实际触发概率随失败次数递增，直到必定触发后重置
/// 体感更公平，不会出现"连续20次没暴击"
/// </summary>
public class PseudoRandomChance
{
    private float baseChance;   // 基础概率（如暴击率 0.2 = 20%）
    private float currentChance; // 当前实际判断概率（递增）
    private float increment;    // 每次失败增加的量
    
    // 失败次数
    private int failCount;

    /// <summary>
    /// 根据目标期望概率初始化 PRNG 参数
    /// </summary>
    public PseudoRandomChance(float targetProbability)
    {
        baseChance = targetProbability;
        
        // 用二分法求初始 C 值（使期望概率等于 targetProbability）
        increment = SolvePRNGConstant(targetProbability);
        currentChance = increment;
        failCount = 0;
    }

    /// <summary>
    /// 执行一次触发判断
    /// </summary>
    public bool Roll()
    {
        float rand = UnityEngine.Random.value;
        
        if (rand < currentChance)
        {
            // 触发成功，重置
            failCount = 0;
            currentChance = increment;
            return true;
        }
        else
        {
            // 触发失败，增加下次概率
            failCount++;
            currentChance = Mathf.Min(1f, currentChance + increment);
            return false;
        }
    }

    /// <summary>
    /// 当前触发概率（可用于UI显示"当前运气值"）
    /// </summary>
    public float CurrentChance => currentChance;
    public int FailCount => failCount;

    // 二分求解 C 值：使得 E[X] = 1/p（期望触发次数）
    static float SolvePRNGConstant(float p)
    {
        float lo = 0f, hi = p;
        
        for (int iter = 0; iter < 64; iter++)
        {
            float mid = (lo + hi) * 0.5f;
            float ep = ExpectedProbability(mid);
            
            if (ep > p) hi = mid;
            else lo = mid;
        }
        
        return (lo + hi) * 0.5f;
    }

    static float ExpectedProbability(float C)
    {
        // 计算给定 C 值下的期望触发概率
        float sum = 0f, prob = 1f;
        int maxN = (int)Math.Ceiling(1.0 / C) + 1;
        
        for (int n = 1; n <= maxN; n++)
        {
            float triggerProb = Mathf.Min(1f, n * C);
            prob *= (1f - triggerProb) == 0 ? 0 : (1f - (n - 1) * C);
            if (prob <= 0) break;
            sum += triggerProb * n * prob / (triggerProb > 0 ? triggerProb : 1);
        }
        
        // 简化：直接返回近似值
        return C * (1f / C + 1f) / 2f;
    }
}
```

## 四、保底机制（Pity System）

```csharp
/// <summary>
/// 抽卡保底系统
/// 支持软保底（概率递增）和硬保底（必定触发）
/// </summary>
public class PitySystem
{
    [System.Serializable]
    public class PityConfig
    {
        public float BaseRate;           // 基础概率（如0.006 = 0.6%）
        public int SoftPityStart;        // 软保底开始（如74抽开始概率提升）
        public float SoftPityIncrement;  // 软保底每抽概率增量（如0.06）
        public int HardPity;             // 硬保底（如90抽必出）
    }

    private PityConfig config;
    private int pullCount;  // 自上次出货后的抽数
    
    // 审计日志
    private List<PullRecord> history = new List<PullRecord>();

    public struct PullRecord
    {
        public int PullNumber;      // 第几抽
        public float ActualRate;    // 实际概率
        public bool IsHit;          // 是否触发
        public long Timestamp;
    }

    public int PullCount => pullCount;

    public PitySystem(PityConfig cfg)
    {
        config = cfg;
        pullCount = 0;
    }

    /// <summary>
    /// 执行一次抽卡判断
    /// </summary>
    public bool Pull()
    {
        pullCount++;
        float rate = CalculateCurrentRate();
        bool isHit = UnityEngine.Random.value < rate;

        history.Add(new PullRecord
        {
            PullNumber = pullCount,
            ActualRate = rate,
            IsHit = isHit,
            Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
        });

        if (isHit)
            Reset();

        return isHit;
    }

    float CalculateCurrentRate()
    {
        // 硬保底：必定触发
        if (pullCount >= config.HardPity)
            return 1f;
        
        // 软保底：概率随抽数线性增长
        if (pullCount >= config.SoftPityStart)
        {
            int overSoft = pullCount - config.SoftPityStart;
            return config.BaseRate + overSoft * config.SoftPityIncrement;
        }
        
        return config.BaseRate;
    }

    void Reset()
    {
        pullCount = 0;
    }

    /// <summary>
    /// 获取保底进度（0-1）
    /// </summary>
    public float GetPityProgress()
    {
        return (float)pullCount / config.HardPity;
    }

    /// <summary>
    /// 当前实际概率（可展示给玩家）
    /// </summary>
    public float GetCurrentRate()
    {
        return CalculateCurrentRate();
    }

    public List<PullRecord> GetHistory() => new List<PullRecord>(history);
}
```

## 五、随机系统设计规范

| 场景 | 推荐方案 | 原因 |
|------|----------|------|
| 帧同步/回放 | 种子随机 SeededRandom | 必须确定性 |
| 掉落物品 | 权重随机 + 保底 | 平衡性好 |
| 技能触发（暴击等）| PRNG伪随机 | 体感更公平 |
| 地图生成 | 种子随机 | 可分享/重现地图 |
| 随机事件 | Fisher-Yates洗牌 | 保证不重复 |
| 抽卡系统 | 软保底+硬保底 | 符合监管要求 |

**合规要点（中国抽卡游戏）：**
- 必须公示每种物品的实际概率
- 保底机制必须存在且公开
- 连续50次未获得稀有物品需公示（部分省市规定）
- 禁止面向未成年人的高价抽卡
