## 一、第一人称摄像机

```csharp
using UnityEngine;

/// <summary>
/// 第一人称摄像机控制器（FPS）
/// </summary>
public class FPSCameraController : MonoBehaviour
{
    [Header("灵敏度")]
    [SerializeField] private float mouseSensitivity = 2f;
    [SerializeField] private float aimSensitivityMultiplier = 0.4f; // 瞄准时灵敏度
    
    [Header("角度限制")]
    [SerializeField] private float minPitch = -85f;  // 俯视最大角度
    [SerializeField] private float maxPitch = 85f;   // 仰视最大角度
    
    [Header("头部晃动")]
    [SerializeField] private bool enableHeadBob = true;
    [SerializeField] private float bobFrequency = 1.8f;   // 晃动频率
    [SerializeField] private float bobAmplitude = 0.05f;  // 晃动幅度
    [SerializeField] private float bobLerpSpeed = 10f;
    
    [Header("FOV")]
    [SerializeField] private float normalFOV = 70f;
    [SerializeField] private float aimFOV = 40f;
    [SerializeField] private float fovLerpSpeed = 10f;
    
    [Header("引用")]
    [SerializeField] private Transform playerBody; // 玩家身体（水平旋转用）
    
    private Camera cam;
    private float pitch;  // 当前俯仰角
    private float yaw;    // 当前偏航角
    
    // 头部晃动
    private float bobTimer;
    private Vector3 bobOffset;
    private Vector3 camDefaultPos;
    
    private bool isAiming;
    
    void Awake()
    {
        cam = GetComponent<Camera>();
        camDefaultPos = transform.localPosition;
        
        // 锁定鼠标
        Cursor.lockState = CursorLockMode.Locked;
        Cursor.visible = false;
    }

    void Update()
    {
        HandleMouseLook();
        HandleFOV();
        
        if (enableHeadBob)
            HandleHeadBob();
    }

    void HandleMouseLook()
    {
        float sensitivity = isAiming ? 
            mouseSensitivity * aimSensitivityMultiplier : mouseSensitivity;
        
        float mouseX = Input.GetAxis("Mouse X") * sensitivity;
        float mouseY = Input.GetAxis("Mouse Y") * sensitivity;
        
        // 俯仰角（摄像机上下转）
        pitch -= mouseY;
        pitch = Mathf.Clamp(pitch, minPitch, maxPitch);
        
        // 偏航角（玩家身体左右转）
        yaw += mouseX;
        
        transform.localRotation = Quaternion.Euler(pitch, 0f, 0f);
        
        if (playerBody != null)
            playerBody.rotation = Quaternion.Euler(0f, yaw, 0f);
    }

    void HandleFOV()
    {
        float targetFOV = isAiming ? aimFOV : normalFOV;
        cam.fieldOfView = Mathf.Lerp(cam.fieldOfView, targetFOV, 
            fovLerpSpeed * Time.deltaTime);
    }

    void HandleHeadBob()
    {
        bool isMoving = Input.GetAxis("Horizontal") != 0 || Input.GetAxis("Vertical") != 0;
        
        if (isMoving)
        {
            bobTimer += Time.deltaTime * bobFrequency;
            float bobX = Mathf.Sin(bobTimer) * bobAmplitude * 0.5f;
            float bobY = Mathf.Abs(Mathf.Sin(bobTimer)) * bobAmplitude;
            bobOffset = new Vector3(bobX, bobY, 0f);
        }
        else
        {
            bobOffset = Vector3.Lerp(bobOffset, Vector3.zero, bobLerpSpeed * Time.deltaTime);
        }
        
        transform.localPosition = Vector3.Lerp(
            transform.localPosition, 
            camDefaultPos + bobOffset, 
            bobLerpSpeed * Time.deltaTime);
    }

    public void SetAiming(bool aiming) => isAiming = aiming;
    
    /// <summary>
    /// 施加震屏（爆炸/受击反馈）
    /// </summary>
    public void Shake(float intensity, float duration)
    {
        StartCoroutine(ShakeCoroutine(intensity, duration));
    }
    
    System.Collections.IEnumerator ShakeCoroutine(float intensity, float duration)
    {
        float elapsed = 0;
        while (elapsed < duration)
        {
            float x = Random.Range(-intensity, intensity);
            float y = Random.Range(-intensity, intensity);
            transform.localPosition = camDefaultPos + new Vector3(x, y, 0);
            elapsed += Time.deltaTime;
            yield return null;
        }
        transform.localPosition = camDefaultPos;
    }
}
```

## 二、第三人称摄像机

```csharp
/// <summary>
/// 第三人称摄像机（带弹簧臂碰撞检测）
/// </summary>
public class TPSCameraController : MonoBehaviour
{
    [Header("跟随目标")]
    [SerializeField] private Transform target;
    [SerializeField] private Vector3 targetOffset = new Vector3(0, 1.6f, 0);
    
    [Header("弹簧臂")]
    [SerializeField] private float armLength = 4f;          // 目标臂长
    [SerializeField] private float minArmLength = 1f;       // 最小距离（碰撞收缩）
    [SerializeField] private float armShrinkSpeed = 10f;    // 收缩速度
    [SerializeField] private LayerMask collisionMask;       // 碰撞层
    [SerializeField] private float collisionRadius = 0.2f;  // 碰撞检测球半径
    
    [Header("旋转")]
    [SerializeField] private float horizontalSensitivity = 3f;
    [SerializeField] private float verticalSensitivity = 2f;
    [SerializeField] private float minPitch = -30f;
    [SerializeField] private float maxPitch = 70f;
    
    [Header("平滑")]
    [SerializeField] private float positionSmoothTime = 0.1f;
    [SerializeField] private float rotationSmoothTime = 0.05f;
    
    [Header("目标锁定")]
    [SerializeField] private bool enableLockOn = true;
    [SerializeField] private float lockOnRange = 15f;
    [SerializeField] private LayerMask lockOnMask;
    
    private float currentYaw;
    private float currentPitch;
    private float currentArmLength;
    
    private Vector3 smoothPositionVelocity;
    private float smoothYawVelocity;
    private float smoothPitchVelocity;
    
    private Transform lockOnTarget;

    void Start()
    {
        currentArmLength = armLength;
        if (target != null)
        {
            Vector3 angles = transform.eulerAngles;
            currentYaw = angles.y;
            currentPitch = angles.x;
        }
        
        Cursor.lockState = CursorLockMode.Locked;
    }

    void LateUpdate()
    {
        if (target == null) return;
        
        if (lockOnTarget != null)
            UpdateLockOnRotation();
        else
            UpdateFreeRotation();
        
        UpdatePosition();
    }

    void UpdateFreeRotation()
    {
        float mouseX = Input.GetAxis("Mouse X") * horizontalSensitivity;
        float mouseY = Input.GetAxis("Mouse Y") * verticalSensitivity;
        
        currentYaw += mouseX;
        currentPitch -= mouseY;
        currentPitch = Mathf.Clamp(currentPitch, minPitch, maxPitch);
    }

    void UpdateLockOnRotation()
    {
        // 朝向目标
        Vector3 dir = (lockOnTarget.position - target.position).normalized;
        float targetYaw = Mathf.Atan2(dir.x, dir.z) * Mathf.Rad2Deg;
        float targetPitch = -Mathf.Asin(dir.y) * Mathf.Rad2Deg * 0.5f; // 稍微抬高视角
        
        currentYaw = Mathf.SmoothDampAngle(currentYaw, targetYaw, 
            ref smoothYawVelocity, rotationSmoothTime);
        currentPitch = Mathf.SmoothDampAngle(currentPitch, targetPitch, 
            ref smoothPitchVelocity, rotationSmoothTime);
    }

    void UpdatePosition()
    {
        Vector3 pivotPoint = target.position + targetOffset;
        Quaternion targetRot = Quaternion.Euler(currentPitch, currentYaw, 0);
        
        // 弹簧臂碰撞检测
        float desiredArmLength = armLength;
        Vector3 armDir = targetRot * Vector3.back;
        
        if (Physics.SphereCast(pivotPoint, collisionRadius, armDir, 
            out RaycastHit hit, armLength, collisionMask))
        {
            desiredArmLength = Mathf.Max(minArmLength, hit.distance - collisionRadius);
        }
        
        // 平滑收缩/恢复
        currentArmLength = Mathf.Lerp(currentArmLength, desiredArmLength, 
            armShrinkSpeed * Time.deltaTime);
        
        // 目标摄像机位置
        Vector3 targetPos = pivotPoint + armDir * currentArmLength;
        
        // 平滑移动
        transform.position = Vector3.SmoothDamp(
            transform.position, targetPos, 
            ref smoothPositionVelocity, positionSmoothTime);
        
        transform.rotation = Quaternion.Lerp(transform.rotation, targetRot, 
            rotationSmoothTime > 0 ? Time.deltaTime / rotationSmoothTime : 1f);
    }

    /// <summary>
    /// 切换目标锁定（Tab键触发）
    /// </summary>
    public void ToggleLockOn()
    {
        if (lockOnTarget != null)
        {
            lockOnTarget = null;
            return;
        }
        
        // 找最近的可锁定目标
        var colliders = Physics.OverlapSphere(target.position, lockOnRange, lockOnMask);
        Transform nearest = null;
        float minDist = float.MaxValue;
        
        foreach (var col in colliders)
        {
            if (col.transform == target) continue;
            
            float dist = Vector3.Distance(target.position, col.transform.position);
            if (dist < minDist)
            {
                minDist = dist;
                nearest = col.transform;
            }
        }
        
        lockOnTarget = nearest;
    }

    public Transform LockOnTarget => lockOnTarget;
}
```

## 三、摄像机震屏系统

```csharp
/// <summary>
/// 摄像机震屏（支持叠加多个震屏效果）
/// </summary>
public class CameraShakeManager : MonoBehaviour
{
    private static CameraShakeManager instance;
    public static CameraShakeManager Instance => instance;

    [System.Serializable]
    public class ShakeEffect
    {
        public float Intensity;
        public float Duration;
        public float Frequency;
        public AnimationCurve FalloffCurve = AnimationCurve.EaseInOut(0, 1, 1, 0);
        
        [System.NonSerialized] public float Elapsed;
        [System.NonSerialized] public float NoiseOffsetX;
        [System.NonSerialized] public float NoiseOffsetY;
    }

    private List<ShakeEffect> activeShakes = new List<ShakeEffect>();
    private Vector3 shakeOffset;
    
    void Awake() { instance = this; }

    void LateUpdate()
    {
        shakeOffset = Vector3.zero;
        
        for (int i = activeShakes.Count - 1; i >= 0; i--)
        {
            var shake = activeShakes[i];
            shake.Elapsed += Time.deltaTime;
            
            if (shake.Elapsed >= shake.Duration)
            {
                activeShakes.RemoveAt(i);
                continue;
            }
            
            float t = shake.Elapsed / shake.Duration;
            float falloff = shake.FalloffCurve.Evaluate(t);
            float intensity = shake.Intensity * falloff;
            
            // Perlin Noise 震屏（比纯随机更流畅）
            float noise_time = shake.Elapsed * shake.Frequency;
            float x = (Mathf.PerlinNoise(noise_time + shake.NoiseOffsetX, 0) - 0.5f) * 2f;
            float y = (Mathf.PerlinNoise(0, noise_time + shake.NoiseOffsetY) - 0.5f) * 2f;
            
            shakeOffset += new Vector3(x, y, 0) * intensity;
        }
        
        // 应用到摄像机
        transform.localPosition = new Vector3(
            shakeOffset.x, shakeOffset.y, transform.localPosition.z);
    }

    public void AddShake(ShakeEffect shake)
    {
        shake.Elapsed = 0;
        shake.NoiseOffsetX = Random.Range(0f, 100f);
        shake.NoiseOffsetY = Random.Range(0f, 100f);
        activeShakes.Add(shake);
    }

    // 预定义震屏类型
    public void SmallShake() => AddShake(new ShakeEffect 
        { Intensity = 0.05f, Duration = 0.3f, Frequency = 20f });
    
    public void MediumShake() => AddShake(new ShakeEffect 
        { Intensity = 0.15f, Duration = 0.5f, Frequency = 15f });
    
    public void ExplosionShake() => AddShake(new ShakeEffect 
        { Intensity = 0.5f, Duration = 0.8f, Frequency = 10f });
}
```

## 四、摄像机方案选型

| 场景 | 推荐方案 |
|------|----------|
| 简单FPS | 自定义 FPSCameraController |
| 复杂TPS（战斗锁定+过场）| Cinemachine |
| 固定角度（2D/俯视）| 自定义简单跟随 |
| 电影级过场动画 | Cinemachine + Timeline |
| 多人游戏分屏 | 多Camera + Viewport Rect |
