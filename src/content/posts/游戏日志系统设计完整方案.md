## 一、日志系统的职责层次

一套完整的游戏日志系统需覆盖以下层次：

```
开发阶段  → 快速调试，输出详尽信息，支持模块过滤
测试阶段  → 捕获异常，输出到文件，支持远程传输
上线阶段  → 最小性能影响，崩溃前保全，远程日志采集
```

## 二、日志分级与模块过滤

### 2.1 日志级别定义

```csharp
public enum LogLevel
{
    Verbose = 0,   // 最详细：每帧状态、逐步流程
    Debug   = 1,   // 调试信息：关键流程节点
    Info    = 2,   // 普通信息：正常业务事件
    Warning = 3,   // 警告：潜在问题，但不影响运行
    Error   = 4,   // 错误：功能异常，需要处理
    Fatal   = 5,   // 致命：程序即将崩溃
    Off     = 99   // 关闭所有日志
}
```

## 三、核心日志管理器

```csharp
using System;
using System.Collections.Concurrent;
using System.IO;
using System.Text;
using System.Threading;
using UnityEngine;

/// <summary>
/// 游戏日志管理器（线程安全）
/// </summary>
public class GameLogger : MonoBehaviour
{
    private static GameLogger instance;
    public static GameLogger Instance => instance;

    [Header("配置")]
    [SerializeField] private LogLevel minimumLevel = LogLevel.Debug;
    [SerializeField] private LogModule enabledModules = LogModule.All;
    [SerializeField] private bool writeToFile = true;
    [SerializeField] private bool asyncWrite = true;
    [SerializeField] private int maxLogFileSizeMB = 20;
    [SerializeField] private int maxLogFileCount = 5;
    [SerializeField] private bool encryptLog = false;
    [SerializeField] private bool compressOnExit = true;

    // 输出目标
    private List<ILogSink> sinks = new List<ILogSink>();
    
    // 异步写入队列
    private ConcurrentQueue<LogEntry> writeQueue = new ConcurrentQueue<LogEntry>();
    private Thread writeThread;
    private volatile bool isRunning;
    
    // 日志文件管理
    private string currentLogPath;
    private StreamWriter fileWriter;
    private long currentFileSize;
    private int fileIndex;
    
    // 性能计数
    private long totalLogsWritten;
    private long totalLogsDropped;

    void Awake()
    {
        if (instance != null) { Destroy(gameObject); return; }
        instance = this;
        DontDestroyOnLoad(gameObject);
        
        Initialize();
        Application.logMessageReceivedThreaded += OnUnityLogReceived;
        Application.quitting += OnApplicationQuitting;
    }

    void Initialize()
    {
        // 注册输出目标
        sinks.Add(new ConsoleSink());
        
        if (writeToFile)
        {
            InitializeFileWriter();
            sinks.Add(new FileSink(this));
        }
        
        #if !UNITY_EDITOR
        sinks.Add(new RemoteSink()); // 线上版本额外上传日志
        #endif
        
        // 启动异步写入线程
        if (asyncWrite)
        {
            isRunning = true;
            writeThread = new Thread(AsyncWriteLoop)
            {
                Name = "LogWriteThread",
                IsBackground = true
            };
            writeThread.Start();
        }
        
        Info(LogModule.None, "GameLogger initialized. Path: " + currentLogPath);
    }

    // ============ 公共接口 ============

    public static void V(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Verbose, module, msg, args);
    
    public static void D(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Debug, module, msg, args);
    
    public static void I(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Info, module, msg, args);
    
    public static void W(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Warning, module, msg, args);
    
    public static void E(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Error, module, msg, args);
    
    public static void F(LogModule module, string msg, params object[] args)
        => Instance?.Log(LogLevel.Fatal, module, msg, args);

    // 直接使用实例方法
    public void Info(LogModule module, string msg)
        => Log(LogLevel.Info, module, msg);

    void Log(LogLevel level, LogModule module, string message, object[] args = null)
    {
        // 级别过滤
        if (level < minimumLevel) return;
        
        // 模块过滤
        if (module != LogModule.None && (enabledModules & module) == 0) return;
        
        string formatted = args != null && args.Length > 0 ? 
            string.Format(message, args) : message;
        
        var entry = new LogEntry
        {
            Level = level,
            Module = module,
            Message = formatted,
            Timestamp = DateTime.Now,
            FrameCount = Time.frameCount,
            ThreadId = Thread.CurrentThread.ManagedThreadId
        };
        
        if (asyncWrite)
        {
            writeQueue.Enqueue(entry);
        }
        else
        {
            WriteEntry(entry);
        }
    }

    void AsyncWriteLoop()
    {
        while (isRunning || !writeQueue.IsEmpty)
        {
            while (writeQueue.TryDequeue(out var entry))
            {
                WriteEntry(entry);
                Interlocked.Increment(ref totalLogsWritten);
            }
            Thread.Sleep(8); // ~120fps 节奏写入
        }
    }

    void WriteEntry(LogEntry entry)
    {
        foreach (var sink in sinks)
        {
            try { sink.Write(entry); }
            catch { /* 日志系统本身不能抛出异常 */ }
        }
    }

    // ============ 文件管理 ============

    void InitializeFileWriter()
    {
        string logDir = Path.Combine(Application.persistentDataPath, "Logs");
        Directory.CreateDirectory(logDir);
        
        // 清理旧日志
        CleanOldLogs(logDir);
        
        fileIndex = 0;
        OpenNewLogFile(logDir);
    }

    void OpenNewLogFile(string dir)
    {
        string timestamp = DateTime.Now.ToString("yyyyMMdd_HHmmss");
        currentLogPath = Path.Combine(dir, $"game_{timestamp}_{fileIndex:D2}.log");
        
        fileWriter = new StreamWriter(currentLogPath, append: false, 
            encoding: Encoding.UTF8)
        {
            AutoFlush = false  // 手动控制flush，提升性能
        };
        currentFileSize = 0;
    }

    void CleanOldLogs(string dir)
    {
        var files = new DirectoryInfo(dir).GetFiles("*.log")
            .OrderByDescending(f => f.CreationTime)
            .Skip(maxLogFileCount - 1)
            .ToArray();
        
        foreach (var f in files)
        {
            try { f.Delete(); } catch { }
        }
    }

    public void WriteToFile(string line)
    {
        if (fileWriter == null) return;
        
        // 文件大小检测，超出则滚动
        currentFileSize += line.Length * 2; // 粗略估算UTF8大小
        if (currentFileSize > maxLogFileSizeMB * 1024 * 1024)
        {
            fileWriter.Flush();
            fileWriter.Dispose();
            fileIndex++;
            OpenNewLogFile(Path.GetDirectoryName(currentLogPath));
        }
        
        fileWriter.WriteLine(line);
        
        // 每100条强制flush一次
        if (totalLogsWritten % 100 == 0)
            fileWriter.Flush();
    }

    // ============ Unity日志接管 ============

    void OnUnityLogReceived(string condition, string stackTrace, LogType type)
    {
        var level = type switch
        {
            LogType.Log => LogLevel.Debug,
            LogType.Warning => LogLevel.Warning,
            LogType.Error => LogLevel.Error,
            LogType.Exception => LogLevel.Fatal,
            LogType.Assert => LogLevel.Fatal,
            _ => LogLevel.Debug
        };
        
        string msg = type == LogType.Exception ? 
            $"{condition}\n{stackTrace}" : condition;
        
        Log(level, LogModule.None, $"[Unity] {msg}");
    }

    void OnApplicationQuitting()
    {
        isRunning = false;
        writeThread?.Join(3000); // 最多等3秒写完
        
        // 强制刷新所有剩余日志
        while (writeQueue.TryDequeue(out var entry))
            WriteEntry(entry);
        
        fileWriter?.Flush();
        fileWriter?.Dispose();
        
        // 压缩日志文件
        if (compressOnExit && File.Exists(currentLogPath))
        {
            CompressLogFile(currentLogPath);
        }
    }

    void CompressLogFile(string path)
    {
        try
        {
            using var source = File.OpenRead(path);
            using var dest = File.Create(path + ".gz");
            using var gz = new System.IO.Compression.GZipStream(
                dest, System.IO.Compression.CompressionMode.Compress);
            source.CopyTo(gz);
            File.Delete(path);
        }
        catch (Exception e)
        {
            Debug.LogError($"Log compress failed: {e.Message}");
        }
    }

    void OnDestroy()
    {
        Application.logMessageReceivedThreaded -= OnUnityLogReceived;
        OnApplicationQuitting();
    }
    
    public string CurrentLogPath => currentLogPath;
    public long TotalLogsWritten => totalLogsWritten;
}

/// <summary>
/// 日志条目
/// </summary>
public struct LogEntry
{
    public LogLevel Level;
    public LogModule Module;
    public string Message;
    public DateTime Timestamp;
    public int FrameCount;
    public int ThreadId;
    
    public string Format()
    {
        string levelStr = Level.ToString()[0].ToString();
        string moduleStr = Module == LogModule.None ? "" : $"[{Module}]";
        return $"{Timestamp:HH:mm:ss.fff} {levelStr} F{FrameCount} T{ThreadId} {moduleStr} {Message}";
    }
}

// ============ 输出目标接口 ============

public interface ILogSink
{
    void Write(LogEntry entry);
}

public class ConsoleSink : ILogSink
{
    public void Write(LogEntry entry)
    {
        switch (entry.Level)
        {
            case LogLevel.Warning:
                Debug.LogWarning(entry.Message); break;
            case LogLevel.Error:
            case LogLevel.Fatal:
                Debug.LogError(entry.Message); break;
            default:
                Debug.Log(entry.Message); break;
        }
    }
}

public class FileSink : ILogSink
{
    private GameLogger logger;
    public FileSink(GameLogger l) { logger = l; }
    
    public void Write(LogEntry entry)
    {
        logger.WriteToFile(entry.Format());
    }
}

public class RemoteSink : ILogSink
{
    private ConcurrentQueue<LogEntry> uploadQueue = new ConcurrentQueue<LogEntry>();
    private const int MaxUploadQueue = 500;
    
    public void Write(LogEntry entry)
    {
        // 仅上传 Warning 及以上
        if (entry.Level < LogLevel.Warning) return;
        
        if (uploadQueue.Count < MaxUploadQueue)
            uploadQueue.Enqueue(entry);
    }
    
    // 定时批量上传（由外部定时器触发）
    public LogEntry[] FlushBatch(int maxCount = 50)
    {
        var batch = new List<LogEntry>();
        while (batch.Count < maxCount && uploadQueue.TryDequeue(out var entry))
            batch.Add(entry);
        return batch.ToArray();
    }
}
```

## 四、崩溃前日志保全

```csharp
/// <summary>
/// 崩溃前保全最近N条日志
/// </summary>
public class CrashLogPreserver
{
    private static readonly Queue<string> recentLogs = new Queue<string>();
    private const int MaxPreservedLogs = 200;
    
    public static void AddLog(string log)
    {
        lock (recentLogs)
        {
            recentLogs.Enqueue(log);
            if (recentLogs.Count > MaxPreservedLogs)
                recentLogs.Dequeue();
        }
    }
    
    public static void SaveCrashLog(string crashReason, string stackTrace)
    {
        string crashDir = Path.Combine(Application.persistentDataPath, "CrashLogs");
        Directory.CreateDirectory(crashDir);
        
        string path = Path.Combine(crashDir, 
            $"crash_{DateTime.Now:yyyyMMdd_HHmmss}.txt");
        
        var sb = new StringBuilder();
        sb.AppendLine("=== CRASH REPORT ===");
        sb.AppendLine($"Time: {DateTime.Now}");
        sb.AppendLine($"Version: {Application.version}");
        sb.AppendLine($"Platform: {Application.platform}");
        sb.AppendLine($"Reason: {crashReason}");
        sb.AppendLine();
        sb.AppendLine("=== STACK TRACE ===");
        sb.AppendLine(stackTrace);
        sb.AppendLine();
        sb.AppendLine("=== RECENT LOGS ===");
        
        lock (recentLogs)
        {
            foreach (var log in recentLogs)
                sb.AppendLine(log);
        }
        
        File.WriteAllText(path, sb.ToString());
        
        // 下次启动时上传
        PlayerPrefs.SetString("PendingCrashReport", path);
    }
}
```

## 五、远程日志诊断

```csharp
/// <summary>
/// 远程日志级别动态调整（运营后台控制）
/// </summary>
public class RemoteLogConfig
{
    [Serializable]
    public class LogConfig
    {
        public string MinLevel;
        public string[] EnabledModules;
        public bool EnableRemoteUpload;
        public string RemoteEndpoint;
    }
    
    public static void ApplyRemoteConfig(string configJson)
    {
        var config = JsonUtility.FromJson<LogConfig>(configJson);
        
        if (Enum.TryParse<LogLevel>(config.MinLevel, out var level))
        {
            // 动态调整最低日志级别
            GameLogger.Instance.SetMinLevel(level);
        }
        
        // 按需开启特定模块的详细日志
        if (config.EnabledModules != null)
        {
            LogModule modules = LogModule.None;
            foreach (var m in config.EnabledModules)
            {
                if (Enum.TryParse<LogModule>(m, out var mod))
                    modules |= mod;
            }
            GameLogger.Instance.SetEnabledModules(modules);
        }
    }
}
```

## 六、日志系统最佳实践

| 场景 | 推荐做法 |
|------|----------|
| 开发环境 | Level=Verbose, All模块, 彩色输出 |
| QA测试 | Level=Debug, 写文件, 保留5个滚动文件 |
| 预发布 | Level=Info, 异步写文件+远程Warning+ |
| 正式上线 | Level=Warning, 异步写文件, 崩溃上报 |
| 线上诊断 | 远程动态开启特定模块Debug日志 |

**性能原则：**
1. 字符串格式化延迟执行（仅在确认输出时才拼接）
2. 异步批量写盘，不阻塞主线程
3. 日志文件滚动，防止磁盘占满
4. 线上版本默认只输出Warning+，减少IO开销
5. 使用结构化日志（JSON格式）便于后端检索分析
