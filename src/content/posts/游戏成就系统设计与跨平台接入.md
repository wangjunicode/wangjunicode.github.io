## 一、成就系统的价值

成就系统是提升玩家留存和游戏时长的重要机制：

1. **目标感**：给玩家设定短中长期目标
2. **探索引导**：引导玩家探索游戏内容
3. **炫耀分享**：社交分享驱动拉新
4. **平台集成**：Steam/主机平台成就增加游戏曝光

## 三、成就管理器

```csharp
/// <summary>
/// 本地成就进度存储
/// </summary>
[Serializable]
public class AchievementProgress
{
    public string AchievementId;
    public int CurrentProgress;
    public bool IsUnlocked;
    public DateTime UnlockTime;
    public bool IsNewlyUnlocked;  // 未展示的新解锁
}

/// <summary>
/// 成就管理器
/// </summary>
public class AchievementManager : MonoBehaviour
{
    private static AchievementManager instance;
    public static AchievementManager Instance => instance;

    [Header("成就数据")]
    [SerializeField] private AchievementData[] allAchievements;

    [Header("UI")]
    [SerializeField] private AchievementUnlockPopup unlockPopup;

    // 进度存储
    private Dictionary<string, AchievementProgress> progressMap 
        = new Dictionary<string, AchievementProgress>();
    
    // 索引
    private Dictionary<string, AchievementData> achievementIndex 
        = new Dictionary<string, AchievementData>();

    // 平台服务
    private IAchievementPlatform platformService;

    // 事件
    public event Action<AchievementData> OnAchievementUnlocked;

    void Awake()
    {
        instance = this;
        
        foreach (var ach in allAchievements)
            achievementIndex[ach.AchievementId] = ach;
        
        LoadProgress();
        InitializePlatformService();
    }

    void InitializePlatformService()
    {
        #if UNITY_STEAM
        platformService = new SteamAchievementService();
        #elif UNITY_ANDROID
        platformService = new GooglePlayAchievementService();
        #elif UNITY_IOS
        platformService = new GameCenterAchievementService();
        #else
        platformService = new LocalAchievementService(); // 编辑器/其他平台
        #endif
        
        platformService?.Initialize();
    }

    /// <summary>
    /// 上报进度（核心接口）
    /// </summary>
    public void ReportProgress(string achievementId, int progressDelta = 1)
    {
        if (!achievementIndex.TryGetValue(achievementId, out var data))
        {
            Debug.LogWarning($"Achievement not found: {achievementId}");
            return;
        }
        
        if (!progressMap.ContainsKey(achievementId))
        {
            progressMap[achievementId] = new AchievementProgress 
            { AchievementId = achievementId };
        }
        
        var progress = progressMap[achievementId];
        
        // 已解锁不再处理
        if (progress.IsUnlocked) return;
        
        progress.CurrentProgress += progressDelta;
        progress.CurrentProgress = Mathf.Min(progress.CurrentProgress, 
            data.RequiredProgress);
        
        // 通知平台更新进度（百分比）
        float percent = (float)progress.CurrentProgress / data.RequiredProgress * 100f;
        platformService?.ReportProgress(GetPlatformId(data), percent);
        
        // 检查是否解锁
        if (progress.CurrentProgress >= data.RequiredProgress)
        {
            UnlockAchievement(achievementId, data, progress);
        }
        
        SaveProgress();
    }

    /// <summary>
    /// 直接解锁（无需进度的一次性成就）
    /// </summary>
    public void Unlock(string achievementId)
    {
        ReportProgress(achievementId, int.MaxValue / 2);
    }

    void UnlockAchievement(string id, AchievementData data, AchievementProgress progress)
    {
        progress.IsUnlocked = true;
        progress.UnlockTime = DateTime.Now;
        progress.IsNewlyUnlocked = true;
        
        // 发放奖励
        GrantRewards(data);
        
        // 平台成就解锁
        platformService?.UnlockAchievement(GetPlatformId(data));
        
        // 显示解锁弹窗
        unlockPopup?.ShowUnlock(data);
        
        // 触发事件
        OnAchievementUnlocked?.Invoke(data);
        
        Debug.Log($"Achievement Unlocked: {data.Title}");
    }

    void GrantRewards(AchievementData data)
    {
        foreach (var reward in data.Rewards ?? Array.Empty<AchievementReward>())
        {
            if (!string.IsNullOrEmpty(reward.ItemId))
                InventorySystem.Instance?.AddItem(reward.ItemId, reward.Count);
            
            if (reward.Currency > 0)
                CurrencySystem.Instance?.AddCurrency(reward.Currency);
            
            if (!string.IsNullOrEmpty(reward.TitleUnlockId))
                TitleSystem.Instance?.UnlockTitle(reward.TitleUnlockId);
        }
    }

    string GetPlatformId(AchievementData data)
    {
        #if UNITY_STEAM
        return data.SteamApiName;
        #elif UNITY_ANDROID
        return data.GooglePlayId;
        #elif UNITY_IOS
        return data.AppleGameCenterId;
        #else
        return data.AchievementId;
        #endif
    }

    public int GetProgress(string id)
    {
        return progressMap.TryGetValue(id, out var p) ? p.CurrentProgress : 0;
    }

    public bool IsUnlocked(string id)
    {
        return progressMap.TryGetValue(id, out var p) && p.IsUnlocked;
    }

    public List<AchievementData> GetAllAchievements(bool includeSecret = false)
    {
        var list = new List<AchievementData>();
        foreach (var ach in allAchievements)
        {
            if (!includeSecret && ach.IsSecret && !IsUnlocked(ach.AchievementId))
                continue;
            list.Add(ach);
        }
        return list;
    }

    void LoadProgress()
    {
        string json = PlayerPrefs.GetString("AchievementProgress", "");
        if (string.IsNullOrEmpty(json)) return;
        
        var list = JsonUtility.FromJson<AchievementProgressList>(json);
        if (list?.Items != null)
        {
            foreach (var item in list.Items)
                progressMap[item.AchievementId] = item;
        }
    }

    void SaveProgress()
    {
        var list = new AchievementProgressList
        {
            Items = new List<AchievementProgress>(progressMap.Values)
        };
        PlayerPrefs.SetString("AchievementProgress", JsonUtility.ToJson(list));
        PlayerPrefs.Save();
    }

    [Serializable]
    class AchievementProgressList
    {
        public List<AchievementProgress> Items;
    }
}
```

```csharp
/// <summary>
/// 成就平台服务接口
/// </summary>
public interface IAchievementPlatform
{
    void Initialize();
    void UnlockAchievement(string platformId);
    void ReportProgress(string platformId, float percentComplete);
    void ResetAchievement(string platformId); // 仅开发调试用
}

/// <summary>
/// Steam 成就服务
/// </summary>
public class SteamAchievementService : IAchievementPlatform
{
    public void Initialize()
    {
        // SteamManager 初始化，Steamworks.NET
        #if STEAMWORKS_NET
        if (!Steamworks.SteamManager.Initialized)
            Debug.LogError("Steam not initialized!");
        #endif
    }

    public void UnlockAchievement(string platformId)
    {
        #if STEAMWORKS_NET
        Steamworks.SteamUserStats.SetAchievement(platformId);
        Steamworks.SteamUserStats.StoreStats();
        #endif
    }

    public void ReportProgress(string platformId, float percentComplete)
    {
        // Steam 不需要单独报告进度，直接 SetAchievement 即可
        // 如果要显示进度条，需要使用 Steam Stats
        #if STEAMWORKS_NET
        if (percentComplete >= 100f)
        {
            Steamworks.SteamUserStats.SetAchievement(platformId);
            Steamworks.SteamUserStats.StoreStats();
        }
        #endif
    }

    public void ResetAchievement(string platformId)
    {
        #if STEAMWORKS_NET && UNITY_EDITOR
        Steamworks.SteamUserStats.ClearAchievement(platformId);
        Steamworks.SteamUserStats.StoreStats();
        #endif
    }
}

/// <summary>
/// Google Play Games 成就服务
/// </summary>
public class GooglePlayAchievementService : IAchievementPlatform
{
    public void Initialize()
    {
        // Google Play Games Services 初始化
        // PlayGamesPlatform.Activate();
    }

    public void UnlockAchievement(string platformId)
    {
        // Social.ReportProgress(platformId, 100.0, success => {});
        UnityEngine.Social.ReportProgress(platformId, 100.0, success =>
        {
            if (!success) Debug.LogWarning($"Failed to unlock achievement: {platformId}");
        });
    }

    public void ReportProgress(string platformId, float percentComplete)
    {
        UnityEngine.Social.ReportProgress(platformId, percentComplete, _ => {});
    }

    public void ResetAchievement(string platformId) { }
}

/// <summary>
/// Apple Game Center 成就服务
/// </summary>
public class GameCenterAchievementService : IAchievementPlatform
{
    public void Initialize()
    {
        UnityEngine.Social.localUser.Authenticate(success =>
        {
            Debug.Log($"Game Center auth: {success}");
        });
    }

    public void UnlockAchievement(string platformId)
    {
        UnityEngine.Social.ReportProgress(platformId, 100.0, _ => {});
    }

    public void ReportProgress(string platformId, float percentComplete)
    {
        UnityEngine.Social.ReportProgress(platformId, percentComplete, _ => {});
    }

    public void ResetAchievement(string platformId) { }
}

/// 本地成就服务（开发模式）

    public void ResetAchievement(string id) => Debug.Log($"[Achievement] Reset: {id}");
```

## 五、成就解锁弹窗

```csharp
/// <summary>
/// 成就解锁弹窗（队列显示）
/// </summary>
public class AchievementUnlockPopup : MonoBehaviour
{
    [SerializeField] private GameObject popupRoot;
    [SerializeField] private Image iconImage;
    [SerializeField] private Text titleText;
    [SerializeField] private Text descText;
    [SerializeField] private float displayDuration = 3f;
    [SerializeField] private AnimationCurve slideCurve;

    private Queue<AchievementData> displayQueue = new Queue<AchievementData>();
    private bool isShowing;

    public void ShowUnlock(AchievementData data)
    {
        displayQueue.Enqueue(data);
        if (!isShowing) StartCoroutine(ProcessQueue());
    }

    System.Collections.IEnumerator ProcessQueue()
    {
        isShowing = true;
        
        while (displayQueue.Count > 0)
        {
            var data = displayQueue.Dequeue();
            yield return ShowSingle(data);
            yield return new WaitForSeconds(0.3f);
        }
        
        isShowing = false;
    }

    System.Collections.IEnumerator ShowSingle(AchievementData data)
    {
        iconImage.sprite = data.Icon;
        titleText.text = data.Title;
        descText.text = data.Description;
        
        popupRoot.SetActive(true);
        
        // 滑入动画
        float elapsed = 0;
        Vector3 startPos = new Vector3(300, 0, 0);
        Vector3 endPos = Vector3.zero;
        
        while (elapsed < 0.3f)
        {
            elapsed += Time.deltaTime;
            float t = slideCurve.Evaluate(elapsed / 0.3f);
            popupRoot.transform.localPosition = Vector3.Lerp(startPos, endPos, t);
            yield return null;
        }
        
        yield return new WaitForSeconds(displayDuration);
        
        // 滑出
        elapsed = 0;
        while (elapsed < 0.3f)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / 0.3f;
            popupRoot.transform.localPosition = Vector3.Lerp(endPos, startPos, t);
            yield return null;
        }
        
        popupRoot.SetActive(false);
    }
}
```

```csharp
/// <summary>
/// 成就追踪器示例：集成到各业务系统
/// </summary>
public class AchievementTrackers : MonoBehaviour
{
    void Start()
    {
        // 监听各业务事件
        CombatSystem.Instance.OnEnemyKilled += OnEnemyKilled;
        PlayerController.Instance.OnLevelUp += OnLevelUp;
        InventorySystem.Instance.OnItemCollected += OnItemCollected;
        DungeonManager.Instance.OnDungeonCompleted += OnDungeonCompleted;
    }

    void OnEnemyKilled(EnemyType type, int totalKills)
    {
        AchievementManager.Instance.ReportProgress("kill_100_enemies");
        
        if (type == EnemyType.Boss)
            AchievementManager.Instance.Unlock("first_boss_kill");
        
        // 里程碑成就
        if (totalKills == 100) AchievementManager.Instance.Unlock("slayer_100");
        if (totalKills == 1000) AchievementManager.Instance.Unlock("slayer_1000");
    }

    void OnLevelUp(int newLevel)
    {
        if (newLevel >= 60) AchievementManager.Instance.Unlock("reach_max_level");
    }

    void OnItemCollected(string itemId)
    {
        AchievementManager.Instance.ReportProgress("collect_items_1000");
        
        // 检查是否集齐套装
        CheckSetCompletion(itemId);
    }

    void OnDungeonCompleted(DungeonInstance instance)
    {
        AchievementManager.Instance.ReportProgress("clear_10_dungeons");
        
        if (instance.ElapsedTime < 600f) // 10分钟内通关
            AchievementManager.Instance.Unlock("speedrunner");
    }

    void CheckSetCompletion(string itemId) { }
}
```

## 七、设计要点

| 模块 | 建议 |
|------|------|
| ID 规范 | 平台 API ID 与本地 ID 分开，避免平台绑定 |
| 进度存储 | 本地缓存 + 服务端同步，防止刷新页面丢失 |
| 队列显示 | 同时解锁多个成就时排队显示，不要叠加弹窗 |
| 调试工具 | 提供 Editor 窗口手动触发/重置成就，方便测试 |
| 平台测试 | 使用 Sandbox 模式测试平台成就，不污染线上数据 |
