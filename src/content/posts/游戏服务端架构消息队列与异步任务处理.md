## 一、消息队列在游戏中的应用场景

| 场景 | 队列类型 | 延迟要求 | 重要程度 |
|------|----------|----------|----------|
| 游戏内邮件发送 | 普通队列 | < 5s | 中 |
| 充值订单发货 | 优先队列+重试 | < 10s | 极高 |
| 赛季结算奖励 | 延迟队列 | 允许分钟级 | 高 |
| 跨服拍卖行同步 | 发布订阅 | 分钟级 | 中 |
| 游戏日志上报 | 批量队列 | 分钟级 | 低 |
| 段位晋级通知 | 即时队列 | < 1s | 中 |

## 二、充值订单可靠发货（生产者-消费者）

```python
# Python 伪代码展示架构（实际可用 Go/Java 实现）

import json
import time
import pika  # RabbitMQ Python 客户端

# ============ 消息结构 ============

class OrderMessage:
    def __init__(self, order_id, player_id, product_id, amount, currency):
        self.order_id = order_id
        self.player_id = player_id  
        self.product_id = product_id
        self.amount = amount
        self.currency = currency
        self.create_time = int(time.time())
        self.retry_count = 0

# ============ 生产者（订单服务）============

class OrderProducer:
    EXCHANGE_NAME = "game.orders"
    ROUTING_KEY = "order.payment.success"
    
    def publish_order(self, order: OrderMessage):
        connection = pika.BlockingConnection(
            pika.ConnectionParameters('localhost'))
        channel = connection.channel()
        
        # 声明交换机（持久化）
        channel.exchange_declare(
            exchange=self.EXCHANGE_NAME,
            exchange_type='direct',
            durable=True
        )
        
        message = json.dumps(order.__dict__)
        
        # 发布消息（持久化，重启不丢失）
        channel.basic_publish(
            exchange=self.EXCHANGE_NAME,
            routing_key=self.ROUTING_KEY,
            body=message,
            properties=pika.BasicProperties(
                delivery_mode=2,           # 持久化消息
                message_id=order.order_id,
                content_type='application/json'
            )
        )
        
        print(f"[Producer] Published order: {order.order_id}")
        connection.close()

# ============ 消费者（发货服务）============

class OrderDeliveryConsumer:
    QUEUE_NAME = "game.order.delivery"
    DLX_EXCHANGE = "game.orders.dlx"     # 死信交换机
    MAX_RETRIES = 3
    
    def process_order(self, channel, method, properties, body):
        order = json.loads(body)
        order_id = order['order_id']
        player_id = order['player_id']
        
        # 幂等性检查（防止重复发货）
        if self.is_already_delivered(order_id):
            print(f"[Consumer] Order {order_id} already delivered, ack and skip")
            channel.basic_ack(delivery_tag=method.delivery_tag)
            return
        
        try:
            # 执行发货
            self.deliver_items(player_id, order['product_id'], order['amount'])
            
            # 标记已发货（原子操作）
            self.mark_delivered(order_id)
            
            # 确认消息（从队列删除）
            channel.basic_ack(delivery_tag=method.delivery_tag)
            print(f"[Consumer] Order {order_id} delivered successfully")
            
        except Exception as e:
            retry_count = order.get('retry_count', 0)
            
            if retry_count < self.MAX_RETRIES:
                # 重试：延迟后重新放回队列
                order['retry_count'] = retry_count + 1
                time.sleep(2 ** retry_count)  # 指数退避
                
                channel.basic_nack(delivery_tag=method.delivery_tag, requeue=True)
                print(f"[Consumer] Order {order_id} retry {retry_count + 1}/{self.MAX_RETRIES}")
            else:
                # 超过重试次数，发送到死信队列（人工处理）
                channel.basic_nack(delivery_tag=method.delivery_tag, requeue=False)
                self.alert_ops(order_id, str(e))  # 触发运维告警
                print(f"[Consumer] Order {order_id} moved to DLQ after {self.MAX_RETRIES} retries")
    
    def is_already_delivered(self, order_id) -> bool:
        # Redis 查询是否已发货（SET NX 原子操作）
        return False  # 简化
    
    def mark_delivered(self, order_id):

        pass
    
    def deliver_items(self, player_id, product_id, amount):

        pass
    
    def alert_ops(self, order_id, error):
```

## 三、Kafka 游戏事件流

```go
// Go 代码示例：游戏事件通过 Kafka 驱动多个下游服务

package main

import (
    "encoding/json"
    "log"
    "github.com/segmentio/kafka-go"
)

// 游戏事件定义
type GameEvent struct {
    EventType string          `json:"event_type"`
    PlayerId  string          `json:"player_id"`
    ServerId  string          `json:"server_id"`
    Timestamp int64           `json:"timestamp"`
    Data      json.RawMessage `json:"data"`
}

// 常见事件类型
const (
    EventPlayerLevelUp     = "player.level_up"
    EventPlayerKill        = "pvp.player_kill"
    EventQuestComplete     = "quest.complete"
    EventSeasonRankChange  = "season.rank_change"
    EventPaymentSuccess    = "payment.success"
)

// 生产者：发布游戏事件
func PublishEvent(writer *kafka.Writer, event GameEvent) error {
    data, err := json.Marshal(event)
    if err != nil {
        return err
    }
    
    return writer.WriteMessages(nil, kafka.Message{
        Key:   []byte(event.PlayerId),  // 按玩家ID分区，保证同一玩家消息有序
        Value: data,
    })
}

// 消费者组示例
// 成就系统、排行榜系统、数据分析系统可以各自消费同一个 Topic
// Kafka 支持多个消费者组独立消费，互不影响
```

## 四、异步任务队列（赛季结算）

```python
# 赛季结算异步任务（可能涉及数百万玩家）

class SeasonSettlementTask:
    """
    赛季结算不能同步处理（数百万玩家同时处理会超时）
    使用分批异步任务：
    1. 拉取所有参与玩家列表（分页）
    2. 每批 1000 人推入队列
    3. Worker 并行消费，每人计算奖励并发邮件
    """
    
    BATCH_SIZE = 1000
    
    def start_settlement(self, season_id: str):
        # 获取总玩家数
        total = self.get_participant_count(season_id)
        batches = (total + self.BATCH_SIZE - 1) // self.BATCH_SIZE
        
        print(f"[Settlement] Season {season_id}: {total} players, {batches} batches")
        
        for batch_index in range(batches):
            offset = batch_index * self.BATCH_SIZE
            players = self.get_players_page(season_id, offset, self.BATCH_SIZE)
            
            # 推入消息队列
            self.queue.publish({
                "type": "season_settlement_batch",
                "season_id": season_id,
                "player_ids": players,
                "batch_index": batch_index,
                "total_batches": batches
            })
        
        print(f"[Settlement] All {batches} batches queued")
    
    def process_batch(self, message):
        for player_id in message['player_ids']:
            try:
                rank, score = self.get_player_rank(message['season_id'], player_id)
                rewards = self.calculate_rewards(rank, score)
                self.send_mail_reward(player_id, rewards, message['season_id'])
            except Exception as e:
                # 单个玩家失败不影响整批
                self.log_error(player_id, str(e))
```

## 五、架构要点

1. **幂等性是第一公民**：充值发货必须保证 exactly-once，使用分布式锁/Redis SET NX
2. **死信队列必须告警**：DLQ 中的消息需要人工干预，要有监控告警
3. **批量处理大任务**：赛季结算/活动奖励等批量操作必须异步分批
4. **消费速率控制**：防止消费过快压垮下游（限流/背压机制）
5. **消息顺序性**：Kafka 按 PlayerId 分区保证同一玩家消息有序
