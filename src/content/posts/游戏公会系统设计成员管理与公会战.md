## 一、公会数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 公会成员职位
/// </summary>
public enum GuildRank
{
    Member     = 0,  // 成员（最基础权限）
    Elite      = 1,  // 精英（可邀请新成员）
    Officer    = 2,  // 干部（可管理成员）
    ViceLeader = 3,  // 副会长
    Leader     = 4,  // 会长（最高权限）
}

/// <summary>
/// 公会权限
/// </summary>
[Flags]
public enum GuildPermission
{
    None            = 0,
    InviteMembers   = 1 << 0,  // 邀请成员
    KickMembers     = 1 << 1,  // 踢人
    ManageRanks     = 1 << 2,  // 管理职位
    StartGuildWar   = 1 << 3,  // 发起公会战
    ManageNotice    = 1 << 4,  // 修改公告
    ManageFunds     = 1 << 5,  // 管理公会资金
    DisbandGuild    = 1 << 6,  // 解散公会（仅会长）
}

/// <summary>
/// 公会信息
/// </summary>
[Serializable]
public class GuildInfo
{
    public string GuildId;
    public string Name;
    public string Tag;             // 公会标签（如[战锋]）
    public string Description;
    public string LeaderPlayerId;
    public int Level;              // 公会等级
    public long Experience;        // 公会经验
    public int MaxMembers;         // 最大成员数（随公会等级增加）
    public int MemberCount;
    public long TotalContribution; // 历史总贡献度
    public GuildFlags Flags;
    public long CreateTime;
    public string Announcement;   // 公会公告
}

[Flags]
public enum GuildFlags
{
    None            = 0,
    AutoAccept      = 1 << 0,   // 自动通过申请
    RequireInterview = 1 << 1,  // 需要面试
}

/// <summary>
/// 公会成员
/// </summary>
[Serializable]
public class GuildMember
{
    public string PlayerId;
    public string PlayerName;
    public int PlayerLevel;
    public GuildRank Rank;
    public long Contribution;       // 本周贡献度
    public long TotalContribution;  // 历史总贡献度
    public long JoinTime;
    public long LastActiveTime;
    public bool IsOnline;
}
```

## 二、公会权限系统

```csharp
/// <summary>
/// 公会权限管理
/// </summary>
public static class GuildPermissionHelper
{
    // 各职位拥有的权限
    private static readonly Dictionary<GuildRank, GuildPermission> RankPermissions 
        = new Dictionary<GuildRank, GuildPermission>
    {
        [GuildRank.Member]     = GuildPermission.None,
        [GuildRank.Elite]      = GuildPermission.InviteMembers,
        [GuildRank.Officer]    = GuildPermission.InviteMembers | 
                                  GuildPermission.KickMembers |
                                  GuildPermission.ManageNotice,
        [GuildRank.ViceLeader] = GuildPermission.InviteMembers | 
                                  GuildPermission.KickMembers |
                                  GuildPermission.ManageRanks |
                                  GuildPermission.ManageNotice |
                                  GuildPermission.StartGuildWar |
                                  GuildPermission.ManageFunds,
        [GuildRank.Leader]     = (GuildPermission)0x7FFFFFFF, // 所有权限
    };

    public static bool HasPermission(GuildRank rank, GuildPermission permission)
    {
        return (RankPermissions[rank] & permission) != 0;
    }

    public static bool CanKickMember(GuildRank operatorRank, GuildRank targetRank)
    {
        // 只能踢比自己职位低的成员
        return HasPermission(operatorRank, GuildPermission.KickMembers) &&
               operatorRank > targetRank;
    }

    public static bool CanPromote(GuildRank operatorRank, GuildRank targetCurrentRank, 
        GuildRank targetNewRank)
    {
        // 只能提升到不超过自己职位-1的位置
        return HasPermission(operatorRank, GuildPermission.ManageRanks) &&
               targetCurrentRank < operatorRank &&
               targetNewRank < operatorRank;
    }
}
```

## 三、公会管理器（客户端）

```csharp
/// <summary>
/// 公会管理器（客户端缓存）
/// </summary>
public class GuildManager : MonoBehaviour
{
    private static GuildManager instance;
    public static GuildManager Instance => instance;

    private GuildInfo myGuild;
    private List<GuildMember> members = new List<GuildMember>();
    private GuildMember myMember; // 当前玩家的公会成员信息
    
    public GuildInfo MyGuild => myGuild;
    public bool IsInGuild => myGuild != null;
    
    public event Action OnGuildInfoUpdated;
    public event Action OnMembersUpdated;
    public event Action<string> OnMemberJoined;   // playerId
    public event Action<string> OnMemberLeft;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    public void SetGuildData(GuildInfo guild, List<GuildMember> memberList, 
        string myPlayerId)
    {
        myGuild = guild;
        members = memberList;
        myMember = members.Find(m => m.PlayerId == myPlayerId);
        
        OnGuildInfoUpdated?.Invoke();
        OnMembersUpdated?.Invoke();
    }

    public bool HasPermission(GuildPermission permission)
    {
        if (myMember == null) return false;
        return GuildPermissionHelper.HasPermission(myMember.Rank, permission);
    }

    public async System.Threading.Tasks.Task<bool> InviteMember(string targetPlayerId)
    {
        if (!HasPermission(GuildPermission.InviteMembers))
        {
            UIManager.Instance?.ShowMessage("权限不足：无法邀请成员");
            return false;
        }
        
        try
        {
            await NetworkService.GuildInvite(myGuild.GuildId, targetPlayerId);
            UIManager.Instance?.ShowMessage("邀请已发送");
            return true;
        }
        catch (Exception e)
        {
            UIManager.Instance?.ShowMessage($"邀请失败: {e.Message}");
            return false;
        }
    }

    public async System.Threading.Tasks.Task<bool> KickMember(string targetPlayerId)
    {
        var target = members.Find(m => m.PlayerId == targetPlayerId);
        if (target == null) return false;
        
        if (!GuildPermissionHelper.CanKickMember(myMember.Rank, target.Rank))
        {
            UIManager.Instance?.ShowMessage("权限不足：无法踢出该成员");
            return false;
        }
        
        bool confirmed = await UIManager.Instance.ShowConfirmDialog(
            $"确定要踢出 {target.PlayerName}？");
        if (!confirmed) return false;
        
        await NetworkService.GuildKick(myGuild.GuildId, targetPlayerId);
        members.Remove(target);
        OnMembersUpdated?.Invoke();
        OnMemberLeft?.Invoke(targetPlayerId);
        
        return true;
    }

    public List<GuildMember> GetOnlineMembers() =>
        members.FindAll(m => m.IsOnline);

    public List<GuildMember> GetMembersByRank(GuildRank rank) =>
        members.FindAll(m => m.Rank == rank);
}
```

## 五、公会系统设计要点

| 要点 | 方案 |
|------|------|
| 权限分级 | 5级权限，会长拥有所有权限 |
| 防低级成员刷票 | 踢人/提升只能操作低职位成员 |
| 活跃度激励 | 贡献度排行榜，每周重置，激励活跃 |
| 公会战保护 | 开战前报名期，双方人数均衡才开打 |
| 成员上限 | 随公会等级增加，激励升级 |
| 公告同步 | 登录时同步，减少实时推送压力 |
