## 一、存档数据结构

```csharp
using System;
using System.IO;
using Newtonsoft.Json;
using UnityEngine;

[Serializable]
public class SaveData
{
    public int Version = 1;          // 存档版本（用于迁移）
    public string PlayerId;
    public long SaveTimestamp;       // UTC时间戳
    public string DeviceId;          // 最后保存的设备
    
    // 游戏数据
    public PlayerSaveData Player;
    public InventorySaveData Inventory;
    public QuestSaveData Quests;
    public WorldSaveData World;
}

[Serializable] public class PlayerSaveData { public int Level; public long Exp; public float PosX, PosY, PosZ; public string CurrentScene; }
[Serializable] public class InventorySaveData { public ItemStack[] Items; }
[Serializable] public class ItemStack { public string ItemId; public int Count; }
[Serializable] public class QuestSaveData { public string[] CompletedQuestIds; public ActiveQuestState[] ActiveQuests; }
[Serializable] public class ActiveQuestState { public string QuestId; public System.Collections.Generic.Dictionary<string, int> Progress; }
[Serializable] public class WorldSaveData { public string[] UnlockedAreas; }
```

## 二、本地存档管理

```csharp
public class SaveManager : MonoBehaviour
{
    private static SaveManager instance;
    public static SaveManager Instance => instance;

    private const string SAVE_FILE = "save.dat";
    private const string BACKUP_FILE = "save.bak";
    private static readonly byte[] EncryptKey = new byte[] { 0x47, 0x61, 0x6D, 0x65 };

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    string SavePath => Path.Combine(Application.persistentDataPath, SAVE_FILE);
    string BackupPath => Path.Combine(Application.persistentDataPath, BACKUP_FILE);

    public void Save(SaveData data)
    {
        data.SaveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        data.DeviceId = SystemInfo.deviceUniqueIdentifier;
        
        string json = JsonConvert.SerializeObject(data, Formatting.None);
        byte[] bytes = System.Text.Encoding.UTF8.GetBytes(json);
        bytes = XorEncrypt(bytes); // 简单加密防普通篡改
        
        // 先写备份，再写主文件（防止写入中断导致存档损坏）
        if (File.Exists(SavePath))
            File.Copy(SavePath, BackupPath, true);
        
        File.WriteAllBytes(SavePath, bytes);
        Debug.Log("[Save] 存档保存成功");
    }

    public SaveData Load()
    {
        try
        {
            if (File.Exists(SavePath))
                return LoadFromFile(SavePath);
        }
        catch (Exception e)
        {
            Debug.LogWarning($"[Save] 主存档损坏，尝试读取备份: {e.Message}");
            if (File.Exists(BackupPath))
                return LoadFromFile(BackupPath);
        }
        return null;
    }

    SaveData LoadFromFile(string path)
    {
        byte[] bytes = File.ReadAllBytes(path);
        bytes = XorEncrypt(bytes);
        string json = System.Text.Encoding.UTF8.GetString(bytes);
        var data = JsonConvert.DeserializeObject<SaveData>(json);
        return MigrateSaveData(data);
    }

    SaveData MigrateSaveData(SaveData data)
    {
        // 版本迁移
        if (data.Version < 2)
        {
            // v1→v2迁移逻辑
            data.Version = 2;
        }
        return data;
    }

    static byte[] XorEncrypt(byte[] data)
    {
        var result = new byte[data.Length];
        for (int i = 0; i < data.Length; i++)
            result[i] = (byte)(data[i] ^ EncryptKey[i % EncryptKey.Length]);
        return result;
    }

    public void DeleteSave()
    {
        if (File.Exists(SavePath)) File.Delete(SavePath);
        if (File.Exists(BackupPath)) File.Delete(BackupPath);
    }
}
```

## 三、云存档同步策略

| 场景 | 策略 |
|------|------|
| 新设备登录 | 检测云端时间戳，优先下载最新 |
| 双端冲突 | 弹窗让玩家选择（本地/云端）|
| 网络失败 | 降级本地存档，后台重试 |
| 存档压缩 | JSON gzip 压缩减少流量 |
| 自动同步 | 每10分钟+每次关卡结束同步 |
