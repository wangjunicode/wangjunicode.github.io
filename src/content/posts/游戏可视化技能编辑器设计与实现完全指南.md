# 游戏可视化技能编辑器设计与实现完全指南

## 一、技能编辑器整体架构

### 1.1 为什么需要可视化技能编辑器？

在大型游戏项目中，技能数量往往超过 500 个，且需要策划人员频繁调整参数和逻辑。如果将技能逻辑硬编码在程序中，每次修改都需要程序员介入、重新编译，效率低下且风险高。

**可视化技能编辑器的核心价值：**
- 策划可独立配置复杂技能逻辑，无需程序干预
- 支持运行时热修改，快速迭代
- 可视化调试，清晰展示技能执行流程
- 数据驱动，技能作为资产（ScriptableObject/JSON）管理

### 1.2 架构分层

```
┌─────────────────────────────────────────┐
│          Editor 层（Unity编辑器扩展）      │
│  ┌──────────────┐  ┌────────────────┐   │
│  │  Node Graph  │  │  Inspector 面板 │   │
│  │   (UIToolkit)│  │  参数配置       │   │
│  └──────────────┘  └────────────────┘   │
├─────────────────────────────────────────┤
│          数据层（序列化/反序列化）          │
│  SkillGraph.asset  ←→  SkillGraphData   │
│  (ScriptableObject)    (JSON Schema)    │
├─────────────────────────────────────────┤
│          运行时层（技能执行引擎）           │
│  SkillGraphRunner → NodeExecutor        │
│  技能状态机 ← → 节点图解析器             │
└─────────────────────────────────────────┘
```

```csharp
public class NodePort
{

    public string portName;
    public PortDirection direction;  // Input / Output
    public PortType type;            // Execution / Value
    public string valueType;         // 值类型（用于类型检查）
    public List<string> connections = new List<string>(); // 连接的端口ID
}

public enum PortDirection { Input, Output }
public enum PortType { Execution, Value }

// 技能节点基类
[Serializable]
public abstract class SkillNode
{
    [HideInInspector] public string nodeId;
    [HideInInspector] public string nodeType;
    [HideInInspector] public Vector2 position;   // 编辑器中的位置
    [HideInInspector] public string displayName;
    [HideInInspector] public Color  nodeColor = Color.white;
    
    public List<NodePort> inputPorts  = new List<NodePort>();
    public List<NodePort> outputPorts = new List<NodePort>();
    
    // 运行时执行接口
    public abstract void Execute(SkillExecuteContext context);
    
    // 获取连接的输出端口值（用于值类型端口）
    protected T GetInputValue<T>(string portId, SkillExecuteContext ctx, T defaultValue = default)
    {
        var port = inputPorts.Find(p => p.portId == portId);
        if (port == null || port.connections.Count == 0) return defaultValue;
        return ctx.GetPortValue<T>(port.connections[0]);
    }
}

[Serializable, SkillNodeMeta("Delay", "等待", "#2196F3")]
public class DelayNode : SkillNode
{
    public float delayTime = 0.5f;
    
    public override void Execute(SkillExecuteContext ctx)
    {
        ctx.StartCoroutineNode(nodeId, DelayCoroutine(ctx));
    }
    
    private System.Collections.IEnumerator DelayCoroutine(SkillExecuteContext ctx)
    {
        yield return new WaitForSeconds(delayTime);
        ctx.ExecuteNextNode(nodeId, "execution_out");
    }
}

// 播放动画节点
[Serializable, SkillNodeMeta("PlayAnimation", "播放动画", "#FF9800")]
public class PlayAnimationNode : SkillNode
{
    public string animationName;
    public float  crossFadeTime = 0.1f;
    public bool   waitForComplete = true;
    
    public override void Execute(SkillExecuteContext ctx)
    {
        var animator = ctx.Caster.GetComponent<Animator>();
        if (animator != null)
        {
            animator.CrossFadeInFixedTime(animationName, crossFadeTime);
            
            if (waitForComplete)
                ctx.StartCoroutineNode(nodeId, WaitAnimComplete(ctx, animator));
            else
                ctx.ExecuteNextNode(nodeId, "execution_out");
        }
    }
    
    private System.Collections.IEnumerator WaitAnimComplete(SkillExecuteContext ctx, Animator anim)
    {
        yield return null; // 等待一帧让动画生效
        var info = anim.GetCurrentAnimatorStateInfo(0);
        yield return new WaitForSeconds(info.length * (1f - info.normalizedTime));
        ctx.ExecuteNextNode(nodeId, "execution_out");
    }
}

// 造成伤害节点
[Serializable, SkillNodeMeta("DealDamage", "造成伤害", "#F44336")]
public class DealDamageNode : SkillNode
{
    public float baseDamage = 100f;
    public DamageType damageType = DamageType.Physical;
    public float damageRadius = 0f; // 0=单目标，>0=AOE范围
    
    public override void Execute(SkillExecuteContext ctx)
    {
        // 从值端口获取目标（支持动态绑定）
        var target = GetInputValue<GameObject>("target_in", ctx);
        
        if (damageRadius > 0f)
        {
            // AOE 伤害
            var hits = Physics.OverlapSphere(ctx.Caster.transform.position, damageRadius, ctx.EnemyLayerMask);
            foreach (var hit in hits)
            {
                ApplyDamage(hit.gameObject, ctx);
            }
        }
        else if (target != null)
        {
            ApplyDamage(target, ctx);
        }
        else if (ctx.Target != null)
        {
            ApplyDamage(ctx.Target, ctx);
        }
        
        ctx.ExecuteNextNode(nodeId, "execution_out");
    }
    
    private void ApplyDamage(GameObject target, SkillExecuteContext ctx)
    {
        float finalDamage = baseDamage * ctx.Caster.GetComponent<ICharacterStats>().AttackMultiplier;
        var damageable = target.GetComponent<IDamageable>();
        damageable?.TakeDamage(new DamageInfo
        {
            damage = finalDamage,
            type   = damageType,
            source = ctx.Caster,
        });
    }
}

// 播放特效节点
[Serializable, SkillNodeMeta("SpawnVFX", "生成特效", "#9C27B0")]
public class SpawnVFXNode : SkillNode
{
    public GameObject vfxPrefab;
    public VFXSpawnPosition spawnPosition = VFXSpawnPosition.OnCaster;
    public float lifetime = 3f;
    public bool attachToCaster = false;
    
    public enum VFXSpawnPosition { OnCaster, OnTarget, AtPosition }
    
    public override void Execute(SkillExecuteContext ctx)
    {
        if (vfxPrefab == null) { ctx.ExecuteNextNode(nodeId, "execution_out"); return; }
        
        Vector3 pos = spawnPosition switch
        {
            VFXSpawnPosition.OnTarget   => ctx.Target?.transform.position ?? ctx.Caster.transform.position,
            VFXSpawnPosition.OnCaster   => ctx.Caster.transform.position,
            _                           => ctx.Caster.transform.position,
        };
        
        var vfxGO = UnityEngine.Object.Instantiate(vfxPrefab, pos, Quaternion.identity);
        if (attachToCaster) vfxGO.transform.SetParent(ctx.Caster.transform, true);
        UnityEngine.Object.Destroy(vfxGO, lifetime);
        
        ctx.ExecuteNextNode(nodeId, "execution_out");
    }
}

// 条件分支节点
[Serializable, SkillNodeMeta("Branch", "条件分支", "#607D8B")]
public class BranchNode : SkillNode
{
    public enum ConditionType { TargetHPBelow, CasterMPAbove, RandomChance, IsBuffActive }
    
    public ConditionType conditionType;
    public float conditionValue = 0.5f;
    public string buffId;
    
    public override void Execute(SkillExecuteContext ctx)
    {
        bool result = EvaluateCondition(ctx);
        ctx.ExecuteNextNode(nodeId, result ? "true_out" : "false_out");
    }
    
    private bool EvaluateCondition(SkillExecuteContext ctx)
    {
        return conditionType switch
        {
            ConditionType.TargetHPBelow  => ctx.Target?.GetComponent<ICharacterStats>().HPRatio < conditionValue,
            ConditionType.CasterMPAbove  => ctx.Caster.GetComponent<ICharacterStats>().MPRatio > conditionValue,
            ConditionType.RandomChance   => UnityEngine.Random.value < conditionValue,
            ConditionType.IsBuffActive   => ctx.Caster.GetComponent<IBuffSystem>().HasBuff(buffId),
            _                            => false
        };
    }
}
```

```csharp
// SkillExecuteContext.cs
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class SkillExecuteContext
{
    public GameObject Caster { get; set; }
    public GameObject Target { get; set; }
    public Vector3    CastPosition { get; set; }
    public LayerMask  EnemyLayerMask { get; set; }
    
    // 当前执行的节点图
    private SkillGraphData _graphData;
    private Dictionary<string, SkillNode> _nodeMap;
    
    // 端口值缓存（用于值类型端口传递）
    private readonly Dictionary<string, object> _portValues = new Dictionary<string, object>();
    
    // 协程宿主（节点需要跑协程时使用）
    private MonoBehaviour _coroutineHost;
    
    // 等待节点完成的回调
    private readonly Dictionary<string, Action> _nodeCompleteCallbacks = new Dictionary<string, Action>();
    
    public SkillExecuteContext(MonoBehaviour host)
    {
        _coroutineHost = host;
    }
    
    public void Initialize(SkillGraphAsset graph, GameObject caster, GameObject target = null)
    {
        Caster = caster;
        Target = target;
        _graphData = graph.GetGraphData();
        _nodeMap = BuildNodeMap(_graphData);
    }
    
    // 执行指定节点的下一个连接节点
    public void ExecuteNextNode(string currentNodeId, string outputPortId)
    {
        var connection = _graphData.connections.Find(c => 
            c.fromNodeId == currentNodeId && c.fromPortId == outputPortId);
        
        if (connection == null) return; // 无后续节点，技能流程结束
        
        if (_nodeMap.TryGetValue(connection.toNodeId, out var nextNode))
        {
            nextNode.Execute(this);
        }
    }
    
    // 启动节点协程（异步执行）
    public void StartCoroutineNode(string nodeId, IEnumerator coroutine)
    {
        _coroutineHost.StartCoroutine(coroutine);
    }
    
    // 设置端口值（供值类型端口使用）
    public void SetPortValue(string portId, object value)
    {
        _portValues[portId] = value;
    }
    
    // 获取端口值
    public T GetPortValue<T>(string portId)
    {
        if (_portValues.TryGetValue(portId, out var value) && value is T typed)
            return typed;
        return default;
    }
    
    private Dictionary<string, SkillNode> BuildNodeMap(SkillGraphData data)
    {
        var map = new Dictionary<string, SkillNode>();
        foreach (var serialized in data.nodes)
        {
            var node = SkillNodeFactory.Deserialize(serialized);
            if (node != null) map[node.nodeId] = node;
        }
        return map;
    }
}
```

```csharp
// SkillNodeFactory.cs - 节点类型注册与反序列化
using System;
using System.Collections.Generic;
using System.Reflection;
using UnityEngine;

public static class SkillNodeFactory
{
    private static Dictionary<string, Type> _nodeTypeRegistry;
    
    static SkillNodeFactory()
    {
        _nodeTypeRegistry = new Dictionary<string, Type>();
        
        // 自动扫描所有带 SkillNodeMeta 标记的类
        foreach (var assembly in AppDomain.CurrentDomain.GetAssemblies())
        {
            foreach (var type in assembly.GetTypes())
            {
                var meta = type.GetCustomAttribute<SkillNodeMetaAttribute>();
                if (meta != null)
                {
                    _nodeTypeRegistry[meta.TypeId] = type;
                    Debug.Log($"[SkillNodeFactory] 注册节点: {meta.TypeId} -> {type.Name}");
                }
            }
        }
    }
    
    public static SkillNode Deserialize(SerializedNode serialized)
    {
        if (!_nodeTypeRegistry.TryGetValue(serialized.nodeType, out var type))
        {
            Debug.LogWarning($"[SkillNodeFactory] 未知节点类型: {serialized.nodeType}");
            return null;
        }
        
        var node = (SkillNode)JsonUtility.FromJson(serialized.jsonData, type);
        node.nodeId   = serialized.nodeId;
        node.nodeType = serialized.nodeType;
        node.position = serialized.position;
        return node;
    }
    
    public static SerializedNode Serialize(SkillNode node)
    {
        return new SerializedNode
        {
            nodeId   = node.nodeId,
            nodeType = node.nodeType,
            position = node.position,
            jsonData = JsonUtility.ToJson(node),
        };
    }
    
    // 获取所有已注册节点类型（Editor使用）
    public static IReadOnlyDictionary<string, Type> GetAllNodeTypes() => _nodeTypeRegistry;
    
    // 创建新节点实例
    public static SkillNode CreateNode(string typeId)
    {
        if (!_nodeTypeRegistry.TryGetValue(typeId, out var type)) return null;
        
        var node = (SkillNode)Activator.CreateInstance(type);
        node.nodeId   = Guid.NewGuid().ToString("N")[..8]; // 8位短ID
        node.nodeType = typeId;
        
        var meta = type.GetCustomAttribute<SkillNodeMetaAttribute>();
        node.displayName = meta?.DisplayName ?? typeId;
        if (ColorUtility.TryParseHtmlString(meta?.HexColor ?? "#FFFFFF", out var c))
            node.nodeColor = c;
        
        return node;
    }
}
```

```csharp
// SkillComponent.cs - 角色技能组件
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class SkillComponent : MonoBehaviour
{
    [Header("技能槽位")]
    public SkillGraphAsset[] skillSlots = new SkillGraphAsset[6];
    
    // 技能冷却状态
    private readonly Dictionary<string, float> _cooldowns = new Dictionary<string, float>();
    private readonly Dictionary<string, Coroutine> _runningSkills = new Dictionary<string, Coroutine>();
    
    private SkillExecuteContext _sharedContext;
    
    private void Awake()
    {
        _sharedContext = new SkillExecuteContext(this);
    }
    
    // 释放技能（通过槽位索引）
    public bool CastSkill(int slotIndex, GameObject target = null)
    {
        if (slotIndex < 0 || slotIndex >= skillSlots.Length) return false;
        var skill = skillSlots[slotIndex];
        if (skill == null) return false;
        
        return CastSkillById(skill.skillId, target);
    }
    
    // 释放技能（通过ID）
    public bool CastSkillById(string skillId, GameObject target = null)
    {
        var skill = System.Array.Find(skillSlots, s => s != null && s.skillId == skillId);
        if (skill == null) return false;
        
        // 检查冷却
        if (_cooldowns.TryGetValue(skillId, out float cdEnd) && Time.time < cdEnd)
        {
            Debug.Log($"[SkillComponent] 技能 {skillId} 冷却中，剩余: {cdEnd - Time.time:F1}s");
            return false;
        }
        
        // 检查 MP
        var stats = GetComponent<ICharacterStats>();
        if (stats != null && skill.manaCost > 0 && stats.CurrentMP < skill.manaCost)
        {
            Debug.Log($"[SkillComponent] 技能 {skillId} 魔法值不足");
            return false;
        }
        
        // 消耗 MP
        stats?.ConsumeMana(skill.manaCost);
        
        // 开始冷却
        _cooldowns[skillId] = Time.time + skill.cooldown;
        
        // 执行技能图
        _sharedContext.Initialize(skill, gameObject, target);
        var entry = FindEntryNode(skill.GetGraphData());
        if (entry == null) { Debug.LogWarning($"技能 {skillId} 没有入口节点"); return false; }
        
        entry.Execute(_sharedContext);
        return true;
    }
    
    public float GetCooldownRatio(string skillId)
    {
        if (!_cooldowns.TryGetValue(skillId, out float cdEnd)) return 0f;
        var skill = System.Array.Find(skillSlots, s => s != null && s.skillId == skillId);
        if (skill == null) return 0f;
        float elapsed = cdEnd - Time.time;
        return Mathf.Clamp01(elapsed / skill.cooldown);
    }
    
    private SkillNode FindEntryNode(SkillGraphData data)
    {
        var entrySerialized = data.nodes.Find(n => n.nodeType == "Entry");
        return entrySerialized != null ? SkillNodeFactory.Deserialize(entrySerialized) : null;
    }
}
```

```csharp
        // 右键菜单
        nodeCreationRequest = ctx => ShowNodeCreationMenu(ctx.screenMousePosition);
        
        // 注册连接验证回调
        graphViewChanged = OnGraphViewChanged;
    }
    
    private void ShowNodeCreationMenu(Vector2 screenPos)
    {
        var searchWindow = ScriptableObject.CreateInstance<SkillNodeSearchWindow>();
        searchWindow.Initialize(this, screenPos);
        SearchWindow.Open(new SearchWindowContext(screenPos), searchWindow);
    }
    
    private GraphViewChange OnGraphViewChanged(GraphViewChange change)
    {
        // 可以在这里处理节点删除、连接创建等事件
        return change;
    }
    
    public override List<Port> GetCompatiblePorts(Port startPort, NodeAdapter nodeAdapter)
    {
        var compatible = new List<Port>();
        ports.ForEach(port =>
        {
            // 不能连接到自身，方向必须相反，类型必须匹配
            if (startPort != port && 
                startPort.node != port.node && 
                startPort.direction != port.direction)
            {
                compatible.Add(port);
            }
        });
        return compatible;
    }
    
    public void LoadGraph(SkillGraphData data)
    {
        ClearAll();
        var nodeViewMap = new Dictionary<string, SkillNodeView>();
        
        // 创建节点视图
        foreach (var serialized in data.nodes)
        {
            var node = SkillNodeFactory.Deserialize(serialized);
            if (node == null) continue;
            var view = CreateNodeView(node);
            nodeViewMap[serialized.nodeId] = view;
        }
        
        // 恢复连接
        foreach (var conn in data.connections)
        {
            if (!nodeViewMap.TryGetValue(conn.fromNodeId, out var fromView)) continue;
            if (!nodeViewMap.TryGetValue(conn.toNodeId, out var toView)) continue;
            
            var fromPort = fromView.GetOutputPort(conn.fromPortId);
            var toPort   = toView.GetInputPort(conn.toPortId);
            
            if (fromPort != null && toPort != null)
            {
                var edge = fromPort.ConnectTo(toPort);
                AddElement(edge);
            }
        }
    }
    
    public SkillGraphData SaveGraph()
    {
        var data = new SkillGraphData();
        
        // 收集节点
        nodes.ForEach(n =>
        {
            if (n is SkillNodeView nodeView)
            {
                data.nodes.Add(SkillNodeFactory.Serialize(nodeView.GetSkillNode()));
            }
        });
        
        // 收集连接
        edges.ForEach(e =>
        {
            if (e.output.node is SkillNodeView fromView && e.input.node is SkillNodeView toView)
            {
                data.connections.Add(new NodeConnection
                {
                    fromNodeId = fromView.GetSkillNode().nodeId,
                    fromPortId = e.output.portName,
                    toNodeId   = toView.GetSkillNode().nodeId,
                    toPortId   = e.input.portName,
                });
            }
        });
        
        return data;
    }
    
    private SkillNodeView CreateNodeView(SkillNode node)
    {
        var view = new SkillNodeView(node);
        AddElement(view);
        return view;
    }
    
    public void ClearAll()
    {
        DeleteElements(graphElements.ToList());
    }
}
```

### 6.1 设计原则

| 原则 | 实践方式 |

### 6.2 扩展节点的步骤

```csharp
// 1. 继承 SkillNode，添加 SkillNodeMeta 标记
[Serializable, SkillNodeMeta("LaunchProjectile", "发射弹丸", "#E91E63")]
public class LaunchProjectileNode : SkillNode
{
    public GameObject projectilePrefab;
    public float      speed = 20f;
    public int        count = 1;
    public float      spreadAngle = 0f;
    
    // 2. 实现 Execute 方法
    public override void Execute(SkillExecuteContext ctx)
    {
        Vector3 dir = ctx.Target != null 
            ? (ctx.Target.transform.position - ctx.Caster.transform.position).normalized 
            : ctx.Caster.transform.forward;
        
        for (int i = 0; i < count; i++)
        {
            float angle = count > 1 
                ? Mathf.Lerp(-spreadAngle * 0.5f, spreadAngle * 0.5f, (float)i / (count - 1)) 
                : 0f;
            
            var spawnDir = Quaternion.Euler(0, angle, 0) * dir;
            var go = Object.Instantiate(projectilePrefab, 
                ctx.Caster.transform.position + spawnDir * 0.5f, 
                Quaternion.LookRotation(spawnDir));
            
            if (go.TryGetComponent<IProjectile>(out var proj))
            {
                proj.Initialize(ctx.Caster, speed, ctx.Target);
            }
        }
        
        ctx.ExecuteNextNode(nodeId, "execution_out");
    }
}
// 3. 自动注册！无需手动修改工厂或配置文件
```

## 七、总结

可视化技能编辑器是大型游戏项目的核心基础设施，关键要点：

1. **节点图 + 数据驱动**：将技能逻辑与代码解耦，策划可独立迭代
2. **ScriptableObject 存储**：利用 Unity 资产系统天然支持版本管理和引用
3. **工厂模式 + 反射注册**：添加新节点类型只需继承基类+标记属性，零改动框架层
4. **执行上下文**：统一传递施法者、目标、环境信息，节点间通过上下文通信
5. **运行时调试**：编辑器节点高亮显示执行流，极大提升调试效率
6. **异步节点**：延迟、等待动画完成等异步操作通过协程驱动，保持节点接口统一

结合 Unity UI Toolkit 的 GraphView，可以构建出媲美商业工具（如 Behavior Designer、Visual Scripting）的技能编辑体验。
