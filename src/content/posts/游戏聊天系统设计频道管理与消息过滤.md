## 二、聊天管理器

```csharp
public class ChatManager : MonoBehaviour
{
    private static ChatManager instance;
    public static ChatManager Instance => instance;

    private const int MAX_MESSAGES_PER_CHANNEL = 200;
    private const float SEND_COOLDOWN = 1f; // 发消息冷却

    private Dictionary<ChatChannel, List<ChatMessage>> messages 
        = new Dictionary<ChatChannel, List<ChatMessage>>();
    private float lastSendTime;
    private SensitiveWordFilter filter;

    public event Action<ChatMessage> OnMessageReceived;

    void Awake()
    {
        instance = this;
        filter = new SensitiveWordFilter();
        
        foreach (ChatChannel channel in Enum.GetValues(typeof(ChatChannel)))
            messages[channel] = new List<ChatMessage>();
    }

    public bool SendMessage(ChatChannel channel, string content)
    {
        if (Time.time - lastSendTime < SEND_COOLDOWN)
        {
            UIManager.Instance?.ShowMessage("发言太频繁，请稍后再试");
            return false;
        }
        
        if (string.IsNullOrWhiteSpace(content)) return false;
        if (content.Length > 200) { UIManager.Instance?.ShowMessage("消息过长"); return false; }
        
        // 发送到服务端（服务端会做二次校验）
        NetworkService.SendChatMessage(channel, content);
        lastSendTime = Time.time;
        return true;
    }

    /// <summary>
    /// 接收服务端推送的消息
    /// </summary>
    public void OnReceiveMessage(ChatMessage msg)
    {
        // 客户端过滤（服务端也会过滤，双重保障）
        msg.Content = filter.Filter(msg.Content);
        
        var channelMessages = messages[msg.Channel];
        channelMessages.Add(msg);
        
        // 保持上限
        if (channelMessages.Count > MAX_MESSAGES_PER_CHANNEL)
            channelMessages.RemoveAt(0);
        
        OnMessageReceived?.Invoke(msg);
    }

    public List<ChatMessage> GetMessages(ChatChannel channel)
        => messages.TryGetValue(channel, out var list) ? list : new List<ChatMessage>();
}

/// <summary>
/// 敏感词过滤（DFA算法）
/// </summary>
public class SensitiveWordFilter
{
    private Dictionary<char, object> root = new Dictionary<char, object>();

    public SensitiveWordFilter()
    {
        // 实际项目从服务端加载词库
        AddWords(new[] { "作弊", "外挂", "广告词" });
    }

    void AddWords(string[] words)
    {
        foreach (var word in words)
        {
            var node = root;
            foreach (char c in word)
            {
                if (!node.ContainsKey(c))
                    node[c] = new Dictionary<char, object>();
                node = (Dictionary<char, object>)node[c];
            }
            node['$'] = null; // 结束标记
        }
    }

    public string Filter(string text, char replacement = '*')
    {
        var chars = text.ToCharArray();
        int i = 0;
        while (i < chars.Length)
        {
            var node = root;
            int j = i;
            int lastEnd = -1;
            
            while (j < chars.Length && node.ContainsKey(chars[j]))
            {
                node = (Dictionary<char, object>)node[chars[j]];
                j++;
                if (node.ContainsKey('$')) lastEnd = j;
            }
            
            if (lastEnd > i)
            {
                for (int k = i; k < lastEnd; k++) chars[k] = replacement;
                i = lastEnd;
            }
            else i++;
        }
        return new string(chars);
    }
}
```

## 三、聊天UI设计

```csharp
public class ChatUIController : MonoBehaviour
{
    [SerializeField] private Transform messageContainer;
    [SerializeField] private GameObject messagePrefab;
    [SerializeField] private UnityEngine.UI.InputField inputField;
    [SerializeField] private ChatChannel currentChannel = ChatChannel.World;

    void Start()
    {
        ChatManager.Instance.OnMessageReceived += OnMessageReceived;
    }

    void OnMessageReceived(ChatMessage msg)
    {
        if (msg.Channel != currentChannel && 
            msg.Channel != ChatChannel.System) return;
        
        var go = Instantiate(messagePrefab, messageContainer);
        var item = go.GetComponent<ChatMessageItem>();
        item?.Setup(msg);
        
        // 滚动到底部
        Canvas.ForceUpdateCanvases();
        var scroll = messageContainer.GetComponentInParent<UnityEngine.UI.ScrollRect>();
        if (scroll != null) scroll.verticalNormalizedPosition = 0;
    }

    public void OnSendButton()
    {
        string content = inputField.text.Trim();
        if (string.IsNullOrEmpty(content)) return;
        
        ChatManager.Instance.SendMessage(currentChannel, content);
        inputField.text = "";
        inputField.ActivateInputField();
    }
}
```

## 四、聊天系统设计规范

| 规范 | 说明 |
|------|------|
| 发言冷却 | 防刷屏，1-3秒冷却 |
| 消息长度 | 200字符上限 |
| 敏感词 | DFA双重过滤（客户端+服务端）|
| 消息历史 | 客户端保留最近200条 |
| 频道隔离 | 世界/公会/队伍独立 |
| 举报功能 | 右键消息可举报 |
