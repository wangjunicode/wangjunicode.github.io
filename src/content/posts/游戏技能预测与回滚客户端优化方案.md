## 一、问题背景

在有网络延迟的多人游戏中，如果等待服务端确认后才执行技能：

```
玩家按键 → 发包 → 服务端处理(50-200ms) → 返回结果 → 客户端播放效果
```

这会导致明显的操作延迟感，破坏游戏体验。

**客户端预测**方案：
```
玩家按键 → 立即执行本地效果（乐观执行）
         → 同时发包给服务端
         → 服务端确认 → 微调最终状态
         → 服务端拒绝 → 回滚本地状态
```

## 二、技能预测框架

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 技能操作记录（用于回滚）
/// </summary>
public class SkillOperation
{
    public uint SequenceId;         // 序列号（单调递增）
    public string SkillId;
    public Vector3 CastPosition;
    public Vector3 CastDirection;
    public float CastTime;
    public OperationState State;
    
    // 执行前的状态快照（用于回滚）
    public PlayerStateSnapshot PreState;
    
    // 预测执行产生的效果列表
    public List<PredictedEffect> PredictedEffects = new List<PredictedEffect>();
    
    public enum OperationState
    {
        Pending,    // 已发送，等待服务端确认
        Confirmed,  // 服务端确认
        Rejected,   // 服务端拒绝，需要回滚
        Expired     // 超时未收到回应
    }
}

/// <summary>
/// 玩家状态快照
/// </summary>
[Serializable]
public class PlayerStateSnapshot
{
    public float Hp;
    public float Mp;
    public Dictionary<string, float> SkillCooldowns = new Dictionary<string, float>();
    public Vector3 Position;
    public List<string> ActiveBuffs = new List<string>();
    public long Timestamp;
}

/// <summary>
/// 预测产生的效果（待确认）
/// </summary>
public class PredictedEffect
{
    public string EffectType;   // "damage", "heal", "buff", "knockback"
    public string TargetId;
    public object Data;
    public bool IsConfirmed;
    
    // 对应的游戏对象（用于回滚时移除）
    public List<GameObject> SpawnedObjects = new List<GameObject>();
}

/// <summary>
/// 技能预测管理器
/// </summary>
public class SkillPredictionManager : MonoBehaviour
{
    private static SkillPredictionManager instance;
    public static SkillPredictionManager Instance => instance;

    [Header("配置")]
    [SerializeField] private float operationTimeout = 3f;   // 操作超时时间
    [SerializeField] private float rollbackBlendTime = 0.2f; // 回滚混合时间
    [SerializeField] private bool enablePrediction = true;
    [SerializeField] private bool logPredictions = false;

    private uint nextSequenceId = 1;
    
    // 待确认的操作队列（按序列号排序）
    private Dictionary<uint, SkillOperation> pendingOperations 
        = new Dictionary<uint, SkillOperation>();
    
    private INetworkClient networkClient;
    private ISkillExecutor skillExecutor;
    private PlayerStateManager stateManager;

    void Awake()
    {
        instance = this;
        networkClient = GetComponent<INetworkClient>();
        skillExecutor = GetComponent<ISkillExecutor>();
        stateManager = GetComponent<PlayerStateManager>();
    }

    void Update()
    {
        // 检测超时的操作
        float now = Time.time;
        var timedOut = new List<uint>();
        
        foreach (var kvp in pendingOperations)
        {
            if (now - kvp.Value.CastTime > operationTimeout)
                timedOut.Add(kvp.Key);
        }
        
        foreach (var id in timedOut)
        {
            var op = pendingOperations[id];
            op.State = SkillOperation.OperationState.Expired;
            HandleOperationResult(op, false, "timeout");
            pendingOperations.Remove(id);
        }
    }

    /// <summary>
    /// 尝试释放技能（主入口）
    /// </summary>
    public bool TryCastSkill(string skillId, Vector3 targetPos, Vector3 castDir)
    {
        // 本地前置检查（快速失败）
        if (!LocalPreCheck(skillId))
            return false;
        
        uint seqId = nextSequenceId++;
        
        // 保存当前状态快照
        var preState = stateManager.CaptureSnapshot();
        
        // 创建操作记录
        var operation = new SkillOperation
        {
            SequenceId = seqId,
            SkillId = skillId,
            CastPosition = transform.position,
            CastDirection = castDir,
            CastTime = Time.time,
            State = SkillOperation.OperationState.Pending,
            PreState = preState
        };
        
        pendingOperations[seqId] = operation;
        
        if (enablePrediction)
        {
            // 乐观执行：立即在本地播放效果
            ExecuteLocally(operation);
        }
        
        // 发送到服务端（异步）
        networkClient.SendSkillRequest(seqId, skillId, targetPos, castDir);
        
        if (logPredictions)
            Debug.Log($"[Prediction] Skill {skillId} cast, seq={seqId}");
        
        return true;
    }

    /// <summary>
    /// 服务端确认回调
    /// </summary>
    public void OnServerConfirm(uint seqId, ServerSkillResult result)
    {
        if (!pendingOperations.TryGetValue(seqId, out var op))
        {
            // 可能是超时后到达的确认，忽略
            return;
        }
        
        pendingOperations.Remove(seqId);
        
        if (result.Approved)
        {
            op.State = SkillOperation.OperationState.Confirmed;
            
            // 对比服务端结果与预测结果，修正偏差
            ReconcileWithServer(op, result);
            
            if (logPredictions)
                Debug.Log($"[Prediction] Skill {op.SkillId} confirmed, seq={seqId}");
        }
        else
        {
            op.State = SkillOperation.OperationState.Rejected;
            
            // 回滚本地预测效果
            Rollback(op, result.RejectionReason);
            
            if (logPredictions)
                Debug.Log($"[Prediction] Skill {op.SkillId} REJECTED: {result.RejectionReason}");
        }
        
        HandleOperationResult(op, result.Approved, result.RejectionReason);
    }

    bool LocalPreCheck(string skillId)
    {
        // 快速本地检查（不需要网络确认的逻辑）
        // 1. 冷却检查
        if (!CooldownManager.Instance.CanUse(skillId)) return false;
        
        // 2. 资源检查（MP等）
        var skillData = SkillDatabase.GetSkill(skillId);
        if (skillData != null && stateManager.CurrentMp < skillData.MpCost) return false;
        
        // 3. 状态检查（是否被沉默、眩晕）
        if (stateManager.HasDebuff("silence") || stateManager.HasDebuff("stun")) return false;
        
        return true;
    }

    void ExecuteLocally(SkillOperation op)
    {
        var skillData = SkillDatabase.GetSkill(op.SkillId);
        if (skillData == null) return;
        
        // 立即触发冷却（乐观）
        CooldownManager.Instance.TriggerCooldown(op.SkillId);
        
        // 立即消耗 MP（乐观）
        stateManager.ConsumeMp(skillData.MpCost);
        
        // 播放施法动画
        AnimationController.Instance?.PlayCastAnimation(op.SkillId);
        
        // 预测伤害效果（用虚线或半透明效果标记）
        skillExecutor.ExecutePrediction(op);
    }

    void ReconcileWithServer(SkillOperation op, ServerSkillResult result)
    {
        // 对比预测状态与服务端实际状态
        float hpDiff = Mathf.Abs(stateManager.CurrentHp - result.ServerHp);
        float mpDiff = Mathf.Abs(stateManager.CurrentMp - result.ServerMp);
        
        // 误差在阈值内：平滑修正
        if (hpDiff < 5f && mpDiff < 10f)
        {
            stateManager.SmoothCorrect(result.ServerHp, result.ServerMp, 0.1f);
        }
        else
        {
            // 误差较大：强制修正
            Debug.LogWarning($"[Prediction] Large discrepancy: hp diff={hpDiff}, mp diff={mpDiff}");
            stateManager.ForceCorrect(result.ServerHp, result.ServerMp);
        }
        
        // 更新冷却时间（服务端时间为准）
        if (result.ActualCooldown > 0)
            CooldownManager.Instance.ForceSetCooldown(op.SkillId, result.ActualCooldown);
        
        // 确认预测的效果（将虚线效果替换为实线效果）
        foreach (var effect in op.PredictedEffects)
        {
            effect.IsConfirmed = true;
            skillExecutor.ConfirmEffect(effect);
        }
    }

    void Rollback(SkillOperation op, string reason)
    {
        Debug.Log($"[Prediction] Rolling back skill {op.SkillId}: {reason}");
        
        // 恢复冷却
        CooldownManager.Instance.ResetCooldown(op.SkillId);
        
        // 恢复 MP
        stateManager.RestoreSnapshot(op.PreState);
        
        // 移除预测产生的游戏对象
        foreach (var effect in op.PredictedEffects)
        {
            foreach (var go in effect.SpawnedObjects)
            {
                if (go != null) Destroy(go);
            }
        }
        
        // 显示回滚反馈（避免玩家困惑）
        UIManager.Instance?.ShowMessage(GetRejectionMessage(reason));
        
        // 播放打断动画
        AnimationController.Instance?.PlayInterruptAnimation();
    }

    string GetRejectionMessage(string reason) => reason switch
    {
        "cooldown" => "技能冷却中",
        "insufficient_mp" => "法力值不足",
        "out_of_range" => "目标超出范围",
        "invalid_target" => "无效目标",
        "silenced" => "被沉默，无法使用技能",
        _ => "技能释放失败"
    };

    void HandleOperationResult(SkillOperation op, bool success, string reason) { }
}

/// <summary>
/// 服务端技能处理结果
/// </summary>
public class ServerSkillResult
{
    public bool Approved;
    public string RejectionReason;
    public float ServerHp;
    public float ServerMp;
    public float ActualCooldown;
    public List<ServerEffect> Effects;
}

public class ServerEffect
{
    public string TargetId;
    public string EffectType;
    public float Value;
}

// 接口定义
public interface INetworkClient
{
    void SendSkillRequest(uint seqId, string skillId, Vector3 targetPos, Vector3 dir);
}

public interface ISkillExecutor
{
    void ExecutePrediction(SkillOperation op);
    void ConfirmEffect(PredictedEffect effect);
}
```

## 三、高延迟降级策略

```csharp
/// <summary>
/// 延迟检测与预测策略自适应调整
/// </summary>
public class NetworkQualityAdaptor : MonoBehaviour
{
    [SerializeField] private float highLatencyThreshold = 150f; // ms
    
    private float currentLatency;
    
    void Update()
    {
        // 实时测量 RTT
        // currentLatency = networkClient.GetAverageRTT();
    }
    
    public PredictionStrategy GetCurrentStrategy()
    {
        return currentLatency switch
        {
            < 50f => PredictionStrategy.FullPrediction,    // <50ms: 全预测
            < 150f => PredictionStrategy.PartialPrediction, // <150ms: 部分预测
            < 300f => PredictionStrategy.DelayedPrediction, // <300ms: 延迟执行
            _ => PredictionStrategy.ServerAuthoritative     // >300ms: 等待服务端
        };
    }
}

public enum PredictionStrategy
{
    FullPrediction,       // 完全乐观执行 + 后台校验
    PartialPrediction,    // 动画立即，伤害数字等服务端确认
    DelayedPrediction,    // 50ms 后执行（期望在服务端返回前执行）
    ServerAuthoritative   // 完全等待服务端确认
}
```

## 四、设计要点总结

| 网络状况 | 推荐策略 | 玩家体验 |
|----------|----------|----------|
| <50ms | 全预测（乐观执行） | 与单机无区别 |
| 50-150ms | 动画预测+状态等服务端 | 略有感知但可接受 |
| 150-300ms | 延迟执行（凑近服务端时间） | 明显延迟 |
| >300ms | 服务端权威（等确认再执行） | 网络太差，无法平滑 |

**核心原则：**
1. 预测失败率决定玩家体验——前置校验越准确，回滚越少
2. 回滚时提供清晰的视觉/文字反馈，避免玩家困惑
3. 状态误差小于阈值时平滑修正，避免位置跳变
4. 序列号保证操作顺序，防止乱序响应导致错误回滚
