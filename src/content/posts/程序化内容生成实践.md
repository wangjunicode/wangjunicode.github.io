## 程序化内容生成的价值

```
手工设计内容 vs 程序化生成：

手工设计：
  优点：精确控制体验，质量有保证
  缺点：耗时（一关卡可能需要数周），重玩性低

程序化生成：
  优点：自动生成大量内容，高重玩性（Roguelike/Roguelite）
  缺点：难以保证质量，需要大量调参

适用场景：
  ✅ Roguelike 游戏（每次运行随机关卡）
  ✅ 开放世界地形（随机但可控的地形）
  ✅ 辅助工具（为关卡设计师提供初稿）
  ✅ 测试环境（快速生成大量测试场景）
```

## 一、随机数管理

### 1.1 可重复的随机

```csharp
/// <summary>
/// 可重复随机数生成器
/// 相同的 seed → 相同的随机序列 → 相同的地图
/// 这是实现"分享地图种子"功能的基础
/// </summary>
public class SeededRandom
{
    private System.Random _random;
    private readonly int _seed;
    
    public SeededRandom(int seed)
    {
        _seed = seed;
        _random = new System.Random(seed);
    }
    
    public void Reset() => _random = new System.Random(_seed);
    
    public int Next(int min, int max) => _random.Next(min, max);
    public float NextFloat(float min = 0, float max = 1)
        => (float)(_random.NextDouble() * (max - min) + min);
    public bool NextBool(float probability = 0.5f) => NextFloat() < probability;
    
    // 从列表中随机选择一个元素
    public T Choose<T>(IList<T> list) => list[Next(0, list.Count)];
    
    // Shuffle（Fisher-Yates 算法）
    public void Shuffle<T>(IList<T> list)
    {
        for (int i = list.Count - 1; i > 0; i--)
        {
            int j = Next(0, i + 1);
            (list[i], list[j]) = (list[j], list[i]);
        }
    }
}

// 使用：保存 seed，可以复现任何地图
var rng = new SeededRandom(seed: 12345);
GenerateLevel(rng); // 每次传入相同 seed，生成相同地图
```

## 二、地牢生成算法

### 2.1 BSP（Binary Space Partitioning）

```csharp
/// <summary>
/// BSP 地牢生成：递归分割空间，创建房间，用走廊连接
/// </summary>
public class BSPDungeonGenerator
{
    public int Width, Height;
    public int MinRoomSize, MaxRoomSize;
    
    private SeededRandom _rng;
    private int[,] _grid; // 0=墙, 1=地板
    
    public class BSPNode
    {
        public RectInt Area;       // 本节点覆盖的区域
        public RectInt Room;       // 本节点内的房间（叶节点才有）
        public BSPNode Left, Right; // 子节点
        
        public bool IsLeaf => Left == null && Right == null;
    }
    
    public int[,] Generate(int seed)
    {
        _rng = new SeededRandom(seed);
        _grid = new int[Width, Height];
        
        var root = new BSPNode { Area = new RectInt(0, 0, Width, Height) };
        Split(root, 4); // 最多分割4层
        
        CreateRooms(root);
        ConnectRooms(root);
        
        return _grid;
    }
    
    private void Split(BSPNode node, int maxDepth)
    {
        if (maxDepth == 0 || node.Area.width < MinRoomSize * 2 || node.Area.height < MinRoomSize * 2)
            return;
        
        bool splitHorizontal = _rng.NextBool();
        
        // 如果一个方向太窄，强制另一方向分割
        if (node.Area.width > node.Area.height * 1.25f) splitHorizontal = false;
        if (node.Area.height > node.Area.width * 1.25f) splitHorizontal = true;
        
        if (splitHorizontal)
        {
            int splitY = _rng.Next(node.Area.y + MinRoomSize, node.Area.yMax - MinRoomSize);
            
            node.Left = new BSPNode 
            { 
                Area = new RectInt(node.Area.x, node.Area.y, node.Area.width, splitY - node.Area.y) 
            };
            node.Right = new BSPNode 
            { 
                Area = new RectInt(node.Area.x, splitY, node.Area.width, node.Area.yMax - splitY) 
            };
        }
        else
        {
            int splitX = _rng.Next(node.Area.x + MinRoomSize, node.Area.xMax - MinRoomSize);
            
            node.Left = new BSPNode 
            { 
                Area = new RectInt(node.Area.x, node.Area.y, splitX - node.Area.x, node.Area.height) 
            };
            node.Right = new BSPNode 
            { 
                Area = new RectInt(splitX, node.Area.y, node.Area.xMax - splitX, node.Area.height) 
            };
        }
        
        Split(node.Left, maxDepth - 1);
        Split(node.Right, maxDepth - 1);
    }
    
    private void CreateRooms(BSPNode node)
    {
        if (node.IsLeaf)
        {
            // 在叶节点区域内创建一个随机大小的房间
            int roomW = _rng.Next(MinRoomSize, Mathf.Min(MaxRoomSize, node.Area.width - 1));
            int roomH = _rng.Next(MinRoomSize, Mathf.Min(MaxRoomSize, node.Area.height - 1));
            int roomX = _rng.Next(node.Area.x, node.Area.xMax - roomW);
            int roomY = _rng.Next(node.Area.y, node.Area.yMax - roomH);
            
            node.Room = new RectInt(roomX, roomY, roomW, roomH);
            
            // 填充房间地板
            for (int x = roomX; x < roomX + roomW; x++)
                for (int y = roomY; y < roomY + roomH; y++)
                    _grid[x, y] = 1;
        }
        else
        {
            if (node.Left != null) CreateRooms(node.Left);
            if (node.Right != null) CreateRooms(node.Right);
        }
    }
    
    private Vector2Int GetCenter(BSPNode node)
    {
        if (node.IsLeaf)
            return new Vector2Int(node.Room.x + node.Room.width / 2, node.Room.y + node.Room.height / 2);
        
        var leftCenter = node.Left != null ? GetCenter(node.Left) : Vector2Int.zero;
        var rightCenter = node.Right != null ? GetCenter(node.Right) : Vector2Int.zero;
        return (leftCenter + rightCenter) / 2;
    }
    
    private void ConnectRooms(BSPNode node)
    {
        if (node.IsLeaf) return;
        
        if (node.Left != null) ConnectRooms(node.Left);
        if (node.Right != null) ConnectRooms(node.Right);
        
        // 连接两个子节点的中心
        if (node.Left != null && node.Right != null)
        {
            var a = GetCenter(node.Left);
            var b = GetCenter(node.Right);
            CarveCorridorBetween(a, b);
        }
    }
    
    private void CarveCorridorBetween(Vector2Int a, Vector2Int b)
    {
        // L 形走廊
        if (_rng.NextBool())
        {
            CarveLine(a.x, b.x, a.y, horizontal: true);   // 先横
            CarveLine(a.y, b.y, b.x, horizontal: false);  // 再竖
        }
        else
        {
            CarveLine(a.y, b.y, a.x, horizontal: false);  // 先竖
            CarveLine(a.x, b.x, b.y, horizontal: true);   // 再横
        }
    }
    
    private void CarveLine(int from, int to, int fixed_, bool horizontal)
    {
        int min = Mathf.Min(from, to);
        int max = Mathf.Max(from, to);
        
        for (int i = min; i <= max; i++)
        {
            int x = horizontal ? i : fixed_;
            int y = horizontal ? fixed_ : i;
            
            if (x >= 0 && x < Width && y >= 0 && y < Height)
                _grid[x, y] = 1;
        }
    }
}
```

## 三、Wave Function Collapse（WFC）

### 3.1 WFC 基本原理

```
WFC 是从量子力学概念借鉴的算法：

基本思想：
  定义一组 Tile（小块）和它们之间的连接规则
  （例：草地 Tile 只能与草地/路/水相邻）
  
  算法从所有可能性中，逐步"坍缩"到唯一确定的结果
  
步骤：
  1. 初始化：每个格子包含所有可能的 Tile
  2. 选择熵最低的格子（可能性最少的）
  3. 坍缩：随机选择一种可能
  4. 传播：根据连接规则，更新相邻格子的可能性
  5. 重复 2~4，直到所有格子确定

优点：
  生成结果"局部正确"（符合约束规则）
  可以学习已有的地图样式（从示例提取规则）

缺点：
  有时候会遇到矛盾（无解），需要回溯
  规则复杂时性能较慢
```

### 3.2 简单 WFC 实现

```csharp
/// <summary>
/// 2D WFC 地图生成（简化版）
/// </summary>
public class SimpleWFC
{
    // 每种 Tile 允许在哪些方向连接哪些 Tile
    private Dictionary<int, HashSet<int>[]> _constraints;
    // [tileId][direction] = 可以相邻的 Tile ID 集合
    // direction: 0=右, 1=下, 2=左, 3=上
    
    private int[,] _grid;
    private HashSet<int>[,] _possibilities; // 每个格子当前可能的 Tile
    private int _width, _height;
    private SeededRandom _rng;
    
    public void SetConstraints(Dictionary<int, HashSet<int>[]> constraints)
    {
        _constraints = constraints;
    }
    
    public int[,] Generate(int width, int height, int seed)
    {
        _width = width;
        _height = height;
        _rng = new SeededRandom(seed);
        _grid = new int[width, height];
        
        // 初始化：每个格子都可能是任何 Tile
        _possibilities = new HashSet<int>[width, height];
        var allTiles = new HashSet<int>(_constraints.Keys);
        
        for (int x = 0; x < width; x++)
        for (int y = 0; y < height; y++)
            _possibilities[x, y] = new HashSet<int>(allTiles);
        
        // 逐步坍缩
        while (true)
        {
            var (cx, cy) = FindLowestEntropy();
            if (cx < 0) break; // 全部确定，完成
            
            Collapse(cx, cy);
            Propagate(cx, cy);
        }
        
        return _grid;
    }
    
    private (int, int) FindLowestEntropy()
    {
        int minEntropy = int.MaxValue;
        int resultX = -1, resultY = -1;
        
        for (int x = 0; x < _width; x++)
        for (int y = 0; y < _height; y++)
        {
            int count = _possibilities[x, y].Count;
            if (count > 1 && count < minEntropy)
            {
                minEntropy = count;
                resultX = x;
                resultY = y;
            }
        }
        
        return (resultX, resultY);
    }
    
    private void Collapse(int x, int y)
    {
        var choices = _possibilities[x, y].ToList();
        int chosen = _rng.Choose(choices);
        
        _possibilities[x, y] = new HashSet<int> { chosen };
        _grid[x, y] = chosen;
    }
    
    private void Propagate(int startX, int startY)
    {
        var queue = new Queue<(int, int)>();
        queue.Enqueue((startX, startY));
        
        while (queue.Count > 0)
        {
            var (x, y) = queue.Dequeue();
            int[] dx = { 1, 0, -1, 0 };
            int[] dy = { 0, 1, 0, -1 };
            
            for (int dir = 0; dir < 4; dir++)
            {
                int nx = x + dx[dir];
                int ny = y + dy[dir];
                
                if (nx < 0 || nx >= _width || ny < 0 || ny >= _height) continue;
                if (_possibilities[nx, ny].Count <= 1) continue;
                
                // 计算 (nx, ny) 处合法的 Tile：
                // 必须与 (x, y) 当前所有可能 Tile 在 dir 方向上兼容
                var validTiles = new HashSet<int>();
                int oppDir = (dir + 2) % 4; // 反方向
                
                foreach (int srcTile in _possibilities[x, y])
                {
                    if (_constraints.TryGetValue(srcTile, out var compatible))
                        validTiles.UnionWith(compatible[dir]);
                }
                
                // 去除不合法的 Tile
                int before = _possibilities[nx, ny].Count;
                _possibilities[nx, ny].IntersectWith(validTiles);
                
                if (_possibilities[nx, ny].Count < before)
                    queue.Enqueue((nx, ny)); // 有变化，继续传播
            }
        }
    }
}
```

## 四、实用：程序化地形生成

### 4.1 Perlin Noise 地形

```csharp
public class ProceduralTerrain : MonoBehaviour
{
    [Header("地形参数")]
    [SerializeField] private int width = 256;
    [SerializeField] private int height = 256;
    [SerializeField] private float scale = 50f;
    [SerializeField] private int octaves = 4;    // 细节层数
    [SerializeField] private float persistence = 0.5f; // 每层振幅衰减
    [SerializeField] private float lacunarity = 2f;    // 每层频率倍增
    [SerializeField] private int seed;
    
    [SerializeField] private Terrain _terrain;
    
    void GenerateTerrain()
    {
        float[,] heightMap = new float[width, height];
        
        var rng = new SeededRandom(seed);
        Vector2 offset = new Vector2(rng.NextFloat(-100, 100), rng.NextFloat(-100, 100));
        
        float maxHeight = 0, minHeight = float.MaxValue;
        
        for (int y = 0; y < height; y++)
        for (int x = 0; x < width; x++)
        {
            float amplitude = 1f;
            float frequency = 1f;
            float noiseHeight = 0f;
            
            // 多层 Perlin Noise 叠加（Fractal Brownian Motion）
            for (int i = 0; i < octaves; i++)
            {
                float sampleX = (x + offset.x) / scale * frequency;
                float sampleY = (y + offset.y) / scale * frequency;
                
                noiseHeight += Mathf.PerlinNoise(sampleX, sampleY) * amplitude;
                
                amplitude *= persistence;
                frequency *= lacunarity;
            }
            
            heightMap[x, y] = noiseHeight;
            maxHeight = Mathf.Max(maxHeight, noiseHeight);
            minHeight = Mathf.Min(minHeight, noiseHeight);
        }
        
        // 归一化到 [0, 1]
        for (int y = 0; y < height; y++)
        for (int x = 0; x < width; x++)
            heightMap[x, y] = Mathf.InverseLerp(minHeight, maxHeight, heightMap[x, y]);
        
        // 应用到 Unity Terrain
        _terrain.terrainData.SetHeights(0, 0, heightMap);
    }
}
```

## 总结

PCG 技术的选型指南：

| 需求 | 推荐算法 |
|------|---------|
| 房间地牢（Roguelike） | BSP 分割 |
| 洞穴地图 | Cellular Automata |
| 有风格约束的地图 | WFC（波函数坍缩） |
| 开放世界地形 | Perlin Noise + FBM |
| 道路网络 | L-System / 随机游走 |

**设计建议**：
- PCG 不是"随机"，是"可控的随机"
- 好的 PCG 系统有丰富的参数让设计师调整
- 必须有 Seed 机制，让有趣的结果可以重现和分享
