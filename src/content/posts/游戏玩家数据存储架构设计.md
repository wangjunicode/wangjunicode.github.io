## 一、玩家数据的分层架构

游戏玩家数据按照性质分为以下层次：

```
第一层（设备本地）：显示设置、操作配置、画质设置
第二层（账户级）：角色数据、背包道具、任务进度
第三层（服务端权威）：货币、排行榜、PvP战绩、充值记录
第四层（实时同步）：在线状态、组队信息、副本状态
```

## 二、本地存储方案

### 2.1 存储方案选择

```csharp
// 方案对比：
// PlayerPrefs    → 键值对，仅适合简单配置（不要存游戏存档！）
// JSON文件       → 可读性好，适合调试，生产环境需加密
// BinaryFormatter → 已弃用（Unity 2022+不推荐），安全性差
// MessagePack    → 高性能二进制序列化（推荐）
// SQLite         → 复杂查询场景（背包搜索等）

// 推荐方案：重要存档用 MessagePack + AES加密
```

## 三、存档管理器

```csharp
/// <summary>
/// 存档管理器（支持本地读写、加密、版本迁移）
/// </summary>
public class SaveManager : MonoBehaviour
{
    private static SaveManager instance;
    public static SaveManager Instance => instance;

    [Header("存档配置")]
    [SerializeField] private bool encryptSave = true;
    [SerializeField] private int autoSaveIntervalSeconds = 30;
    [SerializeField] private int maxBackupCount = 3;

    private PlayerSaveData currentSave;
    private float autoSaveTimer;
    
    // AES 加密密钥（实际项目应从服务端获取或设备绑定）
    private const string EncryptionKey = "GameSaveKey2026!"; // 16字节
    
    private string SavePath => 
        System.IO.Path.Combine(Application.persistentDataPath, "saves", "player.sav");
    
    private string BackupPath(int index) => 
        System.IO.Path.Combine(Application.persistentDataPath, "saves", $"backup_{index}.sav");

    void Awake()
    {
        instance = this;
        System.IO.Directory.CreateDirectory(
            System.IO.Path.GetDirectoryName(SavePath));
    }

    void Update()
    {
        autoSaveTimer += Time.deltaTime;
        if (autoSaveTimer >= autoSaveIntervalSeconds)
        {
            autoSaveTimer = 0;
            SaveLocal(isAutoSave: true);
        }
    }

    /// <summary>
    /// 加载存档
    /// </summary>
    public PlayerSaveData LoadOrCreate(string playerId)
    {
        if (System.IO.File.Exists(SavePath))
        {
            try
            {
                var loaded = LoadFromFile(SavePath);
                if (loaded != null && loaded.PlayerId == playerId)
                {
                    currentSave = loaded;
                    MigrateIfNeeded(currentSave);
                    return currentSave;
                }
            }
            catch (Exception e)
            {
                Debug.LogError($"Save load failed: {e.Message}, trying backup...");
                currentSave = TryLoadBackup(playerId);
            }
        }
        
        if (currentSave == null)
        {
            // 创建新存档
            currentSave = CreateNewSave(playerId);
        }
        
        return currentSave;
    }

    PlayerSaveData LoadFromFile(string path)
    {
        byte[] bytes = System.IO.File.ReadAllBytes(path);
        
        if (encryptSave)
            bytes = DecryptBytes(bytes);
        
        string json = System.Text.Encoding.UTF8.GetString(bytes);
        return JsonUtility.FromJson<PlayerSaveData>(json);
    }

    /// <summary>
    /// 保存到本地
    /// </summary>
    public void SaveLocal(bool isAutoSave = false)
    {
        if (currentSave == null) return;
        
        currentSave.LastSaveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        
        try
        {
            // 备份旧存档
            if (!isAutoSave)
                RotateBackups();
            
            // 序列化
            string json = JsonUtility.ToJson(currentSave, prettyPrint: false);
            byte[] bytes = System.Text.Encoding.UTF8.GetBytes(json);
            
            if (encryptSave)
                bytes = EncryptBytes(bytes);
            
            // 原子写入（先写临时文件，再替换）
            string tempPath = SavePath + ".tmp";
            System.IO.File.WriteAllBytes(tempPath, bytes);
            System.IO.File.Move(tempPath, SavePath);
            
            if (!isAutoSave)
                Debug.Log($"Save successful: {bytes.Length} bytes");
        }
        catch (Exception e)
        {
            Debug.LogError($"Save failed: {e.Message}");
        }
    }

    void RotateBackups()
    {
        for (int i = maxBackupCount - 1; i >= 1; i--)
        {
            string prev = BackupPath(i - 1);
            string next = BackupPath(i);
            if (System.IO.File.Exists(prev))
            {
                if (System.IO.File.Exists(next))
                    System.IO.File.Delete(next);
                System.IO.File.Copy(prev, next);
            }
        }
        
        if (System.IO.File.Exists(SavePath))
            System.IO.File.Copy(SavePath, BackupPath(0), overwrite: true);
    }

    PlayerSaveData TryLoadBackup(string playerId)
    {
        for (int i = 0; i < maxBackupCount; i++)
        {
            string path = BackupPath(i);
            if (!System.IO.File.Exists(path)) continue;
            
            try
            {
                var backup = LoadFromFile(path);
                if (backup?.PlayerId == playerId) return backup;
            }
            catch { }
        }
        return null;
    }

    PlayerSaveData CreateNewSave(string playerId)
    {
        return new PlayerSaveData
        {
            PlayerId = playerId,
            SaveVersion = CurrentSaveVersion,
            LastSaveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
            Character = new CharacterData
            {
                Name = "Player",
                Level = 1,
                Hp = 100, MaxHp = 100,
                Mp = 50, MaxMp = 50,
                LastScene = "Town",
                LastPosition = Vector3Data.From(Vector3.zero)
            },
            Inventory = new List<InventoryItemData>(),
            Quests = new Dictionary<string, QuestProgress>(),
            WorldFlags = new Dictionary<string, bool>(),
            AchievementProgress = new Dictionary<string, int>(),
            UnlockedItems = new List<string>(),
            Statistics = new PlayerStatistics()
        };
    }

    // ============ AES 加密 ============

    byte[] EncryptBytes(byte[] data)
    {
        using var aes = System.Security.Cryptography.Aes.Create();
        aes.Key = System.Text.Encoding.UTF8.GetBytes(EncryptionKey);
        aes.GenerateIV();
        
        using var ms = new System.IO.MemoryStream();
        ms.Write(aes.IV, 0, aes.IV.Length); // 前16字节存IV
        
        using (var cs = new System.Security.Cryptography.CryptoStream(
            ms, aes.CreateEncryptor(), System.Security.Cryptography.CryptoStreamMode.Write))
        {
            cs.Write(data, 0, data.Length);
        }
        
        return ms.ToArray();
    }

    byte[] DecryptBytes(byte[] data)
    {
        using var aes = System.Security.Cryptography.Aes.Create();
        aes.Key = System.Text.Encoding.UTF8.GetBytes(EncryptionKey);
        
        byte[] iv = new byte[16];
        Array.Copy(data, 0, iv, 0, 16);
        aes.IV = iv;
        
        using var ms = new System.IO.MemoryStream(data, 16, data.Length - 16);
        using var cs = new System.Security.Cryptography.CryptoStream(
            ms, aes.CreateDecryptor(), System.Security.Cryptography.CryptoStreamMode.Read);
        using var result = new System.IO.MemoryStream();
        cs.CopyTo(result);
        return result.ToArray();
    }

    // ============ 存档版本迁移 ============

    private const int CurrentSaveVersion = 3;

    void MigrateIfNeeded(PlayerSaveData save)
    {
        if (save.SaveVersion >= CurrentSaveVersion) return;
        
        Debug.Log($"Migrating save from v{save.SaveVersion} to v{CurrentSaveVersion}");
        
        // 版本迁移链
        if (save.SaveVersion < 2)
            MigrateV1ToV2(save);
        
        if (save.SaveVersion < 3)
            MigrateV2ToV3(save);
        
        save.SaveVersion = CurrentSaveVersion;
        SaveLocal();
    }

    void MigrateV1ToV2(PlayerSaveData save)
    {
        // V1->V2：新增统计数据字段
        if (save.Statistics == null)
            save.Statistics = new PlayerStatistics();
    }

    void MigrateV2ToV3(PlayerSaveData save)
    {
        // V2->V3：背包数据结构变更，添加 UniqueId
        foreach (var item in save.Inventory ?? new List<InventoryItemData>())
        {
            if (string.IsNullOrEmpty(item.UniqueId))
                item.UniqueId = Guid.NewGuid().ToString("N");
        }
    }

    public PlayerSaveData CurrentSave => currentSave;
}
```

## 四、服务器同步策略

```csharp
/// <summary>
/// 玩家数据服务器同步器
/// 采用"离线优先"策略：本地操作立即生效，后台同步
/// </summary>
public class PlayerDataSyncService : MonoBehaviour
{
    [SerializeField] private float syncInterval = 60f;  // 每60秒同步一次
    [SerializeField] private bool syncOnFocusLost = true;

    private float syncTimer;
    private bool isSyncing;

    void Start()
    {
        Application.focusChanged += OnFocusChanged;
    }

    void Update()
    {
        syncTimer += Time.deltaTime;
        if (syncTimer >= syncInterval)
        {
            syncTimer = 0;
            StartCoroutine(SyncToServer());
        }
    }

    void OnFocusChanged(bool hasFocus)
    {
        if (!hasFocus && syncOnFocusLost)
            StartCoroutine(SyncToServer());
    }

    System.Collections.IEnumerator SyncToServer()
    {
        if (isSyncing) yield break;
        
        var save = SaveManager.Instance.CurrentSave;
        if (save == null || save.DirtyModules.Count == 0) yield break;
        
        isSyncing = true;
        
        // 只同步有改动的模块（增量同步）
        var payload = BuildSyncPayload(save);
        
        // 发送到服务器
        yield return SendToServer(payload, response =>
        {
            if (response.Success)
            {
                save.LastSyncTimestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
                save.DirtyModules.Clear();
            }
            else if (response.HasConflict)
            {
                // 冲突解决：服务器版本更新 -> 拉取服务器数据
                StartCoroutine(PullFromServer());
            }
        });
        
        isSyncing = false;
    }

    object BuildSyncPayload(PlayerSaveData save)
    {
        // 构建只包含脏数据的增量同步包
        var payload = new Dictionary<string, object>
        {
            ["playerId"] = save.PlayerId,
            ["clientTimestamp"] = save.LastSaveTimestamp
        };
        
        if (save.DirtyModules.Contains("character"))
            payload["character"] = save.Character;
        
        if (save.DirtyModules.Contains("inventory"))
            payload["inventory"] = save.Inventory;
        
        if (save.DirtyModules.Contains("quests"))
            payload["quests"] = save.Quests;
        
        return payload;
    }

    System.Collections.IEnumerator PullFromServer()
    {
        // 从服务器拉取最新数据，处理冲突
        Debug.Log("Conflict detected, pulling server data...");
        yield return null;
    }

    System.Collections.IEnumerator SendToServer(object payload, Action<SyncResponse> callback)
    {
        // 使用 UnityWebRequest 发送数据
        string json = JsonUtility.ToJson(payload);
        // var request = UnityWebRequest.Put(serverUrl, json);
        // yield return request.SendWebRequest();
        yield return null;
        callback?.Invoke(new SyncResponse { Success = true });
    }
}

public class SyncResponse
{
    public bool Success;
    public bool HasConflict;
    public string ServerData;
}
```

## 五、架构最佳实践

| 数据类型 | 存储位置 | 同步策略 |
|----------|----------|----------|
| 操作设置、画质 | 本地PlayerPrefs | 不需要同步 |
| 角色/背包/任务 | 本地文件+服务器 | 离线优先，定期同步 |
| 货币、排行榜 | 服务端权威 | 每次操作立即验证 |
| 实时状态 | 内存+服务器 | WebSocket实时 |

**设计原则：**
1. 使用原子写入防止存档损坏（写临时文件再替换）
2. 维护多份备份，出错时自动降级
3. 存档版本号 + 迁移链，支持无痛升级数据格式
4. 脏标记实现增量同步，减少带宽消耗
5. 货币等关键数据服务端验证，防止客户端作弊
