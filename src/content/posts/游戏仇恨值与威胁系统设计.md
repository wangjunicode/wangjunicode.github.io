## 一、仇恨系统的设计目标

仇恨（Aggro/Threat）系统是 MMORPG 团队玩法的核心机制：

1. **坦克职业的价值**：高仇恨量使坦克持续吸引怪物攻击
2. **治疗威胁**：治疗行为产生威胁，需要坦克控制节奏
3. **输出节奏**：输出职业不能超过坦克的仇恨，否则被怪攻击
4. **团队协作**：仇恨管理成为团队协作的核心要素

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using UnityEngine;

/// <summary>
/// 单个目标的仇恨记录
/// </summary>
public class ThreatEntry
{
    public string TargetId;
    public GameObject Target;
    public float ThreatValue;           // 当前仇恨值
    public float LastInteractionTime;   // 最后互动时间（用于仇恨衰减）
    public bool IsTaunted;              // 是否被嘲讽锁定
    public float TauntExpireTime;       // 嘲讽到期时间
}

/// <summary>
/// 仇恨修改器（来自技能/Buff）
/// </summary>
[Serializable]
public class ThreatModifier
{
    public string ModifierId;
    public ThreatModifierType Type;
    public float Value;               // 乘数或加值
    public float ExpireTime;          // 到期时间

    public enum ThreatModifierType
    {
        Multiplier,      // 仇恨乘数（嘲讽 x3，远程 x1.2 等）
        FlatBonus,       // 固定加值
        Reduction,       // 仇恨减少百分比
    }
}

/// <summary>
/// 仇恨产生事件
/// </summary>
public struct ThreatEvent
{
    public string SourceId;
    public ThreatSource Source;
    public float BaseThreatAmount;    // 基础仇恨量
    
    public enum ThreatSource
    {
        Damage,      // 造成伤害
        Healing,     // 治疗（通常威胁值 = 治疗量 * 0.5）
        Taunt,       // 嘲讽（立即拉满仇恨）
        Skill,       // 特殊技能（固定威胁值）
        Movement,    // 移动进入仇恨范围
    }
}
```

## 三、仇恨管理组件

```csharp
/// <summary>
/// AI 仇恨管理器（挂载在怪物/Boss上）
/// </summary>
public class ThreatManager : MonoBehaviour
{
    [Header("仇恨配置")]
    [SerializeField] private float threatDecayRate = 0.02f;   // 仇恨每秒衰减率
    [SerializeField] private float threatDecayDelay = 5f;     // 多久不互动后开始衰减
    [SerializeField] private float aggroRange = 15f;          // 初始仇恨范围
    [SerializeField] private float leashRange = 40f;          // 追击极限范围（超出则重置）
    [SerializeField] private float threatSwitchThreshold = 1.1f; // 超过当前目标仇恨多少倍才切换
    
    private Dictionary<string, ThreatEntry> threatTable 
        = new Dictionary<string, ThreatEntry>();
    
    private ThreatEntry currentTarget;  // 当前主要攻击目标
    
    // 仇恨修改器
    private List<ThreatModifier> globalModifiers = new List<ThreatModifier>();
    
    // 事件
    public event Action<ThreatEntry, ThreatEntry> OnTargetChanged; // oldTarget, newTarget
    public event Action<string> OnTargetLost; // targetId

    void Update()
    {
        float dt = Time.deltaTime;
        
        // 更新仇恨衰减
        DecayThreat(dt);
        
        // 清除过期的修改器
        CleanExpiredModifiers();
        
        // 重新评估目标
        EvaluateCurrentTarget();
    }

    /// <summary>
    /// 添加仇恨（核心接口）
    /// </summary>
    public void AddThreat(GameObject source, float amount, ThreatEvent.ThreatSource sourceType)
    {
        if (source == null) return;
        
        string id = GetTargetId(source);
        
        if (!threatTable.ContainsKey(id))
        {
            threatTable[id] = new ThreatEntry
            {
                TargetId = id,
                Target = source,
                ThreatValue = 0
            };
        }
        
        var entry = threatTable[id];
        entry.LastInteractionTime = Time.time;
        
        // 计算最终仇恨值（考虑修改器）
        float finalThreat = amount * GetThreatMultiplier(id, sourceType);
        entry.ThreatValue += finalThreat;
        
        // 触发目标重评估
        EvaluateCurrentTarget();
    }

    float GetThreatMultiplier(string targetId, ThreatEvent.ThreatSource sourceType)
    {
        float multiplier = 1f;
        
        // 全局修改器
        foreach (var mod in globalModifiers)
        {
            if (mod.Type == ThreatModifier.ThreatModifierType.Multiplier)
                multiplier *= mod.Value;
        }
        
        // 来源类型修改器
        multiplier *= sourceType switch
        {
            ThreatEvent.ThreatSource.Healing => 0.5f,   // 治疗产生50%仇恨
            ThreatEvent.ThreatSource.Damage  => 1f,
            ThreatEvent.ThreatSource.Taunt   => 100f,   // 嘲讽产生极高仇恨
            ThreatEvent.ThreatSource.Skill   => 1.5f,
            _ => 1f
        };
        
        return multiplier;
    }

    /// <summary>
    /// 嘲讽（立即成为首要目标）
    /// </summary>
    public void Taunt(GameObject source, float duration = 3f)
    {
        string id = GetTargetId(source);
        
        if (!threatTable.ContainsKey(id))
        {
            threatTable[id] = new ThreatEntry
            {
                TargetId = id,
                Target = source,
                ThreatValue = 0
            };
        }
        
        var entry = threatTable[id];
        
        // 设置嘲讽状态
        entry.IsTaunted = true;
        entry.TauntExpireTime = Time.time + duration;
        
        // 将嘲讽目标的仇恨值提升到最高 + 一定量
        float maxThreat = threatTable.Values.Max(e => e.ThreatValue);
        entry.ThreatValue = maxThreat * 1.1f + 500f; // 确保成为首要目标
        
        ForceSetTarget(entry);
        
        Debug.Log($"[Threat] {source.name} taunted for {duration}s");
    }

    void DecayThreat(float dt)
    {
        float now = Time.time;
        var toRemove = new List<string>();
        
        foreach (var kvp in threatTable)
        {
            var entry = kvp.Value;
            
            // 嘲讽检查（到期则清除）
            if (entry.IsTaunted && now > entry.TauntExpireTime)
            {
                entry.IsTaunted = false;
            }
            
            // 超出一段时间没有互动才开始衰减
            if (now - entry.LastInteractionTime > threatDecayDelay)
            {
                entry.ThreatValue *= (1f - threatDecayRate * dt);
                
                // 仇恨值归零，移除记录
                if (entry.ThreatValue < 1f)
                    toRemove.Add(kvp.Key);
            }
            
            // 检查目标是否超出追击范围
            if (entry.Target != null && 
                Vector3.Distance(transform.position, entry.Target.transform.position) > leashRange)
            {
                toRemove.Add(kvp.Key);
                OnTargetLost?.Invoke(entry.TargetId);
            }
        }
        
        foreach (var key in toRemove)
            threatTable.Remove(key);
    }

    void EvaluateCurrentTarget()
    {
        // 找到仇恨值最高的目标
        ThreatEntry highestThreat = null;
        float highest = 0;
        
        foreach (var entry in threatTable.Values)
        {
            if (entry.Target == null || !entry.Target.activeInHierarchy) continue;
            
            // 嘲讽目标优先
            if (entry.IsTaunted && Time.time < entry.TauntExpireTime)
            {
                highestThreat = entry;
                break;
            }
            
            if (entry.ThreatValue > highest)
            {
                highest = entry.ThreatValue;
                highestThreat = entry;
            }
        }
        
        if (highestThreat == null) return;
        
        // 只有当新目标仇恨明显更高时才切换（防止频繁切换）
        bool shouldSwitch = currentTarget == null ||
            highestThreat.TargetId != currentTarget.TargetId &&
            highestThreat.ThreatValue > currentTarget.ThreatValue * threatSwitchThreshold;
        
        if (shouldSwitch)
        {
            var oldTarget = currentTarget;
            currentTarget = highestThreat;
            OnTargetChanged?.Invoke(oldTarget, currentTarget);
        }
    }

    void ForceSetTarget(ThreatEntry entry)
    {
        var old = currentTarget;
        currentTarget = entry;
        if (old != entry) OnTargetChanged?.Invoke(old, entry);
    }

    void CleanExpiredModifiers()
    {
        globalModifiers.RemoveAll(m => Time.time > m.ExpireTime);
    }

    public void AddGlobalModifier(ThreatModifier mod) => globalModifiers.Add(mod);

    public ThreatEntry CurrentTarget => currentTarget;
    
    public float GetThreatOf(string targetId) =>
        threatTable.TryGetValue(targetId, out var e) ? e.ThreatValue : 0;
    
    public List<ThreatEntry> GetThreatTable() => 
        threatTable.Values.OrderByDescending(e => e.ThreatValue).ToList();
    
    string GetTargetId(GameObject obj) => obj.GetInstanceID().ToString();
}
```

## 四、Boss 阶段切换 AI

```csharp
/// <summary>
/// Boss AI 状态机（带仇恨感知）
/// </summary>
public class BossAI : MonoBehaviour
{
    [Header("Boss 配置")]
    [SerializeField] private float maxHp = 500000f;
    [SerializeField] private BossPhaseConfig[] phases;

    private float currentHp;
    private ThreatManager threatManager;
    private BossPhaseConfig currentPhase;
    private int currentPhaseIndex = 0;
    private BossState state = BossState.Idle;

    [SerializeField]
    [System.Serializable]
    public class BossPhaseConfig
    {
        public string PhaseName;
        [Range(0, 1)] public float HpThreshold;    // 血量百分比触发点
        public float[] AbilityWeights;              // 各技能权重
        public string[] PhaseTransitionSkills;      // 阶段转换时释放的技能
        public float AttackSpeedMultiplier = 1f;
        public float DamageMultiplier = 1f;
        public bool BecomesEnraged = false;         // 狂暴状态
    }

    public enum BossState
    {
        Idle,           // 未激活
        Engaging,       // 战斗中
        PhaseTransition, // 阶段过渡（无敌）
        Enraged,        // 狂暴
        Dead
    }

    void Awake()
    {
        currentHp = maxHp;
        threatManager = GetComponent<ThreatManager>();
        threatManager.OnTargetChanged += OnTargetChanged;
    }

    void Update()
    {
        if (state != BossState.Engaging && state != BossState.Enraged) return;
        
        var target = threatManager.CurrentTarget;
        if (target?.Target == null)
        {
            state = BossState.Idle;
            return;
        }
        
        // 面向目标
        Vector3 dir = (target.Target.transform.position - transform.position).normalized;
        dir.y = 0;
        if (dir != Vector3.zero)
            transform.rotation = Quaternion.LookRotation(dir);
        
        // 检查阶段切换
        CheckPhaseTransition();
        
        // 选择并执行技能
        SelectAndUseAbility(target.Target);
    }

    public void TakeDamage(float damage, GameObject source)
    {
        currentHp -= damage;
        
        // 添加仇恨
        threatManager.AddThreat(source, damage, ThreatEvent.ThreatSource.Damage);
        
        if (currentHp <= 0)
        {
            Die();
        }
    }

    void CheckPhaseTransition()
    {
        if (currentPhaseIndex >= phases.Length - 1) return;
        
        float hpPercent = currentHp / maxHp;
        var nextPhase = phases[currentPhaseIndex + 1];
        
        if (hpPercent <= nextPhase.HpThreshold)
        {
            StartCoroutine(ExecutePhaseTransition(nextPhase));
        }
    }

    System.Collections.IEnumerator ExecutePhaseTransition(BossPhaseConfig newPhase)
    {
        state = BossState.PhaseTransition;
        currentPhaseIndex++;
        currentPhase = newPhase;
        
        // 播放阶段转换动画
        Debug.Log($"[Boss] Entering phase: {newPhase.PhaseName}");
        
        // 释放阶段转换技能
        foreach (var skillId in newPhase.PhaseTransitionSkills ?? Array.Empty<string>())
        {
            yield return UseSkill(skillId);
            yield return new WaitForSeconds(1f);
        }
        
        state = newPhase.BecomesEnraged ? BossState.Enraged : BossState.Engaging;
    }

    System.Collections.IEnumerator UseSkill(string skillId)
    {
        Debug.Log($"[Boss] Using skill: {skillId}");
        yield return new WaitForSeconds(1.5f);
    }

    void SelectAndUseAbility(GameObject target)
    {
        // 基于当前阶段的技能权重随机选择技能
        if (currentPhase == null) return;
        
        // 简化：每隔一段时间使用随机技能
    }

    void OnTargetChanged(ThreatEntry oldTarget, ThreatEntry newTarget)
    {
        Debug.Log($"[Boss] Target switched: {oldTarget?.TargetId} -> {newTarget?.TargetId}");
    }

    void Die()
    {
        state = BossState.Dead;
        // 触发死亡事件、掉落奖励、成就检查等
        EventBus<EnemyKilledEvent>.Publish(new EnemyKilledEvent
        {
            EnemyId = gameObject.name,
            Type = EnemyType.Boss,
            Position = transform.position
        });
    }
}

public enum EnemyType { Normal, Elite, Boss }
```

```csharp
/// <summary>
/// 团队仇恨排行榜 UI
/// </summary>
public class ThreatMeterUI : MonoBehaviour
{
    [SerializeField] private Transform entryContainer;
    [SerializeField] private GameObject entryPrefab;
    [SerializeField] private float refreshInterval = 0.2f;
    
    private ThreatManager currentTargetThreat;
    private float refreshTimer;
    
    void Update()
    {
        refreshTimer += Time.deltaTime;
        if (refreshTimer >= refreshInterval && currentTargetThreat != null)
        {
            refreshTimer = 0;
            RefreshDisplay();
        }
    }
    
    void RefreshDisplay()
    {
        var table = currentTargetThreat.GetThreatTable();
        float maxThreat = table.Count > 0 ? table[0].ThreatValue : 1f;
        
        // 清空旧条目
        foreach (Transform child in entryContainer)
            Destroy(child.gameObject);
        
        foreach (var entry in table.Take(8)) // 最多显示前8名
        {
            var go = Instantiate(entryPrefab, entryContainer);
            var nameText = go.transform.Find("Name")?.GetComponent<UnityEngine.UI.Text>();
            var bar = go.transform.Find("Bar")?.GetComponent<UnityEngine.UI.Image>();
            
            if (nameText != null) 
                nameText.text = entry.Target?.name ?? "Unknown";
            if (bar != null) 
                bar.fillAmount = entry.ThreatValue / maxThreat;
        }
    }
}
```

## 六、设计总结

| 机制 | 设计意图 | 实现要点 |
|------|----------|----------|
| 仇恨累积 | 鼓励持续输出 | 伤害/治疗按比例产生仇恨 |
| 仇恨衰减 | 防止长久遗留仇恨 | 离开战斗后缓慢衰减 |
| 嘲讽技能 | 坦克的核心工具 | 强制锁定，短时间无法被超越 |
| 切换阈值 | 防止频繁换目标 | 需超过当前目标1.1倍才切换 |
| 治疗威胁 | 让治疗也有压力 | 治疗量*0.5产生仇恨 |
| Boss阶段AI | 提升战斗多样性 | 血量阈值触发阶段切换 |
