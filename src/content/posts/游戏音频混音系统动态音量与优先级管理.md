## 一、AudioMixer 层次架构

```
Master Mixer
├── BGM Group（背景音乐）
│   ├── ExposedParam: BGM_Volume
│   └── Send → Reverb / Echo
├── SFX Group（音效）
│   ├── Action SFX（战斗/交互音效）
│   │   └── ExposedParam: Action_Volume
│   ├── Ambient SFX（环境音效）
│   │   └── ExposedParam: Ambient_Volume
│   └── UI SFX（界面音效）
│       └── ExposedParam: UI_Volume
├── Voice Group（语音/对话）
│   └── ExposedParam: Voice_Volume
└── Master Volume（全局音量）
```

## 二、音频管理器

```csharp
using UnityEngine;
using UnityEngine.Audio;
using System.Collections;
using System.Collections.Generic;

/// <summary>
/// 游戏音频管理器
/// </summary>
public class AudioManager : MonoBehaviour
{
    private static AudioManager instance;
    public static AudioManager Instance => instance;

    [Header("Mixer 引用")]
    [SerializeField] private AudioMixer masterMixer;
    
    [Header("音频组参数名（Exposed Parameters）")]
    private const string PARAM_MASTER   = "Master_Volume";
    private const string PARAM_BGM      = "BGM_Volume";
    private const string PARAM_SFX      = "SFX_Volume";
    private const string PARAM_VOICE    = "Voice_Volume";
    private const string PARAM_UI       = "UI_Volume";
    
    [Header("BGM 淡入淡出")]
    [SerializeField] private float bgmFadeDuration = 1.5f;
    
    [Header("音频池")]
    [SerializeField] private int sfxPoolSize = 30;
    [SerializeField] private AudioMixerGroup sfxGroup;
    [SerializeField] private AudioMixerGroup bgmGroup;
    [SerializeField] private AudioMixerGroup voiceGroup;
    
    private AudioSource bgmSource;
    private AudioSource bgmSourceB;     // 用于交叉淡入淡出
    private bool usingSourceA = true;
    
    private Queue<AudioSource> sfxPool = new Queue<AudioSource>();
    private List<AudioSource> activeSfxSources = new List<AudioSource>();

    // 音量缓存（-1到1 对数刻度，转为dB）
    private float masterVolume = 1f;
    private float bgmVolume = 1f;
    private float sfxVolume = 1f;

    void Awake()
    {
        instance = this;
        InitializeAudioSources();
        LoadVolumeSettings();
    }

    void InitializeAudioSources()
    {
        // BGM 双源（交叉淡入淡出）
        bgmSource = CreateAudioSource("BGM_A", bgmGroup, true);
        bgmSourceB = CreateAudioSource("BGM_B", bgmGroup, true);
        
        // SFX 对象池
        for (int i = 0; i < sfxPoolSize; i++)
        {
            var source = CreateAudioSource($"SFX_{i}", sfxGroup, false);
            source.gameObject.SetActive(false);
            sfxPool.Enqueue(source);
        }
    }

    AudioSource CreateAudioSource(string name, AudioMixerGroup group, bool loop)
    {
        var go = new GameObject(name);
        go.transform.SetParent(transform);
        var source = go.AddComponent<AudioSource>();
        source.outputAudioMixerGroup = group;
        source.loop = loop;
        source.playOnAwake = false;
        return source;
    }

    // ============ BGM 控制 ============

    public void PlayBGM(AudioClip clip, bool crossfade = true)
    {
        if (clip == null) return;
        
        AudioSource current = usingSourceA ? bgmSource : bgmSourceB;
        AudioSource next = usingSourceA ? bgmSourceB : bgmSource;
        
        if (current.clip == clip && current.isPlaying) return;
        
        next.clip = clip;
        next.volume = 0f;
        next.Play();
        
        if (crossfade)
            StartCoroutine(CrossfadeBGM(current, next, bgmFadeDuration));
        else
        {
            current.Stop();
            next.volume = 1f;
        }
        
        usingSourceA = !usingSourceA;
    }

    IEnumerator CrossfadeBGM(AudioSource from, AudioSource to, float duration)
    {
        float timer = 0;
        float fromStartVol = from.volume;
        
        while (timer < duration)
        {
            timer += Time.unscaledDeltaTime;
            float t = timer / duration;
            from.volume = Mathf.Lerp(fromStartVol, 0f, t);
            to.volume = Mathf.Lerp(0f, 1f, t);
            yield return null;
        }
        
        from.Stop();
        from.clip = null;
        to.volume = 1f;
    }

    public void StopBGM(float fadeOut = 1f)
    {
        var current = usingSourceA ? bgmSource : bgmSourceB;
        StartCoroutine(FadeOut(current, fadeOut));
    }

    IEnumerator FadeOut(AudioSource source, float duration)
    {
        float startVol = source.volume;
        float timer = 0;
        
        while (timer < duration)
        {
            timer += Time.unscaledDeltaTime;
            source.volume = Mathf.Lerp(startVol, 0f, timer / duration);
            yield return null;
        }
        
        source.Stop();
    }

    // ============ SFX 控制 ============

    public AudioSource PlaySFX(AudioClip clip, Vector3? position = null, 
        float volume = 1f, float pitch = 1f, float spatialBlend = 0f)
    {
        if (clip == null) return null;
        
        AudioSource source = GetSFXSource();
        if (source == null) return null;
        
        source.gameObject.SetActive(true);
        source.clip = clip;
        source.volume = volume;
        source.pitch = pitch + UnityEngine.Random.Range(-0.05f, 0.05f); // 轻微随机音高
        source.spatialBlend = spatialBlend;
        source.loop = false;
        
        if (position.HasValue)
            source.transform.position = position.Value;
        
        source.Play();
        activeSfxSources.Add(source);
        
        // 播放完自动归还池
        StartCoroutine(ReturnToPool(source, clip.length / Mathf.Abs(pitch)));
        
        return source;
    }

    public AudioSource PlaySFX3D(AudioClip clip, Vector3 position, 
        float minDistance = 1f, float maxDistance = 20f, float volume = 1f)
    {
        var source = PlaySFX(clip, position, volume, 1f, 1f);
        if (source != null)
        {
            source.minDistance = minDistance;
            source.maxDistance = maxDistance;
            source.rolloffMode = AudioRolloffMode.Logarithmic;
        }
        return source;
    }

    IEnumerator ReturnToPool(AudioSource source, float delay)
    {
        yield return new WaitForSeconds(delay);
        source.Stop();
        source.gameObject.SetActive(false);
        source.transform.position = transform.position;
        activeSfxSources.Remove(source);
        sfxPool.Enqueue(source);
    }

    AudioSource GetSFXSource()
    {
        if (sfxPool.Count > 0)
            return sfxPool.Dequeue();
        
        // 池用尽：抢占最旧的活跃源
        if (activeSfxSources.Count > 0)
        {
            var oldest = activeSfxSources[0];
            activeSfxSources.RemoveAt(0);
            oldest.Stop();
            return oldest;
        }
        
        Debug.LogWarning("[Audio] SFX pool exhausted!");
        return null;
    }

    // ============ 音量控制 ============

    public void SetMasterVolume(float volume)
    {
        masterVolume = Mathf.Clamp01(volume);
        SetMixerVolume(PARAM_MASTER, masterVolume);
        PlayerPrefs.SetFloat("vol_master", masterVolume);
    }

    public void SetBGMVolume(float volume)
    {
        bgmVolume = Mathf.Clamp01(volume);
        SetMixerVolume(PARAM_BGM, bgmVolume);
        PlayerPrefs.SetFloat("vol_bgm", bgmVolume);
    }

    public void SetSFXVolume(float volume)
    {
        sfxVolume = Mathf.Clamp01(volume);
        SetMixerVolume(PARAM_SFX, sfxVolume);
        PlayerPrefs.SetFloat("vol_sfx", sfxVolume);
    }

    void SetMixerVolume(string param, float linearVolume)
    {
        // 线性音量转dB（-80dB到0dB）
        float dB = linearVolume > 0.001f ? 
            Mathf.Log10(linearVolume) * 20f : -80f;
        masterMixer.SetFloat(param, dB);
    }

    void LoadVolumeSettings()
    {
        SetMasterVolume(PlayerPrefs.GetFloat("vol_master", 1f));
        SetBGMVolume(PlayerPrefs.GetFloat("vol_bgm", 0.8f));
        SetSFXVolume(PlayerPrefs.GetFloat("vol_sfx", 1f));
    }

    // ============ 动态混音 ============

    /// <summary>
    /// 战斗时压低背景音乐（Ducking效果）
    /// </summary>
    public void SetCombatMode(bool inCombat)
    {
        float targetBGM = inCombat ? bgmVolume * 0.4f : bgmVolume;
        SetMixerVolume(PARAM_BGM, targetBGM);
        
        // 也可以通过 AudioMixer Snapshot 实现
        // string snapshotName = inCombat ? "Combat" : "Exploration";
        // masterMixer.FindSnapshot(snapshotName)?.TransitionTo(0.5f);
    }

    /// <summary>
    /// 使用 Snapshot 过渡音频状态
    /// </summary>
    public void TransitionToSnapshot(string snapshotName, float transitionTime = 0.5f)
    {
        var snapshot = masterMixer.FindSnapshot(snapshotName);
        snapshot?.TransitionTo(transitionTime);
    }
}
```

## 三、AudioMixer Snapshot 状态管理

```csharp
/// <summary>
/// 音频状态管理（Snapshot 切换）
/// </summary>
public class AudioStateManager : MonoBehaviour
{
    [Header("Snapshot 配置")]
    [SerializeField] private AudioMixer mixer;
    
    public enum AudioState
    {
        Normal,         // 正常探索
        Combat,         // 战斗中（BGM变快/音效更强）
        Underwater,     // 水下（低通滤波）
        Menu,           // 菜单（SFX降低）
        Paused,         // 暂停（全部降低）
        Cinematic       // 过场动画（关闭SFX）
    }

    private AudioState currentState = AudioState.Normal;
    
    private readonly Dictionary<AudioState, string> snapshotNames = 
        new Dictionary<AudioState, string>
        {
            { AudioState.Normal,     "Normal"     },
            { AudioState.Combat,     "Combat"     },
            { AudioState.Underwater, "Underwater" },
            { AudioState.Menu,       "Menu"       },
            { AudioState.Paused,     "Paused"     },
            { AudioState.Cinematic,  "Cinematic"  },
        };

    public void SetAudioState(AudioState newState, float transitionTime = 0.5f)
    {
        if (newState == currentState) return;
        
        currentState = newState;
        
        if (snapshotNames.TryGetValue(newState, out string name))
        {
            var snapshot = mixer.FindSnapshot(name);
            snapshot?.TransitionTo(transitionTime);
        }
        
        Debug.Log($"[Audio] State → {newState}");
    }
}
```

## 四、移动端音频优化

| 优化项 | 推荐设置 | 说明 |
|--------|----------|------|
| AudioClip 格式 | Vorbis（质量 70%） | iOS用AAC |
| 加载类型 | Compressed In Memory（短音效） | BGM用Streaming |
| 最大并发声音数 | 24（移动端）/ 48（PC）| Unity音频预算 |
| 3D音效距离衰减 | 对数（Logarithmic Rolloff）| 更自然 |
| 预加载音频 | 常用SFX提前加载 | 避免首次播放延迟 |
| 采样率 | 22050Hz（移动端SFX）| 人耳难辨区别 |

**关键原则：**
1. 线性音量值转dB公式：`dB = 20 * log10(linearValue)`
2. 使用 Snapshot 管理状态，比手动改参数更优雅
3. SFX 对象池规模按实际峰值并发量设定（不要无限扩展）
4. 音频是情感驱动系统，在预算范围内尽量保持高质量
