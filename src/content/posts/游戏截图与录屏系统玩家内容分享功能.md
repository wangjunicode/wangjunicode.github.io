## 一、截图系统

```csharp
using System;
using System.Collections;
using System.IO;
using UnityEngine;
using UnityEngine.UI;

/// <summary>
/// 游戏截图管理器
/// </summary>
public class ScreenshotManager : MonoBehaviour
{
    private static ScreenshotManager instance;
    public static ScreenshotManager Instance => instance;

    [Header("截图配置")]
    [SerializeField] private int superSamplingFactor = 2;   // 超采样（2=2K分辨率）
    [SerializeField] private string screenshotFolder = "Screenshots";
    
    [Header("水印配置")]
    [SerializeField] private Sprite watermarkSprite;        // 游戏Logo水印
    [SerializeField] private Vector2 watermarkAnchor = new Vector2(1, 0); // 右下角
    [SerializeField, Range(0, 1)] private float watermarkOpacity = 0.6f;

    private string SaveFolder => 
        Path.Combine(Application.persistentDataPath, screenshotFolder);

    void Awake() 
    { 
        instance = this;
        if (!Directory.Exists(SaveFolder))
            Directory.CreateDirectory(SaveFolder);
    }

    /// <summary>
    /// 截取游戏画面（含UI）
    /// </summary>
    public void TakeScreenshot(bool includeUI = true, Action<Texture2D> onComplete = null)
    {
        StartCoroutine(CaptureScreenshot(includeUI, onComplete));
    }

    IEnumerator CaptureScreenshot(bool includeUI, Action<Texture2D> onComplete)
    {
        yield return new WaitForEndOfFrame();
        
        Texture2D screenshot;
        
        if (includeUI)
        {
            // 完整截图（含UI）
            screenshot = ScreenCapture.CaptureScreenshotAsTexture(superSamplingFactor);
        }
        else
        {
            // 仅游戏画面（不含UI）
            screenshot = CaptureGameOnly();
        }
        
        // 叠加水印
        if (watermarkSprite != null)
            AddWatermark(screenshot, watermarkSprite.texture);
        
        onComplete?.Invoke(screenshot);
    }

    Texture2D CaptureGameOnly()
    {
        // 临时禁用 UI Canvas
        var canvases = FindObjectsOfType<Canvas>();
        var wasActive = new bool[canvases.Length];
        
        for (int i = 0; i < canvases.Length; i++)
        {
            wasActive[i] = canvases[i].enabled;
            if (canvases[i].renderMode == RenderMode.ScreenSpaceOverlay)
                canvases[i].enabled = false;
        }
        
        // 截图
        var screenshot = ScreenCapture.CaptureScreenshotAsTexture(superSamplingFactor);
        
        // 恢复 Canvas
        for (int i = 0; i < canvases.Length; i++)
            canvases[i].enabled = wasActive[i];
        
        return screenshot;
    }

    void AddWatermark(Texture2D target, Texture2D watermark)
    {
        int w = watermark.width / 4;  // 水印大小为截图的1/4宽
        int h = Mathf.RoundToInt(w * (float)watermark.height / watermark.width);
        
        // 计算位置（右下角，带边距）
        int margin = 20;
        int x = target.width - w - margin;
        int y = margin;
        
        // 混合水印像素
        Color[] watermarkPixels = watermark.GetPixels();
        
        for (int py = 0; py < h; py++)
        {
            for (int px = 0; px < w; px++)
            {
                int srcX = Mathf.RoundToInt((float)px / w * watermark.width);
                int srcY = Mathf.RoundToInt((float)py / h * watermark.height);
                
                Color watermarkPixel = watermarkPixels[srcY * watermark.width + srcX];
                watermarkPixel.a *= watermarkOpacity;
                
                Color targetPixel = target.GetPixel(x + px, y + py);
                Color blended = Color.Lerp(targetPixel, watermarkPixel, watermarkPixel.a);
                target.SetPixel(x + px, y + py, blended);
            }
        }
        
        target.Apply();
    }

    /// <summary>
    /// 保存截图到本地
    /// </summary>
    public string SaveScreenshot(Texture2D screenshot, string filename = null)
    {
        if (filename == null)
            filename = $"Screenshot_{DateTime.Now:yyyyMMdd_HHmmss}.png";
        
        string fullPath = Path.Combine(SaveFolder, filename);
        byte[] pngData = screenshot.EncodeToPNG();
        File.WriteAllBytes(fullPath, pngData);
        
        // Android 需要通知媒体库扫描新文件
        #if UNITY_ANDROID
        AndroidJavaClass mediaScanner = new AndroidJavaClass("android.media.MediaScannerConnection");
        AndroidJavaClass unityPlayer = new AndroidJavaClass("com.unity3d.player.UnityPlayer");
        AndroidJavaObject activity = unityPlayer.GetStatic<AndroidJavaObject>("currentActivity");
        mediaScanner.CallStatic("scanFile", activity, new string[] { fullPath }, null, null);
        #endif
        
        Debug.Log($"[Screenshot] Saved: {fullPath}");
        return fullPath;
    }
}
```

## 二、社交分享功能

```csharp
/// <summary>
/// 社交分享管理器
/// </summary>
public class SocialShareManager : MonoBehaviour
{
    private static SocialShareManager instance;
    public static SocialShareManager Instance => instance;

    void Awake() { instance = this; }

    /// <summary>
    /// 分享截图（调用原生系统分享）
    /// </summary>
    public void ShareScreenshot(string imagePath, string shareText = "")
    {
        #if UNITY_IOS
        ShareOniOS(imagePath, shareText);
        #elif UNITY_ANDROID
        ShareOnAndroid(imagePath, shareText);
        #else
        Debug.Log($"[Share] Simulating share: {imagePath}");
        #endif
    }

    /// <summary>
    /// 先截图再分享
    /// </summary>
    public void TakeAndShare(bool includeUI = true, string shareText = "")
    {
        ScreenshotManager.Instance.TakeScreenshot(includeUI, screenshot =>
        {
            string path = ScreenshotManager.Instance.SaveScreenshot(screenshot);
            ShareScreenshot(path, shareText);
            
            // 释放内存
            Destroy(screenshot);
        });
    }

    #if UNITY_IOS
    [System.Runtime.InteropServices.DllImport("__Internal")]
    static extern void _ShareImage(string imagePath, string text);
    
    void ShareOniOS(string imagePath, string text)
    {
        // 调用 Objective-C 插件的 UIActivityViewController
        _ShareImage(imagePath, text);
    }
    #endif

    #if UNITY_ANDROID
    void ShareOnAndroid(string imagePath, string text)
    {
        // 使用 Android Intent 分享
        AndroidJavaClass intentClass = new AndroidJavaClass("android.content.Intent");
        AndroidJavaObject intent = new AndroidJavaObject("android.content.Intent");
        
        intent.Call<AndroidJavaObject>("setAction", 
            intentClass.GetStatic<string>("ACTION_SEND"));
        intent.Call<AndroidJavaObject>("setType", "image/*");
        
        // 需要先通过 FileProvider 获取 URI（Android 7+）
        AndroidJavaClass fileProviderClass = new AndroidJavaClass(
            "androidx.core.content.FileProvider");
        AndroidJavaClass unityPlayer = new AndroidJavaClass(
            "com.unity3d.player.UnityPlayer");
        AndroidJavaObject activity = unityPlayer.GetStatic<AndroidJavaObject>("currentActivity");
        
        AndroidJavaObject file = new AndroidJavaObject("java.io.File", imagePath);
        AndroidJavaObject uri = fileProviderClass.CallStatic<AndroidJavaObject>(
            "getUriForFile", activity, 
            Application.identifier + ".fileprovider", file);
        
        intent.Call<AndroidJavaObject>("putExtra", 
            intentClass.GetStatic<string>("EXTRA_STREAM"), uri);
        
        if (!string.IsNullOrEmpty(text))
            intent.Call<AndroidJavaObject>("putExtra", 
                intentClass.GetStatic<string>("EXTRA_TEXT"), text);
        
        intent.Call<AndroidJavaObject>("addFlags", 1); // FLAG_GRANT_READ_URI_PERMISSION
        
        AndroidJavaObject chooser = intentClass.CallStatic<AndroidJavaObject>(
            "createChooser", intent, "分享游戏截图");
        
        activity.Call("startActivity", chooser);
    }
    #endif

    /// <summary>
    /// 生成带玩家信息的分享文案
    /// </summary>
    public string GenerateShareText(string template = null)
    {
        var player = PlayerDataService.GetLocalPlayerData();
        string defaultTemplate = $"我在《游戏名称》达到了 Lv.{player?.Level} 级！快来一起玩吧！";
        return template ?? defaultTemplate;
    }
}
```

## 三、截图相框系统

```csharp
/// <summary>
/// 截图相框合成（节日主题/角色海报等）
/// </summary>
public class ScreenshotFrameComposer : MonoBehaviour
{
    [System.Serializable]
    public class FrameTemplate
    {
        public string FrameId;
        public string DisplayName;
        public Sprite FrameImage;       // 相框图片（PNG，中间透明）
        public bool SeasonalOnly;       // 是否限时活动专属
        public long AvailableUntil;     // 有效期
    }

    [SerializeField] private FrameTemplate[] frames;
    [SerializeField] private RawImage previewImage;

    public Texture2D ComposeWithFrame(Texture2D screenshot, string frameId)
    {
        var frame = System.Array.Find(frames, f => f.FrameId == frameId);
        if (frame?.FrameImage == null) return screenshot;
        
        // 创建合成纹理
        int width = screenshot.width;
        int height = screenshot.height;
        
        Texture2D result = new Texture2D(width, height, TextureFormat.RGBA32, false);
        
        // 1. 填充截图内容
        result.SetPixels(screenshot.GetPixels());
        
        // 2. 叠加相框（PNG透明度混合）
        Texture2D frameTexture = ResizeTexture(frame.FrameImage.texture, width, height);
        BlendTextures(result, frameTexture);
        
        result.Apply();
        Destroy(frameTexture);
        
        return result;
    }

    Texture2D ResizeTexture(Texture2D source, int width, int height)
    {
        RenderTexture rt = RenderTexture.GetTemporary(width, height);
        Graphics.Blit(source, rt);
        
        RenderTexture prev = RenderTexture.active;
        RenderTexture.active = rt;
        
        Texture2D result = new Texture2D(width, height);
        result.ReadPixels(new Rect(0, 0, width, height), 0, 0);
        result.Apply();
        
        RenderTexture.active = prev;
        RenderTexture.ReleaseTemporary(rt);
        
        return result;
    }

    void BlendTextures(Texture2D base_, Texture2D overlay)
    {
        Color[] basePixels = base_.GetPixels();
        Color[] overlayPixels = overlay.GetPixels();
        
        for (int i = 0; i < basePixels.Length; i++)
        {
            Color src = overlayPixels[i];
            Color dst = basePixels[i];
            
            // Alpha 混合
            float alpha = src.a;
            basePixels[i] = new Color(
                src.r * alpha + dst.r * (1 - alpha),
                src.g * alpha + dst.g * (1 - alpha),
                src.b * alpha + dst.b * (1 - alpha),
                1f);
        }
        
        base_.SetPixels(basePixels);
    }
}
```

## 四、隐私合规注意事项

| 平台 | 权限要求 | 说明 |
|------|----------|------|
| Android | WRITE_EXTERNAL_STORAGE | Android 9以下需要 |
| Android 10+ | 无需存储权限 | 使用 MediaStore API |
| iOS | Photo Library 权限 | 保存到相册需要 |
| 所有平台 | 用户数据说明 | 不允许截取其他APP内容 |

**最佳实践：**
1. 截图中不要包含玩家真实姓名（只显示游戏昵称）
2. 提供选项让玩家决定是否分享（不强制）
3. 水印添加游戏名而非个人信息
4. 分享文案让玩家预览后再发送（不自动发布）
