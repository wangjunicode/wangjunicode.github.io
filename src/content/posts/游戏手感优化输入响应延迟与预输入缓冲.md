## 一、手感的工程本质

```
输入延迟链路（帧级分析）：

物理按键按下
    ↓ USB/蓝牙采集延迟（1-4ms）
    ↓ 操作系统输入处理（约 1 帧）
    ↓ Unity 输入轮询（Update 帧头）
    ↓ 游戏逻辑处理（本帧）
    ↓ 渲染提交（本帧末尾）
    ↓ GPU 渲染（约 1-2 帧）
    ↓ 显示器扫描（1/60 秒）
    ↓ 显示器响应时间（1-5ms）
物理像素变化

总延迟：约 3-5 帧 = 50-83ms（60fps）

手感差的游戏通常在「游戏逻辑处理」环节额外增加了延迟
（如：攻击动画的前摇太长，玩家按下攻击后要等很久才有反馈）
```

## 二、预输入缓冲系统

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 预输入缓冲（Input Buffer）
/// 在特定时间窗口内预先接收输入，等待合适时机执行
/// 典型用例：玩家在攻击动画未结束时提前按了下次攻击
/// </summary>
public class InputBuffer : MonoBehaviour
{
    [Header("缓冲配置")]
    [SerializeField] private float bufferWindow = 0.15f;   // 预输入有效窗口（秒）
    [SerializeField] private int maxBufferSize = 3;        // 最多缓存几个输入

    private struct BufferedInput
    {
        public string Action;
        public float Timestamp;
    }

    private Queue<BufferedInput> buffer = new Queue<BufferedInput>();

    /// <summary>
    /// 接收输入（即使当前不能立刻响应，也缓冲起来）
    /// </summary>
    public void BufferInput(string action)
    {
        // 超出缓冲大小，丢弃最旧的
        while (buffer.Count >= maxBufferSize)
            buffer.Dequeue();
        
        buffer.Enqueue(new BufferedInput
        {
            Action = action,
            Timestamp = Time.time
        });
        
        Debug.Log($"[InputBuffer] Buffered: {action}");
    }

    /// <summary>
    /// 查询是否有有效的缓冲输入（并消耗它）
    /// </summary>
    public bool ConsumeBuffered(string action)
    {
        // 清理过期输入
        while (buffer.Count > 0 && Time.time - buffer.Peek().Timestamp > bufferWindow)
            buffer.Dequeue();
        
        foreach (var buffered in buffer)
        {
            if (buffered.Action == action)
            {
                // 找到匹配输入，从缓冲区移除
                var tempQueue = new Queue<BufferedInput>(buffer);
                buffer.Clear();
                foreach (var item in tempQueue)
                {
                    if (item.Action == action && !HasConsumed(item)) continue;
                    buffer.Enqueue(item);
                }
                return true;
            }
        }
        
        return false;
    }

    private bool hasConsumedThisFrame;
    bool HasConsumed(BufferedInput input) => hasConsumedThisFrame;

    public bool HasBuffered(string action)
    {
        while (buffer.Count > 0 && Time.time - buffer.Peek().Timestamp > bufferWindow)
            buffer.Dequeue();
        
        foreach (var buffered in buffer)
            if (buffered.Action == action) return true;
        
        return false;
    }

    public void ClearAll() => buffer.Clear();
    public void ClearAction(string action)
    {
        var newQueue = new Queue<BufferedInput>();
        foreach (var item in buffer)
            if (item.Action != action) newQueue.Enqueue(item);
        buffer = newQueue;
    }
}

/// <summary>
/// 土狼时间（Coyote Time）和跳跃缓冲
/// 让玩家在走出平台边缘后仍可短暂跳跃，消除不公平感
/// </summary>
public class PlatformerJumpSystem : MonoBehaviour
{
    [Header("土狼时间")]
    [SerializeField] private float coyoteTime = 0.12f;        // 离地后多久内还能跳
    [SerializeField] private float jumpBufferTime = 0.15f;    // 提前多久按跳跃仍有效

    private float coyoteTimer;
    private float jumpBufferTimer;
    private bool isGrounded;
    private bool wasGrounded;
    private CharacterController cc;
    private float verticalVelocity;

    [SerializeField] private float jumpForce = 12f;

    void Awake() => cc = GetComponent<CharacterController>();

    void Update()
    {
        wasGrounded = isGrounded;
        isGrounded = cc.isGrounded;
        
        // 土狼时间计时
        if (wasGrounded && !isGrounded)
        {
            // 刚刚离地，开始土狼时间计时
            coyoteTimer = coyoteTime;
        }
        else if (coyoteTimer > 0)
        {
            coyoteTimer -= Time.deltaTime;
        }
        
        // 跳跃输入缓冲
        if (Input.GetButtonDown("Jump"))
        {
            jumpBufferTimer = jumpBufferTime;
        }
        else if (jumpBufferTimer > 0)
        {
            jumpBufferTimer -= Time.deltaTime;
        }
        
        // 判断是否应该跳跃
        bool canJump = isGrounded || coyoteTimer > 0;  // 实际在地面 OR 土狼时间内
        bool wantsJump = jumpBufferTimer > 0;           // 当前帧 OR 之前缓冲的跳跃输入
        
        if (canJump && wantsJump)
        {
            verticalVelocity = jumpForce;
            coyoteTimer = 0;    // 消耗土狼时间
            jumpBufferTimer = 0; // 消耗缓冲输入
        }
        
        // 重力
        if (!isGrounded)
            verticalVelocity -= 25f * Time.deltaTime;
        else if (verticalVelocity < 0)
            verticalVelocity = -2f;
        
        cc.Move(new Vector3(0, verticalVelocity * Time.deltaTime, 0));
    }
}
```

## 三、方向输入平滑

```csharp
/// <summary>
/// 方向输入平滑（消除原始摇杆抖动，但保留响应性）
/// </summary>
public class InputSmoother : MonoBehaviour
{
    [Header("平滑配置")]
    [SerializeField] private float inputSmoothing = 0.1f;       // 平滑时间（越小越灵敏）
    [SerializeField] private float deadzone = 0.15f;             // 死区（过滤微小输入）
    [SerializeField] private float accelerationCurve = 2f;       // 输入加速曲线指数
    
    private Vector2 rawInput;
    private Vector2 smoothInput;
    private Vector2 inputVelocity;

    public Vector2 GetSmoothedInput()
    {
        // 获取原始输入
        rawInput = new Vector2(
            Input.GetAxis("Horizontal"),
            Input.GetAxis("Vertical"));
        
        // 应用死区（圆形死区比方形更自然）
        float magnitude = rawInput.magnitude;
        if (magnitude < deadzone)
        {
            rawInput = Vector2.zero;
        }
        else
        {
            // 重映射到 0-1（死区外的范围）
            rawInput = rawInput.normalized * ((magnitude - deadzone) / (1f - deadzone));
        }
        
        // 应用输入曲线（让小幅度输入更精细，大幅度输入更有力）
        if (rawInput.magnitude > 0)
        {
            float curved = Mathf.Pow(rawInput.magnitude, accelerationCurve);
            rawInput = rawInput.normalized * curved;
        }
        
        // 平滑过渡
        smoothInput = Vector2.SmoothDamp(smoothInput, rawInput, 
            ref inputVelocity, inputSmoothing);
        
        return smoothInput;
    }
}
```

## 四、手感量化指标

| 指标 | 测量方法 | 目标值 |
|------|----------|--------|
| 输入响应延迟 | 高速摄像机测量按键到画面响应 | < 3帧（50ms@60fps）|
| 攻击前摇时间 | 按下攻击到角色产生判定的帧数 | ≤ 3帧（动作游戏）|
| 跳跃手感 | 到达最高点时间 / 下落速度比 | 上升比下落慢（自然感）|
| 连招输入窗口 | 能接受预输入的帧数 | 4-8帧（约67-133ms）|
| 方向响应延迟 | 输入方向到角色完全转向的帧数 | 3-5帧 |

**核心原则：手感是细节的总和，每一帧的延迟都有意义。优秀的手感往往来自微调，而非架构变化。**
