## 一、MaterialPropertyBlock 动态染色

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 角色染色系统（使用 MaterialPropertyBlock，不影响 Draw Call 合批）
/// </summary>
public class CharacterColorizer : MonoBehaviour
{
    [Header("材质颜色属性")]
    [SerializeField] private string primaryColorProperty = "_PrimaryColor";
    [SerializeField] private string secondaryColorProperty = "_SecondaryColor";
    [SerializeField] private string metalColorProperty = "_MetalColor";
    [SerializeField] private string emissionColorProperty = "_EmissionColor";

    private Renderer[] renderers;
    private MaterialPropertyBlock propertyBlock;
    
    // 颜色缓存
    private Color primaryColor = Color.white;
    private Color secondaryColor = Color.gray;
    private Color metalColor = new Color(0.8f, 0.8f, 0.9f);
    private Color emissionColor = Color.black;

    void Awake()
    {
        renderers = GetComponentsInChildren<Renderer>(true);
        propertyBlock = new MaterialPropertyBlock();
    }

    public void ApplyColorScheme(SkinColorScheme scheme)
    {
        primaryColor = scheme.PrimaryColor;
        secondaryColor = scheme.SecondaryColor;
        metalColor = scheme.MetalColor;
        emissionColor = scheme.EmissionColor;
        ApplyColors();
    }

    void ApplyColors()
    {
        foreach (var renderer in renderers)
        {
            if (renderer == null) continue;
            
            renderer.GetPropertyBlock(propertyBlock);
            
            propertyBlock.SetColor(primaryColorProperty, primaryColor);
            propertyBlock.SetColor(secondaryColorProperty, secondaryColor);
            propertyBlock.SetColor(metalColorProperty, metalColor);
            propertyBlock.SetColor(emissionColorProperty, emissionColor);
            
            renderer.SetPropertyBlock(propertyBlock);
        }
    }

    [ContextMenu("Test Gold Color Scheme")]
    void TestGoldScheme()
    {
        ApplyColorScheme(new SkinColorScheme
        {
            PrimaryColor = new Color(1f, 0.8f, 0.2f),
            SecondaryColor = new Color(0.8f, 0.6f, 0.1f),
            MetalColor = new Color(1f, 0.9f, 0.5f),
            EmissionColor = new Color(0.3f, 0.2f, 0f)
        });
    }
}

/// <summary>
/// 皮肤颜色方案
/// </summary>
[Serializable]
public class SkinColorScheme
{
    public Color PrimaryColor = Color.white;
    public Color SecondaryColor = Color.gray;
    public Color MetalColor = new Color(0.8f, 0.8f, 0.9f);
    public Color EmissionColor = Color.black;
    public Texture2D PatternTexture;    // 额外的图案纹理
}
```

## 二、皮肤配置数据

```csharp
/// <summary>
/// 皮肤数据（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "Skin", menuName = "Game/Skin Data")]
public class SkinData : ScriptableObject
{
    [Header("基础信息")]
    public string SkinId;
    public string DisplayName;
    [TextArea(1, 3)] public string Description;
    public Sprite ThumbnailImage;          // 皮肤缩略图
    public SkinRarity Rarity;
    
    [Header("获取方式")]
    public SkinSource Source;              // 商店/赛季/成就/活动
    public int PremiumCurrencyPrice;       // 高级货币价格（0=非商店获取）
    
    [Header("皮肤资源")]
    public GameObject PreviewPrefab;       // 预览用Prefab（展台旋转展示）
    public SkinColorScheme ColorScheme;    // 颜色方案
    public RuntimeAnimatorController AnimatorOverride; // 动画覆盖（可选）
    
    [Header("限时")]
    public bool IsLimited;                 // 是否限时皮肤
    public long LimitedUntil;              // 有效期（UTC时间戳）
    
    public bool IsObtainable => !IsLimited || 
        DateTimeOffset.UtcNow.ToUnixTimeSeconds() < LimitedUntil;
}

public enum SkinRarity { Common, Uncommon, Rare, Epic, Legendary }
public enum SkinSource { Shop, SeasonPass, Achievement, Event, Craft, Gift }
```

## 三、换装系统（SkinnedMesh 替换）

```csharp
/// <summary>
/// 换装系统（替换角色身体部位的 SkinnedMeshRenderer）
/// </summary>
public class CharacterOutfitManager : MonoBehaviour
{
    [System.Serializable]
    public class OutfitSlot
    {
        public string SlotName;             // "Head" / "Body" / "Legs" / "Weapon"
        public SkinnedMeshRenderer Current; // 当前装备的网格
        public Transform[] BoneMap;         // 骨骼映射（用于绑定）
    }

    [SerializeField] private OutfitSlot[] slots;
    [SerializeField] private Transform rootBone;          // 角色骨骼根节点
    [SerializeField] private Transform[] allBones;        // 所有骨骼（用于换装绑定）

    Dictionary<string, OutfitSlot> slotMap = new Dictionary<string, OutfitSlot>();

    void Awake()
    {
        foreach (var slot in slots)
            slotMap[slot.SlotName] = slot;
    }

    /// <summary>
    /// 替换指定部位的装备
    /// </summary>
    public void EquipItem(string slotName, SkinnedMeshRenderer newMesh)
    {
        if (!slotMap.TryGetValue(slotName, out var slot))
        {
            Debug.LogWarning($"[Outfit] Unknown slot: {slotName}");
            return;
        }
        
        // 禁用旧网格
        if (slot.Current != null)
            slot.Current.gameObject.SetActive(false);
        
        if (newMesh == null) return;
        
        // 实例化新网格
        var instance = Instantiate(newMesh, transform);
        
        // 重新绑定骨骼（确保使用角色自己的骨骼）
        RebindBones(instance);
        
        slot.Current = instance;
        instance.gameObject.SetActive(true);
    }

    void RebindBones(SkinnedMeshRenderer target)
    {
        // 根据骨骼名称匹配并重新绑定
        var newBones = new Transform[target.bones.Length];
        
        for (int i = 0; i < target.bones.Length; i++)
        {
            string boneName = target.bones[i].name;
            
            // 在角色骨骼中查找同名骨骼
            foreach (var bone in allBones)
            {
                if (bone != null && bone.name == boneName)
                {
                    newBones[i] = bone;
                    break;
                }
            }
        }
        
        target.bones = newBones;
        target.rootBone = rootBone;
    }
}
```

## 四、皮肤管理与预览

```csharp
/// <summary>
/// 皮肤管理器
/// </summary>
public class SkinManager : MonoBehaviour
{
    private static SkinManager instance;
    public static SkinManager Instance => instance;

    private Dictionary<string, SkinData> allSkins = new Dictionary<string, SkinData>();
    private HashSet<string> ownedSkins = new HashSet<string>();
    private string equippedSkinId;

    void Awake()
    {
        instance = this;
        LoadSkinData();
        LoadOwnedSkins();
    }

    void LoadSkinData()
    {
        // 从 Resources 或 Addressables 加载所有皮肤配置
        var skins = Resources.LoadAll<SkinData>("Skins");
        foreach (var skin in skins)
            allSkins[skin.SkinId] = skin;
    }

    void LoadOwnedSkins()
    {
        // 从本地/服务端加载已拥有的皮肤
        string json = PlayerPrefs.GetString("owned_skins", "");
        // 解析并填充 ownedSkins
    }

    public bool OwnsSkin(string skinId) => ownedSkins.Contains(skinId);

    public void UnlockSkin(string skinId)
    {
        ownedSkins.Add(skinId);
        // 保存到服务端
    }

    public void EquipSkin(string skinId)
    {
        if (!OwnsSkin(skinId))
        {
            Debug.LogWarning($"[Skin] Don't own skin: {skinId}");
            return;
        }
        
        equippedSkinId = skinId;
        PlayerPrefs.SetString("equipped_skin", skinId);
        
        // 应用皮肤到当前角色
        ApplySkinToPlayer(skinId);
    }

    void ApplySkinToPlayer(string skinId)
    {
        if (!allSkins.TryGetValue(skinId, out var skin)) return;
        
        var player = FindObjectOfType<PlayerController>();
        if (player == null) return;
        
        var colorizer = player.GetComponent<CharacterColorizer>();
        colorizer?.ApplyColorScheme(skin.ColorScheme);
        
        // 如果有动画覆盖
        if (skin.AnimatorOverride != null)
        {
            var animator = player.GetComponent<Animator>();
            if (animator != null)
                animator.runtimeAnimatorController = skin.AnimatorOverride;
        }
    }

    public List<SkinData> GetOwnedSkins()
    {
        var result = new List<SkinData>();
        foreach (var skinId in ownedSkins)
        {
            if (allSkins.TryGetValue(skinId, out var skin))
                result.Add(skin);
        }
        return result;
    }
}
```

## 五、皮肤系统设计参考

| 类型 | 示例 | 特点 |
|------|------|------|
| 颜色皮肤 | 金色/彩虹版本 | 低成本，用于免费奖励 |
| 全新造型 | 完全不同的角色外观 | 高成本，旗舰皮肤 |
| 动画皮肤 | 替换攻击/移动动画 | 独特感强，技术复杂 |
| 语音皮肤 | 替换角色配音 | 本地化内容较多 |
| 特效皮肤 | 击中/死亡特效替换 | 提升爽感 |

**皮肤变现建议：**
1. 保证每种稀有度都有对应价格（普通-史诗-传说）
2. 限时皮肤制造紧迫感（但不过度，避免玩家反感）
3. 皮肤不影响游戏平衡（不能让付费玩家有优势）
4. 提供预览功能（买前看清楚外观）
