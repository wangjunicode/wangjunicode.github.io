# 游戏战斗系统架构设计：技能框架与伤害计算

战斗系统是 ARPG、MOBA、RPG 等游戏的核心。一个设计良好的战斗系统需要同时满足：策划可配置、程序易扩展、网络可同步三大要求。本文以 UE5 GAS（Gameplay Ability System）的设计思想为参考，讲解中小型项目可落地的战斗系统架构。

## 一、战斗系统全景架构

```
战斗系统架构图：

实体层（CombatEntity）
├── 属性组件（AttributeComponent）  → HP/MP/攻击/防御/速度...
├── 技能组件（AbilityComponent）    → 技能列表/冷却管理
├── Buff组件（BuffComponent）       → Buff/Debuff状态
└── 战斗事件（CombatEventBus）      → 发布/订阅战斗事件

技能层（GameplayAbility）
├── 技能配置（AbilityConfig）       → 数据驱动，策划配置
├── 技能执行（AbilityExecution）    → 逻辑执行
├── 技能效果（GameplayEffect）      → 属性修改/Buff施加
└── 技能条件（AbilityCondition）    → 前置条件检查

效果层（GameplayEffect）
├── 即时效果（Instant）             → 一次性属性修改（造成伤害）
├── 持续效果（Duration）            → 有限时间的Buff
└── 无限效果（Infinite）            → 装备/天赋永久加成

属性层（AttributeSet）
├── 基础值（BaseValue）             → 原始属性
├── 修改器（Modifier）              → 加法/乘法/覆盖
└── 当前值（CurrentValue）          → 计算后的实际值
```

## 二、属性系统设计

### 2.1 属性集合（AttributeSet）

```csharp
// 属性定义：支持基础值和修改器的叠加计算
[Serializable]
public class AttributeValue
{
    public float BaseValue;           // 基础值（角色等级属性）
    
    private List<AttributeModifier> _addModifiers = new();  // 加法修改器（+50攻击力）
    private List<AttributeModifier> _mulModifiers = new();  // 乘法修改器（攻击+20%）
    private List<AttributeModifier> _overrideModifiers = new(); // 覆盖修改器
    
    // 计算最终值：BaseValue * (1 + 乘法修改器之和) + 加法修改器之和
    public float FinalValue
    {
        get
        {
            if (_overrideModifiers.Count > 0)
                return _overrideModifiers.Max(m => m.Value);
            
            float addTotal = _addModifiers.Sum(m => m.Value);
            float mulTotal = _mulModifiers.Sum(m => m.Value);
            
            return (BaseValue + addTotal) * (1f + mulTotal);
        }
    }
    
    public void AddModifier(AttributeModifier modifier)
    {
        switch (modifier.Type)
        {
            case ModifierType.Add:      _addModifiers.Add(modifier); break;
            case ModifierType.Multiply: _mulModifiers.Add(modifier); break;
            case ModifierType.Override: _overrideModifiers.Add(modifier); break;
        }
    }
    
    public void RemoveModifier(AttributeModifier modifier)
    {
        _addModifiers.Remove(modifier);
        _mulModifiers.Remove(modifier);
        _overrideModifiers.Remove(modifier);
    }
}

// 角色属性集
public class CharacterAttributeSet
{
    public AttributeValue MaxHP = new() { BaseValue = 100 };
    public AttributeValue MaxMP = new() { BaseValue = 50 };
    public AttributeValue Attack = new() { BaseValue = 20 };
    public AttributeValue Defense = new() { BaseValue = 10 };
    public AttributeValue CritRate = new() { BaseValue = 0.05f };
    public AttributeValue CritDamage = new() { BaseValue = 1.5f };
    public AttributeValue Speed = new() { BaseValue = 5f };
    
    // 当前值（HP/MP 受到修改）
    private float _currentHP;
    public float CurrentHP
    {
        get => _currentHP;
        set => _currentHP = Mathf.Clamp(value, 0, MaxHP.FinalValue);
    }
}
```

### 2.2 伤害计算公式

```csharp
public class DamageCalculator
{
    // 伤害公式：参考主流 ARPG 设计
    public static DamageResult Calculate(
        CombatEntity attacker, 
        CombatEntity defender, 
        DamageSpec spec)
    {
        var result = new DamageResult();
        
        // 1. 基础伤害 = 技能系数 * 攻击力
        float baseDamage = spec.SkillCoefficient * attacker.Attributes.Attack.FinalValue;
        
        // 2. 暴击判定
        float critRoll = UnityEngine.Random.value;
        bool isCrit = critRoll < attacker.Attributes.CritRate.FinalValue;
        
        if (isCrit)
            baseDamage *= attacker.Attributes.CritDamage.FinalValue;
        
        result.IsCritical = isCrit;
        
        // 3. 防御减伤公式（参考《原神》）
        // 减伤比 = 防御力 / (防御力 + 5 * 攻击者等级 + 500)
        float defenderDef = defender.Attributes.Defense.FinalValue;
        float attackerLevel = attacker.Level;
        float defenseMultiplier = 1f - defenderDef / (defenderDef + 5 * attackerLevel + 500);
        
        baseDamage *= defenseMultiplier;
        
        // 4. 元素抗性/伤害类型减免
        float resistance = GetResistance(defender, spec.DamageType);
        baseDamage *= (1f - resistance);
        
        // 5. 随机波动（±10%）
        float variance = UnityEngine.Random.Range(0.9f, 1.1f);
        baseDamage *= variance;
        
        result.FinalDamage = Mathf.Max(1, Mathf.RoundToInt(baseDamage)); // 最低伤害1
        
        return result;
    }
    
    private static float GetResistance(CombatEntity entity, DamageType damageType)
    {
        return damageType switch
        {
            DamageType.Physical  => entity.PhysicalResistance,
            DamageType.Fire      => entity.FireResistance,
            DamageType.Ice       => entity.IceResistance,
            DamageType.Lightning => entity.LightningResistance,
            _ => 0f
        };
    }
}
```

### 3.1 技能基类

```csharp
// 技能配置（策划配置）
[CreateAssetMenu(menuName = "Combat/Ability Config")]
public class AbilityConfig : ScriptableObject
{
    public string AbilityId;
    public string DisplayName;
    public Sprite Icon;
    
    [Header("代价")]
    public float MPCost;
    public float Cooldown;
    public float CastTime; // 0 = 瞬发
    
    [Header("效果")]
    public List<GameplayEffectConfig> Effects;
    
    [Header("条件")]
    public List<AbilityConditionConfig> Conditions;
    
    [Header("表现")]
    public string AnimationTrigger;
    public GameObject CastVFX;
    public AudioClip CastSFX;
}

// 技能执行基类
public abstract class GameplayAbility
{
    protected AbilityConfig _config;
    protected CombatEntity _owner;
    protected CombatEntity _target;
    
    // 检查技能是否可激活
    public virtual bool CanActivate()
    {
        // 检查 MP
        if (_owner.Attributes.CurrentMP < _config.MPCost) return false;
        
        // 检查冷却
        if (IsOnCooldown()) return false;
        
        // 检查自定义条件
        foreach (var condition in _config.Conditions)
        {
            if (!condition.IsMet(_owner, _target)) return false;
        }
        
        return true;
    }
    
    // 激活技能
    public async Task ActivateAsync(CombatEntity target)
    {
        _target = target;
        
        // 消耗资源
        _owner.Attributes.CurrentMP -= _config.MPCost;
        StartCooldown(_config.Cooldown);
        
        // 播放前摇
        if (_config.CastTime > 0)
        {
            await WaitForCastTime(_config.CastTime);
        }
        
        // 播放特效/音效
        PlayVisualEffects();
        
        // 执行技能效果
        await ExecuteAbilityAsync();
    }
    
    // 子类实现具体技能逻辑
    protected abstract Task ExecuteAbilityAsync();
    
    // 应用效果
    protected void ApplyEffects(CombatEntity target)
    {
        foreach (var effectConfig in _config.Effects)
        {
            var effect = GameplayEffectFactory.Create(effectConfig);
            effect.Apply(_owner, target);
        }
    }
}

// 具体技能：近战斩击
public class MeleeSlashAbility : GameplayAbility
{
    protected override async Task ExecuteAbilityAsync()
    {
        // 获取攻击范围内的所有目标
        var targets = GetTargetsInArc(
            origin: _owner.transform.position,
            direction: _owner.transform.forward,
            angle: 120f,
            radius: 2.5f
        );
        
        // 对每个目标应用效果
        foreach (var target in targets)
        {
            ApplyEffects(target);
        }
        
        // 等待动画完成
        await WaitForAnimation(_config.AnimationTrigger);
    }
}
```

### 3.2 Buff/Debuff 系统

```csharp
// Buff 基类
[Serializable]
public abstract class GameplayBuff
{
    public string BuffId;
    public string DisplayName;
    public Sprite Icon;
    public BuffType Type; // Buff / Debuff / Neutral
    
    public float Duration;          // -1 = 永久
    public float RemainingTime;
    public int StackCount = 1;      // 当前层数
    public int MaxStackCount = 1;   // 最大层数
    
    protected CombatEntity _target;
    protected CombatEntity _source;
    
    public virtual void OnApply(CombatEntity target, CombatEntity source)
    {
        _target = target;
        _source = source;
        RemainingTime = Duration;
        OnFirstApply();
    }
    
    // 首次施加时调用
    protected virtual void OnFirstApply() { }
    
    // 已有 Buff 时再次施加（叠层）
    public virtual void OnStack()
    {
        if (StackCount < MaxStackCount)
        {
            StackCount++;
            RemainingTime = Duration; // 刷新时间
        }
    }
    
    // 每帧 Tick
    public virtual void OnTick(float deltaTime)
    {
        if (Duration > 0)
        {
            RemainingTime -= deltaTime;
        }
    }
    
    public bool IsExpired => Duration > 0 && RemainingTime <= 0;
    
    // Buff 被移除时
    public virtual void OnRemove() { }
}

// 持续伤害 Buff（燃烧/中毒）
public class DamageOverTimeBuff : GameplayBuff
{
    public float DamagePerTick;
    public float TickInterval;
    private float _tickTimer;
    
    public override void OnTick(float deltaTime)
    {
        base.OnTick(deltaTime);
        
        _tickTimer += deltaTime;
        while (_tickTimer >= TickInterval)
        {
            _tickTimer -= TickInterval;
            ApplyTickDamage();
        }
    }
    
    private void ApplyTickDamage()
    {
        var spec = new DamageSpec
        {
            SkillCoefficient = DamagePerTick,
            DamageType = DamageType.Fire,
            IsDOT = true // 标记为 DOT 伤害（不触发某些效果）
        };
        
        var damage = DamageCalculator.Calculate(_source, _target, spec);
        _target.TakeDamage(damage);
    }
}

// Buff 管理器
public class BuffComponent
{
    private List<GameplayBuff> _activeBuffs = new();
    private CombatEntity _owner;
    
    public void ApplyBuff(GameplayBuff buff, CombatEntity source)
    {
        // 检查是否已有同类 Buff
        var existing = _activeBuffs.FirstOrDefault(b => b.BuffId == buff.BuffId);
        
        if (existing != null)
        {
            existing.OnStack();
        }
        else
        {
            buff.OnApply(_owner, source);
            _activeBuffs.Add(buff);
            OnBuffApplied?.Invoke(buff);
        }
    }
    
    public void Tick(float deltaTime)
    {
        for (int i = _activeBuffs.Count - 1; i >= 0; i--)
        {
            var buff = _activeBuffs[i];
            buff.OnTick(deltaTime);
            
            if (buff.IsExpired)
            {
                buff.OnRemove();
                _activeBuffs.RemoveAt(i);
                OnBuffRemoved?.Invoke(buff);
            }
        }
    }
    
    public void RemoveBuff(string buffId)
    {
        var buff = _activeBuffs.FirstOrDefault(b => b.BuffId == buffId);
        if (buff != null)
        {
            buff.OnRemove();
            _activeBuffs.Remove(buff);
            OnBuffRemoved?.Invoke(buff);
        }
    }
    
    public bool HasBuff(string buffId) 
        => _activeBuffs.Any(b => b.BuffId == buffId);
    
    public event Action<GameplayBuff> OnBuffApplied;
    public event Action<GameplayBuff> OnBuffRemoved;
}
```

```csharp
// 事件总线（解耦战斗各模块）
public static class CombatEventBus
{
    // 伤害事件
    public static event Action<DamageEvent> OnDamageDealt;
    public static event Action<HealEvent> OnHealApplied;
    public static event Action<KillEvent> OnEntityKilled;
    public static event Action<BuffEvent> OnBuffChanged;
    public static event Action<SkillEvent> OnSkillUsed;
    
    public static void PublishDamage(DamageEvent evt) => OnDamageDealt?.Invoke(evt);
    public static void PublishHeal(HealEvent evt) => OnHealApplied?.Invoke(evt);
    public static void PublishKill(KillEvent evt) => OnEntityKilled?.Invoke(evt);
}

// 事件数据
public record DamageEvent(
    CombatEntity Attacker,
    CombatEntity Defender,
    int Damage,
    DamageType DamageType,
    bool IsCritical,
    Vector3 HitPosition
);

// 战斗UI订阅事件
public class CombatUIManager : MonoBehaviour
{
    private void OnEnable()
    {
        CombatEventBus.OnDamageDealt += ShowDamageNumber;
        CombatEventBus.OnEntityKilled += ShowKillEffect;
    }
    
    private void OnDisable()
    {
        CombatEventBus.OnDamageDealt -= ShowDamageNumber;
        CombatEventBus.OnEntityKilled -= ShowKillEffect;
    }
    
    private void ShowDamageNumber(DamageEvent evt)
    {
        var popup = DamagePopupPool.Get();
        popup.Show(
            damage: evt.Damage,
            position: evt.HitPosition,
            isCrit: evt.IsCritical,
            color: GetDamageColor(evt.DamageType)
        );
    }
}
```

## 六、网络同步要点

```csharp
// 联机战斗：服务端权威，客户端预测
public class NetworkedCombatEntity
{
    // 技能激活请求（客户端 → 服务端）
    [ClientRpc]
    public void RequestAbility(string abilityId, int targetId)
    {
        // 客户端预测执行（视觉效果立即播放）
        PredictAbilityExecution(abilityId, targetId);
    }
    
    // 服务端验证并广播结果
    [ServerRpc]
    public void ServerActivateAbility(string abilityId, int targetId, ulong clientId)
    {
        // 服务端验证
        if (!ValidateAbilityUse(abilityId, clientId)) return;
        
        // 执行战斗计算
        var result = ExecuteAbility(abilityId, targetId);
        
        // 广播结果给所有客户端
        BroadcastAbilityResult(result);
    }
}
```

> 💡 **设计原则**：战斗系统的可配置性比可扩展性更重要。好的战斗框架应该让策划能在不修改代码的情况下完成 80% 的新技能设计。为此，要把技能逻辑的"边界"设计好——哪些归代码（伤害计算公式、目标选取逻辑），哪些归配置（技能系数、目标半径、特效资源）。
