## 一、工具链的价值

### 1.1 为什么技术负责人要重视工具链

```
传统工作方式的痛点：
- 策划修改数值：改配置表 → 找程序导入 → 等待构建 → 测试（半天）
- 美术检查资源：手动对照规范 → 发现问题 → 修改 → 再次检查（1-2小时/次）
- 程序调试AI：修改代码 → 构建 → 进游戏 → 测试（15-30分钟/次）

工具链优化后：
- 策划修改数值：改配置表 → 自动导入 → 实时预览（5分钟）
- 美术检查资源：一键批量检查 → 自动报告 → 针对性修改（10分钟）
- 程序调试AI：可视化行为树 → 实时查看AI状态 → 即时调整（2分钟）

ROI（投资回报率）：
工具开发时间：1周
节省时间：每天2-3小时 × 团队5人 = 10-15人时/天
理论收益：1周投入 → 1个月回本
```

[CustomEditor(typeof(EnemyConfig))]
public class EnemyConfigEditor : Editor
{
    private EnemyConfig _target;
    private bool _showSkillsFoldout = true;
    private bool _showPreviewFoldout = true;
    
    private void OnEnable()
    {
        _target = (EnemyConfig)target;
    }
    
    public override void OnInspectorGUI()
    {
        serializedObject.Update();
        
        // ===== 标题区域 =====
        EditorGUILayout.Space(5);
        
        using (new EditorGUILayout.HorizontalScope())
        {
            // 显示图标
            if (_target.icon != null)
            {
                GUILayout.Label(_target.icon.texture, GUILayout.Width(64), GUILayout.Height(64));
            }
            
            using (new EditorGUILayout.VerticalScope())
            {
                // 名称（高亮显示）
                GUIStyle titleStyle = new GUIStyle(EditorStyles.boldLabel);
                titleStyle.fontSize = 16;
                EditorGUILayout.LabelField(_target.enemyName, titleStyle);
                
                // 类型标签
                EditorGUILayout.LabelField($"类型: {_target.type}", EditorStyles.miniLabel);
            }
        }
        
        EditorGUILayout.Space(5);
        EditorGUILayout.LabelField("", GUI.skin.horizontalSlider);
        
        // ===== 基础属性 =====
        EditorGUILayout.LabelField("基础属性", EditorStyles.boldLabel);
        
        using (new EditorGUI.IndentLevelScope())
        {
            EditorGUILayout.PropertyField(serializedObject.FindProperty("enemyName"), new GUIContent("名称"));
            EditorGUILayout.PropertyField(serializedObject.FindProperty("level"), new GUIContent("等级"));
            
            // HP显示（带颜色条）
            EditorGUILayout.PropertyField(serializedObject.FindProperty("maxHP"), new GUIContent("最大HP"));
            
            // 攻击/防御并排显示
            using (new EditorGUILayout.HorizontalScope())
            {
                EditorGUILayout.PropertyField(serializedObject.FindProperty("attackPower"), new GUIContent("攻击力"));
                EditorGUILayout.PropertyField(serializedObject.FindProperty("defense"), new GUIContent("防御力"));
            }
        }
        
        EditorGUILayout.Space(5);
        
        // ===== 技能列表（可折叠）=====
        _showSkillsFoldout = EditorGUILayout.Foldout(_showSkillsFoldout, $"技能列表 ({_target.skills?.Count ?? 0})");
        if (_showSkillsFoldout)
        {
            using (new EditorGUI.IndentLevelScope())
            {
                EditorGUILayout.PropertyField(serializedObject.FindProperty("skills"), true);
            }
        }
        
        EditorGUILayout.Space(5);
        
        // ===== 操作按钮 =====
        using (new EditorGUILayout.HorizontalScope())
        {
            if (GUILayout.Button("预览怪物", GUILayout.Height(30)))
            {
                PreviewEnemy();
            }
            
            if (GUILayout.Button("复制配置", GUILayout.Height(30)))
            {
                DuplicateConfig();
            }
            
            // 危险操作用红色按钮
            GUI.backgroundColor = Color.red;
            if (GUILayout.Button("重置", GUILayout.Height(30), GUILayout.Width(60)))
            {
                if (EditorUtility.DisplayDialog("确认重置", "确定要重置所有配置吗？", "确定", "取消"))
                    ResetConfig();
            }
            GUI.backgroundColor = Color.white;
        }
        
        serializedObject.ApplyModifiedProperties();
    }
    
    private void PreviewEnemy()
    {
        if (_target.prefab == null) { Debug.LogWarning("没有设置Prefab"); return; }
        // 在场景中临时生成预览
        var preview = PrefabUtility.InstantiatePrefab(_target.prefab) as GameObject;
        Selection.activeGameObject = preview;
        SceneView.lastActiveSceneView?.FrameSelected();
    }
    
    private void DuplicateConfig()
    {
        string path = AssetDatabase.GetAssetPath(_target);
        string newPath = path.Replace(".asset", "_Copy.asset");
        AssetDatabase.CopyAsset(path, newPath);
        AssetDatabase.SaveAssets();
    }
    
    private void ResetConfig()
    {
        Undo.RecordObject(_target, "Reset Enemy Config");
        _target.maxHP = 100;
        _target.attackPower = 10;
        _target.defense = 5;
        EditorUtility.SetDirty(_target);
    }
}

// 实用工具：批量设置Texture的压缩格式
public class TextureBatchProcessor : EditorWindow
{
    [MenuItem("Tools/资源工具/批量Texture设置")]
    public static void ShowWindow()
    {
        GetWindow<TextureBatchProcessor>("批量Texture设置");
    }
    
    // 工具参数
    private string _searchPath = "Assets";
    private TextureImporterFormat _androidFormat = TextureImporterFormat.ASTC_6x6;
    private TextureImporterFormat _iosFormat = TextureImporterFormat.ASTC_6x6;
    private int _maxTextureSize = 1024;
    private bool _enableMipMaps = false;
    private bool _dryRun = true; // 预演模式，不真正修改
    
    private Vector2 _scrollPos;
    private List<string> _previewResults = new();
    
    void OnGUI()
    {
        EditorGUILayout.LabelField("批量Texture压缩格式设置", EditorStyles.boldLabel);
        EditorGUILayout.Space();
        
        // 搜索路径
        using (new EditorGUILayout.HorizontalScope())
        {
            _searchPath = EditorGUILayout.TextField("搜索路径", _searchPath);
            if (GUILayout.Button("选择", GUILayout.Width(50)))
            {
                string selected = EditorUtility.OpenFolderPanel("选择文件夹", _searchPath, "");
                if (!string.IsNullOrEmpty(selected))
                    _searchPath = "Assets" + selected.Substring(Application.dataPath.Length);
            }
        }
        
        EditorGUILayout.Space();
        EditorGUILayout.LabelField("压缩设置", EditorStyles.boldLabel);
        _androidFormat = (TextureImporterFormat)EditorGUILayout.EnumPopup("Android格式", _androidFormat);
        _iosFormat = (TextureImporterFormat)EditorGUILayout.EnumPopup("iOS格式", _iosFormat);
        _maxTextureSize = EditorGUILayout.IntPopup("最大尺寸", _maxTextureSize, 
            new[] { "256", "512", "1024", "2048" }, new[] { 256, 512, 1024, 2048 });
        _enableMipMaps = EditorGUILayout.Toggle("开启MipMaps", _enableMipMaps);
        
        EditorGUILayout.Space();
        _dryRun = EditorGUILayout.Toggle("预演模式（不修改文件）", _dryRun);
        
        EditorGUILayout.Space();
        
        using (new EditorGUILayout.HorizontalScope())
        {
            if (GUILayout.Button("预览", GUILayout.Height(30)))
                RunBatchProcess(true);
            
            GUI.enabled = !_dryRun;
            if (GUILayout.Button("执行", GUILayout.Height(30)))
            {
                if (EditorUtility.DisplayDialog("确认执行", 
                    $"将修改 {_previewResults.Count} 个Texture，此操作不可撤销。确定吗？", 
                    "执行", "取消"))
                    RunBatchProcess(false);
            }
            GUI.enabled = true;
        }
        
        EditorGUILayout.Space();
        
        // 预览结果列表
        if (_previewResults.Count > 0)
        {
            EditorGUILayout.LabelField($"找到 {_previewResults.Count} 个需要修改的Texture:", EditorStyles.boldLabel);
            _scrollPos = EditorGUILayout.BeginScrollView(_scrollPos, GUILayout.Height(200));
            foreach (var result in _previewResults)
                EditorGUILayout.LabelField(result, EditorStyles.miniLabel);
            EditorGUILayout.EndScrollView();
        }
    }
    
    void RunBatchProcess(bool dryRun)
    {
        _previewResults.Clear();
        
        string[] guids = AssetDatabase.FindAssets("t:Texture2D", new[] { _searchPath });
        int processedCount = 0;
        
        try
        {
            for (int i = 0; i < guids.Length; i++)
            {
                string path = AssetDatabase.GUIDToAssetPath(guids[i]);
                
                // 进度条
                if (EditorUtility.DisplayCancelableProgressBar(
                    "处理中...", 
                    $"处理: {path}", 
                    (float)i / guids.Length))
                {
                    break;
                }
                
                var importer = AssetImporter.GetAtPath(path) as TextureImporter;
                if (importer == null) continue;
                
                bool needsChange = false;
                var changes = new List<string>();
                
                // 检查Android设置
                var androidSettings = importer.GetPlatformTextureSettings("Android");
                if (!androidSettings.overridden || androidSettings.format != _androidFormat)
                {
                    changes.Add($"Android格式: {androidSettings.format} → {_androidFormat}");
                    needsChange = true;
                }
                
                // 如果不是预演，实际修改
                if (!dryRun && needsChange)
                {
                    var newAndroidSettings = new TextureImporterPlatformSettings
                    {
                        name = "Android",
                        overridden = true,
                        format = _androidFormat,
                        maxTextureSize = _maxTextureSize
                    };
                    importer.SetPlatformTextureSettings(newAndroidSettings);
                    importer.mipmapEnabled = _enableMipMaps;
                    importer.SaveAndReimport();
                    processedCount++;
                }
                
                if (needsChange)
                {
                    _previewResults.Add($"{path}: {string.Join(", ", changes)}");
                }
            }
        }
        finally
        {
            EditorUtility.ClearProgressBar();
        }
        
        if (!dryRun)
            EditorUtility.DisplayDialog("完成", $"成功处理 {processedCount} 个Texture", "OK");
    }
}

// 基础配置基类
public abstract class GameConfigBase : ScriptableObject
{

    [SerializeField] private string _displayName;

    public string DisplayName => _displayName;
}

// 技能配置
[CreateAssetMenu(fileName = "Skill_", menuName = "GameConfig/Skill", order = 1)]
public class SkillConfig : GameConfigBase
{
    [Header("基础参数")]
    public float cooldown = 1f;
    public float castRange = 5f;
    public SkillType type;
    
    [Header("效果")]
    public List<EffectConfig> effects;
    
    [Header("表现")]
    public GameObject castVFX;
    public AudioClip castSFX;
}

// 全局配置数据库（提供快速查找）
[CreateAssetMenu(fileName = "SkillDatabase", menuName = "GameConfig/SkillDatabase")]
public class SkillDatabase : ScriptableObject
{
    [SerializeField] private List<SkillConfig> _skills;
    
    private Dictionary<int, SkillConfig> _skillMap;
    
    // 编辑器中就可以验证
    private void OnValidate()
    {
        // 检查ID唯一性
        var ids = _skills.Select(s => s.ID);
        var duplicates = ids.GroupBy(id => id).Where(g => g.Count() > 1).Select(g => g.Key).ToList();
        if (duplicates.Count > 0)
        {
            Debug.LogError($"SkillDatabase存在重复ID: {string.Join(", ", duplicates)}");
        }
    }
    
    // 运行时初始化
    public void Initialize()
    {
        _skillMap = _skills.ToDictionary(s => s.ID);
    }
    
    public SkillConfig GetSkill(int id)
    {
        if (_skillMap.TryGetValue(id, out var skill))
            return skill;
        Debug.LogWarning($"技能ID不存在: {id}");
        return null;
    }
}

// 配置数据库编辑器工具
[CustomEditor(typeof(SkillDatabase))]
public class SkillDatabaseEditor : Editor
{
    public override void OnInspectorGUI()
    {
        base.OnInspectorGUI();
        
        EditorGUILayout.Space();
        EditorGUILayout.LabelField("工具", EditorStyles.boldLabel);
        
        using (new EditorGUILayout.HorizontalScope())
        {
            if (GUILayout.Button("自动收集所有技能配置"))
            {
                AutoCollectSkills();
            }
            
            if (GUILayout.Button("导出为CSV"))
            {
                ExportToCSV();
            }
        }
    }
    
    void AutoCollectSkills()
    {
        var db = (SkillDatabase)target;
        var sp = serializedObject.FindProperty("_skills");
        sp.ClearArray();
        
        string[] guids = AssetDatabase.FindAssets("t:SkillConfig");
        foreach (var guid in guids)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            var skill = AssetDatabase.LoadAssetAtPath<SkillConfig>(path);
            sp.arraySize++;
            sp.GetArrayElementAtIndex(sp.arraySize - 1).objectReferenceValue = skill;
        }
        
        serializedObject.ApplyModifiedProperties();
        EditorUtility.SetDirty(db);
        AssetDatabase.SaveAssets();
        
        Debug.Log($"✅ 自动收集完成，共找到 {sp.arraySize} 个技能配置");
    }
    
    void ExportToCSV()
    {
        var db = (SkillDatabase)target;
        // 反射获取字段...
        string csvContent = "ID,Name,Cooldown,CastRange,Type\n";
        // ... 导出逻辑
        
        string path = EditorUtility.SaveFilePanel("导出CSV", "", "skills", "csv");
        if (!string.IsNullOrEmpty(path))
        {
            File.WriteAllText(path, csvContent);
            Debug.Log($"导出成功: {path}");
        }
    }
}

// 让游戏逻辑在Scene视图中可见，加速调试
public class EnemyPatrolZone : MonoBehaviour
{
    [Header("巡逻设置")]
    public float patrolRadius = 10f;
    public List<Vector3> patrolWaypoints = new();
    
    [Header("攻击设置")]
    public float detectionRange = 8f;
    public float attackRange = 2f;
    
    // 在Scene视图中绘制可视化提示
    void OnDrawGizmos()
    {
        // 绘制巡逻范围（蓝色圆）
        Gizmos.color = new Color(0, 0, 1, 0.2f);
        Gizmos.DrawSphere(transform.position, patrolRadius);
        Gizmos.color = Color.blue;
        Gizmos.DrawWireSphere(transform.position, patrolRadius);
        
        // 绘制侦测范围（黄色圆）
        Gizmos.color = new Color(1, 1, 0, 0.1f);
        Gizmos.DrawSphere(transform.position, detectionRange);
        Gizmos.color = Color.yellow;
        Gizmos.DrawWireSphere(transform.position, detectionRange);
        
        // 绘制攻击范围（红色圆）
        Gizmos.color = new Color(1, 0, 0, 0.2f);
        Gizmos.DrawSphere(transform.position, attackRange);
        Gizmos.color = Color.red;
        Gizmos.DrawWireSphere(transform.position, attackRange);
        
        // 绘制巡逻路径
        if (patrolWaypoints.Count > 1)
        {
            Gizmos.color = Color.green;
            for (int i = 0; i < patrolWaypoints.Count; i++)
            {
                Vector3 wp = transform.position + patrolWaypoints[i];
                Gizmos.DrawSphere(wp, 0.3f);
                
                Vector3 next = transform.position + patrolWaypoints[(i + 1) % patrolWaypoints.Count];
                Gizmos.DrawLine(wp, next);
            }
        }
    }
    
    // 仅在选中时显示更多信息
    void OnDrawGizmosSelected()
    {
        // 用Handles绘制更复杂的UI
        Handles.color = Color.white;
        Handles.Label(transform.position + Vector3.up * 2, 
            $"巡逻半径: {patrolRadius}m\n侦测: {detectionRange}m\n攻击: {attackRange}m");
    }
}

// 编辑器时可拖拽的巡逻路径点
[CustomEditor(typeof(EnemyPatrolZone))]
public class EnemyPatrolZoneEditor : Editor
{
    private EnemyPatrolZone _target;
    
    void OnEnable() => _target = (EnemyPatrolZone)target;
    
    void OnSceneGUI()
    {
        // 让每个路径点可以在Scene中直接拖拽
        for (int i = 0; i < _target.patrolWaypoints.Count; i++)
        {
            Vector3 worldPos = _target.transform.position + _target.patrolWaypoints[i];
            
            EditorGUI.BeginChangeCheck();
            Vector3 newWorldPos = Handles.PositionHandle(worldPos, Quaternion.identity);
            
            if (EditorGUI.EndChangeCheck())
            {
                Undo.RecordObject(_target, "Move Waypoint");
                _target.patrolWaypoints[i] = newWorldPos - _target.transform.position;
            }
        }
    }
}

public class GameDevTools
{
    // ===== 快速工具 =====
    
    [MenuItem("Tools/清理/删除空GameObject")]
    static void RemoveEmptyGameObjects()
    {
        var emptyObjects = new List<GameObject>();
        var allObjects = Resources.FindObjectsOfTypeAll<GameObject>();
        
        foreach (var go in allObjects)
        {
            if (go.hideFlags != HideFlags.None) continue;
            if (go.transform.childCount == 0 && 
                go.GetComponents<Component>().Length == 1) // 只有Transform
            {
                emptyObjects.Add(go);
            }
        }
        
        if (emptyObjects.Count > 0 && 
            EditorUtility.DisplayDialog("删除空节点", 
                $"找到 {emptyObjects.Count} 个空节点，确认删除？", "删除", "取消"))
        {
            foreach (var go in emptyObjects)
                Undo.DestroyObjectImmediate(go);
        }
    }
    
    [MenuItem("Tools/场景/截图")]
    static void TakeEditorScreenshot()
    {
        string path = EditorUtility.SaveFilePanel("保存截图", "", "screenshot", "png");
        if (string.IsNullOrEmpty(path)) return;
        
        var sceneView = SceneView.lastActiveSceneView;
        int width = (int)sceneView.position.width;
        int height = (int)sceneView.position.height;
        
        var rt = new RenderTexture(width, height, 24);
        sceneView.camera.targetTexture = rt;
        sceneView.camera.Render();
        
        RenderTexture.active = rt;
        var screenshot = new Texture2D(width, height, TextureFormat.RGB24, false);
        screenshot.ReadPixels(new Rect(0, 0, width, height), 0, 0);
        screenshot.Apply();
        
        File.WriteAllBytes(path, screenshot.EncodeToPNG());
        
        // 清理
        sceneView.camera.targetTexture = null;
        RenderTexture.active = null;
        Object.DestroyImmediate(rt);
        Object.DestroyImmediate(screenshot);
        
        Debug.Log($"截图已保存: {path}");
    }
    
    // 查找并替换材质
    [MenuItem("Tools/材质/批量替换材质")]
    static void BatchReplaceMaterial()
    {
        var window = GetWindow<MaterialReplacerWindow>("材质替换");
        window.Show();
    }
    
    // 选中对象在文件系统中打开
    [MenuItem("Assets/在文件管理器中打开")]
    static void OpenInExplorer()
    {
        string path = AssetDatabase.GetAssetPath(Selection.activeObject);
        if (!string.IsNullOrEmpty(path))
        {
            string fullPath = Path.GetFullPath(path);
            EditorUtility.RevealInFinder(fullPath);
        }
    }
    
    // 快速设置对象的Layer
    [MenuItem("GameObject/设置Layer/Enemy", false, 0)]
    static void SetLayerEnemy()
    {
        foreach (var go in Selection.gameObjects)
            SetLayerRecursively(go, LayerMask.NameToLayer("Enemy"));
    }
    
    static void SetLayerRecursively(GameObject go, int layer)
    {
        go.layer = layer;
        foreach (Transform child in go.transform)
            SetLayerRecursively(child.gameObject, layer);
    }

```csharp
// 将Excel配置表自动转为ScriptableObject
// 策划改表 → 一键导入 → 游戏生效
using NPOI.SS.UserModel;
using NPOI.XSSF.UserModel;

public class ExcelImporter : EditorWindow
{
    [MenuItem("Tools/配置/导入Excel配置表")]
    static void ShowWindow()
    {
        GetWindow<ExcelImporter>("Excel导入");
    }
    
    void OnGUI()
    {
        if (GUILayout.Button("导入技能配置", GUILayout.Height(40)))
            ImportSkillConfig();
        
        if (GUILayout.Button("导入怪物配置", GUILayout.Height(40)))
            ImportEnemyConfig();
        
        if (GUILayout.Button("导入全部配置", GUILayout.Height(40)))
        {
            ImportSkillConfig();
            ImportEnemyConfig();
            AssetDatabase.SaveAssets();
            Debug.Log("✅ 全部配置导入完成");
        }
    }
    
    void ImportSkillConfig()
    {
        string excelPath = "Assets/Configs/Excel/skills.xlsx";
        
        using var file = new FileStream(excelPath, FileMode.Open, FileAccess.Read);
        var workbook = new XSSFWorkbook(file);
        var sheet = workbook.GetSheetAt(0);
        
        // 第一行是表头
        var header = sheet.GetRow(0);
        
        for (int rowIdx = 1; rowIdx <= sheet.LastRowNum; rowIdx++)
        {
            var row = sheet.GetRow(rowIdx);
            if (row == null) continue;
            
            int id = (int)row.GetCell(0).NumericCellValue;
            string name = row.GetCell(1).StringCellValue;
            float cooldown = (float)row.GetCell(2).NumericCellValue;
            
            // 查找或创建ScriptableObject
            string assetPath = $"Assets/Configs/Skills/Skill_{id:D3}.asset";
            var config = AssetDatabase.LoadAssetAtPath<SkillConfig>(assetPath);
            
            if (config == null)
            {
                config = ScriptableObject.CreateInstance<SkillConfig>();
                AssetDatabase.CreateAsset(config, assetPath);
            }
            
            // 更新数据
            // config.id = id;
            // config.displayName = name;
            config.cooldown = cooldown;
            
            EditorUtility.SetDirty(config);
        }
        
        Debug.Log($"技能配置导入完成，共处理 {sheet.LastRowNum} 条记录");
    }

## 总结：工具链建设优先级

```
Phase 1（立即可带来收益）：
✅ 批量资源检查工具（美术质量保障）
✅ 配置数据自动导入（策划效率）
✅ 常用快捷操作MenuItem（日常效率）

Phase 2（中期投入）：
✅ 可视化调试工具（程序调试效率）
✅ 场景辅助工具（场景搭建效率）
✅ 性能报告工具（QA测试效率）

Phase 3（长期建设）：
✅ 完整的编辑器内测试框架
✅ 热更新发布工具（运营效率）
✅ 数据可视化分析工具（数值调平）

技术负责人的工具哲学：
"工具是给团队用的，不是给自己秀技术的"
→ 优先考虑最能解放团队生产力的工具
→ 工具要简单易用（美术策划也能用）
→ 工具要有文档（新人能上手）
