### 2.1 自定义 EditorWindow

```csharp
using UnityEngine;
using UnityEditor;
using System.Collections.Generic;

public class LevelEditorWindow : EditorWindow
{
    // 工具栏状态
    private enum EditMode { Select, Paint, Erase, Fill, Picker }
    private EditMode _currentMode = EditMode.Paint;
    
    // 当前编辑数据
    private LevelData _levelData;
    private LayerData _activeLayer;
    private TileInfo _selectedTile;
    
    // 视图状态
    private Vector2 _scrollPos;
    private float _zoom = 1.0f;
    private Vector2 _panOffset;
    
    // 画笔设置
    private int _brushSize = 1;
    private bool _showGrid = true;
    private bool _showCollision = false;
    
    // 撤销重做栈
    private CommandHistory _commandHistory = new CommandHistory();
    
    [MenuItem("Tools/Level Editor")]
    public static void ShowWindow()
    {
        var window = GetWindow<LevelEditorWindow>("关卡编辑器");
        window.minSize = new Vector2(1000, 700);
        window.Show();
    }
    
    private void OnGUI()
    {
        DrawToolbar();
        
        EditorGUILayout.BeginHorizontal();
        {
            // 左侧：图层面板
            DrawLayerPanel();
            
            // 中间：主视口
            DrawMainViewport();
            
            // 右侧：Tile选择面板
            DrawTilePalette();
        }
        EditorGUILayout.EndHorizontal();
        
        DrawStatusBar();
        HandleKeyboardShortcuts();
    }
    
    private void DrawToolbar()
    {
        EditorGUILayout.BeginHorizontal(EditorStyles.toolbar);
        {
            // 文件操作
            if (GUILayout.Button("新建", EditorStyles.toolbarButton, GUILayout.Width(50)))
                NewLevel();
            if (GUILayout.Button("打开", EditorStyles.toolbarButton, GUILayout.Width(50)))
                OpenLevel();
            if (GUILayout.Button("保存", EditorStyles.toolbarButton, GUILayout.Width(50)))
                SaveLevel();
            
            GUILayout.Space(10);
            
            // 编辑模式
            string[] modeNames = { "选择", "绘制", "擦除", "填充", "拾取" };
            _currentMode = (EditMode)GUILayout.Toolbar((int)_currentMode, modeNames,
                EditorStyles.toolbarButton, GUILayout.Width(250));
            
            GUILayout.Space(10);
            
            // 撤销重做
            GUI.enabled = _commandHistory.CanUndo;
            if (GUILayout.Button("撤销(Ctrl+Z)", EditorStyles.toolbarButton))
                _commandHistory.Undo();
            
            GUI.enabled = _commandHistory.CanRedo;
            if (GUILayout.Button("重做(Ctrl+Y)", EditorStyles.toolbarButton))
                _commandHistory.Redo();
            GUI.enabled = true;
            
            GUILayout.FlexibleSpace();
            
            // 视图控制
            _showGrid = GUILayout.Toggle(_showGrid, "网格", EditorStyles.toolbarButton);
            _showCollision = GUILayout.Toggle(_showCollision, "碰撞", EditorStyles.toolbarButton);
            
            GUILayout.Label($"缩放: {_zoom:F1}x", EditorStyles.toolbarLabel);
        }
        EditorGUILayout.EndHorizontal();
    }
    
    private void DrawMainViewport()
    {
        // 获取视口区域
        var viewportRect = GUILayoutUtility.GetRect(GUIContent.none, GUIStyle.none,
            GUILayout.ExpandWidth(true), GUILayout.ExpandHeight(true));
        
        if (_levelData == null) return;
        
        // 绘制背景
        EditorGUI.DrawRect(viewportRect, new Color(0.2f, 0.2f, 0.2f));
        
        // 裁剪到视口区域
        GUI.BeginClip(viewportRect);
        {
            // 应用缩放和平移变换
            var matrix = GUI.matrix;
            GUIUtility.ScaleAroundPivot(Vector2.one * _zoom, viewportRect.size * 0.5f);
            
            // 绘制网格
            if (_showGrid)
                DrawGrid(viewportRect);
            
            // 绘制所有图层
            foreach (var layer in _levelData.layers)
            {
                if (!layer.isVisible) continue;
                DrawLayer(layer, viewportRect);
            }
            
            // 绘制碰撞可视化
            if (_showCollision)
                DrawCollisionOverlay(viewportRect);
            
            // 绘制实体
            DrawEntities(viewportRect);
            
            // 绘制画笔预览
            DrawBrushPreview(viewportRect);
            
            GUI.matrix = matrix;
        }
        GUI.EndClip();
        
        // 处理鼠标事件（在视口内）
        HandleViewportMouseEvents(viewportRect);
    }
    
    private void DrawGrid(Rect viewportRect)
    {
        if (_levelData == null) return;
        
        Handles.color = new Color(0.5f, 0.5f, 0.5f, 0.3f);
        float tileSize = _levelData.tileSize * _zoom * EditorGUIUtility.pixelsPerPoint;
        
        // 垂直线
        for (int x = 0; x <= _levelData.mapSize.x; x++)
        {
            float xPos = x * tileSize + _panOffset.x;
            Handles.DrawLine(
                new Vector3(xPos, 0),
                new Vector3(xPos, _levelData.mapSize.y * tileSize)
            );
        }
        
        // 水平线
        for (int y = 0; y <= _levelData.mapSize.y; y++)
        {
            float yPos = y * tileSize + _panOffset.y;
            Handles.DrawLine(
                new Vector3(0, yPos),
                new Vector3(_levelData.mapSize.x * tileSize, yPos)
            );
        }
    }
    
    private Vector2Int WorldToTile(Vector2 worldPos)
    {
        return new Vector2Int(
            Mathf.FloorToInt((worldPos.x - _panOffset.x) / (_levelData.tileSize * _zoom)),
            Mathf.FloorToInt((worldPos.y - _panOffset.y) / (_levelData.tileSize * _zoom))
        );
    }
    
    private void HandleViewportMouseEvents(Rect viewportRect)
    {
        var e = Event.current;
        if (!viewportRect.Contains(e.mousePosition)) return;
        
        switch (e.type)
        {
            case EventType.ScrollWheel:
                // 缩放
                float zoomDelta = -e.delta.y * 0.1f;
                _zoom = Mathf.Clamp(_zoom + zoomDelta, 0.25f, 4f);
                e.Use();
                Repaint();
                break;
                
            case EventType.MouseDrag when e.button == 2: // 中键平移
                _panOffset += e.delta;
                e.Use();
                Repaint();
                break;
                
            case EventType.MouseDown:
            case EventType.MouseDrag when e.button == 0:
                if (_activeLayer == null || _activeLayer.isLocked) break;
                var tilePos = WorldToTile(e.mousePosition - viewportRect.position);
                HandleTileEdit(tilePos, e);
                break;
        }
    }
    
    private void HandleTileEdit(Vector2Int tilePos, Event e)
    {
        if (!IsValidTilePos(tilePos)) return;
        
        switch (_currentMode)
        {
            case EditMode.Paint:
                if (_selectedTile != null)
                {
                    var cmd = new PaintTileCommand(_activeLayer, tilePos, _selectedTile, _brushSize);
                    _commandHistory.Execute(cmd);
                }
                break;
                
            case EditMode.Erase:
                var eraseCmd = new EraseTileCommand(_activeLayer, tilePos, _brushSize);
                _commandHistory.Execute(eraseCmd);
                break;
                
            case EditMode.Fill:
                if (e.type == EventType.MouseDown)
                {
                    var fillCmd = new FloodFillCommand(_activeLayer, tilePos, _selectedTile);
                    _commandHistory.Execute(fillCmd);
                }
                break;
                
            case EditMode.Picker:
                if (_activeLayer.tiles.TryGetValue(tilePos, out var tile))
                {
                    _selectedTile = new TileInfo
                    {
                        tileId = tile.tileId,
                        rotation = tile.rotation,
                        flipX = tile.flipX,
                        flipY = tile.flipY
                    };
                    _currentMode = EditMode.Paint; // 拾取后切换回绘制
                }
                break;
        }
        
        Repaint();
    }
    
    private bool IsValidTilePos(Vector2Int pos)
    {
        return pos.x >= 0 && pos.y >= 0 &&
               pos.x < _levelData.mapSize.x &&
               pos.y < _levelData.mapSize.y;
    }
    
    private void HandleKeyboardShortcuts()
    {
        var e = Event.current;
        if (e.type != EventType.KeyDown) return;
        
        if (e.control)
        {
            switch (e.keyCode)
            {
                case KeyCode.Z: _commandHistory.Undo(); e.Use(); break;
                case KeyCode.Y: _commandHistory.Redo(); e.Use(); break;
                case KeyCode.S: SaveLevel(); e.Use(); break;
            }
        }
        else
        {
            switch (e.keyCode)
            {
                case KeyCode.Q: _currentMode = EditMode.Select; Repaint(); break;
                case KeyCode.W: _currentMode = EditMode.Paint; Repaint(); break;
                case KeyCode.E: _currentMode = EditMode.Erase; Repaint(); break;
                case KeyCode.R: _currentMode = EditMode.Fill; Repaint(); break;
                case KeyCode.T: _currentMode = EditMode.Picker; Repaint(); break;
            }
        }
    }
}
```

### 2.2 命令模式实现撤销重做

```csharp
// 命令接口
public interface ILevelEditCommand
{
    string Description { get; }
    void Execute();
    void Undo();
}

// 命令历史管理
public class CommandHistory
{
    private readonly Stack<ILevelEditCommand> _undoStack = new();
    private readonly Stack<ILevelEditCommand> _redoStack = new();
    private const int MaxHistorySize = 100;
    
    public bool CanUndo => _undoStack.Count > 0;
    public bool CanRedo => _redoStack.Count > 0;
    
    public void Execute(ILevelEditCommand command)
    {
        command.Execute();
        _undoStack.Push(command);
        _redoStack.Clear(); // 执行新命令后清空重做栈
        
        // 限制历史记录大小
        if (_undoStack.Count > MaxHistorySize)
        {
            var temp = _undoStack.ToArray();
            _undoStack.Clear();
            for (int i = 0; i < MaxHistorySize - 1; i++)
                _undoStack.Push(temp[i]);
        }
    }
    
    public void Undo()
    {
        if (!CanUndo) return;
        var cmd = _undoStack.Pop();
        cmd.Undo();
        _redoStack.Push(cmd);
    }
    
    public void Redo()
    {
        if (!CanRedo) return;
        var cmd = _redoStack.Pop();
        cmd.Execute();
        _undoStack.Push(cmd);
    }
}

// 绘制Tile命令
public class PaintTileCommand : ILevelEditCommand
{
    private readonly LayerData _layer;
    private readonly Vector2Int _center;
    private readonly TileInfo _newTile;
    private readonly int _brushSize;
    private readonly Dictionary<Vector2Int, TileInfo> _previousTiles = new();
    
    public string Description => $"绘制Tile at {_center} (brush:{_brushSize})";
    
    public PaintTileCommand(LayerData layer, Vector2Int center, TileInfo tile, int brushSize)
    {
        _layer = layer;
        _center = center;
        _newTile = tile;
        _brushSize = brushSize;
    }
    
    public void Execute()
    {
        _previousTiles.Clear();
        int half = _brushSize / 2;
        for (int dx = -half; dx <= half; dx++)
        {
            for (int dy = -half; dy <= half; dy++)
            {
                var pos = _center + new Vector2Int(dx, dy);
                // 保存原始状态用于撤销
                _previousTiles[pos] = _layer.tiles.TryGetValue(pos, out var old) ? old : null;
                _layer.tiles[pos] = new TileInfo
                {
                    tileId = _newTile.tileId,
                    rotation = _newTile.rotation,
                    flipX = _newTile.flipX,
                    flipY = _newTile.flipY
                };
            }
        }
    }
    
    public void Undo()
    {
        foreach (var kv in _previousTiles)
        {
            if (kv.Value == null)
                _layer.tiles.Remove(kv.Key);
            else
                _layer.tiles[kv.Key] = kv.Value;
        }
    }
}

// 洪水填充命令
public class FloodFillCommand : ILevelEditCommand
{
    private readonly LayerData _layer;
    private readonly Vector2Int _startPos;
    private readonly TileInfo _fillTile;
    private readonly Dictionary<Vector2Int, TileInfo> _changes = new();
    private string _targetTileId;
    
    public string Description => $"填充 at {_startPos}";
    
    public FloodFillCommand(LayerData layer, Vector2Int startPos, TileInfo fillTile)
    {
        _layer = layer;
        _startPos = startPos;
        _fillTile = fillTile;
    }
    
    public void Execute()
    {
        _changes.Clear();
        _layer.tiles.TryGetValue(_startPos, out var startTile);
        _targetTileId = startTile?.tileId;
        
        FloodFill(_startPos, _targetTileId);
    }
    
    private void FloodFill(Vector2Int pos, string targetId)
    {
        var queue = new Queue<Vector2Int>();
        var visited = new HashSet<Vector2Int>();
        queue.Enqueue(pos);
        
        while (queue.Count > 0)
        {
            var current = queue.Dequeue();
            if (visited.Contains(current)) continue;
            visited.Add(current);
            
            // 检查当前格子是否是目标ID
            _layer.tiles.TryGetValue(current, out var existingTile);
            string existingId = existingTile?.tileId;
            if (existingId != targetId) continue;
            
            // 记录变更
            _changes[current] = existingTile;
            _layer.tiles[current] = new TileInfo
            {
                tileId = _fillTile.tileId,
                rotation = _fillTile.rotation,
                flipX = _fillTile.flipX,
                flipY = _fillTile.flipY
            };
            
            // 四向扩展
            var neighbors = new[]
            {
                current + Vector2Int.up,
                current + Vector2Int.down,
                current + Vector2Int.left,
                current + Vector2Int.right
            };
            
            foreach (var n in neighbors)
            {
                if (!visited.Contains(n))
                    queue.Enqueue(n);
            }
        }
    }
    
    public void Undo()
    {
        foreach (var kv in _changes)
        {
            if (kv.Value == null)
                _layer.tiles.Remove(kv.Key);
            else
                _layer.tiles[kv.Key] = kv.Value;
        }
    }
}
```

### 3.1 JSON序列化（人类可读，适合版本控制）

```csharp
using Newtonsoft.Json;
using Newtonsoft.Json.Converters;
using System.IO;

public static class LevelSerializer
{
    private static readonly JsonSerializerSettings _settings = new JsonSerializerSettings
    {
        Formatting = Formatting.Indented,
        NullValueHandling = NullValueHandling.Ignore,
        DefaultValueHandling = DefaultValueHandling.Ignore,
        Converters = new List<JsonConverter>
        {
            new StringEnumConverter(),
            new Vector2IntConverter(),
            new Vector2Converter()
        }
    };
    
    public static void SaveToJson(LevelData level, string path)
    {
        var json = JsonConvert.SerializeObject(level, _settings);
        File.WriteAllText(path, json);
        Debug.Log($"[LevelSerializer] 保存成功: {path} ({json.Length} bytes)");
    }
    
    public static LevelData LoadFromJson(string path)
    {
        if (!File.Exists(path))
            throw new FileNotFoundException($"关卡文件不存在: {path}");
        
        var json = File.ReadAllText(path);
        var level = JsonConvert.DeserializeObject<LevelData>(json, _settings);
        ValidateLevelData(level);
        return level;
    }
    
    // 二进制序列化（运行时用，体积小、加载快）
    public static byte[] SerializeBinary(LevelData level)
    {
        using var ms = new MemoryStream();
        using var bw = new BinaryWriter(ms);
        
        bw.Write(level.levelId);
        bw.Write(level.version);
        bw.Write(level.mapSize.x);
        bw.Write(level.mapSize.y);
        bw.Write(level.tileSize);
        bw.Write(level.layers.Count);
        
        foreach (var layer in level.layers)
        {
            bw.Write(layer.layerId);
            bw.Write(layer.layerName);
            bw.Write((int)layer.layerType);
            bw.Write(layer.sortingOrder);
            bw.Write(layer.tiles.Count);
            
            foreach (var (pos, tile) in layer.tiles)
            {
                bw.Write((short)pos.x);
                bw.Write((short)pos.y);
                bw.Write(tile.tileId);
                bw.Write((byte)tile.rotation);
                bw.Write(tile.flipX);
                bw.Write(tile.flipY);
            }
        }
        
        return ms.ToArray();
    }
    
    private static void ValidateLevelData(LevelData level)
    {
        if (level == null) throw new System.Exception("关卡数据为null");
        if (string.IsNullOrEmpty(level.levelId)) throw new System.Exception("levelId为空");
        if (level.mapSize.x <= 0 || level.mapSize.y <= 0) 
            throw new System.Exception("地图尺寸非法");
        level.layers ??= new List<LayerData>();
        level.entities ??= new List<EntityData>();
        level.triggers ??= new List<TriggerData>();
    }
}

// 自定义JSON转换器
public class Vector2IntConverter : JsonConverter<Vector2Int>
{
    public override Vector2Int ReadJson(JsonReader reader, Type objectType,
        Vector2Int existingValue, bool hasExistingValue, JsonSerializer serializer)
    {
        var token = JToken.Load(reader);
        return new Vector2Int(token["x"].Value<int>(), token["y"].Value<int>());
    }
    
    public override void WriteJson(JsonWriter writer, Vector2Int value, JsonSerializer serializer)
    {
        writer.WriteStartObject();
        writer.WritePropertyName("x"); writer.WriteValue(value.x);
        writer.WritePropertyName("y"); writer.WriteValue(value.y);
        writer.WriteEndObject();
    }
}
```

### 4.1 基于噪声的地形生成器

```csharp
public class ProceduralTerrainGenerator
{
    [System.Serializable]
    public class TerrainConfig
    {
        public int width = 64;
        public int height = 64;
        public float scale = 20f;
        public int octaves = 4;
        public float persistence = 0.5f;
        public float lacunarity = 2.0f;
        public int seed = 42;
        public Vector2 offset;
        
        // 高度阈值映射到TileID
        public List<TerrainLayer> layers;
    }
    
    [System.Serializable]
    public class TerrainLayer
    {
        public string tileName;   // 如 "deep_water", "water", "sand", "grass", "mountain"
        public float heightThreshold; // 0~1，低于此值使用该层
        public Color debugColor;
    }
    
    public static LayerData GenerateTerrain(TerrainConfig config)
    {
        var heightMap = GenerateHeightMap(config);
        var layer = new LayerData
        {
            layerId = System.Guid.NewGuid().ToString(),
            layerName = "Terrain_Generated",
            layerType = LayerType.Ground,
            tiles = new Dictionary<Vector2Int, TileInfo>()
        };
        
        for (int x = 0; x < config.width; x++)
        {
            for (int y = 0; y < config.height; y++)
            {
                float height = heightMap[x, y];
                string tileId = GetTileIdForHeight(height, config.layers);
                
                layer.tiles[new Vector2Int(x, y)] = new TileInfo
                {
                    tileId = tileId,
                    rotation = 0,
                    flipX = false,
                    flipY = false
                };
            }
        }
        
        // 后处理：水岸自动拼接
        ApplyAutoTilingRules(layer, config);
        
        return layer;
    }
    
    private static float[,] GenerateHeightMap(TerrainConfig config)
    {
        var heightMap = new float[config.width, config.height];
        var rng = new System.Random(config.seed);
        
        // 生成各个octave的随机偏移
        var octaveOffsets = new Vector2[config.octaves];
        for (int i = 0; i < config.octaves; i++)
        {
            octaveOffsets[i] = new Vector2(
                rng.Next(-100000, 100000) + config.offset.x,
                rng.Next(-100000, 100000) + config.offset.y
            );
        }
        
        float maxHeight = float.MinValue;
        float minHeight = float.MaxValue;
        
        for (int x = 0; x < config.width; x++)
        {
            for (int y = 0; y < config.height; y++)
            {
                float amplitude = 1f;
                float frequency = 1f;
                float noiseValue = 0f;
                
                for (int o = 0; o < config.octaves; o++)
                {
                    float sampleX = (x - config.width * 0.5f) / config.scale * frequency + octaveOffsets[o].x;
                    float sampleY = (y - config.height * 0.5f) / config.scale * frequency + octaveOffsets[o].y;
                    
                    float perlin = Mathf.PerlinNoise(sampleX, sampleY) * 2 - 1;
                    noiseValue += perlin * amplitude;
                    
                    amplitude *= config.persistence;
                    frequency *= config.lacunarity;
                }
                
                heightMap[x, y] = noiseValue;
                maxHeight = Mathf.Max(maxHeight, noiseValue);
                minHeight = Mathf.Min(minHeight, noiseValue);
            }
        }
        
        // 归一化到 [0, 1]
        for (int x = 0; x < config.width; x++)
            for (int y = 0; y < config.height; y++)
                heightMap[x, y] = Mathf.InverseLerp(minHeight, maxHeight, heightMap[x, y]);
        
        return heightMap;
    }
    
    private static string GetTileIdForHeight(float height, List<TerrainLayer> layers)
    {
        // layers 按 heightThreshold 升序排列
        foreach (var layer in layers)
        {
            if (height <= layer.heightThreshold)
                return layer.tileName;
        }
        return layers[layers.Count - 1].tileName;
    }
    
    // 自动拼接规则（Wang Tile / 位图遮罩方式）
    private static void ApplyAutoTilingRules(LayerData layer, TerrainConfig config)
    {
        // 检测岸边：若某格为沙地，且相邻有水，应用对应方向的过渡Tile
        var positions = new List<Vector2Int>(layer.tiles.Keys);
        foreach (var pos in positions)
        {
            var tile = layer.tiles[pos];
            if (tile.tileId != "sand") continue;
            
            // 计算4邻域遮罩
            int mask = 0;
            if (GetTileId(layer, pos + Vector2Int.up) == "water" ||
                GetTileId(layer, pos + Vector2Int.up) == "deep_water") mask |= 1; // N
            if (GetTileId(layer, pos + Vector2Int.right) == "water" ||
                GetTileId(layer, pos + Vector2Int.right) == "deep_water") mask |= 2; // E
            if (GetTileId(layer, pos + Vector2Int.down) == "water" ||
                GetTileId(layer, pos + Vector2Int.down) == "deep_water") mask |= 4; // S
            if (GetTileId(layer, pos + Vector2Int.left) == "water" ||
                GetTileId(layer, pos + Vector2Int.left) == "deep_water") mask |= 8; // W
            
            if (mask > 0)
                tile.tileId = $"sand_transition_{mask:D2}";
        }
    }
    
    private static string GetTileId(LayerData layer, Vector2Int pos)
    {
        return layer.tiles.TryGetValue(pos, out var t) ? t.tileId : null;
    }
}
```

## 五、关卡异步流式加载

### 5.1 大型关卡分块加载系统

```csharp
public class LevelStreamingManager : MonoBehaviour
{
    [Header("流加载配置")]
    [SerializeField] private int _chunkSize = 16;          // 每块尺寸（格子数）
    [SerializeField] private int _loadRadius = 3;           // 加载半径（块）
    [SerializeField] private int _unloadRadius = 5;         // 卸载半径（块）
    [SerializeField] private int _loadBudgetPerFrame = 2;  // 每帧最多加载块数
    
    private Transform _playerTransform;
    private Dictionary<Vector2Int, LevelChunk> _loadedChunks = new();
    private HashSet<Vector2Int> _loadingChunks = new();
    private Queue<Vector2Int> _loadQueue = new();
    private LevelData _levelData;
    
    public struct LevelChunk
    {
        public Vector2Int chunkPos;
        public GameObject root;
        public bool isLoaded;
    }
    
    private void Update()
    {
        if (_playerTransform == null) return;
        
        var playerChunk = WorldToChunk(_playerTransform.position);
        
        // 计算需要加载/卸载的块
        UpdateLoadQueue(playerChunk);
        
        // 执行卸载
        UnloadDistantChunks(playerChunk);
        
        // 执行加载（受帧预算限制）
        ProcessLoadQueue();
    }
    
    private void UpdateLoadQueue(Vector2Int centerChunk)
    {
        for (int dx = -_loadRadius; dx <= _loadRadius; dx++)
        {
            for (int dy = -_loadRadius; dy <= _loadRadius; dy++)
            {
                var chunkPos = centerChunk + new Vector2Int(dx, dy);
                if (!_loadedChunks.ContainsKey(chunkPos) &&
                    !_loadingChunks.Contains(chunkPos) &&
                    !_loadQueue.Contains(chunkPos))
                {
                    _loadQueue.Enqueue(chunkPos);
                }
            }
        }
    }
    
    private void ProcessLoadQueue()
    {
        int loaded = 0;
        while (_loadQueue.Count > 0 && loaded < _loadBudgetPerFrame)
        {
            var chunkPos = _loadQueue.Dequeue();
            if (!_loadedChunks.ContainsKey(chunkPos))
            {
                StartCoroutine(LoadChunkAsync(chunkPos));
                loaded++;
            }
        }
    }
    
    private IEnumerator LoadChunkAsync(Vector2Int chunkPos)
    {
        _loadingChunks.Add(chunkPos);
        
        // 使用异步操作避免卡帧
        yield return null;
        
        var chunkGo = new GameObject($"Chunk_{chunkPos.x}_{chunkPos.y}");
        chunkGo.transform.parent = transform;
        chunkGo.transform.position = ChunkToWorld(chunkPos);
        
        // 获取该块内的所有Tile数据
        var tilesInChunk = GetTilesInChunk(chunkPos);
        
        int tileCount = 0;
        foreach (var (pos, tile) in tilesInChunk)
        {
            InstantiateTile(chunkGo.transform, pos, tile);
            
            tileCount++;
            if (tileCount % 50 == 0)
                yield return null; // 每50个Tile让出一帧
        }
        
        _loadedChunks[chunkPos] = new LevelChunk
        {
            chunkPos = chunkPos,
            root = chunkGo,
            isLoaded = true
        };
        
        _loadingChunks.Remove(chunkPos);
    }
    
    private void UnloadDistantChunks(Vector2Int centerChunk)
    {
        var toUnload = new List<Vector2Int>();
        foreach (var kv in _loadedChunks)
        {
            int dist = Mathf.Max(
                Mathf.Abs(kv.Key.x - centerChunk.x),
                Mathf.Abs(kv.Key.y - centerChunk.y)
            );
            if (dist > _unloadRadius)
                toUnload.Add(kv.Key);
        }
        
        foreach (var pos in toUnload)
        {
            if (_loadedChunks.TryGetValue(pos, out var chunk))
            {
                Destroy(chunk.root);
                _loadedChunks.Remove(pos);
            }
        }
    }
    
    private Vector2Int WorldToChunk(Vector3 worldPos)
    {
        float chunkWorldSize = _chunkSize * (_levelData?.tileSize ?? 1f);
        return new Vector2Int(
            Mathf.FloorToInt(worldPos.x / chunkWorldSize),
            Mathf.FloorToInt(worldPos.y / chunkWorldSize)
        );
    }
    
    private Vector3 ChunkToWorld(Vector2Int chunkPos)
    {
        float chunkWorldSize = _chunkSize * (_levelData?.tileSize ?? 1f);
        return new Vector3(chunkPos.x * chunkWorldSize, chunkPos.y * chunkWorldSize, 0);
    }
    
    private Dictionary<Vector2Int, TileInfo> GetTilesInChunk(Vector2Int chunkPos)
    {
        var result = new Dictionary<Vector2Int, TileInfo>();
        if (_levelData == null) return result;
        
        int startX = chunkPos.x * _chunkSize;
        int startY = chunkPos.y * _chunkSize;
        
        foreach (var layer in _levelData.layers)
        {
            foreach (var kv in layer.tiles)
            {
                if (kv.Key.x >= startX && kv.Key.x < startX + _chunkSize &&
                    kv.Key.y >= startY && kv.Key.y < startY + _chunkSize)
                {
                    result[kv.Key] = kv.Value;
                }
            }
        }
        
        return result;
    }
    
    private void InstantiateTile(Transform parent, Vector2Int tilePos, TileInfo tile)
    {
        // 实际项目中通过对象池管理
        var prefabPath = TileRegistry.GetPrefabPath(tile.tileId);
        if (string.IsNullOrEmpty(prefabPath)) return;
        
        var prefab = Resources.Load<GameObject>(prefabPath);
        if (prefab == null) return;
        
        var worldPos = new Vector3(
            tilePos.x * _levelData.tileSize,
            tilePos.y * _levelData.tileSize,
            0
        );
        
        var go = Instantiate(prefab, worldPos, Quaternion.Euler(0, 0, tile.rotation), parent);
        go.transform.localScale = new Vector3(
            tile.flipX ? -1 : 1,
            tile.flipY ? -1 : 1,
            1
        );
    }
}
```

```csharp
#if UNITY_EDITOR
public static class LevelValidator
{
    [System.Serializable]
    public class ValidationResult
    {
        public bool isValid;
        public List<string> errors = new();
        public List<string> warnings = new();
        public LevelStats stats;
    }
    
    [System.Serializable]
    public class LevelStats
    {
        public int totalTiles;
        public int entityCount;
        public int triggerCount;
        public bool hasSpawnPoint;
        public bool hasExitPoint;
        public float estimatedPlayTime; // 秒
    }
    
    public static ValidationResult Validate(LevelData level)
    {
        var result = new ValidationResult();
        result.stats = CollectStats(level);
        
        // 必要性检查
        if (level.layers.Count == 0)
            result.errors.Add("关卡没有任何图层");
        
        if (!result.stats.hasSpawnPoint)
            result.errors.Add("缺少玩家出生点（Spawn点）");
        
        if (!result.stats.hasExitPoint)
            result.warnings.Add("缺少关卡出口（Exit点），请确认是否为必要");
        
        // 可达性检查（BFS验证玩家能到达所有必经区域）
        var reachabilityErrors = CheckReachability(level);
        result.errors.AddRange(reachabilityErrors);
        
        // Tile引用检查
        var missingTiles = FindMissingTileReferences(level);
        if (missingTiles.Count > 0)
            result.warnings.Add($"存在 {missingTiles.Count} 个无效TileID引用");
        
        // 性能检查
        if (result.stats.totalTiles > 50000)
            result.warnings.Add($"Tile数量过多（{result.stats.totalTiles}），可能影响加载性能，建议使用流加载");
        
        result.isValid = result.errors.Count == 0;
        return result;
    }
    
    private static LevelStats CollectStats(LevelData level)
    {
        var stats = new LevelStats();
        
        foreach (var layer in level.layers)
            stats.totalTiles += layer.tiles.Count;
        
        stats.entityCount = level.entities.Count;
        stats.triggerCount = level.triggers.Count;
        
        stats.hasSpawnPoint = level.entities.Exists(e => e.entityType == "Spawn");
        stats.hasExitPoint = level.entities.Exists(e => e.entityType == "Exit");
        
        return stats;
    }
    
    private static List<string> CheckReachability(LevelData level)
    {
        var errors = new List<string>();
        var spawnPoint = level.entities.Find(e => e.entityType == "Spawn");
        if (spawnPoint == null) return errors;
        
        // 找到所有地面层
        var groundLayer = level.layers.Find(l => l.layerType == LayerType.Ground);
        if (groundLayer == null) return errors;
        
        // 从出生点BFS，检查是否能到达所有Exit
        var start = new Vector2Int(
            Mathf.RoundToInt(spawnPoint.position.x),
            Mathf.RoundToInt(spawnPoint.position.y)
        );
        
        var reachable = BFS(start, groundLayer);
        
        foreach (var entity in level.entities)
        {
            if (entity.entityType != "Exit") continue;
            var exitPos = new Vector2Int(
                Mathf.RoundToInt(entity.position.x),
                Mathf.RoundToInt(entity.position.y)
            );
            if (!reachable.Contains(exitPos))
                errors.Add($"出口 {entity.entityId} 从出生点不可达！");
        }
        
        return errors;
    }
    
    private static HashSet<Vector2Int> BFS(Vector2Int start, LayerData groundLayer)
    {
        var visited = new HashSet<Vector2Int>();
        var queue = new Queue<Vector2Int>();
        queue.Enqueue(start);
        
        while (queue.Count > 0)
        {
            var pos = queue.Dequeue();
            if (visited.Contains(pos)) continue;
            if (!groundLayer.tiles.ContainsKey(pos)) continue; // 不是可行走格子
            
            visited.Add(pos);
            
            queue.Enqueue(pos + Vector2Int.up);
            queue.Enqueue(pos + Vector2Int.down);
            queue.Enqueue(pos + Vector2Int.left);
            queue.Enqueue(pos + Vector2Int.right);
        }
        
        return visited;
    }
    
    private static List<string> FindMissingTileReferences(LevelData level)
    {
        var missing = new List<string>();
        var registry = TileRegistry.GetAllTileIds();
        
        foreach (var layer in level.layers)
        {
            foreach (var tile in layer.tiles.Values)
            {
                if (!registry.Contains(tile.tileId) && !missing.Contains(tile.tileId))
                    missing.Add(tile.tileId);
            }
        }
        
        return missing;
    }
}
```

## 七、最佳实践总结

### 7.1 编辑器性能优化

| 优化手段 | 说明 |
|---------|------|
| **稀疏存储** | 只记录非空格子，避免全量数组内存 |
| **脏标记** | 变更追踪，仅重绘变化区域 |
| **视口裁剪** | 只渲染可见范围内的格子 |
| **LOD** | 小缩放比例下合并显示，减少绘制调用 |
| **对象池** | 场景Tile对象复用，避免频繁GC |

### 7.2 工作流建议

```
策划流程：
  新建关卡 → 生成地形基础 → 手动细化 → 
  摆放实体 → 配置触发器 → 验证通过 → 导出

工具支持：
  ✓ 一键生成底图地形
  ✓ 自动拼接Tile（AutoTiling）
  ✓ 可达性自动验证
  ✓ 性能指标预估
  ✓ 关卡截图/预览图自动生成
  ✓ 多人协同（基于文本格式便于Diff与合并）
```

### 7.3 版本管理策略

- **JSON格式**：方便Git Diff，适合关卡脚本和配置数据
- **Binary格式**：体积小、加载快，适合Runtime发布包
- **构建管线**：提交JSON → CI自动转换Binary → 打包进AssetBundle
