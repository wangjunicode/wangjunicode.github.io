# 编辑器工具与美术工具链

## 1. 系统概述

本项目提供了一套完整的 Unity 编辑器工具链，涵盖战斗动画导出、Timeline 编辑、VFX 特效控制、资源分析等多个方向，极大提升了美术和策划的工作效率。

**主要工具分类：**
- **战斗录制与动画导出**：将战斗回放数据转换为 Maya/Max 可用的动画文件
- **Timeline 工具**：批量操作 Timeline Clip 的延时和帧率
- **特效控制编辑器**：预览和调试全局 Volume 效果（景深、Bloom 等）
- **资源分析工具**：AssetBundle 包体大小分析
- **动画层级编辑器**：批量修改动画 Hierarchy 绑定

## 2. 战斗动画导出工具

```csharp
// 位置：Assets/Scripts/Editor/Animation/BattleRecordAnimationGenerator.cs
// 功能：读取战斗录制的 JSON 数据，将摄像机动画 + 角色动画导出为 FBX/AnimationClip
// 用途：美术用 Maya/Max 制作过场动画时，需要基于真实战斗数据做参考
[MenuItem("ArtTools/战斗动画导出工具")]
public class BattleRecordAnimationGenerator : EditorWindow
{
    // 录制数据 JSON 路径（战斗录制后生成）
    string jsonPath = "Assets/Recordings/2024_11_14-15_55_58_View.json";
    
    // 导出帧范围控制
    private int exportFrameStart = 0;           // 从第几帧开始导出
    private int exportFrameAllCount = -1;        // -1 = 导出全部帧
    private bool onlyExportCameraAnim = false;   // true = 只导出摄像机动画（不含角色）
    
    public static void GenerateAnimation(string jsonPath, bool isOnlyExportCamera = false,
        int exportFrameStart = 0, int exportFrameAllCount = -1)
    {
        float frameRate = 30;  // 30FPS 导出
        
        // 1. 创建输出文件夹（与 JSON 同目录）
        string fileName = Path.GetFileNameWithoutExtension(jsonPath);
        string newFolderPath = Path.Combine(Path.GetDirectoryName(jsonPath), fileName);
        Directory.CreateDirectory(newFolderPath);
        
        // 2. 扫描项目内所有动画 Clip（用于匹配角色播放的动画名）
        Dictionary<string, AnimationClip> clipDict = new Dictionary<string, AnimationClip>();
        string[] animFiles = Directory.GetFiles(
            Path.Combine(Application.dataPath, "ArtRes/Characters/"), 
            "*.anim", SearchOption.AllDirectories);
        
        foreach (var file in animFiles)
        {
            if (file.Contains("Demochar")) continue;  // 跳过 Demo 资源
            string assetPath = "Assets" + file.Substring(Application.dataPath.Length);
            AnimationClip clip = AssetDatabase.LoadAssetAtPath<AnimationClip>(assetPath);
            if (clip != null && !clipDict.ContainsKey(clip.name))
            {
                clipDict[clip.name] = clip;
            }
        }
        
        // 3. 解析录制 JSON
        // JSON 包含每帧的：摄像机位置/旋转、各角色位置/旋转/动画状态
        var recordData = Serializer.DeserializeFromJson<BattleRecordData>(
            File.ReadAllText(jsonPath));
        
        // 4. 生成摄像机 AnimationClip
        var cameraClip = GenerateCameraAnimation(recordData, frameRate, 
            exportFrameStart, exportFrameAllCount);
        // 保存到项目
        AssetDatabase.CreateAsset(cameraClip, 
            $"{ChangeToAssetPath(newFolderPath)}/Camera.anim");
        
        // 5. 如果不只导摄像机，生成各角色 AnimationClip
        if (!isOnlyExportCamera)
        {
            foreach (var character in recordData.Characters)
            {
                var charClip = GenerateCharacterAnimation(
                    character, clipDict, frameRate, exportFrameStart, exportFrameAllCount);
                if (charClip != null)
                {
                    AssetDatabase.CreateAsset(charClip, 
                        $"{ChangeToAssetPath(newFolderPath)}/{character.Name}.anim");
                }
            }
        }
        
        AssetDatabase.SaveAssets();
        AssetDatabase.Refresh();
    }
}
```

## 3. Timeline Clip 批量延时工具

```csharp
// 位置：Assets/Scripts/Editor/Timeline/EffectDelayTimeToTimelineClipTools.cs
// 功能：批量给 Timeline 中的 Clip 添加延时偏移（如特效需要统一延后 N 帧）
public class EffectDelayTimeToTimelineClipTools : EditorWindow
{
    private TimelineAsset _targetTimeline;  // 目标 Timeline 资产
    private float _delayTime = 0.033f;      // 延时量（秒），默认 1 帧
    private bool _applyToAllTracks = true;  // 是否应用到所有轨道
    
    [MenuItem("ArtTools/Timeline/批量延时工具")]
    public static void ShowWindow()
    {
        GetWindow<EffectDelayTimeToTimelineClipTools>("Timeline 批量延时");
    }
    
    private void OnGUI()
    {
        _targetTimeline = (TimelineAsset)EditorGUILayout.ObjectField(
            "Timeline", _targetTimeline, typeof(TimelineAsset), false);
        _delayTime = EditorGUILayout.FloatField("延时（秒）", _delayTime);
        _applyToAllTracks = EditorGUILayout.Toggle("所有轨道", _applyToAllTracks);
        
        if (GUILayout.Button("执行批量延时"))
        {
            ApplyDelay();
        }
    }
    
    private void ApplyDelay()
    {
        if (_targetTimeline == null) return;
        
        Undo.RecordObject(_targetTimeline, "Apply Timeline Delay");
        
        foreach (var track in _targetTimeline.GetOutputTracks())
        {
            if (!_applyToAllTracks && !track.GetType().Name.Contains("Signal")) continue;
            
            foreach (var clip in track.GetClips())
            {
                clip.start += _delayTime;  // 所有 Clip 统一后移
            }
        }
        
        EditorUtility.SetDirty(_targetTimeline);
        AssetDatabase.SaveAssets();
        Debug.Log($"已对 {_targetTimeline.name} 的所有 Clip 延时 {_delayTime}s");
    }
}
```

## 4. 景深/全局 Volume 可视化编辑器

```csharp
// 位置：Assets/Scripts/Editor/VFX/GlobalVolumeControllerEditor.cs
// 功能：在 Inspector 中为 GlobalVolumeControllerCutscene 提供可视化景深参数控制
// 景深模式：Gaussian（高斯，性能好）vs Bokeh（物理光圈，效果更好）
[CanEditMultipleObjects]
public class GlobalVolumeControllerEditor : Editor
{
    // Bokeh 景深参数
    SerializedProperty focusDistance;    // 对焦距离（m）
    SerializedProperty focalLength;      // 焦距（mm），越大景深越浅
    SerializedProperty aperture;         // 光圈（f值），越小虚化越强
    SerializedProperty bladeCount;       // 光圈叶片数（影响焦外形状）
    SerializedProperty bladeCurvature;   // 叶片弯曲度（1=圆形）
    SerializedProperty bladeRotation;    // 叶片旋转角度
    
    // Gaussian 景深参数（性能版）
    SerializedProperty gaussianStart;    // 模糊开始距离
    SerializedProperty gaussianEnd;      // 完全模糊距离
    SerializedProperty gaussianMaxRadius;// 最大模糊半径（像素）
    SerializedProperty highQualitySampling; // 高质量采样（更平滑但更慢）
    
    // 是否忽略角色（角色不参与景深虚化）
    SerializedProperty isIgnoreCharacter;
    
    public override void OnInspectorGUI()
    {
        serializedObject.Update();
        
        // 模式选择：Bokeh 或 Gaussian
        _controllerCutscene.dofMode = (DepthOfFieldMode)EditorGUILayout.EnumPopup(
            "景深模式", _controllerCutscene.dofMode);
        
        switch (_controllerCutscene.dofMode)
        {
            case DepthOfFieldMode.Bokeh:
                EditorGUILayout.PropertyField(focusDistance, new GUIContent("对焦距离"));
                EditorGUILayout.PropertyField(focalLength, new GUIContent("焦距 (mm)"));
                EditorGUILayout.PropertyField(aperture, new GUIContent("光圈 (f/)"));
                EditorGUILayout.PropertyField(bladeCount, new GUIContent("叶片数"));
                EditorGUILayout.PropertyField(bladeCurvature, new GUIContent("叶片弯曲度"));
                EditorGUILayout.PropertyField(bladeRotation, new GUIContent("叶片旋转"));
                break;
                
            case DepthOfFieldMode.Gaussian:
                EditorGUILayout.PropertyField(gaussianStart, new GUIContent("虚化开始距离"));
                EditorGUILayout.PropertyField(gaussianEnd, new GUIContent("完全虚化距离"));
                EditorGUILayout.PropertyField(gaussianMaxRadius, new GUIContent("最大模糊半径"));
                EditorGUILayout.PropertyField(highQualitySampling, new GUIContent("高质量采样"));
                break;
        }
        
        EditorGUILayout.PropertyField(isIgnoreCharacter, new GUIContent("忽略角色"));
        
        serializedObject.ApplyModifiedProperties();
    }
}
```

## 5. 摄像机动画曲线工具

```csharp
// 位置：Assets/Scripts/Editor/Animation/CameraAnimationCurvesTool.cs
// 功能：将摄像机路径上的关键帧曲线可视化并批量调整
// 使用场景：过场动画的摄像机运动曲线优化（曲线太抖，需要平滑处理）
public class CameraAnimationCurvesTool : EditorWindow
{
    [MenuItem("ArtTools/摄像机动画曲线工具")]
    public static void ShowWindow()
    {
        GetWindow<CameraAnimationCurvesTool>("摄像机曲线");
    }
    
    // 平滑摄像机动画曲线（减少关键帧抖动）
    private void SmoothCurves(AnimationClip clip)
    {
        var bindings = AnimationUtility.GetCurveBindings(clip);
        foreach (var binding in bindings)
        {
            if (!binding.path.Contains("Camera")) continue;
            
            var curve = AnimationUtility.GetEditorCurve(clip, binding);
            
            // 对每个关键帧应用平滑切线
            for (int i = 0; i < curve.keys.Length; i++)
            {
                AnimationUtility.SetKeyLeftTangentMode(
                    curve, i, AnimationUtility.TangentMode.ClampedAuto);
                AnimationUtility.SetKeyRightTangentMode(
                    curve, i, AnimationUtility.TangentMode.ClampedAuto);
            }
            
            AnimationUtility.SetEditorCurve(clip, binding, curve);
        }
        EditorUtility.SetDirty(clip);
    }
}
```

## 6. 动画 Hierarchy 批量编辑器

```csharp
// 位置：Assets/Scripts/Editor/Animation/AnimationHierarchyEditor.cs
// 功能：当骨架层级发生变更时，批量修正已有 AnimationClip 的绑定路径
// 场景：美术重构了角色骨骼结构，原有所有动画需要更新路径

public class AnimationHierarchyEditor : EditorWindow
{
    [MenuItem("ArtTools/动画层级路径修复")]
    public static void ShowWindow()
    {
        GetWindow<AnimationHierarchyEditor>("动画路径修复");
    }
    
    // 将旧路径批量替换为新路径
    private void ReplaceAllPaths(AnimationClip clip, 
        string oldPath, string newPath)
    {
        var bindings = AnimationUtility.GetCurveBindings(clip);
        foreach (var binding in bindings)
        {
            if (!binding.path.StartsWith(oldPath)) continue;
            
            var newBinding = binding;
            newBinding.path = newPath + binding.path.Substring(oldPath.Length);
            
            var curve = AnimationUtility.GetEditorCurve(clip, binding);
            AnimationUtility.SetEditorCurve(clip, binding, null);  // 删除旧绑定
            AnimationUtility.SetEditorCurve(clip, newBinding, curve);  // 添加新绑定
        }
    }
}
```

## 8. 常见问题

**Q: 战斗动画导出工具运行很慢？**  
A: 因为需要扫描所有 `.anim` 文件并加载到内存。可以通过缓存 clipDict（序列化到 EditorPrefs）加速二次运行。

**Q: Timeline 批量延时后，事件 Signal 的时间也变了吗？**  
A: 是的，Signal Track 也会受到影响。如果只想延迟特效 Track，需要在工具中加 Track 类型过滤。

**Q: 如何在 CI/CD 中使用这些工具？**  
A: 通过 `Unity -batchmode -executeMethod ClassName.MethodName` 调用静态方法。注意 Editor 工具类需要在 `UnityEditor` 命名空间内，不会打包进运行时。
