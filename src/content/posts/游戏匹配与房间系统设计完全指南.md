## 一、匹配系统的核心挑战

1. **公平性**：实力相近的玩家相互匹配
2. **速度**：等待时间不能过长（通常目标<30秒）
3. **质量 vs 速度**：随等待时间延长适当放宽匹配条件
4. **防刷分**：防止玩家人为操纵 MMR
5. **地区优化**：优先匹配同地区玩家，保证低延迟

## 二、ELO/MMR 评分系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// MMR（匹配分）计算系统
/// </summary>
public class MMRCalculator
{
    // ELO K 因子（决定分数变化幅度）
    private const float K_FACTOR_NEW = 32f;     // 新玩家（<30场）
    private const float K_FACTOR_NORMAL = 20f;  // 普通玩家
    private const float K_FACTOR_HIGH = 15f;    // 高分玩家（>2000分）
    
    private const float BASE_MMR = 1500f;       // 新账号初始分
    private const float PLACEMENT_GAMES = 10;   // 定级赛场次

    /// <summary>
    /// 计算 ELO 期望胜率
    /// </summary>
    public float ExpectedWinRate(float playerMMR, float opponentMMR)
    {
        return 1f / (1f + Mathf.Pow(10f, (opponentMMR - playerMMR) / 400f));
    }

    /// <summary>
    /// 根据比赛结果更新 MMR
    /// </summary>
    public MMRUpdateResult UpdateMMR(PlayerMMRData player, float opponentMMR, 
        MatchResult result)
    {
        float kFactor = GetKFactor(player);
        float expected = ExpectedWinRate(player.MMR, opponentMMR);
        float actual = result switch
        {
            MatchResult.Win  => 1f,
            MatchResult.Loss => 0f,
            MatchResult.Draw => 0.5f,
            _ => 0.5f
        };
        
        float delta = kFactor * (actual - expected);
        
        // 防止新号快速上分（定级赛加成）
        if (player.TotalGames < PLACEMENT_GAMES)
            delta *= 2f;
        
        // 防止满分段玩家分数波动过大
        if (player.MMR > 3000 && delta > 0)
            delta *= 0.5f;
        
        float newMMR = Mathf.Max(0, player.MMR + delta);
        
        return new MMRUpdateResult
        {
            OldMMR = player.MMR,
            NewMMR = newMMR,
            Delta = delta,
            KFactor = kFactor
        };
    }

    float GetKFactor(PlayerMMRData player)
    {
        if (player.TotalGames < 30) return K_FACTOR_NEW;
        if (player.MMR > 2000) return K_FACTOR_HIGH;
        return K_FACTOR_NORMAL;
    }

    /// <summary>
    /// 定级赛：根据初始几场结果快速定位 MMR
    /// </summary>
    public float CalculatePlacementMMR(int wins, int losses, float avgOpponentMMR)
    {
        float winRate = (float)wins / (wins + losses);
        // 基于胜率和对手强度估算初始 MMR
        float estimatedMMR = avgOpponentMMR + (winRate - 0.5f) * 400f;
        return Mathf.Clamp(estimatedMMR, 800f, 2200f);
    }
}

[Serializable]
public class PlayerMMRData
{
    public string PlayerId;
    public float MMR;
    public int TotalGames;
    public int Wins;
    public int Losses;
    public float Uncertainty;  // 不确定性（Glicko-2系统使用）
    public string Rank;        // 段位显示（根据MMR映射）
    
    public string GetRankDisplay()
    {
        return MMR switch
        {
            < 800  => "铁",
            < 1000 => "铜",
            < 1200 => "银",
            < 1500 => "金",
            < 1800 => "铂金",
            < 2100 => "钻石",
            < 2400 => "大师",
            _      => "王者"
        };
    }
}

public class MMRUpdateResult
{
    public float OldMMR;
    public float NewMMR;
    public float Delta;
    public float KFactor;
}

public enum MatchResult { Win, Loss, Draw }
```

## 三、多维度匹配系统

```csharp
/// <summary>
/// 匹配请求
/// </summary>
public class MatchmakingRequest
{
    public string PlayerId;
    public float MMR;
    public string Region;           // 地区（cn, us, eu）
    public string GameMode;         // 游戏模式（ranked, normal, aram）
    public float EnqueueTime;       // 入队时间（用于计算等待时长）
    public int[] PreferredLatency;  // 可接受的服务器延迟
    public string[] PartyMembers;   // 组队成员（需要打包匹配）
    
    // 当前可接受的匹配范围（随等待时间扩大）
    public float CurrentMMRTolerance;
    public float MaxMMRTolerance = 500f;
}

/// <summary>
/// 匹配管理器（服务端逻辑，这里演示客户端交互部分）
/// </summary>
public class MatchmakingManager : MonoBehaviour
{
    private static MatchmakingManager instance;
    public static MatchmakingManager Instance => instance;

    [Header("匹配配置")]
    [SerializeField] private float initialMMRTolerance = 100f;
    [SerializeField] private float mmrToleranceGrowthRate = 50f; // 每秒增加的容忍度
    [SerializeField] private float maxWaitTime = 120f;            // 最大等待时间（秒）
    [SerializeField] private float estimatedMatchTime = 15f;      // 预估等待时间

    // 匹配状态
    private MatchmakingState currentState = MatchmakingState.Idle;
    private float queueStartTime;
    private MatchmakingRequest currentRequest;
    
    // 事件
    public event Action<MatchmakingState> OnStateChanged;
    public event Action<MatchFoundData> OnMatchFound;
    public event Action<string> OnMatchmakingFailed;
    public event Action<float, int> OnQueueStatusUpdate; // 等待时间, 在队人数

    public enum MatchmakingState
    {
        Idle,
        Searching,
        MatchFound,
        Connecting,
        Cancelled,
        Failed
    }

    void Update()
    {
        if (currentState != MatchmakingState.Searching) return;
        
        float elapsed = Time.time - queueStartTime;
        
        // 动态扩大匹配范围
        currentRequest.CurrentMMRTolerance = Mathf.Min(
            initialMMRTolerance + elapsed * mmrToleranceGrowthRate,
            currentRequest.MaxMMRTolerance
        );
        
        // 超时降级处理
        if (elapsed > maxWaitTime)
        {
            OnMaxWaitReached();
        }
    }

    /// <summary>
    /// 开始匹配
    /// </summary>
    public void StartMatchmaking(string gameMode)
    {
        if (currentState != MatchmakingState.Idle)
        {
            Debug.LogWarning("[Matchmaking] Already in queue");
            return;
        }
        
        var playerData = PlayerDataService.GetLocalPlayerData();
        
        currentRequest = new MatchmakingRequest
        {
            PlayerId = playerData.PlayerId,
            MMR = playerData.MMR,
            Region = GetPreferredRegion(),
            GameMode = gameMode,
            EnqueueTime = Time.time,
            CurrentMMRTolerance = initialMMRTolerance
        };
        
        queueStartTime = Time.time;
        ChangeState(MatchmakingState.Searching);
        
        // 发送匹配请求到服务器
        NetworkService.SendMatchmakingRequest(currentRequest, OnServerMatchmakingResponse);
        
        // 开始轮询队列状态
        StartCoroutine(PollQueueStatus());
    }

    /// <summary>
    /// 取消匹配
    /// </summary>
    public void CancelMatchmaking()
    {
        if (currentState != MatchmakingState.Searching) return;
        
        StopAllCoroutines();
        NetworkService.CancelMatchmaking(currentRequest.PlayerId);
        ChangeState(MatchmakingState.Cancelled);
        ChangeState(MatchmakingState.Idle);
    }

    System.Collections.IEnumerator PollQueueStatus()
    {
        while (currentState == MatchmakingState.Searching)
        {
            yield return new WaitForSeconds(5f);
            
            // 每5秒查询一次队列状态
            NetworkService.GetQueueStatus(currentRequest.PlayerId, status =>
            {
                if (status != null)
                {
                    OnQueueStatusUpdate?.Invoke(status.EstimatedWaitTime, status.PlayersInQueue);
                }
            });
        }
    }

    void OnServerMatchmakingResponse(MatchmakingResponse response)
    {
        if (response.MatchFound)
        {
            ChangeState(MatchmakingState.MatchFound);
            
            var matchData = new MatchFoundData
            {
                MatchId = response.MatchId,
                ServerAddress = response.ServerAddress,
                ServerPort = response.ServerPort,
                Players = response.Players,
                GameMode = response.GameMode
            };
            
            OnMatchFound?.Invoke(matchData);
            
            // 连接到游戏服务器
            StartCoroutine(ConnectToGameServer(matchData));
        }
        else if (response.Error != null)
        {
            ChangeState(MatchmakingState.Failed);
            OnMatchmakingFailed?.Invoke(response.Error);
        }
    }

    System.Collections.IEnumerator ConnectToGameServer(MatchFoundData data)
    {
        ChangeState(MatchmakingState.Connecting);
        
        Debug.Log($"[Matchmaking] Connecting to {data.ServerAddress}:{data.ServerPort}");
        
        // 连接到专用服务器
        // NetworkManager.singleton.networkAddress = data.ServerAddress;
        // NetworkManager.singleton.StartClient();
        
        yield return new WaitForSeconds(2f); // 模拟连接
        
        // 加载游戏场景
        // SceneManager.LoadScene(data.GameMode);
    }

    void OnMaxWaitReached()
    {
        Debug.Log("[Matchmaking] Max wait time reached, relaxing criteria...");
        
        // 可以选择：
        // 1. 继续等待（放宽 MMR 限制）
        // 2. 加入机器人凑局
        // 3. 通知玩家等待时间过长
        
        UIManager.Instance?.ShowMessage("等待时间较长，正在调整匹配条件...");
    }

    void ChangeState(MatchmakingState newState)
    {
        currentState = newState;
        OnStateChanged?.Invoke(newState);
    }

    string GetPreferredRegion()
    {
        // 根据玩家IP或手动选择确定地区
        return "cn";
    }
    
    public MatchmakingState CurrentState => currentState;
    public float WaitTime => currentState == MatchmakingState.Searching ? 
        Time.time - queueStartTime : 0f;
}

public class MatchFoundData
{
    public string MatchId;
    public string ServerAddress;
    public int ServerPort;
    public List<PlayerMatchInfo> Players;
    public string GameMode;
}

public class PlayerMatchInfo
{
    public string PlayerId;
    public string Nickname;
    public float MMR;
    public int TeamIndex;
}

public class MatchmakingResponse
{
    public bool MatchFound;
    public string MatchId;
    public string ServerAddress;
    public int ServerPort;
    public List<PlayerMatchInfo> Players;
    public string GameMode;
    public string Error;
}

public class QueueStatus
{
    public float EstimatedWaitTime;
    public int PlayersInQueue;
}
```

## 四、房间系统（自定义房间）

```csharp
/// <summary>
/// 游戏房间管理（支持快速匹配与自定义房间并存）
/// </summary>
public class RoomManager : MonoBehaviour
{
    public RoomData CurrentRoom { get; private set; }
    
    // 事件
    public event Action<RoomData> OnRoomCreated;
    public event Action<RoomData> OnRoomJoined;
    public event Action<RoomPlayerData> OnPlayerJoined;
    public event Action<string> OnPlayerLeft;
    public event Action OnRoomDisbanded;
    public event Action OnGameStarted;

    /// <summary>
    /// 创建自定义房间
    /// </summary>
    public void CreateRoom(RoomConfig config)
    {
        // 发送到服务端创建房间
        NetworkService.CreateRoom(config, response =>
        {
            if (response.Success)
            {
                CurrentRoom = response.Room;
                OnRoomCreated?.Invoke(CurrentRoom);
            }
        });
    }

    /// <summary>
    /// 通过房间码加入
    /// </summary>
    public void JoinRoomByCode(string roomCode)
    {
        NetworkService.JoinRoom(roomCode, response =>
        {
            if (response.Success)
            {
                CurrentRoom = response.Room;
                OnRoomJoined?.Invoke(CurrentRoom);
            }
            else
            {
                UIManager.Instance?.ShowError($"加入失败: {response.Error}");
            }
        });
    }

    public void SetReady(bool ready)
    {
        if (CurrentRoom == null) return;
        NetworkService.SetReady(CurrentRoom.RoomId, ready);
    }

    public void StartGame()
    {
        // 仅房主可以开始游戏
        if (CurrentRoom == null || !CurrentRoom.IsLocalPlayerHost) return;
        NetworkService.StartRoom(CurrentRoom.RoomId);
    }
}

[Serializable]
public class RoomConfig
{
    public string GameMode;
    public int MaxPlayers;
    public bool IsPrivate;
    public string Password;
    public string MapId;
    public Dictionary<string, string> CustomSettings;
}

[Serializable]
public class RoomData
{
    public string RoomId;
    public string RoomCode;         // 4-6位邀请码
    public string HostPlayerId;
    public List<RoomPlayerData> Players;
    public RoomConfig Config;
    public RoomState State;
    public bool IsLocalPlayerHost;
}

[Serializable]
public class RoomPlayerData
{
    public string PlayerId;
    public string Nickname;
    public bool IsReady;
    public bool IsHost;
    public int TeamIndex;
    public float MMR;
}

public enum RoomState { Waiting, Starting, InGame, Ended }
```

## 五、工程方案选型

| 方案 | 优点 | 缺点 | 适用场景 |
|------|------|------|----------|
| Mirror (开源) | 完全可控、免费 | 需要自建服务器 | 独立游戏、中小型项目 |
| Photon PUN/Fusion | 开箱即用、有托管服务 | 收费、有依赖 | 快速原型、中型项目 |
| PlayFab Multiplayer | 完整后端服务 | 复杂、收费 | 商业项目 |
| 自研 | 完全定制 | 开发成本高 | 大型商业MMORPG |

匹配系统是多人游戏的核心体验，其设计质量直接影响玩家留存。建议早期使用成熟的第三方服务（Photon/PlayFab），待验证商业价值后再考虑自研优化。
