## 二、存档管理器

```csharp
using System;
using System.IO;
using System.Security.Cryptography;
using System.Text;
using UnityEngine;

/// <summary>
/// 存档管理器
/// </summary>
public class SaveManager : MonoBehaviour
{
    private static SaveManager instance;
    public static SaveManager Instance => instance;

    [Header("存档配置")]
    [SerializeField] private int maxSaveSlots = 3;
    [SerializeField] private bool enableEncryption = true;
    [SerializeField] private bool enableAutoSave = true;
    [SerializeField] private float autoSaveInterval = 300f; // 5分钟自动保存

    private const string ENCRYPTION_KEY = "YourSecretKey123"; // 实际项目请用更安全的密钥管理
    private const int CURRENT_VERSION = 2;
    
    private float autoSaveTimer;
    private string SaveDirectory => Path.Combine(Application.persistentDataPath, "Saves");

    void Awake()
    {
        instance = this;
        Directory.CreateDirectory(SaveDirectory);
    }

    void Update()
    {
        if (!enableAutoSave) return;
        
        autoSaveTimer += Time.deltaTime;
        if (autoSaveTimer >= autoSaveInterval)
        {
            autoSaveTimer = 0;
            AutoSave();
        }
    }

    // ============ 保存 ============

    public bool Save(int slotIndex = 0)
    {
        try
        {
            var data = CollectSaveData();
            data.SaveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
            data.Version = CURRENT_VERSION;
            
            // 计算校验和
            string json = JsonUtility.ToJson(data);
            data.Checksum = ComputeChecksum(json);
            json = JsonUtility.ToJson(data); // 含校验和的最终JSON
            
            // 加密
            byte[] bytes = enableEncryption ? Encrypt(json) : Encoding.UTF8.GetBytes(json);
            
            // 保存前备份旧存档
            BackupSlot(slotIndex);
            
            string path = GetSavePath(slotIndex);
            File.WriteAllBytes(path, bytes);
            
            Debug.Log($"[Save] 存档 Slot{slotIndex} 保存成功");
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Save] 保存失败: {e.Message}");
            return false;
        }
    }

    void AutoSave()
    {
        // 自动存档到专用槽（不占用手动存档槽）
        Save(-1); // -1 = 自动存档槽
        Debug.Log("[AutoSave] 自动存档完成");
    }

    // ============ 读取 ============

    public GameSaveData Load(int slotIndex = 0)
    {
        string path = GetSavePath(slotIndex);
        
        if (!File.Exists(path))
        {
            Debug.Log($"[Save] Slot{slotIndex} 无存档");
            return null;
        }
        
        try
        {
            byte[] bytes = File.ReadAllBytes(path);
            string json = enableEncryption ? Decrypt(bytes) : Encoding.UTF8.GetString(bytes);
            
            var data = JsonUtility.FromJson<GameSaveData>(json);
            
            // 校验完整性
            if (!VerifyChecksum(data))
            {
                Debug.LogWarning("[Save] 存档校验失败，尝试恢复备份");
                return LoadBackup(slotIndex);
            }
            
            // 版本迁移
            data = MigrateToCurrentVersion(data);
            
            return data;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Save] 读取失败: {e.Message}，尝试恢复备份");
            return LoadBackup(slotIndex);
        }
    }

    // ============ 版本迁移 ============

    GameSaveData MigrateToCurrentVersion(GameSaveData data)
    {
        while (data.Version < CURRENT_VERSION)
        {
            data = MigrateOneVersion(data);
        }
        return data;
    }

    GameSaveData MigrateOneVersion(GameSaveData data)
    {
        switch (data.Version)
        {
            case 1:
                // v1 → v2：新增 Stats 字段
                if (data.Stats == null)
                    data.Stats = new GameStatsSaveData();
                data.Version = 2;
                break;
        }
        return data;
    }

    // ============ 加密/解密 ============

    byte[] Encrypt(string plainText)
    {
        using var aes = Aes.Create();
        var key = PadKey(ENCRYPTION_KEY);
        aes.Key = key;
        aes.GenerateIV();
        
        using var encryptor = aes.CreateEncryptor();
        byte[] plainBytes = Encoding.UTF8.GetBytes(plainText);
        byte[] encrypted = encryptor.TransformFinalBlock(plainBytes, 0, plainBytes.Length);
        
        // IV + 密文
        byte[] result = new byte[aes.IV.Length + encrypted.Length];
        Buffer.BlockCopy(aes.IV, 0, result, 0, aes.IV.Length);
        Buffer.BlockCopy(encrypted, 0, result, aes.IV.Length, encrypted.Length);
        return result;
    }

    string Decrypt(byte[] cipherBytes)
    {
        using var aes = Aes.Create();
        aes.Key = PadKey(ENCRYPTION_KEY);
        
        byte[] iv = new byte[16];
        byte[] cipher = new byte[cipherBytes.Length - 16];
        Buffer.BlockCopy(cipherBytes, 0, iv, 0, 16);
        Buffer.BlockCopy(cipherBytes, 16, cipher, 0, cipher.Length);
        aes.IV = iv;
        
        using var decryptor = aes.CreateDecryptor();
        byte[] plain = decryptor.TransformFinalBlock(cipher, 0, cipher.Length);
        return Encoding.UTF8.GetString(plain);
    }

    byte[] PadKey(string key)
    {
        byte[] keyBytes = Encoding.UTF8.GetBytes(key);
        Array.Resize(ref keyBytes, 16); // AES-128
        return keyBytes;
    }

    string ComputeChecksum(string data)
    {
        using var sha = SHA256.Create();
        byte[] hash = sha.ComputeHash(Encoding.UTF8.GetBytes(data + ENCRYPTION_KEY));
        return Convert.ToBase64String(hash).Substring(0, 16);
    }

    bool VerifyChecksum(GameSaveData data)
    {
        string savedChecksum = data.Checksum;
        data.Checksum = "";
        string json = JsonUtility.ToJson(data);
        string computedChecksum = ComputeChecksum(json);
        data.Checksum = savedChecksum;
        return savedChecksum == computedChecksum;
    }

    // ============ 工具方法 ============

    string GetSavePath(int slot) =>
        Path.Combine(SaveDirectory, slot == -1 ? "autosave.sav" : $"save_{slot}.sav");

    void BackupSlot(int slot)
    {
        string src = GetSavePath(slot);
        if (!File.Exists(src)) return;
        string backup = src + ".bak";
        File.Copy(src, backup, overwrite: true);
    }

    GameSaveData LoadBackup(int slot)
    {
        string backup = GetSavePath(slot) + ".bak";
        if (!File.Exists(backup)) return null;
        
        try
        {
            byte[] bytes = File.ReadAllBytes(backup);
            string json = enableEncryption ? Decrypt(bytes) : Encoding.UTF8.GetString(bytes);
            return JsonUtility.FromJson<GameSaveData>(json);
        }
        catch
        {
            return null;
        }
    }

    GameSaveData CollectSaveData()
    {
        // 从各系统收集存档数据
        return new GameSaveData
        {
            PlayerId = PlayerDataService.GetLocalPlayerData()?.PlayerId ?? "",
            Player = CollectPlayerData(),
            World = CollectWorldData(),
            Quests = CollectQuestData(),
        };
    }

    PlayerSaveData CollectPlayerData() => new PlayerSaveData { Level = 1 };
    WorldSaveData CollectWorldData() => new WorldSaveData();
    QuestSaveData CollectQuestData() => new QuestSaveData();
}
```

## 三、云端存档同步

```csharp
/// <summary>
/// 云端存档同步（上传/下载/冲突解决）
/// </summary>
public class CloudSaveSync : MonoBehaviour
{
    public enum ConflictResolution { UseLocal, UseCloud, UseNewer }
    
    [SerializeField] private ConflictResolution conflictStrategy = ConflictResolution.UseNewer;

    public async System.Threading.Tasks.Task SyncWithCloud(int slotIndex)
    {
        var localData = SaveManager.Instance.Load(slotIndex);
        
        // 下载云端存档
        GameSaveData cloudData = await DownloadCloudSave(slotIndex);
        
        if (cloudData == null)
        {
            // 云端无存档，直接上传本地
            if (localData != null)
                await UploadToCloud(slotIndex, localData);
            return;
        }
        
        if (localData == null)
        {
            // 本地无存档，使用云端
            ApplyCloudSave(cloudData);
            return;
        }
        
        // 冲突解决
        GameSaveData winner = ResolveConflict(localData, cloudData);
        ApplyCloudSave(winner);
        await UploadToCloud(slotIndex, winner);
    }

    GameSaveData ResolveConflict(GameSaveData local, GameSaveData cloud)
    {
        switch (conflictStrategy)
        {
            case ConflictResolution.UseLocal:  return local;
            case ConflictResolution.UseCloud:  return cloud;
            case ConflictResolution.UseNewer:
                return local.SaveTimestamp >= cloud.SaveTimestamp ? local : cloud;
            default: return local;
        }
    }

    async System.Threading.Tasks.Task<GameSaveData> DownloadCloudSave(int slot)
    {
        // 实现云端下载（Firebase/自建服务器/平台云存档）

        return null;
    }

    async System.Threading.Tasks.Task UploadToCloud(int slot, GameSaveData data)
    {
        await System.Threading.Tasks.Task.Delay(100);
    }

    void ApplyCloudSave(GameSaveData data)
    {

    }
```

## 四、存档系统设计要点

| 要点 | 方案 |
|------|------|
| 防篡改 | SHA256校验和 + AES加密 |
| 版本兼容 | 版本号字段 + 逐版本迁移函数 |
| 备份恢复 | 每次保存前备份.bak |
| 自动存档 | 定时 + 关键节点触发 |
| 云端冲突 | 时间戳比较取最新 |
| 存档损坏 | 校验失败自动尝试备份 |
