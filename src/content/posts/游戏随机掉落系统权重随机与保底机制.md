## 一、权重随机系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 掉落物品配置
/// </summary>
[Serializable]
public class LootEntry
{
    public string ItemId;
    public int MinCount;
    public int MaxCount;
    public float BaseWeight;   // 基础权重（越高越容易掉）
    public ItemRarity Rarity;
    public bool IsGuaranteed; // 是否保底
}

/// <summary>
/// 掉落表（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "LootTable", menuName = "Game/Loot Table")]
public class LootTable : ScriptableObject
{
    public string TableId;
    public int RollCount = 1;                    // 一次开几次
    public LootEntry[] Entries;
    
    [Header("保底配置")]
    public bool HasGuarantee;                    // 是否有保底
    public int GuaranteeRolls = 90;              // 几抽保底
    public ItemRarity GuaranteeMinRarity = ItemRarity.Rare;
}

/// <summary>
/// 权重随机抽取器
/// </summary>
public class WeightedRandomSelector<T>
{
    private List<T> items = new List<T>();
    private List<float> weights = new List<float>();
    private float totalWeight;
    
    // 预计算的累积权重（用于O(logn)二分查找）
    private float[] cumulativeWeights;
    private bool isDirty = true;

    public void Add(T item, float weight)
    {
        items.Add(item);
        weights.Add(weight);
        totalWeight += weight;
        isDirty = true;
    }

    public void Clear()
    {
        items.Clear();
        weights.Clear();
        totalWeight = 0;
        isDirty = true;
    }

    /// <summary>
    /// 随机抽取一个（O(logn)二分查找）
    /// </summary>
    public T Select()
    {
        if (items.Count == 0) throw new InvalidOperationException("没有可选的项目");
        
        if (isDirty) BuildCumulative();
        
        float random = UnityEngine.Random.Range(0f, totalWeight);
        
        // 二分查找
        int lo = 0, hi = cumulativeWeights.Length - 1;
        while (lo < hi)
        {
            int mid = (lo + hi) / 2;
            if (cumulativeWeights[mid] < random) lo = mid + 1;
            else hi = mid;
        }
        
        return items[lo];
    }

    /// <summary>
    /// 随机抽取多个（不重复）
    /// </summary>
    public List<T> SelectMultiple(int count)
    {
        count = Mathf.Min(count, items.Count);
        var selected = new List<T>();
        
        // 复制一份用于临时抽取（抽过的权重清0）
        var tempWeights = new float[weights.Count];
        weights.CopyTo(tempWeights);
        float tempTotal = totalWeight;
        
        for (int i = 0; i < count; i++)
        {
            float random = UnityEngine.Random.Range(0f, tempTotal);
            float cumulative = 0;
            
            for (int j = 0; j < items.Count; j++)
            {
                cumulative += tempWeights[j];
                if (random <= cumulative && tempWeights[j] > 0)
                {
                    selected.Add(items[j]);
                    tempTotal -= tempWeights[j];
                    tempWeights[j] = 0; // 已选，下次权重为0
                    break;
                }
            }
        }
        
        return selected;
    }

    void BuildCumulative()
    {
        cumulativeWeights = new float[weights.Count];
        float cumulative = 0;
        for (int i = 0; i < weights.Count; i++)
        {
            cumulative += weights[i];
            cumulativeWeights[i] = cumulative;
        }
        isDirty = false;
    }
}
```

## 二、掉落管理器（含保底）

```csharp
/// <summary>
/// 掉落管理器（包含保底机制）
/// </summary>
public class LootManager : MonoBehaviour
{
    private static LootManager instance;
    public static LootManager Instance => instance;

    // 玩家保底计数器（持久化存储）
    private Dictionary<string, int> guaranteeCounters = new Dictionary<string, int>();
    
    // 掉落历史（用于统计）
    private Dictionary<string, List<string>> lootHistory = new Dictionary<string, List<string>>();

    void Awake() { instance = this; LoadGuaranteeCounters(); }

    /// <summary>
    /// 执行掉落
    /// </summary>
    public List<LootResult> Roll(LootTable table, string playerId)
    {
        var results = new List<LootResult>();
        
        for (int roll = 0; roll < table.RollCount; roll++)
        {
            var result = RollOnce(table, playerId);
            if (result != null) results.Add(result);
        }
        
        // 记录掉落历史
        RecordHistory(table.TableId, results, playerId);
        
        // 持久化保底计数
        SaveGuaranteeCounters();
        
        return results;
    }

    LootResult RollOnce(LootTable table, string playerId)
    {
        string counterKey = $"{playerId}_{table.TableId}";
        
        if (!guaranteeCounters.ContainsKey(counterKey))
            guaranteeCounters[counterKey] = 0;
        
        guaranteeCounters[counterKey]++;
        int currentCounter = guaranteeCounters[counterKey];
        
        // 检查是否触发保底
        bool isGuaranteed = table.HasGuarantee && currentCounter >= table.GuaranteeRolls;
        
        // 构建权重选择器
        var selector = new WeightedRandomSelector<LootEntry>();
        
        foreach (var entry in table.Entries)
        {
            float weight = entry.BaseWeight;
            
            // 保底时，低稀有度权重降为0
            if (isGuaranteed && entry.Rarity < table.GuaranteeMinRarity)
                weight = 0;
            
            if (weight > 0)
                selector.Add(entry, weight);
        }
        
        var selected = selector.Select();
        
        // 抽到稀有以上重置保底计数
        if (selected.Rarity >= table.GuaranteeMinRarity)
            guaranteeCounters[counterKey] = 0;
        
        int count = UnityEngine.Random.Range(selected.MinCount, selected.MaxCount + 1);
        
        return new LootResult
        {
            ItemId = selected.ItemId,
            Count = count,
            Rarity = selected.Rarity,
            IsGuaranteed = isGuaranteed,
            RollsToGuarantee = table.HasGuarantee ? 
                table.GuaranteeRolls - guaranteeCounters[counterKey] : -1
        };
    }

    public int GetGuaranteeProgress(string tableId, string playerId)
    {
        string key = $"{playerId}_{tableId}";
        return guaranteeCounters.TryGetValue(key, out int count) ? count : 0;
    }

    void RecordHistory(string tableId, List<LootResult> results, string playerId)
    {
        if (!lootHistory.ContainsKey(playerId))
            lootHistory[playerId] = new List<string>();
        
        foreach (var result in results)
            lootHistory[playerId].Add($"{tableId}:{result.ItemId}");
    }

    void SaveGuaranteeCounters()
    {
        foreach (var kv in guaranteeCounters)
            PlayerPrefs.SetInt($"guarantee_{kv.Key}", kv.Value);
        PlayerPrefs.Save();
    }

    void LoadGuaranteeCounters()
    {
        // 实际项目从服务端加载保底状态，防止客户端篡改
    }
}

[Serializable]
public class LootResult
{
    public string ItemId;
    public int Count;
    public ItemRarity Rarity;
    public bool IsGuaranteed;
    public int RollsToGuarantee; // 还需多少抽到保底（-1=无保底）
}
```

## 三、保底系统示例（类天井）

```
标准抽卡保底示例（1-90抽）：

基础概率：
  SSR: 0.6%
  SR:  5.1%
  R:   94.3%

软保底（从第76抽开始）：
  第76抽：SSR概率 = 0.6% + (76-75) × 6% = 6.6%
  第77抽：SSR概率 = 12.6%
  ...
  第89抽：SSR概率 = 90.6%（接近100%）

硬保底（第90抽）：
  第90抽：强制出SSR，100%概率

实现代码：
  float baseRate = 0.006f;
  if (rollCount >= 75)
  {
      float bonus = (rollCount - 74) * 0.06f;
      currentRate = baseRate + bonus;
  }
  if (rollCount >= 90) currentRate = 1.0f; // 强制保底
```

## 四、掉落系统设计要点

| 要点 | 方案 |
|------|------|
| 权重数据化 | ScriptableObject配置，美术/策划可调整 |
| 保底服务端存储 | 保底计数必须存服务端，防止清缓存重置 |
| 概率展示 | 界面显示各物品掉落概率（法规要求）|
| 掉落历史 | 可查看历史记录，提升透明度 |
| 预览系统 | 开箱前可查看可能掉落的物品列表 |
| 防客户端作弊 | 掉落结果由服务端计算，客户端只展示 |
