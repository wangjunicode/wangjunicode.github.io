## 一、登录流程

```csharp
using System;
using System.Threading.Tasks;
using UnityEngine;

public enum LoginType { Guest, Phone, WeChat, Apple, Google }

[Serializable]
public class LoginResult
{
    public bool Success;
    public string PlayerId;
    public string AccessToken;
    public string RefreshToken;
    public float ExpiresIn;
    public bool IsNewPlayer;
    public string ErrorCode;
    public string ErrorMessage;
}

public class AccountManager : MonoBehaviour
{
    private static AccountManager instance;
    public static AccountManager Instance => instance;
    
    private string currentPlayerId;
    private LoginType currentLoginType;
    
    public event Action<LoginResult> OnLoginComplete;
    public event Action OnLogout;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    /// <summary>
    /// 游客登录（快速进入，无注册门槛）
    /// </summary>
    public async Task<LoginResult> LoginAsGuest()
    {
        string deviceId = SystemInfo.deviceUniqueIdentifier;
        string savedGuestId = PlayerPrefs.GetString("guest_id", "");
        
        var result = await NetworkService.LoginGuest(deviceId, savedGuestId);
        if (result.Success)
        {
            if (string.IsNullOrEmpty(savedGuestId))
                PlayerPrefs.SetString("guest_id", result.PlayerId);
            
            OnLoginSuccess(result, LoginType.Guest);
        }
        
        OnLoginComplete?.Invoke(result);
        return result;
    }

    /// <summary>
    /// 手机号登录
    /// </summary>
    public async Task<LoginResult> LoginWithPhone(string phone, string verifyCode)
    {
        var result = await NetworkService.LoginPhone(phone, verifyCode);
        if (result.Success)
            OnLoginSuccess(result, LoginType.Phone);
        
        OnLoginComplete?.Invoke(result);
        return result;
    }

    /// <summary>
    /// 微信登录
    /// </summary>
    public async Task<LoginResult> LoginWithWeChat()
    {
        string wechatCode = await WeChatSDK.GetAuthCode();
        if (string.IsNullOrEmpty(wechatCode)) 
            return new LoginResult { Success = false, ErrorCode = "WECHAT_CANCELLED" };
        
        var result = await NetworkService.LoginWeChat(wechatCode);
        if (result.Success)
            OnLoginSuccess(result, LoginType.WeChat);
        
        OnLoginComplete?.Invoke(result);
        return result;
    }

    void OnLoginSuccess(LoginResult result, LoginType loginType)
    {
        currentPlayerId = result.PlayerId;
        currentLoginType = loginType;
        
        // 存储Token
        HttpClient.Instance?.SetTokens(result.AccessToken, result.RefreshToken, result.ExpiresIn);
        
        // 本地保存登录状态
        PlayerPrefs.SetString("last_login_type", loginType.ToString());
        PlayerPrefs.SetString("last_player_id", result.PlayerId);
        
        Debug.Log($"[Account] 登录成功: {result.PlayerId} via {loginType}");
    }

    /// <summary>
    /// 游客账号绑定手机号（升级正式账号）
    /// </summary>
    public async Task<bool> BindPhone(string phone, string verifyCode)
    {
        if (currentLoginType != LoginType.Guest) return false;
        
        bool success = await NetworkService.BindPhone(currentPlayerId, phone, verifyCode);
        if (success)
        {
            currentLoginType = LoginType.Phone;
            UIManager.Instance?.ShowToast("绑定成功！游客账号已升级");
        }
        return success;
    }

    public bool IsGuest() => currentLoginType == LoginType.Guest;
    public string GetPlayerId() => currentPlayerId;
    
    public void Logout()
    {
        currentPlayerId = null;
        PlayerPrefs.DeleteKey("last_login_type");
        HttpClient.Instance?.SetTokens("", "", 0);
        OnLogout?.Invoke();
    }
}
```

## 二、异地登录检测

```csharp
public class SessionSecurityManager : MonoBehaviour
{
    private string sessionId;
    private string loginDeviceId;

    void Start()
    {
        // 订阅服务端推送的安全事件
        NetworkService.OnMessageReceived += HandleSecurityMessage;
    }

    void HandleSecurityMessage(string type, string payload)
    {
        switch (type)
        {
            case "session_revoked":
                // 账号在其他设备登录，当前Session被踢
                ShowForceLogoutDialog("您的账号在其他设备登录，当前设备已退出");
                break;
            case "suspicious_login":
                // 异常登录提醒
                UIManager.Instance?.ShowDialog("安全提醒", "检测到异地登录，是否为本人操作？");
                break;
            case "account_banned":
                var banInfo = JsonUtility.FromJson<BanInfo>(payload);
                ShowForceLogoutDialog($"账号已被封禁至{banInfo.UnbanTime}\n原因：{banInfo.Reason}");
                break;
        }
    }

    void ShowForceLogoutDialog(string message)
    {
        UIManager.Instance?.ShowDialog("账号异常", message, () =>
        {
            AccountManager.Instance?.Logout();
            // 跳转登录页
        });
    }
}

[Serializable] class BanInfo { public string Reason; public string UnbanTime; }
```

## 三、账号系统设计规范

| 要点 | 方案 |
|------|------|
| 游客保留 | 游客数据至少保留30天 |
| 升级绑定 | 绑定后游客数据迁移到正式账号 |
| Token短期 | AccessToken 2小时有效 |
| RefreshToken | 7天有效，用于续期 |
| 多设备 | 最多N台设备同时在线 |
| 封禁 | 服务端封禁，客户端被动响应 |
