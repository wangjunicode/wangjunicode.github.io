# 游戏配置表系统设计：从Excel到运行时数据

配置表是策划与程序协作的核心工具。一套设计良好的配置表系统，能让策划在不修改代码的情况下调整 90% 的游戏数值，同时让程序员以类型安全的方式访问数据。

## 一、配置表体系架构

```
配置表工作流：

策划编辑
  ↓
Excel 文件（可读性强）
  ↓ 导出工具（Python/C#脚本）
JSON/Binary 数据文件
  ↓ + 自动生成
强类型 C# 数据类
  ↓
运行时数据管理器（查询/缓存）
  ↓
游戏逻辑调用
```

## 二、Excel 配置表规范

### 2.1 表头设计规范

```
技能配置表（SkillConfig.xlsx）：

行1（字段名）：id | name | mpCost | cooldown | damage | buffId | description
行2（类型）：  int | string | float | float | float | string | string
行3（描述）：  技能ID | 技能名称 | MP消耗 | 冷却时间(秒) | 伤害系数 | 施加BuffID | 描述
行4（数据）：  1001 | 斩击 | 0 | 0.5 | 2.0 | - | 挥剑斩击...
行5（数据）：  1002 | 火球 | 30 | 3.0 | 3.0 | burn | 发射火球...

规范：
- 第1行：程序字段名（英文，camelCase）
- 第2行：数据类型（int/float/string/bool/json）
- 第3行：中文描述（方便策划理解）
- 第4行起：实际数据
- 注释行：以 # 开头（导出时忽略）
```

### 2.2 多级数据设计

```
复杂数据用 JSON 格式内嵌：

技能效果字段（effectJson）：
  单效果：{"type":"damage","value":2.0}
  多效果：[{"type":"damage","value":2.0},{"type":"buff","id":"burn","duration":3}]

装备属性字段（attrs）：
  {"atk":100,"def":50,"hp":200,"crit":0.05}

掉落配置（drops）：
  [{"itemId":1001,"weight":100},{"itemId":1002,"weight":30,"count":"2~5"}]
```

## 三、配置表导出工具

### 3.1 Python 导出脚本

```python
#!/usr/bin/env python3
# excel_exporter.py

import openpyxl
import json
import os
from typing import Dict, Any

class ExcelExporter:
    TYPE_MAP = {
        'int': int,
        'float': float,
        'string': str,
        'bool': bool,
        'json': json.loads,
        'intlist': lambda s: [int(x) for x in s.split('|')],
        'stringlist': lambda s: s.split('|'),
    }
    
    def export_sheet(self, ws) -> list:
        """导出单个Sheet"""
        rows = list(ws.iter_rows(values_only=True))
        
        if len(rows) < 4:
            return []
        
        field_names = rows[0]   # 字段名
        field_types = rows[1]   # 类型
        # rows[2] 是描述行，跳过
        
        results = []
        
        for row in rows[3:]:
            if row[0] is None:  # 空行跳过
                continue
            if str(row[0]).startswith('#'):  # 注释行跳过
                continue
            
            record = {}
            for i, (name, type_str, value) in enumerate(zip(field_names, field_types, row)):
                if name is None or type_str is None:
                    continue
                
                converter = self.TYPE_MAP.get(str(type_str).lower(), str)
                
                try:
                    if value is None or value == '-' or value == '':
                        record[name] = self.get_default_value(str(type_str))
                    else:
                        record[name] = converter(value)
                except Exception as e:
                    print(f"转换失败: 字段={name}, 类型={type_str}, 值={value}: {e}")
                    record[name] = value
            
            results.append(record)
        
        return results
    
    def get_default_value(self, type_str: str):
        defaults = {
            'int': 0, 'float': 0.0, 'string': '', 
            'bool': False, 'json': None,
            'intlist': [], 'stringlist': []
        }
        return defaults.get(type_str.lower(), None)
    
    def export_workbook(self, excel_path: str, output_dir: str):
        wb = openpyxl.load_workbook(excel_path, data_only=True)
        
        for sheet_name in wb.sheetnames:
            if sheet_name.startswith('_'):  # _ 开头的 Sheet 跳过
                continue
            
            ws = wb[sheet_name]
            data = self.export_sheet(ws)
            
            if not data:
                continue
            
            # 输出 JSON
            output_path = os.path.join(output_dir, f"{sheet_name}.json")
            with open(output_path, 'w', encoding='utf-8') as f:
                json.dump(data, f, ensure_ascii=False, indent=2)
            
            print(f"✅ 导出 {sheet_name}: {len(data)} 条记录 → {output_path}")

# 运行
if __name__ == '__main__':
    exporter = ExcelExporter()
    exporter.export_workbook(
        excel_path='../Design/GameConfig.xlsx',
        output_dir='../Assets/Resources/Configs/'
    )
```

### 3.2 C# 代码自动生成

```python
# code_generator.py：根据表头自动生成 C# 数据类

def generate_csharp_class(sheet_name: str, fields: list, types: list) -> str:
    lines = [
        "// Auto-generated by ExcelExporter. DO NOT EDIT!",
        "using System;",
        "using System.Collections.Generic;",
        "",
        "[Serializable]",
        f"public class {sheet_name}",
        "{",
    ]
    
    csharp_type_map = {
        'int': 'int',
        'float': 'float',
        'string': 'string',
        'bool': 'bool',
        'json': 'string',  # JSON 字段保留为 string，运行时按需解析
        'intlist': 'int[]',
        'stringlist': 'string[]',
    }
    
    for name, type_str in zip(fields, types):
        if name and type_str:
            cs_type = csharp_type_map.get(str(type_str).lower(), 'string')
            lines.append(f"    public {cs_type} {name};")
    
    lines.append("}")
    return '\n'.join(lines)
```

## 四、运行时配置管理器

```csharp
// 配置表管理器
public class ConfigManager : MonoBehaviour
{
    public static ConfigManager Instance { get; private set; }
    
    private Dictionary<Type, object> _configDicts = new();
    
    private void Awake()
    {
        Instance = this;
        // 预加载所有配置
        LoadAllConfigs();
    }
    
    private async void LoadAllConfigs()
    {
        // 从 Addressables 或 Resources 加载
        await LoadConfig<SkillConfig>("Configs/SkillConfig");
        await LoadConfig<ItemConfig>("Configs/ItemConfig");
        await LoadConfig<EnemyConfig>("Configs/EnemyConfig");
        await LoadConfig<LevelConfig>("Configs/LevelConfig");
        
        Debug.Log("所有配置表加载完成");
    }
    
    private async Task LoadConfig<T>(string path) where T : class
    {
        var json = await LoadJsonAsync(path);
        var list = JsonConvert.DeserializeObject<List<T>>(json);
        
        // 建立以 ID 为 Key 的字典（O(1) 查询）
        var dict = new Dictionary<int, T>();
        
        foreach (var item in list)
        {
            // 使用反射获取 id 字段
            var idProperty = typeof(T).GetField("id");
            int id = (int)(idProperty?.GetValue(item) ?? 0);
            dict[id] = item;
        }
        
        _configDicts[typeof(T)] = dict;
    }
    
    // 按 ID 查询
    public T Get<T>(int id) where T : class
    {
        if (_configDicts.TryGetValue(typeof(T), out var dictObj))
        {
            var dict = (Dictionary<int, T>)dictObj;
            return dict.TryGetValue(id, out var config) ? config : null;
        }
        return null;
    }
    
    // 获取所有配置
    public IEnumerable<T> GetAll<T>() where T : class
    {
        if (_configDicts.TryGetValue(typeof(T), out var dictObj))
        {
            return ((Dictionary<int, T>)dictObj).Values;
        }
        return Enumerable.Empty<T>();
    }
}

// 使用示例
public class SkillSystem
{
    public void UseSkill(int skillId, CombatEntity caster)
    {
        // O(1) 查询
        var config = ConfigManager.Instance.Get<SkillConfig>(skillId);
        if (config == null)
        {
            Debug.LogError($"找不到技能配置: {skillId}");
            return;
        }
        
        Debug.Log($"使用技能: {config.name}，MP消耗: {config.mpCost}");
    }
}
```

## 五、配置数据验证

```csharp
// 配置验证器：在导出/加载时检查数据合法性
public class ConfigValidator
{
    public static List<string> ValidateSkillConfigs(List<SkillConfig> configs)
    {
        var errors = new List<string>();
        var idSet = new HashSet<int>();
        
        foreach (var config in configs)
        {
            // 检查 ID 唯一性
            if (!idSet.Add(config.id))
                errors.Add($"技能 ID {config.id} 重复");
            
            // 检查数值范围
            if (config.mpCost < 0)
                errors.Add($"技能 {config.id}: MP消耗不能为负数 ({config.mpCost})");
            
            if (config.cooldown < 0)
                errors.Add($"技能 {config.id}: 冷却时间不能为负数");
            
            if (config.damage < 0 || config.damage > 100)
                errors.Add($"技能 {config.id}: 伤害系数 {config.damage} 超出合理范围 [0, 100]");
            
            // 检查引用完整性
            if (!string.IsNullOrEmpty(config.buffId))
            {
                var buffConfig = ConfigManager.Instance.Get<BuffConfig>(config.buffId);
                if (buffConfig == null)
                    errors.Add($"技能 {config.id}: 引用的 Buff ID '{config.buffId}' 不存在");
            }
            
            // 检查必填字段
            if (string.IsNullOrEmpty(config.name))
                errors.Add($"技能 {config.id}: 名称为空");
        }
        
        return errors;
    }
}

// Unity Editor 验证工具
#if UNITY_EDITOR
[MenuItem("Tools/Config/Validate All")]
```

## 六、多语言配置

```csharp
// 本地化配置表
// LocalizationConfig.json：
// {"key": "skill_slash_name", "zh-cn": "斩击", "en": "Slash", "ja": "斬撃"}

public class LocalizationManager
{
    private Dictionary<string, string> _currentLocale = new();
    
    public async Task LoadLocale(string language)
    {
        var json = await LoadJsonAsync($"Configs/Localization_{language}");
        var entries = JsonConvert.DeserializeObject<List<LocalizationEntry>>(json);
        
        _currentLocale.Clear();
        foreach (var entry in entries)
        {
            _currentLocale[entry.key] = entry.value;
        }
    }
    
    public string GetText(string key, params object[] args)
    {
        if (_currentLocale.TryGetValue(key, out var template))
        {
            return args.Length > 0 ? string.Format(template, args) : template;
        }
        
        // 找不到时返回 key 本身，方便调试
        Debug.LogWarning($"找不到本地化Key: {key}");
        return $"[MISSING: {key}]";
    }
}
```

## 七、热重载配置（开发期效率）

```csharp
// 开发期间热重载：修改 Excel 后无需重启游戏
#if UNITY_EDITOR
public class ConfigHotReloader : MonoBehaviour
{
    private FileSystemWatcher _watcher;
    
    private void Start()
    {
        _watcher = new FileSystemWatcher(Application.streamingAssetsPath + "/Configs/");
        _watcher.Filter = "*.json";
        _watcher.Changed += OnConfigFileChanged;
        _watcher.EnableRaisingEvents = true;
    }
    
    private void OnConfigFileChanged(object sender, FileSystemEventArgs e)
    {
        // 在主线程上重新加载（FileSystemWatcher 回调在子线程）
        UnityMainThreadDispatcher.Instance.Enqueue(async () =>
        {
            string configName = Path.GetFileNameWithoutExtension(e.Name);
            await ConfigManager.Instance.ReloadConfig(configName);
            Debug.Log($"⚡ 热重载配置: {configName}");
        });
    }
    
    private void OnDestroy()
    {
        _watcher?.Dispose();
    }
}
#endif
```

## 八、配置表最佳实践

| 实践 | 说明 |
|------|------|
| ID 管理 | 每类配置有独立 ID 段（技能1000+，道具2000+，敌人3000+）|
| 版本控制 | 配置 JSON 纳入 Git，便于 diff 和回滚 |
| 数据验证 | CI 流水线中运行验证，避免错误配置上线 |
| 注释规范 | Excel 中保留中文描述行，方便策划理解字段含义 |
| 向下兼容 | 新增字段设置默认值，旧数据不会报错 |
| 引用完整性 | 导出时自动检查跨表引用是否存在 |
| 性能 | 启动时一次性加载全部配置到内存，运行时 O(1) 查询 |

> 💡 **策划协作建议**：建立配置表修改的 Code Review 流程（类似代码 PR），避免"一个数字改错影响整个游戏平衡性"的事故。同时为配置表建立自动化测试（覆盖关键数值边界），让策划改配置时有信心，而不是"先上线再看"。
