## 二、客户端排行榜 SDK

```csharp
using System;
using System.Collections.Generic;
using System.Threading.Tasks;
using UnityEngine;

/// <summary>
/// 排行榜条目
/// </summary>
[Serializable]
public class LeaderboardEntry
{
    public string PlayerId;
    public string Nickname;
    public string AvatarUrl;
    public long Score;
    public int Rank;           // 排名（1-based）
    public long UpdateTime;    // 最后更新时间（Unix ms）
    public bool IsLocalPlayer; // 是否是本地玩家
    public Dictionary<string, string> ExtraData; // 额外展示数据（角色/等级）
}

/// <summary>
/// 排行榜查询参数
/// </summary>
public class LeaderboardQuery
{
    public string LeaderboardId;  // 排行榜ID（global/season/friend）
    public int PageIndex;         // 页码（0-based）
    public int PageSize;          // 每页数量（建议20-50）
    public string Region;         // 地区过滤（可选）
    public bool IncludeMyRank;    // 是否同时返回自己的排名
}

/// <summary>
/// 排行榜查询结果
/// </summary>
public class LeaderboardResult
{
    public string LeaderboardId;
    public List<LeaderboardEntry> Entries;
    public int TotalCount;          // 总参与人数
    public LeaderboardEntry MyEntry; // 我的排名（IncludeMyRank=true时有效）
    public long LastUpdateTime;
    public bool IsFromCache;        // 是否来自本地缓存
}

/// <summary>
/// 积分提交结果
/// </summary>
public class ScoreSubmitResult
{
    public bool Success;
    public long NewScore;
    public int NewRank;
    public long OldScore;
    public bool IsPersonalBest;
    public string Message;
}

/// <summary>
/// 排行榜客户端管理器
/// </summary>
public class LeaderboardManager : MonoBehaviour
{
    private static LeaderboardManager instance;
    public static LeaderboardManager Instance => instance;

    [Header("缓存配置")]
    [SerializeField] private float cacheValiditySeconds = 60f;  // 缓存有效期（秒）
    [SerializeField] private bool enableOfflineCache = true;    // 离线缓存

    private Dictionary<string, LeaderboardCache> cacheMap 
        = new Dictionary<string, LeaderboardCache>();
    
    // 事件
    public event Action<string, int> OnRankUpdated; // leaderboardId, newRank
    public event Action<ScoreSubmitResult> OnScoreSubmitted;

    void Awake() { instance = this; }

    // ============ 核心接口 ============

    /// <summary>
    /// 查询排行榜
    /// </summary>
    public async Task<LeaderboardResult> GetLeaderboardAsync(LeaderboardQuery query)
    {
        string cacheKey = $"{query.LeaderboardId}_{query.PageIndex}_{query.Region}";
        
        // 检查缓存
        if (TryGetCache(cacheKey, out var cached))
        {
            cached.IsFromCache = true;
            return cached;
        }
        
        try
        {
            var result = await NetworkService.GetLeaderboardAsync(query);
            
            // 标记本地玩家
            string localId = PlayerDataService.GetLocalPlayerData()?.PlayerId;
            if (localId != null)
            {
                foreach (var entry in result.Entries)
                    entry.IsLocalPlayer = entry.PlayerId == localId;
            }
            
            // 存入缓存
            SetCache(cacheKey, result);
            
            return result;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Leaderboard] Query failed: {e.Message}");
            
            // 降级：返回离线缓存（即使过期）
            if (enableOfflineCache && cacheMap.TryGetValue(cacheKey, out var fallback))
            {
                fallback.Data.IsFromCache = true;
                return fallback.Data;
            }
            
            return null;
        }
    }

    /// <summary>
    /// 提交分数
    /// </summary>
    public async Task<ScoreSubmitResult> SubmitScoreAsync(string leaderboardId, long score,
        Dictionary<string, string> extraData = null)
    {
        var localPlayer = PlayerDataService.GetLocalPlayerData();
        
        var request = new ScoreSubmitRequest
        {
            PlayerId = localPlayer.PlayerId,
            LeaderboardId = leaderboardId,
            Score = score,
            ClientTimestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
            ExtraData = extraData,
            // 防作弊：客户端签名
            Signature = GenerateScoreSignature(localPlayer.PlayerId, score, leaderboardId)
        };
        
        try
        {
            var result = await NetworkService.SubmitScoreAsync(request);
            
            if (result.Success)
            {
                // 使排行榜缓存失效
                InvalidateCacheForLeaderboard(leaderboardId);
                
                OnRankUpdated?.Invoke(leaderboardId, result.NewRank);
                OnScoreSubmitted?.Invoke(result);
                
                Debug.Log($"[Leaderboard] Score submitted: {score}, rank={result.NewRank}");
            }
            
            return result;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Leaderboard] Submit failed: {e.Message}");
            return new ScoreSubmitResult { Success = false, Message = e.Message };
        }
    }

    /// <summary>
    /// 获取自己的排名（不需要翻页查找）
    /// </summary>
    public async Task<LeaderboardEntry> GetMyRankAsync(string leaderboardId)
    {
        string playerId = PlayerDataService.GetLocalPlayerData()?.PlayerId;
        if (playerId == null) return null;
        
        return await NetworkService.GetPlayerRankAsync(leaderboardId, playerId);
    }

    // ============ 防作弊签名 ============
    
    string GenerateScoreSignature(string playerId, long score, string leaderboardId)
    {
        // 简单示例：实际项目应使用更安全的HMAC算法
        string raw = $"{playerId}:{score}:{leaderboardId}:{GetSecretKey()}";
        byte[] bytes = System.Text.Encoding.UTF8.GetBytes(raw);
        
        using var sha256 = System.Security.Cryptography.SHA256.Create();
        var hash = sha256.ComputeHash(bytes);
        return Convert.ToBase64String(hash);
    }
    
    string GetSecretKey()
    {
        // 实际项目：从服务端获取并定期刷新，不要硬编码
        return PlayerPrefs.GetString("client_secret", "default_key");
    }

    // ============ 缓存管理 ============

    bool TryGetCache(string key, out LeaderboardResult result)
    {
        result = null;
        if (!cacheMap.TryGetValue(key, out var cache)) return false;
        
        bool isValid = Time.realtimeSinceStartup - cache.CacheTime < cacheValiditySeconds;
        if (isValid)
        {
            result = cache.Data;
            return true;
        }
        return false;
    }

    void SetCache(string key, LeaderboardResult data)
    {
        cacheMap[key] = new LeaderboardCache
        {
            Data = data,
            CacheTime = Time.realtimeSinceStartup
        };
    }

    void InvalidateCacheForLeaderboard(string leaderboardId)
    {
        var keysToRemove = new List<string>();
        foreach (var key in cacheMap.Keys)
        {
            if (key.StartsWith(leaderboardId))
                keysToRemove.Add(key);
        }
        foreach (var k in keysToRemove)
            cacheMap.Remove(k);
    }

    class LeaderboardCache
    {
        public LeaderboardResult Data;
        public float CacheTime;
    }
}

public class ScoreSubmitRequest
{
    public string PlayerId;
    public string LeaderboardId;
    public long Score;
    public long ClientTimestamp;
    public Dictionary<string, string> ExtraData;
    public string Signature;
}
```

## 三、排行榜 UI 高性能渲染

```csharp
/// <summary>
/// 虚拟列表排行榜 UI（仅渲染可见条目）
/// </summary>
public class LeaderboardUI : MonoBehaviour
{
    [SerializeField] private RectTransform contentContainer;
    [SerializeField] private GameObject entryPrefab;
    [SerializeField] private float entryHeight = 80f;
    [SerializeField] private int visibleBuffer = 5; // 可视区域外额外渲染的条目数

    private List<LeaderboardEntry> allEntries = new List<LeaderboardEntry>();
    private List<LeaderboardEntryUI> entryPool = new List<LeaderboardEntryUI>();
    private ScrollRect scrollRect;
    private float lastScrollY = -1;

    void Awake()
    {
        scrollRect = GetComponent<ScrollRect>();
        scrollRect.onValueChanged.AddListener(OnScroll);
    }

    public async void LoadLeaderboard(string leaderboardId)
    {
        var query = new LeaderboardQuery
        {
            LeaderboardId = leaderboardId,
            PageIndex = 0,
            PageSize = 200, // 一次加载足够多的数据
            IncludeMyRank = true
        };
        
        var result = await LeaderboardManager.Instance.GetLeaderboardAsync(query);
        if (result == null) return;
        
        allEntries = result.Entries;
        
        // 设置内容高度
        contentContainer.sizeDelta = new Vector2(
            contentContainer.sizeDelta.x,
            allEntries.Count * entryHeight);
        
        // 初始化可见区域
        RefreshVisibleEntries();
        
        // 滚动到自己的排名
        if (result.MyEntry != null)
            ScrollToRank(result.MyEntry.Rank);
    }

    void OnScroll(Vector2 _) => RefreshVisibleEntries();

    void RefreshVisibleEntries()
    {
        float scrollY = contentContainer.anchoredPosition.y;
        if (Mathf.Abs(scrollY - lastScrollY) < entryHeight / 2) return;
        lastScrollY = scrollY;
        
        float viewHeight = ((RectTransform)scrollRect.transform).rect.height;
        
        int firstVisible = Mathf.Max(0, 
            Mathf.FloorToInt(scrollY / entryHeight) - visibleBuffer);
        int lastVisible = Mathf.Min(allEntries.Count - 1, 
            Mathf.CeilToInt((scrollY + viewHeight) / entryHeight) + visibleBuffer);
        
        // 确保有足够的 UI 对象
        int needed = lastVisible - firstVisible + 1;
        while (entryPool.Count < needed)
        {
            var go = Instantiate(entryPrefab, contentContainer);
            entryPool.Add(go.GetComponent<LeaderboardEntryUI>());
        }
        
        // 隐藏不可见的
        for (int i = 0; i < entryPool.Count; i++)
            entryPool[i].gameObject.SetActive(i < needed);
        
        // 更新可见的
        for (int i = firstVisible; i <= lastVisible; i++)
        {
            if (i >= allEntries.Count) break;
            
            var ui = entryPool[i - firstVisible];
            var entry = allEntries[i];
            
            // 定位
            var rect = ui.GetComponent<RectTransform>();
            rect.anchoredPosition = new Vector2(0, -i * entryHeight);
            
            // 填充数据
            ui.SetData(entry);
        }
    }

    void ScrollToRank(int rank)
    {
        float targetY = (rank - 1) * entryHeight;
        contentContainer.anchoredPosition = new Vector2(0, -targetY);
        RefreshVisibleEntries();
    }
}

public class LeaderboardEntryUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Text rankText;
    [SerializeField] private UnityEngine.UI.Text nameText;
    [SerializeField] private UnityEngine.UI.Text scoreText;
    [SerializeField] private UnityEngine.UI.Image avatarImage;
    [SerializeField] private GameObject highlightPanel; // 本地玩家高亮

    public void SetData(LeaderboardEntry entry)
    {
        rankText.text = entry.Rank <= 3 ? 
            (entry.Rank == 1 ? "🥇" : entry.Rank == 2 ? "🥈" : "🥉") : 
            $"#{entry.Rank}";
        
        nameText.text = entry.Nickname;
        scoreText.text = FormatScore(entry.Score);
        highlightPanel.SetActive(entry.IsLocalPlayer);
        
        // 异步加载头像
        if (!string.IsNullOrEmpty(entry.AvatarUrl))
            _ = LoadAvatarAsync(entry.AvatarUrl);
    }

    async System.Threading.Tasks.Task LoadAvatarAsync(string url)
    {
        using var req = UnityEngine.Networking.UnityWebRequestTexture.GetTexture(url);
        await req.SendWebRequest();
        
        if (req.result == UnityEngine.Networking.UnityWebRequest.Result.Success && 
            avatarImage != null)
        {
            var tex = UnityEngine.Networking.DownloadHandlerTexture.GetContent(req);
            avatarImage.sprite = Sprite.Create(tex, 
                new Rect(0, 0, tex.width, tex.height), Vector2.one * 0.5f);
        }
    }

    string FormatScore(long score)
    {
        if (score >= 1000000) return $"{score / 1000000.0:F1}M";
        if (score >= 1000) return $"{score / 1000.0:F1}K";
        return score.ToString();
    }
}
```

## 四、服务端 Redis ZSet 实现要点

```
# Redis ZSet 排行榜核心命令

# 提交/更新分数
ZADD leaderboard:season1 NX 9800 "player_123"  # 仅当不存在时添加
ZADD leaderboard:season1 GT 10200 "player_123"  # 仅当新分数更高时更新（GT选项）

# 获取排名（0-based）
ZREVRANK leaderboard:season1 "player_123"       # 获取名次（降序）

# 获取前100名
ZREVRANGE leaderboard:season1 0 99 WITHSCORES

# 获取分数
ZSCORE leaderboard:season1 "player_123"

# 获取总人数
ZCARD leaderboard:season1

# 赛季重置（快照后清空）
RENAME leaderboard:season1 leaderboard:season1:archive
DEL leaderboard:season1
```

## 五、赛季重置与奖励

```csharp
/// <summary>
/// 赛季结束奖励计算
/// </summary>
public class SeasonRewardCalculator
{
    public static List<SeasonReward> CalculateRewards(int finalRank, int totalPlayers)
    {
        var rewards = new List<SeasonReward>();
        float percentile = (float)(totalPlayers - finalRank) / totalPlayers;
        
        // 根据百分位段给奖励
        string tier = percentile switch
        {
            > 0.997f => "Challenger",  // 前0.3%
            > 0.99f  => "GrandMaster", // 前1%
            > 0.97f  => "Master",      // 前3%
            > 0.90f  => "Diamond",     // 前10%
            > 0.70f  => "Platinum",    // 前30%
            > 0.40f  => "Gold",        // 前60%
            _        => "Silver"       // 其余
        };
        
        // 添加对应段位的奖励
        rewards.Add(GetTierReward(tier, finalRank));
        
        // 额外排名奖励（前10名）
        if (finalRank <= 10)
            rewards.Add(new SeasonReward { Type = "exclusive_title", Value = $"Top{finalRank}" });
        
        return rewards;
    }

    static SeasonReward GetTierReward(string tier, int rank) => new SeasonReward
    {
        Tier = tier,
        Type = "currency",
        Value = tier switch
        {
            "Challenger"  => "50000",
            "GrandMaster" => "20000",
            "Master"      => "10000",
            "Diamond"     => "5000",
            "Platinum"    => "2000",
            "Gold"        => "1000",
            _             => "500"
        }
    };
}

public class SeasonReward
{
    public string Tier;
    public string Type;
    public string Value;
}
```

## 六、设计要点总结

| 维度 | 方案 |
|------|------|
| 防作弊 | 服务端验证 + 客户端签名 + 历史轨迹对比 |
| 高并发写入 | Redis ZSet（GT选项，只保留最高分） |
| 实时性 | 全球实时榜延迟 <5s，好友榜 <30s |
| UI性能 | 虚拟列表，只渲染可视区域 |
| 缓存策略 | 客户端缓存60s，提交后失效 |
| 离线降级 | 无网络时展示最近一次缓存 |
