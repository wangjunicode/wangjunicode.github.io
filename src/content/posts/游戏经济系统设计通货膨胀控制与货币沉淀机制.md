## 一、游戏经济系统核心概念

```
游戏经济循环：

[货币来源（水龙头）]
├── 怪物掉落金币
├── 完成任务奖励
├── 出售物品给NPC
└── 活动/日常奖励

[货币消耗（水槽）]
├── 购买NPC商品
├── 技能升级/装备强化消耗
├── 交易税/拍卖行手续费
└── 维修费用/死亡惩罚

平衡条件：水龙头总量 ≤ 水槽总量（防通胀）
```

## 二、货币系统数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 货币类型定义
/// </summary>
public enum CurrencyType
{
    Gold,           // 软货币（可通过游戏获得）
    Diamond,        // 硬货币（付费获得，少量免费）
    HonorPoint,     // 荣誉点（PVP专属）
    GuildContrib,   // 公会贡献
    EventToken,     // 活动代币（有时限）
    BindingToken,   // 绑定代币（不可交易）
}

/// <summary>
/// 货币交易记录（用于经济分析）
/// </summary>
[Serializable]
public class CurrencyTransaction
{
    public string TransactionId;
    public string PlayerId;
    public CurrencyType Currency;
    public long Amount;             // 正数=获得，负数=消耗
    public string Source;           // 来源（"quest_reward"/"monster_drop"/"shop_buy"等）
    public long Timestamp;
    public string RelatedItemId;    // 关联物品（如果有）
}

/// <summary>
/// 玩家货币钱包
/// </summary>
public class CurrencyWallet
{
    private Dictionary<CurrencyType, long> balances = new Dictionary<CurrencyType, long>();
    
    public event Action<CurrencyType, long, long> OnBalanceChanged; // type, oldBalance, newBalance

    public long GetBalance(CurrencyType type)
    {
        return balances.TryGetValue(type, out long balance) ? balance : 0;
    }

    public bool CanAfford(CurrencyType type, long amount)
    {
        return GetBalance(type) >= amount;
    }

    public bool TryDeduct(CurrencyType type, long amount, string reason)
    {
        if (!CanAfford(type, amount))
        {
            Debug.Log($"[Currency] Insufficient {type}: need {amount}, have {GetBalance(type)}");
            return false;
        }
        
        long old = GetBalance(type);
        balances[type] = old - amount;
        
        RecordTransaction(type, -amount, reason);
        OnBalanceChanged?.Invoke(type, old, balances[type]);
        return true;
    }

    public void Add(CurrencyType type, long amount, string reason)
    {
        if (amount <= 0) return;
        
        long old = GetBalance(type);
        balances[type] = old + amount;
        
        RecordTransaction(type, amount, reason);
        OnBalanceChanged?.Invoke(type, old, balances[type]);
    }

    void RecordTransaction(CurrencyType type, long amount, string reason)
    {
        var tx = new CurrencyTransaction
        {
            TransactionId = Guid.NewGuid().ToString("N"),
            PlayerId = "local", // 从PlayerData获取
            Currency = type,
            Amount = amount,
            Source = reason,
            Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
        };
        
        // 发送到服务端记录（防作弊审计）
        // NetworkService.RecordTransaction(tx);
    }
}
```

## 三、商店价格系统

```csharp
/// <summary>
/// 动态商店价格管理器
/// 支持浮动价格、时限折扣、批量折扣
/// </summary>
public class ShopPriceManager : MonoBehaviour
{
    [System.Serializable]
    public class PriceConfig
    {
        public string ItemId;
        public CurrencyType Currency;
        public long BasePrice;
        
        // 批量折扣
        [System.Serializable]
        public class BulkDiscount { public int Quantity; public float Discount; }
        public BulkDiscount[] BulkDiscounts;
        
        // 时限折扣
        public float SaleDiscount;          // 折扣幅度（0.2 = 8折）
        public long SaleEndTime;            // 折扣结束时间
        
        // 动态浮动
        public bool DynamicPricing;         // 是否启用动态定价
        public float MinMultiplier = 0.5f;  // 最低价格倍数
        public float MaxMultiplier = 2f;    // 最高价格倍数
    }

    [SerializeField] private List<PriceConfig> priceConfigs;
    
    // 动态价格影响因素（服务端下发）
    private Dictionary<string, float> demandMultipliers = new Dictionary<string, float>();

    public long GetPrice(string itemId, int quantity = 1)
    {
        var config = priceConfigs.Find(p => p.ItemId == itemId);
        if (config == null) return 0;
        
        long price = config.BasePrice;
        
        // 动态定价
        if (config.DynamicPricing && demandMultipliers.TryGetValue(itemId, out float mult))
        {
            mult = Mathf.Clamp(mult, config.MinMultiplier, config.MaxMultiplier);
            price = (long)(price * mult);
        }
        
        // 时限折扣
        if (config.SaleDiscount > 0 && 
            DateTimeOffset.UtcNow.ToUnixTimeSeconds() < config.SaleEndTime)
        {
            price = (long)(price * (1f - config.SaleDiscount));
        }
        
        // 批量折扣
        if (config.BulkDiscounts != null && quantity > 1)
        {
            float bestDiscount = 0f;
            foreach (var bulk in config.BulkDiscounts)
            {
                if (quantity >= bulk.Quantity && bulk.Discount > bestDiscount)
                    bestDiscount = bulk.Discount;
            }
            price = (long)(price * (1f - bestDiscount));
        }
        
        return price * quantity;
    }
}
```

```csharp
/// <summary>
/// 拍卖行管理器（含税收沉淀机制）
/// </summary>
public class AuctionHouseManager : MonoBehaviour
{
    [Header("税率配置")]
    [SerializeField] private float listingFeePercent = 0.01f;   // 上架手续费（1%）
    [SerializeField] private float transactionTaxPercent = 0.05f; // 成交税（5%，货币沉淀）
    
    [Header("价格锚定")]
    [SerializeField] private float priceAnomalyThreshold = 5f;   // 超过市场均价5倍报警

    // 获取上架费
    public long GetListingFee(long startingBid)
    {
        return Mathf.Max(1, (long)(startingBid * listingFeePercent));
    }

    // 计算成交税
    public long CalculateTransactionTax(long salePrice)
    {
        return (long)(salePrice * transactionTaxPercent);
    }

    // 获取卖家实收
    public long GetSellerProceeds(long salePrice)
    {
        return salePrice - CalculateTransactionTax(salePrice);
    }

    /// <summary>
    /// 检测异常价格（价格操纵检测）
    /// </summary>
    public bool IsPriceAnomaly(string itemId, long price)
    {
        long marketAvg = GetMarketAveragePrice(itemId);
        if (marketAvg <= 0) return false;
        
        return price > marketAvg * priceAnomalyThreshold ||
               price < marketAvg / priceAnomalyThreshold;
    }

    {
        // 从服务端获取最近24小时均价
        return 0;
    }
}
```

```csharp
/// <summary>
/// 经济健康监控（服务端逻辑，此处为设计参考）
/// </summary>
public class EconomyHealthMonitor
{
    // 监控指标
    public class EconomyMetrics
    {
        public long TotalGoldInCirculation;  // 全服总金币量
        public long DailyGoldInput;          // 日金币产出
        public long DailyGoldOutput;         // 日金币消耗
        public float InflationRate;          // 通胀率（每日变化）
        public float AveragePriceIndex;      // 商品价格指数
        
        public bool IsHealthy => 
            InflationRate < 0.05f &&         // 日通胀率 < 5%
            DailyGoldOutput >= DailyGoldInput * 0.8f; // 消耗至少80%的产出
    }

    /// <summary>
    /// 根据通胀率自动调整水龙头
    /// </summary>
    public void AutoAdjust(EconomyMetrics metrics)
    {
        if (metrics.InflationRate > 0.1f)
        {
            // 通胀过高：降低怪物金币掉落量
            Debug.Log($"[Economy] 通胀率 {metrics.InflationRate:P0} 过高，降低掉落量");
            // ConfigService.Set("monster_gold_multiplier", 0.8f);
        }
        else if (metrics.InflationRate < -0.05f)
        {
            // 通缩：增加任务奖励
            Debug.Log($"[Economy] 通缩，增加任务奖励");
        }
    }
}
```

## 六、双货币体系设计

| 货币类型 | 获取方式 | 用途 | 是否可交易 |
|----------|----------|------|------------|
| 金币（软）| 游戏内活动 | 基础物品/强化 | 可交易（拍卖行）|
| 钻石（硬）| 付费/少量活动 | 高级物品/皮肤 | 不可交易 |
| 绑定代币 | 活动奖励 | 限定内容 | 不可交易 |
| 荣誉点 | PVP奖励 | PVP专属装备 | 不可交易 |

**双货币体系核心原则：**
1. 硬货币不直接换取战斗力（避免P2W投诉）
2. 软货币通过税收回收，控制总量
3. 绑定道具保证付费玩家的体验，但不破坏公平
4. 时限活动代币制造稀缺感，同时避免长期囤积
