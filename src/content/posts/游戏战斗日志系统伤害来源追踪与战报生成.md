## 一、战斗事件定义

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 战斗事件基类
/// </summary>
public abstract class CombatEvent
{
    public float Timestamp;  // 游戏内时间
    public string SourceId;  // 造成者
    public string TargetId;  // 受者
    public string SkillId;   // 技能/能力来源
}

/// <summary>
/// 伤害事件
/// </summary>
public class DamageEvent : CombatEvent
{
    public float RawDamage;    // 原始伤害
    public float FinalDamage;  // 最终伤害（减伤后）
    public DamageType Type;
    public bool IsCritical;
    public bool IsKillingBlow; // 致死一击
    public float TargetHP;     // 受伤后目标血量
    public float TargetMaxHP;
}

/// <summary>
/// 治疗事件
/// </summary>
public class HealEvent : CombatEvent
{
    public float RawHeal;
    public float FinalHeal;    // 超量治疗不计入
    public bool IsOverheal;
}

/// <summary>
/// 死亡事件
/// </summary>
public class DeathEvent : CombatEvent
{
    public string KillerSkillId;
    public float SurvivalTime; // 从战斗开始到死亡的存活时间
}

/// <summary>
/// Buff事件
/// </summary>
public class BuffEvent : CombatEvent
{
    public string BuffId;
    public bool IsApply; // true=施加，false=移除
    public float Duration;
}
```

## 二、战斗日志管理器

```csharp
/// <summary>
/// 战斗日志管理器
/// </summary>
public class CombatLogger : MonoBehaviour
{
    private static CombatLogger instance;
    public static CombatLogger Instance => instance;

    [SerializeField] private int maxEventCount = 2000;  // 最多保留N个事件
    [SerializeField] private bool enableDetailedLog = true;

    private List<CombatEvent> events = new List<CombatEvent>();
    private float battleStartTime;
    private bool isBattleActive;

    // 实体名称缓存
    private Dictionary<string, string> entityNames = new Dictionary<string, string>();

    void Awake() { instance = this; }

    public void StartBattle()
    {
        events.Clear();
        battleStartTime = Time.time;
        isBattleActive = true;
        Debug.Log("[CombatLog] 战斗开始");
    }

    public void EndBattle()
    {
        isBattleActive = false;
        Debug.Log($"[CombatLog] 战斗结束，共{events.Count}个事件");
    }

    void LogEvent(CombatEvent evt)
    {
        if (!isBattleActive) return;
        
        evt.Timestamp = Time.time - battleStartTime;
        events.Add(evt);
        
        // 限制日志大小
        if (events.Count > maxEventCount)
            events.RemoveAt(0);
    }

    public void LogDamage(string sourceId, string targetId, string skillId,
        float rawDamage, float finalDamage, DamageType type, bool isCrit = false)
    {
        var evt = new DamageEvent
        {
            SourceId = sourceId,
            TargetId = targetId,
            SkillId = skillId,
            RawDamage = rawDamage,
            FinalDamage = finalDamage,
            Type = type,
            IsCritical = isCrit,
        };
        LogEvent(evt);
    }

    public void LogHeal(string sourceId, string targetId, string skillId,
        float rawHeal, float finalHeal)
    {
        LogEvent(new HealEvent
        {
            SourceId = sourceId,
            TargetId = targetId,
            SkillId = skillId,
            RawHeal = rawHeal,
            FinalHeal = finalHeal,
            IsOverheal = rawHeal > finalHeal
        });
    }

    public void LogDeath(string targetId, string killerId, string killerSkill)
    {
        LogEvent(new DeathEvent
        {
            SourceId = killerId,
            TargetId = targetId,
            SkillId = killerSkill,
            SurvivalTime = Time.time - battleStartTime,
            IsKillingBlow = true
        });
    }

    // ============ 战报生成 ============

    public BattleReport GenerateReport(List<string> participantIds)
    {
        var report = new BattleReport
        {
            Duration = Time.time - battleStartTime,
            ParticipantStats = new Dictionary<string, ParticipantStats>()
        };
        
        // 初始化所有参与者统计
        foreach (var id in participantIds)
            report.ParticipantStats[id] = new ParticipantStats { EntityId = id };
        
        // 汇总伤害/治疗
        foreach (var evt in events)
        {
            if (evt is DamageEvent dmgEvt)
            {
                if (report.ParticipantStats.ContainsKey(dmgEvt.SourceId))
                {
                    var stats = report.ParticipantStats[dmgEvt.SourceId];
                    stats.TotalDamageDealt += dmgEvt.FinalDamage;
                    stats.DamageEventCount++;
                    if (dmgEvt.IsCritical) stats.CritCount++;
                    stats.MaxSingleDamage = Mathf.Max(stats.MaxSingleDamage, dmgEvt.FinalDamage);
                }
                
                if (report.ParticipantStats.ContainsKey(dmgEvt.TargetId))
                    report.ParticipantStats[dmgEvt.TargetId].TotalDamageTaken += dmgEvt.FinalDamage;
            }
            else if (evt is HealEvent healEvt)
            {
                if (report.ParticipantStats.ContainsKey(healEvt.SourceId))
                    report.ParticipantStats[healEvt.SourceId].TotalHealDone += healEvt.FinalHeal;
            }
            else if (evt is DeathEvent deathEvt)
            {
                if (report.ParticipantStats.ContainsKey(deathEvt.TargetId))
                    report.ParticipantStats[deathEvt.TargetId].DeathCount++;
                if (report.ParticipantStats.ContainsKey(deathEvt.SourceId))
                    report.ParticipantStats[deathEvt.SourceId].KillCount++;
            }
        }
        
        // 计算DPS
        if (report.Duration > 0)
        {
            foreach (var stats in report.ParticipantStats.Values)
                stats.AverageDPS = stats.TotalDamageDealt / report.Duration;
        }
        
        // 找MVP（总伤害最高）
        float maxDamage = 0;
        foreach (var kv in report.ParticipantStats)
        {
            if (kv.Value.TotalDamageDealt > maxDamage)
            {
                maxDamage = kv.Value.TotalDamageDealt;
                report.MVPPlayerId = kv.Key;
            }
        }
        
        return report;
    }
}

[Serializable]
public class BattleReport
{
    public float Duration;
    public string MVPPlayerId;
    public Dictionary<string, ParticipantStats> ParticipantStats;
}

[Serializable]
public class ParticipantStats
{
    public string EntityId;
    public float TotalDamageDealt;
    public float TotalDamageTaken;
    public float TotalHealDone;
    public float AverageDPS;
    public float MaxSingleDamage;
    public int DamageEventCount;
    public int CritCount;
    public int KillCount;
    public int DeathCount;
    
    public float CritRate => DamageEventCount > 0 ? (float)CritCount / DamageEventCount : 0;
}
```

## 三、悬浮伤害数字

```csharp
/// <summary>
/// 悬浮伤害数字管理器
/// </summary>
public class DamageNumberManager : MonoBehaviour
{
    private static DamageNumberManager instance;
    public static DamageNumberManager Instance => instance;

    [SerializeField] private GameObject normalDamagePrefab;
    [SerializeField] private GameObject critDamagePrefab;
    [SerializeField] private GameObject healPrefab;
    [SerializeField] private Canvas worldCanvas;
    
    private ObjectPool<DamageNumberUI> normalPool;
    private ObjectPool<DamageNumberUI> critPool;

    void Awake() { instance = this; }

    public void ShowDamage(Vector3 worldPos, float damage, bool isCrit, DamageType type)
    {
        // 转换到屏幕空间
        Vector3 screenPos = Camera.main.WorldToScreenPoint(worldPos);
        if (screenPos.z < 0) return; // 在摄像机后方，不显示
        
        // 随机偏移防止重叠
        Vector2 offset = UnityEngine.Random.insideUnitCircle * 30f;
        
        GameObject prefab = isCrit ? critDamagePrefab : normalDamagePrefab;
        var go = Instantiate(prefab, worldCanvas.transform);
        var rt = go.GetComponent<RectTransform>();
        rt.position = screenPos + (Vector3)offset;
        
        var ui = go.GetComponent<DamageNumberUI>();
        ui?.Show(damage, isCrit, type);
    }

    public void ShowHeal(Vector3 worldPos, float amount)
    {
        Vector3 screenPos = Camera.main.WorldToScreenPoint(worldPos);
        if (screenPos.z < 0) return;
        
        var go = Instantiate(healPrefab, worldCanvas.transform);
        go.GetComponent<RectTransform>().position = screenPos;
        go.GetComponent<DamageNumberUI>()?.ShowHeal(amount);
    }
}

/// <summary>
/// 单个伤害数字UI
/// </summary>
public class DamageNumberUI : MonoBehaviour
{
    [SerializeField] private TMPro.TextMeshProUGUI text;
    [SerializeField] private float duration = 1f;
    [SerializeField] private float floatDistance = 80f;

    public void Show(float damage, bool isCrit, DamageType type)
    {
        string formatted = damage >= 1000 ? $"{damage / 1000f:F1}K" : $"{(int)damage}";
        text.text = isCrit ? $"<b>{formatted}!</b>" : formatted;
        text.color = GetDamageColor(type, isCrit);
        
        if (isCrit) transform.localScale = Vector3.one * 1.3f;
        
        // 上浮 + 淡出动画
        var canvasGroup = GetComponent<CanvasGroup>() ?? gameObject.AddComponent<CanvasGroup>();
        DG.Tweening.DOTween.Sequence()
            .Join(transform.DOLocalMoveY(transform.localPosition.y + floatDistance, duration))
            .Join(canvasGroup.DOFade(0, duration * 0.5f).SetDelay(duration * 0.5f))
            .OnComplete(() => Destroy(gameObject));
    }

    public void ShowHeal(float amount)
    {
        text.text = $"+{(int)amount}";
        text.color = new Color(0.2f, 1f, 0.4f);
        
        var canvasGroup = GetComponent<CanvasGroup>() ?? gameObject.AddComponent<CanvasGroup>();
        DG.Tweening.DOTween.Sequence()
            .Join(transform.DOLocalMoveY(transform.localPosition.y + floatDistance, duration))
            .Join(canvasGroup.DOFade(0, duration * 0.5f).SetDelay(duration * 0.5f))
            .OnComplete(() => Destroy(gameObject));
    }

    Color GetDamageColor(DamageType type, bool isCrit)
    {
        if (isCrit) return new Color(1f, 0.8f, 0f);
        return type switch
        {
            DamageType.Fire    => new Color(1f, 0.4f, 0f),
            DamageType.Ice     => new Color(0.4f, 0.8f, 1f),
            DamageType.Thunder => new Color(0.8f, 0.6f, 1f),
            _ => Color.white
        };
    }
}
```

## 四、战斗日志设计要点

| 要点 | 方案 |
|------|------|
| 事件队列限制 | 设置最大事件数，防止内存溢出 |
| 战报统计 | 结束时一次性汇总，不要实时计算 |
| DPS实时显示 | 用滑动窗口（最近10秒）而非全程平均 |
| 伤害数字优化 | 超过屏幕外不显示，同位置数字合并 |
| 关键事件标记 | 暴击/击杀特殊高亮显示 |
| 服务端日志 | 服务端同步记录，用于事后作弊审计 |
