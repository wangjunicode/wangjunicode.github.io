## 一、好友数据模型

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

public enum FriendStatus { None, PendingSent, PendingReceived, Friends, Blocked }
public enum OnlineStatus { Offline, Online, InGame, Away }

[Serializable]
public class FriendInfo
{
    public string PlayerId;
    public string PlayerName;
    public string AvatarUrl;
    public FriendStatus RelationStatus;
    public OnlineStatus OnlineStatus;
    public string CurrentActivity;  // "正在游玩：第三关"
    public long LastActiveTime;
    public int Level;
}

[Serializable]
public class FriendListData
{
    public List<FriendInfo> Friends = new List<FriendInfo>();
    public List<FriendInfo> PendingRequests = new List<FriendInfo>();
    public int OnlineCount;
}
```

## 二、好友管理器

```csharp
public class FriendManager : MonoBehaviour
{
    private static FriendManager instance;
    public static FriendManager Instance => instance;
    
    private FriendListData friendList = new FriendListData();
    
    public event Action<FriendInfo> OnFriendOnline;
    public event Action<FriendInfo> OnFriendOffline;
    public event Action<FriendInfo> OnFriendRequest;
    public event Action OnFriendListUpdated;

    void Awake() { instance = this; }

    public async System.Threading.Tasks.Task LoadFriendList()
    {
        try
        {
            friendList = await NetworkService.GetFriendList();
            OnFriendListUpdated?.Invoke();
        }
        catch (Exception e)
        {
            Debug.LogError($"[Friend] 获取好友列表失败: {e.Message}");
        }
    }

    public async System.Threading.Tasks.Task<bool> SendFriendRequest(string targetPlayerId)
    {
        bool success = await NetworkService.SendFriendRequest(targetPlayerId);
        if (success) await LoadFriendList();
        return success;
    }

    public async System.Threading.Tasks.Task<bool> AcceptFriendRequest(string requesterId)
    {
        bool success = await NetworkService.AcceptFriendRequest(requesterId);
        if (success)
        {
            await LoadFriendList();
            // 发送通知给对方
        }
        return success;
    }

    public async System.Threading.Tasks.Task<bool> RemoveFriend(string friendId)
    {
        bool success = await NetworkService.RemoveFriend(friendId);
        if (success) await LoadFriendList();
        return success;
    }

    public async System.Threading.Tasks.Task BlockPlayer(string playerId)
    {
        await NetworkService.BlockPlayer(playerId);
        await LoadFriendList();
    }

    /// <summary>
    /// 获取排序后的好友列表（在线优先，最近活跃优先）
    /// </summary>
    public List<FriendInfo> GetSortedFriends()
    {
        var sorted = new List<FriendInfo>(friendList.Friends);
        sorted.Sort((a, b) =>
        {
            // 在线的优先
            bool aOnline = a.OnlineStatus != OnlineStatus.Offline;
            bool bOnline = b.OnlineStatus != OnlineStatus.Offline;
            if (aOnline != bOnline) return bOnline.CompareTo(aOnline);
            
            // 同为在线：游戏中的排前面
            bool aInGame = a.OnlineStatus == OnlineStatus.InGame;
            bool bInGame = b.OnlineStatus == OnlineStatus.InGame;
            if (aInGame != bInGame) return bInGame.CompareTo(aInGame);
            
            // 都离线：最近活跃排前
            return b.LastActiveTime.CompareTo(a.LastActiveTime);
        });
        return sorted;
    }

    /// <summary>
    /// 接收服务端推送的状态变化
    /// </summary>
    public void OnFriendStatusChanged(string playerId, OnlineStatus newStatus)
    {
        var friend = friendList.Friends.Find(f => f.PlayerId == playerId);
        if (friend == null) return;
        
        var oldStatus = friend.OnlineStatus;
        friend.OnlineStatus = newStatus;
        
        if (oldStatus == OnlineStatus.Offline && newStatus != OnlineStatus.Offline)
            OnFriendOnline?.Invoke(friend);
        else if (oldStatus != OnlineStatus.Offline && newStatus == OnlineStatus.Offline)
            OnFriendOffline?.Invoke(friend);
        
        OnFriendListUpdated?.Invoke();
    }

    public List<FriendInfo> GetPendingRequests() => friendList.PendingRequests;
    public int GetOnlineCount() => friendList.OnlineCount;
    public bool IsFriend(string playerId) => 
        friendList.Friends.Exists(f => f.PlayerId == playerId);
}
```

## 三、在线状态实时推送

```csharp
/// <summary>
/// 好友状态实时推送处理（WebSocket消息处理）
/// </summary>
public class FriendPresenceHandler : MonoBehaviour
{
    void Start()
    {
        // 订阅WebSocket好友状态消息
        NetworkService.OnMessageReceived += HandleServerMessage;
    }
    
    void HandleServerMessage(string type, string payload)
    {
        switch (type)
        {
            case "friend_online":
                var onlineData = JsonUtility.FromJson<FriendStatusMsg>(payload);
                FriendManager.Instance?.OnFriendStatusChanged(
                    onlineData.playerId, OnlineStatus.Online);
                break;
                
            case "friend_offline":
                var offlineData = JsonUtility.FromJson<FriendStatusMsg>(payload);
                FriendManager.Instance?.OnFriendStatusChanged(
                    offlineData.playerId, OnlineStatus.Offline);
                break;
                
            case "friend_request":
                var reqData = JsonUtility.FromJson<FriendInfo>(payload);
                FriendManager.Instance?.OnFriendRequest?.Invoke(reqData);
                UIManager.Instance?.ShowToast($"{reqData.PlayerName} 请求添加你为好友");
                break;
        }
    }
}

[Serializable] class FriendStatusMsg { public string playerId; public string status; }
```
