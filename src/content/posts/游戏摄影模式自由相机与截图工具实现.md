## 一、自由摄影模式控制器

```csharp
using UnityEngine;
using UnityEngine.Rendering;
using UnityEngine.Rendering.Universal;
using System.Collections;

/// <summary>
/// 游戏内摄影模式（Photo Mode）
/// </summary>
public class PhotoMode : MonoBehaviour
{
    private static PhotoMode instance;
    public static PhotoMode Instance => instance;

    [Header("自由相机配置")]
    [SerializeField] private float moveSpeed = 10f;
    [SerializeField] private float sprintMultiplier = 3f;
    [SerializeField] private float rotateSpeed = 2f;
    [SerializeField] private float moveSmoothing = 0.1f;
    [SerializeField] private float rotateSmoothing = 0.05f;

    [Header("景深与后处理")]
    [SerializeField] private Volume postProcessVolume;
    
    [Header("超采样截图")]
    [SerializeField] private int supersamplingFactor = 2;  // 2x = 4倍分辨率

    private bool isActive;
    private Camera mainCamera;
    private Transform originalCamParent;
    private Vector3 originalCamPos;
    private Quaternion originalCamRot;
    
    // 自由相机移动
    private Vector3 targetPosition;
    private Quaternion targetRotation;
    private Vector3 velocityPos;
    private float yaw, pitch, roll;
    
    // 后处理组件
    private DepthOfField dof;
    private ColorAdjustments colorAdj;

    public bool IsActive => isActive;

    void Awake()
    {
        instance = this;
        mainCamera = Camera.main;
    }

    /// <summary>
    /// 进入摄影模式
    /// </summary>
    public void Enter()
    {
        if (isActive) return;
        
        isActive = true;
        
        // 保存原始摄像机状态
        originalCamParent = mainCamera.transform.parent;
        originalCamPos = mainCamera.transform.position;
        originalCamRot = mainCamera.transform.rotation;
        
        // 解放摄像机
        mainCamera.transform.SetParent(null);
        targetPosition = mainCamera.transform.position;
        targetRotation = mainCamera.transform.rotation;
        yaw = targetRotation.eulerAngles.y;
        pitch = targetRotation.eulerAngles.x;
        
        // 暂停游戏（时间停止）
        Time.timeScale = 0f;
        
        // 解锁光标
        Cursor.lockState = CursorLockMode.None;
        Cursor.visible = true;
        
        // 初始化后处理
        if (postProcessVolume != null)
        {
            postProcessVolume.profile.TryGet(out dof);
            postProcessVolume.profile.TryGet(out colorAdj);
        }
        
        // 显示摄影模式UI
        PhotoModeUI.Instance?.Show();
    }

    /// <summary>
    /// 退出摄影模式
    /// </summary>
    public void Exit()
    {
        if (!isActive) return;
        
        isActive = false;
        
        // 恢复摄像机
        mainCamera.transform.SetParent(originalCamParent);
        mainCamera.transform.position = originalCamPos;
        mainCamera.transform.rotation = originalCamRot;
        
        // 恢复时间
        Time.timeScale = 1f;
        
        // 恢复光标
        Cursor.lockState = CursorLockMode.Locked;
        Cursor.visible = false;
        
        PhotoModeUI.Instance?.Hide();
    }

    void Update()
    {
        if (!isActive) return;
        
        HandleMovement();
        HandleRotation();
        
        // 退出快捷键
        if (Input.GetKeyDown(KeyCode.Escape))
            Exit();
        
        // 截图快捷键
        if (Input.GetKeyDown(KeyCode.F12))
            TakePhoto();
    }

    void HandleMovement()
    {
        float speed = moveSpeed * (Input.GetKey(KeyCode.LeftShift) ? sprintMultiplier : 1f);
        
        // WASD + QE 六自由度移动
        Vector3 input = Vector3.zero;
        if (Input.GetKey(KeyCode.W)) input += Vector3.forward;
        if (Input.GetKey(KeyCode.S)) input -= Vector3.forward;
        if (Input.GetKey(KeyCode.A)) input -= Vector3.right;
        if (Input.GetKey(KeyCode.D)) input += Vector3.right;
        if (Input.GetKey(KeyCode.E)) input += Vector3.up;
        if (Input.GetKey(KeyCode.Q)) input -= Vector3.up;
        
        targetPosition += mainCamera.transform.TransformDirection(input) * speed * Time.unscaledDeltaTime;
        
        // 平滑移动
        mainCamera.transform.position = Vector3.SmoothDamp(
            mainCamera.transform.position, targetPosition, 
            ref velocityPos, moveSmoothing, float.MaxValue, Time.unscaledDeltaTime);
    }

    void HandleRotation()
    {
        // 右键拖拽旋转
        if (Input.GetMouseButton(1))
        {
            yaw   += Input.GetAxis("Mouse X") * rotateSpeed;
            pitch -= Input.GetAxis("Mouse Y") * rotateSpeed;
            pitch  = Mathf.Clamp(pitch, -89f, 89f);
        }
        
        targetRotation = Quaternion.Euler(pitch, yaw, roll);
        
        mainCamera.transform.rotation = Quaternion.Slerp(
            mainCamera.transform.rotation, targetRotation,
            rotateSmoothing > 0 ? Time.unscaledDeltaTime / rotateSmoothing : 1f);
    }

    // ============ 后处理控制 ============
    
    public void SetFocalDistance(float distance)
    {
        if (dof != null) dof.focusDistance.value = distance;
    }

    public void SetAperture(float aperture)
    {
        if (dof != null) dof.aperture.value = aperture;
    }

    public void SetExposure(float exposure)
    {
        if (colorAdj != null) colorAdj.postExposure.value = exposure;
    }

    public void SetSaturation(float saturation)
    {
        if (colorAdj != null) colorAdj.saturation.value = saturation;
    }

    public void SetContrast(float contrast)
    {
        if (colorAdj != null) colorAdj.contrast.value = contrast;
    }

    // ============ 截图 ============
    
    public void TakePhoto()
    {
        StartCoroutine(CaptureScreenshot());
    }

    IEnumerator CaptureScreenshot()
    {
        // 隐藏UI
        PhotoModeUI.Instance?.SetVisible(false);
        
        yield return new WaitForEndOfFrame();
        
        // 超采样截图（提升分辨率）
        int width  = Screen.width  * supersamplingFactor;
        int height = Screen.height * supersamplingFactor;
        
        RenderTexture rt = new RenderTexture(width, height, 24, RenderTextureFormat.ARGB32);
        rt.antiAliasing = 8;
        
        mainCamera.targetTexture = rt;
        mainCamera.Render();
        
        RenderTexture.active = rt;
        Texture2D screenshot = new Texture2D(width, height, TextureFormat.RGB24, false);
        screenshot.ReadPixels(new Rect(0, 0, width, height), 0, 0);
        screenshot.Apply();
        
        mainCamera.targetTexture = null;
        RenderTexture.active = null;
        Destroy(rt);
        
        // 保存文件
        string fileName = $"Screenshot_{System.DateTime.Now:yyyyMMdd_HHmmss}.png";
        string path = System.IO.Path.Combine(
            System.Environment.GetFolderPath(System.Environment.SpecialFolder.MyPictures),
            "GameScreenshots", fileName);
        
        System.IO.Directory.CreateDirectory(System.IO.Path.GetDirectoryName(path));
        System.IO.File.WriteAllBytes(path, screenshot.EncodeToPNG());
        Destroy(screenshot);
        
        // 恢复UI
        PhotoModeUI.Instance?.SetVisible(true);
        
        // 提示
        UIManager.Instance?.ShowMessage($"截图已保存！\n{fileName}");
        
        Debug.Log($"[PhotoMode] Screenshot saved to: {path}");
    }
}
```

## 二、摄影模式 UI

```csharp
/// <summary>
/// 摄影模式 UI 面板
/// </summary>
public class PhotoModeUI : MonoBehaviour
{
    private static PhotoModeUI instance;
    public static PhotoModeUI Instance => instance;

    [SerializeField] private UnityEngine.UI.Slider focalDistanceSlider;
    [SerializeField] private UnityEngine.UI.Slider apertureSlider;
    [SerializeField] private UnityEngine.UI.Slider exposureSlider;
    [SerializeField] private UnityEngine.UI.Slider saturationSlider;
    [SerializeField] private UnityEngine.UI.Slider rollSlider;           // 倾斜角
    [SerializeField] private UnityEngine.UI.Button photoButton;
    [SerializeField] private UnityEngine.UI.Button exitButton;
    [SerializeField] private CanvasGroup canvasGroup;

    void Awake()
    {
        instance = this;
        
        focalDistanceSlider.onValueChanged.AddListener(v => PhotoMode.Instance.SetFocalDistance(v));
        apertureSlider.onValueChanged.AddListener(v => PhotoMode.Instance.SetAperture(v));
        exposureSlider.onValueChanged.AddListener(v => PhotoMode.Instance.SetExposure(v));
        saturationSlider.onValueChanged.AddListener(v => PhotoMode.Instance.SetSaturation(v));
        photoButton.onClick.AddListener(() => PhotoMode.Instance.TakePhoto());
        exitButton.onClick.AddListener(() => PhotoMode.Instance.Exit());
        
        gameObject.SetActive(false);
    }

    public void Show() => gameObject.SetActive(true);
    public void Hide() => gameObject.SetActive(false);
    public void SetVisible(bool visible) => canvasGroup.alpha = visible ? 1f : 0f;
}
```
