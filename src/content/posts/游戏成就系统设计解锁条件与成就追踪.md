## 一、成就数据定义

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 成就配置
/// </summary>
[CreateAssetMenu(fileName = "Achievement", menuName = "Game/Achievement")]
public class AchievementData : ScriptableObject
{
    public string Id;
    public string Title;
    [TextArea(2, 4)] public string Description;
    public Sprite Icon;
    public AchievementType Type;
    public AchievementRarity Rarity;
    public bool IsHidden;          // 隐藏成就（未达成时不显示条件）
    
    // 进度型成就配置
    public int TargetCount;        // 目标数量（如"击杀100个敌人"）
    public string TrackingEvent;   // 追踪的事件类型
    
    // 奖励
    public AchievementReward[] Rewards;
    
    // 平台成就ID（Google Play/Steam等）
    public string PlatformAchievementId;
}

public enum AchievementType 
{ 
    OneTime,     // 单次完成（如"完成主线任务"）
    Cumulative,  // 累计（如"累计击杀100个敌人"）
    Tiered,      // 分层（如"击杀10/100/1000"分别解锁）
}

public enum AchievementRarity { Common, Uncommon, Rare, Epic, Legendary }

[Serializable]
public class AchievementReward
{
    public RewardType Type;
    public int Amount;
    public string ItemId;
}
```

## 二、成就管理器

```csharp
/// <summary>
/// 成就管理器
/// </summary>
public class AchievementManager : MonoBehaviour
{
    private static AchievementManager instance;
    public static AchievementManager Instance => instance;

    [SerializeField] private AchievementData[] allAchievements;
    
    private Dictionary<string, AchievementData> achievementMap 
        = new Dictionary<string, AchievementData>();
    
    // 玩家成就状态
    private Dictionary<string, AchievementProgress> progressMap 
        = new Dictionary<string, AchievementProgress>();
    
    public event Action<AchievementData> OnAchievementUnlocked;

    void Awake()
    {
        instance = this;
        foreach (var ach in allAchievements)
            achievementMap[ach.Id] = ach;
        
        LoadProgress();
    }

    /// <summary>
    /// 上报成就事件（主要接口）
    /// </summary>
    public void ReportEvent(string eventType, int value = 1, Dictionary<string, object> metadata = null)
    {
        foreach (var ach in allAchievements)
        {
            if (ach.TrackingEvent != eventType) continue;
            
            var progress = GetOrCreateProgress(ach.Id);
            if (progress.IsUnlocked) continue;
            
            switch (ach.Type)
            {
                case AchievementType.OneTime:
                    if (value > 0)
                        UnlockAchievement(ach.Id);
                    break;
                    
                case AchievementType.Cumulative:
                    progress.CurrentCount += value;
                    if (progress.CurrentCount >= ach.TargetCount)
                        UnlockAchievement(ach.Id);
                    else
                        SaveProgress();
                    break;
            }
        }
    }

    public void UnlockAchievement(string achievementId)
    {
        if (!achievementMap.TryGetValue(achievementId, out var ach)) return;
        
        var progress = GetOrCreateProgress(achievementId);
        if (progress.IsUnlocked) return; // 防重复解锁
        
        progress.IsUnlocked = true;
        progress.UnlockTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        
        // 发放奖励
        GrantRewards(ach);
        
        // 触发UI动画
        OnAchievementUnlocked?.Invoke(ach);
        AchievementToastUI.Instance?.ShowUnlock(ach);
        
        // 同步到平台（Google Play/Steam）
        SyncToPlatform(ach);
        
        // 保存
        SaveProgress();
        
        Debug.Log($"[Achievement] 解锁成就：{ach.Title}");
    }

    void GrantRewards(AchievementData ach)
    {
        if (ach.Rewards == null) return;
        
        foreach (var reward in ach.Rewards)
        {
            switch (reward.Type)
            {
                case RewardType.Experience:
                    ExperienceManager.Instance?.AddExp(reward.Amount);
                    break;
                case RewardType.Currency:
                    // 发放货币
                    break;
                case RewardType.Item:
                    // 发放物品
                    break;
            }
        }
    }

    void SyncToPlatform(AchievementData ach)
    {
        if (string.IsNullOrEmpty(ach.PlatformAchievementId)) return;
        
#if UNITY_ANDROID
        // Google Play Games
        // Social.ReportProgress(ach.PlatformAchievementId, 100f, success => {});
#elif UNITY_IOS
        // Apple Game Center
        // Social.ReportProgress(ach.PlatformAchievementId, 100f, success => {});
#endif
    }

    AchievementProgress GetOrCreateProgress(string id)
    {
        if (!progressMap.TryGetValue(id, out var progress))
        {
            progress = new AchievementProgress { AchievementId = id };
            progressMap[id] = progress;
        }
        return progress;
    }

    public bool IsUnlocked(string achievementId) =>
        progressMap.TryGetValue(achievementId, out var p) && p.IsUnlocked;

    public float GetProgress(string achievementId)
    {
        if (!achievementMap.TryGetValue(achievementId, out var ach)) return 0;
        if (!progressMap.TryGetValue(achievementId, out var prog)) return 0;
        if (prog.IsUnlocked) return 1f;
        return ach.TargetCount > 0 ? (float)prog.CurrentCount / ach.TargetCount : 0;
    }

    void SaveProgress()
    {
        var list = new List<AchievementProgress>(progressMap.Values);
        PlayerPrefs.SetString("achievements", 
            JsonUtility.ToJson(new AchievementProgressList { Items = list }));
        PlayerPrefs.Save();
    }

    void LoadProgress()
    {
        string json = PlayerPrefs.GetString("achievements", "");
        if (string.IsNullOrEmpty(json)) return;
        
        var list = JsonUtility.FromJson<AchievementProgressList>(json);
        if (list?.Items == null) return;
        
        foreach (var item in list.Items)
            progressMap[item.AchievementId] = item;
    }
}

[Serializable]
public class AchievementProgress
{
    public string AchievementId;
    public bool IsUnlocked;
    public int CurrentCount;
    public long UnlockTime;
}

[Serializable]
public class AchievementProgressList
{
    public List<AchievementProgress> Items;
}
```

## 三、成就解锁Toast UI

```csharp
/// <summary>
/// 成就解锁弹窗（Toast通知）
/// </summary>
public class AchievementToastUI : MonoBehaviour
{
    private static AchievementToastUI instance;
    public static AchievementToastUI Instance => instance;

    [SerializeField] private GameObject toastRoot;
    [SerializeField] private UnityEngine.UI.Image icon;
    [SerializeField] private UnityEngine.UI.Text titleText;
    [SerializeField] private UnityEngine.UI.Text descText;
    [SerializeField] private AudioClip unlockSFX;
    [SerializeField] private float displayDuration = 3f;

    private Queue<AchievementData> pendingToasts = new Queue<AchievementData>();
    private bool isShowing;

    void Awake() { instance = this; }

    public void ShowUnlock(AchievementData ach)
    {
        pendingToasts.Enqueue(ach);
        if (!isShowing)
            StartCoroutine(ShowNextToast());
    }

    System.Collections.IEnumerator ShowNextToast()
    {
        while (pendingToasts.Count > 0)
        {
            isShowing = true;
            var ach = pendingToasts.Dequeue();
            
            // 设置内容
            if (icon != null && ach.Icon != null) icon.sprite = ach.Icon;
            if (titleText != null) titleText.text = ach.Title;
            if (descText != null) descText.text = ach.Description;
            
            // 播放音效
            if (unlockSFX != null)
                AudioManager.Instance?.PlaySFX(unlockSFX, 0.8f);
            
            // 显示动画
            toastRoot.SetActive(true);
            // DG.Tweening.DOTween 动画...
            
            yield return new WaitForSeconds(displayDuration);
            
            // 隐藏动画
            toastRoot.SetActive(false);
            yield return new WaitForSeconds(0.2f);
        }
        isShowing = false;
    }
}
```

## 四、成就事件上报示例

```csharp
// 在游戏各系统中上报成就事件
public class AchievementEventReporter : MonoBehaviour
{
    // 击杀敌人
    public void OnEnemyKilled(string enemyId)
    {
        AchievementManager.Instance?.ReportEvent("kill_enemy");
        AchievementManager.Instance?.ReportEvent($"kill_{enemyId}"); // 特定敌人成就
    }
    
    // 完成关卡
    public void OnLevelCompleted(int level, float clearTime)
    {
        AchievementManager.Instance?.ReportEvent("complete_level");
        
        // 速通成就
        if (clearTime < 60f)
            AchievementManager.Instance?.ReportEvent("speedrun_60s");
    }
    
    // 达到等级
    public void OnLevelUp(int newLevel)
    {
        if (newLevel >= 10) AchievementManager.Instance?.UnlockAchievement("reach_level_10");
        if (newLevel >= 50) AchievementManager.Instance?.UnlockAchievement("reach_level_50");
        if (newLevel >= 100) AchievementManager.Instance?.UnlockAchievement("max_level");
    }
}
```

## 五、成就系统设计原则

| 原则 | 说明 |
|------|------|
| 分层难度 | 简单/中等/困难成就保证各类玩家都有成就感 |
| 隐藏成就 | 用于惊喜/彩蛋/高难度挑战 |
| 进度反馈 | 进度型成就需显示进度条（如23/100）|
| 平台同步 | Google Play/Steam/App Store同步，增加粘性 |
| 防刷机制 | 服务端验证，防止批量刷成就 |
| 奖励设计 | 稀有成就给稀有奖励，增加解锁动力 |
