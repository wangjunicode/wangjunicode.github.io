## 一、商品数据结构

```csharp
using System;
using UnityEngine;

public enum ShopItemType { Item, Skin, Bundle, Currency, BattlePass }
public enum PriceType { Gold, Gem, RealMoney }

[Serializable]
public class ShopItem
{
    public string ItemId;
    public string Name;
    [TextArea] public string Description;
    public Sprite Icon;
    public ShopItemType Type;
    
    [Header("价格")]
    public PriceType PriceType;
    public long OriginalPrice;
    public long SalePrice;          // 折扣价（0=无折扣）
    public float DiscountPercent;   // 展示折扣数字（10=9折）
    
    [Header("限购")]
    public int DailyLimit;          // 每日限购次数（0=无限）
    public int WeeklyLimit;         // 每周限购次数
    public int TotalLimit;          // 总限购次数（0=无限）
    public bool IsFirstBuyOnly;     // 仅首购
    
    [Header("时效")]
    public long SaleStartTime;      // 折扣开始时间
    public long SaleEndTime;        // 折扣结束时间（0=永久）
    
    [Header("标签")]
    public bool IsNew;
    public bool IsHot;
    public bool IsLimited;
    
    [Header("内容")]
    public ShopItemContent[] Contents;  // 礼包内容
    
    public bool IsOnSale()
    {
        if (SalePrice <= 0) return false;
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        if (SaleStartTime > 0 && now < SaleStartTime) return false;
        if (SaleEndTime > 0 && now > SaleEndTime) return false;
        return true;
    }
    
    public long GetCurrentPrice() => IsOnSale() ? SalePrice : OriginalPrice;
}

[Serializable]
public class ShopItemContent
{
    public string ItemId;
    public int Count;
}
```

## 二、商城管理器

```csharp
public class ShopManager : MonoBehaviour
{
    private static ShopManager instance;
    public static ShopManager Instance => instance;
    
    private List<ShopItem> shopItems = new List<ShopItem>();
    private Dictionary<string, int> purchaseCount = new Dictionary<string, int>();
    
    public event Action OnShopUpdated;

    void Awake() { instance = this; }

    public async System.Threading.Tasks.Task LoadShop()
    {
        shopItems = await NetworkService.GetShopItems();
        purchaseCount = await NetworkService.GetPurchaseHistory();
        OnShopUpdated?.Invoke();
    }

    public int GetRemainingPurchases(ShopItem item)
    {
        int bought = purchaseCount.TryGetValue(item.ItemId, out int n) ? n : 0;
        
        // 取最严格的限制
        if (item.TotalLimit > 0 && bought >= item.TotalLimit) return 0;
        if (item.IsFirstBuyOnly && bought > 0) return 0;
        
        if (item.DailyLimit > 0)
        {
            int todayBought = GetTodayPurchaseCount(item.ItemId);
            if (todayBought >= item.DailyLimit) return 0;
            return item.DailyLimit - todayBought;
        }
        
        if (item.WeeklyLimit > 0)
        {
            int weekBought = GetWeekPurchaseCount(item.ItemId);
            return Math.Max(0, item.WeeklyLimit - weekBought);
        }
        
        return item.TotalLimit > 0 ? item.TotalLimit - bought : int.MaxValue;
    }

    public async System.Threading.Tasks.Task<bool> Purchase(ShopItem item, int quantity = 1)
    {
        if (GetRemainingPurchases(item) < quantity)
        {
            UIManager.Instance?.ShowMessage("已达到购买上限");
            return false;
        }
        
        // 检查货币
        long totalCost = item.GetCurrentPrice() * quantity;
        if (!CheckBalance(item.PriceType, totalCost)) return false;
        
        // 显示确认弹窗
        bool confirmed = await UIManager.Instance?.ShowPurchaseConfirm(item, quantity, totalCost);
        if (!confirmed) return false;
        
        // 服务端购买
        var result = await NetworkService.ShopPurchase(item.ItemId, quantity);
        if (result?.Success == true)
        {
            // 更新本地购买记录
            if (!purchaseCount.ContainsKey(item.ItemId)) purchaseCount[item.ItemId] = 0;
            purchaseCount[item.ItemId] += quantity;
            
            UIManager.Instance?.ShowPurchaseSuccess(item, quantity);
            OnShopUpdated?.Invoke();
        }
        
        return result?.Success == true;
    }

    bool CheckBalance(PriceType type, long amount)
    {
        switch (type)
        {
            case PriceType.Gold:
                return CurrencyManager.Instance.HasEnough(CurrencyType.Gold, amount);
            case PriceType.Gem:
                return CurrencyManager.Instance.HasEnough(CurrencyType.Gem, amount);
        }
        return true;
    }

    int GetTodayPurchaseCount(string itemId) => 0;   // 需要追踪每日购买
    int GetWeekPurchaseCount(string itemId) => 0;    // 需要追踪每周购买

    public List<ShopItem> GetItemsByType(ShopItemType type) =>
        shopItems.FindAll(i => i.Type == type);
    
    public List<ShopItem> GetSaleItems() =>
        shopItems.FindAll(i => i.IsOnSale());
}
```

## 三、商城转化率优化

| 策略 | 实现 | 提升幅度 |
|------|------|---------|
| 限时折扣 | 倒计时显示，制造紧迫感 | +15-30% |
| 首购优惠 | 首次购买享受特别折扣 | +20%+ |
| 礼包捆绑 | 打包销售性价比高于单买 | +25% |
| 推荐算法 | 展示玩家可能需要的商品 | +10% |
| 社会证明 | "X人已购买" 标签 | +8% |
| 库存稀缺感 | "仅剩X件" 标签 | +12% |
