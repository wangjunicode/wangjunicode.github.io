# 游戏存档系统设计：从本地存档到云同步

存档系统是游戏的"记忆"。设计差的存档系统会导致存档损坏、数据丢失、版本升级崩溃等严重问题，直接影响玩家体验和用户留存。本文系统讲解从本地存档到云同步的完整技术方案。

## 一、存档系统架构设计

```
存档系统分层：

应用层（游戏逻辑）
  ↓ 调用 ISaveService 接口
存档服务层
  ├── LocalSaveService  （本地文件）
  ├── CloudSaveService  （云端同步）
  └── HybridSaveService （本地+云端混合）
  ↓
序列化层
  ├── JsonSerializer    （可读，调试方便）
  ├── BinarySerializer  （性能好，体积小）
  └── MessagePack       （二进制+类型安全，推荐）
  ↓
存储层
  ├── PlayerPrefs       （简单键值，小数据）
  ├── 文件系统          （JSON/二进制文件）
  └── SQLite            （关系型，复杂查询）
```

    }
    
    public bool VerifyChecksum()
    {

    }

## 三、本地存档实现

### 3.1 存档服务实现

```csharp
public class LocalSaveService : ISaveService
{
    private string SaveDirectory => 
        Path.Combine(Application.persistentDataPath, "SaveData");
    
    private readonly ISerializer _serializer;
    private readonly IEncryptor _encryptor;
    
    public LocalSaveService(ISerializer serializer, IEncryptor encryptor = null)
    {
        _serializer = serializer;
        _encryptor = encryptor;
        Directory.CreateDirectory(SaveDirectory);
    }
    
    // 保存存档
    public async Task SaveAsync(SaveFile saveData, int slotIndex)
    {
        try
        {
            saveData.Metadata.SaveTime = DateTime.UtcNow;
            saveData.Metadata.SlotIndex = slotIndex;
            saveData.GenerateChecksum();
            
            byte[] data = _serializer.Serialize(saveData);
            
            // 可选加密
            if (_encryptor != null)
                data = _encryptor.Encrypt(data);
            
            string filePath = GetSavePath(slotIndex);
            
            // 原子写入：先写临时文件，成功后替换（防止写入中断导致损坏）
            string tempPath = filePath + ".tmp";
            await File.WriteAllBytesAsync(tempPath, data);
            
            // 保留备份
            if (File.Exists(filePath))
                File.Copy(filePath, filePath + ".bak", overwrite: true);
            
            File.Move(tempPath, filePath, overwrite: true);
            
            Debug.Log($"存档槽 {slotIndex} 保存成功");
        }
        catch (Exception e)
        {
            Debug.LogError($"存档失败: {e.Message}");
            throw new SaveException("存档写入失败", e);
        }
    }
    
    // 加载存档
    public async Task<SaveFile> LoadAsync(int slotIndex)
    {
        string filePath = GetSavePath(slotIndex);
        
        if (!File.Exists(filePath))
            return null;
        
        try
        {
            byte[] data = await File.ReadAllBytesAsync(filePath);
            
            if (_encryptor != null)
                data = _encryptor.Decrypt(data);
            
            var saveFile = _serializer.Deserialize<SaveFile>(data);
            
            // 验证完整性
            if (!saveFile.VerifyChecksum())
            {
                Debug.LogWarning($"存档槽 {slotIndex} 校验失败，尝试恢复备份...");
                return await LoadFromBackup(slotIndex);
            }
            
            // 版本迁移
            saveFile = await MigrateIfNeeded(saveFile);
            
            return saveFile;
        }
        catch (Exception e)
        {
            Debug.LogError($"加载存档失败: {e.Message}，尝试备份...");
            return await LoadFromBackup(slotIndex);
        }
    }
    
    // 从备份恢复
    private async Task<SaveFile> LoadFromBackup(int slotIndex)
    {
        string backupPath = GetSavePath(slotIndex) + ".bak";
        if (!File.Exists(backupPath)) return null;
        
        byte[] data = await File.ReadAllBytesAsync(backupPath);
        if (_encryptor != null) data = _encryptor.Decrypt(data);
        
        return _serializer.Deserialize<SaveFile>(data);
    }
    
    private string GetSavePath(int slotIndex) 
        => Path.Combine(SaveDirectory, $"save_{slotIndex:D2}.dat");
}
```

### 3.2 存档加密

```csharp
public class AESEncryptor : IEncryptor
{
    // 密钥要保密！不能硬编码明文，建议运行时生成并用设备唯一ID保护
    private readonly byte[] _key;
    private readonly byte[] _iv;
    
    public AESEncryptor()
    {
        // 基于设备唯一标识生成密钥（单机反改档）
        string deviceId = SystemInfo.deviceUniqueIdentifier;
        using var sha256 = SHA256.Create();
        _key = sha256.ComputeHash(Encoding.UTF8.GetBytes(deviceId + "SaveKey_V1"));
        _iv = _key[..16]; // 取前16字节作为IV
    }
    
    public byte[] Encrypt(byte[] data)
    {
        using var aes = Aes.Create();
        aes.Key = _key;
        aes.IV = _iv;
        
        using var ms = new MemoryStream();
        using var cs = new CryptoStream(ms, aes.CreateEncryptor(), CryptoStreamMode.Write);
        cs.Write(data, 0, data.Length);
        cs.FlushFinalBlock();
        return ms.ToArray();
    }
    
    public byte[] Decrypt(byte[] data)
    {
        using var aes = Aes.Create();
        aes.Key = _key;
        aes.IV = _iv;
        
        using var ms = new MemoryStream(data);
        using var cs = new CryptoStream(ms, aes.CreateDecryptor(), CryptoStreamMode.Read);
        using var result = new MemoryStream();
        cs.CopyTo(result);
        return result.ToArray();
    }
}
```

## 四、存档版本迁移

游戏迭代时，存档格式可能会变化，必须处理旧版存档的兼容性：

```csharp
public class SaveMigrationManager
{
    private const int CURRENT_VERSION = 3;
    
    public async Task<SaveFile> MigrateIfNeeded(SaveFile save)
    {
        int version = save.Metadata.Version;
        
        if (version == CURRENT_VERSION) return save;
        
        Debug.Log($"存档版本 {version} → {CURRENT_VERSION}，开始迁移...");
        
        // 链式迁移：v1→v2→v3
        if (version < 2) save = await MigrateV1ToV2(save);
        if (version < 3) save = await MigrateV2ToV3(save);
        
        save.Metadata.Version = CURRENT_VERSION;
        save.GenerateChecksum();
        
        return save;
    }
    
    // v1 → v2：新增云同步字段，背包格式重构
    private Task<SaveFile> MigrateV1ToV2(SaveFile save)
    {
        Debug.Log("迁移 v1 → v2");
        
        // 假设 v2 新增了装备栏系统
        if (save.Player.Inventory == null)
            save.Player.Inventory = new List<ItemSaveData>();
        
        // 将旧格式物品转换到新格式
        // save.Player.OldItemIds → save.Player.Inventory
        
        return Task.FromResult(save);
    }
    
    // v2 → v3：技能系统重构
    private Task<SaveFile> MigrateV2ToV3(SaveFile save)
    {
        Debug.Log("迁移 v2 → v3");
        
        // 技能ID体系变化：旧ID * 100 = 新ID
        if (save.Player.UnlockedSkillIds != null)
        {
            save.Player.UnlockedSkillIds = save.Player.UnlockedSkillIds
                .Select(id => id * 100)
                .ToList();
        }
        
        return Task.FromResult(save);
    }
}
```

## 五、云存档同步

```csharp
public class CloudSaveService : ISaveService
{
    private readonly LocalSaveService _local;
    private readonly ICloudBackend _cloud; // 可对接 Firebase/自建服务器
    
    // 云端优先策略：本地快速响应，后台同步
    public async Task<SaveFile> LoadAsync(int slotIndex)
    {
        // 1. 立即返回本地存档
        var localSave = await _local.LoadAsync(slotIndex);
        
        // 2. 后台拉取云端存档
        _ = SyncFromCloudAsync(slotIndex, localSave);
        
        return localSave;
    }
    
    private async Task SyncFromCloudAsync(int slotIndex, SaveFile localSave)
    {
        try
        {
            var cloudSave = await _cloud.FetchSaveAsync(slotIndex);
            if (cloudSave == null) return;
            
            // 冲突解决：比较时间戳
            if (cloudSave.Metadata.SaveTime > localSave?.Metadata.SaveTime)
            {
                // 云端更新，询问玩家
                var choice = await ShowConflictDialog(localSave, cloudSave);
                if (choice == ConflictResolution.UseCloud)
                {
                    await _local.SaveAsync(cloudSave, slotIndex);
                    OnCloudSaveApplied?.Invoke(cloudSave);
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogWarning($"云存档同步失败（将使用本地存档）: {e.Message}");
        }
    }
    
    public async Task SaveAsync(SaveFile saveData, int slotIndex)
    {
        // 1. 先保存本地（保证可靠性）
        await _local.SaveAsync(saveData, slotIndex);
        
        // 2. 异步上传云端（不阻塞游戏）
        _ = UploadToCloudAsync(saveData, slotIndex);
    }
    
    private async Task UploadToCloudAsync(SaveFile saveData, int slotIndex)
    {
        int retryCount = 3;
        while (retryCount > 0)
        {
            try
            {
                await _cloud.UploadSaveAsync(saveData, slotIndex);
                return;
            }
            catch
            {
                retryCount--;
                await Task.Delay(1000 * (4 - retryCount)); // 指数退避
            }
        }
        Debug.LogWarning("云存档上传失败，将在下次启动时重试");
    }
}
```

## 六、自动存档策略

```csharp
public class AutoSaveManager : MonoBehaviour
{
    [SerializeField] private float _autoSaveInterval = 300f; // 每5分钟
    [SerializeField] private int _autoSaveSlot = 0; // 自动存档用第0槽
    
    private float _timeSinceLastSave;
    private bool _isSaving;
    
    // 触发自动存档的时机
    private void Update()
    {
        _timeSinceLastSave += Time.deltaTime;
        
        if (_timeSinceLastSave >= _autoSaveInterval)
        {
            TriggerAutoSave("定时自动存档");
        }
    }
    
    // 场景切换时存档
    private void OnSceneChanging(string nextScene)
    {
        TriggerAutoSave($"场景切换至 {nextScene}");
    }
    
    // 过场动画后存档（通关检查点）
    public void OnCheckpointReached(string checkpointId)
    {
        TriggerAutoSave($"抵达存档点 {checkpointId}");
    }
    
    private async void TriggerAutoSave(string reason)
    {
        if (_isSaving) return;
        
        _isSaving = true;
        _timeSinceLastSave = 0;
        
        // 显示存档图标（不打断游戏）
        UIManager.Instance.ShowAutoSaveIcon();
        
        try
        {
            var saveData = GameDataCollector.Instance.CollectAllData();
            saveData.Metadata.DisplayName = $"自动存档 - {reason}";
            await SaveSystem.Instance.SaveAsync(saveData, _autoSaveSlot);
        }
        finally
        {
            _isSaving = false;
            UIManager.Instance.HideAutoSaveIcon();
        }
    }
}
```

## 七、存档系统设计清单

| 要点 | 实现方案 |
|------|----------|
| 原子写入 | 先写临时文件，成功后替换 |
| 数据备份 | 每次保存前创建 .bak 备份文件 |
| 完整性校验 | MD5/SHA256 校验码 |
| 防篡改 | AES 加密（单机）/ 服务端验证（联网） |
| 版本迁移 | 链式迁移函数 + 版本号管理 |
| 云同步 | 本地优先 + 后台上传 + 冲突解决 |
| 自动存档 | 定时 + 检查点 + 场景切换 |
| 截图预览 | 存档时截图，选档界面展示 |

> 💡 **最重要的原则**：**永远不要让玩家丢失进度**。存档系统必须极度保守：先写本地再写云端，先备份再覆盖，有疑问就保留两份让玩家选。宁可多占几 MB 磁盘空间，也不能让玩家的十几个小时游戏时间付诸东流。
