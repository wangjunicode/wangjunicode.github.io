## 一、属性体系设计

```
基础属性（Primary Stats）
├── 力量（Strength）
├── 敏捷（Agility）
├── 智力（Intelligence）
├── 体力（Vitality）
└── 幸运（Luck）

衍生属性（Derived Stats）
├── 生命值   = 体力 × 10 + 装备加成
├── 攻击力   = 力量 × 2 + 武器加成
├── 防御力   = 体力 × 1.5 + 护甲加成
├── 暴击率   = 敏捷 × 0.1% + 额外加成
├── 暴击倍率 = 150% + 额外加成
├── 攻击速度 = 基础速度 × (1 + 敏捷×0.001)
└── 魔法回复 = 智力 × 0.5 + 额外加成
```

## 二、属性系统实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 属性ID定义（避免魔法字符串）
/// </summary>
public static class StatId
{
    // 基础属性
    public const string Strength    = "strength";
    public const string Agility     = "agility";
    public const string Intelligence = "intelligence";
    public const string Vitality    = "vitality";
    public const string Luck        = "luck";
    
    // 战斗属性（衍生）
    public const string MaxHP       = "max_hp";
    public const string MaxMP       = "max_mp";
    public const string Attack      = "attack";
    public const string MagicAttack = "magic_attack";
    public const string Defense     = "defense";
    public const string MagicDef    = "magic_def";
    public const string CritRate    = "crit_rate";    // 暴击率 (0-1)
    public const string CritDamage  = "crit_damage";  // 暴击倍率 (1.5 = 150%)
    public const string AtkSpeed    = "atk_speed";    // 攻速
    public const string MoveSpeed   = "move_speed";
    public const string DodgeRate   = "dodge_rate";
    public const string LifeSteal   = "life_steal";   // 吸血率
}

/// <summary>
/// 属性修改器
/// </summary>
public class StatModifierV2
{
    public enum ModType
    {
        Flat,       // 加法：最先计算，所有Flat相加
        Percent,    // 乘法百分比：(1 + sum of all percents)
        Override    // 覆盖：最后计算，直接设定值
    }

    public string StatId;
    public float Value;
    public ModType Type;
    public string Source;           // 来源（装备ID/技能ID/Buff ID）
    public int Order;               // 相同类型内的计算顺序

    public StatModifierV2(string statId, float value, ModType type, string source, int order = 0)
    {
        StatId = statId;
        Value = value;
        Type = type;
        Source = source;
        Order = order;
    }
}

/// <summary>
/// 角色属性容器（支持多层修改器叠加）
/// </summary>
public class CharacterStats : MonoBehaviour
{
    [Header("基础属性（角色原始值）")]
    [SerializeField] private Dictionary<string, float> baseStats 
        = new Dictionary<string, float>();
    
    // 修改器列表（装备/Buff/技能等）
    private List<StatModifierV2> modifiers = new List<StatModifierV2>();
    
    // 缓存的最终属性值（避免每次查询都重算）
    private Dictionary<string, float> cachedStats = new Dictionary<string, float>();
    private bool isDirty = true;
    
    public event Action<string, float, float> OnStatChanged; // statId, oldValue, newValue

    void Awake()
    {
        InitializeBaseStats();
    }

    void InitializeBaseStats()
    {
        // 角色基础属性初始值
        baseStats[StatId.Strength]     = 10f;
        baseStats[StatId.Agility]      = 10f;
        baseStats[StatId.Intelligence] = 10f;
        baseStats[StatId.Vitality]     = 10f;
        baseStats[StatId.Luck]         = 5f;
        
        // 衍生属性（会被公式动态计算）
        MarkDirty();
    }

    // ============ 修改器管理 ============

    public void AddModifier(StatModifierV2 modifier)
    {
        modifiers.Add(modifier);
        MarkDirty();
        NotifyChange(modifier.StatId);
    }

    public void RemoveModifier(StatModifierV2 modifier)
    {
        if (modifiers.Remove(modifier))
        {
            MarkDirty();
            NotifyChange(modifier.StatId);
        }
    }

    public void RemoveModifiersBySource(string source)
    {
        int removed = modifiers.RemoveAll(m => m.Source == source);
        if (removed > 0)
        {
            MarkDirty();
            // 可能影响多个属性，全部触发
        }
    }

    void MarkDirty() => isDirty = true;

    // ============ 属性计算 ============

    /// <summary>
    /// 获取最终属性值（带缓存）
    /// </summary>
    public float Get(string statId)
    {
        if (isDirty)
        {
            RecalculateAll();
            isDirty = false;
        }
        
        return cachedStats.TryGetValue(statId, out float val) ? val : 0f;
    }

    void RecalculateAll()
    {
        var old = new Dictionary<string, float>(cachedStats);
        
        // 1. 计算基础属性（含Flat修改器）
        foreach (var kv in baseStats)
        {
            cachedStats[kv.Key] = CalculateStat(kv.Key, kv.Value);
        }
        
        // 2. 计算衍生属性（依赖基础属性）
        RecalculateDerivedStats();
        
        // 3. 触发变化事件
        foreach (var kv in cachedStats)
        {
            float oldVal = old.TryGetValue(kv.Key, out float o) ? o : 0f;
            if (Mathf.Abs(kv.Value - oldVal) > 0.001f)
                OnStatChanged?.Invoke(kv.Key, oldVal, kv.Value);
        }
    }

    float CalculateStat(string statId, float baseValue)
    {
        // 获取该属性的所有修改器
        float flatBonus = 0f;
        float percentBonus = 0f;
        bool hasOverride = false;
        float overrideValue = 0f;
        
        // 排序后计算（确保Order正确）
        var relevantMods = modifiers
            .FindAll(m => m.StatId == statId)
            .ToArray();
        
        foreach (var mod in relevantMods)
        {
            switch (mod.Type)
            {
                case StatModifierV2.ModType.Flat:
                    flatBonus += mod.Value;
                    break;
                case StatModifierV2.ModType.Percent:
                    percentBonus += mod.Value;
                    break;
                case StatModifierV2.ModType.Override:
                    hasOverride = true;
                    overrideValue = mod.Value;
                    break;
            }
        }
        
        if (hasOverride) return overrideValue;
        
        // 计算顺序：(base + flat) * (1 + percent)
        float result = (baseValue + flatBonus) * (1f + percentBonus);
        
        // 应用上下限
        return ApplyClamp(statId, result);
    }

    void RecalculateDerivedStats()
    {
        float str = cachedStats.TryGetValue(StatId.Strength, out float s) ? s : 10f;
        float agi = cachedStats.TryGetValue(StatId.Agility, out float a) ? a : 10f;
        float intel = cachedStats.TryGetValue(StatId.Intelligence, out float i) ? i : 10f;
        float vit = cachedStats.TryGetValue(StatId.Vitality, out float v) ? v : 10f;
        float luk = cachedStats.TryGetValue(StatId.Luck, out float l) ? l : 5f;
        
        // 衍生属性公式（基础部分，再叠加修改器）
        float baseMaxHP = vit * 10f + 100f;
        float baseAtk = str * 2f;
        float baseDef = vit * 1.5f;
        float baseCritRate = agi * 0.001f + luk * 0.002f;
        float baseMoveSpeed = 5f + agi * 0.02f;
        
        cachedStats[StatId.MaxHP]      = CalculateStat(StatId.MaxHP, baseMaxHP);
        cachedStats[StatId.Attack]     = CalculateStat(StatId.Attack, baseAtk);
        cachedStats[StatId.Defense]    = CalculateStat(StatId.Defense, baseDef);
        cachedStats[StatId.CritRate]   = CalculateStat(StatId.CritRate, baseCritRate);
        cachedStats[StatId.CritDamage] = CalculateStat(StatId.CritDamage, 1.5f);
        cachedStats[StatId.MoveSpeed]  = CalculateStat(StatId.MoveSpeed, baseMoveSpeed);
        cachedStats[StatId.DodgeRate]  = CalculateStat(StatId.DodgeRate, agi * 0.0005f);
    }

    float ApplyClamp(string statId, float value)
    {
        return statId switch
        {
            StatId.CritRate   => Mathf.Clamp(value, 0f, 0.95f), // 暴击率最高95%
            StatId.DodgeRate  => Mathf.Clamp(value, 0f, 0.75f), // 闪避最高75%
            StatId.MoveSpeed  => Mathf.Max(value, 1f),            // 速度最低1
            _ => Mathf.Max(value, 0f)                             // 不允许负值
        };
    }

    void NotifyChange(string statId)
    {
        float old = cachedStats.TryGetValue(statId, out float o) ? o : 0f;
        float newVal = Get(statId);
        if (Mathf.Abs(newVal - old) > 0.001f)
            OnStatChanged?.Invoke(statId, old, newVal);
    }

    // ============ 快捷接口 ============

    public int MaxHP         => Mathf.RoundToInt(Get(StatId.MaxHP));
    public float Attack      => Get(StatId.Attack);
    public float Defense     => Get(StatId.Defense);
    public float CritRate    => Get(StatId.CritRate);
    public float CritDamage  => Get(StatId.CritDamage);
    public float MoveSpeed   => Get(StatId.MoveSpeed);

    public void SetBase(string statId, float value)
    {
        baseStats[statId] = value;
        MarkDirty();
    }
}
```

## 三、伤害计算公式

```csharp
/// <summary>
/// 战斗数值计算器
/// </summary>
public static class CombatMath
{
    /// <summary>
    /// 计算最终伤害
    /// </summary>
    public static float CalculateDamage(CharacterStats attacker, CharacterStats defender,
        float skillMultiplier = 1f, bool forceCrit = false)
    {
        float attack = attacker.Attack;
        float defense = defender.Defense;
        
        // 基础伤害公式：攻击力 / (攻击力 + 防御力) × 攻击力
        // 这个公式的特点：防御永不能减免100%，有上限
        float baseDamage = attack * attack / (attack + defense);
        
        // 技能倍率
        baseDamage *= skillMultiplier;
        
        // 暴击判断
        bool isCrit = forceCrit || UnityEngine.Random.value < attacker.CritRate;
        if (isCrit)
            baseDamage *= attacker.CritDamage;
        
        // 随机浮动（±10%）
        baseDamage *= UnityEngine.Random.Range(0.9f, 1.1f);
        
        // 最小伤害1点
        return Mathf.Max(1f, Mathf.RoundToInt(baseDamage));
    }

    /// <summary>
    /// 计算理论 TTK（Time to Kill）
    /// </summary>
    public static float CalculateTTK(CharacterStats attacker, CharacterStats defender,
        float attacksPerSecond)
    {
        float avgDamagePerHit = CalculateDamage(attacker, defender);
        // 暴击期望值修正
        float avgDamage = avgDamagePerHit * 
            (1 + (attacker.CritDamage - 1) * attacker.CritRate);
        
        float dps = avgDamage * attacksPerSecond;
        return defender.MaxHP / dps;
    }
}
```

## 四、不同 RPG 类型属性对比

| RPG类型 | 核心属性 | 特点 |
|---------|----------|------|
| 动作RPG（暗黑类）| 技能伤害/冷却/暴击 | 属性多，数值大，计算复杂 |
| MMORPG | 面板属性 + 评分 | 追求属性提升感 |
| 卡牌RPG | 攻击/生命/技能 | 简洁，易于理解 |
| 战棋RPG | 移动力/射程 | 战术属性更重要 |
| Roguelike | 随机属性/词缀 | 组合构建乐趣 |

**平衡建议：**
1. 防御减伤公式建议使用非线性（如 `def / (def + C)` 形式），避免防御堆叠后无敌
2. 暴击率有上限（95%）保留一定不确定性
3. 衍生属性用公式计算，不要直接配置，方便调参
4. 定期用Excel模拟各级别的DPS/TTK，确保数值曲线合理
