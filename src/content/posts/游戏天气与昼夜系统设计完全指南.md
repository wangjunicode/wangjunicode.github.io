## 一、昼夜系统核心架构

```

天体运动   → 太阳/月亮旋转、星空显示

天空盒     → 动态天空盒混合、云层
后处理     → 曝光、色调映射、Bloom
```

## 二、时间管理器

```csharp
using System;
using UnityEngine;

/// <summary>
/// 游戏时间管理器
/// </summary>
public class GameTimeManager : MonoBehaviour
{
    private static GameTimeManager instance;
    public static GameTimeManager Instance => instance;

    [Header("时间配置")]
    [SerializeField] private float realSecondsPerGameHour = 120f; // 现实2分钟 = 游戏1小时
    [SerializeField] private float startHour = 8f;                // 初始时间（24小时制）
    [SerializeField] private bool autoAdvanceTime = true;
    [SerializeField, Range(0.1f, 100f)] private float timeScale = 1f;

    // 当前游戏时间（0-24小时）
    private float currentHour;
    private int currentDay;
    private DayOfWeek currentDayOfWeek;
    private Season currentSeason;

    // 事件
    public event Action<float> OnHourChanged;     // 小时变化（0-23整数）
    public event Action<int> OnDayChanged;         // 天数变化
    public event Action<Season> OnSeasonChanged;   // 季节变化

    public float CurrentHour => currentHour;
    public int CurrentDay => currentDay;
    public Season CurrentSeason => currentSeason;
    
    // 时间标准化（0-1，表示一天内的进度）
    public float DayProgress => currentHour / 24f;
    
    // 是否白天（6点到18点）
    public bool IsDaytime => currentHour >= 6f && currentHour < 18f;

    void Awake()
    {
        instance = this;
        currentHour = startHour;
    }

    void Update()
    {
        if (!autoAdvanceTime) return;
        
        float prevHour = currentHour;
        float advance = (1f / realSecondsPerGameHour) * Time.deltaTime * timeScale;
        currentHour += advance * 24f; // 换算成小时
        
        // 满24小时换天
        if (currentHour >= 24f)
        {
            currentHour -= 24f;
            currentDay++;
            UpdateSeason();
            OnDayChanged?.Invoke(currentDay);
        }
        
        // 整小时触发事件
        int prevHourInt = Mathf.FloorToInt(prevHour);
        int currHourInt = Mathf.FloorToInt(currentHour);
        if (currHourInt != prevHourInt)
        {
            OnHourChanged?.Invoke(currHourInt);
        }
    }

    void UpdateSeason()
    {
        int dayOfYear = currentDay % 360; // 简化：一年360天
        Season newSeason = dayOfYear switch
        {
            < 90 => Season.Spring,
            < 180 => Season.Summer,
            < 270 => Season.Autumn,
            _ => Season.Winter
        };
        
        if (newSeason != currentSeason)
        {
            currentSeason = newSeason;
            OnSeasonChanged?.Invoke(currentSeason);
        }
    }

    /// <summary>
    /// 快速设置时间（用于测试或剧情需要）
    /// </summary>
    public void SetTime(float hour)
    {
        currentHour = Mathf.Clamp(hour, 0f, 24f);
    }

    public string GetTimeString() => 
        $"{Mathf.FloorToInt(currentHour):D2}:{Mathf.FloorToInt((currentHour % 1f) * 60):D2}";
}

public enum Season { Spring, Summer, Autumn, Winter }
```

## 三、太阳与月亮控制器

```csharp
/// <summary>
/// 天体运动控制器
/// </summary>
public class CelestialBodyController : MonoBehaviour
{
    [Header("太阳")]
    [SerializeField] private Light sunLight;
    [SerializeField] private Transform sunPivot;
    [SerializeField] private Gradient sunColor;        // 颜色随时间变化
    [SerializeField] private AnimationCurve sunIntensity; // 强度随时间变化
    [SerializeField] private float maxSunIntensity = 1.5f;

    [Header("月亮")]
    [SerializeField] private Light moonLight;
    [SerializeField] private Transform moonPivot;
    [SerializeField] private float moonIntensity = 0.3f;

    [Header("天空")]
    [SerializeField] private Material skyboxMaterial;  // 程序化天空盒
    [SerializeField] private Gradient ambientColor;
    [SerializeField] private Gradient fogColor;

    [Header("星空")]
    [SerializeField] private GameObject starsParticle;
    [SerializeField] private AnimationCurve starVisibility; // 星空可见度曲线

    private GameTimeManager timeManager;

    void Start()
    {
        timeManager = GameTimeManager.Instance;
        timeManager.OnHourChanged += _ => UpdateCelestials();
        UpdateCelestials();
    }

    void Update()
    {
        // 每帧更新，确保平滑过渡
        UpdateCelestials();
    }

    void UpdateCelestials()
    {
        float t = timeManager.DayProgress; // 0-1，一天进度
        
        UpdateSun(t);
        UpdateMoon(t);
        UpdateAmbient(t);
        UpdateFog(t);
        UpdateStars(t);
        UpdateSkybox(t);
    }

    void UpdateSun(float t)
    {
        if (sunLight == null) return;
        
        // 太阳从东方升起（t=0.25 日出，t=0.75 日落）
        // 旋转范围：-90°（午夜地平线）到 90°（正午）到 -90°（次日）
        float sunAngle = (t - 0.25f) * 360f; // 将一天的进度映射到360度
        sunPivot.localRotation = Quaternion.Euler(sunAngle, 0, 0);
        
        // 颜色（日出日落橙红，正午白色）
        sunLight.color = sunColor.Evaluate(t);
        
        // 强度（夜间为0，白天根据曲线变化）
        float intensityCurve = sunIntensity.Evaluate(t);
        sunLight.intensity = intensityCurve * maxSunIntensity;
        
        // 是否开启阴影（强度足够时才开启，节省性能）
        sunLight.shadows = sunLight.intensity > 0.1f ? 
            LightShadows.Soft : LightShadows.None;
    }

    void UpdateMoon(float t)
    {
        if (moonLight == null) return;
        
        // 月亮与太阳相差180度
        float moonAngle = (t - 0.25f + 0.5f) * 360f;
        moonPivot.localRotation = Quaternion.Euler(moonAngle, 0, 0);
        
        // 夜间才显示月光
        bool isNight = t < 0.25f || t > 0.75f;
        moonLight.intensity = isNight ? moonIntensity : 0f;
    }

    void UpdateAmbient(float t)
    {
        RenderSettings.ambientLight = ambientColor.Evaluate(t);
    }

    void UpdateFog(float t)
    {
        RenderSettings.fogColor = fogColor.Evaluate(t);
        
        // 清晨薄雾，白天减淡，夜晚加重
        float fogDensity = t switch
        {
            float x when x < 0.2f => Mathf.Lerp(0.02f, 0.005f, x / 0.2f),  // 清晨薄雾
            float x when x < 0.8f => 0.005f,                                  // 白天清澈
            _ => Mathf.Lerp(0.005f, 0.015f, (t - 0.8f) / 0.2f)              // 傍晚加雾
        };
        
        RenderSettings.fogDensity = fogDensity;
    }

    void UpdateStars(float t)
    {
        if (starsParticle == null) return;
        
        float visibility = starVisibility.Evaluate(t);
        var renderer = starsParticle.GetComponent<ParticleSystemRenderer>();
        if (renderer != null)
        {
            Color starColor = renderer.material.color;
            starColor.a = visibility;
            renderer.material.color = starColor;
        }
        starsParticle.SetActive(visibility > 0.01f);
    }

    void UpdateSkybox(float t)
    {
        if (skyboxMaterial == null) return;
        
        // 更新程序化天空盒参数
        skyboxMaterial.SetFloat("_AtmosphereThickness", 
            Mathf.Lerp(0.8f, 1.2f, Mathf.Sin(t * Mathf.PI)));
        skyboxMaterial.SetFloat("_Exposure", 
            Mathf.Lerp(0.3f, 1.5f, sunLight.intensity / maxSunIntensity));
    }
}
```

## 四、天气系统

```csharp
/// <summary>
/// 天气类型
/// </summary>
public enum WeatherType
{
    Clear,      // 晴天
    Overcast,   // 阴天
    LightRain,  // 小雨
    HeavyRain,  // 大雨
    Thunderstorm, // 雷暴
    LightSnow,  // 小雪
    HeavySnow,  // 大雪
    Fog,        // 雾
    Sandstorm   // 沙尘暴
}

/// <summary>
/// 天气配置
/// </summary>
[CreateAssetMenu(fileName = "WeatherConfig", menuName = "Game/Weather Config")]
public class WeatherConfig : ScriptableObject
{
    public WeatherType Type;
    
    [Header("视觉参数")]
    public Color ambientColorMultiplier = Color.white;
    public float fogDensityMultiplier = 1f;
    public Color fogColorTint = Color.white;
    public float sunIntensityMultiplier = 1f;
    
    [Header("粒子效果")]
    public GameObject precipitationPrefab;   // 降水粒子系统
    public float precipitationIntensity;     // 降水量
    
    [Header("音效")]
    public string ambientSoundKey;          // 环境音效ID
    public float ambientSoundVolume = 1f;
    
    [Header("玩法影响")]
    public float moveSpeedMultiplier = 1f;  // 移动速度影响
    public float visibilityRange = 100f;    // 能见度
    public bool disableFlying;              // 是否禁止飞行
    
    [Header("随机转换")]
    public WeatherTransitionRule[] possibleTransitions; // 可能的天气转换
}

[Serializable]
public class WeatherTransitionRule
{
    public WeatherType NextWeather;
    public float Probability;             // 每小时转换概率
    public float MinDuration;             // 最短持续时间（分钟）
}

/// <summary>
/// 天气系统管理器
/// </summary>
public class WeatherSystem : MonoBehaviour
{
    private static WeatherSystem instance;
    public static WeatherSystem Instance => instance;

    [Header("天气配置")]
    [SerializeField] private WeatherConfig[] weatherConfigs;
    [SerializeField] private WeatherType startWeather = WeatherType.Clear;
    [SerializeField] private float transitionDuration = 30f; // 天气切换过渡时间（秒）
    
    [Header("特效")]
    [SerializeField] private Transform precipitationParent;
    [SerializeField] private ParticleSystem rainSystem;
    [SerializeField] private ParticleSystem snowSystem;
    [SerializeField] private ParticleSystem fogSystem;

    private WeatherType currentWeather;
    private WeatherType targetWeather;
    private float transitionProgress;
    private float currentWeatherDuration;
    private WeatherConfig currentConfig;
    private WeatherConfig targetConfig;
    
    private GameObject activePrecipitation;
    
    public event Action<WeatherType, WeatherType> OnWeatherChanging;
    public event Action<WeatherType> OnWeatherChanged;
    
    public WeatherType CurrentWeather => currentWeather;

    void Awake()
    {
        instance = this;
        SetWeather(startWeather, immediate: true);
    }

    void Start()
    {
        GameTimeManager.Instance.OnHourChanged += OnHourChanged;
    }

    void Update()
    {
        if (currentWeather != targetWeather)
        {
            // 平滑过渡
            transitionProgress += Time.deltaTime / transitionDuration;
            transitionProgress = Mathf.Clamp01(transitionProgress);
            
            ApplyWeatherBlend(currentConfig, targetConfig, transitionProgress);
            
            if (transitionProgress >= 1f)
            {
                currentWeather = targetWeather;
                currentConfig = targetConfig;
                OnWeatherChanged?.Invoke(currentWeather);
            }
        }
        
        currentWeatherDuration += Time.deltaTime;
    }

    void OnHourChanged(float hour)
    {
        // 每小时检查天气转换
        if (currentConfig == null) return;
        
        foreach (var rule in currentConfig.possibleTransitions ?? Array.Empty<WeatherTransitionRule>())
        {
            if (currentWeatherDuration / 60f < rule.MinDuration) continue;
            
            if (UnityEngine.Random.value < rule.Probability)
            {
                SetWeather(rule.NextWeather);
                break;
            }
        }
    }

    public void SetWeather(WeatherType type, bool immediate = false)
    {
        if (type == currentWeather && !immediate) return;
        
        var newConfig = GetConfig(type);
        if (newConfig == null) return;
        
        OnWeatherChanging?.Invoke(currentWeather, type);
        
        targetWeather = type;
        targetConfig = newConfig;
        currentWeatherDuration = 0f;
        
        if (immediate)
        {
            currentWeather = type;
            currentConfig = newConfig;
            transitionProgress = 1f;
            ApplyWeatherBlend(newConfig, newConfig, 1f);
        }
        else
        {
            transitionProgress = 0f;
        }
        
        // 更新粒子系统
        UpdatePrecipitation(type);
        
        // 更新音效
        if (!string.IsNullOrEmpty(newConfig.ambientSoundKey))
            AudioManager.Instance?.PlayAmbient(newConfig.ambientSoundKey, newConfig.ambientSoundVolume);
    }

    void ApplyWeatherBlend(WeatherConfig from, WeatherConfig to, float t)
    {
        if (from == null || to == null) return;
        
        float fogDensity = Mathf.Lerp(
            RenderSettings.fogDensity * from.fogDensityMultiplier,
            RenderSettings.fogDensity * to.fogDensityMultiplier,
            t);
        
        // 注意：不要直接修改 RenderSettings.fogDensity，
        // 因为昼夜系统也在修改它，需要通过乘数的形式叠加
        WeatherFogMultiplier = Mathf.Lerp(from.fogDensityMultiplier, to.fogDensityMultiplier, t);
        WeatherSunMultiplier = Mathf.Lerp(from.sunIntensityMultiplier, to.sunIntensityMultiplier, t);
    }

    void UpdatePrecipitation(WeatherType type)
    {
        // 根据天气类型控制粒子系统
        if (rainSystem != null)
        {
            bool showRain = type is WeatherType.LightRain or WeatherType.HeavyRain 
                or WeatherType.Thunderstorm;
            
            var emission = rainSystem.emission;
            emission.enabled = showRain;
            
            if (showRain)
            {
                float rate = type == WeatherType.LightRain ? 500 :
                    type == WeatherType.HeavyRain ? 2000 : 5000;
                emission.rateOverTime = rate;
            }
        }
        
        if (snowSystem != null)
        {
            bool showSnow = type is WeatherType.LightSnow or WeatherType.HeavySnow;
            var emission = snowSystem.emission;
            emission.enabled = showSnow;
        }
    }

    WeatherConfig GetConfig(WeatherType type)
    {
        foreach (var config in weatherConfigs)
        {
            if (config.Type == type) return config;
        }
        return null;
    }

    // 供昼夜系统读取的天气修正系数
    public float WeatherFogMultiplier { get; private set; } = 1f;
    public float WeatherSunMultiplier { get; private set; } = 1f;
}
```

## 五、风场系统

```csharp
/// <summary>
/// 全局风场系统（影响草地、树木Shader）
/// </summary>
public class WindSystem : MonoBehaviour
{
    [SerializeField] private float baseWindStrength = 0.5f;
    [SerializeField] private float windGustFrequency = 0.3f;
    [SerializeField] private float windGustIntensity = 1.5f;
    
    private Vector3 windDirection = new Vector3(1, 0, 0.3f).normalized;
    private float currentWindStrength;
    
    // Shader全局变量
    private static readonly int WindDirectionId = Shader.PropertyToID("_WindDirection");
    private static readonly int WindStrengthId = Shader.PropertyToID("_WindStrength");

    void Update()
    {
        // 使用柏林噪声模拟阵风
        float gust = Mathf.PerlinNoise(Time.time * windGustFrequency, 0f);
        currentWindStrength = baseWindStrength + gust * windGustIntensity * 0.3f;
        
        // 同步天气强度
        if (WeatherSystem.Instance != null)
        {
            currentWindStrength *= WeatherSystem.Instance.CurrentWeather switch
            {
                WeatherType.Clear => 0.5f,
                WeatherType.HeavyRain => 1.5f,
                WeatherType.Thunderstorm => 2.5f,
                WeatherType.Sandstorm => 3f,
                _ => 1f
            };
        }
        
        // 全局Shader参数（所有使用风效果的Shader自动获取）
        Shader.SetGlobalVector(WindDirectionId, windDirection * currentWindStrength);
        Shader.SetGlobalFloat(WindStrengthId, currentWindStrength);
    }
}
```

## 六、天气对玩法的影响

```csharp
/// <summary>
/// 天气玩法效果控制器
/// </summary>
public class WeatherGameplayEffects : MonoBehaviour
{
    void Start()
    {
        WeatherSystem.Instance.OnWeatherChanged += OnWeatherChanged;
    }
    
    void OnWeatherChanged(WeatherType newWeather)
    {
        switch (newWeather)
        {
            case WeatherType.HeavyRain:
            case WeatherType.Thunderstorm:
                // 大雨：火焰技能减弱，雷系技能增强
                SkillSystem.Instance?.SetElementMultiplier("Fire", 0.5f);
                SkillSystem.Instance?.SetElementMultiplier("Thunder", 1.5f);
                // 地面湿滑：减少摩擦力
                PhysicsSystem.SetGlobalFriction(0.4f);
                break;
            
            case WeatherType.HeavySnow:
                // 大雪：移动速度下降，冰系技能增强
                PlayerStats.Instance?.AddModifier("MoveSpeed", 0.7f, "snow");
                SkillSystem.Instance?.SetElementMultiplier("Ice", 1.5f);
                break;
            
            case WeatherType.Sandstorm:
                // 沙尘暴：视野降低，地图显示受限
                UIManager.Instance?.ShowWeatherAlert("视野受限！");
                MiniMapSystem.Instance?.SetViewRadius(30f);
                break;
                
            case WeatherType.Clear:
                // 清天：恢复默认
                SkillSystem.Instance?.ClearElementModifiers();
                PlayerStats.Instance?.RemoveModifier("MoveSpeed", "snow");
                PhysicsSystem.SetGlobalFriction(1f);
                MiniMapSystem.Instance?.SetViewRadius(100f);
                break;
        }
    }
    
    void OnDestroy()
    {
        if (WeatherSystem.Instance != null)
            WeatherSystem.Instance.OnWeatherChanged -= OnWeatherChanged;
    }
}
```

## 七、设计要点总结

| 系统 | 关键技术 | 性能建议 |
|------|----------|----------|
| 昼夜循环 | AnimationCurve + Gradient | 每帧更新光照，避免每帧重建Shader |
| 天气过渡 | 参数线性插值 | 过渡时间30-60秒，避免突变 |
| 雨雪粒子 | GPU粒子系统 | 使用粒子LOD，超出相机范围降低发射率 |
| 风场 | Shader全局参数 | 全局参数只需一次设置，所有材质自动接收 |
| 天气玩法 | 事件驱动 | 通过事件松耦合，天气系统不依赖玩法系统 |

昼夜天气系统是提升游戏沉浸感的重要手段，建议早期就设计好接口，为后续扩展（季节特效、节日活动天气）预留空间。
