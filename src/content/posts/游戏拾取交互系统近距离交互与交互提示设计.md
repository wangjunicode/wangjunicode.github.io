## 一、可交互对象接口

```csharp
using UnityEngine;

/// <summary>
/// 可交互对象接口
/// </summary>
public interface IInteractable
{
    string InteractionPrompt { get; }    // 交互提示文本（如"拾取 [物品名]" "开门 [E]"）
    bool CanInteract(GameObject interactor); // 是否可以被交互
    void Interact(GameObject interactor);    // 执行交互
    float InteractionDuration { get; }   // 交互持续时间（0=即时，>0=长按）
    void OnInteractionCancelled();       // 长按被取消
}

/// <summary>
/// 可拾取物品
/// </summary>
public class PickupItem : MonoBehaviour, IInteractable
{
    [SerializeField] private ItemData itemData;
    [SerializeField] private int count = 1;
    [SerializeField] private float autoPickupRadius = -1f; // -1=不自动拾取
    
    [Header("特效")]
    [SerializeField] private GameObject pickupVFX;
    [SerializeField] private AudioClip pickupSFX;
    
    // 旋转动画
    [SerializeField] private float rotateSpeed = 90f;
    [SerializeField] private float floatAmplitude = 0.2f;
    [SerializeField] private float floatFrequency = 1f;

    private Vector3 startPos;
    private bool isPickedUp;

    public string InteractionPrompt => $"拾取 {itemData?.DisplayName}";
    public float InteractionDuration => 0f; // 即时拾取
    
    public bool CanInteract(GameObject interactor)
    {
        return !isPickedUp && itemData != null;
    }

    void Start()
    {
        startPos = transform.position;
    }

    void Update()
    {
        if (isPickedUp) return;
        
        // 漂浮旋转动画
        transform.Rotate(Vector3.up * rotateSpeed * Time.deltaTime);
        transform.position = startPos + Vector3.up * 
            (Mathf.Sin(Time.time * floatFrequency * Mathf.PI * 2) * floatAmplitude);
        
        // 自动拾取检测
        if (autoPickupRadius > 0)
            CheckAutoPickup();
    }

    void CheckAutoPickup()
    {
        var cols = Physics.OverlapSphere(transform.position, autoPickupRadius);
        foreach (var col in cols)
        {
            if (col.CompareTag("Player"))
            {
                Interact(col.gameObject);
                break;
            }
        }
    }

    public void Interact(GameObject interactor)
    {
        if (isPickedUp) return;
        
        // 检查背包是否有空间
        var inventory = interactor.GetComponent<InventorySystem>();
        if (inventory != null && !inventory.HasSpace(itemData))
        {
            UIManager.Instance?.ShowMessage("背包已满！");
            return;
        }
        
        isPickedUp = true;
        
        inventory?.AddItem(itemData, count);
        
        // 飞向背包图标
        var uiManager = UIManager.Instance;
        if (uiManager != null)
        {
            VFXPool.Instance?.Play("item_pickup_fly",
                transform.position,
                parent: null);
        }
        
        // 音效
        if (pickupSFX != null)
            AudioManager.Instance?.PlaySFX(pickupSFX, 0.8f);
        
        // 特效
        if (pickupVFX != null)
            Instantiate(pickupVFX, transform.position, Quaternion.identity);
        
        Destroy(gameObject, 0.1f);
    }

    public void OnInteractionCancelled() { }
}

/// <summary>
/// 可开启的门
/// </summary>
public class Door : MonoBehaviour, IInteractable
{
    [SerializeField] private bool isLocked;
    [SerializeField] private string requiredKeyId;
    [SerializeField] private float openAngle = 90f;
    [SerializeField] private float openDuration = 1f;
    
    private bool isOpen;
    
    public string InteractionPrompt => isOpen ? "关闭" : (isLocked ? "开锁" : "开门");
    public float InteractionDuration => isLocked ? 2f : 0f; // 锁着的门需要长按开锁
    
    public bool CanInteract(GameObject interactor)
    {
        if (isLocked)
        {
            var inv = interactor.GetComponent<InventorySystem>();
            return inv?.HasItem(requiredKeyId) ?? false;
        }
        return true;
    }

    public void Interact(GameObject interactor)
    {
        if (isLocked)
        {
            // 消耗钥匙
            interactor.GetComponent<InventorySystem>()?.RemoveItem(requiredKeyId, 1);
            isLocked = false;
        }
        
        isOpen = !isOpen;
        StartCoroutine(OpenDoorCoroutine(isOpen));
    }

    System.Collections.IEnumerator OpenDoorCoroutine(bool opening)
    {
        float startAngle = transform.localEulerAngles.y;
        float targetAngle = opening ? openAngle : 0f;
        float t = 0;
        
        while (t < openDuration)
        {
            t += Time.deltaTime;
            float angle = Mathf.LerpAngle(startAngle, targetAngle, t / openDuration);
            transform.localEulerAngles = new Vector3(0, angle, 0);
            yield return null;
        }
        
        transform.localEulerAngles = new Vector3(0, targetAngle, 0);
    }

    public void OnInteractionCancelled() { }
}
```

## 二、交互系统控制器

```csharp
/// <summary>
/// 玩家交互系统控制器
/// </summary>
public class PlayerInteractionSystem : MonoBehaviour
{
    [Header("检测配置")]
    [SerializeField] private float interactionRadius = 2.5f;
    [SerializeField] private LayerMask interactionMask;
    [SerializeField] private float updateInterval = 0.1f;

    [Header("UI引用")]
    [SerializeField] private InteractionPromptUI promptUI;

    private IInteractable currentTarget;
    private Collider[] nearbyColliders = new Collider[10];
    private float updateTimer;
    
    // 长按进度
    private float interactionProgress;
    private bool isInteracting;
    
    // 事件
    public event Action<IInteractable> OnTargetChanged;

    void Update()
    {
        updateTimer += Time.deltaTime;
        if (updateTimer >= updateInterval)
        {
            updateTimer = 0;
            FindBestTarget();
        }
        
        HandleInteractionInput();
    }

    void FindBestTarget()
    {
        int count = Physics.OverlapSphereNonAlloc(
            transform.position, interactionRadius, nearbyColliders, interactionMask);
        
        IInteractable best = null;
        float bestScore = float.MaxValue;
        
        for (int i = 0; i < count; i++)
        {
            var interactable = nearbyColliders[i].GetComponent<IInteractable>();
            if (interactable == null || !interactable.CanInteract(gameObject)) continue;
            
            float dist = Vector3.Distance(transform.position, 
                nearbyColliders[i].transform.position);
            
            // 评分：距离越近越好，面向目标加分
            Vector3 dir = (nearbyColliders[i].transform.position - transform.position).normalized;
            float dot = Vector3.Dot(transform.forward, dir);
            float score = dist - dot * 1f; // 面向目标减少分数（更优先）
            
            if (score < bestScore)
            {
                bestScore = score;
                best = interactable;
            }
        }
        
        if (best != currentTarget)
        {
            currentTarget = best;
            OnTargetChanged?.Invoke(currentTarget);
            promptUI?.UpdatePrompt(currentTarget);
        }
    }

    void HandleInteractionInput()
    {
        if (currentTarget == null) return;
        
        bool interactKey = Input.GetKey(KeyCode.E) || 
            (InputManager.Instance?.ConsumeJumpBuffer() ?? false);
        
        if (interactKey)
        {
            if (currentTarget.InteractionDuration <= 0)
            {
                // 即时交互
                if (Input.GetKeyDown(KeyCode.E))
                    currentTarget.Interact(gameObject);
            }
            else
            {
                // 长按交互
                isInteracting = true;
                interactionProgress += Time.deltaTime / currentTarget.InteractionDuration;
                promptUI?.SetProgress(interactionProgress);
                
                if (interactionProgress >= 1f)
                {
                    interactionProgress = 0;
                    isInteracting = false;
                    currentTarget.Interact(gameObject);
                }
            }
        }
        else if (isInteracting)
        {
            // 松开按键，取消长按
            isInteracting = false;
            interactionProgress = 0;
            currentTarget.OnInteractionCancelled();
            promptUI?.SetProgress(0);
        }
    }

    void OnDrawGizmosSelected()
    {
        Gizmos.color = new Color(0, 1, 0, 0.3f);
        Gizmos.DrawWireSphere(transform.position, interactionRadius);
    }
}

/// <summary>
/// 交互提示UI
/// </summary>
public class InteractionPromptUI : MonoBehaviour
{
    [SerializeField] private GameObject root;
    [SerializeField] private UnityEngine.UI.Text promptText;
    [SerializeField] private UnityEngine.UI.Slider progressBar;
    [SerializeField] private UnityEngine.UI.Image keyIcon;

    public void UpdatePrompt(IInteractable interactable)
    {
        if (interactable == null)
        {
            root.SetActive(false);
            return;
        }
        
        root.SetActive(true);
        promptText.text = interactable.InteractionPrompt;
        progressBar.gameObject.SetActive(interactable.InteractionDuration > 0);
        progressBar.value = 0;
    }

    public void SetProgress(float progress)
    {
        progressBar.value = progress;
    }
}
```

## 三、交互系统设计要点

| 要点 | 实现方案 |
|------|----------|
| 优先级排序 | 距离+面向综合评分 |
| 即时交互 | 按键按下瞬间响应 |
| 长按交互 | 进度条反馈，松手取消 |
| 多个可交互物重叠 | 只高亮最优先目标 |
| 背包满时提示 | 交互前预检查，给出明确提示 |
| 网络竞争拾取 | 服务端加锁，防止同时拾取 |
