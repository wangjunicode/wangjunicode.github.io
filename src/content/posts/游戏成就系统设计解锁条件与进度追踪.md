## 一、成就数据配置

```csharp
using System;
using UnityEngine;

[CreateAssetMenu(menuName = "Game/Achievement")]
public class AchievementData : ScriptableObject
{
    public string Id;
    public string Title;
    [TextArea] public string Description;
    public Sprite Icon;
    public AchievementType Type;
    public bool IsHidden;       // 隐藏成就（解锁前不显示）
    public int TargetCount;     // 目标进度（累计类）
    public AchievementReward[] Rewards;
}

public enum AchievementType { OneTime, Cumulative, Milestone }

[Serializable]
public class AchievementReward { public string ItemId; public int Amount; }
```

## 二、成就管理器

```csharp
public class AchievementManager : MonoBehaviour
{
    private static AchievementManager instance;
    public static AchievementManager Instance => instance;

    [SerializeField] private AchievementData[] achievements;
    private Dictionary<string, int> progress = new Dictionary<string, int>();
    private HashSet<string> unlocked = new HashSet<string>();
    
    public event Action<AchievementData> OnAchievementUnlocked;

    void Awake() { instance = this; }

    public void ReportProgress(string id, int increment = 1)
    {
        if (unlocked.Contains(id)) return;
        
        if (!progress.ContainsKey(id)) progress[id] = 0;
        progress[id] += increment;
        
        var data = GetData(id);
        if (data != null && progress[id] >= data.TargetCount)
            Unlock(id, data);
    }

    void Unlock(string id, AchievementData data)
    {
        unlocked.Add(id);
        OnAchievementUnlocked?.Invoke(data);
        
        // 发放奖励
        foreach (var r in data.Rewards)
            Debug.Log($"[Achievement] 奖励: {r.ItemId} x{r.Amount}");
        
        Debug.Log($"[Achievement] 解锁: {data.Title}");
    }

    AchievementData GetData(string id)
    {
        foreach (var a in achievements)
            if (a.Id == id) return a;
        return null;
    }

    public float GetProgress(string id)
    {
        var data = GetData(id);
        if (data == null || data.TargetCount <= 0) return 0;
        int p = progress.TryGetValue(id, out int v) ? v : 0;
        return (float)p / data.TargetCount;
    }
}
```

## 三、成就通知UI

```csharp
public class AchievementToastUI : MonoBehaviour
{
    [SerializeField] private GameObject toastPrefab;
    [SerializeField] private Transform toastContainer;
    [SerializeField] private float displayDuration = 4f;

    void Start()
    {
        AchievementManager.Instance.OnAchievementUnlocked += ShowToast;
    }

    void ShowToast(AchievementData data)
    {
        var go = Instantiate(toastPrefab, toastContainer);
        // 设置图标/标题
        StartCoroutine(AutoHide(go, displayDuration));
    }

    System.Collections.IEnumerator AutoHide(GameObject go, float delay)
    {
        yield return new WaitForSeconds(delay);
        Destroy(go);
    }
}
```
