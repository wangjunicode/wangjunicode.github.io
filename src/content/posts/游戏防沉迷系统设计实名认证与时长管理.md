## 一、防沉迷系统架构

```csharp
using System;
using UnityEngine;

/// <summary>
/// 玩家认证状态
/// </summary>
public enum AuthStatus 
{ 
    Unknown,        // 未认证
    Verified,       // 已实名（成年）
    Minor,          // 已实名（未成年）
    GuestMinor      // 游客（视为未成年）
}

[Serializable]
public class AntiAddictionState
{
    public AuthStatus Status;
    public int AgeGroup;              // 0=未知, 1=<8岁, 2=8-16岁, 3=16-18岁, 4=成年
    public int TodayPlayMinutes;      // 今日游戏时长（分钟）
    public int DailyLimitMinutes;     // 每日时长限制
    public long SessionStartTime;     // 本次游戏开始时间
    public bool IsPlayAllowed;        // 当前是否允许游戏
    public long ResumeTime;           // 下次可以游戏的时间（宵禁后）
}

public class AntiAddictionManager : MonoBehaviour
{
    private static AntiAddictionManager instance;
    public static AntiAddictionManager Instance => instance;
    
    private AntiAddictionState state;
    private float checkInterval = 60f;  // 每分钟检查
    private float checkTimer;
    
    public event Action<int> OnPlayTimeWarning;     // 还剩N分钟
    public event Action OnForcedOffline;             // 强制下线

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    void Update()
    {
        if (state == null || !state.IsPlayAllowed) return;
        
        checkTimer += Time.deltaTime;
        if (checkTimer >= checkInterval)
        {
            checkTimer = 0;
            CheckPlayTime();
        }
    }

    public async System.Threading.Tasks.Task Initialize(string playerId)
    {
        state = await NetworkService.GetAntiAddictionState(playerId);
        
        if (state.Status == AuthStatus.Unknown || state.Status == AuthStatus.GuestMinor)
        {
            // 强制实名认证（弹出实名窗口）
            UIManager.Instance?.ShowRealNameAuthPanel();
        }
        else
        {
            ValidateCurrentTime();
        }
    }

    void ValidateCurrentTime()
    {
        if (state == null) return;
        
        var now = DateTime.Now;
        
        // 宵禁检查（22:00-8:00禁止未成年游戏）
        if (state.AgeGroup > 0 && state.AgeGroup < 4)
        {
            if (now.Hour >= 22 || now.Hour < 8)
            {
                state.IsPlayAllowed = false;
                var resumeTime = now.Hour >= 22 
                    ? DateTime.Today.AddDays(1).AddHours(8) 
                    : DateTime.Today.AddHours(8);
                state.ResumeTime = ((DateTimeOffset)resumeTime).ToUnixTimeSeconds();
                
                ForceOffline("宵禁时段（22:00-8:00）未成年人无法游戏");
                return;
            }
        }
        
        state.IsPlayAllowed = true;
        state.SessionStartTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
    }

    void CheckPlayTime()
    {
        if (state == null || state.DailyLimitMinutes <= 0) return;
        
        int remainingMinutes = state.DailyLimitMinutes - state.TodayPlayMinutes;
        
        // 提醒（剩余30/10/5分钟）
        if (remainingMinutes == 30 || remainingMinutes == 10 || remainingMinutes == 5)
            OnPlayTimeWarning?.Invoke(remainingMinutes);
        
        if (remainingMinutes <= 0)
            ForceOffline("今日游戏时长已达上限");
        else
            state.TodayPlayMinutes++;
    }

    void ForceOffline(string reason)
    {
        state.IsPlayAllowed = false;
        Debug.Log($"[AntiAddiction] 强制下线: {reason}");
        
        UIManager.Instance?.ShowAntiAddictionNotice(reason);
        OnForcedOffline?.Invoke();
        
        // 延迟退出游戏
        Invoke(nameof(QuitGame), 5f);
    }

    void QuitGame() => Application.Quit();
}
```

## 二、中国防沉迷合规要求

```
中国未成年人保护条例（2021年起执行）：

年龄分类：
├── <8岁：每日最多1小时，每次不超过30分钟
├── 8-16岁：每日最多2小时
├── 16-18岁：每日最多3小时
└── 成年（18+）：无限制

宵禁：22:00-8:00禁止所有未成年人游戏

付费限制：
├── 8-16岁：单次不超过50元，每月不超过200元
└── 16-18岁：单次不超过100元，每月不超过400元

技术要求：
├── 必须接入实名认证系统（国家网络游戏防沉迷实名验证系统）
├── 服务端计时（不能靠客户端）
├── 用户认证信息必须与公安系统核验
└── 不得以任何手段规避（如修改系统时间）
```

## 三、防沉迷系统最佳实践

| 要点 | 方案 |
|------|------|
| 认证时机 | 首次登录时强制认证，不可跳过 |
| 服务端计时 | 游戏时长必须服务端记录 |
| 多设备共享 | 同账号多端时长累加计算 |
| 提醒阶梯 | 剩30/10/5分钟提前提醒 |
| 强制下线 | 到时间立即强制退出，无法绕过 |
| 付费拦截 | 未成年人付费在服务端严格限制 |
