## 一、埋点事件规范

```csharp
// 事件命名规范：{模块}_{动作}_{对象}
// 例：
// guide_start_step        新手引导开始某步骤
// guide_complete_step     新手引导完成某步骤
// pay_start_purchase      开始购买
// pay_success_purchase    购买成功
// level_enter_dungeon     进入副本
// level_complete_dungeon  完成副本

public static class AnalyticsEvents

    // 新手引导

    public const string PAY_CONFIRM  = "pay_confirm_purchase";
    public const string PAY_SUCCESS  = "pay_success";

}
```

## 二、埋点SDK封装

```csharp
public class AnalyticsManager : MonoBehaviour
{
    private static AnalyticsManager instance;
    public static AnalyticsManager Instance => instance;
    
    private List<AnalyticsEvent> eventQueue = new List<AnalyticsEvent>();
    private const int BATCH_SIZE = 30;       // 批量上报大小
    private const float FLUSH_INTERVAL = 30f; // 定时上报间隔
    private float flushTimer;
    private bool isEnabled = true;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    void Update()
    {
        flushTimer += Time.unscaledDeltaTime;
        if (flushTimer >= FLUSH_INTERVAL)
        {
            flushTimer = 0;
            FlushEvents();
        }
    }

    public void Track(string eventName, Dictionary<string, object> props = null)
    {
        if (!isEnabled) return;
        
        var evt = new AnalyticsEvent
        {
            EventName = eventName,
            Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
            SessionId = GameSession.Current.SessionId,
            PlayerId = PlayerDataService.GetLocalPlayerId(),
            Properties = props ?? new Dictionary<string, object>()
        };
        
        // 自动添加公共属性
        evt.Properties["platform"] = Application.platform.ToString();
        evt.Properties["game_version"] = Application.version;
        evt.Properties["player_level"] = PlayerDataService.GetLocalPlayerData()?.Level ?? 0;
        
        eventQueue.Add(evt);
        
        if (eventQueue.Count >= BATCH_SIZE)
            FlushEvents();
    }

    async void FlushEvents()
    {
        if (eventQueue.Count == 0) return;
        
        var toSend = new List<AnalyticsEvent>(eventQueue);
        eventQueue.Clear();
        
        try
        {
            await NetworkService.SendAnalyticsEvents(toSend);
        }
        catch (Exception e)
        {
            Debug.LogWarning($"[Analytics] 上报失败: {e.Message}，事件将丢弃");
            // 实际项目可做本地缓存，重试时再发
        }
    }

    void OnApplicationPause(bool paused)
    {
        if (paused) FlushEvents(); // 切后台立即上报
    }

    void OnApplicationQuit()
    {
        FlushEvents();
    }
}

[Serializable]
public class AnalyticsEvent
{
    public string EventName;
    public long Timestamp;
    public string SessionId;
    public string PlayerId;
    public Dictionary<string, object> Properties;
}
```

## 三、关键埋点位置

```csharp
// 新手引导埋点示例
public class GuideAnalytics : MonoBehaviour
{
    void OnGuideStepComplete(string stepId, float duration)
    {
        AnalyticsManager.Instance?.Track(AnalyticsEvents.GUIDE_STEP, new Dictionary<string, object>
        {
            { "step_id", stepId },
            { "duration_seconds", duration },
            { "is_skip", false }
        });
    }
    
    void OnGuideFinish(float totalDuration, int totalSteps)
    {
        AnalyticsManager.Instance?.Track(AnalyticsEvents.GUIDE_FINISH, new Dictionary<string, object>
        {
            { "total_duration", totalDuration },
            { "total_steps", totalSteps }
        });
    }
}

// 关卡埋点
public class LevelAnalytics : MonoBehaviour
{
    private float levelStartTime;
    
    public void OnLevelStart(string levelId, int difficulty)
    {
        levelStartTime = Time.realtimeSinceStartup;
        AnalyticsManager.Instance?.Track(AnalyticsEvents.LEVEL_START, new Dictionary<string, object>
        {
            { "level_id", levelId },
            { "difficulty", difficulty },
            { "attempt_count", GetAttemptCount(levelId) }
        });
    }
    
    public void OnLevelPass(string levelId, int stars, long score)
    {
        float duration = Time.realtimeSinceStartup - levelStartTime;
        AnalyticsManager.Instance?.Track(AnalyticsEvents.LEVEL_PASS, new Dictionary<string, object>
        {
            { "level_id", levelId },
            { "stars", stars },
            { "score", score },
            { "duration_seconds", duration }
        });
    }
    
    int GetAttemptCount(string levelId) => PlayerPrefs.GetInt($"attempt_{levelId}", 0);
}
```

## 四、核心分析指标

| 指标 | 定义 | 目标 |
|------|------|------|
| 次日留存 | Day1登录/Day0登录 | >40% |
| 7日留存 | Day7/Day0 | >20% |
| 30日留存 | Day30/Day0 | >10% |
| 首日付费率 | 首日付费人数/首日注册 | >5% |
| ARPU | 总收入/活跃用户 | 持续增长 |
| 引导完成率 | 引导完成人数/开始人数 | >80% |
| 卡关率 | 关卡失败/尝试次数 | <30% |
