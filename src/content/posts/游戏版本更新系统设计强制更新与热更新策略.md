## 一、版本号规范

```
游戏版本号格式：MAJOR.MINOR.PATCH.BUILD

MAJOR:  大版本（重大功能/架构变更，需强制更新）
MINOR:  小版本（新功能/赛季内容，建议更新）
PATCH:  补丁（Bug修复/平衡调整，热更新）
BUILD:  构建号（CI/CD自动递增，仅内部追踪）

示例：
1.0.0.100  → 正式上线版本
1.0.1.105  → Bug修复补丁（Patch，热更新）
1.1.0.120  → 新赛季（Minor，建议更新）
2.0.0.200  → 大版本（Major，强制更新）
```

## 二、版本检查系统

```csharp
using System;
using UnityEngine;

/// <summary>
/// 服务端返回的版本信息
/// </summary>
[Serializable]
public class ServerVersionInfo
{
    public string MinVersion;          // 最低支持版本（低于此必须强制更新）
    public string LatestVersion;       // 最新版本（低于此建议更新）
    public string ChangeLog;           // 更新日志
    public string StoreUrl;            // 商店链接
    public bool IsServerMaintaining;  // 是否维护中
    public long MaintenanceEndTime;   // 维护结束时间（Unix时间戳）
}

/// <summary>
/// 版本更新类型
/// </summary>
public enum UpdateType
{
    UpToDate,       // 已是最新
    Optional,       // 建议更新（可跳过）
    Required,       // 强制更新（不可跳过）
    Maintenance     // 服务器维护中
}

/// <summary>
/// 版本管理器
/// </summary>
public class VersionManager : MonoBehaviour
{
    private static VersionManager instance;
    public static VersionManager Instance => instance;

    [SerializeField] private string versionCheckUrl = "https://api.example.com/version";
    
    // 当前客户端版本（从 Application.version 读取）
    public string ClientVersion => Application.version;
    
    public event Action<UpdateType, ServerVersionInfo> OnVersionCheckComplete;

    void Awake() { instance = this; }

    /// <summary>
    /// 在游戏启动时检查版本
    /// </summary>
    public async System.Threading.Tasks.Task CheckVersion()
    {
        ServerVersionInfo serverInfo;
        
        try
        {
            using var req = UnityEngine.Networking.UnityWebRequest.Get(versionCheckUrl);
            req.SetRequestHeader("X-App-Version", ClientVersion);
            req.SetRequestHeader("X-Platform", Application.platform.ToString());
            req.timeout = 10;
            
            var op = req.SendWebRequest();
            while (!op.isDone)
                await System.Threading.Tasks.Task.Yield();
            
            if (req.result != UnityEngine.Networking.UnityWebRequest.Result.Success)
            {
                Debug.LogWarning($"[Version] 版本检查失败: {req.error}，跳过检查");
                OnVersionCheckComplete?.Invoke(UpdateType.UpToDate, null);
                return;
            }
            
            serverInfo = JsonUtility.FromJson<ServerVersionInfo>(req.downloadHandler.text);
        }
        catch (Exception e)
        {
            Debug.LogError($"[Version] 版本检查异常: {e.Message}");
            OnVersionCheckComplete?.Invoke(UpdateType.UpToDate, null);
            return;
        }
        
        // 检查维护状态
        if (serverInfo.IsServerMaintaining)
        {
            OnVersionCheckComplete?.Invoke(UpdateType.Maintenance, serverInfo);
            return;
        }
        
        UpdateType updateType = DetermineUpdateType(serverInfo);
        OnVersionCheckComplete?.Invoke(updateType, serverInfo);
    }

    UpdateType DetermineUpdateType(ServerVersionInfo serverInfo)
    {
        Version client = Version.Parse(ClientVersion.Split('.').Length < 4 ? 
            ClientVersion : string.Join(".", ClientVersion.Split('.')[..3]));
        
        if (!string.IsNullOrEmpty(serverInfo.MinVersion))
        {
            Version min = Version.Parse(serverInfo.MinVersion.Split('.').Length < 4 ? 
                serverInfo.MinVersion : string.Join(".", serverInfo.MinVersion.Split('.')[..3]));
            
            if (client < min)
                return UpdateType.Required;
        }
        
        if (!string.IsNullOrEmpty(serverInfo.LatestVersion))
        {
            Version latest = Version.Parse(serverInfo.LatestVersion.Split('.').Length < 4 ? 
                serverInfo.LatestVersion : string.Join(".", serverInfo.LatestVersion.Split('.')[..3]));
            
            if (client < latest)
                return UpdateType.Optional;
        }
        
        return UpdateType.UpToDate;
    }

    public void OpenStoreUrl(string storeUrl)
    {
        Application.OpenURL(storeUrl);
    }
}
```

## 三、Addressables 热更新流程

```csharp
using UnityEngine;
using UnityEngine.AddressableAssets;
using UnityEngine.ResourceManagement.AsyncOperations;
using System.Collections;
using System.Collections.Generic;

/// <summary>
/// 资源热更新管理器（基于 Addressables）
/// </summary>
public class HotUpdateManager : MonoBehaviour
{
    private static HotUpdateManager instance;
    public static HotUpdateManager Instance => instance;

    public event Action<long> OnDownloadSizeCalculated;   // 需要下载的总大小
    public event Action<float> OnDownloadProgress;         // 下载进度（0-1）
    public event Action OnUpdateComplete;
    public event Action<string> OnUpdateFailed;

    void Awake() { instance = this; }

    /// <summary>
    /// 完整热更新流程
    /// </summary>
    public IEnumerator CheckAndUpdate()
    {
        // 第一步：初始化 Addressables
        var initOp = Addressables.InitializeAsync();
        yield return initOp;
        
        if (initOp.Status != AsyncOperationStatus.Succeeded)
        {
            OnUpdateFailed?.Invoke("Addressables 初始化失败");
            yield break;
        }

        // 第二步：检查 Catalog 更新
        var checkOp = Addressables.CheckForCatalogUpdates(false);
        yield return checkOp;
        
        if (checkOp.Status != AsyncOperationStatus.Succeeded)
        {
            Debug.LogWarning("[HotUpdate] Catalog检查失败，使用本地资源");
            OnUpdateComplete?.Invoke();
            yield break;
        }
        
        var catalogs = checkOp.Result;
        
        if (catalogs == null || catalogs.Count == 0)
        {
            Debug.Log("[HotUpdate] 无需更新 Catalog");
            yield return CheckAssetBundleUpdates();
            yield break;
        }
        
        // 第三步：更新 Catalog
        Debug.Log($"[HotUpdate] 更新 {catalogs.Count} 个 Catalog");
        var updateCatalogOp = Addressables.UpdateCatalogs(catalogs, false);
        yield return updateCatalogOp;
        
        Addressables.Release(checkOp);
        
        // 第四步：检查 AssetBundle 更新
        yield return CheckAssetBundleUpdates();
    }

    IEnumerator CheckAssetBundleUpdates()
    {
        // 获取所有需要更新的 key（可以指定特定 label）
        List<object> updateKeys = new List<object>();
        
        // 使用 "default" 标签检查所有默认资源
        var sizeOp = Addressables.GetDownloadSizeAsync(updateKeys.Count > 0 ? 
            (object)updateKeys : "default");
        yield return sizeOp;
        
        long downloadSize = sizeOp.Result;
        Addressables.Release(sizeOp);
        
        if (downloadSize <= 0)
        {
            Debug.Log("[HotUpdate] 无资源需要更新");
            OnUpdateComplete?.Invoke();
            yield break;
        }
        
        Debug.Log($"[HotUpdate] 需要下载 {FormatBytes(downloadSize)}");
        OnDownloadSizeCalculated?.Invoke(downloadSize);
        
        // 询问用户是否在 WiFi 环境下下载（或等待 WiFi）
        bool shouldDownload = await PromptUserForDownload(downloadSize);
        if (!shouldDownload)
        {
            OnUpdateComplete?.Invoke(); // 跳过下载，使用旧资源
            yield break;
        }
        
        // 第五步：下载更新
        var downloadOp = Addressables.DownloadDependenciesAsync(
            updateKeys.Count > 0 ? (object)updateKeys : "default", 
            Addressables.MergeMode.Union, false);
        
        while (!downloadOp.IsDone)
        {
            float progress = downloadOp.GetDownloadStatus().Percent;
            OnDownloadProgress?.Invoke(progress);
            yield return null;
        }
        
        if (downloadOp.Status == AsyncOperationStatus.Succeeded)
        {
            Debug.Log("[HotUpdate] 下载完成！");
            OnUpdateComplete?.Invoke();
        }
        else
        {
            string error = downloadOp.OperationException?.Message ?? "未知错误";
            Debug.LogError($"[HotUpdate] 下载失败: {error}");
            OnUpdateFailed?.Invoke(error);
        }
        
        Addressables.Release(downloadOp);
    }

    async System.Threading.Tasks.Task<bool> PromptUserForDownload(long size)
    {
        // 判断网络类型
        bool isWifi = Application.internetReachability == 
                      NetworkReachability.ReachableViaLocalAreaNetwork;
        
        string sizeStr = FormatBytes(size);
        string networkType = isWifi ? "WiFi" : "移动数据";
        
        bool confirm = await UIManager.Instance.ShowConfirmDialog(
            "发现更新",
            $"检测到 {sizeStr} 的更新内容\n当前使用 {networkType} 网络\n是否立即下载？");
        
        return confirm;
    }

    string FormatBytes(long bytes)
    {
        if (bytes < 1024) return $"{bytes}B";
        if (bytes < 1024 * 1024) return $"{bytes / 1024f:F1}KB";
        if (bytes < 1024 * 1024 * 1024) return $"{bytes / (1024f * 1024f):F1}MB";
        return $"{bytes / (1024f * 1024f * 1024f):F2}GB";
    }
}
```

## 四、更新进度UI

```csharp
/// <summary>
/// 热更新进度界面
/// </summary>
public class HotUpdateUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Slider progressBar;
    [SerializeField] private UnityEngine.UI.Text progressText;
    [SerializeField] private UnityEngine.UI.Text statusText;
    [SerializeField] private UnityEngine.UI.Button retryButton;
    [SerializeField] private UnityEngine.UI.Button skipButton;

    void Start()
    {
        retryButton.gameObject.SetActive(false);
        skipButton.gameObject.SetActive(false);
        
        var manager = HotUpdateManager.Instance;
        manager.OnDownloadSizeCalculated += size =>
        {
            statusText.text = $"发现更新，正在下载({FormatBytes(size)})...";
        };
        
        manager.OnDownloadProgress += progress =>
        {
            progressBar.value = progress;
            progressText.text = $"{progress * 100:F0}%";
            statusText.text = $"正在更新...";
        };
        
        manager.OnUpdateComplete += () =>
        {
            progressBar.value = 1f;
            progressText.text = "100%";
            statusText.text = "更新完成！";
            // 进入主场景
            UnityEngine.SceneManagement.SceneManager.LoadScene("MainMenu");
        };
        
        manager.OnUpdateFailed += error =>
        {
            statusText.text = $"更新失败：{error}";
            retryButton.gameObject.SetActive(true);
            skipButton.gameObject.SetActive(true);
        };
        
        retryButton.onClick.AddListener(() =>
        {
            retryButton.gameObject.SetActive(false);
            StartCoroutine(manager.CheckAndUpdate());
        });
        
        StartCoroutine(manager.CheckAndUpdate());
    }

    string FormatBytes(long bytes) => 
        bytes < 1024 * 1024 ? $"{bytes / 1024f:F1}KB" : $"{bytes / (1024f * 1024f):F1}MB";
}
```

## 五、平台更新策略对比

| 平台 | 强制更新方式 | 热更新支持 | 时效 |
|------|------------|-----------|------|
| iOS | 跳转 App Store | 仅资源热更（代码不行）| 审核周期 1-3天 |
| Android | 跳转 Google Play 或 APK直链 | 资源+脚本 | 无审核 |
| PC（Steam）| 自动更新 | 全量 | 分钟级 |
| 微信小游戏 | 自动更新 | 全量 | 分钟级 |

**iOS 合规要点：**
- 禁止通过热更新引入新的主要功能（违反 App Store 规定）
- 资源更新（音频/图片/配置）合法
- Lua/JS 脚本热更处于灰色地带，需评估风险
