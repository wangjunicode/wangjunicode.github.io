## 一、伤害飘字系统

```csharp
using System.Collections;
using UnityEngine;
using TMPro;

public class DamageNumber : MonoBehaviour
{
    [SerializeField] private TextMeshPro text;
    [SerializeField] private float floatHeight = 2f;
    [SerializeField] private float duration = 1.2f;
    [SerializeField] private AnimationCurve scaleCurve;
    [SerializeField] private AnimationCurve alphaCurve;

    public void Setup(long damage, DamageNumberType type)
    {
        text.text = FormatDamage(damage, type);
        text.color = GetColor(type);
        text.fontSize = GetFontSize(type);
        
        // 随机偏移（多个飘字不重叠）
        transform.position += new Vector3(
            Random.Range(-0.5f, 0.5f), 0, 0);
        
        StartCoroutine(AnimateFloat());
    }

    string FormatDamage(long value, DamageNumberType type)
    {
        string str = value >= 10000 ? $"{value/10000.0f:F1}万" : value.ToString();
        if (type == DamageNumberType.Critical) str = $"暴击！{str}";
        if (type == DamageNumberType.Heal) str = $"+{str}";
        if (type == DamageNumberType.Miss) str = "Miss";
        if (type == DamageNumberType.Block) str = "格挡";
        return str;
    }

    Color GetColor(DamageNumberType type) => type switch
    {
        DamageNumberType.Critical => new Color(1f, 0.5f, 0f),
        DamageNumberType.Heal => Color.green,
        DamageNumberType.Miss => Color.gray,
        DamageNumberType.Block => Color.cyan,
        DamageNumberType.Dot => new Color(1f, 0.3f, 0.3f),
        _ => Color.white
    };

    float GetFontSize(DamageNumberType type) => type switch
    {
        DamageNumberType.Critical => 5f,
        DamageNumberType.Heal => 3f,
        _ => 4f
    };

    IEnumerator AnimateFloat()
    {
        Vector3 startPos = transform.position;
        Vector3 endPos = startPos + Vector3.up * floatHeight;
        float t = 0;

        while (t < duration)
        {
            t += Time.deltaTime;
            float progress = t / duration;
            
            transform.position = Vector3.Lerp(startPos, endPos, progress);
            float scale = scaleCurve.Evaluate(progress);
            transform.localScale = Vector3.one * scale;
            
            var c = text.color;
            c.a = alphaCurve.Evaluate(progress);
            text.color = c;
            
            // 始终面向摄像机
            transform.LookAt(transform.position + Camera.main.transform.forward);
            
            yield return null;
        }

        Destroy(gameObject);
    }
}

public enum DamageNumberType { Normal, Critical, Heal, Miss, Block, Dot, Shield }

public class DamageNumberSpawner : MonoBehaviour
{
    private static DamageNumberSpawner instance;
    public static DamageNumberSpawner Instance => instance;
    
    [SerializeField] private DamageNumber prefab;

    void Awake() { instance = this; }

    public void Spawn(Vector3 worldPos, long value, DamageNumberType type)
    {
        var dn = Instantiate(prefab, worldPos + Vector3.up * 1.5f, Quaternion.identity);
        dn.Setup(value, type);
    }
}
```

## 二、DPS统计器

```csharp
public class CombatStatTracker : MonoBehaviour
{
    private float combatStartTime;
    private long totalDamageDealt;
    private long totalHealingDone;
    private int killCount;
    
    private Dictionary<string, long> damageBySkill = new Dictionary<string, long>();

    public void OnCombatStart()
    {
        combatStartTime = Time.time;
        totalDamageDealt = 0;
        totalHealingDone = 0;
        killCount = 0;
        damageBySkill.Clear();
    }

    public void RecordDamage(long damage, string skillId = "auto")
    {
        totalDamageDealt += damage;
        if (!damageBySkill.ContainsKey(skillId)) damageBySkill[skillId] = 0;
        damageBySkill[skillId] += damage;
    }

    public void RecordHeal(long heal) => totalHealingDone += heal;
    public void RecordKill() => killCount++;

    public float GetDPS()
    {
        float elapsed = Time.time - combatStartTime;
        return elapsed > 0 ? totalDamageDealt / elapsed : 0;
    }

    public float GetHPS()
    {
        float elapsed = Time.time - combatStartTime;
        return elapsed > 0 ? totalHealingDone / elapsed : 0;
    }

    public CombatSummary GetSummary() => new CombatSummary
    {
        TotalDamage = totalDamageDealt,
        TotalHeal = totalHealingDone,
        DPS = GetDPS(),
        HPS = GetHPS(),
        KillCount = killCount,
        CombatDuration = Time.time - combatStartTime
    };
}

[Serializable]
public class CombatSummary
{
    public long TotalDamage;
    public long TotalHeal;
    public float DPS;
    public float HPS;
    public int KillCount;
    public float CombatDuration;
}
```

## 三、击杀播报Feed

```csharp
public class KillFeedManager : MonoBehaviour
{
    [SerializeField] private Transform feedContainer;
    [SerializeField] private GameObject feedItemPrefab;
    [SerializeField] private int maxFeedItems = 5;
    [SerializeField] private float itemDuration = 4f;

    private Queue<GameObject> feedItems = new Queue<GameObject>();

    public void AddKillFeed(string killerId, string victimId, string weaponId)
    {
        if (feedItems.Count >= maxFeedItems)
        {
            var oldest = feedItems.Dequeue();
            Destroy(oldest);
        }

        var go = Instantiate(feedItemPrefab, feedContainer);
        // 设置击杀者/受害者名称、武器图标
        feedItems.Enqueue(go);
        
        StartCoroutine(AutoRemove(go, itemDuration));
    }

    IEnumerator AutoRemove(GameObject go, float delay)
    {
        yield return new WaitForSeconds(delay);
        if (go != null)
        {
            feedItems.TryDequeue(out _);
            Destroy(go);
        }
    }
}
```
