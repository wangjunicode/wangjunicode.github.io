## 一、埋点系统的价值

数据驱动游戏设计的核心工具：

1. **玩家行为分析**：了解玩家在哪里流失、哪里停留
2. **A/B测试**：验证不同设计方案的效果
3. **性能监控**：帧率、加载时间、崩溃率
4. **商业分析**：付费转化率、ARPU、LTV
5. **内容优化**：哪些关卡太难、哪些副本少人玩

## 二、埋点事件分级模型

```csharp
/// <summary>
/// 埋点事件优先级（决定实时性和存储策略）
/// </summary>
public enum AnalyticsEventPriority
{
    Critical = 0,   // 关键事件（付费、账号操作）：立即上报，不允许丢失
    High = 1,       // 高优先级（关卡完成、首次行为）：队列优先上报
    Normal = 2,     // 普通事件（普通操作）：批量上报（每30秒一次）
    Low = 3,        // 低优先级（高频行为）：合并聚合后上报
    Debug = 4,      // 调试事件（仅开发环境上报）
}

/// <summary>
/// 标准埋点事件
/// </summary>
[System.Serializable]
public class AnalyticsEvent
{
    // 基础字段（每个事件都有）
    public string EventName;
    public long Timestamp;            // Unix毫秒时间戳
    public string SessionId;          // 会话ID
    public string UserId;             // 用户ID（匿名或账号）
    public string DeviceId;           // 设备唯一ID
    public string Platform;           // ios/android/pc/console
    public string AppVersion;         // 游戏版本
    public string Country;            // 国家/地区
    public int SequenceNumber;        // 序列号（检测丢失）
    
    // 扩展参数
    public Dictionary<string, object> Properties 
        = new Dictionary<string, object>();
    
    // 内部字段（不上报）
    [System.NonSerialized]
    public AnalyticsEventPriority Priority;
    
    [System.NonSerialized]
    public int RetryCount;
}
```

## 三、埋点管理器

```csharp
using System;
using System.Collections;
using System.Collections.Generic;
using System.IO;
using UnityEngine;
using UnityEngine.Networking;

/// <summary>
/// 游戏埋点管理器（客户端 SDK 核心）
/// </summary>
public class AnalyticsManager : MonoBehaviour
{
    private static AnalyticsManager instance;
    public static AnalyticsManager Instance => instance;

    [Header("上报配置")]
    [SerializeField] private string serverEndpoint = "https://analytics.example.com/collect";
    [SerializeField] private int batchSize = 50;                // 批量上报条数
    [SerializeField] private float batchInterval = 30f;         // 批量上报间隔（秒）
    [SerializeField] private int maxCacheSize = 5000;           // 最大缓存条数
    [SerializeField] private bool enableInEditor = false;       // 编辑器中是否真实上报
    
    [Header("隐私合规")]
    [SerializeField] private bool gdprEnabled = true;
    [SerializeField] private bool collectPersonalData = false;  // 用户是否同意收集个人数据

    // 事件队列（按优先级）
    private Queue<AnalyticsEvent>[] priorityQueues;
    private const int QueueCount = 4; // Critical, High, Normal, Low

    // 离线缓存（APP重启后继续上报未发送的事件）
    private List<AnalyticsEvent> offlineCache = new List<AnalyticsEvent>();
    private string cachePath;

    // 会话信息
    private string sessionId;
    private int sequenceNumber = 0;
    private float sessionStartTime;

    // 用户属性（超级属性）
    private Dictionary<string, object> superProperties = new Dictionary<string, object>();

    void Awake()
    {
        instance = this;
        
        cachePath = Path.Combine(Application.persistentDataPath, "analytics_cache.json");
        
        // 初始化优先级队列
        priorityQueues = new Queue<AnalyticsEvent>[QueueCount];
        for (int i = 0; i < QueueCount; i++)
            priorityQueues[i] = new Queue<AnalyticsEvent>();
        
        // 生成会话ID
        sessionId = Guid.NewGuid().ToString("N").Substring(0, 16);
        sessionStartTime = Time.realtimeSinceStartup;
        
        // 设置超级属性（每个事件都会携带）
        SetSuperProperty("device_model", SystemInfo.deviceModel);
        SetSuperProperty("os_version", SystemInfo.operatingSystem);
        SetSuperProperty("screen_resolution", $"{Screen.width}x{Screen.height}");
        SetSuperProperty("app_version", Application.version);
        SetSuperProperty("platform", Application.platform.ToString());
        
        // 加载离线缓存
        LoadOfflineCache();
        
        // 启动批量上报
        StartCoroutine(BatchUploadRoutine());
        
        // 注册 APP 退出事件
        Application.quitting += OnApplicationQuitting;
    }

    // ============ 核心埋点接口 ============

    /// <summary>
    /// 追踪事件（通用接口）
    /// </summary>
    public void Track(string eventName, Dictionary<string, object> properties = null,
        AnalyticsEventPriority priority = AnalyticsEventPriority.Normal)
    {
        if (!collectPersonalData && gdprEnabled) return;
        
        #if UNITY_EDITOR
        if (!enableInEditor)
        {
            Debug.Log($"[Analytics][{priority}] {eventName}: {DictToString(properties)}");
            return;
        }
        #endif
        
        var evt = CreateEvent(eventName, priority, properties);
        EnqueueEvent(evt);
        
        // 关键事件立即上报
        if (priority == AnalyticsEventPriority.Critical)
        {
            StartCoroutine(UploadBatch(new List<AnalyticsEvent> { evt }));
        }
    }

    /// <summary>
    /// 追踪页面/场景访问
    /// </summary>
    public void TrackScreen(string screenName, Dictionary<string, object> extraProps = null)
    {
        var props = new Dictionary<string, object>(extraProps ?? new Dictionary<string, object>())
        {
            ["screen_name"] = screenName
        };
        Track("screen_view", props, AnalyticsEventPriority.High);
    }

    /// <summary>
    /// 追踪购买事件（符合 Firebase/AppsFlyer 规范）
    /// </summary>
    public void TrackPurchase(string itemId, float price, string currency, string orderId)
    {
        Track("purchase", new Dictionary<string, object>
        {
            ["item_id"] = itemId,
            ["price"] = price,
            ["currency"] = currency,
            ["order_id"] = orderId,
            ["payment_method"] = "iap"
        }, AnalyticsEventPriority.Critical);
    }

    /// <summary>
    /// 追踪关卡事件
    /// </summary>
    public void TrackLevelStart(string levelId, int attempt = 1)
    {
        Track("level_start", new Dictionary<string, object>
        {
            ["level_id"] = levelId,
            ["attempt_number"] = attempt
        }, AnalyticsEventPriority.High);
    }

    public void TrackLevelComplete(string levelId, float completionTime, int stars = 3)
    {
        Track("level_complete", new Dictionary<string, object>
        {
            ["level_id"] = levelId,
            ["completion_time_sec"] = completionTime,
            ["stars"] = stars
        }, AnalyticsEventPriority.High);
    }

    public void TrackLevelFail(string levelId, string failReason, int attempt)
    {
        Track("level_fail", new Dictionary<string, object>
        {
            ["level_id"] = levelId,
            ["fail_reason"] = failReason,
            ["attempt_number"] = attempt
        }, AnalyticsEventPriority.High);
    }

    // ============ 内部实现 ============

    AnalyticsEvent CreateEvent(string name, AnalyticsEventPriority priority,
        Dictionary<string, object> properties)
    {
        var evt = new AnalyticsEvent
        {
            EventName = name,
            Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
            SessionId = sessionId,
            UserId = GetUserId(),
            DeviceId = SystemInfo.deviceUniqueIdentifier,
            Platform = Application.platform.ToString(),
            AppVersion = Application.version,
            SequenceNumber = ++sequenceNumber,
            Priority = priority,
            Properties = new Dictionary<string, object>(properties ?? new Dictionary<string, object>())
        };
        
        // 注入超级属性
        foreach (var sp in superProperties)
        {
            if (!evt.Properties.ContainsKey(sp.Key))
                evt.Properties[sp.Key] = sp.Value;
        }
        
        // 注入会话时长
        evt.Properties["session_duration_sec"] = 
            Mathf.RoundToInt(Time.realtimeSinceStartup - sessionStartTime);
        
        return evt;
    }

    void EnqueueEvent(AnalyticsEvent evt)
    {
        int queueIdx = (int)evt.Priority;
        if (queueIdx >= QueueCount) queueIdx = QueueCount - 1;
        
        var queue = priorityQueues[queueIdx];
        
        // 队列满时丢弃低优先级事件
        if (queue.Count > maxCacheSize / QueueCount)
        {
            if (evt.Priority >= AnalyticsEventPriority.Normal)
            {
                Debug.LogWarning($"[Analytics] Queue full, dropping event: {evt.EventName}");
                return;
            }
        }
        
        queue.Enqueue(evt);
    }

    IEnumerator BatchUploadRoutine()
    {
        while (true)
        {
            yield return new WaitForSeconds(batchInterval);
            
            if (Application.internetReachability == NetworkReachability.NotReachable)
            {
                // 无网络，缓存到磁盘
                SaveOfflineCache();
                continue;
            }
            
            // 从优先级队列收集一批事件
            var batch = CollectBatch(batchSize);
            if (batch.Count > 0)
            {
                yield return UploadBatch(batch);
            }
        }
    }

    List<AnalyticsEvent> CollectBatch(int maxCount)
    {
        var batch = new List<AnalyticsEvent>();
        
        // 优先从高优先级队列取
        for (int p = 0; p < QueueCount && batch.Count < maxCount; p++)
        {
            var queue = priorityQueues[p];
            while (queue.Count > 0 && batch.Count < maxCount)
            {
                batch.Add(queue.Dequeue());
            }
        }
        
        // 也包含离线缓存
        while (offlineCache.Count > 0 && batch.Count < maxCount)
        {
            batch.Add(offlineCache[0]);
            offlineCache.RemoveAt(0);
        }
        
        return batch;
    }

    IEnumerator UploadBatch(List<AnalyticsEvent> batch)
    {
        if (batch.Count == 0) yield break;
        
        string payload = SerializeBatch(batch);
        
        using var req = new UnityWebRequest(serverEndpoint, "POST");
        req.uploadHandler = new UploadHandlerRaw(
            System.Text.Encoding.UTF8.GetBytes(payload));
        req.downloadHandler = new DownloadHandlerBuffer();
        req.SetRequestHeader("Content-Type", "application/json");
        req.SetRequestHeader("X-Client-Version", Application.version);
        
        yield return req.SendWebRequest();
        
        if (req.result != UnityWebRequest.Result.Success)
        {
            Debug.LogWarning($"[Analytics] Upload failed: {req.error}");
            
            // 重试逻辑：将事件放回队列（最多重试3次）
            foreach (var evt in batch)
            {
                if (evt.RetryCount < 3)
                {
                    evt.RetryCount++;
                    offlineCache.Add(evt);
                }
            }
        }
    }

    string SerializeBatch(List<AnalyticsEvent> batch)
    {
        // 简化序列化（实际项目可用 MessagePack 或自定义格式）
        var events = batch.ConvertAll(e => new
        {
            event_name = e.EventName,
            timestamp = e.Timestamp,
            session_id = e.SessionId,
            user_id = e.UserId,
            device_id = e.DeviceId,
            seq = e.SequenceNumber,
            props = e.Properties
        });
        return JsonUtility.ToJson(new { events });
    }

    void SaveOfflineCache()
    {
        // 将队列中的事件保存到磁盘
        var allEvents = CollectBatch(maxCacheSize);
        offlineCache.AddRange(allEvents);
        
        // 写入磁盘（限制大小）
        if (offlineCache.Count > maxCacheSize)
            offlineCache = offlineCache.GetRange(0, maxCacheSize);
    }

    void LoadOfflineCache()
    {
        if (!File.Exists(cachePath)) return;
        // 从磁盘加载未发送的事件...
    }

    void OnApplicationQuitting()
    {
        Track("session_end", new Dictionary<string, object>
        {
            ["session_duration_sec"] = Time.realtimeSinceStartup - sessionStartTime
        }, AnalyticsEventPriority.High);
        
        SaveOfflineCache();
    }

    public void SetSuperProperty(string key, object value) => superProperties[key] = value;

    public void SetUser(string userId)
    {
        SetSuperProperty("user_id", userId);
        Track("user_login", new Dictionary<string, object> { ["user_id"] = userId },
            AnalyticsEventPriority.High);
    }

    /// <summary>
    /// GDPR 合规：用户撤回同意
    /// </summary>
    public void OptOut()
    {
        collectPersonalData = false;
        // 清除已收集的个人数据
        offlineCache.Clear();
        Track("user_opt_out", null, AnalyticsEventPriority.Critical);
        PlayerPrefs.SetInt("analytics_consent", 0);
    }

    public void OptIn()
    {
        collectPersonalData = true;
        PlayerPrefs.SetInt("analytics_consent", 1);
        Track("user_opt_in", null, AnalyticsEventPriority.Critical);
    }

    string GetUserId() => superProperties.TryGetValue("user_id", out var id) ? 
        id.ToString() : "anonymous";

    string DictToString(Dictionary<string, object> dict) =>
        dict == null ? "{}" : string.Join(", ", 
            System.Linq.Enumerable.Select(dict, kv => $"{kv.Key}={kv.Value}"));
}
```

## 四、Firebase Analytics 集成

```csharp
/// <summary>
/// Firebase Analytics 适配器
/// </summary>
public class FirebaseAnalyticsAdapter : MonoBehaviour
{
    void Start()
    {
        // 集成到主埋点管理器
        // AnalyticsManager.Instance 会在 Track 时调用此适配器
    }
    
    public void TrackFirebaseEvent(string eventName, Dictionary<string, object> props)
    {
        #if FIREBASE_ANALYTICS
        var parameters = new List<Firebase.Analytics.Parameter>();
        
        foreach (var kv in props ?? new Dictionary<string, object>())
        {
            switch (kv.Value)
            {
                case string s:
                    parameters.Add(new Firebase.Analytics.Parameter(kv.Key, s));
                    break;
                case float f:
                    parameters.Add(new Firebase.Analytics.Parameter(kv.Key, f));
                    break;
                case double d:
                    parameters.Add(new Firebase.Analytics.Parameter(kv.Key, d));
                    break;
                case int i:
                    parameters.Add(new Firebase.Analytics.Parameter(kv.Key, (long)i));
                    break;
            }
        }
        
        Firebase.Analytics.FirebaseAnalytics.LogEvent(eventName, parameters.ToArray());
        #endif
    }
}
```

## 六、设计要点

| 维度 | 建议 |
|------|------|
| 事件命名 | 统一规范：snake_case，使用常量，避免拼写错误 |
| 批量上报 | 合并30-50条事件一次发送，减少HTTP请求 |
| 离线缓存 | APP崩溃后重启继续上报未发送的事件 |
| 隐私合规 | GDPR地区需要用户明确同意，提供OptOut入口 |
| 测试验证 | 本地调试模式只打日志，不真实上报 |
| 数据质量 | 序列号检测乱序/重复，服务端做去重 |
