## 一、卡牌数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 卡牌稀有度
/// </summary>
public enum CardRarity { Common, Uncommon, Rare, Epic, Legendary }

/// <summary>
/// 卡牌类型
/// </summary>
public enum CardType { Creature, Spell, Equipment, Trap, Land }

/// <summary>
/// 卡牌关键词（如：穿透/连击/吸血/嘲讽）
/// </summary>
[Flags]
public enum CardKeyword
{
    None      = 0,
    Rush      = 1 << 0,    // 急袭：可立即攻击
    Taunt     = 1 << 1,    // 嘲讽：必须先攻击
    Lifesteal = 1 << 2,    // 吸血：造成伤害等量回血
    Poisonous = 1 << 3,    // 剧毒：任意伤害秒杀
    DivineShield = 1 << 4, // 圣盾：免疫一次伤害
    Stealth  = 1 << 5,     // 潜行：无法被指定攻击目标
    Charge   = 1 << 6,     // 冲锋（旧版Rush）
    Battlecry = 1 << 7,    // 战吼：打出时触发
    Deathrattle = 1 << 8,  // 亡语：死亡时触发
}

/// <summary>
/// 卡牌配置数据（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "Card", menuName = "CardGame/Card Data")]
public class CardData : ScriptableObject
{
    [Header("基础信息")]
    public string CardId;
    public string DisplayName;
    [TextArea(2, 4)] public string CardText;  // 效果描述文本
    public Sprite Artwork;
    public CardRarity Rarity;
    public CardType Type;
    
    [Header("费用/属性")]
    public int ManaCost;        // 费用
    public int Attack;          // 攻击力（生物牌）
    public int Health;          // 生命值（生物牌）
    
    [Header("关键词")]
    public CardKeyword Keywords;
    
    [Header("效果")]
    public List<CardEffect> Effects;
    
    [Header("牌组限制")]
    public int MaxCopiesInDeck = 2;  // 牌组中最多几张
    
    // 关键词检测
    public bool HasKeyword(CardKeyword keyword) => (Keywords & keyword) != 0;
}

/// <summary>
/// 卡牌效果（可视化配置）
/// </summary>
[Serializable]
public class CardEffect
{
    public CardEffectType EffectType;
    public CardEffectTrigger Trigger;
    public int Value;
    public string TargetId;     // 作用目标
    public bool IsRandom;       // 随机目标
}

public enum CardEffectType
{
    DealDamage,         // 造成伤害
    Heal,               // 回复生命
    DrawCard,           // 抽牌
    AddMana,            // 增加费用
    SummonMinion,       // 召唤随从
    BuffAttack,         // 增加攻击
    BuffHealth,         // 增加生命
    Silence,            // 沉默
    Destroy,            // 消灭
    Transform,          // 变形
    Discover,           // 发现（三选一）
}

public enum CardEffectTrigger
{
    OnPlay,             // 打出时（战吼）
    OnDeath,            // 死亡时（亡语）
    OnAttack,           // 攻击时
    OnDamaged,          // 受伤时
    OnTurnStart,        // 回合开始
    OnTurnEnd,          // 回合结束
    OnOtherCardPlayed,  // 其他牌打出时
    Passive,            // 持续效果
}
```

## 二、牌堆管理器

```csharp
/// <summary>
/// 牌堆（手牌/弃牌堆/牌库）
/// </summary>
public class CardDeck
{
    private List<CardData> library = new List<CardData>();  // 牌库
    private List<CardData> hand = new List<CardData>();     // 手牌
    private List<CardData> graveyard = new List<CardData>(); // 弃牌堆
    private List<CardData> played = new List<CardData>();   // 已打出的牌（场上）

    private System.Random rng;

    public IReadOnlyList<CardData> Hand => hand;
    public int LibraryCount => library.Count;
    public int HandCount => hand.Count;
    
    public event Action<CardData> OnCardDrawn;
    public event Action<CardData> OnCardPlayed;
    public event Action<CardData> OnCardDiscarded;
    public event Action OnLibraryEmpty;

    public CardDeck(List<CardData> deckList, int seed = -1)
    {
        library.AddRange(deckList);
        rng = seed >= 0 ? new System.Random(seed) : new System.Random();
        Shuffle();
    }

    /// <summary>
    /// 洗牌（Fisher-Yates 算法）
    /// </summary>
    public void Shuffle()
    {
        for (int i = library.Count - 1; i > 0; i--)
        {
            int j = rng.Next(0, i + 1);
            (library[i], library[j]) = (library[j], library[i]);
        }
    }

    /// <summary>
    /// 抽牌
    /// </summary>
    public CardData DrawCard()
    {
        if (library.Count == 0)
        {
            // 疲劳机制：牌库空时受到伤害
            OnLibraryEmpty?.Invoke();
            return null;
        }
        
        // 手牌上限检测（通常为10张）
        if (hand.Count >= 10)
        {
            // 爆牌：超出手牌上限的牌进入墓地（燃牌）
            CardData burned = library[0];
            library.RemoveAt(0);
            graveyard.Add(burned);
            Debug.Log($"[CardGame] 爆牌：{burned.DisplayName}");
            return null;
        }
        
        CardData card = library[0];
        library.RemoveAt(0);
        hand.Add(card);
        OnCardDrawn?.Invoke(card);
        return card;
    }

    public void DrawCards(int count)
    {
        for (int i = 0; i < count; i++)
            DrawCard();
    }

    /// <summary>
    /// 打出手牌
    /// </summary>
    public bool PlayCard(CardData card, int manaCost)
    {
        if (!hand.Contains(card))
        {
            Debug.LogError($"[CardGame] 试图打出不在手牌中的牌：{card.DisplayName}");
            return false;
        }
        
        hand.Remove(card);
        played.Add(card);
        OnCardPlayed?.Invoke(card);
        return true;
    }

    /// <summary>
    /// 弃牌（从手牌移到弃牌堆）
    /// </summary>
    public bool DiscardCard(CardData card)
    {
        if (!hand.Contains(card)) return false;
        
        hand.Remove(card);
        graveyard.Add(card);
        OnCardDiscarded?.Invoke(card);
        return true;
    }

    public bool DiscardRandom()
    {
        if (hand.Count == 0) return false;
        int idx = rng.Next(0, hand.Count);
        return DiscardCard(hand[idx]);
    }

    /// <summary>
    /// 从牌库中搜索特定卡牌（搜牌）
    /// </summary>
    public CardData SearchCard(Func<CardData, bool> predicate)
    {
        var found = library.Find(c => predicate(c));
        if (found != null)
        {
            library.Remove(found);
            hand.Add(found);
            OnCardDrawn?.Invoke(found);
        }
        return found;
    }

    /// <summary>
    /// 将弃牌堆洗回牌库
    /// </summary>
    public void RecycleGraveyard()
    {
        library.AddRange(graveyard);
        graveyard.Clear();
        Shuffle();
    }
}
```

## 三、费用系统

```csharp
/// <summary>
/// 法力费用管理（类似炉石传说）
/// </summary>
public class ManaSystem
{
    private int maxMana = 0;       // 最大法力水晶
    private int currentMana = 0;   // 当前可用法力
    private int overloadedMana = 0; // 过载锁定量（下回合少的法力）
    private int lockedMana = 0;    // 当前锁定量
    
    public const int MAX_MANA_CRYSTALS = 10;
    
    public int MaxMana => maxMana;
    public int CurrentMana => currentMana;
    public int EmptyMana => maxMana - currentMana;
    
    public event Action<int, int> OnManaChanged; // current, max

    public void OnTurnStart()
    {
        // 回合开始：恢复法力，处理过载
        if (maxMana < MAX_MANA_CRYSTALS)
            maxMana++;
        
        currentMana = maxMana - lockedMana;
        lockedMana = overloadedMana; // 上回合的过载这回合生效
        overloadedMana = 0;
        
        OnManaChanged?.Invoke(currentMana, maxMana);
    }

    public bool CanAfford(int cost) => currentMana >= cost;

    public bool Spend(int cost)
    {
        if (!CanAfford(cost)) return false;
        currentMana -= cost;
        OnManaChanged?.Invoke(currentMana, maxMana);
        return true;
    }

    public void AddMana(int amount)
    {
        currentMana = Mathf.Min(currentMana + amount, maxMana);
        OnManaChanged?.Invoke(currentMana, maxMana);
    }

    public void Overload(int amount)
    {
        overloadedMana += amount;
    }

    public void SetMaxMana(int max)
    {
        maxMana = Mathf.Clamp(max, 0, MAX_MANA_CRYSTALS);
        currentMana = Mathf.Min(currentMana, maxMana);
        OnManaChanged?.Invoke(currentMana, maxMana);
    }
}
```

## 四、效果链触发

```csharp
/// <summary>
/// 效果链执行器（战吼/亡语/触发效果的按序执行）
/// </summary>
public class CardEffectChain
{
    private Queue<CardEffectContext> pendingEffects = new Queue<CardEffectContext>();
    private bool isProcessing;

    public class CardEffectContext
    {
        public CardData Source;
        public CardEffect Effect;
        public GameObject Target;
        public int Value;
    }

    public void Enqueue(CardData source, CardEffect effect, GameObject target = null)
    {
        pendingEffects.Enqueue(new CardEffectContext
        {
            Source = source,
            Effect = effect,
            Target = target,
            Value = effect.Value
        });
        
        if (!isProcessing)
            ProcessNext();
    }

    async void ProcessNext()
    {
        if (pendingEffects.Count == 0)
        {
            isProcessing = false;
            return;
        }
        
        isProcessing = true;
        var ctx = pendingEffects.Dequeue();
        
        // 执行效果
        await ExecuteEffect(ctx);
        
        // 处理下一个
        ProcessNext();
    }

    async System.Threading.Tasks.Task ExecuteEffect(CardEffectContext ctx)
    {
        // 等待效果动画播放
        await System.Threading.Tasks.Task.Delay(300);
        
        switch (ctx.Effect.EffectType)
        {
            case CardEffectType.DealDamage:
                Debug.Log($"[CardEffect] {ctx.Source.DisplayName} 造成 {ctx.Value} 点伤害");
                break;
            case CardEffectType.DrawCard:
                Debug.Log($"[CardEffect] 抽 {ctx.Value} 张牌");
                break;
            case CardEffectType.Heal:
                Debug.Log($"[CardEffect] 回复 {ctx.Value} 点生命");
                break;
        }
    }
}
```

## 五、卡牌游戏核心循环

```
回合开始
  ↓
增加法力水晶（+1，最多10）
  ↓  
抽一张牌
  ↓
玩家操作阶段
  ├── 打出手牌（消耗费用）
  ├── 用随从攻击
  └── 使用英雄技能
  ↓
回合结束（触发回合结束效果）
  ↓
切换到对手回合
```

**卡牌游戏设计建议：**
1. 随机数使用种子（固定种子可重现对局，支持录像）
2. 爆牌（燃牌）机制增加手牌管理的决策深度
3. 过载机制限制强力效果的连续使用
4. 战吼 vs 亡语 形成进攻与防守的权衡
