## 战斗系统的本质

战斗系统是游戏中最核心、最复杂的子系统之一。它的复杂性来自：

1. **规则多变**：策划随时可能调整数值、添加新的效果
2. **交互复杂**：多个技能/Buff 同时作用，相互影响
3. **性能敏感**：战斗帧可能每帧处理数百个实体
4. **准确性要求高**：伤害计算错误直接影响游戏体验

好的战斗架构需要同时满足：**高扩展性**（应对策划的无尽需求）+ **高性能**（维持流畅帧率）+ **高可靠性**（计算结果正确）。

```csharp
/// <summary>
/// 伤害公式（支持简单的表达式）
/// </summary>
[Serializable]
public class DamageFormula
{
    // 基础伤害 = BaseMultiplier * 攻击力 + BaseFlatBonus
    public float BaseMultiplier;   // 攻击力系数，如 2.5f
    public int BaseFlatBonus;      // 固定加成，如 100
    
    // 示例：火球 = 2.5 * 攻击力 + 100
    // 普攻 = 1.0 * 攻击力 + 0
    
    public int Calculate(int attackPower)
    {
        return (int)(attackPower * BaseMultiplier) + BaseFlatBonus;
    }
}
```

### 1.3 技能执行器

```csharp
/// <summary>
/// 技能执行器：根据 SkillConfig 执行技能效果
/// </summary>
public class SkillExecutor
{
    private readonly BuffSystem _buffSystem;
    private readonly DamageSystem _damageSystem;
    private readonly EffectSystem _effectSystem;
    
    public SkillExecutor(BuffSystem buffSystem, DamageSystem damageSystem, EffectSystem effectSystem)
    {
        _buffSystem = buffSystem;
        _damageSystem = damageSystem;
        _effectSystem = effectSystem;
    }
    
    public async UniTask ExecuteAsync(SkillConfig skill, BattleUnit caster, 
                                       BattleUnit primaryTarget, CancellationToken ct)
    {
        // 1. 吟唱阶段
        if (skill.CastTime > 0)
        {
            caster.PlayAnimation(skill.CastAnimName);
            await UniTask.Delay(TimeSpan.FromSeconds(skill.CastTime), cancellationToken: ct);
        }
        
        // 2. 收集目标
        var targets = CollectTargets(skill, caster, primaryTarget);
        
        // 3. 依次执行效果
        foreach (var effect in skill.Effects)
        {
            await ExecuteEffect(effect, caster, targets, ct);
        }
    }
    
    private List<BattleUnit> CollectTargets(SkillConfig skill, BattleUnit caster, BattleUnit primary)
    {
        switch (skill.TargetType)
        {
            case TargetType.Single:
                return new List<BattleUnit> { primary };
                
            case TargetType.AOE:
                // 在主目标周围收集目标
                return BattleWorld.GetUnitsInRadius(
                    primary.Position, 
                    skill.Effects[0].AOERadius,
                    caster.IsEnemy(primary) ? UnitFilter.Enemy : UnitFilter.Ally,
                    maxCount: skill.Effects[0].MaxTargets
                );
                
            case TargetType.Self:
                return new List<BattleUnit> { caster };
                
            default:
                return new List<BattleUnit>();
        }
    }
    
    private async UniTask ExecuteEffect(SkillEffect effect, BattleUnit caster, 
                                          List<BattleUnit> targets, CancellationToken ct)
    {
        switch (effect.Type)
        {
            case EffectType.Damage:
                int baseDamage = effect.DamageFormula.Calculate(caster.AttackPower);
                foreach (var target in targets)
                    _damageSystem.ApplyDamage(new DamageInfo
                    {
                        Source = caster,
                        Target = target,
                        BaseDamage = baseDamage,
                        DamageType = effect.DamageType
                    });
                break;
                
            case EffectType.AddBuff:
                var buffConfig = BuffConfig.Get(effect.BuffId);
                foreach (var target in targets)
                    _buffSystem.AddBuff(target, buffConfig, effect.BuffDuration, caster);
                break;
                
            case EffectType.Heal:
                int healAmount = effect.DamageFormula.Calculate(caster.AttackPower);
                foreach (var target in targets)
                    target.Heal(healAmount);
                break;
                
            case EffectType.Knockback:
                foreach (var target in targets)
                {
                    var direction = (target.Position - caster.Position).normalized;
                    await target.ApplyKnockbackAsync(
                        direction * effect.KnockbackDistance, 
                        effect.KnockbackDuration, 
                        ct);
                }
                break;
        }
    }
}
```

## 二、伤害计算链：责任链模式

### 2.1 伤害流水线设计

```
原始伤害 → 攻击加成 → 防御减免 → 暴击计算 → Buff修正 → 最终伤害

每个步骤都是一个"处理器"，可以独立扩展
```

```csharp
/// <summary>
/// 伤害信息：在整个计算链中流转
/// </summary>
public class DamageInfo
{
    public BattleUnit Source;
    public BattleUnit Target;
    public int BaseDamage;
    public DamageType DamageType;
    
    // 计算过程中的中间状态
    public float AttackMultiplier = 1f;
    public float DefenseReduction = 0f;
    public bool IsCritical = false;
    public float CriticalMultiplier = 2f;
    public float BuffMultiplier = 1f;
    
    // 最终伤害（只读）
    public int FinalDamage { get; private set; }
    
    public void ComputeFinal()
    {
        float damage = BaseDamage * AttackMultiplier;
        damage -= damage * DefenseReduction;
        if (IsCritical) damage *= CriticalMultiplier;
        damage *= BuffMultiplier;
        FinalDamage = Math.Max(1, (int)damage);
    }
}

/// <summary>
/// 伤害处理器接口
/// </summary>
public interface IDamageProcessor
{
    void Process(DamageInfo damage);
}

/// <summary>
/// 攻击力加成处理器
/// </summary>
public class AttackBonusProcessor : IDamageProcessor
{
    public void Process(DamageInfo damage)
    {
        float bonus = damage.Source.GetBuffedStat(StatType.AttackMultiplier);
        damage.AttackMultiplier *= bonus;
    }
}

/// <summary>
/// 防御减免处理器
/// </summary>
public class DefenseProcessor : IDamageProcessor
{
    public void Process(DamageInfo damage)
    {
        int defense = damage.Target.GetEffectiveDefense(damage.DamageType);
        // 防御转化为百分比减免（有上限）
        damage.DefenseReduction = Mathf.Clamp(defense / (defense + 100f), 0f, 0.75f);
    }
}

/// <summary>
/// 暴击处理器
/// </summary>
public class CriticalProcessor : IDamageProcessor
{
    public void Process(DamageInfo damage)
    {
        float critRate = damage.Source.GetEffectiveStat(StatType.CriticalRate);
        damage.IsCritical = UnityEngine.Random.value < critRate;
        
        if (damage.IsCritical)
        {
            damage.CriticalMultiplier = damage.Source.GetEffectiveStat(StatType.CriticalDamage);
        }
    }
}

/// <summary>
/// 伤害计算链（责任链模式）
/// </summary>
public class DamageSystem
{
    private readonly List<IDamageProcessor> _processors = new();
    
    public DamageSystem()
    {
        // 注册处理器（顺序重要）
        _processors.Add(new AttackBonusProcessor());
        _processors.Add(new DefenseProcessor());
        _processors.Add(new CriticalProcessor());
        _processors.Add(new BuffDamageModifierProcessor());
        _processors.Add(new ShieldAbsorptionProcessor());
        // 可以轻松添加新的处理器
    }
    
    public void ApplyDamage(DamageInfo damage)
    {
        // 依次通过所有处理器
        foreach (var processor in _processors)
            processor.Process(damage);
        
        // 计算最终伤害
        damage.ComputeFinal();
        
        // 应用伤害
        damage.Target.HP -= damage.FinalDamage;
        
        // 发布伤害事件（UI 飘字、特效等订阅）
        EventBus.Publish(new DamageDealtEvent
        {
            Source = damage.Source,
            Target = damage.Target,
            FinalDamage = damage.FinalDamage,
            IsCritical = damage.IsCritical
        });
        
        // 检查死亡
        if (damage.Target.HP <= 0)
        {
            EventBus.Publish(new UnitDiedEvent { Unit = damage.Target, Killer = damage.Source });
        }
    }
}
```

### 3.1 Buff 的数据模型

```csharp
/// <summary>
/// Buff 配置（配置表）
/// </summary>
[Serializable]
public class BuffConfig
{
    public int BuffId;
    public string BuffName;
    public BuffType Type;               // 增益/减益/中性
    public int MaxStacks;               // 最大层数（1=不可叠加）
    public StackBehavior StackBehavior; // 叠加行为
    
    public StatModifier[] StatModifiers;  // 属性修改
    public PeriodicEffect PeriodicEffect; // 持续效果（每 X 秒触发）
}

public enum BuffType { Buff, Debuff, Neutral }
public enum StackBehavior 
{ 
    Refresh,        // 刷新持续时间
    Independent,    // 独立计时
    AddStack,       // 增加层数
}

/// <summary>
/// 属性修改器：可以是加法或乘法
/// </summary>
[Serializable]
public struct StatModifier
{
    public StatType Stat;           // 修改哪个属性
    public ModifierType ModType;    // 加法/乘法/覆盖
    public float Value;             // 修改值
}

public enum ModifierType { Additive, Multiplicative, Override }

/// <summary>
/// 持续效果（如每秒燃烧伤害）
/// </summary>
[Serializable]
public class PeriodicEffect
{
    public float Interval;      // 触发间隔（秒）
    public EffectType Type;     // 效果类型
    public DamageFormula Formula;
}

/// <summary>
/// 运行时 Buff 实例
/// </summary>
public class BuffInstance
{
    public BuffConfig Config;
    public BattleUnit Caster;       // 施加者（用于计算伤害）
    public float RemainingDuration;
    public int CurrentStacks;
    private float _periodicTimer;
    
    public void Update(float deltaTime, BattleUnit owner)
    {
        RemainingDuration -= deltaTime;
        
        // 处理持续效果
        if (Config.PeriodicEffect != null)
        {
            _periodicTimer += deltaTime;
            while (_periodicTimer >= Config.PeriodicEffect.Interval)
            {
                _periodicTimer -= Config.PeriodicEffect.Interval;
                TriggerPeriodicEffect(owner);
            }
        }
    }
    
    private void TriggerPeriodicEffect(BattleUnit owner)
    {
        if (Config.PeriodicEffect.Type == EffectType.Damage)
        {
            int damage = Config.PeriodicEffect.Formula.Calculate(Caster.AttackPower);
            BattleWorld.DamageSystem.ApplyDamage(new DamageInfo
            {
                Source = Caster,
                Target = owner,
                BaseDamage = damage,
                DamageType = DamageType.Magical
            });
        }
    }
}
```

### 3.2 Buff 系统管理器

```csharp
/// <summary>
/// Buff 系统：管理单个 BattleUnit 的所有 Buff
/// </summary>
public class UnitBuffSystem
{
    private readonly BattleUnit _owner;
    private readonly List<BuffInstance> _buffs = new(8);
    
    // 属性缓存（避免每帧重新计算）
    private readonly Dictionary<StatType, float> _cachedStats = new();
    private bool _statsDirty = true;
    
    public UnitBuffSystem(BattleUnit owner) => _owner = owner;
    
    public void AddBuff(BuffConfig config, float duration, BattleUnit caster)
    {
        // 查找是否已有同类 Buff
        var existing = _buffs.Find(b => b.Config.BuffId == config.BuffId);
        
        if (existing != null)
        {
            switch (config.StackBehavior)
            {
                case StackBehavior.Refresh:
                    existing.RemainingDuration = duration;
                    break;
                    
                case StackBehavior.AddStack:
                    if (existing.CurrentStacks < config.MaxStacks)
                    {
                        existing.CurrentStacks++;
                        existing.RemainingDuration = duration;
                        _statsDirty = true;
                    }
                    break;
                    
                case StackBehavior.Independent:
                    // 添加新的独立实例
                    AddNewInstance(config, duration, caster);
                    break;
            }
        }
        else
        {
            AddNewInstance(config, duration, caster);
        }
        
        _statsDirty = true;
    }
    
    public void RemoveBuff(int buffId)
    {
        _buffs.RemoveAll(b => b.Config.BuffId == buffId);
        _statsDirty = true;
    }
    
    public void Update(float deltaTime)
    {
        for (int i = _buffs.Count - 1; i >= 0; i--)
        {
            _buffs[i].Update(deltaTime, _owner);
            
            if (_buffs[i].RemainingDuration <= 0)
            {
                _buffs.RemoveAt(i);
                _statsDirty = true;
            }
        }
    }
    
    /// <summary>
    /// 获取某属性的最终修正值（使用缓存）
    /// </summary>
    public float GetStatMultiplier(StatType stat)
    {
        if (_statsDirty) RebuildStatCache();
        return _cachedStats.TryGetValue(stat, out float value) ? value : 1f;
    }
    
    private void RebuildStatCache()
    {
        _cachedStats.Clear();
        
        // 收集所有 Buff 的属性修改器
        var additiveSums = new Dictionary<StatType, float>();
        var multiplicativeProducts = new Dictionary<StatType, float>();
        
        foreach (var buff in _buffs)
        {
            foreach (var modifier in buff.Config.StatModifiers)
            {
                float stackMultiplier = buff.CurrentStacks;
                
                switch (modifier.ModType)
                {
                    case ModifierType.Additive:
                        if (!additiveSums.ContainsKey(modifier.Stat))
                            additiveSums[modifier.Stat] = 0;
                        additiveSums[modifier.Stat] += modifier.Value * stackMultiplier;
                        break;
                        
                    case ModifierType.Multiplicative:
                        if (!multiplicativeProducts.ContainsKey(modifier.Stat))
                            multiplicativeProducts[modifier.Stat] = 1f;
                        multiplicativeProducts[modifier.Stat] *= 
                            Mathf.Pow(1f + modifier.Value, stackMultiplier);
                        break;
                }
            }
        }
        
        // 合并：先加法，再乘法
        var allStats = new HashSet<StatType>(additiveSums.Keys);
        allStats.UnionWith(multiplicativeProducts.Keys);
        
        foreach (var stat in allStats)
        {
            float additive = additiveSums.GetValueOrDefault(stat, 0f);
            float multiplicative = multiplicativeProducts.GetValueOrDefault(stat, 1f);
            _cachedStats[stat] = (1f + additive) * multiplicative;
        }
        
        _statsDirty = false;
    }
    
    private void AddNewInstance(BuffConfig config, float duration, BattleUnit caster)
    {
        _buffs.Add(new BuffInstance
        {
            Config = config,
            Caster = caster,
            RemainingDuration = duration,
            CurrentStacks = 1
        });
    }
}
```

## 四、战斗状态机

### 4.1 战斗单元的状态图

```
                    ┌─────────────────────────────────┐
                    ↓                                 │
Idle ──────→ Casting ──────→ AttackRecovery ─────────┘
  ↑↓           │                   
  │            ↓ 被打断            
  │        Interrupted ─────→ Idle
  │            
  ↓ 受到伤害            
Stunned ──────────────────────────────────→ Idle
  │
  ↓ HP = 0
Dying ──────────────────────────────────→ Dead
```

```csharp
public enum BattleUnitState
{
    Idle,           // 待机
    Moving,         // 移动
    Casting,        // 吟唱技能
    AttackRecovery, // 攻击后摇
    Stunned,        // 眩晕
    Knocked,        // 被击飞
    Dying,          // 死亡动画
    Dead,           // 已死亡
}

/// <summary>
/// 战斗单元状态机
/// </summary>
public class BattleUnitStateMachine
{
    private BattleUnitState _currentState;
    private float _stateTimer;       // 当前状态已持续时间
    private float _stateDuration;    // 当前状态持续时间（0=无限制）
    
    public BattleUnitState Current => _currentState;
    
    // 状态变化事件
    public event Action<BattleUnitState, BattleUnitState> OnStateChanged;
    
    public bool CanCastSkill => _currentState == BattleUnitState.Idle || 
                                 _currentState == BattleUnitState.Moving;
    
    public bool CanMove => _currentState == BattleUnitState.Idle || 
                           _currentState == BattleUnitState.Moving;
    
    public bool CanReceiveDamage => _currentState != BattleUnitState.Dead;
    
    public void TransitionTo(BattleUnitState newState, float duration = 0f)
    {
        if (_currentState == newState) return;
        
        // 验证状态转换合法性
        if (!IsTransitionValid(_currentState, newState))
        {
            Debug.LogWarning($"无效的状态转换: {_currentState} → {newState}");
            return;
        }
        
        var prevState = _currentState;
        _currentState = newState;
        _stateTimer = 0f;
        _stateDuration = duration;
        
        OnStateChanged?.Invoke(prevState, newState);
    }
    
    public void Update(float deltaTime)
    {
        _stateTimer += deltaTime;
        
        // 有时间限制的状态自动结束
        if (_stateDuration > 0 && _stateTimer >= _stateDuration)
        {
            OnStateTimeout();
        }
    }
    
    private void OnStateTimeout()
    {
        switch (_currentState)
        {
            case BattleUnitState.Stunned:
            case BattleUnitState.Knocked:
            case BattleUnitState.AttackRecovery:
                TransitionTo(BattleUnitState.Idle);
                break;
                
            case BattleUnitState.Dying:
                TransitionTo(BattleUnitState.Dead);
                break;
        }
    }
    
    private bool IsTransitionValid(BattleUnitState from, BattleUnitState to)
    {
        // 死亡状态不可逃离
        if (from == BattleUnitState.Dead) return false;
        
        // 吟唱中只能被打断
        if (from == BattleUnitState.Casting && 
            to != BattleUnitState.Stunned && 
            to != BattleUnitState.Knocked &&
            to != BattleUnitState.Dying)
            return false;
        
        return true;
    }
}
```

```csharp
    public readonly BattleUnitStateMachine StateMachine;
    public readonly SkillManager SkillManager;

    public FixedVector2 Position { get; set; }
    
    public BattleUnit(int unitId, BattleUnitConfig config)
    {
        UnitId = unitId;
        MaxHP = HP = config.MaxHP;
        BaseAttackPower = config.AttackPower;
        BaseDefense = config.Defense;
        BaseMoveSpeed = config.MoveSpeed;
        
        BuffSystem = new UnitBuffSystem(this);
        StateMachine = new BattleUnitStateMachine();
        SkillManager = new SkillManager(this);
    }
    
    /// <summary>
    /// 获取经过 Buff 修正后的属性
    /// </summary>
    public int AttackPower
    {
        get
        {
            float multiplier = BuffSystem.GetStatMultiplier(StatType.Attack);
            return (int)(BaseAttackPower * multiplier);
        }
    }
    
    public int GetEffectiveDefense(DamageType damageType)
    {
        float multiplier = BuffSystem.GetStatMultiplier(
            damageType == DamageType.Physical ? StatType.PhysicalDefense : StatType.MagicDefense);
        return (int)(BaseDefense * multiplier);
    }
    
    public void Update(float deltaTime)
    {
        StateMachine.Update(deltaTime);
        BuffSystem.Update(deltaTime);
        SkillManager.Update(deltaTime);
    }
    
    public bool IsEnemy(BattleUnit other) => /* 阵营判断逻辑 */ true;
    
    public void Heal(int amount) 
    { 
        HP = Math.Min(MaxHP, HP + amount);
        EventBus.Publish(new HealEvent { Unit = this, Amount = amount });
    }
}
```

## 六、性能优化：大量单位场景

```csharp
/// <summary>
/// 战场管理器：批量更新所有战斗单元
/// 优化：避免每个单位都有 MonoBehaviour Update
/// </summary>
public class BattleWorld
{
    private readonly List<BattleUnit> _units = new(256);
    public static DamageSystem DamageSystem { get; private set; }
    
    // 空间索引：快速找到范围内的单位（避免全场遍历）
    private readonly SpatialGrid _spatialGrid;
    
    public void Update(float deltaTime)
    {
        // 批量更新所有单位
        int count = _units.Count;
        for (int i = 0; i < count; i++)
            _units[i].Update(deltaTime);
    }
    
    /// <summary>
    /// 获取范围内的单位（使用空间网格加速）
    /// </summary>
    public List<BattleUnit> GetUnitsInRadius(FixedVector2 center, Fixed64 radius, 
                                               UnitFilter filter, int maxCount = int.MaxValue)
    {
        // 使用空间网格避免 O(n²) 遍历
        return _spatialGrid.QueryRadius(center, radius, filter, maxCount);
    }
}

/// <summary>
/// 空间网格：将战场划分为网格，每个格子存储其中的单位
/// </summary>
public class SpatialGrid
{
    private const int CELL_SIZE = 5; // 每格5米
    private readonly Dictionary<(int, int), List<BattleUnit>> _cells = new();
    
    public void UpdateUnit(BattleUnit unit, FixedVector2 oldPos, FixedVector2 newPos)
    {
        var oldCell = GetCell(oldPos);
        var newCell = GetCell(newPos);
        
        if (oldCell == newCell) return;
        
        if (_cells.TryGetValue(oldCell, out var oldList))
            oldList.Remove(unit);
        
        if (!_cells.TryGetValue(newCell, out var newList))
        {
            newList = new List<BattleUnit>(4);
            _cells[newCell] = newList;
        }
        newList.Add(unit);
    }
    
    public List<BattleUnit> QueryRadius(FixedVector2 center, Fixed64 radius, 
                                          UnitFilter filter, int maxCount)
    {
        var result = new List<BattleUnit>();
        int cellRadius = (int)(radius.ToFloat() / CELL_SIZE) + 1;
        var centerCell = GetCell(center);
        
        for (int dx = -cellRadius; dx <= cellRadius && result.Count < maxCount; dx++)
        {
            for (int dy = -cellRadius; dy <= cellRadius && result.Count < maxCount; dy++)
            {
                var cell = (centerCell.Item1 + dx, centerCell.Item2 + dy);
                if (!_cells.TryGetValue(cell, out var units)) continue;
                
                foreach (var unit in units)
                {
                    if (result.Count >= maxCount) break;
                    if (!MatchesFilter(unit, filter)) continue;
                    
                    // 精确距离检查
                    var diff = unit.Position - center;
                    var distSq = diff.X * diff.X + diff.Y * diff.Y;
                    if (distSq <= radius * radius)
                        result.Add(unit);
                }
            }
        }
        
        return result;
    }
    
    private (int, int) GetCell(FixedVector2 pos)
    {
        return ((int)(pos.X.ToFloat() / CELL_SIZE), (int)(pos.Y.ToFloat() / CELL_SIZE));
    }
    
    private bool MatchesFilter(BattleUnit unit, UnitFilter filter)
    {
        return filter switch
        {
            UnitFilter.All => true,
            UnitFilter.Enemy => true, // 具体逻辑根据项目实现
            UnitFilter.Ally => false,
            _ => true
        };
    }
}
```

## 总结

好的战斗系统架构需要把握以下核心原则：

1. **数据驱动**：把变化的（数值、效果）放到配置，把不变的（计算逻辑）放到代码
2. **职责单一**：伤害计算、Buff管理、状态机各司其职，互不侵入
3. **扩展优先**：通过责任链/策略模式添加新效果，不修改已有代码
4. **性能意识**：空间索引、批量更新、属性缓存，预先考虑大量单位场景
