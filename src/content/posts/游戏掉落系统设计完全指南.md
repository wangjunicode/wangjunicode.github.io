## 一、掉落系统的核心问题

掉落系统设计需要回答以下问题：

1. **随机性来源**：伪随机、种子随机还是服务端随机？
2. **概率模型**：等概率、权重池还是独立概率？
3. **保底机制**：连续未出时如何提升概率？
4. **数量分布**：固定数量、区间随机还是根据怪物等级？
5. **防作弊**：如何防止客户端篡改掉落结果？

### 2.1 权重池随机

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 权重池随机选择器
/// </summary>
public class WeightedRandomSelector<T>
{
    private List<(T item, float weight, float cumulativeWeight)> entries 
        = new List<(T, float, float)>();
    private float totalWeight;

    public void Add(T item, float weight)
    {
        if (weight <= 0) return;
        totalWeight += weight;
        entries.Add((item, weight, totalWeight));
    }

    public T Select(System.Random rng = null)
    {
        if (entries.Count == 0) return default;
        
        double roll = rng != null ? 
            rng.NextDouble() * totalWeight : 
            UnityEngine.Random.value * totalWeight;
        
        // 二分查找（大池子时性能好）
        int lo = 0, hi = entries.Count - 1;
        while (lo < hi)
        {
            int mid = (lo + hi) / 2;
            if (entries[mid].cumulativeWeight < roll)
                lo = mid + 1;
            else
                hi = mid;
        }
        
        return entries[lo].item;
    }

    /// <summary>
    /// 不放回随机选择（每次选出后从池中临时移除）
    /// </summary>
    public List<T> SelectMultipleWithoutReplacement(int count, System.Random rng = null)
    {
        if (count >= entries.Count) 
            return entries.ConvertAll(e => e.item);
        
        var remaining = new List<(T item, float weight)>(
            entries.ConvertAll(e => (e.item, e.weight)));
        
        var results = new List<T>();
        
        for (int i = 0; i < count && remaining.Count > 0; i++)
        {
            float total = 0;
            foreach (var r in remaining) total += r.weight;
            
            double roll = rng != null ? rng.NextDouble() * total : 
                UnityEngine.Random.value * total;
            
            float cumulative = 0;
            for (int j = 0; j < remaining.Count; j++)
            {
                cumulative += remaining[j].weight;
                if (cumulative >= roll)
                {
                    results.Add(remaining[j].item);
                    remaining.RemoveAt(j);
                    break;
                }
            }
        }
        
        return results;
    }
    
    public void Clear() { entries.Clear(); totalWeight = 0; }
    public int Count => entries.Count;
    public float TotalWeight => totalWeight;
}
```

### 2.2 独立概率模型

```csharp
/// <summary>
/// 独立概率掉落（每个物品单独判定）
/// 多个物品可以同时掉落
/// </summary>
public class IndependentDropChecker
{
    [Serializable]
    public class DropEntry
    {
        public string ItemId;
        public float Probability;   // 0-1
        public int MinCount;
        public int MaxCount;
    }

    public List<(string itemId, int count)> Roll(List<DropEntry> entries)
    {
        var results = new List<(string, int)>();
        
        foreach (var entry in entries)
        {
            if (UnityEngine.Random.value <= entry.Probability)
            {
                int count = UnityEngine.Random.Range(entry.MinCount, entry.MaxCount + 1);
                results.Add((entry.ItemId, count));
            }
        }
        
        return results;
    }
}
```

```csharp
    public enum DropTableMode
    {
        WeightedPool,      // 权重池：抽出N个
        IndependentChance, // 独立概率：每个单独判定
        Mixed              // 混合：保证+权重池+独立稀有
    }
}

[Serializable]
public class GuaranteedDrop
{
    public string ItemId;
    public int CountMin = 1;
    public int CountMax = 1;
    [Tooltip("只有满足条件才保底掉落")]
    public string Condition; // 空=始终
}

[Serializable]
public class RandomDropEntry
{
    public string ItemId;
    public float Weight = 100f;
    public ItemRarity Rarity = ItemRarity.Common;
    public int CountMin = 1;
    public int CountMax = 1;
    [Tooltip("每个结果的子掉落表（支持嵌套）")]
    public DropTable SubTable;
}

[Serializable]
public class RareDrop
{
    public string ItemId;
    [Range(0f, 1f)] public float Probability = 0.01f;
    public ItemRarity Rarity = ItemRarity.Rare;
}

[Serializable]
public class PityConfig
{
    public bool Enabled = false;
    public string PityKey;              // 保底计数器Key（每个掉落表独立）
    public int PityThreshold = 50;      // 多少次未出触发保底
    public string PityItemId;           // 保底物品ID
    public float PityRampStartAt = 30;  // 从第几次开始提升概率（软保底）
    public float PityRampPerRoll = 0.02f; // 每次额外增加的概率
}

public enum ItemRarity 
{ 
    Common, Uncommon, Rare, Epic, Legendary, Mythic 
}
```

## 四、掉落管理器

```csharp
/// <summary>
/// 掉落管理器（核心逻辑）
/// </summary>
public class DropManager : MonoBehaviour
{
    private static DropManager instance;
    public static DropManager Instance => instance;

    // 保底计数器（玩家唯一，持久化）
    private Dictionary<string, int> pityCounters = new Dictionary<string, int>();
    
    // 随机数生成器（使用固定种子保证复现性，用于调试）
    private System.Random rng = new System.Random();

    void Awake()
    {
        instance = this;
        LoadPityCounters();
    }

    /// <summary>
    /// 执行掉落
    /// </summary>
    public List<DropResult> ExecuteDrop(DropTable table, DropContext context = null)
    {
        var results = new List<DropResult>();
        
        if (table == null) return results;
        
        // 1. 必掉物品
        foreach (var guaranteed in table.GuaranteedDrops ?? Array.Empty<GuaranteedDrop>())
        {
            if (!CheckCondition(guaranteed.Condition, context)) continue;
            
            int count = UnityEngine.Random.Range(guaranteed.CountMin, guaranteed.CountMax + 1);
            results.Add(new DropResult { ItemId = guaranteed.ItemId, Count = count });
        }
        
        // 2. 主要掉落（根据模式）
        switch (table.Mode)
        {
            case DropTable.DropTableMode.WeightedPool:
                RollWeightedPool(table, context, results);
                break;
            case DropTable.DropTableMode.IndependentChance:
                RollIndependent(table, context, results);
                break;
            case DropTable.DropTableMode.Mixed:
                RollWeightedPool(table, context, results);
                RollIndependent(table, context, results); // 再跑稀有独立
                break;
        }
        
        // 3. 稀有独立掉落
        foreach (var rare in table.RareDrops ?? Array.Empty<RareDrop>())
        {
            if ((float)rng.NextDouble() <= rare.Probability)
            {
                results.Add(new DropResult { ItemId = rare.ItemId, Count = 1, IsRareProc = true });
            }
        }
        
        // 4. 保底检查
        if (table.Pity.Enabled)
        {
            var pityResult = CheckPity(table.Pity, results);
            if (pityResult != null)
                results.Add(pityResult);
        }
        
        // 5. 处理嵌套掉落表
        var subDrops = new List<DropResult>();
        foreach (var result in results)
        {
            var entry = Array.Find(table.RandomDrops, 
                e => e.ItemId == result.ItemId && e.SubTable != null);
            if (entry != null)
            {
                subDrops.AddRange(ExecuteDrop(entry.SubTable, context));
            }
        }
        results.AddRange(subDrops);
        
        return results;
    }

    void RollWeightedPool(DropTable table, DropContext context, List<DropResult> results)
    {
        if (table.RandomDrops == null || table.RandomDrops.Length == 0) return;
        
        var selector = new WeightedRandomSelector<RandomDropEntry>();
        
        foreach (var entry in table.RandomDrops)
        {
            float weight = entry.Weight;
            // 应用上下文修正（如击杀精英怪提升稀有度概率）
            if (context != null)
                weight *= GetLuckMultiplier(context, entry.Rarity);
            
            selector.Add(entry, weight);
        }
        
        int dropCount = UnityEngine.Random.Range(table.MinDropCount, table.MaxDropCount + 1);
        
        for (int i = 0; i < dropCount; i++)
        {
            var chosen = selector.Select(rng);
            if (chosen != null)
            {
                int count = UnityEngine.Random.Range(chosen.CountMin, chosen.CountMax + 1);
                results.Add(new DropResult 
                { 
                    ItemId = chosen.ItemId, 
                    Count = count,
                    Rarity = chosen.Rarity 
                });
            }
        }
    }

    void RollIndependent(DropTable table, DropContext context, List<DropResult> results)
    {
        foreach (var entry in table.RandomDrops ?? Array.Empty<RandomDropEntry>())
        {
            float prob = entry.Weight / 100f; // 这里weight当概率用（0-100）
            if (context != null) prob *= GetLuckMultiplier(context, entry.Rarity);
            prob = Mathf.Clamp01(prob);
            
            if ((float)rng.NextDouble() <= prob)
            {
                int count = UnityEngine.Random.Range(entry.CountMin, entry.CountMax + 1);
                results.Add(new DropResult { ItemId = entry.ItemId, Count = count });
            }
        }
    }

    DropResult CheckPity(PityConfig pity, List<DropResult> currentResults)
    {
        // 本次有没有出保底物品
        bool hitPityItem = currentResults.Exists(r => r.ItemId == pity.PityItemId 
            || r.IsRareProc);
        
        if (!pityCounters.ContainsKey(pity.PityKey))
            pityCounters[pity.PityKey] = 0;
        
        if (hitPityItem)
        {
            // 重置计数器
            pityCounters[pity.PityKey] = 0;
            SavePityCounters();
            return null;
        }
        
        pityCounters[pity.PityKey]++;
        int counter = pityCounters[pity.PityKey];
        
        // 软保底：从 PityRampStartAt 开始每次额外增加概率
        if (counter >= pity.PityRampStartAt)
        {
            float extraProb = (counter - pity.PityRampStartAt) * pity.PityRampPerRoll;
            if ((float)rng.NextDouble() <= extraProb)
            {
                pityCounters[pity.PityKey] = 0;
                SavePityCounters();
                return new DropResult { ItemId = pity.PityItemId, Count = 1, IsPityDrop = true };
            }
        }
        
        // 硬保底：达到阈值必出
        if (counter >= pity.PityThreshold)
        {
            pityCounters[pity.PityKey] = 0;
            SavePityCounters();
            return new DropResult { ItemId = pity.PityItemId, Count = 1, IsPityDrop = true };
        }
        
        SavePityCounters();
        return null;
    }

    float GetLuckMultiplier(DropContext context, ItemRarity rarity)
    {
        if (context == null) return 1f;
        
        return rarity switch
        {
            ItemRarity.Rare => 1f + context.LuckBonus * 0.5f,
            ItemRarity.Epic => 1f + context.LuckBonus * 0.3f,
            ItemRarity.Legendary => 1f + context.LuckBonus * 0.1f,
            _ => 1f
        };
    }

    bool CheckCondition(string condition, DropContext context)
    {

/// </summary>
```

```csharp
/// <summary>
/// 掉落概率展示（向玩家公示概率，合规要求）
/// </summary>
public class DropOddsDisplay : MonoBehaviour
{
    [SerializeField] private DropTable dropTable;

    public string GetOddsText()
    {
        var sb = new System.Text.StringBuilder();
        sb.AppendLine("掉落概率：");
        
        float totalWeight = 0;
        foreach (var entry in dropTable.RandomDrops)
            totalWeight += entry.Weight;
        
        foreach (var entry in dropTable.RandomDrops)
        {
            float prob = entry.Weight / totalWeight * 100f;
            sb.AppendLine($"{entry.ItemId}（{entry.Rarity}）: {prob:F2}%");
        }
        
        foreach (var rare in dropTable.RareDrops)
        {
            sb.AppendLine($"{rare.ItemId}（{rare.Rarity}）: {rare.Probability * 100:F4}%");
        }
        
        if (dropTable.Pity.Enabled)
        {
            sb.AppendLine($"保底：每{dropTable.Pity.PityThreshold}次必出 {dropTable.Pity.PityItemId}");
            sb.AppendLine($"软保底：第{dropTable.Pity.PityRampStartAt}次起每次额外+{dropTable.Pity.PityRampPerRoll*100:F1}%");
        }
        
        return sb.ToString();
    }
}
```

## 六、设计要点汇总

| 概率模型 | 适用场景 | 优缺点 |
|----------|----------|--------|
| 等权重随机 | 简单材料、金币 | 简单，无稀有感 |
| 权重池 | 主要战利品、装备 | 灵活可配，支持稀有度分层 |
| 独立概率 | 稀有材料、特殊道具 | 多物品可同时掉落 |
| 保底机制 | 抽卡、稀有装备 | 合规要求，防止极端运气差体验 |
| 嵌套掉落表 | 宝箱、BOSS | 支持复杂组合，便于内容管理 |

**关键设计原则：**
1. 服务端执行掉落判定，客户端只做展示
2. 保底计数器服务端存储，防止删档重置
3. 按法规要求向玩家公示抽卡/掉落概率
4. 支持运营活动期间临时调整掉落率（不修改代码）
