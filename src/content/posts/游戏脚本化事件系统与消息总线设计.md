# 游戏脚本化事件系统与消息总线设计

游戏各模块之间的通信是架构设计的核心挑战。直接引用导致强耦合，改一处牵动全身。事件系统/消息总线是解耦的利器，让模块之间"只需知道发生了什么，不需要知道谁在处理"。

## 一、为什么需要事件系统

### 1.1 直接引用的问题

```csharp
// ❌ 强耦合：敌人死亡时，需要知道所有"感兴趣的系统"
public class Enemy
{
    // 这里直接引用了所有依赖它的系统 → 高耦合
    public QuestManager questManager;
    public AchievementSystem achievement;
    public AudioSystem audio;
    public UISystem ui;
    public DropSystem dropSystem;
    
    void Die()
    {
        questManager.OnEnemyKilled(this);      // 更新任务
        achievement.OnEnemyKilled(this);       // 更新成就
        audio.PlayDeathSound(transform.position);  // 播放音效
        ui.ShowKillEffect(transform.position); // 显示击杀特效
        dropSystem.SpawnDrops(transform.position); // 生成掉落
    }
}
// 问题：添加新系统（如排行榜）→ 必须改 Enemy 代码！
```

```csharp
// ✅ 事件驱动：敌人只需发布事件，不关心谁处理
public class Enemy
{
    void Die()
    {
        // 只需发布一个事件
        EventBus.Publish(new EnemyDiedEvent(this));
    }
}

// 各系统独立订阅，Enemy 完全不知道它们的存在
// 新增系统时，只需订阅事件，零修改 Enemy 代码 ✅
```

## 二、类型安全事件系统

### 2.1 泛型事件总线

```csharp
// 类型安全的事件总线（编译期类型检查）
public static class EventBus
{
    // 每种事件类型有独立的订阅者集合
    private static readonly Dictionary<Type, List<Delegate>> _handlers = new();
    
    // 订阅
    public static void Subscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);
        if (!_handlers.TryGetValue(type, out var list))
        {
            list = new List<Delegate>();
            _handlers[type] = list;
        }
        list.Add(handler);
    }
    
    // 取消订阅
    public static void Unsubscribe<T>(Action<T> handler) where T : struct
    {
        var type = typeof(T);
        if (_handlers.TryGetValue(type, out var list))
            list.Remove(handler);
    }
    
    // 发布事件
    public static void Publish<T>(T eventData) where T : struct
    {
        var type = typeof(T);
        if (!_handlers.TryGetValue(type, out var list)) return;
        
        // 遍历副本（处理过程中可能有取消订阅）
        var handlers = list.ToArray();
        foreach (var handler in handlers)
        {
            try
            {
                ((Action<T>)handler).Invoke(eventData);
            }
            catch (Exception e)
            {
                Debug.LogError($"事件处理异常 [{typeof(T).Name}]: {e}");
            }
        }
    }
    
    // 清除所有订阅（场景切换时调用）
    public static void Clear() => _handlers.Clear();
    
    // 清除特定类型的订阅
    public static void Clear<T>() where T : struct 
        => _handlers.Remove(typeof(T));
}

// 事件定义（值类型，零GC）
public struct EnemyDiedEvent
{
    public int EnemyId;
    public Vector3 Position;
    public int ExperienceReward;
    public string EnemyType;
}

public struct PlayerLevelUpEvent
{
    public int PlayerId;
    public int NewLevel;
    public int OldLevel;
}

public struct ItemPickedUpEvent
{
    public int PlayerId;
    public int ItemId;
    public int Count;
}
```

### 2.2 订阅者管理

```csharp
// 使用 MonoBehaviour 基类自动管理订阅生命周期
public abstract class EventSubscriber : MonoBehaviour
{
    private readonly List<Action> _unsubscribeActions = new();
    
    protected void Subscribe<T>(Action<T> handler) where T : struct
    {
        EventBus.Subscribe(handler);
        _unsubscribeActions.Add(() => EventBus.Unsubscribe(handler));
    }
    
    private void OnDestroy()
    {
        // 自动取消所有订阅，防止内存泄漏
        foreach (var unsubscribe in _unsubscribeActions)
            unsubscribe();
        
        _unsubscribeActions.Clear();
    }
}

// 使用示例
public class QuestSystem : EventSubscriber
{
    private void OnEnable()
    {
        Subscribe<EnemyDiedEvent>(OnEnemyDied);
        Subscribe<ItemPickedUpEvent>(OnItemPickedUp);
        Subscribe<PlayerLevelUpEvent>(OnPlayerLevelUp);
    }
    
    private void OnEnemyDied(EnemyDiedEvent evt)
    {
        // 更新"击杀X个敌人"类型的任务
        UpdateKillQuests(evt.EnemyType, evt.EnemyId);
    }
    
    private void OnItemPickedUp(ItemPickedUpEvent evt)
    {
        // 更新"收集X个物品"类型的任务
        UpdateCollectQuests(evt.ItemId, evt.Count);
    }
    
    private void OnPlayerLevelUp(PlayerLevelUpEvent evt)
    {
        // 检查并解锁新任务
        UnlockQuestsForLevel(evt.NewLevel);
    }
}
```

## 三、ScriptableObject 事件（Unity 推荐方案）

```csharp
// GameEvent：基于 ScriptableObject 的事件（可在 Inspector 配置）
[CreateAssetMenu(menuName = "Events/Game Event")]
public class GameEvent : ScriptableObject
{
    private List<GameEventListener> _listeners = new();
    
    public void Raise()
    {
        // 倒序遍历（允许在回调中反注册）
        for (int i = _listeners.Count - 1; i >= 0; i--)
        {
            _listeners[i].OnEventRaised();
        }
    }
    
    public void RegisterListener(GameEventListener listener)
        => _listeners.Add(listener);
    
    public void UnregisterListener(GameEventListener listener)
        => _listeners.Remove(listener);
}

// 带参数的泛型事件
[CreateAssetMenu(menuName = "Events/Int Event")]
public class IntGameEvent : GameEvent<int> { }

public abstract class GameEvent<T> : ScriptableObject
{
    private List<GameEventListener<T>> _listeners = new();
    
    public void Raise(T value)
    {
        for (int i = _listeners.Count - 1; i >= 0; i--)
            _listeners[i].OnEventRaised(value);
    }
}

// 监听组件（挂在 GameObject 上）
public class GameEventListener : MonoBehaviour
{
    [SerializeField] private GameEvent _event;
    [SerializeField] private UnityEvent _response; // 在 Inspector 中配置响应
    
    private void OnEnable() => _event.RegisterListener(this);
    private void OnDisable() => _event.UnregisterListener(this);
    
    public void OnEventRaised() => _response?.Invoke();
}
```

## 四、延迟事件与事件队列

```csharp
// 延迟事件系统：避免在帧中途修改游戏状态
public class DeferredEventQueue
{
    private Queue<Action> _pendingEvents = new();
    private bool _isProcessing = false;
    
    // 将事件加入队列（而非立即执行）
    public void Enqueue<T>(T eventData) where T : struct
    {
        _pendingEvents.Enqueue(() => EventBus.Publish(eventData));
    }
    
    // 在帧末处理所有待发事件（FixedUpdate 之后，LateUpdate 之前）
    public void ProcessPendingEvents()
    {
        if (_isProcessing) return; // 防止递归
        
        _isProcessing = true;
        
        while (_pendingEvents.Count > 0)
        {
            var action = _pendingEvents.Dequeue();
            action();
        }
        
        _isProcessing = false;
    }
}

// 带优先级的事件系统
public class PriorityEventBus
{
    private struct PriorityHandler
    {
        public int Priority;  // 数字越小越先执行
        public Delegate Handler;
    }
    
    private Dictionary<Type, List<PriorityHandler>> _handlers = new();
    
    public void Subscribe<T>(Action<T> handler, int priority = 0) where T : struct
    {
        var type = typeof(T);
        if (!_handlers.ContainsKey(type))
            _handlers[type] = new List<PriorityHandler>();
        
        _handlers[type].Add(new PriorityHandler { Priority = priority, Handler = handler });
        
        // 按优先级排序
        _handlers[type].Sort((a, b) => a.Priority.CompareTo(b.Priority));
    }
    
    public void Publish<T>(T eventData) where T : struct
    {
        var type = typeof(T);
        if (!_handlers.TryGetValue(type, out var list)) return;
        
        foreach (var item in list)
            ((Action<T>)item.Handler).Invoke(eventData);
    }
}
```

```csharp
public struct InterruptibleEventContext<T>
{

public void OnPlayerTakeDamage(ref InterruptibleEventContext<DamageEvent> ctx)
{
    // 护盾系统订阅（优先级高）

    {

        ctx.IsHandled = true; // 中断：伤害被护盾吸收，后续不扣血
    }
}
```

```csharp
// 事件调试：记录所有事件（开发期排查问题）
public static class EventDebugger
{
#if UNITY_EDITOR || DEVELOPMENT_BUILD
    public static readonly List<EventRecord> EventHistory = new();
    public static bool IsRecording = false;
    
    public struct EventRecord
    {
        public string EventType;
        public string SubscriberCount;
        public float Timestamp;
        public string StackTrace;
    }
    
    public static void Record<T>(T eventData, int subscriberCount)
    {
        if (!IsRecording) return;
        
        EventHistory.Add(new EventRecord
        {
            EventType = typeof(T).Name,
            SubscriberCount = subscriberCount.ToString(),
            Timestamp = Time.realtimeSinceStartup,
            StackTrace = Environment.StackTrace
        });
        
        // 最多保留最近 500 条
        if (EventHistory.Count > 500)
            EventHistory.RemoveAt(0);
    }
#endif
}
```

## 七、选型对比

| 方案 | 适用场景 | 优点 | 缺点 |
|------|----------|------|------|
| C# 委托/事件 | 同模块内通信 | 最简单，类型安全 | 无法跨模块/场景 |
| 泛型 EventBus | 跨模块通信 | 灵活，类型安全 | 需手动管理生命周期 |
| ScriptableObject 事件 | Unity 项目，设计师配置 | 可视化，易调试 | 只适合 Unity |
| 消息队列（延迟） | 帧安全的状态修改 | 避免帧内副作用 | 延迟一帧响应 |

> 💡 **核心建议**：对于中大型项目，强烈推荐使用事件总线。它是"高内聚，低耦合"原则的最直接体现。但要注意事件的订阅/取消订阅管理，这是最常见的内存泄漏来源。建立统一的基类（EventSubscriber）自动管理生命周期是最佳实践。
