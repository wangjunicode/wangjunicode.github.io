## 一、输入缓冲系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 输入缓冲队列（存储最近一段时间内的输入，用于连招检测）
/// </summary>
public class InputBuffer
{
    private struct BufferedInput
    {
        public InputAction Action;
        public float Timestamp;
        public Vector2 Direction;  // 方向输入（格斗游戏用）
    }

    [SerializeField] private float bufferWindow = 0.5f;  // 缓冲窗口（秒）
    [SerializeField] private int maxBufferSize = 10;
    
    private List<BufferedInput> buffer = new List<BufferedInput>();

    public InputBuffer(float bufferWindow = 0.5f, int maxSize = 10)
    {
        this.bufferWindow = bufferWindow;
        maxBufferSize = maxSize;
    }

    /// <summary>
    /// 记录输入
    /// </summary>
    public void RecordInput(InputAction action, Vector2 direction = default)
    {
        buffer.Add(new BufferedInput
        {
            Action = action,
            Timestamp = Time.time,
            Direction = direction
        });
        
        // 限制缓冲区大小
        if (buffer.Count > maxBufferSize)
            buffer.RemoveAt(0);
    }

    /// <summary>
    /// 检查缓冲区中是否存在特定输入序列（连招检测）
    /// </summary>
    public bool ConsumeSequence(InputAction[] sequence, float maxInterval = 0.5f)
    {
        if (sequence == null || sequence.Length == 0) return false;
        
        // 清除过期输入
        CleanExpired();
        
        if (buffer.Count < sequence.Length) return false;
        
        // 从缓冲区末尾向前匹配
        int matchIndex = sequence.Length - 1;
        float lastTime = float.MaxValue;
        
        for (int i = buffer.Count - 1; i >= 0 && matchIndex >= 0; i--)
        {
            var input = buffer[i];
            
            if (input.Action == sequence[matchIndex])
            {
                if (lastTime - input.Timestamp <= maxInterval)
                {
                    lastTime = input.Timestamp;
                    matchIndex--;
                    
                    if (matchIndex < 0)
                    {
                        // 匹配成功，消费这些输入
                        ConsumeMatched(i, sequence.Length);
                        return true;
                    }
                }
                else
                {
                    // 时间间隔过长，重新匹配
                    matchIndex = sequence.Length - 1;
                    lastTime = float.MaxValue;
                }
            }
        }
        
        return false;
    }

    /// <summary>
    /// 检查最近是否有特定单个输入
    /// </summary>
    public bool ConsumeInput(InputAction action)
    {
        CleanExpired();
        
        for (int i = buffer.Count - 1; i >= 0; i--)
        {
            if (buffer[i].Action == action)
            {
                buffer.RemoveAt(i);
                return true;
            }
        }
        return false;
    }

    void CleanExpired()
    {
        float cutoff = Time.time - bufferWindow;
        buffer.RemoveAll(b => b.Timestamp < cutoff);
    }

    void ConsumeMatched(int startIndex, int count)
    {
        int end = Mathf.Min(startIndex + count, buffer.Count);
        buffer.RemoveRange(startIndex, end - startIndex);
    }

    public void Clear() => buffer.Clear();
}

public enum InputAction { LightAttack, HeavyAttack, Dodge, Block, Skill1, Skill2, Skill3, Interact }
```

## 二、连招状态机

```csharp
/// <summary>
/// 连招配置（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "Combo", menuName = "Game/Combo")]
public class ComboData : ScriptableObject
{
    [Serializable]
    public class ComboNode
    {
        public string AnimationName;
        public float Duration;            // 动画持续时间
        public float CancelableAt;        // 何时可以接下一招（归一化0-1）
        public InputAction[] NextInputs;  // 可以接的下一个输入
        public ComboNode[] NextNodes;     // 对应的下一个节点
        public float DamageMultiplier = 1f;
        public bool IsFinisher;           // 终结技
    }

    public ComboNode RootNode;           // 起始节点
    public float ResetTime = 1.5f;      // 无输入多久后重置连招
}

/// <summary>
/// 连招执行器
/// </summary>
public class ComboExecutor : MonoBehaviour
{
    [SerializeField] private ComboData comboData;
    [SerializeField] private Animator animator;
    
    private InputBuffer inputBuffer;
    private ComboData.ComboNode currentNode;
    private float comboTimer;
    private float cancelableTimer;
    
    // 连击数
    private int comboCount;
    public int ComboCount => comboCount;
    
    public event Action<int> OnComboCountChanged;
    public event Action OnComboReset;

    void Awake()
    {
        inputBuffer = new InputBuffer(bufferWindow: 0.3f);
    }

    void Update()
    {
        // 更新计时器
        if (currentNode != null)
        {
            comboTimer += Time.deltaTime;
            cancelableTimer = comboTimer / currentNode.Duration;
            
            // 检查是否可以接下一招
            if (cancelableTimer >= currentNode.CancelableAt)
            {
                CheckContinueCombo();
            }
            
            // 连招超时重置
            if (comboTimer >= currentNode.Duration + comboData.ResetTime)
            {
                ResetCombo();
            }
        }
    }

    /// <summary>
    /// 处理攻击输入
    /// </summary>
    public void HandleAttackInput(InputAction input)
    {
        inputBuffer.RecordInput(input);
        
        if (currentNode == null)
        {
            // 开始新连招
            StartCombo(input);
        }
    }

    void StartCombo(InputAction input)
    {
        var root = comboData.RootNode;
        if (root == null) return;
        
        // 检查起始输入是否匹配
        for (int i = 0; i < root.NextInputs.Length; i++)
        {
            if (root.NextInputs[i] == input)
            {
                ExecuteNode(root.NextNodes[i]);
                break;
            }
        }
    }

    void CheckContinueCombo()
    {
        if (currentNode?.NextInputs == null) return;
        
        for (int i = 0; i < currentNode.NextInputs.Length; i++)
        {
            if (inputBuffer.ConsumeInput(currentNode.NextInputs[i]))
            {
                ExecuteNode(currentNode.NextNodes[i]);
                return;
            }
        }
    }

    void ExecuteNode(ComboData.ComboNode node)
    {
        currentNode = node;
        comboTimer = 0;
        cancelableTimer = 0;
        
        comboCount++;
        OnComboCountChanged?.Invoke(comboCount);
        
        // 播放动画
        animator?.CrossFade(node.AnimationName, 0.1f);
        
        if (node.IsFinisher)
            StartCoroutine(FinisherSequence(node));
    }

    System.Collections.IEnumerator FinisherSequence(ComboData.ComboNode node)
    {
        yield return new WaitForSeconds(node.Duration);
        ResetCombo();
    }

    void ResetCombo()
    {
        currentNode = null;
        comboTimer = 0;
        inputBuffer.Clear();
        
        if (comboCount > 0)
        {
            comboCount = 0;
            OnComboReset?.Invoke();
        }
    }
}
```

## 三、格斗游戏方向输入

```csharp
/// <summary>
/// 格斗游戏方向输入检测（↓↘→ + 攻击 = 波动拳）
/// </summary>
public class FightingGameInput : MonoBehaviour
{
    private InputBuffer directionBuffer;
    
    // 9键方向（数字小键盘表示法）
    private const int Neutral = 5;
    private const int Down = 2;
    private const int DownForward = 3;
    private const int Forward = 6;
    private const int Back = 4;
    private const int Up = 8;
    
    void Awake()
    {
        directionBuffer = new InputBuffer(bufferWindow: 0.6f, maxSize: 20);
    }

    void Update()
    {
        Vector2 dir = GetDirectionalInput();
        int numpad = VectorToNumpad(dir);
        
        // 检测波动拳输入（↓↘→）
        if (CheckHadoken())
        {
            Debug.Log("[Fighting] Hadoken!");
        }
    }

    bool CheckHadoken()
    {
        // 简化检测：只需要在缓冲窗口内出现 2 3 6 序列
        var seq = new InputAction[] 
        { 
            (InputAction)2, // Down
            (InputAction)3, // Down-Forward
            (InputAction)6  // Forward
        };
        return directionBuffer.ConsumeSequence(seq, 0.4f);
    }

    Vector2 GetDirectionalInput()
    {
        float h = Input.GetAxisRaw("Horizontal");
        float v = Input.GetAxisRaw("Vertical");
        return new Vector2(h, v).normalized;
    }

    int VectorToNumpad(Vector2 dir)
    {
        if (dir.magnitude < 0.3f) return Neutral;
        float angle = Mathf.Atan2(dir.y, dir.x) * Mathf.Rad2Deg;
        if (angle < -157.5f || angle >= 157.5f) return Back;
        if (angle >= -22.5f && angle < 22.5f) return Forward;
        if (angle >= 67.5f && angle < 112.5f) return Up;
        if (angle >= -112.5f && angle < -67.5f) return Down;
        if (angle >= 22.5f && angle < 67.5f) return 9; // Up-Forward
        if (angle >= -67.5f && angle < -22.5f) return DownForward;
        if (angle >= 112.5f && angle < 157.5f) return 7; // Up-Back
        return 1; // Down-Back
    }
}
```

## 四、连招设计原则

| 原则 | 说明 |
|------|------|
| 输入宽容 | 缓冲窗口150-300ms，玩家不需要精确卡帧 |
| 取消窗口 | 攻击动画70-80%时允许接下一招 |
| 视觉反馈 | 连击计数器、屏幕轻微震动 |
| 容错设计 | 错误输入自动跳到最接近的招式，而非失败 |
| 终结技奖励 | 完成完整连招给予额外伤害奖励 |
| 防止刷连 | 连击数有上限或衰减，防止无限连 |
