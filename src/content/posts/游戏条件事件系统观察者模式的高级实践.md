## 一、类型安全泛型事件总线

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 事件基类标记接口
/// </summary>
public interface IGameEvent { }

/// <summary>
/// 高性能泛型事件总线
/// </summary>
public static class EventBus
{
    private static Dictionary<Type, object> handlers = new Dictionary<Type, object>();
    
    // 订阅
    public static void Subscribe<T>(Action<T> handler) where T : IGameEvent
    {
        var type = typeof(T);
        if (!handlers.ContainsKey(type))
            handlers[type] = new List<Action<T>>();
        
        ((List<Action<T>>)handlers[type]).Add(handler);
    }

    // 取消订阅
    public static void Unsubscribe<T>(Action<T> handler) where T : IGameEvent
    {
        var type = typeof(T);
        if (handlers.TryGetValue(type, out var list))
            ((List<Action<T>>)list).Remove(handler);
    }

    // 发布
    public static void Publish<T>(T gameEvent) where T : IGameEvent
    {
        var type = typeof(T);
        if (!handlers.TryGetValue(type, out var list)) return;
        
        var handlerList = (List<Action<T>>)list;
        for (int i = handlerList.Count - 1; i >= 0; i--)
        {
            try { handlerList[i]?.Invoke(gameEvent); }
            catch (Exception e) { Debug.LogError($"EventBus error: {e}"); }
        }
    }

    // 清理（场景切换时调用）
    public static void Clear() => handlers.Clear();
}

// 具体事件定义
public struct PlayerLevelUpEvent : IGameEvent
{
    public string PlayerId;
    public int OldLevel;
    public int NewLevel;
}

public struct EnemyDeathEvent : IGameEvent
{
    public string EnemyId;
    public string KillerId;
    public Vector3 Position;
}

public struct ItemPickedUpEvent : IGameEvent
{
    public string PlayerId;
    public string ItemId;
    public int Count;
}

// 使用示例
public class ExperienceSystem : MonoBehaviour
{
    void OnEnable()
    {
        EventBus.Subscribe<EnemyDeathEvent>(OnEnemyKilled);
    }

    void OnDisable()
    {
        EventBus.Unsubscribe<EnemyDeathEvent>(OnEnemyKilled);
    }

    void OnEnemyKilled(EnemyDeathEvent e)
    {
        Debug.Log($"Player {e.KillerId} killed enemy {e.EnemyId}");
        // 给杀手加经验
    }
}
```

## 二、带优先级的事件系统

```csharp
/// <summary>
/// 带优先级的事件总线（高优先级处理器先触发）
/// </summary>
public static class PriorityEventBus
{
    private class HandlerEntry<T>
    {
        public Action<T> Handler;
        public int Priority;       // 数字越大优先级越高
        public bool CanIntercept;  // 是否可以拦截（阻止后续处理）
    }

    private static Dictionary<Type, object> handlers = new Dictionary<Type, object>();

    public static void Subscribe<T>(Action<T> handler, int priority = 0, 
        bool canIntercept = false) where T : IGameEvent
    {
        var type = typeof(T);
        if (!handlers.ContainsKey(type))
            handlers[type] = new List<HandlerEntry<T>>();
        
        var list = (List<HandlerEntry<T>>)handlers[type];
        list.Add(new HandlerEntry<T> { Handler = handler, Priority = priority, CanIntercept = canIntercept });
        
        // 按优先级排序（降序，高优先级在前）
        list.Sort((a, b) => b.Priority.CompareTo(a.Priority));
    }

    /// <summary>
    /// 发布事件（返回是否被拦截）
    /// </summary>
    public static bool Publish<T>(T gameEvent) where T : IGameEvent
    {
        var type = typeof(T);
        if (!handlers.TryGetValue(type, out var obj)) return false;
        
        var list = (List<HandlerEntry<T>>)obj;
        
        foreach (var entry in list)
        {
            entry.Handler?.Invoke(gameEvent);
            
            // 如果该处理器声明可拦截，且事件对象实现了拦截接口
            if (entry.CanIntercept && gameEvent is IInterceptableEvent ie && ie.IsIntercepted)
                return true; // 被拦截，后续处理器不再执行
        }
        
        return false;
    }
}

public interface IInterceptableEvent : IGameEvent
{
    bool IsIntercepted { get; }
    void Intercept();
}

// 可拦截事件示例（装备使用前可以被玩家特性拦截修改）
public class ItemUseEvent : IInterceptableEvent
{
    public string ItemId;
    public int Cost;
    private bool intercepted;
    
    public bool IsIntercepted => intercepted;
    public void Intercept() => intercepted = true;
}
```

## 三、延迟事件系统

```csharp
/// <summary>
/// 延迟事件管理器（支持帧延迟和时间延迟）
/// </summary>
public class DelayedEventManager : MonoBehaviour
{
    private static DelayedEventManager instance;
    public static DelayedEventManager Instance => instance;

    private struct DelayedEvent
    {
        public Action Action;
        public float ExecuteTime;       // 按 Time.time 计算
        public int ExecuteFrame;        // 按帧计算（0表示不用）
    }

    private List<DelayedEvent> pendingEvents = new List<DelayedEvent>();

    void Awake() { instance = this; }

    /// <summary>
    /// 在 N 秒后触发事件
    /// </summary>
    public void PublishAfterDelay<T>(T gameEvent, float delaySeconds) where T : IGameEvent
    {
        pendingEvents.Add(new DelayedEvent
        {
            Action = () => EventBus.Publish(gameEvent),
            ExecuteTime = Time.time + delaySeconds
        });
    }

    /// <summary>
    /// 在下 N 帧后触发事件
    /// </summary>
    public void PublishAfterFrames<T>(T gameEvent, int frames) where T : IGameEvent
    {
        pendingEvents.Add(new DelayedEvent
        {
            Action = () => EventBus.Publish(gameEvent),
            ExecuteFrame = Time.frameCount + frames,
            ExecuteTime = float.MaxValue
        });
    }

    void Update()
    {
        float now = Time.time;
        int frame = Time.frameCount;
        
        for (int i = pendingEvents.Count - 1; i >= 0; i--)
        {
            var e = pendingEvents[i];
            bool shouldFire = e.ExecuteFrame > 0 ? frame >= e.ExecuteFrame : now >= e.ExecuteTime;
            
            if (shouldFire)
            {
                e.Action?.Invoke();
                pendingEvents.RemoveAt(i);
            }
        }
    }
}
```

## 四、事件调试工具

```csharp
#if UNITY_EDITOR || DEBUG
/// <summary>
/// 事件日志（调试用，记录最近N个事件）
/// </summary>
public static class EventLogger
{
    private static Queue<string> log = new Queue<string>();
    private const int MAX_LOG = 100;
    
    public static void Log(string eventName, object data)
    {
        string entry = $"[{Time.frameCount}] {eventName}: {data}";
        log.Enqueue(entry);
        if (log.Count > MAX_LOG) log.Dequeue();
        Debug.Log($"[EventBus] {entry}");
    }
    
    public static string GetRecentLogs(int count = 20)
    {
        var recent = new System.Text.StringBuilder();
        var arr = log.ToArray();
        int start = Mathf.Max(0, arr.Length - count);
        for (int i = start; i < arr.Length; i++)
            recent.AppendLine(arr[i]);
        return recent.ToString();
    }
}
#endif
```

## 五、事件系统对比

| 方案 | GC分配 | 类型安全 | 解耦程度 | 适用场景 |
|------|--------|----------|----------|----------|
| C# Action/event | 低 | 高 | 中（直接引用）| 组件内部 |
| UnityEvent | 中（Inspector序列化）| 中 | 高 | Editor配置的事件 |
| 本文泛型EventBus | 低（Struct事件零GC）| 高 | 极高 | 跨模块通信 |
| SendMessage | 高（字符串反射）| 低 | 高 | 不推荐（性能差）|

**最佳实践：**
- 使用 struct 而非 class 定义事件（减少GC）
- 及时 Unsubscribe 防止内存泄漏（OnDisable 中取消订阅）
- 不要在事件处理中再发布相同事件（循环事件）
