## 一、装备系统的核心要素

| 要素 | 作用 | 典型游戏案例 |
|------|------|-------------|
| 基础属性 | 攻击力、防御力等基础数值 | 所有RPG |
| 词缀属性 | 随机附加的额外属性 | 暗黑破坏神、POE |
| 品质/稀有度 | 影响词缀数量和数值范围 | 所有ARG/ARPG |
| 套装系统 | 同套多件获得额外加成 | 暗黑3、命运2 |
| 强化/附魔 | 通过材料强化装备 | MMORPG |
| 词条重铸 | 重新随机词缀 | POE |

## 二、装备数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 装备稀有度
/// </summary>
public enum ItemRarity
{
    Common,     // 普通（白）
    Magic,      // 魔法（蓝，1-2个词缀）
    Rare,       // 稀有（黄，3-6个词缀）
    Unique,     // 独特（橙，固定词缀）
    Set,        // 套装（绿）
    Legendary   // 传说（金）
}

/// <summary>
/// 装备部位
/// </summary>
public enum EquipmentSlot
{
    Weapon, Shield, Helmet, Chest, Gloves, Belt, Boots,
    Ring1, Ring2, Amulet, Offhand
}

/// <summary>
/// 单条属性词缀
/// </summary>
[Serializable]
public class ItemAffix
{
    public string AffixId;         // 词缀ID（用于查表）
    public string DisplayName;     // 展示名称（"火焰伤害"）
    public StatType StatType;      // 影响的属性类型
    public ModifyType ModifyType;  // 修改方式（固定值/百分比）
    public float MinValue;         // 最小值
    public float MaxValue;         // 最大值
    public float RolledValue;      // 实际随机到的值
    public AffixPosition Position; // 前缀/后缀
    
    public float GetValue() => RolledValue;
    
    public string GetDescription()
    {
        return ModifyType switch
        {
            ModifyType.Flat => $"{DisplayName} +{RolledValue:F0}",
            ModifyType.Percent => $"{DisplayName} +{RolledValue:F1}%",
            ModifyType.PercentMore => $"{DisplayName} 增加 {RolledValue:F0}%",
            _ => $"{DisplayName}: {RolledValue}"
        };
    }
}

public enum StatType
{
    Damage, Defense, MaxHp, MaxMp, AttackSpeed, CritChance, CritDamage,
    FireDamage, ColdDamage, LightningDamage, PoisonDamage,
    FireResistance, ColdResistance, LightningResistance,
    MovementSpeed, CooldownReduction, ManaRegen, LifeOnHit
}

public enum ModifyType { Flat, Percent, PercentMore }
public enum AffixPosition { Prefix, Suffix }

/// <summary>
/// 装备实例（运行时对象）
/// </summary>
[Serializable]
public class EquipmentItem
{
    public string ItemId;           // 实例唯一ID
    public string BaseItemId;       // 对应的基础装备模板
    public string Name;
    public ItemRarity Rarity;
    public EquipmentSlot Slot;
    
    // 基础属性（来自模板）
    public List<ItemAffix> ImplicitAffixes = new List<ItemAffix>(); // 隐式词缀（固定）
    
    // 词缀属性（随机生成）
    public List<ItemAffix> Prefixes = new List<ItemAffix>();  // 前缀（最多3个）
    public List<ItemAffix> Suffixes = new List<ItemAffix>();  // 后缀（最多3个）
    
    // 强化
    public int EnhancementLevel;     // 强化等级（0-20）
    public float EnhancementBonus;   // 强化带来的额外加成
    
    // 外观
    public Sprite Icon;
    public GameObject ModelPrefab;
    
    // 装备分数（综合评估）
    public float Score => CalculateScore();
    
    public IEnumerable<ItemAffix> AllAffixes
    {
        get
        {
            foreach (var a in ImplicitAffixes) yield return a;
            foreach (var a in Prefixes) yield return a;
            foreach (var a in Suffixes) yield return a;
        }
    }

    float CalculateScore()
    {
        float score = 0;
        foreach (var affix in AllAffixes)
        {
            // 根据词缀类型赋予不同权重
            float weight = GetAffixWeight(affix.StatType);
            score += affix.GetValue() * weight;
        }
        return score * (1f + EnhancementLevel * 0.05f);
    }

    float GetAffixWeight(StatType type) => type switch
    {
        StatType.Damage     => 1.5f,
        StatType.CritChance => 3f,
        StatType.CritDamage => 2f,
        StatType.MaxHp      => 0.5f,
        _ => 1f
    };
}

/// <summary>
/// 基础装备模板（ScriptableObject）
/// </summary>
[CreateAssetMenu(menuName = "Game/Items/EquipmentTemplate")]
public class EquipmentTemplate : ScriptableObject
{
    [Header("基础信息")]
    public string BaseItemId;
    public string BaseName;
    public EquipmentSlot Slot;
    public Sprite Icon;
    public GameObject ModelPrefab;
    
    [Header("等级要求")]
    public int RequiredLevel;
    public int ItemLevel;         // 决定可以出现的词缀等级
    
    [Header("隐式词缀（固定）")]
    public List<AffixDefinition> ImplicitAffixes;
    
    [Header("词缀池（用于随机词缀）")]
    public List<AffixDefinition> AffixPool;
    
    [Serializable]
    public class AffixDefinition
    {
        public string AffixId;
        public string DisplayName;
        public StatType StatType;
        public ModifyType ModifyType;
        public float MinValue;
        public float MaxValue;
        public AffixPosition Position;
        public float Weight;         // 出现权重
        public int RequiredItemLevel; // 需要的装备等级
    }
}
```

## 三、装备生成器

```csharp
/// <summary>
/// 随机装备生成器
/// </summary>
public class EquipmentGenerator
{
    private System.Random rng;

    public EquipmentGenerator(int seed = 0)
    {
        rng = new System.Random(seed == 0 ? System.Environment.TickCount : seed);
    }

    /// <summary>
    /// 生成一件随机装备
    /// </summary>
    public EquipmentItem Generate(EquipmentTemplate template, ItemRarity rarity, 
        int dropLevel = 0)
    {
        var item = new EquipmentItem
        {
            ItemId = Guid.NewGuid().ToString("N"),
            BaseItemId = template.BaseItemId,
            Name = GenerateName(template, rarity),
            Rarity = rarity,
            Slot = template.Slot
        };
        
        // 生成隐式词缀
        foreach (var affixDef in template.ImplicitAffixes)
        {
            item.ImplicitAffixes.Add(RollAffix(affixDef));
        }
        
        // 根据品质决定词缀数量
        (int minPre, int maxPre, int minSuf, int maxSuf) = GetAffixCounts(rarity);
        int prefixCount = rng.Next(minPre, maxPre + 1);
        int suffixCount = rng.Next(minSuf, maxSuf + 1);
        
        // 从词缀池中随机选取
        var availablePrefixes = template.AffixPool.FindAll(a => 
            a.Position == AffixPosition.Prefix && 
            a.RequiredItemLevel <= (dropLevel > 0 ? dropLevel : template.ItemLevel));
        
        var availableSuffixes = template.AffixPool.FindAll(a => 
            a.Position == AffixPosition.Suffix && 
            a.RequiredItemLevel <= (dropLevel > 0 ? dropLevel : template.ItemLevel));
        
        var selectedPrefixes = WeightedSample(availablePrefixes, prefixCount);
        var selectedSuffixes = WeightedSample(availableSuffixes, suffixCount);
        
        foreach (var p in selectedPrefixes)
            item.Prefixes.Add(RollAffix(p));
        
        foreach (var s in selectedSuffixes)
            item.Suffixes.Add(RollAffix(s));
        
        return item;
    }

    ItemAffix RollAffix(EquipmentTemplate.AffixDefinition def)
    {
        float value = (float)(def.MinValue + (def.MaxValue - def.MinValue) * rng.NextDouble());
        
        return new ItemAffix
        {
            AffixId = def.AffixId,
            DisplayName = def.DisplayName,
            StatType = def.StatType,
            ModifyType = def.ModifyType,
            MinValue = def.MinValue,
            MaxValue = def.MaxValue,
            RolledValue = value,
            Position = def.Position
        };
    }

    (int, int, int, int) GetAffixCounts(ItemRarity rarity) => rarity switch
    {
        ItemRarity.Common  => (0, 0, 0, 0),
        ItemRarity.Magic   => (0, 1, 0, 1),
        ItemRarity.Rare    => (1, 3, 1, 3),
        ItemRarity.Unique  => (0, 0, 0, 0), // Unique 词缀固定
        _                  => (1, 3, 1, 3)
    };

    List<EquipmentTemplate.AffixDefinition> WeightedSample(
        List<EquipmentTemplate.AffixDefinition> pool, int count)
    {
        var result = new List<EquipmentTemplate.AffixDefinition>();
        var remaining = new List<EquipmentTemplate.AffixDefinition>(pool);
        
        for (int i = 0; i < count && remaining.Count > 0; i++)
        {
            float totalWeight = 0;
            foreach (var a in remaining) totalWeight += a.Weight;
            
            float roll = (float)(rng.NextDouble() * totalWeight);
            float accumulated = 0;
            
            foreach (var a in remaining)
            {
                accumulated += a.Weight;
                if (roll <= accumulated)
                {
                    result.Add(a);
                    remaining.Remove(a);
                    break;
                }
            }
        }
        
        return result;
    }

    string GenerateName(EquipmentTemplate template, ItemRarity rarity)
    {
        return rarity switch
        {
            ItemRarity.Common  => template.BaseName,
            ItemRarity.Magic   => $"{GetRandomPrefix()} {template.BaseName}",
            ItemRarity.Rare    => GetRandomRareName(),
            ItemRarity.Unique  => template.BaseName, // Unique 有固定名字
            _ => template.BaseName
        };
    }

    string GetRandomPrefix()
    {
        string[] prefixes = { "狂暴的", "锐利的", "冰冷的", "烈焰的", "暗影的" };
        return prefixes[rng.Next(0, prefixes.Length)];
    }

    string GetRandomRareName()
    {
        string[] names = { "风暴之怒", "深渊裂痕", "龙魂之威", "暗夜荆棘", "雷霆主宰" };
        return names[rng.Next(0, names.Length)];
    }
}
```

## 四、装备合成/锻造系统

```csharp
/// <summary>
/// 装备锻造系统
/// </summary>
public class ForgingSystem
{
    /// <summary>
    /// 强化装备（消耗材料提升强化等级）
    /// </summary>
    public EnhanceResult Enhance(EquipmentItem item, PlayerInventory inventory)
    {
        var cost = GetEnhanceCost(item);
        
        if (!inventory.HasMaterials(cost))
            return new EnhanceResult { Success = false, Message = "材料不足" };
        
        if (item.EnhancementLevel >= 20)
            return new EnhanceResult { Success = false, Message = "已达最大强化等级" };
        
        inventory.ConsumeMaterials(cost);
        
        // 高强化等级有失败概率
        float successRate = GetEnhanceSuccessRate(item.EnhancementLevel);
        
        if (UnityEngine.Random.value <= successRate)
        {
            item.EnhancementLevel++;
            item.EnhancementBonus += GetEnhanceBonus(item.EnhancementLevel);
            return new EnhanceResult
            {
                Success = true,
                NewLevel = item.EnhancementLevel,
                Message = $"强化成功！+{item.EnhancementLevel}"
            };
        }
        else
        {
            // 失败：高等级装备可能降级
            bool downgrade = item.EnhancementLevel >= 15 && 
                UnityEngine.Random.value < 0.3f;
            
            if (downgrade)
            {
                item.EnhancementLevel = Mathf.Max(0, item.EnhancementLevel - 1);
                return new EnhanceResult
                {
                    Success = false,
                    NewLevel = item.EnhancementLevel,
                    Message = "强化失败，等级降低！"
                };
            }
            
            return new EnhanceResult { Success = false, Message = "强化失败！" };
        }
    }

    float GetEnhanceSuccessRate(int currentLevel) => currentLevel switch
    {
        <= 7  => 1.0f,
        <= 10 => 0.9f,
        <= 12 => 0.7f,
        <= 15 => 0.5f,
        <= 18 => 0.3f,
        _     => 0.2f
    };

    float GetEnhanceBonus(int level) => level * 0.05f; // 每级+5%属性

    Dictionary<string, int> GetEnhanceCost(EquipmentItem item)
    {
        int multiplier = item.EnhancementLevel + 1;
        return new Dictionary<string, int>
        {
            ["iron_ore"] = 5 * multiplier,
            ["magic_crystal"] = 1 * multiplier
        };
    }

    /// <summary>
    /// 词条重铸（随机重新生成所有非固定词缀）
    /// </summary>
    public EquipmentItem Reforge(EquipmentItem item, EquipmentTemplate template,
        PlayerInventory inventory, int seed = 0)
    {
        if (!inventory.HasMaterials(new Dictionary<string, int> { ["chaos_orb"] = 1 }))
            return null;
        
        inventory.ConsumeMaterials(new Dictionary<string, int> { ["chaos_orb"] = 1 });
        
        var gen = new EquipmentGenerator(seed);
        var newItem = gen.Generate(template, item.Rarity);
        
        // 保留强化等级
        newItem.EnhancementLevel = item.EnhancementLevel;
        newItem.EnhancementBonus = item.EnhancementBonus;
        newItem.ItemId = item.ItemId; // 保持同一实例ID
        
        return newItem;
    }
}

public class EnhanceResult
{
    public bool Success;
    public int NewLevel;
    public string Message;
}

// 简化接口
public abstract class PlayerInventory
{
    public abstract bool HasMaterials(Dictionary<string, int> cost);
    public abstract void ConsumeMaterials(Dictionary<string, int> cost);
}
```

## 五、装备属性计算系统

```csharp
/// <summary>
/// 角色最终属性计算（考虑所有装备加成）
/// </summary>
public class CharacterStats
{
    // 基础属性（来自角色等级）
    public float BaseAttack = 100f;
    public float BaseDefense = 50f;
    public float BaseMaxHp = 1000f;
    
    // 计算最终属性（装备加成后）
    public FinalStats Calculate(List<EquipmentItem> equippedItems)
    {
        float flatAttack = 0, percentAttack = 0, moreAttack = 1f;
        float flatDefense = 0, percentDefense = 0;
        float flatHp = 0, percentHp = 0;
        float critChance = 5f; // 基础5%暴击
        float critDamage = 150f; // 基础150%暴击伤害
        
        foreach (var item in equippedItems)
        foreach (var affix in item.AllAffixes)
        {
            switch (affix.StatType)
            {
                case StatType.Damage:
                    if (affix.ModifyType == ModifyType.Flat)
                        flatAttack += affix.GetValue();
                    else if (affix.ModifyType == ModifyType.Percent)
                        percentAttack += affix.GetValue();
                    else if (affix.ModifyType == ModifyType.PercentMore)
                        moreAttack *= (1 + affix.GetValue() / 100f);
                    break;
                case StatType.Defense:
                    if (affix.ModifyType == ModifyType.Flat)
                        flatDefense += affix.GetValue();
                    else
                        percentDefense += affix.GetValue();
                    break;
                case StatType.MaxHp:
                    if (affix.ModifyType == ModifyType.Flat)
                        flatHp += affix.GetValue();
                    else
                        percentHp += affix.GetValue();
                    break;
                case StatType.CritChance:
                    critChance += affix.GetValue();
                    break;
                case StatType.CritDamage:
                    critDamage += affix.GetValue();
                    break;
            }
        }
        
        // POE 风格计算：增加(additive) + 增倍(multiplicative)
        float finalAttack = (BaseAttack + flatAttack) * 
            (1f + percentAttack / 100f) * moreAttack;
        
        float finalDefense = (BaseDefense + flatDefense) * 
            (1f + percentDefense / 100f);
        
        float finalHp = (BaseMaxHp + flatHp) * (1f + percentHp / 100f);
        
        return new FinalStats
        {
            Attack = finalAttack,
            Defense = finalDefense,
            MaxHp = finalHp,
            CritChance = Mathf.Min(critChance, 100f), // 暴击上限100%
            CritDamage = critDamage
        };
    }
}

public class FinalStats
{
    public float Attack;
    public float Defense;
    public float MaxHp;
    public float CritChance;
    public float CritDamage;
    
    public float DPS => Attack * (1f + CritChance / 100f * (CritDamage / 100f - 1f));
    
    public override string ToString() =>
        $"攻击:{Attack:F0} 防御:{Defense:F0} 生命:{MaxHp:F0} 暴击:{CritChance:F1}% 爆伤:{CritDamage:F0}%\nDPS: {DPS:F0}";
}
```

## 六、设计要点

| 系统 | 核心设计 | 玩家体验 |
|------|----------|----------|
| 词缀随机 | 加权随机，高稀有度出现更好词缀 | 开箱的惊喜感 |
| 强化系统 | 低等级必成，高等级有失败率 | 成长感 + 赌博刺激感 |
| 词条重铸 | 消耗货币重置词缀 | 追求完美的驱动力 |
| 装备分数 | 综合评估词缀价值 | 清晰的升级对比 |
| 属性叠加 | 加法+乘法分层计算 | 策略深度（POE风格） |
