## 一、词条数据配置

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

public enum AffixTier { Common, Magic, Rare, Legendary }
public enum AffixType { Prefix, Suffix }  // 前缀/后缀分离

[CreateAssetMenu(menuName = "Game/Affix")]
public class AffixData : ScriptableObject
{
    public string AffixId;
    public string Name;
    public AffixType Type;
    public AffixTier Tier;
    public string StatId;           // 影响的属性ID
    public float MinValue;
    public float MaxValue;
    public int Weight;              // 出现权重（越大越常见）
    public string[] ExclusiveGroup; // 互斥词条组（同组只能出现一个）
    
    public float RollValue(System.Random rng = null)
    {
        rng ??= new System.Random();
        return MinValue + (float)rng.NextDouble() * (MaxValue - MinValue);
    }
}

[Serializable]
public class RolledAffix
{
    public string AffixId;
    public float Value;
    public bool IsLocked;  // 锁定（重铸时保留）
}
```

## 二、词条生成与重铸

```csharp
public class AffixSystem : MonoBehaviour
{
    [SerializeField] private AffixData[] affixPool;
    
    private System.Random rng = new System.Random();
    private Dictionary<string, AffixData> affixMap;

    void Awake()
    {
        affixMap = new Dictionary<string, AffixData>();
        foreach (var affix in affixPool)
            affixMap[affix.AffixId] = affix;
    }

    /// <summary>
    /// 为装备随机生成词条（根据稀有度决定词条数量）
    /// </summary>
    public List<RolledAffix> GenerateAffixes(EquipmentRarity rarity)
    {
        int count = rarity switch
        {
            EquipmentRarity.Common    => 1,
            EquipmentRarity.Uncommon  => 2,
            EquipmentRarity.Rare      => 3,
            EquipmentRarity.Epic      => 4,
            EquipmentRarity.Legendary => 5,
            _ => 1
        };
        
        var result = new List<RolledAffix>();
        var usedGroups = new HashSet<string>();
        int maxAttempts = count * 10;
        int attempts = 0;
        
        while (result.Count < count && attempts < maxAttempts)
        {
            attempts++;
            var affix = WeightedRandom(affixPool);
            if (affix == null) continue;
            
            // 检查互斥组
            bool conflicted = false;
            if (affix.ExclusiveGroup != null)
                foreach (var group in affix.ExclusiveGroup)
                    if (usedGroups.Contains(group)) { conflicted = true; break; }
            
            if (conflicted) continue;
            
            // 添加词条
            if (affix.ExclusiveGroup != null)
                foreach (var group in affix.ExclusiveGroup)
                    usedGroups.Add(group);
            
            result.Add(new RolledAffix
            {
                AffixId = affix.AffixId,
                Value = affix.RollValue(rng)
            });
        }
        
        return result;
    }

    /// <summary>
    /// 重铸（重新随机未锁定的词条）
    /// </summary>
    public List<RolledAffix> Reforge(List<RolledAffix> currentAffixes)
    {
        var lockedAffixes = currentAffixes.FindAll(a => a.IsLocked);
        int reforgeCount = currentAffixes.Count - lockedAffixes.Count;
        
        if (reforgeCount <= 0) return currentAffixes;
        
        var newAffixes = new List<RolledAffix>(lockedAffixes);
        var usedGroups = new HashSet<string>();
        
        // 先统计锁定词条的互斥组
        foreach (var locked in lockedAffixes)
        {
            if (affixMap.TryGetValue(locked.AffixId, out var data) && data.ExclusiveGroup != null)
                foreach (var group in data.ExclusiveGroup)
                    usedGroups.Add(group);
        }
        
        // 随机新词条填充
        var newRolled = GenerateAffixesExcluding(reforgeCount, usedGroups);
        newAffixes.AddRange(newRolled);
        
        return newAffixes;
    }

    List<RolledAffix> GenerateAffixesExcluding(int count, HashSet<string> excludeGroups)
    {
        var result = new List<RolledAffix>();
        var usedGroups = new HashSet<string>(excludeGroups);
        
        int attempts = 0;
        while (result.Count < count && attempts < count * 10)
        {
            attempts++;
            var affix = WeightedRandom(affixPool);
            if (affix == null) continue;
            
            bool ok = true;
            if (affix.ExclusiveGroup != null)
                foreach (var g in affix.ExclusiveGroup)
                    if (usedGroups.Contains(g)) { ok = false; break; }
            
            if (!ok) continue;
            
            if (affix.ExclusiveGroup != null)
                foreach (var g in affix.ExclusiveGroup) usedGroups.Add(g);
            
            result.Add(new RolledAffix
            {
                AffixId = affix.AffixId,
                Value = affix.RollValue(rng)
            });
        }
        
        return result;
    }

    AffixData WeightedRandom(AffixData[] pool)
    {
        int totalWeight = 0;
        foreach (var a in pool) totalWeight += a.Weight;
        
        int roll = rng.Next(totalWeight);
        int cumulative = 0;
        
        foreach (var a in pool)
        {
            cumulative += a.Weight;
            if (roll < cumulative) return a;
        }
        return null;
    }
}

public enum EquipmentRarity { Common, Uncommon, Rare, Epic, Legendary }
```

## 三、词条系统设计规范

| Tier | 颜色 | 词条数 | 特点 |
|------|------|--------|------|
| 普通 | 白 | 1 | 简单属性加成 |
| 精良 | 绿 | 2 | 含1个特殊词条 |
| 稀有 | 蓝 | 3 | 强力词条组合 |
| 史诗 | 紫 | 4 | 特殊机制词条 |
| 传奇 | 橙 | 5 | 独一无二词条 |

**重铸设计：** 最多锁定N-1个词条，保证至少有1个词条被重铸，防止玩家无限刷完美词条。
