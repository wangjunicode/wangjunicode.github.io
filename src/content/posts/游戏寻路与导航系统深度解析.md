# 游戏寻路与导航系统深度解析

导航与寻路是几乎所有游戏都需要的基础技术。从简单的 NPC 走到目标点，到大规模 RTS 单位的群体寻路，背后都有一套精心设计的算法体系。

## 一、寻路算法基础

### 1.1 A* 算法原理

A*（A-star）是游戏寻路中最经典的算法，是 Dijkstra 算法（保证最短路径）和贪心最佳优先搜索（速度快）的结合。

```
A* 核心公式：
f(n) = g(n) + h(n)

g(n) = 从起点到节点 n 的实际代价
h(n) = 从节点 n 到终点的估算代价（启发函数）
f(n) = 总估算代价（越小越优先探索）
```

### 1.2 A* 完整实现

```csharp
public class AStarPathfinder
{
    private int[,] _grid;      // 0=可通行, 1=障碍
    private int _width, _height;
    
    public struct Node
    {
        public int X, Y;
        public float G, H, F;
        public Node? Parent;
        public bool IsWalkable;
        
        public float CalculateH(int targetX, int targetY)
        {
            // 曼哈顿距离（4方向移动）
            // H = |dx| + |dy|
            
            // 欧几里得距离（8方向/自由移动）
            float dx = Math.Abs(X - targetX);
            float dy = Math.Abs(Y - targetY);
            return dx + dy + (1.414f - 2) * Math.Min(dx, dy); // Octile 距离
        }
    }
    
    public List<Vector2Int> FindPath(Vector2Int start, Vector2Int end)
    {
        if (!IsWalkable(end.x, end.y)) return null;
        
        // Open 列表：待探索的节点（用最小堆优化）
        var openSet = new SortedSet<(float f, int id)>();
        var nodes = new Node[_width, _height];
        
        // 初始化起点
        nodes[start.x, start.y] = new Node 
        { 
            X = start.x, Y = start.y, G = 0 
        };
        nodes[start.x, start.y].H = nodes[start.x, start.y].CalculateH(end.x, end.y);
        nodes[start.x, start.y].F = nodes[start.x, start.y].H;
        
        openSet.Add((nodes[start.x, start.y].F, start.x * _height + start.y));
        var inOpen = new HashSet<int> { start.x * _height + start.y };
        var inClosed = new HashSet<int>();
        
        // 8方向移动（包括对角线）
        int[] dx = { 0, 1, 0, -1, 1, 1, -1, -1 };
        int[] dy = { 1, 0, -1, 0, 1, -1, 1, -1 };
        float[] costs = { 1f, 1f, 1f, 1f, 1.414f, 1.414f, 1.414f, 1.414f };
        
        while (openSet.Count > 0)
        {
            // 取 F 值最小的节点
            var (_, currentId) = openSet.Min;
            openSet.Remove(openSet.Min);
            
            int cx = currentId / _height;
            int cy = currentId % _height;
            inOpen.Remove(currentId);
            inClosed.Add(currentId);
            
            // 到达终点
            if (cx == end.x && cy == end.y)
                return ReconstructPath(nodes, end);
            
            // 扩展邻居
            for (int i = 0; i < 8; i++)
            {
                int nx = cx + dx[i];
                int ny = cy + dy[i];
                
                if (!IsWalkable(nx, ny)) continue;
                
                int neighborId = nx * _height + ny;
                if (inClosed.Contains(neighborId)) continue;
                
                float newG = nodes[cx, cy].G + costs[i];
                
                if (!inOpen.Contains(neighborId) || newG < nodes[nx, ny].G)
                {
                    nodes[nx, ny].G = newG;
                    nodes[nx, ny].H = nodes[nx, ny].CalculateH(end.x, end.y);
                    nodes[nx, ny].F = newG + nodes[nx, ny].H;
                    nodes[nx, ny].Parent = nodes[cx, cy];
                    
                    if (!inOpen.Contains(neighborId))
                    {
                        openSet.Add((nodes[nx, ny].F, neighborId));
                        inOpen.Add(neighborId);
                    }
                }
            }
        }
        
        return null; // 无法到达
    }
    
    private List<Vector2Int> ReconstructPath(Node[,] nodes, Vector2Int end)
    {
        var path = new List<Vector2Int>();
        var current = nodes[end.x, end.y];
        
        while (current.Parent.HasValue)
        {
            path.Add(new Vector2Int(current.X, current.Y));
            current = current.Parent.Value;
        }
        
        path.Reverse();
        return path;
    }
    
    private bool IsWalkable(int x, int y)
        => x >= 0 && x < _width && y >= 0 && y < _height && _grid[x, y] == 0;
}
```

### 1.3 路径平滑（String Pulling）

```csharp
// 去除 A* 路径的"锯齿"，生成更自然的路径
public class PathSmoother
{
    // 使用射线检测，跳过不必要的中间点
    public List<Vector3> SmoothPath(List<Vector3> rawPath)
    {
        if (rawPath.Count <= 2) return rawPath;
        
        var smoothed = new List<Vector3> { rawPath[0] };
        int current = 0;
        
        while (current < rawPath.Count - 1)
        {
            // 找到从 current 出发能直线到达的最远点
            int furthest = current + 1;
            
            for (int i = current + 2; i < rawPath.Count; i++)
            {
                if (HasClearLine(rawPath[current], rawPath[i]))
                    furthest = i;
                else
                    break;
            }
            
            smoothed.Add(rawPath[furthest]);
            current = furthest;
        }
        
        return smoothed;
    }
    
    private bool HasClearLine(Vector3 from, Vector3 to)
    {
        // NavMesh 射线检测（比普通物理射线更准确）
        if (NavMesh.Raycast(from, to, out _, NavMesh.AllAreas))
            return false; // 有障碍
        
        return true;
    }
}
```

## 二、Unity NavMesh 高级应用

### 2.1 NavMesh 烘焙配置

```
NavMesh 烘焙参数（重要）：

Agent Settings（Navigation 窗口）：
├── Agent Radius：代理半径（影响能通过的最小通道宽度）
├── Agent Height：代理高度（决定能否通过低矮障碍）
├── Max Slope：最大坡度（超过则无法通行）
├── Step Height：台阶高度（小于此值的台阶自动处理）

烘焙参数：
├── Min Region Area：小于此面积的区域被忽略
└── Manual Voxel Size：体素大小（影响精度和烘焙时间）

优化技巧：
- 静态障碍物：Navigation Static 标记
- 复杂模型：用简化代理几何体替代（减少烘焙时间）
- 模块化关卡：使用 NavMeshSurface 局部烘焙
```

```csharp
        switch (type)
        {

                _agent.areaMask = 1 << NavMesh.GetAreaFromName("Water");
                break;
                
            case CharacterType.Flying:
                // 所有区域都可通行
                _agent.areaMask = NavMesh.AllAreas;
                break;
        }
    }
}
```

```csharp
// 手动 OffMeshLink：连接 NavMesh 不连续区域（跳跃/攀爬/传送）
public class JumpLinkHandler : MonoBehaviour
{
    private NavMeshAgent _agent;
    
    private IEnumerator MoveWithJump()
    {
        _agent.SetDestination(_target.position);
        
        while (!_agent.pathPending && _agent.remainingDistance > 0.1f)
        {
            // 检测是否进入 OffMeshLink
            if (_agent.isOnOffMeshLink)
            {
                yield return StartCoroutine(HandleJump());
            }
            
            yield return null;
        }
    }
    
    private IEnumerator HandleJump()
    {
        _agent.isStopped = true;
        
        var link = _agent.currentOffMeshLinkData;
        Vector3 startPos = _agent.transform.position;
        Vector3 endPos = link.endPos;
        
        // 执行跳跃动画曲线
        float jumpHeight = 2f;
        float jumpDuration = 0.5f;
        float elapsed = 0;
        
        while (elapsed < jumpDuration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / jumpDuration;
            
            Vector3 pos = Vector3.Lerp(startPos, endPos, t);
            pos.y += Mathf.Sin(t * Mathf.PI) * jumpHeight;
            
            _agent.transform.position = pos;
            yield return null;
        }
        
        // 完成 OffMeshLink 移动
        _agent.CompleteOffMeshLink();
        _agent.isStopped = false;
    }
}
```

## 三、大规模单位寻路优化

### 3.1 RVO（相互速度障碍）群体避让

```csharp
// 简化版 RVO2：多个代理互相避让，无需集中式协调
public class RVOAgent
{
    public Vector2 Position;
    public Vector2 Velocity;
    public float MaxSpeed;
    public float Radius;
    
    // 计算避让后的优先速度
    public Vector2 ComputeNewVelocity(List<RVOAgent> neighbors, Vector2 preferredVelocity)
    {
        var orcaLines = new List<(Vector2 point, Vector2 direction)>();
        
        foreach (var neighbor in neighbors)
        {
            Vector2 relativePos = neighbor.Position - Position;
            Vector2 relativeVel = Velocity - neighbor.Velocity;
            
            float combinedRadius = Radius + neighbor.Radius;
            float distSq = relativePos.sqrMagnitude;
            
            if (distSq > combinedRadius * combinedRadius)
            {
                // 生成 ORCA 半平面约束
                // （完整 RVO2 算法较复杂，这里是简化示意）
                Vector2 w = relativeVel - relativePos;
                float t = Vector2.Dot(w, relativePos) / distSq;
                
                Vector2 normal = (w - t * relativePos).normalized;
                float u = combinedRadius / Mathf.Sqrt(distSq) - 1;
                
                orcaLines.Add((Velocity + 0.5f * u * normal, normal.Perpendicular()));
            }
        }
        
        // 在约束范围内找最接近优先速度的合法速度
        return FindBestVelocity(preferredVelocity, orcaLines, MaxSpeed);
    }
}
```

### 3.2 分层导航（Hierarchical Pathfinding）

```csharp
// 超大地图（MMORPG/RTS）使用分层导航
// 低层：格子地图精确寻路（局部）
// 高层：区域图粗粒度规划（全局）

public class HierarchicalPathfinder
{
    private Graph _regionGraph;    // 高层：区域连通图
    private Dictionary<int, AStarPathfinder> _localPathfinders; // 低层：区域内精确寻路
    
    public List<Vector3> FindPath(Vector3 start, Vector3 goal)
    {
        // 1. 高层规划：找区域路径
        int startRegion = GetRegionId(start);
        int goalRegion = GetRegionId(goal);
        
        var regionPath = _regionGraph.FindPath(startRegion, goalRegion);
        if (regionPath == null) return null;
        
        // 2. 低层执行：在每个区域内精确寻路
        var fullPath = new List<Vector3>();
        
        for (int i = 0; i < regionPath.Count - 1; i++)
        {
            var localPath = _localPathfinders[regionPath[i]]
                .FindPath(GetRegionEntry(regionPath[i]), GetRegionExit(regionPath[i], regionPath[i + 1]));
            
            fullPath.AddRange(localPath);
        }
        
        return fullPath;
    }
}
```

## 五、导航系统选型建议

| 需求 | 推荐方案 |
|------|----------|
| 普通 NPC（Unity 项目） | Unity NavMesh（零成本上手）|
| 动态地形/程序化地图 | 自实现 A* + 动态更新网格 |
| 大规模单位（RTS） | HPA* + RVO 群体避让 |
| 3D 自由空间（飞行） | 3D A* 或向量场寻路 |
| 网格类游戏 | BFS/Dijkstra（更简单）|

> 💡 **实用建议**：90% 的项目直接用 Unity NavMesh 就够了。自实现 A* 的意义在于理解原理，而不是在生产项目中替代 NavMesh。真正需要自定义寻路的场景：程序化生成地图、特殊移动方式（游泳/飞行/攀爬）、超过 1000 个同时寻路的单位。
