## 一、排行榜客户端实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class LeaderboardEntry
{
    public string PlayerId;
    public string PlayerName;
    public string AvatarUrl;
    public long Score;
    public int Rank;
    public bool IsCurrentPlayer;
}

[Serializable]
public class LeaderboardResponse
{
    public List<LeaderboardEntry> Entries;
    public int CurrentPlayerRank;
    public long CurrentPlayerScore;
    public int TotalPlayers;
    public long RefreshTime;  // 下次刷新时间
}

public class LeaderboardManager : MonoBehaviour
{
    private static LeaderboardManager instance;
    public static LeaderboardManager Instance => instance;

    private const float CACHE_DURATION = 60f; // 缓存60秒
    private Dictionary<string, (LeaderboardResponse data, float cacheTime)> cache 
        = new Dictionary<string, (LeaderboardResponse, float)>();

    void Awake() { instance = this; }

    public async System.Threading.Tasks.Task<LeaderboardResponse> GetLeaderboard(
        string boardId, int page = 0, int pageSize = 20)
    {
        string cacheKey = $"{boardId}_{page}";
        
        if (cache.TryGetValue(cacheKey, out var cached))
        {
            if (Time.time - cached.cacheTime < CACHE_DURATION)
                return cached.data;
        }
        
        try
        {
            var response = await NetworkService.GetLeaderboard(boardId, page, pageSize);
            cache[cacheKey] = (response, Time.time);
            return response;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Leaderboard] 获取排行榜失败: {e.Message}");
            return cached.data; // 返回旧缓存
        }
    }

    /// <summary>
    /// 获取当前玩家周围的排名（常见UI：上5名+本人+下5名）
    /// </summary>
    public async System.Threading.Tasks.Task<List<LeaderboardEntry>> GetSurroundingEntries(
        string boardId, int surroundCount = 5)
    {
        var entries = await NetworkService.GetSurroundingEntries(boardId, surroundCount);
        return entries;
    }

    /// <summary>
    /// 提交分数（需服务端验证）
    /// </summary>
    public async System.Threading.Tasks.Task<bool> SubmitScore(string boardId, long score)
    {
        // 客户端只发送请求，分数由服务端计算验证
        return await NetworkService.SubmitLeaderboardScore(boardId, score);
    }

    public void InvalidateCache(string boardId)
    {
        var toRemove = new List<string>();
        foreach (var key in cache.Keys)
            if (key.StartsWith(boardId)) toRemove.Add(key);
        foreach (var key in toRemove) cache.Remove(key);
    }
}
```

## 二、排行榜UI组件

```csharp
public class LeaderboardUI : MonoBehaviour
{
    [SerializeField] private Transform entryContainer;
    [SerializeField] private GameObject entryPrefab;
    [SerializeField] private UnityEngine.UI.Text playerRankText;
    [SerializeField] private string boardId = "global_score";

    async void OnEnable()
    {
        var response = await LeaderboardManager.Instance.GetLeaderboard(boardId);
        if (response == null) return;
        
        // 清空列表
        foreach (Transform child in entryContainer)
            Destroy(child.gameObject);
        
        // 填充数据
        foreach (var entry in response.Entries)
        {
            var go = Instantiate(entryPrefab, entryContainer);
            var item = go.GetComponent<LeaderboardEntryItem>();
            item?.Setup(entry);
        }
        
        // 更新当前玩家排名
        playerRankText.text = response.CurrentPlayerRank > 0 ? 
            $"我的排名: #{response.CurrentPlayerRank}" : "未上榜";
    }
}
```

## 四、服务端排行榜（Redis）

```
Redis SortedSet 排行榜实现：

ZADD leaderboard:season1 9500 "player_001"
ZADD leaderboard:season1 8800 "player_002"

获取排名（降序第1名）：
ZREVRANK leaderboard:season1 "player_001"  → 0（第1名）

获取Top10：
ZREVRANGE leaderboard:season1 0 9 WITHSCORES

获取玩家周围玩家：
1. ZREVRANK → rank
2. ZREVRANGE rank-5 rank+5 WITHSCORES

时间复杂度：
- ZADD: O(log N)
- ZREVRANK: O(log N)
- ZREVRANGE: O(log N + M)
```
