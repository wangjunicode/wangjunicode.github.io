## 一、技能资源系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

public enum ResourceType { Mana, Rage, Energy, Heat, Combo, Shield }

[Serializable]
public class SkillResourceConfig
{
    public ResourceType Type;
    public float MaxValue;
    public float InitialValue;
    public float RegenPerSecond;      // 每秒自然恢复（可为负=自然消耗）
    public float RegenDelay;          // 停止行动多久后开始恢复
    public bool AllowOverflow;        // 是否允许超出MaxValue（护盾/超充）
    public float OverflowMax;         // 溢出上限
}

public class SkillResourceManager : MonoBehaviour
{
    [SerializeField] private SkillResourceConfig[] resourceConfigs;
    
    private Dictionary<ResourceType, float> currentValues = new Dictionary<ResourceType, float>();
    private Dictionary<ResourceType, float> regenTimers = new Dictionary<ResourceType, float>();
    private Dictionary<ResourceType, SkillResourceConfig> configs = new Dictionary<ResourceType, SkillResourceConfig>();
    
    public event Action<ResourceType, float, float> OnValueChanged; // type, oldVal, newVal

    void Awake()
    {
        foreach (var cfg in resourceConfigs)
        {
            configs[cfg.Type] = cfg;
            currentValues[cfg.Type] = cfg.InitialValue;
            regenTimers[cfg.Type] = 0;
        }
    }

    void Update()
    {
        float dt = Time.deltaTime;
        
        foreach (var cfg in resourceConfigs)
        {
            if (cfg.RegenPerSecond == 0) continue;
            
            var timer = regenTimers.TryGetValue(cfg.Type, out float t) ? t : 0;
            
            if (cfg.RegenDelay > 0)
            {
                timer -= dt;
                regenTimers[cfg.Type] = timer;
                if (timer > 0) continue;
            }
            
            float current = GetValue(cfg.Type);
            float newValue = Mathf.Clamp(current + cfg.RegenPerSecond * dt, 0, cfg.MaxValue);
            
            if (!Mathf.Approximately(current, newValue))
                SetValue(cfg.Type, newValue);
        }
    }

    public bool CanUse(ResourceType type, float cost)
    {
        return GetValue(type) >= cost;
    }

    public bool Consume(ResourceType type, float cost)
    {
        if (!CanUse(type, cost)) return false;
        
        SetValue(type, GetValue(type) - cost);
        
        // 重置恢复延迟
        if (configs.TryGetValue(type, out var cfg) && cfg.RegenDelay > 0)
            regenTimers[type] = cfg.RegenDelay;
        
        return true;
    }

    public void Add(ResourceType type, float amount)
    {
        if (!configs.TryGetValue(type, out var cfg)) return;
        
        float max = cfg.AllowOverflow ? cfg.OverflowMax : cfg.MaxValue;
        float newVal = Mathf.Clamp(GetValue(type) + amount, 0, max);
        SetValue(type, newVal);
    }

    void SetValue(ResourceType type, float value)
    {
        float old = GetValue(type);
        currentValues[type] = value;
        if (!Mathf.Approximately(old, value))
            OnValueChanged?.Invoke(type, old, value);
    }

    public float GetValue(ResourceType type) =>
        currentValues.TryGetValue(type, out float v) ? v : 0;
    
    public float GetPercent(ResourceType type)
    {
        if (!configs.TryGetValue(type, out var cfg)) return 0;
        return GetValue(type) / cfg.MaxValue;
    }
    
    public float GetMax(ResourceType type) =>
        configs.TryGetValue(type, out var cfg) ? cfg.MaxValue : 0;
}
```

## 二、各资源类型设计

| 类型 | 恢复机制 | 游戏案例 |
|------|----------|---------|
| 法力（Mana）| 时间自动恢复 | 法师类技能 |
| 怒气（Rage）| 受击/攻击积累 | 战士类连招 |
| 能量（Energy）| 时间快速恢复（上限低）| 盗贼/刺客 |
| 热度（Heat）| 技能使用积累，自然降低 | 机甲/枪械过热 |
| 连击（Combo）| 连续攻击积累，断开重置 | 动作游戏特技 |
| 护盾（Shield）| 不自然恢复，技能给予 | 任意 |

## 三、资源条UI动画

```csharp
public class ResourceBarUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Slider mainBar;
    [SerializeField] private UnityEngine.UI.Slider ghostBar;  // 延迟跟随（白色气泡效果）
    [SerializeField] private ResourceType resourceType;
    [SerializeField] private SkillResourceManager resourceMgr;
    [SerializeField] private float ghostFollowSpeed = 2f;

    void Start()
    {
        resourceMgr.OnValueChanged += OnValueChanged;
    }

    void OnValueChanged(ResourceType type, float oldVal, float newVal)
    {
        if (type != resourceType) return;
        float percent = resourceMgr.GetPercent(resourceType);
        mainBar.value = percent;
        // ghostBar 通过Update平滑跟随
    }

    void Update()
    {
        if (ghostBar.value > mainBar.value)
            ghostBar.value = Mathf.MoveTowards(
                ghostBar.value, mainBar.value, ghostFollowSpeed * Time.deltaTime);
        else
            ghostBar.value = mainBar.value;
    }
}
```
