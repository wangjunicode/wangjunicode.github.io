## 一、属性定义

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 属性ID（扩展方便，用字符串Key）
/// </summary>
public static class StatId
{
    // 基础属性
    public const string MaxHP       = "max_hp";
    public const string MaxMP       = "max_mp";
    public const string Strength    = "str";
    public const string Agility     = "agi";
    public const string Intelligence = "int";
    
    // 战斗属性
    public const string AttackDamage = "attack_damage";
    public const string MagicDamage  = "magic_damage";
    public const string Defense      = "defense";
    public const string MagicResist  = "magic_resist";
    public const string CritRate     = "crit_rate";
    public const string CritMultiplier = "crit_multiplier";
    public const string AttackSpeed  = "attack_speed";
    public const string MoveSpeed    = "move_speed";
    
    // 抗性
    public const string FireResist   = "fire_resist";
    public const string IceResist    = "ice_resist";
}

/// <summary>
/// 属性修改器类型（计算顺序：Base → FlatAdd → PercentAdd → FinalMultiply）
/// </summary>
public enum StatModifierOrder
{
    BaseValue    = 0,  // 基础值（角色等级/种族基础值）
    FlatBonus    = 100, // 加法加成（装备+50攻击）
    PercentBonus = 200, // 百分比加法（+20%攻击）
    FinalMultiply = 300 // 最终乘法（×1.5，用于BUFF）
}

/// <summary>
/// 属性修改器
/// </summary>
public class AttributeModifier
{
    public string Source;              // 来源（装备ID/BUFF ID）
    public StatModifierOrder Order;
    public float Value;
    
    public AttributeModifier(string source, StatModifierOrder order, float value)
    {
        Source = source;
        Order = order;
        Value = value;
    }
}

/// <summary>
/// 单个属性（支持多层修改器）
/// </summary>
public class Attribute
{
    public string Id;
    private float baseValue;
    private List<AttributeModifier> modifiers = new List<AttributeModifier>();
    private float? cachedValue; // 缓存计算结果
    
    public event Action<float, float> OnValueChanged; // oldValue, newValue

    public Attribute(string id, float baseValue)
    {
        Id = id;
        this.baseValue = baseValue;
    }

    public float BaseValue
    {
        get => baseValue;
        set
        {
            if (Mathf.Approximately(baseValue, value)) return;
            baseValue = value;
            InvalidateCache();
        }
    }

    /// <summary>
    /// 获取最终计算值
    /// </summary>
    public float Value
    {
        get
        {
            if (!cachedValue.HasValue)
                cachedValue = CalculateFinalValue();
            return cachedValue.Value;
        }
    }

    public void AddModifier(AttributeModifier modifier)
    {
        modifiers.Add(modifier);
        InvalidateCache();
    }

    public void RemoveModifier(string source)
    {
        int removed = modifiers.RemoveAll(m => m.Source == source);
        if (removed > 0) InvalidateCache();
    }

    public void RemoveAllModifiers()
    {
        modifiers.Clear();
        InvalidateCache();
    }

    float CalculateFinalValue()
    {
        // 按顺序计算：Base + FlatAdd → × (1 + PercentBonus) → × FinalMultiply
        float flat = 0f;
        float percentBonus = 0f;
        float finalMultiply = 1f;

        foreach (var mod in modifiers)
        {
            switch (mod.Order)
            {
                case StatModifierOrder.FlatBonus:
                    flat += mod.Value;
                    break;
                case StatModifierOrder.PercentBonus:
                    percentBonus += mod.Value;
                    break;
                case StatModifierOrder.FinalMultiply:
                    finalMultiply *= mod.Value;
                    break;
            }
        }

        float result = (baseValue + flat) * (1f + percentBonus / 100f) * finalMultiply;
        return result;
    }

    void InvalidateCache()
    {
        float oldValue = cachedValue ?? baseValue;
        cachedValue = null;
        float newValue = Value;

        if (!Mathf.Approximately(oldValue, newValue))
            OnValueChanged?.Invoke(oldValue, newValue);
    }
}
```

## 二、角色属性管理器

```csharp
/// <summary>
/// 角色属性管理器
/// </summary>
public class CharacterAttributes : MonoBehaviour
{
    [Header("基础属性初始值")]
    [SerializeField] private float baseMaxHP = 500f;
    [SerializeField] private float baseStr = 10f;
    [SerializeField] private float baseAgi = 10f;
    [SerializeField] private float baseInt = 10f;

    private Dictionary<string, Attribute> attributes = new Dictionary<string, Attribute>();
    
    public event Action<string> OnAttributeChanged; // statId

    void Awake()
    {
        InitializeAttributes();
        SetupDerivedAttributes();
    }

    void InitializeAttributes()
    {
        AddAttribute(StatId.MaxHP,        baseMaxHP);
        AddAttribute(StatId.Strength,     baseStr);
        AddAttribute(StatId.Agility,      baseAgi);
        AddAttribute(StatId.Intelligence, baseInt);
        AddAttribute(StatId.AttackDamage, 0f);
        AddAttribute(StatId.Defense,      0f);
        AddAttribute(StatId.CritRate,     5f);   // 基础5%暴击率
        AddAttribute(StatId.CritMultiplier, 150f); // 基础150%暴击倍率
        AddAttribute(StatId.AttackSpeed,  1f);
        AddAttribute(StatId.MoveSpeed,    5f);
    }

    void SetupDerivedAttributes()
    {
        // 力量影响攻击力：每点力量+3攻击
        Get(StatId.Strength).OnValueChanged += (oldVal, newVal) =>
        {
            float bonus = newVal * 3f;
            Get(StatId.AttackDamage).RemoveModifier("strength_scaling");
            Get(StatId.AttackDamage).AddModifier(new AttributeModifier(
                "strength_scaling", StatModifierOrder.FlatBonus, bonus));
        };
        
        // 敏捷影响攻击速度：每10敏捷+0.1攻速
        Get(StatId.Agility).OnValueChanged += (oldVal, newVal) =>
        {
            float bonus = newVal / 10f * 0.1f;
            Get(StatId.AttackSpeed).RemoveModifier("agility_attack_speed");
            Get(StatId.AttackSpeed).AddModifier(new AttributeModifier(
                "agility_attack_speed", StatModifierOrder.PercentBonus, bonus * 100f));
        };
        
        // 智力影响魔法值
        Get(StatId.Intelligence).OnValueChanged += (oldVal, newVal) =>
        {
            float mpBonus = newVal * 5f;
            Get(StatId.MaxMP).RemoveModifier("int_mp");
            Get(StatId.MaxMP).AddModifier(new AttributeModifier(
                "int_mp", StatModifierOrder.FlatBonus, mpBonus));
        };
    }

    Attribute AddAttribute(string id, float baseValue)
    {
        var attr = new Attribute(id, baseValue);
        attr.OnValueChanged += (o, n) => OnAttributeChanged?.Invoke(id);
        attributes[id] = attr;
        return attr;
    }

    public Attribute Get(string statId)
    {
        return attributes.TryGetValue(statId, out var attr) ? attr : null;
    }

    public float GetValue(string statId)
    {
        return Get(statId)?.Value ?? 0f;
    }

    /// <summary>
    /// 装备物品（应用所有装备属性修改器）
    /// </summary>
    public void ApplyEquipment(string equipmentId, List<StatModifier> statMods)
    {
        foreach (var mod in statMods)
        {
            var attr = Get(mod.StatId);
            attr?.AddModifier(new AttributeModifier(
                equipmentId, 
                (StatModifierOrder)mod.Order, 
                mod.Value));
        }
    }

    /// <summary>
    /// 卸装（移除所有该装备的修改器）
    /// </summary>
    public void RemoveEquipment(string equipmentId)
    {
        foreach (var attr in attributes.Values)
            attr.RemoveModifier(equipmentId);
    }

    /// <summary>
    /// 打印所有属性（调试用）
    /// </summary>
    [ContextMenu("Print All Attributes")]
    public void PrintAttributes()
    {
        foreach (var kv in attributes)
            Debug.Log($"[Attribute] {kv.Key}: {kv.Value.Value:F2} (base: {kv.Value.BaseValue:F2})");
    }
}
```

## 三、属性计算示例

```
角色攻击力计算示例：

基础攻击力 = 50（角色基础值）

装备修改器：
  FlatBonus:    +100（长剑）
  FlatBonus:    +50 （戒指）
  PercentBonus: +20%（宝石）

BUFF修改器：
  FinalMultiply: ×1.5（战斗狂怒）

最终计算：
  (50 + 100 + 50) × (1 + 20%) × 1.5
= 200 × 1.2 × 1.5
= 360 攻击力
```

## 四、设计要点

| 要点 | 方案 |
|------|------|
| 缓存计算结果 | 属性变化时才重新计算，避免每帧重算 |
| 修改器来源追踪 | 每个Modifier标注来源，支持精确移除 |
| 派生属性 | 通过事件订阅，力量变化自动更新攻击力 |
| 上下限 | 某些属性需要Clamp（如暴击率0-100%）|
| 服务端校验 | 关键战斗属性必须服务端计算验证 |
