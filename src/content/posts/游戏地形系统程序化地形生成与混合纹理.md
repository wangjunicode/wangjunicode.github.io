## 一、程序化地形生成

```csharp
using UnityEngine;

/// <summary>
/// 程序化地形生成器（多倍频 Perlin Noise + 侵蚀）
/// </summary>
[RequireComponent(typeof(Terrain))]
public class ProceduralTerrainGenerator : MonoBehaviour
{
    [Header("地形尺寸")]
    [SerializeField] private int resolution = 513;   // 必须是 2^n + 1
    [SerializeField] private float terrainWidth = 1000f;
    [SerializeField] private float terrainLength = 1000f;
    [SerializeField] private float terrainHeight = 200f;

    [Header("噪声参数")]
    [SerializeField] private float noiseScale = 0.003f;
    [SerializeField] private int octaves = 6;
    [SerializeField] private float persistence = 0.5f;   // 振幅衰减
    [SerializeField] private float lacunarity = 2f;      // 频率增长
    [SerializeField] private Vector2 noiseOffset;
    [SerializeField] private int seed = 12345;

    [Header("侵蚀")]
    [SerializeField] private bool applyErosion = true;
    [SerializeField] private int erosionIterations = 5000;
    [SerializeField] private float erosionStrength = 0.03f;

    [Header("纹理混合")]
    [SerializeField] private TerrainLayerConfig[] layerConfigs;

    private Terrain terrain;

    [System.Serializable]
    public class TerrainLayerConfig
    {
        public TerrainLayer Layer;
        public float MinHeight;       // 归一化高度 0-1
        public float MaxHeight;
        public float MinSlope;        // 坡度度数
        public float MaxSlope;
        public float Blend = 0.1f;    // 混合过渡宽度
    }

    [ContextMenu("生成地形")]
    public void Generate()
    {
        terrain = GetComponent<Terrain>();
        var data = terrain.terrainData;

        data.heightmapResolution = resolution;
        data.size = new Vector3(terrainWidth, terrainHeight, terrainLength);

        // 生成高度图
        float[,] heights = GenerateHeightmap();

        // 应用侵蚀
        if (applyErosion)
            ApplyHydraulicErosion(heights);

        data.SetHeights(0, 0, heights);

        // 应用纹理
        if (layerConfigs != null && layerConfigs.Length > 0)
            ApplyTextures(data, heights);

        Debug.Log($"[Terrain] 生成完成 {resolution}x{resolution}");
    }

    float[,] GenerateHeightmap()
    {
        var heights = new float[resolution, resolution];

        // 使用种子初始化随机偏移
        var rng = new System.Random(seed);
        Vector2[] octaveOffsets = new Vector2[octaves];
        for (int i = 0; i < octaves; i++)
        {
            float ox = rng.Next(-100000, 100000) + noiseOffset.x;
            float oy = rng.Next(-100000, 100000) + noiseOffset.y;
            octaveOffsets[i] = new Vector2(ox, oy);
        }

        float maxH = float.MinValue, minH = float.MaxValue;
        float[,] raw = new float[resolution, resolution];

        for (int y = 0; y < resolution; y++)
        {
            for (int x = 0; x < resolution; x++)
            {
                float amplitude = 1f, frequency = 1f, noiseVal = 0f;

                for (int o = 0; o < octaves; o++)
                {
                    float sampleX = (x * noiseScale * frequency) + octaveOffsets[o].x;
                    float sampleY = (y * noiseScale * frequency) + octaveOffsets[o].y;
                    noiseVal += Mathf.PerlinNoise(sampleX, sampleY) * amplitude;
                    amplitude *= persistence;
                    frequency *= lacunarity;
                }

                raw[y, x] = noiseVal;
                if (noiseVal > maxH) maxH = noiseVal;
                if (noiseVal < minH) minH = noiseVal;
            }
        }

        // 归一化到 0-1
        for (int y = 0; y < resolution; y++)
            for (int x = 0; x < resolution; x++)
                heights[y, x] = Mathf.InverseLerp(minH, maxH, raw[y, x]);

        return heights;
    }

    /// <summary>
    /// 简化液压侵蚀
    /// </summary>
    void ApplyHydraulicErosion(float[,] heights)
    {
        var rng = new System.Random(seed + 1);

        for (int iter = 0; iter < erosionIterations; iter++)
        {
            // 随机雨滴位置
            int x = rng.Next(1, resolution - 1);
            int y = rng.Next(1, resolution - 1);

            float h = heights[y, x];

            // 找最低邻居
            float minNeighbor = h;
            int minX = x, minY = y;

            for (int dy = -1; dy <= 1; dy++)
            {
                for (int dx = -1; dx <= 1; dx++)
                {
                    if (dx == 0 && dy == 0) continue;
                    int nx = x + dx, ny = y + dy;
                    if (nx < 0 || nx >= resolution || ny < 0 || ny >= resolution) continue;

                    if (heights[ny, nx] < minNeighbor)
                    {
                        minNeighbor = heights[ny, nx];
                        minX = nx; minY = ny;
                    }
                }
            }

            // 搬运高度差的一部分
            if (minX != x || minY != y)
            {
                float diff = (h - minNeighbor) * erosionStrength;
                heights[y, x] -= diff;
                heights[minY, minX] += diff * 0.8f; // 少量沉积
            }
        }
    }

    void ApplyTextures(TerrainData data, float[,] heights)
    {
        if (layerConfigs.Length == 0) return;

        // 设置纹理层
        data.terrainLayers = System.Array.ConvertAll(layerConfigs, c => c.Layer);

        int alphaRes = data.alphamapResolution;
        float[,,] alphaMaps = new float[alphaRes, alphaRes, layerConfigs.Length];

        for (int y = 0; y < alphaRes; y++)
        {
            for (int x = 0; x < alphaRes; x++)
            {
                float normalizedX = (float)x / alphaRes;
                float normalizedY = (float)y / alphaRes;

                // 获取该点高度和坡度
                float h = data.GetInterpolatedHeight(normalizedX, normalizedY) / terrainHeight;
                float slope = data.GetSteepness(normalizedX, normalizedY);

                float[] weights = new float[layerConfigs.Length];

                for (int l = 0; l < layerConfigs.Length; l++)
                {
                    var cfg = layerConfigs[l];
                    float heightWeight = SmoothStep(cfg.MinHeight - cfg.Blend, cfg.MinHeight, h)
                                      * SmoothStep(cfg.MaxHeight + cfg.Blend, cfg.MaxHeight, h);
                    float slopeWeight = SmoothStep(cfg.MinSlope - 5f, cfg.MinSlope, slope)
                                      * SmoothStep(cfg.MaxSlope + 5f, cfg.MaxSlope, slope);
                    weights[l] = heightWeight * slopeWeight;
                }

                // 归一化权重
                float sum = 0;
                foreach (var w in weights) sum += w;
                if (sum <= 0) weights[0] = 1f; // 保底第一层
                else for (int l = 0; l < weights.Length; l++) weights[l] /= sum;

                for (int l = 0; l < layerConfigs.Length; l++)
                    alphaMaps[y, x, l] = weights[l];
            }
        }

        data.SetAlphamaps(0, 0, alphaMaps);
    }

    float SmoothStep(float from, float to, float value)
    {
        float t = Mathf.Clamp01((value - from) / (to - from));
        return t * t * (3f - 2f * t);
    }
}
```

## 二、HeightMap 导入导出

```csharp
/// <summary>
/// HeightMap PNG 导入/导出工具
/// </summary>
public static class TerrainHeightMapIO
{
    /// <summary>
    /// 将 Terrain HeightMap 导出为 16位灰度 PNG
    /// </summary>
    public static void Export(Terrain terrain, string filePath)
    {
        var data = terrain.terrainData;
        int res = data.heightmapResolution;
        float[,] heights = data.GetHeights(0, 0, res, res);

        var texture = new Texture2D(res, res, TextureFormat.R16, false);
        var pixels = new Color[res * res];

        for (int y = 0; y < res; y++)
            for (int x = 0; x < res; x++)
                pixels[y * res + x] = new Color(heights[y, x], 0, 0, 1);

        texture.SetPixels(pixels);
        byte[] bytes = texture.EncodeToPNG();
        System.IO.File.WriteAllBytes(filePath, bytes);

        Object.Destroy(texture);
        Debug.Log($"[Terrain] HeightMap exported to {filePath}");
    }

    /// <summary>
    /// 从 PNG 导入 HeightMap
    /// </summary>
    public static void Import(Terrain terrain, string filePath)
    {
        byte[] bytes = System.IO.File.ReadAllBytes(filePath);
        var texture = new Texture2D(2, 2);
        texture.LoadImage(bytes);

        var data = terrain.terrainData;
        int res = data.heightmapResolution;

        float[,] heights = new float[res, res];

        for (int y = 0; y < res; y++)
        {
            for (int x = 0; x < res; x++)
            {
                float u = (float)x / res;
                float v = (float)y / res;
                heights[y, x] = texture.GetPixelBilinear(u, v).r;
            }
        }

        data.SetHeights(0, 0, heights);
        Object.Destroy(texture);

        Debug.Log($"[Terrain] HeightMap imported from {filePath}");
    }
}
```

## 三、地形性能优化

| 优化项 | 方案 | 效果 |
|--------|------|------|
| 地形 LOD | 调整 Billboard Distance | 远处地形降低分辨率 |
| Draw Instanced | 开启 Terrain.drawInstanced | GPU Instancing支持 |
| 纹理层数 | 最多4-8层 | 超过8层性能急剧下降 |
| Pixel Error | 调大 Pixel Error | 减少多边形数量 |
| 树/草密度 | 限制Detail Distance | 草和树是性能大户 |
| 大世界分块 | 地形分块 + Addressable加载 | 支持超大地图 |

**最佳实践：**
1. 高度图分辨率 513 或 1025（2^n+1）
2. 纹理图分辨率与高度图一致
3. 侵蚀算法可以大幅提升真实感（推荐 Sebastian Lague 侵蚀实现）
4. 地形碰撞体开启 Terrain Collider 精度自动降级
