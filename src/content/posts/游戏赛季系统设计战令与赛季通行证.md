## 一、赛季通行证数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 赛季等级奖励配置
/// </summary>
[Serializable]
public class SeasonLevelReward
{
    public int Level;                     // 需要达到的等级
    
    [Header("免费奖励（所有玩家）")]
    public List<RewardItem> FreeRewards;
    
    [Header("付费奖励（持有通行证）")]
    public List<RewardItem> PremiumRewards;
    
    [Header("超级通行证额外奖励")]
    public List<RewardItem> UltraRewards;
}

[Serializable]
public class RewardItem
{
    public string ItemId;
    public string ItemName;
    public int Quantity;
    public RewardType Type;
    public Sprite Icon;
}

public enum RewardType
{
    Currency,       // 游戏货币
    PremiumCurrency, // 钻石/高级货币
    Skin,           // 皮肤
    Emote,          // 表情
    Banner,         // 旗帜/背景
    NameCard,       // 名片
    XPBoost,        // 经验加速
    GiftBox         // 礼盒
}

/// <summary>
/// 赛季配置
/// </summary>
[CreateAssetMenu(fileName = "Season", menuName = "Game/Season Config")]
public class SeasonConfig : ScriptableObject
{
    [Header("赛季基础")]
    public int SeasonNumber;
    public string SeasonName;           // "第X赛季：xxx主题"
    public Sprite SeasonBanner;
    public Color SeasonThemeColor;
    
    [Header("时间")]
    public long StartTimestamp;         // UTC Unix时间戳
    public long EndTimestamp;
    
    [Header("等级配置")]
    public int MaxLevel = 100;          // 最大赛季等级
    public int[] XpRequiredPerLevel;    // 每级需要的经验（长度=MaxLevel）
    
    [Header("奖励配置")]
    public List<SeasonLevelReward> Rewards;
    
    [Header("价格")]
    public int PremiumPassPrice = 980;  // 钻石
    public int UltraPassPrice = 1980;
    
    public bool IsActive => 
        DateTimeOffset.UtcNow.ToUnixTimeSeconds() >= StartTimestamp &&
        DateTimeOffset.UtcNow.ToUnixTimeSeconds() < EndTimestamp;
    
    public TimeSpan GetRemainingTime() =>
        DateTimeOffset.FromUnixTimeSeconds(EndTimestamp).UtcDateTime - DateTime.UtcNow;
    
    public int GetXpRequiredForLevel(int level)
    {
        if (level <= 0 || level > MaxLevel) return 0;
        return XpRequiredPerLevel != null && level <= XpRequiredPerLevel.Length ?
            XpRequiredPerLevel[level - 1] : 1000;
    }
}
```

## 二、赛季进度管理器

```csharp
/// <summary>
/// 赛季进度管理器
/// </summary>
public class SeasonManager : MonoBehaviour
{
    private static SeasonManager instance;
    public static SeasonManager Instance => instance;

    [SerializeField] private SeasonConfig currentSeason;

    private int playerLevel = 0;       // 当前赛季等级
    private long playerXp = 0;          // 当前赛季经验
    private bool hasPremiumPass;        // 是否持有付费通行证
    private bool hasUltraPass;
    private HashSet<int> claimedLevels = new HashSet<int>(); // 已领取的等级奖励

    public event Action<int, long> OnXpChanged;         // level, xp
    public event Action<int> OnLevelUp;                 // newLevel
    public event Action<SeasonLevelReward> OnRewardAvailable; // 有奖励可领取

    void Awake() 
    { 
        instance = this;
        LoadSeasonProgress();
    }

    /// <summary>
    /// 增加赛季经验（完成任务/对局结束时调用）
    /// </summary>
    public void AddXp(long xpAmount, string source = "")
    {
        if (currentSeason == null || !currentSeason.IsActive) return;
        
        // 经验加速加成
        float xpBonus = GetXpBoostMultiplier();
        long actualXp = (long)(xpAmount * xpBonus);
        
        playerXp += actualXp;
        
        // 检查是否升级
        while (playerLevel < currentSeason.MaxLevel)
        {
            long required = currentSeason.GetXpRequiredForLevel(playerLevel + 1);
            if (playerXp >= required)
            {
                playerXp -= required;
                playerLevel++;
                OnLevelUp?.Invoke(playerLevel);
                
                // 通知可以领取奖励
                var reward = currentSeason.Rewards.Find(r => r.Level == playerLevel);
                if (reward != null && !claimedLevels.Contains(playerLevel))
                    OnRewardAvailable?.Invoke(reward);
            }
            else
            {
                break;
            }
        }
        
        OnXpChanged?.Invoke(playerLevel, playerXp);
        SaveSeasonProgress();
    }

    /// <summary>
    /// 领取等级奖励
    /// </summary>
    public List<RewardItem> ClaimLevelReward(int level)
    {
        if (claimedLevels.Contains(level))
        {
            Debug.LogWarning($"[Season] Level {level} reward already claimed");
            return null;
        }
        
        if (playerLevel < level)
        {
            Debug.LogWarning($"[Season] Level {level} not reached yet");
            return null;
        }
        
        var reward = currentSeason.Rewards.Find(r => r.Level == level);
        if (reward == null) return null;
        
        claimedLevels.Add(level);
        
        var claimedItems = new List<RewardItem>();
        claimedItems.AddRange(reward.FreeRewards);
        
        if (hasPremiumPass)
            claimedItems.AddRange(reward.PremiumRewards);
        
        if (hasUltraPass)
            claimedItems.AddRange(reward.UltraRewards);
        
        // 发放奖励
        foreach (var item in claimedItems)
            InventoryManager.Instance?.AddItem(item.ItemId, item.Quantity);
        
        SaveSeasonProgress();
        return claimedItems;
    }

    /// <summary>
    /// 购买高级通行证（可以追溯领取之前的奖励）
    /// </summary>
    public bool PurchasePremiumPass()
    {
        if (hasPremiumPass) return false;
        
        int price = currentSeason.PremiumPassPrice;
        if (!CurrencyManager.Instance.SpendPremiumCurrency(price)) return false;
        
        hasPremiumPass = true;
        
        // 追溯领取之前未领的付费奖励
        ClaimMissedPremiumRewards();
        
        SaveSeasonProgress();
        return true;
    }

    void ClaimMissedPremiumRewards()
    {
        foreach (var reward in currentSeason.Rewards)
        {
            if (reward.Level <= playerLevel && 
                claimedLevels.Contains(reward.Level) &&
                reward.PremiumRewards?.Count > 0)
            {
                // 补发付费奖励
                foreach (var item in reward.PremiumRewards)
                    InventoryManager.Instance?.AddItem(item.ItemId, item.Quantity);
            }
        }
    }

    float GetXpBoostMultiplier()
    {
        float multiplier = 1f;
        if (hasUltraPass) multiplier += 0.2f;  // Ultra 通行证 +20% XP
        // 周末/活动期间额外加速
        return multiplier;
    }

    // 赛季进度存储（简化）
    void SaveSeasonProgress()
    {
        PlayerPrefs.SetInt("season_level", playerLevel);
        PlayerPrefs.SetString("season_xp", playerXp.ToString());
        PlayerPrefs.SetInt("season_has_premium", hasPremiumPass ? 1 : 0);
    }

    void LoadSeasonProgress()
    {
        playerLevel = PlayerPrefs.GetInt("season_level", 0);
        playerXp = long.TryParse(PlayerPrefs.GetString("season_xp", "0"), out long xp) ? xp : 0;
        hasPremiumPass = PlayerPrefs.GetInt("season_has_premium", 0) == 1;
    }

    public float GetLevelProgress()
    {
        if (playerLevel >= currentSeason.MaxLevel) return 1f;
        long required = currentSeason.GetXpRequiredForLevel(playerLevel + 1);
        return required > 0 ? (float)playerXp / required : 0f;
    }
}
```

## 三、赛季任务系统

```csharp
/// <summary>
/// 赛季任务类型
/// </summary>
public enum SeasonChallengeType
{
    Daily,          // 每日任务（每天刷新）
    Weekly,         // 每周任务（每周一刷新）
    SeasonLong,     // 赛季任务（贯穿整个赛季）
    Special         // 特殊活动任务
}

/// <summary>
/// 赛季任务数据
/// </summary>
public class SeasonChallenge
{
    public string ChallengeId;
    public string Description;
    public SeasonChallengeType Type;
    public int Target;              // 目标数量
    public int Progress;            // 当前进度
    public int XpReward;            // 完成奖励（赛季XP）
    public bool IsCompleted => Progress >= Target;
    public float ProgressRatio => (float)Progress / Target;
}

/// <summary>
/// 赛季任务管理器
/// </summary>
public class SeasonChallengeManager : MonoBehaviour
{
    private List<SeasonChallenge> activeChallenges = new List<SeasonChallenge>();

    /// <summary>
    /// 更新任务进度
    /// </summary>
    public void UpdateProgress(string eventType, int amount = 1)
    {
        foreach (var challenge in activeChallenges)
        {
            if (!challenge.IsCompleted && MatchesChallenge(challenge, eventType))
            {
                challenge.Progress = Mathf.Min(challenge.Progress + amount, challenge.Target);
                
                if (challenge.IsCompleted)
                    CompleteChallengeReward(challenge);
            }
        }
    }

    bool MatchesChallenge(SeasonChallenge challenge, string eventType)
    {
        // 根据任务ID和事件类型匹配
        return challenge.ChallengeId.StartsWith(eventType);
    }

    void CompleteChallengeReward(SeasonChallenge challenge)
    {
        SeasonManager.Instance?.AddXp(challenge.XpReward, challenge.ChallengeId);
        Debug.Log($"[Season] Challenge completed: {challenge.Description}, XP: {challenge.XpReward}");
    }
}
```

## 四、Battle Pass 商业化设计

| 级别 | 价格（人民币）| 内容 |
|------|-------------|------|
| 免费 | ¥0 | 基础奖励路线（约1/3奖励）|
| 标准通行证 | ¥68 | 全部奖励 + 皮肤 |
| 超级通行证 | ¥128 | 全部奖励 + 独占皮肤 + 20%XP加速 |

**运营建议：**
1. 设计「拆包感」——每级都有至少一个让玩家期待的物品
2. 最终皮肤要在第80-90级（给玩家努力方向）
3. 每日任务让休闲玩家也能完成赛季
4. 通行证回本设计（付费货币奖励≥通行证价格，鼓励连续购买）
5. 赛季结束前1周提醒玩家完成任务（召回流失用户）
