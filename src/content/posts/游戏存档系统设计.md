## 存档系统的设计难点

```
存档系统看起来简单，实际上有很多坑：

1. 版本兼容：新版本改了数据结构，旧版本的存档怎么读？
2. 数据安全：玩家修改存档怎么办？
3. 云端同步：多设备存档冲突如何处理？
4. 性能：大量数据的序列化/反序列化性能
5. 容错：存档损坏时如何恢复？
6. 测试：如何快速构造各种存档状态进行测试？
```

## 一、序列化方案对比

### 1.1 主流序列化方案

```
JSON（推荐：开发阶段）：
  优点：可读性好，易于调试，对版本变化容错好
  缺点：文件较大，序列化速度较慢
  适用：PlayerPrefs 替代，配置存档

Binary（推荐：生产）：
  优点：体积小（约 JSON 的 1/3），速度快
  缺点：人类不可读，版本变化容错差
  适用：性能敏感的大型存档

MessagePack（推荐：平衡方案）：
  优点：比 JSON 小，比 Binary 可维护，有版本容错
  缺点：需要第三方库
  适用：商业项目首选

Protobuf：
  优点：跨语言，版本兼容性极好
  缺点：需要预先定义 .proto 文件，使用相对复杂
  适用：需要客户端/服务器共享数据结构
```

### 1.2 JSON 序列化实现

```csharp
using System;
using System.IO;
using Newtonsoft.Json; // 推荐 Newtonsoft.Json，比 Unity JsonUtility 功能更强

[System.Serializable]
public class PlayerSaveData
{
    public int Level { get; set; } = 1;
    public int Experience { get; set; } = 0;
    public long Gold { get; set; } = 100;
    public string CurrentScene { get; set; } = "StartScene";
    public SerializableVector3 Position { get; set; } = SerializableVector3.Zero;
    public List<ItemSaveData> Inventory { get; set; } = new();
    public Dictionary<int, QuestProgress> Quests { get; set; } = new();
    public long SaveTimestamp { get; set; }
    public int SaveVersion { get; set; } = 1; // 用于版本迁移
}

[System.Serializable]
public struct SerializableVector3
{
    public float X, Y, Z;
    public static SerializableVector3 Zero => new() { X = 0, Y = 0, Z = 0 };
    
    // 与 Unity Vector3 互转
    public static implicit operator Vector3(SerializableVector3 v) 
        => new(v.X, v.Y, v.Z);
    public static implicit operator SerializableVector3(Vector3 v) 
        => new() { X = v.x, Y = v.y, Z = v.z };
}

public static class SaveSerializer
{
    private static readonly JsonSerializerSettings Settings = new()
    {
        TypeNameHandling = TypeNameHandling.Auto,    // 支持多态序列化
        NullValueHandling = NullValueHandling.Ignore, // 忽略 null，减小体积
        Formatting = Formatting.None,                 // 不格式化，减小体积
        MissingMemberHandling = MissingMemberHandling.Ignore // 忽略多余字段（向后兼容）
    };
    
    public static string Serialize<T>(T data)
    {
        return JsonConvert.SerializeObject(data, Settings);
    }
    
    public static T Deserialize<T>(string json)
    {
        return JsonConvert.DeserializeObject<T>(json, Settings);
    }
}
```

## 二、存档管理器设计

### 2.1 完整存档系统

```csharp
/// <summary>
/// 存档管理器：负责存档的读写、加密、版本迁移
/// </summary>
public class SaveSystem : MonoBehaviour
{
    private const string SAVE_FOLDER = "Saves";
    private const string BACKUP_SUFFIX = ".bak";
    private const string ENCRYPTION_KEY = "YourGameSecretKey2024"; // 生产环境应从服务器获取
    
    private string SaveDirectory => Path.Combine(Application.persistentDataPath, SAVE_FOLDER);
    
    void Awake()
    {
        Directory.CreateDirectory(SaveDirectory);
    }
    
    /// <summary>
    /// 保存存档
    /// </summary>
    public async UniTask SaveAsync(int slot, PlayerSaveData data, CancellationToken ct = default)
    {
        data.SaveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        
        string path = GetSavePath(slot);
        string backupPath = path + BACKUP_SUFFIX;
        
        try
        {
            string json = SaveSerializer.Serialize(data);
            string encrypted = EncryptionHelper.Encrypt(json, ENCRYPTION_KEY);
            
            // 先写入临时文件，再原子替换（防止写入中断导致存档损坏）
            string tempPath = path + ".tmp";
            await File.WriteAllTextAsync(tempPath, encrypted, ct);
            
            // 备份旧存档
            if (File.Exists(path))
                File.Copy(path, backupPath, overwrite: true);
            
            // 原子替换
            File.Move(tempPath, path, overwrite: true);
            
            Debug.Log($"[Save] Slot {slot} saved successfully");
        }
        catch (Exception e)
        {
            Debug.LogError($"[Save] Failed to save slot {slot}: {e.Message}");
            throw;
        }
    }
    
    /// <summary>
    /// 加载存档（带自动修复）
    /// </summary>
    public async UniTask<SaveResult> LoadAsync(int slot, CancellationToken ct = default)
    {
        string path = GetSavePath(slot);
        
        // 优先加载主存档
        var result = await TryLoadFile(path, ct);
        if (result.Success) return result;
        
        // 主存档失败，尝试备份
        string backupPath = path + BACKUP_SUFFIX;
        if (File.Exists(backupPath))
        {
            Debug.LogWarning($"[Save] Main save corrupted, loading backup for slot {slot}");
            result = await TryLoadFile(backupPath, ct);
            if (result.Success) return result;
        }
        
        // 都失败，返回空结果
        Debug.LogError($"[Save] Failed to load slot {slot}");
        return SaveResult.Failure("存档损坏，无法加载");
    }
    
    private async UniTask<SaveResult> TryLoadFile(string path, CancellationToken ct)
    {
        if (!File.Exists(path)) 
            return SaveResult.Failure("文件不存在");
        
        try
        {
            string encrypted = await File.ReadAllTextAsync(path, ct);
            string json = EncryptionHelper.Decrypt(encrypted, ENCRYPTION_KEY);
            
            // 验证完整性（检查哈希）
            // var hash = EncryptionHelper.ComputeHash(json);
            // ...
            
            var data = SaveSerializer.Deserialize<PlayerSaveData>(json);
            
            // 版本迁移
            data = MigrateIfNeeded(data);
            
            return SaveResult.Ok(data);
        }
        catch (Exception e)
        {
            Debug.LogError($"[Save] Load error: {e.Message}");
            return SaveResult.Failure(e.Message);
        }
    }
    
    /// <summary>
    /// 版本迁移：处理旧版本存档
    /// </summary>
    private PlayerSaveData MigrateIfNeeded(PlayerSaveData data)
    {
        // 从版本 1 迁移到版本 2
        if (data.SaveVersion < 2)
        {
            // 例：v2 增加了新字段 PlayerTitle，旧存档没有这个字段
            // 由于 MissingMemberHandling.Ignore，新字段会保持默认值
            // 如果需要特殊初始化，在这里处理
            data.SaveVersion = 2;
        }
        
        // 从版本 2 迁移到版本 3
        if (data.SaveVersion < 3)
        {
            // 例：v3 把 Gold (int) 改成了 Gold (long)
            // 因为类型改变，需要迁移处理
            // 旧的 int Gold 字段已经被 long Gold 替换，序列化自动转换
            data.SaveVersion = 3;
        }
        
        return data;
    }
    
    private string GetSavePath(int slot) 
        => Path.Combine(SaveDirectory, $"save_{slot:D3}.dat");
}

public class SaveResult
{
    public bool Success { get; private set; }
    public PlayerSaveData Data { get; private set; }
    public string ErrorMessage { get; private set; }
    
    public static SaveResult Ok(PlayerSaveData data) 
        => new() { Success = true, Data = data };
    
    public static SaveResult Failure(string message) 
        => new() { Success = false, ErrorMessage = message };
}
```

### 2.2 加密工具

```csharp
using System.Security.Cryptography;

public static class EncryptionHelper
{
    /// <summary>
    /// AES 加密（简单防修改，非高安全需求）
    /// </summary>
    public static string Encrypt(string plainText, string key)
    {
        byte[] keyBytes = DeriveKey(key);
        
        using var aes = Aes.Create();
        aes.Key = keyBytes;
        aes.GenerateIV();
        
        using var encryptor = aes.CreateEncryptor();
        byte[] plainBytes = System.Text.Encoding.UTF8.GetBytes(plainText);
        byte[] cipherBytes = encryptor.TransformFinalBlock(plainBytes, 0, plainBytes.Length);
        
        // 将 IV 和密文合并
        byte[] result = new byte[aes.IV.Length + cipherBytes.Length];
        Buffer.BlockCopy(aes.IV, 0, result, 0, aes.IV.Length);
        Buffer.BlockCopy(cipherBytes, 0, result, aes.IV.Length, cipherBytes.Length);
        
        return Convert.ToBase64String(result);
    }
    
    public static string Decrypt(string cipherText, string key)
    {
        byte[] keyBytes = DeriveKey(key);
        byte[] fullBytes = Convert.FromBase64String(cipherText);
        
        using var aes = Aes.Create();
        aes.Key = keyBytes;
        
        byte[] iv = new byte[aes.BlockSize / 8];
        byte[] cipherBytes = new byte[fullBytes.Length - iv.Length];
        Buffer.BlockCopy(fullBytes, 0, iv, 0, iv.Length);
        Buffer.BlockCopy(fullBytes, iv.Length, cipherBytes, 0, cipherBytes.Length);
        
        aes.IV = iv;
        using var decryptor = aes.CreateDecryptor();
        byte[] plainBytes = decryptor.TransformFinalBlock(cipherBytes, 0, cipherBytes.Length);
        
        return System.Text.Encoding.UTF8.GetString(plainBytes);
    }
    
    private static byte[] DeriveKey(string password)
    {
        // 使用 PBKDF2 从密码派生密钥
        using var rfc = new Rfc2898DeriveBytes(password, 
            salt: new byte[] { 0x1F, 0x2A, 0x3B, 0x4C, 0x5D, 0x6E, 0x7F, 0x80 }, 
            iterations: 1000, 
            hashAlgorithm: HashAlgorithmName.SHA256);
        return rfc.GetBytes(32); // 256位密钥
    }
    
    /// <summary>
    /// 计算数据哈希（用于完整性验证）
    /// </summary>
    public static string ComputeHash(string data)
    {
        using var sha = SHA256.Create();
        byte[] bytes = System.Text.Encoding.UTF8.GetBytes(data);
        byte[] hash = sha.ComputeHash(bytes);
        return Convert.ToBase64String(hash);
    }
}
```

## 三、云存档同步

### 3.1 冲突解决策略

```
多设备存档冲突场景：
  - 玩家在手机上玩了一会，存档到云端
  - 切换到 PC，加载的是旧的云存档，玩了一会
  - 再切回手机，云端有 PC 的进度，本地有手机的进度

冲突解决策略：
  1. 时间戳优先：最新时间戳的存档获胜（最简单）
  2. 关键数据合并：取最高等级、最多金币等（复杂但公平）
  3. 用户选择：弹出提示让玩家选择（最安全但打断体验）
  4. 服务器权威：服务端验证所有数据（防作弊，但需要后端）
```

```csharp
/// <summary>
/// 云存档管理（基于时间戳的简单策略）
/// </summary>
public class CloudSaveManager : MonoBehaviour
{
    /// <summary>
    /// 同步存档到云端
    /// </summary>
    public async UniTask SyncAsync(int slot, PlayerSaveData localData, CancellationToken ct)
    {
        try
        {
            // 获取云端存档信息（轻量元数据）
            var cloudMeta = await FetchCloudMeta(slot, ct);
            
            if (cloudMeta == null)
            {
                // 云端没有存档，直接上传
                await UploadSave(slot, localData, ct);
                return;
            }
            
            if (localData.SaveTimestamp > cloudMeta.Timestamp)
            {
                // 本地更新，上传到云端
                await UploadSave(slot, localData, ct);
            }
            else if (localData.SaveTimestamp < cloudMeta.Timestamp)
            {
                // 云端更新，下载覆盖本地
                var cloudData = await DownloadSave(slot, ct);
                if (cloudData != null)
                    await _localSave.SaveAsync(slot, cloudData, ct);
            }
            // 时间戳相同：不需要同步
        }
        catch (Exception e)
        {
            Debug.LogError($"[Cloud Save] Sync failed: {e.Message}");
            // 云端同步失败不影响本地游玩
        }
    }
    
    private async UniTask<CloudSaveMeta> FetchCloudMeta(int slot, CancellationToken ct)
    {
        // 实际项目中调用后端 API
        // var response = await HttpClient.GetAsync($"/api/save/{slot}/meta", ct);
        // return response.IsSuccess ? JsonConvert.DeserializeObject<CloudSaveMeta>(response.Body) : null;
        return null; // 占位
    }
    
    private async UniTask UploadSave(int slot, PlayerSaveData data, CancellationToken ct)
    {
        // 调用后端 API 上传
        Debug.Log($"[Cloud Save] Uploading slot {slot}...");
        await UniTask.Delay(500, cancellationToken: ct); // 模拟网络请求

    {

    }

}
```

## 四、自动存档与存档点设计

### 4.1 自动存档系统

```csharp
/// <summary>
/// 自动存档管理器
/// </summary>
public class AutoSaveManager : MonoBehaviour
{
    [SerializeField] private float _autoSaveInterval = 300f; // 5分钟自动存档
    [SerializeField] private SaveSystem _saveSystem;
    
    private float _nextAutoSaveTime;
    private bool _isSaving = false;
    
    void Start()
    {
        _nextAutoSaveTime = Time.time + _autoSaveInterval;
        
        // 注册触发事件
        EventBus.Subscribe<LevelCompletedEvent>(OnLevelCompleted);
        EventBus.Subscribe<AreaTransitionEvent>(OnAreaTransition);
        EventBus.Subscribe<ImportantItemPickedEvent>(OnImportantItemPicked);
    }
    
    void Update()
    {
        // 定时自动存档
        if (Time.time >= _nextAutoSaveTime && !_isSaving)
        {
            _nextAutoSaveTime = Time.time + _autoSaveInterval;
            AutoSave().Forget();
        }
    }
    
    private void OnLevelCompleted(LevelCompletedEvent e) => AutoSave().Forget();
    private void OnAreaTransition(AreaTransitionEvent e) => AutoSave().Forget();
    private void OnImportantItemPicked(ImportantItemPickedEvent e) => AutoSave().Forget();
    
    private async UniTaskVoid AutoSave()
    {
        if (_isSaving) return;
        _isSaving = true;
        
        try
        {
            var saveData = GameStateCollector.CollectCurrentState();
            await _saveSystem.SaveAsync(slot: 0, saveData); // slot 0 = 自动存档槽
            
            // 显示"自动存档中..."提示
            UIManager.ShowToast("游戏已自动保存");
        }
        catch (Exception e)
        {
            Debug.LogError($"[AutoSave] Failed: {e.Message}");
        }
        finally
        {
            _isSaving = false;
        }
    }
    
    void OnApplicationPause(bool paused)
    {
        if (paused) AutoSave().Forget(); // 切到后台时强制存档
    }
    
    void OnApplicationQuit()
    {
        // 游戏退出时同步存档（注意：异步需要特殊处理）
        var saveData = GameStateCollector.CollectCurrentState();
        var json = SaveSerializer.Serialize(saveData);
        File.WriteAllText(GetSavePath(0), json); // 同步写入
    }
}
```

```csharp
[TestFixture]
public class SaveSystemTests
{
    private SaveSystem _saveSystem;

        {
            Level = 15,
            Gold = 9999,
            CurrentScene = "DungeonLevel2"
        };
        
        await _saveSystem.SaveAsync(slot: 0, original);
        var result = await _saveSystem.LoadAsync(slot: 0);

        Assert.AreEqual(15, result.Data.Level);

    {

        await _saveSystem.SaveAsync(0, original);
        
        // 损坏主存档
        string path = _saveSystem.GetSavePath(0);
        await File.WriteAllTextAsync(path, "CORRUPTED_DATA");
        
        // 应该加载备份
        var result = await _saveSystem.LoadAsync(0);
        
        Assert.IsTrue(result.Success);
        Assert.AreEqual(10, result.Data.Level);

    public async Task Migration_OldVersion_UpdatesToLatest()

        // 模拟 v1 存档

        await _saveSystem.SaveAsync(0, oldData);
        var result = await _saveSystem.LoadAsync(0);
        
        // 加载后版本应该被迁移到最新版本
        Assert.AreEqual(SaveSystem.CURRENT_VERSION, result.Data.SaveVersion);
    }
```

## 总结

存档系统设计要点：

| 方面 | 建议 |
|------|------|
| 序列化格式 | JSON（开发期）→ MessagePack（发布期） |
| 安全性 | 加密 + 完整性校验（哈希） |
| 版本兼容 | 版本号 + 迁移函数，每次数据结构变化增加版本 |
| 容灾 | 主存档 + 备份，原子写入（先写临时文件） |
| 云同步 | 时间戳比对，失败不影响本地游玩 |
| 自动存档 | 定时 + 事件触发（关卡完成、区域切换） |

**技术负责人的关注点**：存档系统是游戏的"命脉"，一旦出问题就是重大事故。投入足够的测试覆盖，尤其是异常场景（断电、写入失败、版本迁移）。
