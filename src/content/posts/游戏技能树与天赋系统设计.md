## 一、技能树数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 天赋节点数据（ScriptableObject配置）
/// </summary>
[CreateAssetMenu(fileName = "TalentNode", menuName = "Game/Talent Node")]
public class TalentNodeData : ScriptableObject
{
    [Header("基础信息")]
    public string NodeId;
    public string DisplayName;
    [TextArea(2, 4)] public string Description;
    public Sprite Icon;
    
    [Header("等级配置")]
    public int MaxLevel = 1;            // 最大投入点数
    public int[] CostPerLevel;          // 每级消耗天赋点（默认每级1点）
    
    [Header("前置条件")]
    public string[] RequiredNodeIds;    // 前置天赋（全部满足才可点）
    public int[] RequiredNodeLevels;    // 前置天赋所需等级
    public int RequiredCharacterLevel;  // 需要角色等级
    
    [Header("互斥配置")]
    public string[] MutexGroupIds;      // 同组内只能选一个（互斥组）
    
    [Header("效果配置")]
    public TalentEffect[] Effects;      // 天赋效果列表
    
    [Header("UI布局")]
    public Vector2Int GridPosition;     // 在天赋树中的网格位置
}

/// <summary>
/// 天赋效果
/// </summary>
[Serializable]
public class TalentEffect
{
    public string StatId;           // 影响的属性ID（如 "attack_damage"）
    public float[] ValuesPerLevel;  // 每级的效果值
    public bool IsPercentage;       // 是否为百分比加成
    
    public float GetValue(int level)
    {
        if (level <= 0 || ValuesPerLevel == null || ValuesPerLevel.Length == 0)
            return 0f;
        int idx = Mathf.Clamp(level - 1, 0, ValuesPerLevel.Length - 1);
        return ValuesPerLevel[idx];
    }
}

/// <summary>
/// 玩家天赋状态（运行时数据）
/// </summary>
[Serializable]
public class TalentState
{
    public Dictionary<string, int> NodeLevels = new Dictionary<string, int>();
    public int RemainingPoints;
    
    public int GetLevel(string nodeId) =>
        NodeLevels.TryGetValue(nodeId, out int level) ? level : 0;
}
```

## 二、天赋树管理器

```csharp
/// <summary>
/// 天赋树管理器
/// </summary>
public class TalentTreeManager : MonoBehaviour
{
    private static TalentTreeManager instance;
    public static TalentTreeManager Instance => instance;

    [SerializeField] private TalentNodeData[] allNodes;      // 所有天赋节点配置
    [SerializeField] private int pointsPerLevel = 1;          // 每升一级获得的天赋点数
    
    private Dictionary<string, TalentNodeData> nodeMap 
        = new Dictionary<string, TalentNodeData>();
    private TalentState state = new TalentState();
    
    public event Action<string, int> OnNodeLevelChanged;    // nodeId, newLevel
    public event Action<int> OnPointsChanged;               // remainingPoints

    void Awake()
    {
        instance = this;
        foreach (var node in allNodes)
            nodeMap[node.NodeId] = node;
    }

    public void Initialize(TalentState savedState, int characterLevel)
    {
        state = savedState ?? new TalentState();
        
        // 根据角色等级计算总可用天赋点
        int totalEarned = characterLevel * pointsPerLevel;
        int totalSpent = 0;
        foreach (var kv in state.NodeLevels)
            for (int i = 1; i <= kv.Value; i++)
                totalSpent += GetNodeCost(kv.Key, i);
        
        state.RemainingPoints = totalEarned - totalSpent;
    }

    /// <summary>
    /// 尝试提升天赋节点等级
    /// </summary>
    public bool TryLevelUp(string nodeId)
    {
        if (!nodeMap.TryGetValue(nodeId, out var node))
        {
            Debug.LogWarning($"[Talent] Node not found: {nodeId}");
            return false;
        }
        
        int currentLevel = state.GetLevel(nodeId);
        
        // 检查是否已满级
        if (currentLevel >= node.MaxLevel)
        {
            Debug.Log($"[Talent] Node '{nodeId}' is already max level");
            return false;
        }
        
        // 检查天赋点是否足够
        int cost = GetNodeCost(nodeId, currentLevel + 1);
        if (state.RemainingPoints < cost)
        {
            Debug.Log($"[Talent] Not enough points: need {cost}, have {state.RemainingPoints}");
            return false;
        }
        
        // 检查前置条件
        if (!CheckPrerequisites(node))
            return false;
        
        // 检查互斥
        if (!CheckMutexGroup(node))
            return false;
        
        // 执行升级
        state.NodeLevels[nodeId] = currentLevel + 1;
        state.RemainingPoints -= cost;
        
        OnNodeLevelChanged?.Invoke(nodeId, state.NodeLevels[nodeId]);
        OnPointsChanged?.Invoke(state.RemainingPoints);
        
        return true;
    }

    /// <summary>
    /// 重置天赋（退点并返还所有天赋点）
    /// </summary>
    public void ResetAll()
    {
        int refunded = 0;
        foreach (var kv in state.NodeLevels)
        {
            for (int i = 1; i <= kv.Value; i++)
                refunded += GetNodeCost(kv.Key, i);
        }
        
        state.NodeLevels.Clear();
        state.RemainingPoints += refunded;
        
        foreach (var node in allNodes)
            OnNodeLevelChanged?.Invoke(node.NodeId, 0);
        OnPointsChanged?.Invoke(state.RemainingPoints);
    }

    bool CheckPrerequisites(TalentNodeData node)
    {
        if (node.RequiredNodeIds == null) return true;
        
        for (int i = 0; i < node.RequiredNodeIds.Length; i++)
        {
            string reqId = node.RequiredNodeIds[i];
            int reqLevel = node.RequiredNodeLevels != null && i < node.RequiredNodeLevels.Length
                ? node.RequiredNodeLevels[i] : 1;
            
            if (state.GetLevel(reqId) < reqLevel)
            {
                Debug.Log($"[Talent] Prerequisite not met: {reqId} >= {reqLevel}");
                return false;
            }
        }
        return true;
    }

    bool CheckMutexGroup(TalentNodeData node)
    {
        if (node.MutexGroupIds == null || node.MutexGroupIds.Length == 0) 
            return true;
        
        // 检查同互斥组的其他节点是否已经点了
        foreach (var otherNode in allNodes)
        {
            if (otherNode.NodeId == node.NodeId) continue;
            if (otherNode.MutexGroupIds == null) continue;
            
            foreach (var group in otherNode.MutexGroupIds)
            {
                if (System.Array.Exists(node.MutexGroupIds, g => g == group))
                {
                    if (state.GetLevel(otherNode.NodeId) > 0)
                    {
                        Debug.Log($"[Talent] Mutex conflict: {otherNode.NodeId} is already selected");
                        return false;
                    }
                }
            }
        }
        return true;
    }

    int GetNodeCost(string nodeId, int toLevel)
    {
        if (!nodeMap.TryGetValue(nodeId, out var node)) return 1;
        if (node.CostPerLevel != null && toLevel <= node.CostPerLevel.Length)
            return node.CostPerLevel[toLevel - 1];
        return 1;
    }

    /// <summary>
    /// 计算当前所有天赋效果的总加成
    /// </summary>
    public Dictionary<string, float> CalculateTotalBonuses()
    {
        var bonuses = new Dictionary<string, float>();
        
        foreach (var kv in state.NodeLevels)
        {
            if (!nodeMap.TryGetValue(kv.Key, out var node)) continue;
            if (node.Effects == null) continue;
            
            foreach (var effect in node.Effects)
            {
                float value = effect.GetValue(kv.Value);
                if (!bonuses.ContainsKey(effect.StatId))
                    bonuses[effect.StatId] = 0f;
                
                bonuses[effect.StatId] += value;
            }
        }
        
        return bonuses;
    }

    public TalentState GetCurrentState() => state;
    public int RemainingPoints => state.RemainingPoints;
    public int GetNodeLevel(string nodeId) => state.GetLevel(nodeId);
    
    public bool CanLevelUp(string nodeId)
    {
        if (!nodeMap.TryGetValue(nodeId, out var node)) return false;
        int currentLevel = state.GetLevel(nodeId);
        if (currentLevel >= node.MaxLevel) return false;
        int cost = GetNodeCost(nodeId, currentLevel + 1);
        if (state.RemainingPoints < cost) return false;
        if (!CheckPrerequisites(node)) return false;
        if (!CheckMutexGroup(node)) return false;
        return true;
    }
}
```

## 三、天赋UI节点

```csharp
/// <summary>
/// 天赋树UI节点
/// </summary>
public class TalentNodeUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Image iconImage;
    [SerializeField] private UnityEngine.UI.Text levelText;
    [SerializeField] private UnityEngine.UI.Image[] levelDots;     // 点数显示小圆点
    [SerializeField] private UnityEngine.UI.Button button;
    [SerializeField] private GameObject lockedOverlay;             // 锁定遮罩
    [SerializeField] private GameObject maxLevelGlow;              // 满级特效
    [SerializeField] private UnityEngine.UI.Image borderImage;
    
    [Header("颜色配置")]
    [SerializeField] private Color lockedColor = Color.gray;
    [SerializeField] private Color availableColor = Color.white;
    [SerializeField] private Color investedColor = new Color(1f, 0.8f, 0.2f);
    [SerializeField] private Color maxLevelColor = Color.cyan;

    private string nodeId;
    private TalentNodeData data;

    public void Setup(TalentNodeData nodeData)
    {
        data = nodeData;
        nodeId = nodeData.NodeId;
        
        iconImage.sprite = nodeData.Icon;
        
        button.onClick.RemoveAllListeners();
        button.onClick.AddListener(OnClick);
        
        // 订阅天赋变化事件
        TalentTreeManager.Instance.OnNodeLevelChanged += OnNodeLevelChanged;
        
        UpdateDisplay();
    }

    void OnDestroy()
    {
        if (TalentTreeManager.Instance != null)
            TalentTreeManager.Instance.OnNodeLevelChanged -= OnNodeLevelChanged;
    }

    void OnNodeLevelChanged(string changedNodeId, int newLevel)
    {
        // 可能影响到当前节点的可用性（前置条件变化）
        UpdateDisplay();
    }

    void UpdateDisplay()
    {
        int currentLevel = TalentTreeManager.Instance.GetNodeLevel(nodeId);
        bool canLevelUp = TalentTreeManager.Instance.CanLevelUp(nodeId);
        bool isMaxLevel = currentLevel >= data.MaxLevel;
        
        // 等级小圆点
        for (int i = 0; i < levelDots.Length; i++)
        {
            if (i < data.MaxLevel)
            {
                levelDots[i].gameObject.SetActive(true);
                levelDots[i].color = i < currentLevel ? 
                    investedColor : Color.gray;
            }
            else
            {
                levelDots[i].gameObject.SetActive(false);
            }
        }
        
        // 等级文本
        levelText.text = data.MaxLevel > 1 ? $"{currentLevel}/{data.MaxLevel}" : "";
        
        // 状态显示
        lockedOverlay.SetActive(currentLevel == 0 && !canLevelUp);
        maxLevelGlow.SetActive(isMaxLevel);
        
        // 边框颜色
        if (isMaxLevel)
            borderImage.color = maxLevelColor;
        else if (currentLevel > 0)
            borderImage.color = investedColor;
        else if (canLevelUp)
            borderImage.color = availableColor;
        else
            borderImage.color = lockedColor;
        
        button.interactable = canLevelUp || currentLevel > 0;
    }

    void OnClick()
    {
        if (TalentTreeManager.Instance.TryLevelUp(nodeId))
        {
            // 升级成功，播放动画
            transform.DOPunchScale(Vector3.one * 0.15f, 0.3f);
        }
        else
        {
            // 升级失败，显示提示
            UIManager.Instance?.ShowTooltip(GetTooltipText(), transform.position);
        }
    }

    string GetTooltipText()
    {
        int currentLevel = TalentTreeManager.Instance.GetNodeLevel(nodeId);
        if (currentLevel >= data.MaxLevel) return "已达到最高等级";
        
        // 生成包含效果描述的提示文本
        var sb = new System.Text.StringBuilder();
        sb.AppendLine($"<b>{data.DisplayName}</b>");
        sb.AppendLine(data.Description);
        
        if (data.Effects != null)
        {
            sb.AppendLine();
            foreach (var effect in data.Effects)
            {
                float nextValue = effect.GetValue(currentLevel + 1);
                string suffix = effect.IsPercentage ? "%" : "";
                sb.AppendLine($"+{nextValue}{suffix} {effect.StatId}");
            }
        }
        
        return sb.ToString();
    }
}
```

## 四、经典天赋树设计模式

| 模式 | 代表作 | 特点 |
|------|--------|------|
| 线性天赋树 | 暗黑破坏神2 | 每条路线独立，简单直观 |
| 蜘蛛网天赋 | 暗黑破坏神3（抓技能格）| 更大自由度，复杂度高 |
| 天赋节点Web | 流放之路 | 极高自由度，数百节点 |
| 职业专精 | 魔兽世界天赋 | 三条专精路线选一 |
| 天赋点+互斥 | 《哈迪斯》礼品 | 稀缺性决策，每局不同 |

**设计原则：**
1. 前置条件层数不要超过5层（太深则难以理解）
2. 互斥选择增加决策深度（两个好天赋只能选一个）
3. 末级天赋要有「让人感觉值得」的强力效果
4. 支持预览效果数值（让玩家在点之前知道会加多少）
5. 支持退点/重置（严格控制成本，不然玩家不敢点）
