## 一、经验管理器

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

public class ExperienceManager : MonoBehaviour
{
    private static ExperienceManager instance;
    public static ExperienceManager Instance => instance;

    [Header("等级配置")]
    [SerializeField] private int maxLevel = 100;
    [SerializeField] private AnimationCurve expCurve;  // 经验需求曲线
    [SerializeField] private float baseExp = 100f;
    [SerializeField] private float expMultiplier = 1.2f;

    private long currentExp;
    private int currentLevel;
    private float[] expTable; // 预计算升级所需经验

    public event Action<int, int> OnLevelUp;  // oldLevel, newLevel
    public event Action<long, long> OnExpChanged;  // oldExp, newExp

    void Awake()
    {
        instance = this;
        BuildExpTable();
    }

    void BuildExpTable()
    {
        expTable = new float[maxLevel + 1];
        expTable[0] = 0;
        for (int i = 1; i <= maxLevel; i++)
        {
            // 等差+等比混合曲线
            expTable[i] = expTable[i - 1] + 
                Mathf.RoundToInt(baseExp * Mathf.Pow(expMultiplier, i - 1));
        }
    }

    public void AddExp(long amount, string source = "unknown", float bonusMultiplier = 1f)
    {
        long actualAmount = Mathf.RoundToInt(amount * bonusMultiplier);
        long oldExp = currentExp;
        int oldLevel = currentLevel;
        
        currentExp += actualAmount;
        
        // 检查升级
        while (currentLevel < maxLevel && currentExp >= GetExpForNextLevel())
        {
            currentLevel++;
            OnLevelReached(currentLevel);
        }
        
        if (currentLevel > oldLevel)
            OnLevelUp?.Invoke(oldLevel, currentLevel);
        
        OnExpChanged?.Invoke(oldExp, currentExp);
    }

    void OnLevelReached(int level)
    {
        Debug.Log($"[Level] 升级到 {level}！");
        
        // 解锁内容
        UnlockContentForLevel(level);
    }

    void UnlockContentForLevel(int level)
    {
        // 可扩展：解锁技能/副本/功能
        switch (level)
        {
            case 10: UIManager.Instance?.ShowToast("解锁副本系统！"); break;
            case 20: UIManager.Instance?.ShowToast("解锁公会功能！"); break;
            case 30: UIManager.Instance?.ShowToast("解锁竞技场！"); break;
        }
    }

    public long GetExpForLevel(int level)
    {
        if (level < 0 || level >= expTable.Length) return 0;
        return (long)expTable[level];
    }

    public long GetExpForNextLevel()
    {
        return currentLevel >= maxLevel ? long.MaxValue : 
            (long)expTable[currentLevel + 1];
    }

    public float GetLevelProgress()
    {
        if (currentLevel >= maxLevel) return 1f;
        long thisLevel = GetExpForLevel(currentLevel);
        long nextLevel = GetExpForNextLevel();
        if (nextLevel <= thisLevel) return 1f;
        return (float)(currentExp - thisLevel) / (nextLevel - thisLevel);
    }

    public int GetLevel() => currentLevel;
    public long GetExp() => currentExp;
}
```

## 二、经验加成叠加器

```csharp
public class ExpBoostManager : MonoBehaviour
{
    private List<ExpBoost> activeBoosts = new List<ExpBoost>();

    [Serializable]
    public class ExpBoost
    {
        public string BoostId;
        public float Multiplier;    // 加成倍数（1.5 = +50%）
        public long ExpireTime;
    }

    public float GetTotalMultiplier()
    {
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        float total = 1f;
        
        activeBoosts.RemoveAll(b => b.ExpireTime > 0 && b.ExpireTime < now);
        
        foreach (var boost in activeBoosts)
            total *= boost.Multiplier;
        
        // VIP加成
        if (VIPManager.Instance != null)
            total += VIPManager.Instance.GetPrivilegeValue(PrivilegeType.ExpBonus) / 100f;
        
        return total;
    }

    public void AddBoost(string id, float multiplier, float durationSeconds)
    {
        activeBoosts.RemoveAll(b => b.BoostId == id);
        activeBoosts.Add(new ExpBoost
        {
            BoostId = id,
            Multiplier = multiplier,
            ExpireTime = durationSeconds > 0 ? 
                DateTimeOffset.UtcNow.ToUnixTimeSeconds() + (long)durationSeconds : 0
        });
    }
}
```

## 三、经验成长曲线设计

| 等级段 | 升级所需经验 | 设计意图 |
|--------|-------------|---------|
| 1-10 | 较少（100-500）| 快速成长，建立正反馈 |
| 11-30 | 中等（500-2000）| 稳定成长，熟悉系统 |
| 31-60 | 较多（2000-8000）| 长期目标，维持活跃 |
| 61-100 | 很多（8000+）| 核心玩家目标 |

**关键设计：**
- 每日任务经验约 = 3-5级所需经验的 1/3
- 即活跃玩家3天可升1级（中期）
