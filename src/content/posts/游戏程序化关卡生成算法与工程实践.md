## 一、程序化生成的价值

程序化关卡生成（PCG）的主要优势：

- **无限内容**：每次游戏体验独特（Roguelike 核心机制）
- **节省人力**：减少手工制作关卡的工作量
- **可控随机**：通过种子（Seed）复现特定关卡
- **参数调优**：修改参数即可调整难度、密度

## 二、BSP 地牢生成算法

Binary Space Partitioning（二叉空间分割）是经典地牢生成算法：

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// BSP 地牢生成器
/// </summary>
public class BSPDungeonGenerator : MonoBehaviour
{
    [Header("地牢尺寸")]
    [SerializeField] private int mapWidth = 80;
    [SerializeField] private int mapHeight = 60;
    
    [Header("BSP 参数")]
    [SerializeField] private int minLeafSize = 10;   // 最小分割叶子大小
    [SerializeField] private int maxLeafSize = 24;   // 最大叶子大小
    [SerializeField] private int splitDepth = 5;     // 最大分割深度
    [SerializeField] private float splitRatio = 0.45f; // 分割比例范围
    
    [Header("房间参数")]
    [SerializeField] private int minRoomSize = 4;
    [SerializeField] private int corridorWidth = 2;
    
    [Header("生成")]
    [SerializeField] private TileType[,] tileMap;
    [SerializeField] private int randomSeed = 0;
    [SerializeField] private bool useRandomSeed = true;

    private System.Random rng;
    private List<RectInt> rooms = new List<RectInt>();

    public enum TileType { Wall, Floor, Corridor, Door }

    void Start()
    {
        GenerateDungeon();
    }

    [ContextMenu("Regenerate Dungeon")]
    public void GenerateDungeon()
    {
        if (useRandomSeed)
            randomSeed = System.Environment.TickCount;
        
        rng = new System.Random(randomSeed);
        tileMap = new TileType[mapWidth, mapHeight];
        rooms.Clear();
        
        // 初始化为全墙
        for (int y = 0; y < mapHeight; y++)
        for (int x = 0; x < mapWidth; x++)
            tileMap[x, y] = TileType.Wall;
        
        // BSP 分割
        var root = new BSPLeaf(new RectInt(1, 1, mapWidth - 2, mapHeight - 2));
        root.Split(rng, minLeafSize, maxLeafSize, splitRatio, splitDepth);
        
        // 在每个叶子中创建房间
        root.CreateRooms(rng, minRoomSize);
        
        // 收集所有房间
        root.GetRooms(rooms);
        
        // 挖出房间地板
        foreach (var room in rooms)
            CarveRoom(room);
        
        // 连接相邻房间（走廊）
        ConnectRooms(root);
        
        // 渲染地图
        RenderMap();
        
        Debug.Log($"[BSPDungeon] Generated {rooms.Count} rooms, seed={randomSeed}");
    }

    void CarveRoom(RectInt room)
    {
        for (int y = room.yMin; y < room.yMax; y++)
        for (int x = room.xMin; x < room.xMax; x++)
            SetTile(x, y, TileType.Floor);
    }

    void ConnectRooms(BSPLeaf leaf)
    {
        if (leaf.LeftChild == null || leaf.RightChild == null) return;
        
        // 递归连接子节点
        ConnectRooms(leaf.LeftChild);
        ConnectRooms(leaf.RightChild);
        
        // 连接左右子树的代表房间
        var roomA = leaf.LeftChild.GetRepresentativeRoom();
        var roomB = leaf.RightChild.GetRepresentativeRoom();
        
        if (roomA.HasValue && roomB.HasValue)
            CarveCorridorBetween(roomA.Value, roomB.Value);
    }

    void CarveCorridorBetween(RectInt roomA, RectInt roomB)
    {
        // 取两个房间的中心点
        Vector2Int centerA = new Vector2Int(
            roomA.xMin + roomA.width / 2,
            roomA.yMin + roomA.height / 2);
        
        Vector2Int centerB = new Vector2Int(
            roomB.xMin + roomB.width / 2,
            roomB.yMin + roomB.height / 2);
        
        // L形走廊（先横后竖或先竖后横）
        if (rng.NextDouble() > 0.5)
        {
            CarveHorizontalCorridor(centerA.x, centerB.x, centerA.y);
            CarveVerticalCorridor(centerA.y, centerB.y, centerB.x);
        }
        else
        {
            CarveVerticalCorridor(centerA.y, centerB.y, centerA.x);
            CarveHorizontalCorridor(centerA.x, centerB.x, centerB.y);
        }
    }

    void CarveHorizontalCorridor(int xStart, int xEnd, int y)
    {
        int minX = Mathf.Min(xStart, xEnd);
        int maxX = Mathf.Max(xStart, xEnd);
        
        for (int x = minX; x <= maxX; x++)
        for (int w = 0; w < corridorWidth; w++)
            SetTile(x, y + w, TileType.Corridor);
    }

    void CarveVerticalCorridor(int yStart, int yEnd, int x)
    {
        int minY = Mathf.Min(yStart, yEnd);
        int maxY = Mathf.Max(yStart, yEnd);
        
        for (int y = minY; y <= maxY; y++)
        for (int w = 0; w < corridorWidth; w++)
            SetTile(x + w, y, TileType.Corridor);
    }

    void SetTile(int x, int y, TileType type)
    {
        if (x < 0 || x >= mapWidth || y < 0 || y >= mapHeight) return;
        tileMap[x, y] = type;
    }

    void RenderMap()
    {
        // 清理旧的瓦片
        foreach (Transform child in transform)
            Destroy(child.gameObject);
        
        // 根据 tileMap 实例化对应的 Prefab
        // 实际项目使用 Tilemap 系统会更高效
        for (int y = 0; y < mapHeight; y++)
        for (int x = 0; x < mapWidth; x++)
        {
            if (tileMap[x, y] == TileType.Wall) continue;
            
            // 在 (x, y) 处实例化地板
            // var tile = Instantiate(floorPrefab, new Vector3(x, 0, y), Quaternion.identity, transform);
        }
    }

    public RectInt GetRandomRoom()
    {
        return rooms[rng.Next(0, rooms.Count)];
    }

    public List<RectInt> Rooms => rooms;
}

/// <summary>
/// BSP 树节点
/// </summary>
public class BSPLeaf
{
    public RectInt Bounds;
    public BSPLeaf LeftChild;
    public BSPLeaf RightChild;
    private RectInt? room;

    public BSPLeaf(RectInt bounds) { Bounds = bounds; }

    public void Split(System.Random rng, int minSize, int maxSize, 
        float ratio, int depth)
    {
        if (depth <= 0 || 
            (Bounds.width <= maxSize && Bounds.height <= maxSize &&
             rng.NextDouble() > 0.25))
            return;
        
        bool splitHorizontal = Bounds.width < Bounds.height ||
            (Bounds.width == Bounds.height && rng.NextDouble() > 0.5);
        
        int max = (splitHorizontal ? Bounds.height : Bounds.width) - minSize;
        if (max <= minSize) return;
        
        int splitAt = rng.Next(
            (int)(max * ratio), 
            (int)(max * (1f - ratio)));
        splitAt = Mathf.Max(splitAt, minSize);
        
        if (splitHorizontal)
        {
            LeftChild = new BSPLeaf(new RectInt(
                Bounds.xMin, Bounds.yMin, Bounds.width, splitAt));
            RightChild = new BSPLeaf(new RectInt(
                Bounds.xMin, Bounds.yMin + splitAt, Bounds.width, Bounds.height - splitAt));
        }
        else
        {
            LeftChild = new BSPLeaf(new RectInt(
                Bounds.xMin, Bounds.yMin, splitAt, Bounds.height));
            RightChild = new BSPLeaf(new RectInt(
                Bounds.xMin + splitAt, Bounds.yMin, Bounds.width - splitAt, Bounds.height));
        }
        
        LeftChild.Split(rng, minSize, maxSize, ratio, depth - 1);
        RightChild.Split(rng, minSize, maxSize, ratio, depth - 1);
    }

    public void CreateRooms(System.Random rng, int minRoomSize)
    {
        if (LeftChild != null && RightChild != null)
        {
            LeftChild.CreateRooms(rng, minRoomSize);
            RightChild.CreateRooms(rng, minRoomSize);
        }
        else
        {
            // 叶子节点：在 Bounds 内创建随机大小的房间
            int roomW = rng.Next(minRoomSize, Bounds.width - 1);
            int roomH = rng.Next(minRoomSize, Bounds.height - 1);
            int roomX = Bounds.xMin + rng.Next(0, Bounds.width - roomW);
            int roomY = Bounds.yMin + rng.Next(0, Bounds.height - roomH);
            
            room = new RectInt(roomX, roomY, roomW, roomH);
        }
    }

    public void GetRooms(List<RectInt> list)
    {
        if (room.HasValue) list.Add(room.Value);
        LeftChild?.GetRooms(list);
        RightChild?.GetRooms(list);
    }

    public RectInt? GetRepresentativeRoom()
    {
        if (room.HasValue) return room;
        
        var leftRoom = LeftChild?.GetRepresentativeRoom();
        var rightRoom = RightChild?.GetRepresentativeRoom();
        
        if (leftRoom.HasValue && rightRoom.HasValue)
            return leftRoom; // 随机选一个或选更大的
        
        return leftRoom ?? rightRoom;
    }
}
```

## 三、Cellular Automata 洞穴生成

```csharp
/// <summary>
/// 细胞自动机洞穴生成器
/// </summary>
public class CellularAutomataCave : MonoBehaviour
{
    [SerializeField] private int width = 80, height = 50;
    [SerializeField] private float fillProbability = 0.45f;  // 初始填充率
    [SerializeField] private int smoothIterations = 5;        // 平滑迭代次数
    [SerializeField] private int birthLimit = 4;              // 诞生邻居阈值
    [SerializeField] private int survivalLimit = 3;           // 存活邻居阈值

    private bool[,] map; // true = 墙, false = 空洞

    public void GenerateCave(int seed = 0)
    {
        var rng = new System.Random(seed);
        map = new bool[width, height];
        
        // 随机初始化
        for (int y = 0; y < height; y++)
        for (int x = 0; x < width; x++)
        {
            // 边界强制为墙
            if (x == 0 || x == width - 1 || y == 0 || y == height - 1)
                map[x, y] = true;
            else
                map[x, y] = rng.NextDouble() < fillProbability;
        }
        
        // 迭代平滑
        for (int i = 0; i < smoothIterations; i++)
            SmoothMap();
        
        // 移除孤岛（可选）
        RemoveSmallRegions(10);
    }

    void SmoothMap()
    {
        var newMap = new bool[width, height];
        
        for (int y = 1; y < height - 1; y++)
        for (int x = 1; x < width - 1; x++)
        {
            int neighbors = CountWallNeighbors(x, y);
            
            if (map[x, y])
                newMap[x, y] = neighbors >= survivalLimit;
            else
                newMap[x, y] = neighbors > birthLimit;
        }
        
        map = newMap;
    }

    int CountWallNeighbors(int cx, int cy)
    {
        int count = 0;
        for (int dy = -1; dy <= 1; dy++)
        for (int dx = -1; dx <= 1; dx++)
        {
            if (dx == 0 && dy == 0) continue;
            int nx = cx + dx, ny = cy + dy;
            if (nx < 0 || nx >= width || ny < 0 || ny >= height)
                count++; // 边界算墙
            else if (map[nx, ny])
                count++;
        }
        return count;
    }

    void RemoveSmallRegions(int minSize)
    {
        // 使用 BFS/DFS 找到所有连通区域，删除小于 minSize 的区域
        bool[,] visited = new bool[width, height];
        
        for (int y = 0; y < height; y++)
        for (int x = 0; x < width; x++)
        {
            if (!map[x, y] && !visited[x, y])
            {
                var region = FloodFill(x, y, visited);
                if (region.Count < minSize)
                {
                    foreach (var cell in region)
                        map[cell.x, cell.y] = true; // 填充小区域
                }
            }
        }
    }

    List<Vector2Int> FloodFill(int startX, int startY, bool[,] visited)
    {
        var region = new List<Vector2Int>();
        var queue = new Queue<Vector2Int>();
        queue.Enqueue(new Vector2Int(startX, startY));
        visited[startX, startY] = true;
        
        while (queue.Count > 0)
        {
            var cell = queue.Dequeue();
            region.Add(cell);
            
            Vector2Int[] neighbors = {
                new Vector2Int(cell.x - 1, cell.y),
                new Vector2Int(cell.x + 1, cell.y),
                new Vector2Int(cell.x, cell.y - 1),
                new Vector2Int(cell.x, cell.y + 1)
            };
            
            foreach (var n in neighbors)
            {
                if (n.x >= 0 && n.x < width && n.y >= 0 && n.y < height &&
                    !map[n.x, n.y] && !visited[n.x, n.y])
                {
                    visited[n.x, n.y] = true;
                    queue.Enqueue(n);
                }
            }
        }
        
        return region;
    }

    public bool[,] GetMap() => map;
}
```

## 四、Enemy Placement 系统

```csharp
/// <summary>
/// 基于难度的敌人布置系统
/// </summary>
public class DungeonEnemyPlacer : MonoBehaviour
{
    [SerializeField] private EnemySpawnConfig[] enemyConfigs;
    [SerializeField] private float minDistanceFromStart = 5f;

    [Serializable]
    public class EnemySpawnConfig
    {
        public string EnemyId;
        public GameObject Prefab;
        public int MinCount;
        public int MaxCount;
        public float SpawnWeight;      // 权重（相对概率）
        public int MinRoomSize;        // 需要的最小房间大小
        public bool SpawnInCorridors;  // 是否在走廊生成
    }

    public void PlaceEnemies(BSPDungeonGenerator dungeon, Vector2Int startRoom, int difficulty)
    {
        var rooms = dungeon.Rooms;
        
        foreach (var room in rooms)
        {
            // 跳过起始房间附近
            Vector2Int roomCenter = new Vector2Int(
                room.xMin + room.width / 2, room.yMin + room.height / 2);
            
            if (Vector2Int.Distance(roomCenter, startRoom) < minDistanceFromStart)
                continue;
            
            // 根据房间大小和难度决定生成什么敌人
            float difficultyMul = 1f + difficulty * 0.2f;
            int enemyBudget = Mathf.RoundToInt(
                (room.width * room.height) / 20f * difficultyMul);
            
            PlaceEnemiesInRoom(room, enemyBudget);
        }
    }

    void PlaceEnemiesInRoom(RectInt room, int budget)
    {
        // 使用权重随机选择敌人类型
        var selector = new WeightedRandomSelector<EnemySpawnConfig>();
        foreach (var cfg in enemyConfigs)
        {
            if (room.width * room.height >= cfg.MinRoomSize * cfg.MinRoomSize)
                selector.Add(cfg, cfg.SpawnWeight);
        }
        
        int placed = 0;
        int maxAttempts = budget * 5;
        
        while (placed < budget && maxAttempts-- > 0)
        {
            var config = selector.Select();
            if (config == null) break;
            
            // 在房间内随机找一个可用位置
            Vector3 spawnPos = new Vector3(
                Random.Range(room.xMin + 1, room.xMax - 1),
                0,
                Random.Range(room.yMin + 1, room.yMax - 1));
            
            if (config.Prefab != null)
            {
                Instantiate(config.Prefab, spawnPos, 
                    Quaternion.Euler(0, Random.Range(0, 360), 0));
            }
            placed++;
        }
    }
}
```

## 五、PCG 最佳实践

| 算法 | 效果 | 性能 | 适用场景 |
|------|------|------|----------|
| BSP 分割 | 整齐矩形房间 | 快 | 经典地牢、RPG |
| 细胞自动机 | 有机洞穴 | 快 | 洞穴、矿洞 |
| Drunkard Walk | 蜿蜒走廊 | 快 | 随机地图 |
| Delaunay/MST | 连通图走廊 | 中 | 保证连通性 |
| WFC | 规则约束生成 | 慢 | 高质量瓦片地图 |
| Perlin Noise | 地形高度图 | 快 | 开放世界地形 |

**关键设计原则：**
1. 使用种子（Seed）保证可复现性，便于调试和分享关卡
2. 生成后验证连通性，确保所有区域可达
3. 设置参数边界，防止生成极端情况（全墙/全空）
4. 支持运行时动态生成（Roguelike 核心）
