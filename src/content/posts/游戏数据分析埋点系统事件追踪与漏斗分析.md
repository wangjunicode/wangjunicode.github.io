```csharp
    public const string ErrorOccurred  = "error_occurred";
    public const string LowFPS        = "low_fps_warning";
    public const string CrashRecovery = "crash_recovery";
}

/// <summary>
/// 埋点事件数据
/// </summary>
public class AnalyticsEvent
{
    public string EventName;
    public Dictionary<string, object> Parameters;
    public long Timestamp;
    public bool IsImmediate; // 是否立即上报（不进缓冲队列）
    
    public AnalyticsEvent(string name, Dictionary<string, object> parameters = null)
    {
        EventName = name;
        Parameters = parameters ?? new Dictionary<string, object>();
        Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
    }
}
```

## 二、埋点SDK封装

```csharp
/// <summary>
/// 数据分析管理器
/// </summary>
public class AnalyticsManager : MonoBehaviour
{
    private static AnalyticsManager instance;
    public static AnalyticsManager Instance => instance;

    [Header("上报配置")]
    [SerializeField] private string analyticsEndpoint = "https://api.yourgame.com/analytics";
    [SerializeField] private int batchSize = 20;              // 批量上报条数
    [SerializeField] private float batchInterval = 30f;      // 批量上报间隔（秒）
    [SerializeField] private int maxQueueSize = 500;         // 最大离线队列大小
    [SerializeField] private bool enableAnalytics = true;
    
    // 公共属性（每条事件自动附加）
    private Dictionary<string, object> commonProperties;
    
    // 离线队列
    private Queue<AnalyticsEvent> eventQueue = new Queue<AnalyticsEvent>();
    private float batchTimer;
    
    // 会话追踪
    private float sessionStartTime;
    private long sessionId;

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
        
        InitializeCommonProperties();
        sessionStartTime = Time.realtimeSinceStartup;
        sessionId = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
        
        LoadOfflineQueue();
    }

    void InitializeCommonProperties()
    {
        commonProperties = new Dictionary<string, object>
        {
            ["app_version"]    = Application.version,
            ["platform"]       = Application.platform.ToString(),
            ["device_model"]   = SystemInfo.deviceModel,
            ["os_version"]     = SystemInfo.operatingSystem,
            ["screen_width"]   = Screen.width,
            ["screen_height"]  = Screen.height,
            ["language"]       = Application.systemLanguage.ToString(),
        };
    }

    void Update()
    {
        batchTimer += Time.deltaTime;
        if (batchTimer >= batchInterval && eventQueue.Count > 0)
        {
            batchTimer = 0;
            FlushEvents();
        }
    }

    /// <summary>
    /// 记录埋点事件
    /// </summary>
    public void Track(string eventName, Dictionary<string, object> parameters = null)
    {
        if (!enableAnalytics) return;
        
        var evt = new AnalyticsEvent(eventName, parameters);
        
        // 附加公共属性
        foreach (var kv in commonProperties)
            if (!evt.Parameters.ContainsKey(kv.Key))
                evt.Parameters[kv.Key] = kv.Value;
        
        // 附加会话信息
        evt.Parameters["session_id"] = sessionId;
        evt.Parameters["session_duration"] = 
            Mathf.RoundToInt(Time.realtimeSinceStartup - sessionStartTime);
        
        // 附加用户ID（如果已登录）
        string userId = PlayerDataService.GetLocalPlayerData()?.PlayerId;
        if (!string.IsNullOrEmpty(userId))
            evt.Parameters["user_id"] = userId;
        
        if (evt.IsImmediate)
        {
            // 立即上报（如付费事件）
            StartCoroutine(UploadEvents(new List<AnalyticsEvent> { evt }));
        }
        else
        {
            // 加入队列
            if (eventQueue.Count < maxQueueSize)
                eventQueue.Enqueue(evt);
            
            // 达到批量大小立即上报
            if (eventQueue.Count >= batchSize)
                FlushEvents();
        }
    }

    /// <summary>
    /// 快捷方法：记录带单个参数的事件
    /// </summary>
    public void Track(string eventName, string key, object value)
    {
        Track(eventName, new Dictionary<string, object> { [key] = value });
    }

    public void FlushEvents()
    {
        if (eventQueue.Count == 0) return;
        
        var batch = new List<AnalyticsEvent>();
        while (eventQueue.Count > 0 && batch.Count < batchSize)
            batch.Add(eventQueue.Dequeue());
        
        StartCoroutine(UploadEvents(batch));
    }

    System.Collections.IEnumerator UploadEvents(List<AnalyticsEvent> events)
    {
        if (!Application.internetReachability.Equals(NetworkReachability.NotReachable))
        {
            string json = SerializeEvents(events);
            var request = new UnityEngine.Networking.UnityWebRequest(
                analyticsEndpoint, "POST");
            byte[] bodyRaw = System.Text.Encoding.UTF8.GetBytes(json);
            request.uploadHandler = new UnityEngine.Networking.UploadHandlerRaw(bodyRaw);
            request.downloadHandler = new UnityEngine.Networking.DownloadHandlerBuffer();
            request.SetRequestHeader("Content-Type", "application/json");
            
            yield return request.SendWebRequest();
            
            if (request.result != UnityEngine.Networking.UnityWebRequest.Result.Success)
            {
                Debug.LogWarning($"[Analytics] 上报失败: {request.error}，存入离线队列");
                foreach (var evt in events)
                    SaveToOfflineQueue(evt);
            }
        }
        else
        {
            // 无网络，存入离线队列
            foreach (var evt in events)
                SaveToOfflineQueue(evt);
        }
    }

    string SerializeEvents(List<AnalyticsEvent> events)
    {
        // 简化序列化（实际使用JSON库）
        return JsonUtility.ToJson(new { events = events.Count });
    }

    void SaveToOfflineQueue(AnalyticsEvent evt)
    {
        // 持久化到本地（PlayerPrefs或文件）
    }

    void LoadOfflineQueue()
    {
        // 加载离线队列，下次上线时补发
    }

    void OnApplicationPause(bool pauseStatus)
    {
        if (pauseStatus)
        {
            // 切到后台：立即上报所有缓存事件
            Track(AnalyticsEvents.SessionEnd, new Dictionary<string, object>
            {
                ["duration"] = Mathf.RoundToInt(Time.realtimeSinceStartup - sessionStartTime)
            });
            FlushEvents();
        }
    }
}
```

## 三、关键漏斗埋点示例

```csharp
/// <summary>
/// 购买转化漏斗埋点
/// </summary>
public class PurchaseFunnelTracker : MonoBehaviour
{
    void OnShopOpened()
    {
        AnalyticsManager.Instance?.Track(AnalyticsEvents.ShopOpen);
    }
    
    void OnItemViewed(string itemId, string category, long price)
    {
        AnalyticsManager.Instance?.Track(AnalyticsEvents.ItemView,
            new Dictionary<string, object>
            {
                ["item_id"] = itemId,
                ["category"] = category,
                ["price"] = price,
            });
    }
    
    void OnPurchaseInitiated(string itemId, long price, string currency)
    {
        // 付费相关事件立即上报
        var evt = new AnalyticsEvent(AnalyticsEvents.PurchaseInitiate,
            new Dictionary<string, object>
            {
                ["item_id"] = itemId,
                ["price"] = price,
                ["currency"] = currency,
            });
        evt.IsImmediate = true;
        AnalyticsManager.Instance?.Track(evt.EventName, evt.Parameters);
    }
    
    void OnPurchaseSuccess(string itemId, long price, string orderId)
    {
        var evt = new AnalyticsEvent(AnalyticsEvents.PurchaseSuccess,
            new Dictionary<string, object>
            {
                ["item_id"] = itemId,
                ["price"] = price,
                ["order_id"] = orderId,
            });
        evt.IsImmediate = true;
        AnalyticsManager.Instance?.Track(evt.EventName, evt.Parameters);
    }
}
```

## 四、数据合规设计

| 合规要求 | 实现方案 |
|----------|----------|
| GDPR同意 | 获得用户同意前不采集数据 |
| COPPA（儿童）| 13岁以下不采集设备ID/位置 |
| 数据最小化 | 只采集业务必需数据 |
| 数据删除权 | 提供接口删除用户数据 |
| 匿名化 | 设备ID哈希处理，无法逆向 |
| 隐私政策 | 首次启动展示并需用户确认 |
