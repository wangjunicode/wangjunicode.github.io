```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 兴趣点（Point of Interest）类型
/// </summary>
public enum POIType
{
    // 任务相关
    MainQuest,
    SideQuest,
    CompletedQuest,
    
    // NPC 相关
    NPC_Merchant,
    NPC_QuestGiver,
    NPC_Hostile,
    
    // 地点
    Town,
    Dungeon,
    Landmark,
    
    // 资源
    Resource_Ore,
    Resource_Herb,
    Resource_Chest,
    
    // 战斗
    Boss,
    EliteEnemy,
    
    // 玩家自定义
    PlayerMark_Default,
    PlayerMark_Danger,
    PlayerMark_Treasure,
    PlayerMark_Camp,
}

/// <summary>
/// 兴趣点数据
/// </summary>
[Serializable]
public class POIData
{
    public string Id;
    public POIType Type;
    public Vector3 WorldPosition;
    public string DisplayName;
    public string Description;
    public Sprite Icon;
    public Color IconColor = Color.white;
    
    // 可见性控制
    public bool IsVisible = true;
    public float VisibleRange = -1f;    // -1 = 全局可见
    public bool ShowOnMinimap = true;
    public bool ShowOnWorldMap = true;
    public bool ShowOnHUD = true;       // 世界空间指示器
    
    // 动态属性
    public bool IsDynamic;              // 是否跟随 GameObject 移动
    public Transform FollowTarget;      // 跟随的对象
    
    // 任务关联
    public string LinkedQuestId;
    
    // 玩家标记特有
    public string PlayerNote;           // 玩家备注
    public long CreateTime;
}

/// <summary>
/// 兴趣点配置（图标、颜色等静态配置）
/// </summary>
[CreateAssetMenu(fileName = "POIConfig", menuName = "Game/POI Config")]
public class POIConfig : ScriptableObject
{
    [Serializable]
    public class TypeConfig
    {
        public POIType Type;
        public Sprite Icon;
        public Color Color = Color.white;
        public float DisplayPriority = 1f;    // 优先级（高优先级在聚合时优先显示）
        public float MinimapIconSize = 10f;
        public bool AlwaysShowLabel;          // 是否始终显示名称标签
    }

    public List<TypeConfig> Configs = new List<TypeConfig>();
    
    public TypeConfig Get(POIType type) =>
        Configs.Find(c => c.Type == type);
}
```

## 二、POI 管理器

```csharp
/// <summary>
/// 兴趣点管理器（全局单例）
/// </summary>
public class POIManager : MonoBehaviour
{
    private static POIManager instance;
    public static POIManager Instance => instance;

    [SerializeField] private POIConfig config;
    
    private Dictionary<string, POIData> allPOIs = new Dictionary<string, POIData>();
    
    public event Action<POIData> OnPOIAdded;
    public event Action<string> OnPOIRemoved;
    public event Action<POIData> OnPOIUpdated;

    void Awake() { instance = this; }

    /// <summary>
    /// 添加兴趣点
    /// </summary>
    public void AddPOI(POIData poi)
    {
        if (string.IsNullOrEmpty(poi.Id))
            poi.Id = Guid.NewGuid().ToString("N");
        
        // 填充配置（图标/颜色）
        var typeConfig = config?.Get(poi.Type);
        if (typeConfig != null)
        {
            if (poi.Icon == null) poi.Icon = typeConfig.Icon;
            if (poi.IconColor == Color.white) poi.IconColor = typeConfig.Color;
        }
        
        allPOIs[poi.Id] = poi;
        OnPOIAdded?.Invoke(poi);
    }

    public void RemovePOI(string id)
    {
        if (allPOIs.Remove(id))
            OnPOIRemoved?.Invoke(id);
    }

    public void UpdatePOI(POIData poi)
    {
        if (allPOIs.ContainsKey(poi.Id))
        {
            allPOIs[poi.Id] = poi;
            OnPOIUpdated?.Invoke(poi);
        }
    }

    /// <summary>
    /// 添加玩家自定义标记
    /// </summary>
    public POIData AddPlayerMark(Vector3 worldPos, POIType markType, string note = "")
    {
        var poi = new POIData
        {
            Type = markType,
            WorldPosition = worldPos,
            DisplayName = GetDefaultMarkName(markType),
            PlayerNote = note,
            CreateTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds()
        };
        
        AddPOI(poi);
        SavePlayerMarks();
        return poi;
    }

    /// <summary>
    /// 获取在指定范围内的 POI 列表
    /// </summary>
    public List<POIData> GetNearbyPOIs(Vector3 center, float radius, 
        POIType? filterType = null)
    {
        var result = new List<POIData>();
        float sqrRadius = radius * radius;
        
        foreach (var poi in allPOIs.Values)
        {
            if (!poi.IsVisible) continue;
            if (filterType.HasValue && poi.Type != filterType.Value) continue;
            
            float sqrDist = Vector3.SqrMagnitude(poi.WorldPosition - center);
            if (sqrDist <= sqrRadius)
                result.Add(poi);
        }
        
        // 按距离排序
        result.Sort((a, b) => 
            Vector3.SqrMagnitude(a.WorldPosition - center)
            .CompareTo(Vector3.SqrMagnitude(b.WorldPosition - center)));
        
        return result;
    }

    public List<POIData> GetPOIsByType(POIType type)
    {
        var result = new List<POIData>();
        foreach (var poi in allPOIs.Values)
            if (poi.Type == type && poi.IsVisible)
                result.Add(poi);
        return result;
    }

    string GetDefaultMarkName(POIType type) => type switch
    {
        POIType.PlayerMark_Default  => "标记",
        POIType.PlayerMark_Danger   => "危险区域",
        POIType.PlayerMark_Treasure => "宝藏",
        POIType.PlayerMark_Camp     => "营地",
        _ => "标记"
    };

    void SavePlayerMarks()
    {
        // 序列化玩家自定义标记并保存
        var playerMarks = new List<POIData>();
        foreach (var poi in allPOIs.Values)
        {
            if (poi.Type.ToString().StartsWith("PlayerMark"))
                playerMarks.Add(poi);
        }
        
        string json = JsonUtility.ToJson(new { marks = playerMarks });
        PlayerPrefs.SetString("player_map_marks", json);
    }
}
```

```csharp
        {
            // 更新跟随类型的POI位置
        }
    }

    /// <summary>
    /// 世界坐标转小地图坐标
    /// </summary>
    Vector2 WorldToMinimapPos(Vector3 worldPos)
    {
        Vector3 delta = worldPos - playerTransform.position;
        Vector2 minimapPos = new Vector2(delta.x, delta.z) * mapScale;
        
        // 旋转（小地图随玩家转向时）
        float angle = -playerTransform.eulerAngles.y * Mathf.Deg2Rad;
        float cos = Mathf.Cos(angle);
        float sin = Mathf.Sin(angle);
        
        return new Vector2(
            minimapPos.x * cos - minimapPos.y * sin,
            minimapPos.x * sin + minimapPos.y * cos);
    }

    void OnPOIAdded(POIData poi)
    {
        if (!poi.ShowOnMinimap) return;
        
        var go = Instantiate(iconPrefab, minimapRect);
        var icon = go.GetComponent<RectTransform>();
        
        // 设置图标外观
        var img = go.GetComponent<UnityEngine.UI.Image>();
        img.sprite = poi.Icon;
        img.color = poi.IconColor;
        
        poiIcons[poi.Id] = icon;
    }

    void OnPOIRemoved(string id)
    {
        if (poiIcons.TryGetValue(id, out var icon))
        {
            Destroy(icon.gameObject);
            poiIcons.Remove(id);
        }
    }
}
```

```csharp
/// <summary>
/// 屏幕边缘指示器（目标在视野外时显示方向箭头）
/// </summary>
public class OffScreenIndicator : MonoBehaviour
{
    [SerializeField] private POIType[] showTypes;       // 显示哪些类型的POI

    [SerializeField] private float edgePadding = 40f;   // 距离屏幕边缘的距离

    private Dictionary<string, RectTransform> arrows 
        = new Dictionary<string, RectTransform>();

    void Start()
    {
        mainCamera = Camera.main;
    }

    void Update()
    {
        foreach (var kv in arrows)
        {
            // 检查POI是否在视野外
            // 如果在视野外，显示边缘箭头
        }
    }

    Vector2 GetEdgePosition(Vector3 worldPos)
    {
        Vector3 screenPos = mainCamera.WorldToScreenPoint(worldPos);
        
        // 目标在摄像机后面时翻转
        if (screenPos.z < 0) screenPos *= -1;
        
        // 计算到屏幕中心的方向
        Vector2 screenCenter = new Vector2(Screen.width / 2f, Screen.height / 2f);
        Vector2 direction = ((Vector2)screenPos - screenCenter).normalized;
        
        // 计算边缘位置
        float halfW = Screen.width / 2f - edgePadding;
        float halfH = Screen.height / 2f - edgePadding;
        
        float slope = direction.y / (direction.x + float.Epsilon);
        
        Vector2 edgePos;
        if (Mathf.Abs(slope) < halfH / halfW)
        {
            // 左右边缘
            float x = direction.x > 0 ? halfW : -halfW;
            edgePos = new Vector2(x, x * slope) + screenCenter;
        }
        else
        {
            // 上下边缘
            float y = direction.y > 0 ? halfH : -halfH;
            edgePos = new Vector2(y / slope, y) + screenCenter;
        }
        
        return edgePos;
    }
}
```

## 五、地图设计要点

| 功能 | 优先级 | 说明 |
|------|--------|------|
| 玩家位置实时更新 | 必须 | 小地图核心功能 |
| 任务目标标记 | 高 | 驱动玩家前进的核心 |
| 玩家自定义标记 | 高 | 提升探索体验 |
| POI图标聚合 | 中 | 大量标记时防止混乱 |
| 屏幕边缘指示器 | 中 | 帮助玩家找到目标方向 |
| 地图标注（文字）| 低 | 丰富世界感 |
| 已探索区域显示 | 低 | 探索类游戏核心 |

**设计原则：地图系统的核心是「帮玩家少迷路」，而不是「展示所有信息」。层次化地展示信息（缩放层级、过滤类型）是关键。**
