## 一、配方数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 合成配方（ScriptableObject）
/// </summary>
[CreateAssetMenu(fileName = "Recipe", menuName = "Game/Crafting Recipe")]
public class CraftingRecipe : ScriptableObject
{
    [Header("配方信息")]
    public string RecipeId;
    public string DisplayName;
    [TextArea(2, 3)] public string Description;
    
    [Header("产出")]
    public string OutputItemId;         // 产出物品ID
    public int OutputCount = 1;         // 产出数量
    
    [Header("材料")]
    public List<RecipeMaterial> Materials;
    
    [Header("需求")]
    public string WorkbenchType;        // 需要的工作台类型（空=随处可合成）
    public int RequiredCraftingLevel;   // 需要制作等级
    public List<string> RequiredSkillIds; // 需要的技能/知识
    
    [Header("合成参数")]
    public float CraftTime = 0f;        // 合成时间（秒，0=立即）
    public float BaseSuccessRate = 1f;  // 基础成功率（1=100%）
    public bool CanBatchCraft = true;   // 是否支持批量合成
    
    [Header("发现")]
    public bool IsKnownByDefault = true;    // 是否默认已知配方
    public string UnlockCondition;          // 解锁条件描述
    
    [Header("品质系统")]
    public bool HasQuality;             // 是否有品质随机
    public AnimationCurve QualityDistribution; // 品质分布曲线
}

/// <summary>
/// 配方材料需求
/// </summary>
[Serializable]
public class RecipeMaterial
{
    public string ItemId;
    public int Count;
    public bool IsConsumed = true;      // 是否消耗（false=工具类不消耗）
}

/// <summary>
/// 合成品质枚举
/// </summary>
public enum ItemQuality { Broken, Poor, Normal, Good, Excellent, Perfect }
```

## 二、合成系统管理器

```csharp
/// <summary>
/// 合成系统管理器
/// </summary>
public class CraftingManager : MonoBehaviour
{
    private static CraftingManager instance;
    public static CraftingManager Instance => instance;

    private Dictionary<string, CraftingRecipe> allRecipes 
        = new Dictionary<string, CraftingRecipe>();
    private HashSet<string> knownRecipes = new HashSet<string>(); // 玩家已解锁的配方
    
    // 合成队列（异步合成）
    private Queue<CraftQueueEntry> craftQueue = new Queue<CraftQueueEntry>();
    private CraftQueueEntry currentCrafting;
    private float craftTimer;

    public event Action<string, int, ItemQuality> OnCraftSuccess;  // itemId, count, quality
    public event Action<string> OnCraftFailed;
    public event Action<CraftingRecipe> OnRecipeUnlocked;

    void Awake()
    {
        instance = this;
        LoadAllRecipes();
        LoadKnownRecipes();
    }

    void LoadAllRecipes()
    {
        var recipes = Resources.LoadAll<CraftingRecipe>("Recipes");
        foreach (var r in recipes)
        {
            allRecipes[r.RecipeId] = r;
            if (r.IsKnownByDefault)
                knownRecipes.Add(r.RecipeId);
        }
    }

    void LoadKnownRecipes()
    {
        // 从存档加载已解锁配方
        string json = PlayerPrefs.GetString("known_recipes", "");
        if (!string.IsNullOrEmpty(json))
        {
            var ids = JsonUtility.FromJson<StringList>(json).Items;
            foreach (var id in ids)
                knownRecipes.Add(id);
        }
    }

    /// <summary>
    /// 检查是否可以合成
    /// </summary>
    public CraftCheckResult CanCraft(string recipeId, int count = 1, string workbenchType = null)
    {
        if (!allRecipes.TryGetValue(recipeId, out var recipe))
            return CraftCheckResult.Fail("配方不存在");
        
        if (!knownRecipes.Contains(recipeId))
            return CraftCheckResult.Fail("尚未解锁此配方");
        
        if (!string.IsNullOrEmpty(recipe.WorkbenchType) && 
            recipe.WorkbenchType != workbenchType)
            return CraftCheckResult.Fail($"需要工作台：{recipe.WorkbenchType}");
        
        // 检查材料
        foreach (var mat in recipe.Materials)
        {
            int required = mat.Count * count;
            int owned = InventoryManager.Instance?.GetItemCount(mat.ItemId) ?? 0;
            
            if (owned < required)
            {
                string itemName = ItemDatabase.GetItemName(mat.ItemId);
                return CraftCheckResult.Fail($"{itemName} 不足（{owned}/{required}）");
            }
        }
        
        // 检查技能
        if (recipe.RequiredCraftingLevel > 0)
        {
            int playerLevel = PlayerDataService.GetLocalPlayerData().CraftingLevel;
            if (playerLevel < recipe.RequiredCraftingLevel)
                return CraftCheckResult.Fail($"制作等级不足（需要{recipe.RequiredCraftingLevel}级）");
        }
        
        return CraftCheckResult.OK();
    }

    /// <summary>
    /// 执行合成
    /// </summary>
    public bool Craft(string recipeId, int count = 1, string workbenchType = null)
    {
        var check = CanCraft(recipeId, count, workbenchType);
        if (!check.CanCraft)
        {
            UIManager.Instance?.ShowMessage(check.FailReason);
            return false;
        }
        
        var recipe = allRecipes[recipeId];
        
        if (recipe.CraftTime > 0)
        {
            // 加入合成队列
            craftQueue.Enqueue(new CraftQueueEntry
            {
                RecipeId = recipeId,
                Count = count,
                TotalTime = recipe.CraftTime * count
            });
            return true;
        }
        
        // 立即合成
        ExecuteCraft(recipe, count);
        return true;
    }

    void ExecuteCraft(CraftingRecipe recipe, int count)
    {
        // 消耗材料
        foreach (var mat in recipe.Materials)
        {
            if (mat.IsConsumed)
                InventoryManager.Instance?.RemoveItem(mat.ItemId, mat.Count * count);
        }
        
        // 判断成功/失败
        float successRate = recipe.BaseSuccessRate;
        // 玩家技能可以提升成功率
        successRate = Mathf.Min(1f, successRate + GetSkillBonus());
        
        if (UnityEngine.Random.value <= successRate)
        {
            // 合成成功
            ItemQuality quality = DetermineQuality(recipe, count);
            InventoryManager.Instance?.AddItem(recipe.OutputItemId, recipe.OutputCount * count);
            
            OnCraftSuccess?.Invoke(recipe.OutputItemId, recipe.OutputCount * count, quality);
            UIManager.Instance?.ShowMessage($"合成成功：{ItemDatabase.GetItemName(recipe.OutputItemId)} × {recipe.OutputCount * count}");
            
            // 获得制作经验
            GainCraftExp(recipe, count);
        }
        else
        {
            // 合成失败（材料已消耗）
            OnCraftFailed?.Invoke(recipe.RecipeId);
            UIManager.Instance?.ShowMessage("合成失败！材料已损耗");
        }
    }

    void Update()
    {
        // 处理合成队列
        if (currentCrafting == null && craftQueue.Count > 0)
        {
            currentCrafting = craftQueue.Dequeue();
            craftTimer = currentCrafting.TotalTime;
        }
        
        if (currentCrafting != null)
        {
            craftTimer -= Time.deltaTime;
            
            float progress = 1f - craftTimer / currentCrafting.TotalTime;
            // 通知UI更新进度条
            
            if (craftTimer <= 0)
            {
                // 完成合成
                if (allRecipes.TryGetValue(currentCrafting.RecipeId, out var recipe))
                    ExecuteCraft(recipe, currentCrafting.Count);
                
                currentCrafting = null;
            }
        }
    }

    public void UnlockRecipe(string recipeId)
    {
        if (knownRecipes.Contains(recipeId)) return;
        
        knownRecipes.Add(recipeId);
        
        if (allRecipes.TryGetValue(recipeId, out var recipe))
            OnRecipeUnlocked?.Invoke(recipe);
        
        UIManager.Instance?.ShowMessage($"发现新配方：{allRecipes[recipeId]?.DisplayName}");
    }

    ItemQuality DetermineQuality(CraftingRecipe recipe, int count)
    {
        if (!recipe.HasQuality) return ItemQuality.Normal;
        
        // 使用曲线确定品质
        float rand = UnityEngine.Random.value;
        float quality = recipe.QualityDistribution?.Evaluate(rand) ?? 0.5f;
        
        if (quality < 0.15f) return ItemQuality.Poor;
        if (quality < 0.45f) return ItemQuality.Normal;
        if (quality < 0.75f) return ItemQuality.Good;
        if (quality < 0.92f) return ItemQuality.Excellent;
        return ItemQuality.Perfect;
    }

    float GetSkillBonus() => 0f; // 实际从技能系统读取
    void GainCraftExp(CraftingRecipe recipe, int count) { }

    public List<CraftingRecipe> GetKnownRecipes()
    {
        var result = new List<CraftingRecipe>();
        foreach (var id in knownRecipes)
            if (allRecipes.TryGetValue(id, out var r)) result.Add(r);
        return result;
    }
}

public class CraftCheckResult
{
    public bool CanCraft;
    public string FailReason;
    
    public static CraftCheckResult OK() => new CraftCheckResult { CanCraft = true };
    public static CraftCheckResult Fail(string reason) => 
        new CraftCheckResult { CanCraft = false, FailReason = reason };
}

[Serializable] class StringList { public List<string> Items; }

class CraftQueueEntry
{
    public string RecipeId;
    public int Count;
    public float TotalTime;
}
```

## 三、合成经济平衡原则

| 原则 | 说明 |
|------|------|
| 材料价值 < 产出价值 | 合成有利可图，玩家才会用 |
| 终端物品限制合成 | 防止玩家大量制造顶级装备破坏经济 |
| 成功率与技能挂钩 | 鼓励玩家培养制作专业度 |
| 品质随机增加惊喜 | 完美品质是可遇不可求的稀缺品 |
| 配方发现制造探索欲 | 探索地牢发现稀有配方 |

**合成系统三角定律：**
- 材料（稀缺性）× 时间（沉没成本）× 技能（成长感）= 合成乐趣
