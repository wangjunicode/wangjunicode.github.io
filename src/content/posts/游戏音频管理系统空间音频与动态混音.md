## 一、音频管理器

```csharp
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Audio;

/// <summary>
/// 音频管理器（支持空间音频、音频池、混音器控制）
/// </summary>
public class AudioManager : MonoBehaviour
{
    private static AudioManager instance;
    public static AudioManager Instance => instance;

    [Header("AudioMixer")]
    [SerializeField] private AudioMixer masterMixer;
    [SerializeField] private AudioMixerGroup sfxGroup;
    [SerializeField] private AudioMixerGroup musicGroup;
    [SerializeField] private AudioMixerGroup ambientGroup;
    [SerializeField] private AudioMixerGroup voiceGroup;

    [Header("音频池")]
    [SerializeField] private int sfxPoolSize = 20;
    [SerializeField] private int musicPoolSize = 3;

    [Header("背景音乐")]
    [SerializeField] private float musicFadeDuration = 1.5f;

    // AudioSource 池
    private Queue<AudioSource> sfxPool = new Queue<AudioSource>();
    private List<AudioSource> activeSFX = new List<AudioSource>();
    private AudioSource[] musicSources; // 双缓冲音乐源（淡入淡出用）
    private int currentMusicIndex;

    // 音量设置（持久化）
    private const string PREF_MASTER_VOL = "volume_master";
    private const string PREF_SFX_VOL = "volume_sfx";
    private const string PREF_MUSIC_VOL = "volume_music";

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
        
        InitializePool();
        LoadVolumeSettings();
    }

    void InitializePool()
    {
        // 创建SFX池
        for (int i = 0; i < sfxPoolSize; i++)
        {
            var go = new GameObject($"SFX_Pool_{i}");
            go.transform.SetParent(transform);
            var src = go.AddComponent<AudioSource>();
            src.outputAudioMixerGroup = sfxGroup;
            src.playOnAwake = false;
            sfxPool.Enqueue(src);
        }
        
        // 创建双缓冲音乐源
        musicSources = new AudioSource[2];
        for (int i = 0; i < 2; i++)
        {
            var go = new GameObject($"Music_{i}");
            go.transform.SetParent(transform);
            var src = go.AddComponent<AudioSource>();
            src.outputAudioMixerGroup = musicGroup;
            src.loop = true;
            src.playOnAwake = false;
            musicSources[i] = src;
        }
    }

    // ============ SFX 播放 ============

    /// <summary>
    /// 播放2D音效
    /// </summary>
    public AudioSource PlaySFX(AudioClip clip, float volume = 1f, float pitch = 1f)
    {
        if (clip == null) return null;
        
        var source = GetSFXSource();
        source.clip = clip;
        source.volume = volume;
        source.pitch = pitch;
        source.spatialBlend = 0f; // 2D
        source.Play();
        
        StartCoroutine(ReturnToPoolWhenDone(source, clip.length / pitch));
        return source;
    }

    /// <summary>
    /// 播放3D空间音效
    /// </summary>
    public AudioSource PlaySFX3D(AudioClip clip, Vector3 position,
        float volume = 1f, float minDist = 5f, float maxDist = 30f)
    {
        if (clip == null) return null;
        
        var source = GetSFXSource();
        source.transform.position = position;
        source.clip = clip;
        source.volume = volume;
        source.spatialBlend = 1f; // 全3D
        source.minDistance = minDist;
        source.maxDistance = maxDist;
        source.rolloffMode = AudioRolloffMode.Custom;
        source.Play();
        
        StartCoroutine(ReturnToPoolWhenDone(source, clip.length));
        return source;
    }

    AudioSource GetSFXSource()
    {
        AudioSource source;
        
        if (sfxPool.Count > 0)
        {
            source = sfxPool.Dequeue();
        }
        else
        {
            // 池耗尽：停止最旧的音效
            if (activeSFX.Count > 0)
            {
                source = activeSFX[0];
                activeSFX.RemoveAt(0);
                source.Stop();
            }
            else
            {
                // 紧急创建
                var go = new GameObject("SFX_Emergency");
                go.transform.SetParent(transform);
                source = go.AddComponent<AudioSource>();
                source.outputAudioMixerGroup = sfxGroup;
            }
        }
        
        source.gameObject.SetActive(true);
        activeSFX.Add(source);
        return source;
    }

    IEnumerator ReturnToPoolWhenDone(AudioSource source, float duration)
    {
        yield return new WaitForSeconds(duration + 0.1f);
        
        if (source != null && !source.isPlaying)
        {
            activeSFX.Remove(source);
            source.clip = null;
            sfxPool.Enqueue(source);
        }
    }

    // ============ 背景音乐 ============

    public void PlayMusic(AudioClip clip, bool fade = true)
    {
        if (clip == null) return;
        
        var next = musicSources[1 - currentMusicIndex];
        next.clip = clip;
        
        if (fade)
            StartCoroutine(CrossFadeMusic(next));
        else
        {
            musicSources[currentMusicIndex].Stop();
            next.Play();
            currentMusicIndex = 1 - currentMusicIndex;
        }
    }

    IEnumerator CrossFadeMusic(AudioSource nextSource)
    {
        var current = musicSources[currentMusicIndex];
        nextSource.volume = 0f;
        nextSource.Play();
        
        float t = 0;
        while (t < musicFadeDuration)
        {
            t += Time.unscaledDeltaTime;
            float progress = t / musicFadeDuration;
            current.volume = Mathf.Lerp(1f, 0f, progress);
            nextSource.volume = Mathf.Lerp(0f, 1f, progress);
            yield return null;
        }
        
        current.Stop();
        current.volume = 1f;
        nextSource.volume = 1f;
        currentMusicIndex = 1 - currentMusicIndex;
    }

    public void StopMusic(bool fade = true)
    {
        if (fade)
            StartCoroutine(FadeOutMusic());
        else
            foreach (var src in musicSources)
                src.Stop();
    }

    IEnumerator FadeOutMusic()
    {
        var current = musicSources[currentMusicIndex];
        float startVol = current.volume;
        float t = 0;
        while (t < musicFadeDuration)
        {
            t += Time.unscaledDeltaTime;
            current.volume = Mathf.Lerp(startVol, 0f, t / musicFadeDuration);
            yield return null;
        }
        current.Stop();
        current.volume = 1f;
    }

    // ============ 音量控制 ============

    public void SetMasterVolume(float volume)
    {
        masterMixer.SetFloat("MasterVolume", LinearToDecibel(volume));
        PlayerPrefs.SetFloat(PREF_MASTER_VOL, volume);
    }

    public void SetSFXVolume(float volume)
    {
        masterMixer.SetFloat("SFXVolume", LinearToDecibel(volume));
        PlayerPrefs.SetFloat(PREF_SFX_VOL, volume);
    }

    public void SetMusicVolume(float volume)
    {
        masterMixer.SetFloat("MusicVolume", LinearToDecibel(volume));
        PlayerPrefs.SetFloat(PREF_MUSIC_VOL, volume);
    }

    void LoadVolumeSettings()
    {
        SetMasterVolume(PlayerPrefs.GetFloat(PREF_MASTER_VOL, 1f));
        SetSFXVolume(PlayerPrefs.GetFloat(PREF_SFX_VOL, 1f));
        SetMusicVolume(PlayerPrefs.GetFloat(PREF_MUSIC_VOL, 0.7f));
    }

    float LinearToDecibel(float linear)
    {
        // 0 → -80dB（静音），1 → 0dB（最大）
        return linear > 0.0001f ? 20f * Mathf.Log10(linear) : -80f;
    }

    // ============ AudioMixer 快照（状态切换）============

    public void TransitionToSnapshot(string snapshotName, float transitionTime = 0.5f)
    {
        var snapshot = masterMixer.FindSnapshot(snapshotName);
        snapshot?.TransitionTo(transitionTime);
    }

    // 预定义快照切换
    public void EnterCombat()    => TransitionToSnapshot("Combat");
    public void ExitCombat()     => TransitionToSnapshot("Exploration");
    public void PauseGame()      => TransitionToSnapshot("Paused");    // 低通滤波+静音SFX
    public void EnterCutscene()  => TransitionToSnapshot("Cutscene"); // 提高人声，降低音乐
}
```

## 二、自适应音乐系统

```csharp
/// <summary>
/// 自适应音乐（根据游戏状态动态切换音乐）
/// </summary>
public class AdaptiveMusicSystem : MonoBehaviour
{
    [System.Serializable]
    public class MusicState
    {
        public string StateName;
        public AudioClip Clip;
        public float Volume = 1f;
        public bool Loop = true;
    }

    [SerializeField] private MusicState[] states;
    [SerializeField] private float transitionDelay = 2f; // 状态切换后延迟切换音乐

    private Dictionary<string, MusicState> stateMap = new Dictionary<string, MusicState>();
    private string currentState;
    private Coroutine transitionCoroutine;

    void Awake()
    {
        foreach (var s in states)
            stateMap[s.StateName] = s;
    }

    public void SetState(string stateName)
    {
        if (stateName == currentState) return;
        currentState = stateName;
        
        if (transitionCoroutine != null)
            StopCoroutine(transitionCoroutine);
        
        transitionCoroutine = StartCoroutine(TransitionAfterDelay(stateName));
    }

    IEnumerator TransitionAfterDelay(string stateName)
    {
        yield return new WaitForSeconds(transitionDelay);
        
        if (stateMap.TryGetValue(stateName, out var state))
            AudioManager.Instance?.PlayMusic(state.Clip);
    }
}
```

## 三、移动端音频优化

| 优化项 | 方案 | 节省 |
|--------|------|------|
| 压缩格式 | Android: Vorbis, iOS: AAC | 包体减少50%+ |
| 背景音乐流式 | Load Type: Streaming | 内存减少90% |
| 短音效 | Load Type: Decompress on Load | 延迟更低 |
| 同时播放通道 | 限制最大通道数(32) | 防止CPU过载 |
| 音频距离剔除 | 超出范围不播放 | 减少通道占用 |
| 采样率 | 非人声22050Hz足够 | 文件减小30% |
