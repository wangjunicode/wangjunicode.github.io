## 一、对话系统的功能层次

```
基础层  → 文字显示、名字框、头像
交互层  → 玩家选项、条件分支、变量跟踪
表现层  → 打字机效果、语音同步、表情动画
管理层  → 对话触发、历史记录、存档读档
内容层  → Ink/Yarn 脚本、本地化、版本管理
```

## 二、基础对话数据结构

```csharp
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 对话条目
/// </summary>
[Serializable]
public class DialogueLine
{
    public string SpeakerName;         // 说话者名称
    public string SpeakerId;           // 说话者ID（用于头像、语音）
    public string Text;                // 对话内容
    public string VoiceClipAddress;    // 语音资源地址（Addressables）
    public string Portrait;            // 头像表情（happy/sad/angry等）
    public float DisplayDuration;      // 强制显示时长（0=等待点击）
    public DialogueEffect[] Effects;   // 特效指令
}

/// <summary>
/// 对话特效指令
/// </summary>
[Serializable]
public class DialogueEffect
{
    public EffectType Type;
    public string Parameter;
    
    public enum EffectType
    {
        ShakeCamera,       // 抖动摄像机
        PlayAnimation,     // 播放动画
        SetWeather,        // 改变天气
        SpawnObject,       // 生成对象
        PlaySound,         // 播放音效
        FadeIn,            // 淡入
        FadeOut,           // 淡出
        WaitTime,          // 等待一段时间
        SetVariable,       // 设置变量
    }
}

/// <summary>
/// 对话分支选项
/// </summary>
[Serializable]
public class DialogueChoice
{
    public string Text;           // 选项文本
    public string ConditionKey;   // 显示条件（空=始终显示）
    public object ConditionValue; // 条件值
    public string NextDialogueId; // 选择后跳转的对话
    public DialogueLine[] Consequence; // 选择后的即时对话
}

/// <summary>
/// 完整对话节点
/// </summary>
[CreateAssetMenu(fileName = "Dialogue", menuName = "Game/Dialogue Node")]
public class DialogueNode : ScriptableObject
{
    public string NodeId;
    public DialogueLine[] Lines;
    public DialogueChoice[] Choices;     // 不为空时显示选项
    public string DefaultNextNodeId;      // 无选项时的下一节点
    
    [Header("触发条件")]
    public string[] RequiredFlags;        // 需要的全局标记
    public string[] BlockingFlags;        // 阻止显示的标记
    
    [Header("完成效果")]
    public string[] SetFlagsOnComplete;   // 完成后设置的标记
    public QuestAction[] QuestActions;    // 关联的任务操作
}

[Serializable]
public class QuestAction
{
    public string QuestId;
    public QuestActionType Type;
    public enum QuestActionType { Start, Advance, Complete, Fail }
}
```

## 三、对话管理器

```csharp
/// <summary>
/// 对话管理器（单例）
/// </summary>
public class DialogueManager : MonoBehaviour
{
    private static DialogueManager instance;
    public static DialogueManager Instance => instance;

    [Header("UI 引用")]
    [SerializeField] private DialogueUI dialogueUI;

    [Header("配置")]
    [SerializeField] private float defaultTypingSpeed = 30f;   // 每秒显示字符数
    [SerializeField] private float punctuationDelay = 0.15f;   // 标点停顿时间

    // 对话状态
    private bool isDialogueActive;
    private DialogueNode currentNode;
    private int currentLineIndex;
    private Coroutine typingCoroutine;
    private bool isTyping;
    private bool isSkipRequested;

    // 对话变量存储（跨对话持久化）
    private Dictionary<string, object> dialogueVariables = new Dictionary<string, object>();

    // 事件
    public event Action OnDialogueStarted;
    public event Action OnDialogueEnded;
    public event Action<string, object> OnVariableChanged;
    public event Action<DialogueChoice> OnChoiceMade;

    public bool IsDialogueActive => isDialogueActive;

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
    }

    /// <summary>
    /// 启动对话
    /// </summary>
    public void StartDialogue(DialogueNode node)
    {
        if (isDialogueActive) return;
        if (!CheckConditions(node)) return;
        
        isDialogueActive = true;
        currentNode = node;
        currentLineIndex = 0;
        
        // 暂停游戏（可选）
        // Time.timeScale = 0f;
        
        dialogueUI.Show();
        OnDialogueStarted?.Invoke();
        
        ShowCurrentLine();
    }

    void ShowCurrentLine()
    {
        if (currentLineIndex >= currentNode.Lines.Length)
        {
            // 所有台词展示完毕，显示选项或跳转
            OnAllLinesShown();
            return;
        }
        
        var line = currentNode.Lines[currentLineIndex];
        
        // 更新UI
        dialogueUI.SetSpeaker(line.SpeakerName, line.SpeakerId, line.Portrait);
        
        // 执行特效
        foreach (var effect in line.Effects ?? Array.Empty<DialogueEffect>())
            ExecuteEffect(effect);
        
        // 播放语音
        if (!string.IsNullOrEmpty(line.VoiceClipAddress))
            PlayVoice(line.VoiceClipAddress);
        
        // 打字机效果显示文字
        if (typingCoroutine != null) StopCoroutine(typingCoroutine);
        typingCoroutine = StartCoroutine(TypeLineCoroutine(line));
    }

    IEnumerator TypeLineCoroutine(DialogueLine line)
    {
        isTyping = true;
        isSkipRequested = false;
        
        dialogueUI.SetText("");
        
        float charDelay = 1f / defaultTypingSpeed;
        var sb = new System.Text.StringBuilder();
        
        for (int i = 0; i < line.Text.Length; i++)
        {
            if (isSkipRequested)
            {
                // 跳过打字机，直接显示全文
                dialogueUI.SetText(line.Text);
                break;
            }
            
            char c = line.Text[i];
            sb.Append(c);
            dialogueUI.SetText(sb.ToString());
            
            // 标点符号额外停顿
            float delay = charDelay;
            if (c is '，' or '。' or '！' or '？' or ',' or '.' or '!' or '?')
                delay += punctuationDelay;
            
            yield return new WaitForSecondsRealtime(delay);
        }
        
        isTyping = false;
        
        // 强制显示时长
        if (line.DisplayDuration > 0)
        {
            yield return new WaitForSecondsRealtime(line.DisplayDuration);
            AdvanceLine();
        }
        else
        {
            dialogueUI.ShowContinuePrompt(true);
        }
    }

    /// <summary>
    /// 点击继续（玩家输入触发）
    /// </summary>
    public void OnContinuePressed()
    {
        if (!isDialogueActive) return;
        
        if (isTyping)
        {
            // 第一次按：跳过打字机
            isSkipRequested = true;
        }
        else
        {
            // 第二次按：进入下一条
            AdvanceLine();
        }
    }

    void AdvanceLine()
    {
        dialogueUI.ShowContinuePrompt(false);
        currentLineIndex++;
        ShowCurrentLine();
    }

    void OnAllLinesShown()
    {
        if (currentNode.Choices != null && currentNode.Choices.Length > 0)
        {
            // 显示选项
            var validChoices = GetValidChoices(currentNode.Choices);
            dialogueUI.ShowChoices(validChoices, OnChoiceSelected);
        }
        else if (!string.IsNullOrEmpty(currentNode.DefaultNextNodeId))
        {
            // 跳转到下一节点
            var nextNode = LoadDialogueNode(currentNode.DefaultNextNodeId);
            if (nextNode != null)
            {
                // 完成当前节点的标记
                CompleteNode(currentNode);
                StartDialogue(nextNode);
            }
            else
            {
                EndDialogue();
            }
        }
        else
        {
            EndDialogue();
        }
    }

    DialogueChoice[] GetValidChoices(DialogueChoice[] choices)
    {
        var valid = new List<DialogueChoice>();
        foreach (var choice in choices)
        {
            if (string.IsNullOrEmpty(choice.ConditionKey))
            {
                valid.Add(choice);
            }
            else if (dialogueVariables.TryGetValue(choice.ConditionKey, out var val) 
                && val.Equals(choice.ConditionValue))
            {
                valid.Add(choice);
            }
        }
        return valid.ToArray();
    }

    void OnChoiceSelected(DialogueChoice choice)
    {
        OnChoiceMade?.Invoke(choice);
        
        // 显示选项的即时回应台词
        if (choice.Consequence != null && choice.Consequence.Length > 0)
        {
            var tempNode = ScriptableObject.CreateInstance<DialogueNode>();
            tempNode.Lines = choice.Consequence;
            tempNode.DefaultNextNodeId = choice.NextDialogueId;
            StartDialogue(tempNode);
        }
        else if (!string.IsNullOrEmpty(choice.NextDialogueId))
        {
            var nextNode = LoadDialogueNode(choice.NextDialogueId);
            if (nextNode != null) StartDialogue(nextNode);
            else EndDialogue();
        }
        else
        {
            EndDialogue();
        }
    }

    void EndDialogue()
    {
        CompleteNode(currentNode);
        
        isDialogueActive = false;
        currentNode = null;
        
        dialogueUI.Hide();
        // Time.timeScale = 1f;
        
        OnDialogueEnded?.Invoke();
    }

    void CompleteNode(DialogueNode node)
    {
        if (node == null) return;
        
        // 设置完成标记
        foreach (var flag in node.SetFlagsOnComplete ?? Array.Empty<string>())
            SetVariable(flag, true);
        
        // 执行任务操作
        foreach (var action in node.QuestActions ?? Array.Empty<QuestAction>())
            QuestSystem.Instance?.ProcessDialogueAction(action);
    }

    bool CheckConditions(DialogueNode node)
    {
        foreach (var flag in node.RequiredFlags ?? Array.Empty<string>())
        {
            if (!GetBool(flag)) return false;
        }
        foreach (var flag in node.BlockingFlags ?? Array.Empty<string>())
        {
            if (GetBool(flag)) return false;
        }
        return true;
    }

    public void SetVariable(string key, object value)
    {
        dialogueVariables[key] = value;
        OnVariableChanged?.Invoke(key, value);
    }

    public bool GetBool(string key) =>
        dialogueVariables.TryGetValue(key, out var v) && v is bool b && b;

    public T GetVariable<T>(string key, T defaultVal = default)
    {
        if (dialogueVariables.TryGetValue(key, out var v) && v is T t) return t;
        return defaultVal;
    }

    void ExecuteEffect(DialogueEffect effect)
    {
        switch (effect.Type)
        {
            case DialogueEffect.EffectType.ShakeCamera:
                CameraShakeSystem.ShakeHit(1f, Vector3.forward);
                break;
            case DialogueEffect.EffectType.PlaySound:
                AudioManager.Instance?.PlaySFX(effect.Parameter);
                break;
            case DialogueEffect.EffectType.SetVariable:
                var parts = effect.Parameter.Split('=');
                if (parts.Length == 2) SetVariable(parts[0].Trim(), parts[1].Trim());
                break;
        }
    }

    void PlayVoice(string address)
    {
        // 使用 Addressables 异步加载语音
    }

    DialogueNode LoadDialogueNode(string id)
    {
        // 从 Addressables 或本地缓存加载对话节点
        return null;
    }
}
```

## 四、Ink 脚本集成

Ink 是专为游戏叙事设计的脚本语言，与 Unity 集成方案：

```csharp
#if INK_INTEGRATION
using Ink.Runtime;

/// <summary>
/// Ink 对话运行器
/// </summary>
public class InkDialogueRunner : MonoBehaviour
{
    [SerializeField] private TextAsset inkJsonAsset;
    private Story inkStory;

    void Start()
    {
        inkStory = new Story(inkJsonAsset.text);
        
        // 绑定外部函数
        inkStory.BindExternalFunction("GetPlayerLevel", () => 
            PlayerData.Instance.Level);
        
        inkStory.BindExternalFunction("HasItem", (string itemId) => 
            Inventory.Instance.HasItem(itemId));
        
        inkStory.BindExternalFunction("GiveGold", (int amount) => 
            PlayerData.Instance.AddGold(amount));
    }

    public void ContinueDialogue()
    {
        if (!inkStory.canContinue && inkStory.currentChoices.Count == 0)
        {
            EndDialogue();
            return;
        }
        
        // 获取下一行文本
        if (inkStory.canContinue)
        {
            string text = inkStory.Continue();
            
            // 解析标签（#speaker:老铁匠  #portrait:happy）
            var tags = inkStory.currentTags;
            string speaker = ParseTag(tags, "speaker");
            string portrait = ParseTag(tags, "portrait");
            
            DialogueUI.Instance.ShowLine(speaker, text, portrait);
            
            // 处理选项
            if (!inkStory.canContinue && inkStory.currentChoices.Count > 0)
                ShowChoices();
        }
    }

    void ShowChoices()
    {
        var choices = inkStory.currentChoices;
        string[] choiceTexts = new string[choices.Count];
        
        for (int i = 0; i < choices.Count; i++)
            choiceTexts[i] = choices[i].text;
        
        DialogueUI.Instance.ShowChoices(choiceTexts, OnChoiceSelected);
    }

    void OnChoiceSelected(int index)
    {
        inkStory.ChooseChoiceIndex(index);
        ContinueDialogue();
    }

    string ParseTag(List<string> tags, string key)
    {
        foreach (var tag in tags)
        {
            if (tag.StartsWith(key + ":"))
                return tag.Substring(key.Length + 1).Trim();
        }
        return null;
    }

    void EndDialogue() { }
}
#endif
```

```csharp
/// <summary>
/// 对话本地化管理器
/// </summary>
public class DialogueLocalization
{
    private Dictionary<string, string> localizedTexts = new Dictionary<string, string>();
    private string currentLanguage = "zh-CN";
    
    public string GetLocalizedText(string key)
    {
        if (localizedTexts.TryGetValue(key, out var text))
            return text;
        return key; // 找不到时返回Key本身
    }
    
    public async void LoadLanguage(string langCode)
    {
        currentLanguage = langCode;
        // 从 Addressables 加载对应语言包
        // var handle = Addressables.LoadAssetAsync<TextAsset>($"Dialogue/{langCode}");
        // await handle.Task;
        // ParseLanguageFile(handle.Result.text);
    }
    
    void ParseLanguageFile(string json)
    {
        // 解析 JSON 格式的语言文件
        // { "npc_blacksmith_001": "你好，年轻人", ... }
        var dict = JsonUtility.FromJson<LanguageDict>(json);
        localizedTexts.Clear();
        foreach (var entry in dict.entries)
            localizedTexts[entry.key] = entry.value;
    }
    
    [Serializable]
    struct LanguageDict
    {
        public LocalizedEntry[] entries;
    }
    
    [Serializable]
    struct LocalizedEntry
    {
        public string key;
        public string value;
    }
}
```

## 六、系统设计要点

| 功能 | 技术方案 | 注意事项 |
|------|----------|----------|
| 基础对话 | ScriptableObject + 打字机 | 支持富文本标签 |
| 分支对话 | 选项节点 + 条件变量 | 合并显示条件减少数据量 |
| 叙事脚本 | Ink / Yarn Spinner | 适合复杂叙事，程序员友好 |
| 语音同步 | 语音时长 + 字幕同步 | 需要提前规划语音录制流程 |
| 本地化 | Key-Value语言文件 | 键名规范化，避免运行时找不到 |
| 历史记录 | 环形缓冲队列 | 支持玩家查看对话历史 |
| 存档支持 | 序列化变量字典 | 记录已触发的对话节点 |
