# 游戏玩家社交系统完全指南：好友、公会与社交图谱设计

## 一、社交系统的战略价值

社交系统是游戏**粘性**和**传播**的核心引擎。数据表明：

- 拥有 **≥3 个游戏好友**的玩家，30 日留存率是孤立玩家的 **2.7 倍**
- 加入公会的玩家每日在线时长平均高出 **45%**
- 通过好友推荐进入游戏的用户 LTV（生命周期价值）是广告投放用户的 **3.1 倍**

社交系统的技术挑战：
1. **实时性**：好友在线状态、公会消息需要毫秒级推送
2. **一致性**：多端同步、跨服社交的数据一致性
3. **规模性**：社交图谱随玩家增长呈非线性膨胀
4. **安全性**：防骚扰、封禁、隐私保护

## 三、好友系统核心实现

```csharp
// FriendSystem.cs
using System;
using System.Collections.Generic;
using System.Linq;
using Cysharp.Threading.Tasks;
using UnityEngine;

public class FriendSystem
{
    // 好友列表（有序：标星→在线→离线）
    private readonly Dictionary<ulong, FriendData>  _friends    = new();
    private readonly Dictionary<ulong, FriendRequest> _incomingReq = new();
    private readonly Dictionary<ulong, FriendRequest> _outgoingReq = new();
    
    // 屏蔽列表
    private readonly HashSet<ulong> _blockList = new();

    // 事件
    public event Action<FriendData>    OnFriendAdded;
    public event Action<ulong>         OnFriendRemoved;
    public event Action<ulong, PresenceStatus> OnPresenceChanged;
    public event Action<FriendRequest> OnFriendRequestReceived;

    private readonly INetworkService _network;

    public FriendSystem(INetworkService network)
    {
        _network = network;
        _network.OnMessage("social.friend.request",   HandleIncomingRequest);
        _network.OnMessage("social.friend.accepted",  HandleRequestAccepted);
        _network.OnMessage("social.friend.removed",   HandleFriendRemoved);
        _network.OnMessage("social.presence.update",  HandlePresenceUpdate);
    }

    // ===== 发送好友申请 =====
    public async UniTask<FriendAddResult> SendFriendRequest(ulong targetUid, string message = "")
    {
        if (_friends.ContainsKey(targetUid))
            return FriendAddResult.AlreadyFriend;
        if (_blockList.Contains(targetUid))
            return FriendAddResult.BlockedByYou;
        if (_outgoingReq.ContainsKey(targetUid))
            return FriendAddResult.RequestPending;

        var resp = await _network.SendAsync<FriendRequestResponse>(
            "social.friend.add",
            new { target_uid = targetUid, message = message }
        );

        if (resp.Success)
        {
            _outgoingReq[targetUid] = new FriendRequest
            {
                FromUid     = GetLocalUid(),
                ToUid       = targetUid,
                Message     = message,
                RequestTime = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds()
            };
        }

        return resp.Success ? FriendAddResult.Sent : (FriendAddResult)resp.ErrorCode;
    }

    // ===== 接受/拒绝好友申请 =====
    public async UniTask<bool> AcceptFriendRequest(ulong fromUid)
    {
        if (!_incomingReq.TryGetValue(fromUid, out var req)) return false;

        var resp = await _network.SendAsync<SimpleResponse>(
            "social.friend.accept",
            new { from_uid = fromUid }
        );

        if (resp.Success)
        {
            _incomingReq.Remove(fromUid);
            // 服务端会推送 friend.added 事件，届时统一处理
        }

        return resp.Success;
    }

    public async UniTask RejectFriendRequest(ulong fromUid)
    {
        if (!_incomingReq.ContainsKey(fromUid)) return;
        await _network.SendAsync<SimpleResponse>("social.friend.reject", new { from_uid = fromUid });
        _incomingReq.Remove(fromUid);
    }

    // ===== 删除好友 =====
    public async UniTask<bool> RemoveFriend(ulong uid)
    {
        if (!_friends.ContainsKey(uid)) return false;

        var resp = await _network.SendAsync<SimpleResponse>(
            "social.friend.remove", new { uid = uid });

        if (resp.Success)
        {
            _friends.Remove(uid);
            OnFriendRemoved?.Invoke(uid);
        }

        return resp.Success;
    }

    // ===== 屏蔽玩家 =====
    public async UniTask BlockPlayer(ulong uid)
    {
        _blockList.Add(uid);
        if (_friends.ContainsKey(uid))
        {
            _friends.Remove(uid);
            OnFriendRemoved?.Invoke(uid);
        }
        await _network.SendAsync<SimpleResponse>("social.block", new { uid = uid });
    }

    // ===== 获取排序后的好友列表 =====
    public IReadOnlyList<FriendData> GetSortedFriends()
    {
        return _friends.Values
            .OrderByDescending(f => f.IsStarred)           // 标星优先
            .ThenBy(f => f.Status == PresenceStatus.Offline) // 在线优先
            .ThenBy(f => f.Status)                          // 状态排序
            .ThenBy(f => f.NickName)                        // 名字字母序
            .ToList();
    }

    // ===== 修改备注名 =====
    public async UniTask SetRemark(ulong uid, string remark)
    {
        if (!_friends.TryGetValue(uid, out var friend)) return;
        friend.Remark = remark;
        await _network.SendAsync<SimpleResponse>(
            "social.friend.remark", new { uid = uid, remark = remark });
    }

    // ===== 推荐好友（基于共同游戏记录）=====
    public async UniTask<List<FriendData>> GetRecommendations(int count = 20)
    {
        var resp = await _network.SendAsync<FriendRecommendResponse>(
            "social.recommend", new { count = count });
        return resp.Players ?? new List<FriendData>();
    }

    // ===== 内部事件处理 =====
    private void HandleIncomingRequest(object data)
    {
        var req = ParseRequest<FriendRequest>(data);
        if (_blockList.Contains(req.FromUid)) return; // 屏蔽的人忽略
        _incomingReq[req.FromUid] = req;
        OnFriendRequestReceived?.Invoke(req);
    }

    private void HandleRequestAccepted(object data)
    {
        var friend = ParseRequest<FriendData>(data);
        _friends[friend.Uid] = friend;
        _outgoingReq.Remove(friend.Uid);
        OnFriendAdded?.Invoke(friend);
    }

    private void HandleFriendRemoved(object data)
    {
        ulong uid = ParseUid(data);
        _friends.Remove(uid);
        OnFriendRemoved?.Invoke(uid);
    }

    private void HandlePresenceUpdate(object data)
    {
        var update = ParseRequest<PresenceUpdate>(data);
        if (_friends.TryGetValue(update.Uid, out var friend))
        {
            friend.Status      = update.Status;
            friend.StatusMsg   = update.Msg;
            friend.CurrentScene = update.Scene;
            if (update.Status == PresenceStatus.Offline)
                friend.LastSeen = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds();
            OnPresenceChanged?.Invoke(update.Uid, update.Status);
        }
    }

    private ulong GetLocalUid() => GameSession.Current.Uid;
    private T ParseRequest<T>(object data) => JsonUtility.FromJson<T>(data.ToString());
    private ulong ParseUid(object data) => ulong.Parse(data.ToString());
}

public enum FriendAddResult
{
    Sent = 0, AlreadyFriend = 1, BlockedByYou = 2,
    RequestPending = 3, FriendListFull = 4, TargetListFull = 5
}
```

## 四、在线状态实时订阅（Presence System）

```csharp
// PresenceCache.cs
using System;
using System.Collections.Generic;
using Cysharp.Threading.Tasks;
using UnityEngine;

/// <summary>
/// 好友在线状态缓存器。
/// 使用 Push-on-Change 模式：服务端状态变化时主动推送，
/// 定期全量刷新作为补偿。
/// </summary>
public class PresenceCache
{
    private readonly Dictionary<ulong, PresenceInfo> _cache = new();
    private readonly INetworkService _network;
    
    // 本地假设所有好友的状态TTL（超过此时间未更新视为离线）
    private const int STATUS_TTL_SECONDS = 120;

    public event Action<ulong, PresenceInfo> OnPresenceUpdated;

    public PresenceCache(INetworkService network)
    {
        _network = network;
        _network.OnMessage("presence.push", OnPresencePush);
        StartPeriodicRefresh().Forget();
    }

    /// <summary>订阅一批玩家的状态变化（批量订阅减少网络往返）</summary>
    public async UniTask SubscribePresence(IEnumerable<ulong> uids)
    {
        var uidList = new List<ulong>(uids);
        if (uidList.Count == 0) return;

        var resp = await _network.SendAsync<PresenceBatchResponse>(
            "presence.subscribe", new { uids = uidList });

        foreach (var info in resp.Presences)
        {
            _cache[info.Uid] = info;
            info.LastUpdate = DateTime.UtcNow;
        }
    }

    /// <summary>获取指定玩家状态（本地缓存）</summary>
    public PresenceInfo GetPresence(ulong uid)
    {
        if (_cache.TryGetValue(uid, out var info))
        {
            // TTL检查
            if ((DateTime.UtcNow - info.LastUpdate).TotalSeconds > STATUS_TTL_SECONDS)
                info.Status = PresenceStatus.Offline;
            return info;
        }
        return new PresenceInfo { Uid = uid, Status = PresenceStatus.Offline };
    }

    /// <summary>主动拉取最新状态（用于进入好友列表界面时）</summary>
    public async UniTask RefreshAll(IEnumerable<ulong> uids)
    {
        await SubscribePresence(uids);
    }

    private void OnPresencePush(object data)
    {
        var info = JsonUtility.FromJson<PresenceInfo>(data.ToString());
        info.LastUpdate = DateTime.UtcNow;
        _cache[info.Uid] = info;
        OnPresenceUpdated?.Invoke(info.Uid, info);
    }

    // 定时全量刷新（补偿推送丢失场景）
    private async UniTaskVoid StartPeriodicRefresh()
    {
        while (true)
        {
            await UniTask.Delay(TimeSpan.FromSeconds(60));
            var uids = new List<ulong>(_cache.Keys);
            if (uids.Count > 0)
                await RefreshAll(uids);
        }
    }
}

[Serializable]
public class PresenceInfo
{
    public ulong          Uid;
    public PresenceStatus Status;
    public string         Msg;          // 状态签名
    public string         Scene;        // 当前场景
    public DateTime       LastUpdate;
}
```

## 五、公会系统

```csharp
// GuildSystem.cs
using System;
using System.Collections.Generic;
using System.Linq;
using Cysharp.Threading.Tasks;

public class GuildSystem
{
    private GuildData _myGuild;
    private readonly INetworkService _network;

    public GuildData MyGuild => _myGuild;
    public bool IsInGuild => _myGuild != null;

    public event Action<GuildData>   OnGuildJoined;
    public event Action              OnGuildLeft;
    public event Action<GuildLog>    OnGuildLog;
    public event Action<GuildMember> OnMemberJoined;
    public event Action<ulong>       OnMemberLeft;

    public GuildSystem(INetworkService network)
    {
        _network = network;
        _network.OnMessage("guild.member.join",      HandleMemberJoin);
        _network.OnMessage("guild.member.leave",     HandleMemberLeave);
        _network.OnMessage("guild.log",              HandleGuildLog);
        _network.OnMessage("guild.announcement",     HandleAnnouncement);
    }

    // ===== 创建公会 =====
    public async UniTask<GuildCreateResult> CreateGuild(string name, string tag)
    {
        if (IsInGuild) return GuildCreateResult.AlreadyInGuild;
        
        var resp = await _network.SendAsync<GuildCreateResponse>(
            "guild.create", new { name, tag });

        if (resp.Success)
        {
            _myGuild = resp.Guild;
            OnGuildJoined?.Invoke(_myGuild);
        }

        return resp.Success ? GuildCreateResult.Success : (GuildCreateResult)resp.ErrorCode;
    }

    // ===== 申请加入公会 =====
    public async UniTask<bool> ApplyToGuild(ulong guildId, string message = "")
    {
        if (IsInGuild) return false;
        var resp = await _network.SendAsync<SimpleResponse>(
            "guild.apply", new { guild_id = guildId, message });
        return resp.Success;
    }

    // ===== 邀请玩家 =====
    public async UniTask<bool> InvitePlayer(ulong uid)
    {
        if (!IsInGuild) return false;
        if (!CanManage()) return false;
        var resp = await _network.SendAsync<SimpleResponse>(
            "guild.invite", new { uid });
        return resp.Success;
    }

    // ===== 踢出成员 =====
    public async UniTask<bool> KickMember(ulong uid)
    {
        if (!CanManage()) return false;
        var member = GetMember(uid);
        if (member == null) return false;
        if (member.Role >= GetMyRole()) return false; // 不能踢比自己级别高的

        var resp = await _network.SendAsync<SimpleResponse>(
            "guild.kick", new { uid });
        if (resp.Success)
            _myGuild.Members.RemoveAll(m => m.Uid == uid);

        return resp.Success;
    }

    // ===== 调整职务 =====
    public async UniTask<bool> ChangeRole(ulong uid, GuildRole newRole)
    {
        if (GetMyRole() < GuildRole.Master) return false; // 只有会长可以任命
        var resp = await _network.SendAsync<SimpleResponse>(
            "guild.role", new { uid, role = (int)newRole });
        if (resp.Success)
        {
            var m = GetMember(uid);
            if (m != null) m.Role = newRole;
        }
        return resp.Success;
    }

    // ===== 发布公告 =====
    public async UniTask<bool> PostAnnouncement(string text)
    {
        if (GetMyRole() < GuildRole.Officer) return false;
        if (string.IsNullOrEmpty(text) || text.Length > 500) return false;
        
        var resp = await _network.SendAsync<SimpleResponse>(
            "guild.announcement", new { text });
        if (resp.Success && _myGuild != null)
            _myGuild.Announcement = text;
        return resp.Success;
    }

    // ===== 搜索公会 =====
    public async UniTask<List<GuildSearchResult>> SearchGuilds(
        string keyword = "", int minLevel = 0, bool openOnly = true)
    {
        var resp = await _network.SendAsync<GuildSearchResponse>(
            "guild.search", new { keyword, min_level = minLevel, open_only = openOnly });
        return resp.Results ?? new List<GuildSearchResult>();
    }

    // ===== 退出公会 =====
    public async UniTask<bool> LeaveGuild()
    {
        if (!IsInGuild) return false;
        if (GetMyRole() == GuildRole.Master)
            return false; // 会长必须先转让

        var resp = await _network.SendAsync<SimpleResponse>("guild.leave");
        if (resp.Success)
        {
            _myGuild = null;
            OnGuildLeft?.Invoke();
        }
        return resp.Success;
    }

    // ===== 按贡献排行 =====
    public List<GuildMember> GetRankedMembers(bool weekly = true)
    {
        if (_myGuild == null) return new List<GuildMember>();
        return _myGuild.Members
            .OrderByDescending(m => weekly ? m.WeeklyContrib : m.TotalContrib)
            .ToList();
    }

    // ===== 权限检查 =====
    private bool CanManage()
    {
        return _myGuild != null && GetMyRole() >= GuildRole.Officer;
    }

    private GuildRole GetMyRole()
    {
        var me = GetMember(GameSession.Current.Uid);
        return me?.Role ?? GuildRole.Member;
    }

    private GuildMember GetMember(ulong uid)
        => _myGuild?.Members.Find(m => m.Uid == uid);

    // ===== 服务端事件处理 =====
    private void HandleMemberJoin(object data)
    {
        var member = JsonUtility.FromJson<GuildMember>(data.ToString());
        _myGuild?.Members.Add(member);
        OnMemberJoined?.Invoke(member);
    }

    private void HandleMemberLeave(object data)
    {
        ulong uid = ulong.Parse(data.ToString());
        _myGuild?.Members.RemoveAll(m => m.Uid == uid);
        OnMemberLeft?.Invoke(uid);
    }

    private void HandleGuildLog(object data)
    {
        var log = JsonUtility.FromJson<GuildLog>(data.ToString());
        _myGuild?.Logs.Insert(0, log);
        if (_myGuild?.Logs.Count > 100)
            _myGuild.Logs.RemoveAt(100); // 只保留最近100条
        OnGuildLog?.Invoke(log);
    }

    private void HandleAnnouncement(object data)
    {
        if (_myGuild != null)
            _myGuild.Announcement = data.ToString();
    }
}

// 公会操作日志
[Serializable]
public class GuildLog
{
    public long   Time;
    public string OperatorName;
    public string TargetName;
    public string Action;     // "join" / "leave" / "kick" / "promote" / "contrib"
    public int    Value;      // 贡献值变化等
}

public enum GuildCreateResult { Success=0, AlreadyInGuild=1, NameTaken=2, TagTaken=3, LevelRequired=4 }
```

## 六、社交图谱数据结构

社交关系本质上是**有向图**（我关注你，不代表你关注我），好友是双向边的特例：

```csharp
// SocialGraph.cs
using System;
using System.Collections.Generic;
using System.Linq;

/// <summary>
/// 客户端本地社交图谱（有向图）
/// 节点：玩家UID  边：关系类型
/// 用于好友推荐、共同好友计算、社交路径查找
/// </summary>
public class SocialGraph
{
    // 邻接表：uid → [关联uid → 关系类型]
    private readonly Dictionary<ulong, Dictionary<ulong, RelationType>> _adjList = new();
    private ulong _selfUid;

    public void Initialize(ulong selfUid, IEnumerable<FriendData> friends)
    {
        _selfUid = selfUid;
        _adjList.Clear();
        _adjList[selfUid] = new Dictionary<ulong, RelationType>();

        foreach (var f in friends)
        {
            AddEdge(selfUid, f.Uid, RelationType.Friend);
            AddEdge(f.Uid, selfUid, RelationType.Friend);
        }
    }

    public void AddEdge(ulong from, ulong to, RelationType rel)
    {
        if (!_adjList.ContainsKey(from))
            _adjList[from] = new Dictionary<ulong, RelationType>();
        _adjList[from][to] = rel;
    }

    public void RemoveEdge(ulong from, ulong to)
    {
        if (_adjList.TryGetValue(from, out var edges))
            edges.Remove(to);
    }

    public RelationType GetRelation(ulong from, ulong to)
    {
        if (_adjList.TryGetValue(from, out var edges) && edges.TryGetValue(to, out var rel))
            return rel;
        return RelationType.None;
    }

    /// <summary>计算与目标玩家的共同好友</summary>
    public List<ulong> GetMutualFriends(ulong targetUid)
    {
        var myFriends = GetFriends(_selfUid);
        var targetFriends = GetFriends(targetUid);
        return myFriends.Intersect(targetFriends).ToList();
    }

    /// <summary>二度人脉：好友的好友（社交推荐基础）</summary>
    public List<(ulong Uid, int MutualCount)> GetSecondDegreeConnections(int topN = 50)
    {
        var myFriends    = new HashSet<ulong>(GetFriends(_selfUid));
        var candidates   = new Dictionary<ulong, int>(); // uid → 共同好友数
        
        foreach (var friendUid in myFriends)
        {
            foreach (var fof in GetFriends(friendUid))
            {
                if (fof == _selfUid) continue;
                if (myFriends.Contains(fof)) continue;
                
                candidates.TryGetValue(fof, out var count);
                candidates[fof] = count + 1;
            }
        }

        return candidates
            .OrderByDescending(kv => kv.Value)
            .Take(topN)
            .Select(kv => (kv.Key, kv.Value))
            .ToList();
    }

    /// <summary>BFS寻找两个玩家之间的最短社交路径</summary>
    public List<ulong> FindSocialPath(ulong targetUid, int maxDepth = 4)
    {
        if (!_adjList.ContainsKey(_selfUid)) return null;
        
        var visited = new HashSet<ulong> { _selfUid };
        var queue   = new Queue<List<ulong>>();
        queue.Enqueue(new List<ulong> { _selfUid });
        
        while (queue.Count > 0)
        {
            var path = queue.Dequeue();
            var last = path[^1];
            
            if (last == targetUid) return path;
            if (path.Count >= maxDepth + 1) continue;
            
            foreach (var neighbor in GetFriends(last))
            {
                if (visited.Contains(neighbor)) continue;
                visited.Add(neighbor);
                var newPath = new List<ulong>(path) { neighbor };
                queue.Enqueue(newPath);
            }
        }
        
        return null; // 超出最大深度
    }

    private IEnumerable<ulong> GetFriends(ulong uid)
    {
        if (!_adjList.TryGetValue(uid, out var edges)) yield break;
        foreach (var kv in edges)
            if (kv.Value == RelationType.Friend)
                yield return kv.Key;
    }

    public int GetFriendCount(ulong uid)
    {
        if (!_adjList.TryGetValue(uid, out var edges)) return 0;
        return edges.Count(kv => kv.Value == RelationType.Friend);
    }
}
```

```csharp
        public string  ActionParam; // 点击通知后的动作参数
    }

    private readonly List<SocialNotify>      _notifies = new();
    private readonly Queue<SocialNotify>     _pendingShow = new();

    public event Action<SocialNotify>        OnNewNotify;
    public event Action<int>                 OnUnreadCountChanged;

    public int UnreadCount => _notifies.Count(n => !n.IsRead);

    public void AddNotify(SocialNotify notify)
    {
        _notifies.Insert(0, notify);
        if (_notifies.Count > 200) _notifies.RemoveAt(200);
        
        _pendingShow.Enqueue(notify);
        OnNewNotify?.Invoke(notify);
        OnUnreadCountChanged?.Invoke(UnreadCount);
        
        // 播放通知音效
        PlayNotifySound(notify.Type);
    }

    // 批量标记已读
    public void MarkAllRead()
    {
        foreach (var n in _notifies) n.IsRead = true;
        OnUnreadCountChanged?.Invoke(0);
    }

    public void MarkRead(SocialNotifyType type)
    {
        foreach (var n in _notifies.Where(n => n.Type == type))
            n.IsRead = true;
        OnUnreadCountChanged?.Invoke(UnreadCount);
    }

    public IReadOnlyList<SocialNotify> GetNotifies(SocialNotifyType? filter = null)
    {
        if (filter == null) return _notifies;
        return _notifies.Where(n => n.Type == filter.Value).ToList();

    };
```

```csharp
    {
        if (string.IsNullOrEmpty(text)) return true;

            text = text.Replace(word, new string('*', word.Length));
        return text;
    }
```

```csharp
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using TMPro;

public class FriendListPresenter : MonoBehaviour

        _presenceCache = SocialManager.Instance.PresenceCache;
        
        _friendSystem.OnFriendAdded      += OnFriendAdded;
        _friendSystem.OnFriendRemoved    += OnFriendRemoved;
        _friendSystem.OnPresenceChanged  += OnPresenceChanged;
        
        searchInput.onValueChanged.AddListener(OnSearch);
        addFriendButton.onClick.AddListener(ShowAddFriendPanel);
        
        RefreshList();
    }

    private void RefreshList()
    {
        ClearList();
        var friends = _friendSystem.GetSortedFriends();
        int onlineCount = 0;
        
        foreach (var friend in friends)
        {
            if (friend.Status != PresenceStatus.Offline) onlineCount++;
            CreateItem(friend);
        }
        
        onlineCountText.text = $"{onlineCount}/{friends.Count} 在线";
    }

    private void CreateItem(FriendData friend)
    {
        var go   = Instantiate(friendItemPrefab, friendListContainer);
        var view = go.GetComponent<FriendItemView>();
        view.Bind(friend, OnItemAction);
        _itemViews.Add(view);
    }

    private void ClearList()
    {
        foreach (var v in _itemViews) Destroy(v.gameObject);
        _itemViews.Clear();
    }

    private void OnSearch(string keyword)
    {
        foreach (var view in _itemViews)
        {
            bool match = string.IsNullOrEmpty(keyword)
                || view.Data.NickName.Contains(keyword, System.StringComparison.OrdinalIgnoreCase)
                || (view.Data.Remark?.Contains(keyword, System.StringComparison.OrdinalIgnoreCase) ?? false);
            view.gameObject.SetActive(match);
        }
    }

    private void OnFriendAdded(FriendData f) => RefreshList();
    private void OnFriendRemoved(ulong uid)  => RefreshList();
    private void OnPresenceChanged(ulong uid, PresenceStatus status)
    {
        var view = _itemViews.Find(v => v.Data.Uid == uid);
        view?.UpdatePresence(status);
        RefreshOnlineCount();
    }

    private void RefreshOnlineCount()
    {
        int online = _friendSystem.GetSortedFriends().Count(f => f.Status != PresenceStatus.Offline);
        int total  = _friendSystem.GetSortedFriends().Count;
        onlineCountText.text = $"{online}/{total} 在线";
    }

    private void OnItemAction(FriendData friend, string action)
    {
        switch (action)
        {
            case "profile":   UIManager.Open<PlayerProfilePanel>(friend.Uid); break;
            case "chat":      UIManager.Open<ChatPanel>(friend.Uid); break;
            case "invite":    GameInviteManager.InviteFriend(friend.Uid).Forget(); break;
            case "remark":    UIManager.Open<RemarkPanel>(friend); break;
            case "remove":    ConfirmRemoveFriend(friend); break;
            case "block":     ConfirmBlockPlayer(friend); break;
        }
    }

    private void ShowAddFriendPanel() => UIManager.Open<AddFriendPanel>();
    
    private void ConfirmRemoveFriend(FriendData friend)
    {
        UIManager.ShowConfirm($"确定删除好友 {friend.NickName}？",
            async () => await _friendSystem.RemoveFriend(friend.Uid));
    }
    
    private void ConfirmBlockPlayer(FriendData friend)
    {
        UIManager.ShowConfirm($"屏蔽 {friend.NickName} 后将从好友列表移除，确定？",
            async () => await _friendSystem.BlockPlayer(friend.Uid));
    }

    private void OnDestroy()
    {
        if (_friendSystem != null)
        {
            _friendSystem.OnFriendAdded     -= OnFriendAdded;
            _friendSystem.OnFriendRemoved   -= OnFriendRemoved;
            _friendSystem.OnPresenceChanged -= OnPresenceChanged;
        }
    }
```

## 十、最佳实践总结

| 维度 | 推荐做法 | 反例 |
|------|---------|------|
| **状态推送** | Push-on-Change + 定时全量补偿 | 每秒轮询所有好友状态 |
| **图谱存储** | 客户端维护邻接表，增量更新 | 每次操作全量拉取 |
| **好友排序** | 标星→在线→状态类型→名字 | 无排序直接展示 |
| **权限校验** | 客户端预校验+服务端二次验证 | 只依赖客户端校验 |
| **敏感词过滤** | 服务端动态词库 + 客户端预过滤 | 硬编码词库 |
| **公会日志** | 只保留最近100条，超出时删除 | 无限增长内存 |
| **离线通知** | 利用推送通知（FCM/APNs）| 仅在线时通知 |
| **社交推荐** | 基于共同好友数 + 对战记录 | 纯随机推荐 |
| **隐私控制** | 细粒度可见性设置 | 无隐私控制 |
| **反骚扰** | 屏蔽列表 + 私信限制 + 举报系统 | 无任何限制 |

**核心设计原则**：
1. **事件驱动，不轮询**：在线状态、好友请求全部通过服务端 Push，客户端只订阅
2. **乐观更新**：操作后立即更新 UI，失败时回滚，保持响应性
3. **分层权限**：公会角色权限在服务端严格校验，客户端仅做 UI 层面的预判断
4. **社交图谱轻量化**：客户端仅维护直接好友节点，二度人脉等推荐在服务端计算
5. **隐私优先**：默认严格隐私设置，用户主动放开
6. **跨服社交**：好友 UID 包含服务器标识位，路由层透明处理跨服通信
