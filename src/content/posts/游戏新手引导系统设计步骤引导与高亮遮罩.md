## 一、引导步骤配置

```csharp
using System;
using UnityEngine;

public enum GuideActionType
{
    Tap,            // 点击指定UI
    TapAnywhere,    // 点击任意位置
    Wait,           // 等待条件满足
    ShowDialog,     // 显示对话框
    DragTo,         // 拖拽操作
}

[Serializable]
public class GuideStep
{
    public string StepId;
    public GuideActionType ActionType;
    
    [Header("高亮目标")]
    public string TargetUIName;    // UI对象名（FindObjectByName）
    public Vector2 TargetScreenPos; // 屏幕坐标（可选）
    
    [Header("文字提示")]
    [TextArea] public string TipText;
    public TipPosition TipPos;
    
    [Header("等待条件")]
    public string WaitConditionKey;  // 等待Blackboard中某个Key为true
    
    [Header("选项")]
    public bool CanSkip = false;
    public float AutoCompleteDelay;  // 自动完成延迟（0=不自动）
}

public enum TipPosition { Top, Bottom, Left, Right, Center }
```

## 二、引导管理器

```csharp
public class GuideManager : MonoBehaviour
{
    private static GuideManager instance;
    public static GuideManager Instance => instance;

    [SerializeField] private GuideStep[] allSteps;
    [SerializeField] private GuideOverlayUI overlayUI;
    
    private int currentStepIndex = -1;
    private bool isGuideActive;
    private const string PREF_COMPLETED = "guide_completed";
    private const string PREF_STEP = "guide_current_step";

    void Awake() { instance = this; }

    public void StartGuide()
    {
        if (PlayerPrefs.GetInt(PREF_COMPLETED, 0) == 1) return;
        
        int savedStep = PlayerPrefs.GetInt(PREF_STEP, 0);
        currentStepIndex = savedStep;
        isGuideActive = true;
        
        ShowCurrentStep();
    }

    void ShowCurrentStep()
    {
        if (currentStepIndex >= allSteps.Length)
        {
            CompleteGuide();
            return;
        }
        
        var step = allSteps[currentStepIndex];
        
        // 找到目标UI
        GameObject targetUI = null;
        if (!string.IsNullOrEmpty(step.TargetUIName))
            targetUI = GameObject.Find(step.TargetUIName);
        
        // 显示遮罩和高亮
        overlayUI?.ShowStep(step, targetUI);
        
        // 处理不同动作类型
        if (step.ActionType == GuideActionType.TapAnywhere)
        {
            overlayUI?.OnTapAnywhere(() => NextStep());
        }
        else if (step.ActionType == GuideActionType.ShowDialog)
        {
            if (step.AutoCompleteDelay > 0)
                StartCoroutine(AutoComplete(step.AutoCompleteDelay));
        }
        else if (step.ActionType == GuideActionType.Wait)
        {
            StartCoroutine(WaitForCondition(step.WaitConditionKey));
        }
    }

    System.Collections.IEnumerator WaitForCondition(string key)
    {
        while (true)
        {
            if (GuideConditionChecker.Check(key)) break;
            yield return new WaitForSeconds(0.5f);
        }
        NextStep();
    }

    System.Collections.IEnumerator AutoComplete(float delay)
    {
        yield return new WaitForSeconds(delay);
        NextStep();
    }

    public void NextStep()
    {
        currentStepIndex++;
        PlayerPrefs.SetInt(PREF_STEP, currentStepIndex);
        ShowCurrentStep();
    }

    void CompleteGuide()
    {
        isGuideActive = false;
        overlayUI?.Hide();
        PlayerPrefs.SetInt(PREF_COMPLETED, 1);
        Debug.Log("[Guide] 新手引导完成！");
    }

    public void SkipGuide()
    {
        CompleteGuide();
    }

    public bool IsGuideActive() => isGuideActive;
    
    /// <summary>
    /// 外部触发引导步骤完成（如特定操作完成后）
    /// </summary>
    public void CompleteCurrentStep(string expectedStepId = null)
    {
        if (!isGuideActive || currentStepIndex >= allSteps.Length) return;
        
        var current = allSteps[currentStepIndex];
        if (expectedStepId != null && current.StepId != expectedStepId) return;
        
        NextStep();
    }
}

public static class GuideConditionChecker
{
    private static Dictionary<string, Func<bool>> conditions 
        = new Dictionary<string, Func<bool>>();
    
    public static void Register(string key, Func<bool> condition) 
        => conditions[key] = condition;
    
    public static bool Check(string key) 
        => conditions.TryGetValue(key, out var c) && c();
}
```

## 三、高亮遮罩UI

```csharp
public class GuideOverlayUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.Image overlay;  // 半透明黑色遮罩
    [SerializeField] private UnityEngine.UI.Text tipText;
    [SerializeField] private GameObject handPrefab;  // 手指动画
    [SerializeField] private Material holeMaterial;  // 挖孔Shader材质
    
    public void ShowStep(GuideStep step, GameObject target)
    {
        gameObject.SetActive(true);
        tipText.text = step.TipText;
        
        if (target != null)
        {
            // 计算目标位置，设置遮罩挖孔
            var rt = target.GetComponent<RectTransform>();
            if (rt != null)
            {
                Vector3[] corners = new Vector3[4];
                rt.GetWorldCorners(corners);
                
                // 设置Shader挖孔区域
                Vector2 center = (corners[0] + corners[2]) / 2;
                Vector2 size = corners[2] - corners[0];
                holeMaterial.SetVector("_HoleCenter", center);
                holeMaterial.SetVector("_HoleSize", size);
                
                // 显示手指动画
                ShowHand(center);
            }
        }
    }

    void ShowHand(Vector2 pos)
    {
        var hand = Instantiate(handPrefab, transform);
        hand.transform.position = pos;
        // 播放点击动画
    }

    public void OnTapAnywhere(Action callback)
    {
        // 设置全屏点击区域
    }

    public void Hide()
    {
        gameObject.SetActive(false);
    }
}
```

## 四、引导设计原则

| 原则 | 说明 |
|------|------|
| 步骤最小化 | 只引导核心功能，避免教所有功能 |
| 可跳过 | 老玩家不被强制引导 |
| 进度保存 | 中途退出可续续引导 |
| 实时引导 | 在真实场景操作而非教学场景 |
| 操作反馈 | 操作正确立即给奖励 |
| A/B测试 | 测试不同引导路径的留存差异 |
