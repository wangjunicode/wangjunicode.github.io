## 一、大地图流式加载的挑战

开放世界游戏面临的核心技术挑战：

1. **内存限制**：移动设备可用内存通常仅有 1-3GB
2. **加载卡顿**：同步加载大场景导致明显帧率下降
3. **流畅过渡**：玩家移动时无感知地切换场景块
4. **复杂依赖**：地图块之间存在相互依赖（路径、光照、物件）
5. **多端适配**：PC/主机/移动端的分块粒度不同

## 二、世界分区（World Partition）架构

### 2.1 网格分区设计

```csharp
/// <summary>
/// 世界分区坐标
/// </summary>
[System.Serializable]
public struct ChunkCoord : IEquatable<ChunkCoord>
{
    public int X;
    public int Z;
    
    public ChunkCoord(int x, int z) { X = x; Z = z; }
    
    public static ChunkCoord FromWorldPos(Vector3 worldPos, float chunkSize)
    {
        return new ChunkCoord(
            Mathf.FloorToInt(worldPos.x / chunkSize),
            Mathf.FloorToInt(worldPos.z / chunkSize)
        );
    }
    
    public Vector3 ToWorldCenter(float chunkSize)
    {
        return new Vector3(
            (X + 0.5f) * chunkSize,
            0,
            (Z + 0.5f) * chunkSize
        );
    }
    
    public IEnumerable<ChunkCoord> GetNeighbors(int radius)
    {
        for (int dx = -radius; dx <= radius; dx++)
        for (int dz = -radius; dz <= radius; dz++)
            yield return new ChunkCoord(X + dx, Z + dz);
    }
    
    public float DistanceTo(ChunkCoord other)
    {
        int dx = X - other.X, dz = Z - other.Z;
        return Mathf.Sqrt(dx * dx + dz * dz);
    }
    
    public bool Equals(ChunkCoord other) => X == other.X && Z == other.Z;
    public override int GetHashCode() => X * 73856093 ^ Z * 19349663;
    public override string ToString() => $"Chunk({X},{Z})";
}
```

## 三、流式加载管理器

```csharp
/// <summary>
/// 世界流式加载管理器
/// </summary>
public class WorldStreamingManager : MonoBehaviour
{
    [Header("分区配置")]
    [SerializeField] private float chunkSize = 200f;        // 每块大小（米）
    [SerializeField] private int loadRadius = 2;            // 加载半径（块数）
    [SerializeField] private int lodRadius = 4;             // LOD加载半径
    [SerializeField] private int unloadRadius = 5;          // 卸载半径
    
    [Header("性能配置")]
    [SerializeField] private int maxConcurrentLoads = 3;    // 最大并发加载数
    [SerializeField] private float memoryBudgetMB = 1024f;  // 内存预算
    [SerializeField] private float updateInterval = 0.5f;   // 检测间隔（秒）
    
    [Header("调试")]
    [SerializeField] private bool showDebugGizmos = true;
    
    // 分块状态管理
    private Dictionary<ChunkCoord, ChunkLoadState> chunkStates 
        = new Dictionary<ChunkCoord, ChunkLoadState>();
    
    // 元数据索引
    private Dictionary<ChunkCoord, ChunkMetadata> metadataIndex 
        = new Dictionary<ChunkCoord, ChunkMetadata>();
    
    // 加载队列
    private PriorityQueue<ChunkLoadRequest> loadQueue 
        = new PriorityQueue<ChunkLoadRequest>();
    
    // 当前并发加载数
    private int currentConcurrentLoads;
    
    // 跟踪对象
    private List<Transform> trackedObjects = new List<Transform>();
    private ChunkCoord lastPlayerChunk;
    
    // 当前已加载内存（估算）
    private float currentMemoryUsageMB;
    
    private float updateTimer;

    public enum ChunkLoadState
    {
        Unloaded,
        QueuedForLoad,
        Loading,
        Loaded,
        LoadedLod,
        QueuedForUnload,
        Unloading
    }

    void Start()
    {
        // 加载世界元数据索引
        LoadWorldMetadata();
        
        // 注册玩家作为追踪对象
        var player = GameObject.FindGameObjectWithTag("Player");
        if (player) AddTrackedObject(player.transform);
    }

    void Update()
    {
        updateTimer -= Time.deltaTime;
        if (updateTimer > 0) return;
        updateTimer = updateInterval;
        
        // 检测追踪对象位置变化
        bool needUpdate = false;
        foreach (var tracked in trackedObjects)
        {
            if (tracked == null) continue;
            var newChunk = ChunkCoord.FromWorldPos(tracked.position, chunkSize);
            if (!newChunk.Equals(lastPlayerChunk))
            {
                lastPlayerChunk = newChunk;
                needUpdate = true;
                break;
            }
        }
        
        if (needUpdate)
        {
            EvaluateChunks();
        }
        
        // 处理加载队列
        ProcessLoadQueue();
    }

    void EvaluateChunks()
    {
        // 计算所有追踪对象周围需要加载的分块
        HashSet<ChunkCoord> requiredChunks = new HashSet<ChunkCoord>();
        HashSet<ChunkCoord> lodChunks = new HashSet<ChunkCoord>();
        
        foreach (var tracked in trackedObjects)
        {
            if (tracked == null) continue;
            var centerChunk = ChunkCoord.FromWorldPos(tracked.position, chunkSize);
            
            foreach (var neighbor in centerChunk.GetNeighbors(loadRadius))
                requiredChunks.Add(neighbor);
            
            foreach (var neighbor in centerChunk.GetNeighbors(lodRadius))
                lodChunks.Add(neighbor);
        }
        
        // 需要加载但当前未加载的分块 -> 加入加载队列
        foreach (var coord in requiredChunks)
        {
            var state = GetChunkState(coord);
            if (state == ChunkLoadState.Unloaded)
            {
                EnqueueLoad(coord, false);
            }
        }
        
        // LOD分块处理
        foreach (var coord in lodChunks)
        {
            if (requiredChunks.Contains(coord)) continue;
            var state = GetChunkState(coord);
            if (state == ChunkLoadState.Unloaded)
            {
                EnqueueLoad(coord, true);
            }
        }
        
        // 超出卸载半径的分块 -> 加入卸载队列
        foreach (var kvp in new Dictionary<ChunkCoord, ChunkLoadState>(chunkStates))
        {
            if (kvp.Value != ChunkLoadState.Loaded && 
                kvp.Value != ChunkLoadState.LoadedLod) continue;
            
            bool isRequired = false;
            foreach (var tracked in trackedObjects)
            {
                if (tracked == null) continue;
                var centerChunk = ChunkCoord.FromWorldPos(tracked.position, chunkSize);
                if (kvp.Key.DistanceTo(centerChunk) <= unloadRadius)
                {
                    isRequired = true;
                    break;
                }
            }
            
            if (!isRequired)
            {
                EnqueueUnload(kvp.Key);
            }
        }
    }

    void EnqueueLoad(ChunkCoord coord, bool isLod)
    {
        if (!metadataIndex.TryGetValue(coord, out var meta)) return;
        
        // 内存预算检查
        if (currentMemoryUsageMB + meta.EstimatedMemoryMB > memoryBudgetMB)
        {
            // 内存不足，触发紧急卸载
            TriggerEmergencyUnload(meta.EstimatedMemoryMB);
        }
        
        chunkStates[coord] = ChunkLoadState.QueuedForLoad;
        
        float priority = CalculateLoadPriority(coord, isLod, meta);
        loadQueue.Enqueue(new ChunkLoadRequest
        {
            Coord = coord,
            IsLod = isLod,
            Priority = priority
        });
    }

    float CalculateLoadPriority(ChunkCoord coord, bool isLod, ChunkMetadata meta)
    {
        float distancePriority = 0;
        foreach (var tracked in trackedObjects)
        {
            if (tracked == null) continue;
            var playerChunk = ChunkCoord.FromWorldPos(tracked.position, chunkSize);
            distancePriority = Mathf.Max(distancePriority, 
                1f / (coord.DistanceTo(playerChunk) + 1f));
        }
        
        return distancePriority * (isLod ? 0.5f : 1f) * (meta.LoadPriority / 10f + 0.5f);
    }

    void EnqueueUnload(ChunkCoord coord)
    {
        chunkStates[coord] = ChunkLoadState.QueuedForUnload;
        StartCoroutine(UnloadChunkAsync(coord));
    }

    void ProcessLoadQueue()
    {
        while (currentConcurrentLoads < maxConcurrentLoads && loadQueue.Count > 0)
        {
            var request = loadQueue.Dequeue();
            if (GetChunkState(request.Coord) == ChunkLoadState.QueuedForLoad)
            {
                currentConcurrentLoads++;
                StartCoroutine(LoadChunkAsync(request));
            }
        }
    }

    async System.Threading.Tasks.Task LoadChunkAsync_Task(ChunkLoadRequest request)
    {
        // 使用 UniTask / Addressables 实现
    }

    System.Collections.IEnumerator LoadChunkAsync(ChunkLoadRequest request)
    {
        chunkStates[request.Coord] = ChunkLoadState.Loading;
        
        if (!metadataIndex.TryGetValue(request.Coord, out var meta))
        {
            chunkStates[request.Coord] = ChunkLoadState.Unloaded;
            currentConcurrentLoads--;
            yield break;
        }
        
        // 先加载依赖分块
        foreach (var dep in meta.Dependencies)
        {
            // 等待依赖加载完成
            yield return new WaitUntil(() => IsDependencyLoaded(dep));
        }
        
        // 确定场景地址（LOD vs 完整）
        string sceneAddress = request.IsLod && meta.HasLod(0) ? 
            meta.LodSceneAddresses[0] : meta.SceneAddress;
        
        // 使用 Addressables 异步加载场景
        var handle = UnityEngine.AddressableAssets.Addressables
            .LoadSceneAsync(sceneAddress, 
                UnityEngine.SceneManagement.LoadSceneMode.Additive);
        
        yield return handle;
        
        if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
        {
            chunkStates[request.Coord] = request.IsLod ? 
                ChunkLoadState.LoadedLod : ChunkLoadState.Loaded;
            currentMemoryUsageMB += meta.EstimatedMemoryMB;
            
            // 通知系统分块加载完成
            OnChunkLoaded(request.Coord, handle.Result.Scene);
        }
        else
        {
            Debug.LogError($"Failed to load chunk {request.Coord}: {handle.OperationException}");
            chunkStates[request.Coord] = ChunkLoadState.Unloaded;
        }
        
        currentConcurrentLoads--;
    }

    System.Collections.IEnumerator UnloadChunkAsync(ChunkCoord coord)
    {
        chunkStates[coord] = ChunkLoadState.Unloading;
        
        // 等待当前帧结束，避免卸载正在使用的资源
        yield return new WaitForEndOfFrame();
        
        // 执行卸载
        if (loadedScenes.TryGetValue(coord, out var scene))
        {
            var handle = UnityEngine.SceneManagement.SceneManager
                .UnloadSceneAsync(scene);
            yield return handle;
            
            if (metadataIndex.TryGetValue(coord, out var meta))
                currentMemoryUsageMB -= meta.EstimatedMemoryMB;
            
            loadedScenes.Remove(coord);
        }
        
        chunkStates[coord] = ChunkLoadState.Unloaded;
        chunkStates.Remove(coord);
    }

    private Dictionary<ChunkCoord, UnityEngine.SceneManagement.Scene> loadedScenes 
        = new Dictionary<ChunkCoord, UnityEngine.SceneManagement.Scene>();

    void OnChunkLoaded(ChunkCoord coord, UnityEngine.SceneManagement.Scene scene)
    {
        loadedScenes[coord] = scene;
        
        // 通知其他系统
        EventManager.Trigger("ChunkLoaded", coord);
        
        // 检查POI预加载
        if (metadataIndex.TryGetValue(coord, out var meta))
        {
            foreach (var poi in meta.PointsOfInterest)
            {
                CheckPoiPreload(poi);
            }
        }
    }

    void CheckPoiPreload(PoiData poi)
    {
        // 基于玩家位置决定是否预加载POI周围的块
        foreach (var tracked in trackedObjects)
        {
            if (tracked == null) continue;
            float dist = Vector3.Distance(tracked.position, poi.WorldPosition);
            if (dist < poi.PreloadRadius)
            {
                var poiChunk = ChunkCoord.FromWorldPos(poi.WorldPosition, chunkSize);
                foreach (var neighbor in poiChunk.GetNeighbors(2))
                {
                    if (GetChunkState(neighbor) == ChunkLoadState.Unloaded)
                        EnqueueLoad(neighbor, false);
                }
            }
        }
    }

    void TriggerEmergencyUnload(float requiredMB)
    {
        // 按距离从远到近排序，优先卸载最远的分块
        var candidatesForUnload = new List<(ChunkCoord coord, float dist)>();
        
        foreach (var kvp in chunkStates)
        {
            if (kvp.Value != ChunkLoadState.Loaded && 
                kvp.Value != ChunkLoadState.LoadedLod) continue;
            
            float maxDist = 0;
            foreach (var tracked in trackedObjects)
            {
                if (tracked == null) continue;
                var playerChunk = ChunkCoord.FromWorldPos(tracked.position, chunkSize);
                maxDist = Mathf.Max(maxDist, kvp.Key.DistanceTo(playerChunk));
            }
            candidatesForUnload.Add((kvp.Key, maxDist));
        }
        
        candidatesForUnload.Sort((a, b) => b.dist.CompareTo(a.dist));
        
        float freed = 0;
        foreach (var (coord, _) in candidatesForUnload)
        {
            if (freed >= requiredMB) break;
            if (metadataIndex.TryGetValue(coord, out var meta))
            {
                EnqueueUnload(coord);
                freed += meta.EstimatedMemoryMB;
            }
        }
    }

    ChunkLoadState GetChunkState(ChunkCoord coord)
    {
        return chunkStates.TryGetValue(coord, out var state) ? state : ChunkLoadState.Unloaded;
    }

    bool IsDependencyLoaded(string dep) => true; // 简化实现

    void LoadWorldMetadata()
    {
        // 从 StreamingAssets 或 Addressables 加载世界元数据索引
        // 实际项目中可从 JSON/ScriptableObject 加载
    }

    public void AddTrackedObject(Transform t) => trackedObjects.Add(t);
    public void RemoveTrackedObject(Transform t) => trackedObjects.Remove(t);

    void OnDrawGizmos()
    {
        if (!showDebugGizmos) return;
        
        foreach (var kvp in chunkStates)
        {
            var center = kvp.Key.ToWorldCenter(chunkSize);
            center.y = 0.1f;
            
            Gizmos.color = kvp.Value switch
            {
                ChunkLoadState.Loaded => new Color(0, 1, 0, 0.2f),
                ChunkLoadState.LoadedLod => new Color(0, 0, 1, 0.2f),
                ChunkLoadState.Loading => new Color(1, 1, 0, 0.2f),
                ChunkLoadState.QueuedForLoad => new Color(1, 0.5f, 0, 0.2f),
                _ => new Color(1, 0, 0, 0.1f)
            };
            
            Gizmos.DrawCube(center, new Vector3(chunkSize * 0.95f, 1, chunkSize * 0.95f));
        }
    }
}

/// <summary>
/// 简易优先级队列
/// </summary>
public class PriorityQueue<T> where T : class
{
    private List<(float priority, T item)> items = new List<(float, T)>();
    public int Count => items.Count;
    
    public void Enqueue(T item, float priority = 0)
    {
        items.Add((priority, item));
        items.Sort((a, b) => b.priority.CompareTo(a.priority));
    }
    
    public T Dequeue()
    {
        if (items.Count == 0) return null;
        var item = items[0].item;
        items.RemoveAt(0);
        return item;
    }
}

public class ChunkLoadRequest
{
    public ChunkCoord Coord;
    public bool IsLod;
    public float Priority;
}
```

## 四、距离分级 LOD 策略

```
距离玩家   加载策略
0-200m     完整高质量场景块（全部细节、物理、AI）
200-500m   中等LOD场景块（减少物件、简化碰撞）
500-1000m  远景LOD块（仅静态网格、无碰撞）
1000m+     卸载或仅保留Impostor（布告板）
```

```csharp
/// <summary>
/// 动态LOD调度器
/// </summary>
public class ChunkLodScheduler : MonoBehaviour
{
    private WorldStreamingManager streamingManager;
    
    [Header("LOD距离阈值")]
    public float[] lodDistances = { 200f, 500f, 1000f };
    
    void Update()
    {
        // 每秒检查一次LOD升降级
        // 当分块从LOD变为全质量时，触发详细对象激活
    }
    
    public int GetDesiredLodLevel(float distanceToPlayer)
    {
        for (int i = 0; i < lodDistances.Length; i++)
        {
            if (distanceToPlayer < lodDistances[i]) return i;
        }
        return lodDistances.Length; // 超出范围 -> 卸载
    }
}
```

## 六、最佳实践总结

1. **分块大小选择**：PC 推荐 200-500m，移动端 100-200m
2. **异步优先**：所有加载操作必须异步，避免主线程卡顿
3. **内存预算**：为流式系统设置硬性内存上限，超出时主动卸载
4. **预加载**：根据玩家朝向和速度预测下一个需要加载的分块
5. **数据驱动**：所有分块配置通过 ScriptableObject / JSON 配置，支持热更新
6. **监控告警**：线上监控加载失败率、平均加载时长、内存超限次数
