## 一、Ragdoll布娃娃系统

```csharp
using UnityEngine;

/// <summary>
/// 布娃娃系统（动画↔物理动态切换）
/// </summary>
public class RagdollSystem : MonoBehaviour
{
    [Header("骨骼引用")]
    [SerializeField] private Animator animator;
    [SerializeField] private Rigidbody[] ragdollBodies;    // 所有布娃娃刚体
    [SerializeField] private Collider[] ragdollColliders;  // 所有布娃娃碰撞体
    [SerializeField] private Rigidbody hipBone;            // 根骨骼（臀部）
    
    [Header("起身配置")]
    [SerializeField] private float standUpDelay = 3f;     // 倒地后多久起身
    [SerializeField] private string standUpAnimFront = "StandUp_Front";
    [SerializeField] private string standUpAnimBack  = "StandUp_Back";
    
    private bool isRagdoll;
    private float ragdollStartTime;

    void Awake()
    {
        SetRagdollState(false); // 默认动画驱动
    }

    /// <summary>
    /// 切换布娃娃状态
    /// </summary>
    public void SetRagdollState(bool enable)
    {
        isRagdoll = enable;
        
        // 切换动画器
        animator.enabled = !enable;
        
        // 切换物理
        foreach (var rb in ragdollBodies)
        {
            rb.isKinematic = !enable;
        }
        
        foreach (var col in ragdollColliders)
        {
            col.enabled = enable;
        }
        
        if (enable)
        {
            ragdollStartTime = Time.time;
            StartCoroutine(TryStandUp());
        }
    }

    /// <summary>
    /// 受击触发布娃娃（带冲击力）
    /// </summary>
    public void Ragdoll(Vector3 hitForce, Vector3 hitPoint, ForceMode forceMode = ForceMode.Impulse)
    {
        SetRagdollState(true);
        
        // 找最近的刚体施加力
        Rigidbody closestRb = null;
        float minDist = float.MaxValue;
        
        foreach (var rb in ragdollBodies)
        {
            float dist = Vector3.Distance(rb.position, hitPoint);
            if (dist < minDist)
            {
                minDist = dist;
                closestRb = rb;
            }
        }
        
        closestRb?.AddForceAtPosition(hitForce, hitPoint, forceMode);
    }

    System.Collections.IEnumerator TryStandUp()
    {
        yield return new WaitForSeconds(standUpDelay);
        
        // 等待布娃娃静止
        yield return new WaitUntil(() => IsRagdollStill());
        
        // 根据朝向选择起身动画
        bool faceDown = Vector3.Dot(hipBone.transform.up, Vector3.down) > 0;
        string anim = faceDown ? standUpAnimFront : standUpAnimBack;
        
        StartCoroutine(BlendToAnimation(anim));
    }

    bool IsRagdollStill()
    {
        float totalVelocity = 0;
        foreach (var rb in ragdollBodies)
            totalVelocity += rb.velocity.magnitude;
        return totalVelocity < 0.5f;
    }

    System.Collections.IEnumerator BlendToAnimation(string animName)
    {
        // 1. 对齐根骨骼位置/旋转
        AlignTransformToRagdoll();
        
        // 2. 切回动画
        SetRagdollState(false);
        
        // 3. 播放起身动画
        animator.CrossFade(animName, 0.3f);
        
        yield return null;
    }

    void AlignTransformToRagdoll()
    {
        Vector3 ragdollPos = hipBone.position;
        ragdollPos.y = GetGroundHeight(ragdollPos);
        transform.position = ragdollPos;
        
        // 根据臀部朝向对齐Y轴旋转
        Vector3 forward = hipBone.transform.forward;
        forward.y = 0;
        if (forward.magnitude > 0.1f)
            transform.rotation = Quaternion.LookRotation(forward.normalized);
    }

    float GetGroundHeight(Vector3 pos)
    {
        if (Physics.Raycast(pos + Vector3.up * 2f, Vector3.down, out var hit, 5f))
            return hit.point.y;
        return pos.y;
    }
}
```

## 二、Hit Reaction（受击方向性）

```csharp
/// <summary>
/// 受击反应系统（方向性动画 + 物理冲量混合）
/// </summary>
public class HitReactionSystem : MonoBehaviour
{
    [SerializeField] private Animator animator;
    [SerializeField] private RagdollSystem ragdollSystem;
    [SerializeField] private float ragdollThreshold = 50f;  // 超过此伤害触发布娃娃
    
    private static readonly int HitDirectionX = Animator.StringToHash("HitDirectionX");
    private static readonly int HitDirectionY = Animator.StringToHash("HitDirectionY");
    private static readonly int IsHit = Animator.StringToHash("IsHit");

    public void OnHit(Vector3 hitDirection, float damage, Vector3 hitPoint)
    {
        if (damage >= ragdollThreshold)
        {
            // 强击触发布娃娃
            ragdollSystem?.Ragdoll(hitDirection * damage * 0.5f, hitPoint);
            return;
        }
        
        // 普通受击动画（方向性）
        // 将世界空间击中方向转换到本地空间
        Vector3 localHitDir = transform.InverseTransformDirection(hitDirection);
        
        // 归一化到 -1~1
        float hitX = Mathf.Clamp(localHitDir.x, -1f, 1f);
        float hitY = Mathf.Clamp(localHitDir.z, -1f, 1f); // 前后用Z
        
        animator.SetFloat(HitDirectionX, hitX);
        animator.SetFloat(HitDirectionY, hitY);
        animator.SetTrigger(IsHit);
        
        // 物理推力（不触发布娃娃，但有轻微位移）
        var rb = GetComponent<Rigidbody>();
        if (rb != null)
            rb.AddForce(hitDirection * (damage * 0.3f), ForceMode.Impulse);
    }
}
```

## 三、物理性能优化

```csharp
/// <summary>
/// 物理系统优化配置
/// </summary>
public class PhysicsOptimizer : MonoBehaviour
{
    void Awake()
    {
        // 全局物理设置
        Physics.defaultSolverIterations = 6;         // 默认6，降低可提升性能（但降低精度）
        Physics.defaultSolverVelocityIterations = 1;
        Physics.sleepThreshold = 0.005f;             // 刚体休眠阈值（越大越容易休眠）
        Physics.defaultContactOffset = 0.01f;        // 接触偏移
        Physics.queriesHitTriggers = false;           // 射线检测默认不碰触发器（性能优化）
        
        // 设置层碰撞矩阵（避免不必要的碰撞检测）
        SetupLayerCollisionMatrix();
    }

    void SetupLayerCollisionMatrix()
    {
        // 玩家子弹不与玩家自身碰撞
        int playerBulletLayer = LayerMask.NameToLayer("PlayerBullet");
        int playerLayer = LayerMask.NameToLayer("Player");
        if (playerBulletLayer >= 0 && playerLayer >= 0)
            Physics.IgnoreLayerCollision(playerBulletLayer, playerLayer, true);
        
        // UI射线检测不与物理碰撞
        int uiLayer = LayerMask.NameToLayer("UI");
        if (uiLayer >= 0)
            for (int i = 0; i < 32; i++)
                Physics.IgnoreLayerCollision(uiLayer, i, true);
    }
}
```

## 五、物理系统设计要点

| 要点 | 方案 |
|------|------|
| 布娃娃状态机 | 动画/物理清晰分离，切换时对齐根骨骼 |
| 层碰撞矩阵 | 关闭不必要的层间碰撞，减少碰撞对 |
| CCD连续碰撞 | 高速子弹开启CCD防穿墙，其他物体不开 |
| 刚体休眠 | 合理设置休眠阈值，静止物体零开销 |
| 物理固定步长 | FixedDeltaTime=0.02(50Hz)，移动端可降低到0.033(30Hz) |
