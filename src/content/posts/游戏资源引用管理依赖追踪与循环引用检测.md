## 一、资源引用问题类型

| 问题类型 | 表现 | 影响 |
|----------|------|------|
| 循环依赖 | Bundle A 依赖 B，B 依赖 A | 加载死锁/无法卸载 |
| 冗余打包 | 同一资源被多个Bundle包含 | 包体增大 |
| 空引用 | 序列化字段引用了已删除的资源 | 运行时报错 |
| 未释放引用 | 加载后未调用 Release | 内存泄漏 |
| 跨Bundle直接引用 | Inspector 跨Bundle拖拽引用 | 隐式依赖，难以管理 |

## 二、编辑器资源分析工具

```csharp
#if UNITY_EDITOR
using UnityEngine;
using UnityEditor;
using System.Collections.Generic;
using System.Linq;

/// <summary>
/// 资源依赖关系分析工具
/// </summary>
public class AssetDependencyAnalyzer : EditorWindow
{
    [MenuItem("Tools/Game/资源依赖分析")]
    static void Open() => GetWindow<AssetDependencyAnalyzer>("依赖分析");

    private Vector2 scrollPos;
    private string analysisRoot = "Assets";
    private List<DependencyNode> nodes = new List<DependencyNode>();
    private List<string[]> circularDeps = new List<string[]>();
    private List<RedundantAsset> redundantAssets = new List<RedundantAsset>();

    public class DependencyNode
    {
        public string AssetPath;
        public string AssetGuid;
        public long FileSize;
        public List<string> Dependencies = new List<string>();
        public List<string> Dependents = new List<string>(); // 被哪些资源依赖
    }

    public class RedundantAsset
    {
        public string AssetPath;
        public List<string> IncludedInBundles = new List<string>();
    }

    void OnGUI()
    {
        GUILayout.Label("资源依赖分析工具", EditorStyles.boldLabel);
        
        EditorGUILayout.BeginHorizontal();
        analysisRoot = EditorGUILayout.TextField("分析根目录", analysisRoot);
        if (GUILayout.Button("选择", GUILayout.Width(60)))
        {
            string path = EditorUtility.OpenFolderPanel("选择目录", "Assets", "");
            if (!string.IsNullOrEmpty(path))
                analysisRoot = "Assets" + path.Replace(Application.dataPath, "");
        }
        EditorGUILayout.EndHorizontal();
        
        EditorGUILayout.Space();
        
        EditorGUILayout.BeginHorizontal();
        if (GUILayout.Button("分析循环依赖")) AnalyzeCircularDependencies();
        if (GUILayout.Button("查找冗余资源")) FindRedundantAssets();
        if (GUILayout.Button("查找孤立资源")) FindOrphanAssets();
        EditorGUILayout.EndHorizontal();
        
        EditorGUILayout.Space();
        
        if (circularDeps.Count > 0)
        {
            EditorGUILayout.HelpBox(
                $"发现 {circularDeps.Count} 个循环依赖！", MessageType.Error);
            
            scrollPos = EditorGUILayout.BeginScrollView(scrollPos, GUILayout.Height(200));
            foreach (var cycle in circularDeps)
            {
                EditorGUILayout.LabelField(
                    $"循环: {string.Join(" → ", cycle)} → {cycle[0]}", 
                    EditorStyles.miniLabel);
            }
            EditorGUILayout.EndScrollView();
        }
        
        if (redundantAssets.Count > 0)
        {
            EditorGUILayout.HelpBox(
                $"发现 {redundantAssets.Count} 个冗余资源", MessageType.Warning);
            
            scrollPos = EditorGUILayout.BeginScrollView(scrollPos, GUILayout.Height(200));
            foreach (var asset in redundantAssets)
            {
                EditorGUILayout.BeginHorizontal();
                if (GUILayout.Button(
                    System.IO.Path.GetFileName(asset.AssetPath), 
                    EditorStyles.linkLabel))
                {
                    Selection.activeObject = AssetDatabase.LoadAssetAtPath<Object>(asset.AssetPath);
                    EditorGUIUtility.PingObject(Selection.activeObject);
                }
                EditorGUILayout.LabelField(
                    $"被包含在: {string.Join(", ", asset.IncludedInBundles)}");
                EditorGUILayout.EndHorizontal();
            }
            EditorGUILayout.EndScrollView();
        }
    }

    void AnalyzeCircularDependencies()
    {
        circularDeps.Clear();
        
        // 构建依赖图
        var allAssets = AssetDatabase.FindAssets("", new[] { analysisRoot });
        var dependencyGraph = new Dictionary<string, HashSet<string>>();
        
        int total = allAssets.Length;
        for (int i = 0; i < total; i++)
        {
            if (i % 50 == 0)
                EditorUtility.DisplayProgressBar("分析中...", 
                    $"{i}/{total}", (float)i / total);
            
            string guid = allAssets[i];
            string path = AssetDatabase.GUIDToAssetPath(guid);
            
            if (path.StartsWith("Packages/")) continue;
            
            var deps = AssetDatabase.GetDependencies(path, false);
            dependencyGraph[path] = new HashSet<string>(deps.Where(d => d != path));
        }
        
        EditorUtility.ClearProgressBar();
        
        // DFS 检测循环
        var visited = new HashSet<string>();
        var recursionStack = new List<string>();
        
        foreach (var node in dependencyGraph.Keys)
        {
            if (!visited.Contains(node))
                DetectCycle(node, dependencyGraph, visited, recursionStack);
        }
        
        Debug.Log($"[DependencyAnalyzer] 分析完成，发现 {circularDeps.Count} 个循环依赖");
    }

    void DetectCycle(string node, Dictionary<string, HashSet<string>> graph,
        HashSet<string> visited, List<string> stack)
    {
        visited.Add(node);
        stack.Add(node);
        
        if (graph.TryGetValue(node, out var deps))
        {
            foreach (var dep in deps)
            {
                if (!visited.Contains(dep))
                {
                    DetectCycle(dep, graph, visited, stack);
                }
                else if (stack.Contains(dep))
                {
                    // 发现循环
                    int cycleStart = stack.IndexOf(dep);
                    var cycle = stack.GetRange(cycleStart, stack.Count - cycleStart).ToArray();
                    circularDeps.Add(cycle);
                }
            }
        }
        
        stack.Remove(node);
    }

    void FindRedundantAssets()
    {
        redundantAssets.Clear();
        
        // 获取 Addressables 组信息
        // 简化：检查是否同一资源在多个 AssetBundle 标签中
        var allAssets = AssetDatabase.FindAssets("", new[] { analysisRoot });
        var assetBundleMap = new Dictionary<string, List<string>>();
        
        foreach (var guid in allAssets)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            string bundleName = AssetDatabase.GetImplicitAssetBundleName(path);
            
            if (string.IsNullOrEmpty(bundleName)) continue;
            
            if (!assetBundleMap.ContainsKey(path))
                assetBundleMap[path] = new List<string>();
            assetBundleMap[path].Add(bundleName);
        }
        
        foreach (var kv in assetBundleMap)
        {
            if (kv.Value.Count > 1)
            {
                redundantAssets.Add(new RedundantAsset
                {
                    AssetPath = kv.Key,
                    IncludedInBundles = kv.Value
                });
            }
        }
        
        Debug.Log($"[DependencyAnalyzer] 发现 {redundantAssets.Count} 个冗余资源");
    }

    void FindOrphanAssets()
    {
        // 找出没有被任何其他资源引用的资源（可能是无用资源）
        var allAssets = AssetDatabase.FindAssets("t:Object", new[] { analysisRoot });
        var referencedPaths = new HashSet<string>();
        
        foreach (var guid in allAssets)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            var deps = AssetDatabase.GetDependencies(path, false);
            foreach (var dep in deps)
                referencedPaths.Add(dep);
        }
        
        int orphanCount = 0;
        foreach (var guid in allAssets)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            if (!referencedPaths.Contains(path) && !IsEntryPoint(path))
            {
                orphanCount++;
                Debug.Log($"[Orphan] {path}");
            }
        }
        
        Debug.Log($"[DependencyAnalyzer] 发现 {orphanCount} 个孤立资源");
        EditorUtility.DisplayDialog("孤立资源", 
            $"发现 {orphanCount} 个孤立资源，已输出到 Console", "确定");
    }

    bool IsEntryPoint(string path)
    {
        // 以下类型不算孤立资源（它们是入口点）
        return path.EndsWith(".unity") ||
               path.EndsWith(".prefab") && path.Contains("Resources") ||
               path.Contains("Resources/");
    }
}
#endif
```

## 三、运行时资源引用计数

```csharp
/// <summary>
/// 资源引用计数管理（防止过早卸载或泄漏）
/// </summary>
public class AssetReferenceTracker
{
    private Dictionary<string, int> refCounts = new Dictionary<string, int>();
    private Dictionary<string, object> loadedAssets = new Dictionary<string, object>();

    /// <summary>
    /// 持有资源引用
    /// </summary>
    public void AddRef(string assetPath)
    {
        if (refCounts.ContainsKey(assetPath))
            refCounts[assetPath]++;
        else
            refCounts[assetPath] = 1;
    }

    /// <summary>
    /// 释放资源引用（引用数归零时卸载）
    /// </summary>
    public void Release(string assetPath)
    {
        if (!refCounts.ContainsKey(assetPath))
        {
            Debug.LogWarning($"[AssetTracker] Release 了未追踪的资源: {assetPath}");
            return;
        }
        
        refCounts[assetPath]--;
        
        if (refCounts[assetPath] <= 0)
        {
            refCounts.Remove(assetPath);
            loadedAssets.Remove(assetPath);
            
            // 通知 Addressables 卸载
            Debug.Log($"[AssetTracker] 卸载资源: {assetPath}");
        }
    }

    public int GetRefCount(string assetPath)
    {
        return refCounts.TryGetValue(assetPath, out int count) ? count : 0;
    }

    /// <summary>
    /// 打印所有持有中的资源（调试用）
    /// </summary>
    public void DumpAllRefs()
    {
        Debug.Log($"[AssetTracker] 当前持有 {refCounts.Count} 个资源引用：");
        foreach (var kv in refCounts)
            Debug.Log($"  {kv.Key}: refs={kv.Value}");
    }
}
```

## 四、CI 集成资源检查

```yaml
# .github/workflows/asset-check.yml
# 每次 PR 自动检查资源问题
name: Asset Check

on: [pull_request]

jobs:
  asset-check:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Run Asset Reference Check
        uses: game-ci/unity-test-runner@v4
        with:
          testMode: editmode
          customParameters: -executeMethod AssetCheckCI.RunAllChecks
        env:
          UNITY_LICENSE: ${{ secrets.UNITY_LICENSE }}
      
      - name: Check for failures
        run: |
          if grep -l "ASSET_CHECK_FAILED" test-results/; then
            echo "资源检查发现问题！"
            exit 1
          fi
```

```csharp
#if UNITY_EDITOR
/// <summary>
/// CI 资源检查入口（被 CI/CD 调用）
/// </summary>
public static class AssetCheckCI
{
    public static void RunAllChecks()
    {
        int failCount = 0;
        
        // 检查 Missing Script
        failCount += CheckMissingScripts();
        
        // 检查超大资源
        failCount += CheckOversizedAssets();
        
        // 检查命名规范
        failCount += CheckNamingConventions();
        
        if (failCount > 0)
        {
            Debug.LogError($"ASSET_CHECK_FAILED: {failCount} 个问题");
            EditorApplication.Exit(1);
        }
        else
        {
            Debug.Log("ASSET_CHECK_PASSED");
            EditorApplication.Exit(0);
        }
    }

    static int CheckMissingScripts()
    {
        int count = 0;
        var prefabs = AssetDatabase.FindAssets("t:Prefab");
        
        foreach (var guid in prefabs)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            var prefab = AssetDatabase.LoadAssetAtPath<GameObject>(path);
            
            if (prefab == null) continue;
            
            foreach (var comp in prefab.GetComponentsInChildren<Component>(true))
            {
                if (comp == null)
                {
                    Debug.LogError($"Missing Script in: {path}");
                    count++;
                }
            }
        }
        
        return count;
    }

    static int CheckOversizedAssets()
    {
        int count = 0;
        const long MAX_TEXTURE_SIZE = 4 * 1024 * 1024; // 4MB
        
        var textures = AssetDatabase.FindAssets("t:Texture2D", new[] { "Assets" });
        foreach (var guid in textures)
        {
            string path = AssetDatabase.GUIDToAssetPath(guid);
            long size = new System.IO.FileInfo(path).Length;
            
            if (size > MAX_TEXTURE_SIZE)
            {
                Debug.LogError($"Oversized texture: {path} ({size / 1024f / 1024f:F1}MB)");
                count++;
            }
        }
        
        return count;
    }

    static int CheckNamingConventions()
    {
        // 检查资源命名（自定义规范）
        return 0;
    }
}
#endif
```

## 五、资源管理规范

| 规范 | 要求 |
|------|------|
| 命名规范 | Texture: T_XXX, Prefab: P_XXX, Material: M_XXX |
| Bundle 分组 | 按使用场景分组（不要所有资源一个Bundle）|
| 依赖关系 | 禁止循环依赖，共享资源单独放共享Bundle |
| 冗余检查 | 每次提交 PR 时自动检查 |
| 孤立资源 | 定期清理未被引用的资源 |
