## 一、强化配置数据

```csharp
using System;
using UnityEngine;

[Serializable]
public class EnhanceLevelConfig
{
    public int Level;              // 强化等级（+1到+15）
    public float SuccessRate;      // 成功概率（0-1）
    public float FailRate;         // 失败降级概率（剩余为保持）
    public float DestroyRate;      // 装备毁坏概率（高等级时有）
    public int MaterialCost;       // 强化石消耗
    public long GoldCost;          // 金币消耗
    public float AttackBonus;      // 强化成功后额外攻击加成（累加）
    public float DefenseBonus;     // 防御加成
}

[CreateAssetMenu(menuName = "Game/Enhance Config")]
public class EnhanceConfig : ScriptableObject
{
    public EnhanceLevelConfig[] Levels;
    public int MaxLevel = 15;
    
    public EnhanceLevelConfig GetConfig(int level)
    {
        if (level <= 0 || level > Levels.Length) return null;
        return Levels[level - 1];
    }
}
```

## 二、强化系统

```csharp
public class EquipmentEnhancer : MonoBehaviour
{
    [SerializeField] private EnhanceConfig config;
    
    public event Action<EnhanceResult> OnEnhanceComplete;

    public enum EnhanceResultType { Success, FailKeep, FailDegrade, Destroy }

    [Serializable]
    public class EnhanceResult
    {
        public EnhanceResultType Type;
        public int OldLevel;
        public int NewLevel;
        public string EquipmentId;
    }

    /// <summary>
    /// 执行强化（客户端预演，服务端验证）
    /// </summary>
    public async System.Threading.Tasks.Task<EnhanceResult> Enhance(
        string equipmentId, int currentLevel, bool useProtection)
    {
        var cfg = config.GetConfig(currentLevel + 1);
        if (cfg == null) return null;
        
        // 检查资源
        if (!HasEnoughResources(cfg, useProtection))
        {
            UIManager.Instance?.ShowMessage("资源不足");
            return null;
        }
        
        // 发送服务端请求（服务端计算真实结果）
        var result = await NetworkService.EnhanceEquipment(
            equipmentId, currentLevel, useProtection);
        
        if (result != null)
        {
            OnEnhanceComplete?.Invoke(result);
            ShowEnhanceEffect(result);
        }
        
        return result;
    }

    bool HasEnoughResources(EnhanceLevelConfig cfg, bool useProtection)
    {
        // 检查强化石/金币
        return true; // 实际需要检查背包和货币
    }

    void ShowEnhanceEffect(EnhanceResult result)
    {
        switch (result.Type)
        {
            case EnhanceResultType.Success:
                UIManager.Instance?.ShowMessage($"强化成功！装备达到+{result.NewLevel}");
                // 播放金色爆炸特效
                break;
            case EnhanceResultType.FailDegrade:
                UIManager.Instance?.ShowMessage($"强化失败，装备降为+{result.NewLevel}");
                // 播放失败特效
                break;
            case EnhanceResultType.Destroy:
                UIManager.Instance?.ShowMessage("装备已损毁！");
                // 播放装备碎裂特效
                break;
        }
    }
}
```

## 三、强化等级发光效果

```csharp
public class EquipmentGlowEffect : MonoBehaviour
{
    [SerializeField] private Renderer equipRenderer;
    [SerializeField] private Material[] glowMaterials; // 每5级换一套材质

    public void UpdateGlow(int enhanceLevel)
    {
        if (enhanceLevel <= 0) { ResetGlow(); return; }
        
        // +1~+5: 白色微光
        // +6~+10: 蓝色发光
        // +11~+15: 金色强发光
        int tier = (enhanceLevel - 1) / 5;
        tier = Mathf.Clamp(tier, 0, glowMaterials.Length - 1);
        
        if (equipRenderer != null)
            equipRenderer.material = glowMaterials[tier];
        
        // 调整发光强度
        float intensity = 0.5f + (enhanceLevel % 5) * 0.1f;
        equipRenderer.material.SetFloat("_GlowIntensity", intensity);
    }

    void ResetGlow()
    {
        if (glowMaterials.Length > 0)
            equipRenderer.material = glowMaterials[0];
    }
}
```

## 四、强化系统设计规范

| 强化段位 | 成功率 | 风险 | 保护 |
|----------|--------|------|------|
| +1~+5 | 100% | 无 | 不需要 |
| +6~+8 | 80% | 失败保持 | 可选 |
| +9~+12 | 50% | 失败降级 | 推荐 |
| +13~+15 | 20% | 可能毁坏 | 必须 |

**说明：** 高风险段位是商业化变现的关键节点，玩家为规避风险会购买保护道具。但毁坏机制需谨慎，过高会严重影响玩家体验。
