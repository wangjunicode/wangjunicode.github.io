## 一、RenderTexture 小地图方案

```csharp
using UnityEngine;
using UnityEngine.UI;
using System.Collections.Generic;

/// <summary>
/// 小地图摄像机控制器
/// </summary>
public class MinimapCamera : MonoBehaviour
{
    [Header("摄像机配置")]
    [SerializeField] private Camera minimapCam;
    [SerializeField] private Transform followTarget;       // 跟随的目标（玩家）
    [SerializeField] private float height = 50f;           // 俯视高度
    [SerializeField] private float orthographicSize = 30f; // 视野大小（缩放）
    [SerializeField] private bool rotateWithPlayer = true; // 是否跟随玩家朝向旋转
    
    [Header("RenderTexture")]
    [SerializeField] private RenderTexture minimapRT;      // 320×320 或 512×512
    
    [Header("缩放")]
    [SerializeField] private float minZoom = 10f;
    [SerializeField] private float maxZoom = 100f;
    [SerializeField] private float zoomSpeed = 5f;
    
    private float currentZoom;

    void Awake()
    {
        currentZoom = orthographicSize;
        minimapCam.targetTexture = minimapRT;
        minimapCam.orthographic = true;
        minimapCam.orthographicSize = currentZoom;
        
        // 只渲染指定层（小地图层，减少渲染开销）
        minimapCam.cullingMask = LayerMask.GetMask("Default", "MinimapIcon");
    }

    void LateUpdate()
    {
        if (followTarget == null) return;
        
        // 跟随目标位置
        Vector3 pos = followTarget.position;
        transform.position = new Vector3(pos.x, pos.y + height, pos.z);
        
        // 朝向
        if (rotateWithPlayer)
            transform.rotation = Quaternion.Euler(90f, followTarget.eulerAngles.y, 0f);
        else
            transform.rotation = Quaternion.Euler(90f, 0f, 0f);
        
        // 缩放（鼠标滚轮）
        float scroll = Input.GetAxis("Mouse ScrollWheel");
        if (Mathf.Abs(scroll) > 0.01f)
        {
            currentZoom = Mathf.Clamp(
                currentZoom - scroll * zoomSpeed * currentZoom,
                minZoom, maxZoom);
            minimapCam.orthographicSize = currentZoom;
        }
    }
}
```

## 二、小地图图标系统

```csharp
/// <summary>
/// 小地图图标注册接口（挂在需要显示在小地图上的对象）
/// </summary>
public class MinimapIcon : MonoBehaviour
{
    [SerializeField] private Sprite iconSprite;
    [SerializeField] private Color iconColor = Color.white;
    [SerializeField] private float iconSize = 10f;
    [SerializeField] private MinimapIconType iconType;
    [SerializeField] private bool rotateWithObject = true;  // 图标是否跟随朝向旋转
    [SerializeField] private bool showOnEdge = true;        // 超出小地图时在边缘显示

    public MinimapIconType IconType => iconType;
    public Sprite Sprite => iconSprite;
    public Color Color => iconColor;
    public float Size => iconSize;
    public bool RotateWithObject => rotateWithObject;
    public bool ShowOnEdge => showOnEdge;

    void OnEnable()  => MinimapManager.Instance?.RegisterIcon(this);
    void OnDisable() => MinimapManager.Instance?.UnregisterIcon(this);
}

public enum MinimapIconType
{
    Player,         // 玩家（永远可见）
    Ally,           // 队友
    Enemy,          // 敌人（只在视野内显示）
    NPC,            // NPC
    QuestTarget,    // 任务目标
    Resource,       // 采集点
    POI,            // 兴趣点
    Waypoint        // 路径点
}

/// <summary>
/// 小地图 UI 管理器
/// </summary>
public class MinimapManager : MonoBehaviour
{
    private static MinimapManager instance;
    public static MinimapManager Instance => instance;

    [Header("UI 引用")]
    [SerializeField] private RawImage minimapImage;         // 显示 RenderTexture
    [SerializeField] private Transform iconContainer;       // 图标容器
    [SerializeField] private GameObject iconPrefab;         // 图标预制体
    
    [Header("配置")]
    [SerializeField] private MinimapCamera minimapCamera;
    [SerializeField] private float mapWorldRadius = 100f;   // 小地图覆盖的世界半径

    private Dictionary<MinimapIcon, RectTransform> iconInstances 
        = new Dictionary<MinimapIcon, RectTransform>();
    
    private RectTransform mapRect;

    void Awake()
    {
        instance = this;
        mapRect = minimapImage.GetComponent<RectTransform>();
    }

    public void RegisterIcon(MinimapIcon icon)
    {
        if (iconInstances.ContainsKey(icon)) return;
        
        var go = Instantiate(iconPrefab, iconContainer);
        var img = go.GetComponent<Image>();
        img.sprite = icon.Sprite;
        img.color = icon.Color;
        go.GetComponent<RectTransform>().sizeDelta = Vector2.one * icon.Size;
        
        iconInstances[icon] = go.GetComponent<RectTransform>();
    }

    public void UnregisterIcon(MinimapIcon icon)
    {
        if (iconInstances.TryGetValue(icon, out var rt))
        {
            Destroy(rt.gameObject);
            iconInstances.Remove(icon);
        }
    }

    void LateUpdate()
    {
        foreach (var kv in iconInstances)
        {
            var icon = kv.Key;
            var iconRect = kv.Value;
            
            if (icon == null || icon.gameObject == null) continue;
            
            UpdateIconPosition(icon, iconRect);
        }
    }

    void UpdateIconPosition(MinimapIcon icon, RectTransform iconRect)
    {
        // 计算图标在小地图上的归一化位置
        Vector3 relativePos = icon.transform.position - minimapCamera.followTarget.position;
        
        // 旋转补偿（如果小地图跟随玩家旋转）
        float mapAngle = -minimapCamera.transform.eulerAngles.y;
        float cosA = Mathf.Cos(mapAngle * Mathf.Deg2Rad);
        float sinA = Mathf.Sin(mapAngle * Mathf.Deg2Rad);
        
        float rotX = relativePos.x * cosA - relativePos.z * sinA;
        float rotZ = relativePos.x * sinA + relativePos.z * cosA;
        
        // 归一化到 -1 到 1
        float normX = rotX / minimapCamera.GetComponent<Camera>().orthographicSize;
        float normZ = rotZ / minimapCamera.GetComponent<Camera>().orthographicSize;
        
        // 是否超出小地图范围
        float dist = new Vector2(normX, normZ).magnitude;
        bool isOutside = dist > 1f;
        
        if (isOutside && icon.ShowOnEdge)
        {
            // 边缘指示器
            normX /= dist;
            normZ /= dist;
            normX *= 0.95f; // 略微缩进，不贴边
            normZ *= 0.95f;
        }
        else if (isOutside && !icon.ShowOnEdge)
        {
            iconRect.gameObject.SetActive(false);
            return;
        }
        
        iconRect.gameObject.SetActive(true);
        
        // 转换到 RectTransform 坐标
        float halfW = mapRect.rect.width * 0.5f;
        float halfH = mapRect.rect.height * 0.5f;
        
        iconRect.anchoredPosition = new Vector2(normX * halfW, normZ * halfH);
        
        // 图标朝向
        if (icon.RotateWithObject)
        {
            float iconAngle = icon.transform.eulerAngles.y + mapAngle;
            iconRect.rotation = Quaternion.Euler(0, 0, -iconAngle);
        }
    }
}
```

## 三、迷雾探索集成

```csharp
/// <summary>
/// 小地图迷雾遮罩（探索过的区域显示，未探索显示黑雾）
/// </summary>
public class MinimapFogOfWar : MonoBehaviour
{
    [SerializeField] private int fogTextureSize = 256;      // 迷雾纹理分辨率
    [SerializeField] private float worldSize = 1000f;        // 世界地图总大小
    [SerializeField] private float revealRadius = 50f;       // 每步探索半径
    [SerializeField] private Material fogMaterial;           // 应用到小地图的迷雾材质
    
    private Texture2D fogTexture;
    private Color32[] fogData;
    private Transform player;

    void Start()
    {
        player = FindObjectOfType<PlayerController>()?.transform;
        
        fogTexture = new Texture2D(fogTextureSize, fogTextureSize, TextureFormat.R8, false);
        fogData = new Color32[fogTextureSize * fogTextureSize];
        
        // 初始化全部为黑雾
        for (int i = 0; i < fogData.Length; i++)
            fogData[i] = new Color32(0, 0, 0, 255);
        
        fogTexture.SetPixels32(fogData);
        fogTexture.Apply();
        
        fogMaterial.SetTexture("_FogMap", fogTexture);
    }

    void Update()
    {
        if (player == null) return;
        RevealArea(player.position, revealRadius);
    }

    void RevealArea(Vector3 worldPos, float radius)
    {
        // 世界坐标转纹理坐标
        float texX = (worldPos.x / worldSize + 0.5f) * fogTextureSize;
        float texZ = (worldPos.z / worldSize + 0.5f) * fogTextureSize;
        
        float texRadius = radius / worldSize * fogTextureSize;
        
        int minX = Mathf.Max(0, (int)(texX - texRadius));
        int maxX = Mathf.Min(fogTextureSize - 1, (int)(texX + texRadius));
        int minZ = Mathf.Max(0, (int)(texZ - texRadius));
        int maxZ = Mathf.Min(fogTextureSize - 1, (int)(texZ + texRadius));
        
        bool changed = false;
        
        for (int x = minX; x <= maxX; x++)
        {
            for (int z = minZ; z <= maxZ; z++)
            {
                float dist = Mathf.Sqrt((x - texX) * (x - texX) + (z - texZ) * (z - texZ));
                if (dist <= texRadius)
                {
                    int idx = z * fogTextureSize + x;
                    byte brightness = (byte)Mathf.Min(255, fogData[idx].r + 
                        Mathf.RoundToInt((1f - dist / texRadius) * 30));
                    
                    if (brightness > fogData[idx].r)
                    {
                        fogData[idx].r = brightness;
                        changed = true;
                    }
                }
            }
        }
        
        if (changed)
        {
            fogTexture.SetPixels32(fogData);
            fogTexture.Apply(false); // false=不更新mipmap，节省性能
        }
    }
}
```

## 四、小地图方案对比

| 方案 | 优点 | 缺点 | 适用 |
|------|------|------|------|
| RenderTexture实时渲染 | 实时准确，动态更新 | 额外摄像机渲染开销 | 中低端机需注意性能 |
| 预烘焙贴图 | 无渲染开销 | 不能显示动态建筑变化 | 静态场景，性能敏感 |
| 混合方案 | 背景静态 + 图标动态 | 实现略复杂 | 推荐方案 |

**混合方案（推荐）：**
- 背景：预烘焙的地图贴图（无需每帧渲染）
- 图标：动态UI元素（玩家/NPC/目标等）
- 效果：几乎无性能开销，同样功能完整
