## 一、存档版本迁移系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;
using Newtonsoft.Json;
using Newtonsoft.Json.Linq;

/// <summary>
/// 存档迁移器基类
/// </summary>
public abstract class SaveMigration
{
    public abstract int FromVersion { get; }
    public abstract int ToVersion { get; }
    public abstract JObject Migrate(JObject saveData);
}

/// <summary>
/// v1 → v2 迁移：将 "PlayerName" 字段重命名为 "playerName"
/// </summary>
public class Migration_V1_to_V2 : SaveMigration
{
    public override int FromVersion => 1;
    public override int ToVersion => 2;
    
    public override JObject Migrate(JObject saveData)
    {
        // 字段重命名
        if (saveData.ContainsKey("PlayerName"))
        {
            saveData["playerName"] = saveData["PlayerName"];
            saveData.Remove("PlayerName");
        }
        
        // 添加新字段（带默认值）
        if (!saveData.ContainsKey("soundVolume"))
            saveData["soundVolume"] = 1.0f;
        
        return saveData;
    }
}

/// <summary>
/// v2 → v3 迁移：装备格式变更（string[] → ItemData[]）
/// </summary>
public class Migration_V2_to_V3 : SaveMigration
{
    public override int FromVersion => 2;
    public override int ToVersion => 3;
    
    public override JObject Migrate(JObject saveData)
    {
        // 旧格式：inventory: ["sword_01", "shield_01"]
        // 新格式：inventory: [{itemId: "sword_01", count: 1, level: 1}, ...]
        
        if (saveData["inventory"] is JArray oldInventory)
        {
            var newInventory = new JArray();
            foreach (var item in oldInventory)
            {
                newInventory.Add(new JObject
                {
                    ["itemId"] = item.ToString(),
                    ["count"] = 1,
                    ["level"] = 1,
                    ["equipped"] = false
                });
            }
            saveData["inventory"] = newInventory;
        }
        
        return saveData;
    }
}

/// <summary>
/// 存档迁移管理器
/// </summary>
public class SaveMigrationManager
{
    private List<SaveMigration> migrations = new List<SaveMigration>();
    
    public const int CURRENT_VERSION = 3;

    public SaveMigrationManager()
    {
        // 注册所有迁移
        migrations.Add(new Migration_V1_to_V2());
        migrations.Add(new Migration_V2_to_V3());
        
        // 排序确保按版本顺序执行
        migrations.Sort((a, b) => a.FromVersion.CompareTo(b.FromVersion));
    }

    /// <summary>
    /// 将存档数据迁移到最新版本
    /// </summary>
    public JObject MigrateToLatest(JObject saveData)
    {
        int version = saveData["version"]?.Value<int>() ?? 1;
        
        if (version == CURRENT_VERSION)
            return saveData;
        
        if (version > CURRENT_VERSION)
        {
            Debug.LogWarning($"[Save] 存档版本 {version} 高于当前版本 {CURRENT_VERSION}，可能不兼容");
            return saveData;
        }
        
        Debug.Log($"[Save] 开始迁移存档: v{version} → v{CURRENT_VERSION}");
        
        var current = saveData;
        
        foreach (var migration in migrations)
        {
            if (version < migration.ToVersion)
            {
                Debug.Log($"[Save] 执行迁移: v{migration.FromVersion} → v{migration.ToVersion}");
                
                try
                {
                    current = migration.Migrate(current);
                    version = migration.ToVersion;
                    current["version"] = version;
                }
                catch (Exception e)
                {
                    Debug.LogError($"[Save] 迁移失败: {e.Message}");
                    throw;
                }
            }
        }
        
        Debug.Log($"[Save] 存档迁移完成，当前版本: {version}");
        return current;
    }
}
```

## 二、云端存档同步与冲突解决

```csharp
/// <summary>
/// 云存档同步管理器
/// </summary>
public class CloudSaveManager : MonoBehaviour
{
    private static CloudSaveManager instance;
    public static CloudSaveManager Instance => instance;

    [Header("同步配置")]
    [SerializeField] private float syncInterval = 300f;      // 自动同步间隔（5分钟）
    [SerializeField] private bool syncOnResume = true;        // APP切回前台时同步
    
    private float syncTimer;
    private bool isSyncing;

    public enum ConflictResolution
    {
        UseCloud,       // 使用云端数据
        UseLocal,       // 使用本地数据
        MergeLatest,    // 使用最新修改时间的数据
        AskUser         // 询问用户
    }

    public event Action<CloudSyncResult> OnSyncCompleted;

    void Awake() { instance = this; }

    void OnApplicationFocus(bool hasFocus)
    {
        if (hasFocus && syncOnResume)
            TriggerSync();
    }

    void Update()
    {
        syncTimer += Time.deltaTime;
        if (syncTimer >= syncInterval)
        {
            syncTimer = 0;
            TriggerSync();
        }
    }

    public void TriggerSync()
    {
        if (!isSyncing)
            _ = SyncAsync();
    }

    async System.Threading.Tasks.Task SyncAsync()
    {
        isSyncing = true;
        
        try
        {
            // 1. 获取云端存档元数据
            var cloudMeta = await CloudStorageService.GetSaveMetadata();
            var localMeta = LocalSaveManager.GetSaveMetadata();
            
            if (cloudMeta == null)
            {
                // 云端没有存档，上传本地
                await UploadToCloud();
                OnSyncCompleted?.Invoke(new CloudSyncResult 
                { 
                    Success = true, 
                    Action = "uploaded" 
                });
                return;
            }
            
            // 2. 比较时间戳
            long cloudTimestamp = cloudMeta.LastModified;
            long localTimestamp = localMeta.LastModified;
            
            if (cloudTimestamp == localTimestamp)
            {
                // 一致，无需同步
                return;
            }
            
            if (cloudTimestamp > localTimestamp)
            {
                // 云端更新：下载云端数据
                await ResolveCloudNewer(cloudMeta, localMeta);
            }
            else
            {
                // 本地更新：上传本地数据
                await UploadToCloud();
            }
            
            OnSyncCompleted?.Invoke(new CloudSyncResult { Success = true });
        }
        catch (Exception e)
        {
            Debug.LogError($"[CloudSave] 同步失败: {e.Message}");
            OnSyncCompleted?.Invoke(new CloudSyncResult { Success = false, Error = e.Message });
        }
        finally
        {
            isSyncing = false;
        }
    }

    async System.Threading.Tasks.Task ResolveCloudNewer(SaveMetadata cloudMeta, 
        SaveMetadata localMeta)
    {
        // 自动冲突解决策略：使用最高进度的存档
        var cloudSave = await CloudStorageService.DownloadSave();
        var localSave = LocalSaveManager.LoadRawSave();
        
        // 比较关键进度指标
        bool cloudHasHigherProgress = CompareProgress(cloudSave, localSave) > 0;
        
        // 如果进度差距大，询问用户
        bool progressSignificantlyDifferent = Mathf.Abs(
            cloudSave?.PlayerLevel ?? 0 - localSave?.PlayerLevel ?? 0) > 5;
        
        if (progressSignificantlyDifferent)
        {
            bool useCloud = await ShowConflictDialog(cloudSave, localSave);
            if (useCloud)
                LocalSaveManager.ApplySave(cloudSave);
            else
                await UploadToCloud();
        }
        else if (cloudHasHigherProgress)
        {
            LocalSaveManager.ApplySave(cloudSave);
            Debug.Log("[CloudSave] 已使用云端更高进度的存档");
        }
        else
        {
            await UploadToCloud();
        }
    }

    int CompareProgress(SaveData a, SaveData b)
    {
        if (a == null && b == null) return 0;
        if (a == null) return -1;
        if (b == null) return 1;
        
        // 比较多个维度的进度
        int score = 0;
        if (a.PlayerLevel > b.PlayerLevel) score++;
        else if (a.PlayerLevel < b.PlayerLevel) score--;
        
        if (a.TotalPlayTime > b.TotalPlayTime) score++;
        else if (a.TotalPlayTime < b.TotalPlayTime) score--;
        
        return score;
    }

    async System.Threading.Tasks.Task<bool> ShowConflictDialog(SaveData cloud, SaveData local)
    {
        // 展示冲突解决 UI
        return await UIManager.Instance.ShowSaveConflictDialog(
            $"云端: Lv.{cloud?.PlayerLevel} 游戏时间:{cloud?.TotalPlayTime:F0}h",
            $"本地: Lv.{local?.PlayerLevel} 游戏时间:{local?.TotalPlayTime:F0}h");
    }

    async System.Threading.Tasks.Task UploadToCloud()
    {
        var saveData = LocalSaveManager.LoadRawSave();
        if (saveData != null)
            await CloudStorageService.UploadSave(saveData);
    }
}

public class CloudSyncResult
{
    public bool Success;
    public string Action;
    public string Error;
}

public class SaveMetadata
{
    public long LastModified;
    public string Version;
    public int PlayerLevel;
}
```

## 三、双缓冲存档（防崩溃）

```csharp
/// <summary>
/// 双缓冲存档（防止写入过程中崩溃导致存档损坏）
/// </summary>
public class SafeSaveWriter
{
    private string savePath;
    private string backupPath;
    private string tempPath;

    public SafeSaveWriter(string savePath)
    {
        this.savePath = savePath;
        backupPath = savePath + ".bak";
        tempPath = savePath + ".tmp";
    }

    /// <summary>
    /// 安全写入（原子操作）
    /// </summary>
    public bool SafeWrite(string data)
    {
        try
        {
            // 1. 写入临时文件
            System.IO.File.WriteAllText(tempPath, data, System.Text.Encoding.UTF8);
            
            // 2. 验证临时文件
            string verify = System.IO.File.ReadAllText(tempPath);
            if (verify != data)
            {
                Debug.LogError("[Save] 写入验证失败！");
                System.IO.File.Delete(tempPath);
                return false;
            }
            
            // 3. 将旧存档备份（移动而非复制，速度快）
            if (System.IO.File.Exists(savePath))
            {
                if (System.IO.File.Exists(backupPath))
                    System.IO.File.Delete(backupPath);
                System.IO.File.Move(savePath, backupPath);
            }
            
            // 4. 将临时文件重命名为正式存档
            System.IO.File.Move(tempPath, savePath);
            
            return true;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Save] 写入失败: {e.Message}");
            // 尝试恢复
            TryRecover();
            return false;
        }
    }

    public string SafeRead()
    {
        // 优先读取主存档
        if (System.IO.File.Exists(savePath))
        {
            try
            {
                return System.IO.File.ReadAllText(savePath, System.Text.Encoding.UTF8);
            }
            catch { }
        }
        
        // 主存档不可用，尝试备份
        if (System.IO.File.Exists(backupPath))
        {
            Debug.LogWarning("[Save] 主存档不可用，使用备份存档");
            try
            {
                return System.IO.File.ReadAllText(backupPath, System.Text.Encoding.UTF8);
            }
            catch { }
        }
        
        return null;
    }

    void TryRecover()
    {
        if (System.IO.File.Exists(backupPath) && !System.IO.File.Exists(savePath))
        {
            System.IO.File.Copy(backupPath, savePath);
            Debug.Log("[Save] 已从备份恢复存档");
        }
    }
}
```

## 四、存档加密

```csharp
/// <summary>
/// 存档加密/解密（AES-128）
/// </summary>
public static class SaveEncryption
{
    // 实际项目中密钥应该通过安全方式存储，不要硬编码
    private static readonly byte[] Key = 
        System.Text.Encoding.UTF8.GetBytes("YOUR_32_CHAR_KEY_HERE!_SECURE_123");
    private static readonly byte[] IV = 
        System.Text.Encoding.UTF8.GetBytes("YOUR_16_CHAR_IV!!");

    public static string Encrypt(string plainText)
    {
        using var aes = System.Security.Cryptography.Aes.Create();
        aes.Key = Key[..16]; // 取前16字节（AES-128）
        aes.IV = IV;
        
        var encryptor = aes.CreateEncryptor();
        byte[] plainBytes = System.Text.Encoding.UTF8.GetBytes(plainText);
        byte[] encryptedBytes = encryptor.TransformFinalBlock(plainBytes, 0, plainBytes.Length);
        
        return Convert.ToBase64String(encryptedBytes);
    }

    public static string Decrypt(string encryptedText)
    {
        using var aes = System.Security.Cryptography.Aes.Create();
        aes.Key = Key[..16];
        aes.IV = IV;
        
        var decryptor = aes.CreateDecryptor();
        byte[] encryptedBytes = Convert.FromBase64String(encryptedText);
        byte[] plainBytes = decryptor.TransformFinalBlock(encryptedBytes, 0, encryptedBytes.Length);
        
        return System.Text.Encoding.UTF8.GetString(plainBytes);
    }

    /// <summary>
    /// 计算存档数据签名（防篡改）
    /// </summary>
    public static string ComputeSignature(string data)
    {
        var hmac = new System.Security.Cryptography.HMACSHA256(Key);
        byte[] dataBytes = System.Text.Encoding.UTF8.GetBytes(data);
        byte[] hash = hmac.ComputeHash(dataBytes);
        return Convert.ToBase64String(hash);
    }

    public static bool VerifySignature(string data, string signature)
    {
        string expected = ComputeSignature(data);
        return expected == signature;
    }
}
```

## 五、存档系统设计要点

| 维度 | 方案 |
|------|------|
| 版本兼容 | 每次格式变更注册迁移器，保证旧存档可读 |
| 防崩溃 | 双缓冲写入（临时文件 → 备份 → 主文件） |
| 防篡改 | HMAC签名校验，服务端关键数值验证 |
| 云端同步 | 时间戳比较 + 进度比较 + 用户确认 |
| 性能 | 增量存档（只保存变化的字段），异步IO |
| 调试 | 提供存档查看/编辑工具（仅开发包）|
