## 一、签到数据结构

```csharp
using System;
using UnityEngine;

[Serializable]
public class CheckInData
{
    public int ConsecutiveDays;     // 连续签到天数
    public long LastCheckInTime;    // 最后签到时间（Unix时间戳，UTC）
    public string CurrentMonth;     // 当前月份 "2026-03"
    public bool[] MonthlyCheckIns;  // 当月每天签到状态（长度=当月天数）
    public int TotalCheckIns;       // 历史总签到天数
    public int MakeupCount;         // 剩余补签次数
}

[Serializable]
public class CheckInReward
{
    public int Day;          // 第几天
    public string ItemId;
    public int Count;
    public bool IsSpecial;   // 特殊奖励（7天/30天大奖）
}
```

## 二、签到管理器

```csharp
public class CheckInManager : MonoBehaviour
{
    private static CheckInManager instance;
    public static CheckInManager Instance => instance;
    
    private CheckInData checkInData;
    
    public event Action<int> OnCheckInSuccess;  // 连续天数
    public event Action OnCheckInStateUpdated;

    void Awake() { instance = this; }

    public bool CanCheckInToday()
    {
        if (checkInData == null) return false;
        
        var today = GetServerDate();
        var lastDate = DateTimeOffset.FromUnixTimeSeconds(checkInData.LastCheckInTime).Date;
        
        return today.Date > lastDate;
    }

    public async System.Threading.Tasks.Task<bool> CheckIn()
    {
        if (!CanCheckInToday()) return false;
        
        var result = await NetworkService.CheckIn();
        if (result == null) return false;
        
        checkInData = result;
        
        OnCheckInSuccess?.Invoke(checkInData.ConsecutiveDays);
        OnCheckInStateUpdated?.Invoke();
        
        // 计算奖励
        var reward = GetRewardForDay(checkInData.ConsecutiveDays);
        if (reward != null)
        {
            UIManager.Instance?.ShowCheckInReward(reward);
        }
        
        return true;
    }

    public async System.Threading.Tasks.Task<bool> MakeupCheckIn(int dayOfMonth)
    {
        if (checkInData.MakeupCount <= 0) return false;
        
        bool success = await NetworkService.MakeupCheckIn(dayOfMonth);
        if (success) await LoadCheckInData();
        return success;
    }

    CheckInReward GetRewardForDay(int consecutiveDays)
    {
        // 7天循环奖励
        int cycleDay = ((consecutiveDays - 1) % 7) + 1;

        return null;
    }

    DateTime GetServerDate()
    {
        // 使用服务器时间而非本地时间（防止修改系统时间作弊）
        return NetworkTimeSync.GetServerTime();
    }

    async System.Threading.Tasks.Task LoadCheckInData()
    {
        checkInData = await NetworkService.GetCheckInData();
        OnCheckInStateUpdated?.Invoke();
    }
}
```

## 四、签到系统设计规范

| 设计点 | 最佳实践 |
|--------|----------|
| 时区处理 | 使用服务器UTC时间，0点刷新 |
| 断签处理 | 连续签到重置回0（保留总天数）|
| 补签机制 | 月卡/VIP特权，增加付费价值 |
| 奖励节奏 | 第7/30天奖励要足够吸引人 |
| 视觉设计 | 月历形式让玩家看到全月进度 |
| 防作弊 | 服务端验证，不信任客户端时间 |
