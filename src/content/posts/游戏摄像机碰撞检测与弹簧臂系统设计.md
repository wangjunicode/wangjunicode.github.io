## 一、弹簧臂的作用

第三人称游戏摄像机的核心挑战：

1. **穿墙问题**：摄像机进入墙体内部，画面失真
2. **遮挡问题**：摄像机被障碍物遮挡，看不见角色
3. **抖动问题**：快速移动时摄像机跟随不够平滑
4. **视角控制**：玩家操控水平/垂直旋转，需要合理限制

弹簧臂（Spring Arm）是解决上述问题的经典方案——通过缩短弹簧臂长度来保持摄像机始终位于可见区域内。

## 二、基础弹簧臂实现

```csharp
using UnityEngine;

/// <summary>
/// 弹簧臂组件：防止摄像机穿模的经典方案
/// </summary>
public class SpringArm : MonoBehaviour
{
    [Header("目标追踪")]
    [SerializeField] private Transform target;
    [SerializeField] private Vector3 targetOffset = new Vector3(0, 1.5f, 0);
    
    [Header("弹簧臂参数")]
    [SerializeField] private float armLength = 5f;        // 理想臂长
    [SerializeField] private float minArmLength = 1f;     // 最短臂长
    [SerializeField] private float cameraRadius = 0.2f;   // 摄像机碰撞半径
    [SerializeField] private LayerMask collisionMask;     // 碰撞检测层
    
    [Header("旋转控制")]
    [SerializeField] private float horizontalSensitivity = 2f;
    [SerializeField] private float verticalSensitivity = 2f;
    [SerializeField] private float minVerticalAngle = -30f;
    [SerializeField] private float maxVerticalAngle = 70f;
    
    [Header("平滑参数")]
    [SerializeField] private float followSmoothTime = 0.08f;  // 位置跟随平滑
    [SerializeField] private float rotateSmoothTime = 0.04f;  // 旋转平滑
    [SerializeField] private float armLengthSmoothTime = 0.1f; // 臂长平滑
    
    [Header("碰撞响应")]
    [SerializeField] private float cameraRetractSpeed = 15f;  // 碰撞缩短速度
    [SerializeField] private float cameraExtendSpeed = 3f;    // 恢复伸长速度

    // 旋转角度
    private float yaw;   // 水平旋转
    private float pitch; // 垂直旋转

    // 平滑中间变量
    private Vector3 followVelocity;
    private float currentArmLength;
    private float armLengthVelocity;
    private Vector3 currentPosition;
    private Quaternion targetRotation;
    private Quaternion smoothRotation;

    void Start()
    {
        currentArmLength = armLength;
        currentPosition = target ? target.position : transform.position;
        
        // 初始化为当前旋转角度
        yaw = transform.eulerAngles.y;
        pitch = transform.eulerAngles.x;
        
        Cursor.lockState = CursorLockMode.Locked;
        Cursor.visible = false;
    }

    void LateUpdate()
    {
        if (target == null) return;
        
        HandleInput();
        UpdateFollowPosition();
        HandleCollision();
        ApplyTransform();
    }

    void HandleInput()
    {
        float mouseX = Input.GetAxis("Mouse X") * horizontalSensitivity;
        float mouseY = Input.GetAxis("Mouse Y") * verticalSensitivity;
        
        yaw += mouseX;
        pitch -= mouseY; // 减法：向上移动鼠标，俯仰角减小（看向上方）
        pitch = Mathf.Clamp(pitch, minVerticalAngle, maxVerticalAngle);
    }

    void UpdateFollowPosition()
    {
        // 平滑跟随目标位置
        Vector3 targetPos = target.position + targetOffset;
        currentPosition = Vector3.SmoothDamp(
            currentPosition, targetPos, 
            ref followVelocity, followSmoothTime);
    }

    void HandleCollision()
    {
        // 计算理想摄像机位置
        Quaternion rotation = Quaternion.Euler(pitch, yaw, 0);
        Vector3 desiredDirection = rotation * Vector3.back;
        Vector3 desiredPosition = currentPosition + desiredDirection * armLength;
        
        // SphereCast：从目标点向摄像机方向检测碰撞
        float actualArmLength = armLength;
        
        if (Physics.SphereCast(
            currentPosition,
            cameraRadius,
            desiredDirection,
            out RaycastHit hit,
            armLength,
            collisionMask,
            QueryTriggerInteraction.Ignore))
        {
            // 碰撞点距离 = 摄像机可以到达的最近距离
            actualArmLength = hit.distance;
            actualArmLength = Mathf.Max(actualArmLength, minArmLength);
        }
        
        // 碰撞时快速缩短，恢复时缓慢伸长
        float smoothSpeed = actualArmLength < currentArmLength ? 
            cameraRetractSpeed : cameraExtendSpeed;
        
        currentArmLength = Mathf.Lerp(currentArmLength, actualArmLength, 
            smoothSpeed * Time.deltaTime);
    }

    void ApplyTransform()
    {
        Quaternion rotation = Quaternion.Euler(pitch, yaw, 0);
        
        // 平滑旋转
        smoothRotation = Quaternion.Slerp(smoothRotation, rotation, 
            1f - Mathf.Exp(-1f / rotateSmoothTime * Time.deltaTime));
        
        // 计算最终摄像机位置
        Vector3 offset = smoothRotation * Vector3.back * currentArmLength;
        transform.position = currentPosition + offset;
        transform.rotation = smoothRotation;
    }
}
```

## 三、高级摄像机碰撞处理

```csharp
/// <summary>
/// 多点碰撞检测：处理摄像机进入角落的边缘情况
/// </summary>
public class AdvancedCameraCollision
{
    private float radius;
    private LayerMask mask;
    
    public AdvancedCameraCollision(float r, LayerMask m)
    {
        radius = r;
        mask = m;
    }
    
    /// <summary>
    /// 使用多条射线确保碰撞检测的稳健性
    /// </summary>
    public float GetCollisionAdjustedLength(
        Vector3 pivotPos, Vector3 direction, float desiredLength)
    {
        float result = desiredLength;
        
        // 中心射线
        if (SphereCastLength(pivotPos, direction, desiredLength, out float centerLen))
            result = Mathf.Min(result, centerLen);
        
        // 4个偏移射线（检测摄像机边缘）
        Vector3 right = Vector3.Cross(direction, Vector3.up).normalized * radius;
        Vector3 up = Vector3.up * radius;
        
        Vector3[] offsets = { right, -right, up, -up };
        foreach (var offset in offsets)
        {
            if (SphereCastLength(pivotPos + offset, direction, desiredLength, out float len))
                result = Mathf.Min(result, len);
        }
        
        return Mathf.Max(result, 0.1f);
    }
    
    private bool SphereCastLength(Vector3 origin, Vector3 dir, float maxLen, out float length)
    {
        if (Physics.SphereCast(origin, radius * 0.5f, dir, out RaycastHit hit, 
            maxLen, mask, QueryTriggerInteraction.Ignore))
        {
            length = hit.distance;
            return true;
        }
        length = maxLen;
        return false;
    }
    
    /// <summary>
    /// 透明度处理：角色被遮挡时淡化遮挡物
    /// </summary>
    public void HandleObstructionTransparency(
        Vector3 cameraPos, Vector3 targetPos, float fadeDistance = 2f)
    {
        Vector3 dir = targetPos - cameraPos;
        float dist = dir.magnitude;
        
        RaycastHit[] hits = Physics.RaycastAll(cameraPos, dir.normalized, dist);
        
        foreach (var hit in hits)
        {
            var renderer = hit.collider.GetComponent<Renderer>();
            if (renderer != null)
            {
                // 根据距离设置透明度
                float alpha = Mathf.Clamp01((hit.distance - fadeDistance) / fadeDistance);
                SetRendererAlpha(renderer, alpha);
            }
        }
    }
    
    private void SetRendererAlpha(Renderer r, float alpha)
    {
        // 修改材质透明度（需要材质支持透明）
        foreach (var mat in r.materials)
        {
            Color c = mat.color;
            c.a = alpha;
            mat.color = c;
        }
    }
}
```

## 四、摄像机抖动（Camera Shake）系统

```csharp
/// <summary>
/// 摄像机抖动系统
/// 支持爆炸、受击、地震等多种抖动模式
/// </summary>
public class CameraShakeSystem : MonoBehaviour
{
    private static CameraShakeSystem instance;
    public static CameraShakeSystem Instance => instance;

    [SerializeField] private Transform cameraTarget; // 摄像机跟随的目标

    private List<ShakeInstance> activeShakes = new List<ShakeInstance>();
    private Vector3 shakeOffset;
    private Vector3 shakeRotation;

    void Awake() => instance = this;

    void LateUpdate()
    {
        shakeOffset = Vector3.zero;
        shakeRotation = Vector3.zero;
        
        for (int i = activeShakes.Count - 1; i >= 0; i--)
        {
            var shake = activeShakes[i];
            shake.Update(Time.deltaTime);
            
            shakeOffset += shake.CurrentOffset;
            shakeRotation += shake.CurrentRotation;
            
            if (shake.IsComplete)
                activeShakes.RemoveAt(i);
        }
        
        // 应用抖动偏移
        if (cameraTarget != null)
        {
            cameraTarget.localPosition = shakeOffset;
            cameraTarget.localRotation = Quaternion.Euler(shakeRotation);
        }
    }

    /// <summary>
    /// 触发爆炸抖动
    /// </summary>
    public static void ShakeExplosion(float intensity, float duration, 
        Vector3 epicenter, float maxDistance)
    {
        float distanceFactor = 1f;
        if (Instance.cameraTarget != null)
        {
            float dist = Vector3.Distance(Instance.cameraTarget.position, epicenter);
            distanceFactor = Mathf.Clamp01(1f - dist / maxDistance);
        }
        
        Instance.activeShakes.Add(new ShakeInstance(
            ShakeProfile.Explosion(intensity * distanceFactor, duration)));
    }

    /// <summary>
    /// 触发受击抖动
    /// </summary>
    public static void ShakeHit(float intensity, Vector3 hitDirection)
    {
        Instance.activeShakes.Add(new ShakeInstance(
            ShakeProfile.Hit(intensity, hitDirection)));
    }

    /// <summary>
    /// 触发持续地震抖动
    /// </summary>
    public static CameraShakeHandle ShakeContinuous(float intensity, float frequency)
    {
        var shake = new ShakeInstance(ShakeProfile.Continuous(intensity, frequency));
        Instance.activeShakes.Add(shake);
        return new CameraShakeHandle(shake);
    }
}

[System.Serializable]
public class ShakeProfile
{
    public float Intensity;
    public float Duration;
    public float Frequency;
    public AnimationCurve IntensityCurve;
    public bool UseRotation;
    public Vector3 DirectionalBias; // 方向性偏移

    public static ShakeProfile Explosion(float intensity, float duration) => new ShakeProfile
    {
        Intensity = intensity,
        Duration = duration,
        Frequency = 25f,
        IntensityCurve = AnimationCurve.EaseInOut(0, 1, 1, 0),
        UseRotation = true
    };

    public static ShakeProfile Hit(float intensity, Vector3 dir) => new ShakeProfile
    {
        Intensity = intensity,
        Duration = 0.3f,
        Frequency = 20f,
        DirectionalBias = dir.normalized,
        IntensityCurve = AnimationCurve.Linear(0, 1, 1, 0)
    };

    public static ShakeProfile Continuous(float intensity, float freq) => new ShakeProfile
    {
        Intensity = intensity,
        Duration = float.MaxValue,
        Frequency = freq,
        IntensityCurve = AnimationCurve.Constant(0, 1, 1)
    };
}

public class ShakeInstance
{
    private ShakeProfile profile;
    private float elapsed;
    private float noiseOffset;
    
    public Vector3 CurrentOffset { get; private set; }
    public Vector3 CurrentRotation { get; private set; }
    public bool IsComplete => elapsed >= profile.Duration;
    public float Intensity { get; set; } = 1f;

    public ShakeInstance(ShakeProfile p)
    {
        profile = p;
        noiseOffset = UnityEngine.Random.Range(0f, 1000f);
    }

    public void Update(float dt)
    {
        elapsed += dt;
        float t = Mathf.Clamp01(elapsed / profile.Duration);
        float curveIntensity = profile.IntensityCurve.Evaluate(t);
        float totalIntensity = profile.Intensity * curveIntensity * Intensity;
        
        // 使用柏林噪声产生自然的抖动
        float timeScale = elapsed * profile.Frequency;
        
        float noiseX = (Mathf.PerlinNoise(timeScale + noiseOffset, 0) - 0.5f) * 2f;
        float noiseY = (Mathf.PerlinNoise(0, timeScale + noiseOffset) - 0.5f) * 2f;
        float noiseZ = (Mathf.PerlinNoise(timeScale + noiseOffset, timeScale) - 0.5f) * 2f;
        
        CurrentOffset = new Vector3(noiseX, noiseY, noiseZ) * totalIntensity * 0.15f
            + profile.DirectionalBias * totalIntensity * 0.1f;
        
        if (profile.UseRotation)
            CurrentRotation = new Vector3(noiseY, noiseX, noiseZ) * totalIntensity * 2f;
        else
            CurrentRotation = Vector3.zero;
    }
    
    public void Stop() => elapsed = profile.Duration;
}

/// <summary>
/// 持续抖动的控制句柄
/// </summary>
public class CameraShakeHandle
{
    private ShakeInstance shake;
    public CameraShakeHandle(ShakeInstance s) => shake = s;
    public void SetIntensity(float v) => shake.Intensity = v;
    public void Stop() => shake.Stop();
}
```

## 五、与 Cinemachine 的集成

```csharp
/// <summary>
/// 基于 Cinemachine 的弹簧臂扩展
/// </summary>
#if CINEMACHINE
using Cinemachine;

[AddComponentMenu("")]
public class CinemachineSpringArmExtension : CinemachineExtension
{
    [SerializeField] private float armLength = 5f;
    [SerializeField] private float radius = 0.3f;
    [SerializeField] private LayerMask collisionLayer;
    [SerializeField] private float returnSpeed = 3f;
    
    private float currentLength;
    
    protected override void PostPipelineStageCallback(
        CinemachineVirtualCameraBase vcam,
        CinemachineCore.Stage stage,
        ref CameraState state,
        float deltaTime)
    {
        if (stage != CinemachineCore.Stage.Body) return;
        
        Vector3 pivotPos = state.ReferenceUp; // 获取角色位置
        Vector3 camOffset = state.RawPosition - vcam.Follow.position;
        float desiredLength = camOffset.magnitude;
        
        float actualLength = desiredLength;
        
        if (Physics.SphereCast(vcam.Follow.position, radius,
            camOffset.normalized, out RaycastHit hit, desiredLength, collisionLayer))
        {
            actualLength = Mathf.Max(hit.distance, 0.5f);
        }
        
        float speed = actualLength < currentLength ? 20f : returnSpeed;
        currentLength = Mathf.Lerp(currentLength, actualLength, speed * deltaTime);
        
        state.RawPosition = vcam.Follow.position + 
            camOffset.normalized * currentLength;
    }
}
#endif
```

## 六、摄像机系统设计要点

| 功能 | 关键技术 | 注意事项 |
|------|----------|----------|
| 防穿模 | SphereCast + 收缩臂长 | 用球形检测而非射线，避免细缝穿透 |
| 平滑跟随 | SmoothDamp + Slerp | 位置用SmoothDamp，旋转用Slerp |
| 碰撞恢复 | 快收慢放 | 碰撞时快速收短，恢复时缓慢伸长 |
| 抖动效果 | 柏林噪声 | 比随机数更自然，用于爆炸/受击 |
| 视角限制 | Clamp 俯仰角 | 通常限制 -30° ~ 70° |
| 遮挡透明 | Raycast + 材质Alpha | 需要材质支持透明模式切换 |

摄像机系统是玩家体验的第一窗口，其品质直接影响游戏整体手感。建议在游戏早期就建立一套可配置、可扩展的摄像机系统，而不是依赖单一的 MonoBehaviour 脚本。
