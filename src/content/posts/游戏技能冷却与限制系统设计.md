## 一、冷却系统的核心概念

技能冷却（Cooldown）是几乎所有动作/RPG游戏的基础机制，其核心职责包括：

1. **防止技能滥用**：限制技能释放频率，保证游戏平衡
2. **策略深度**：冷却管理成为玩家技术的一部分
3. **资源维度**：与能量/法力值共同构成技能释放条件
4. **反馈清晰**：通过UI直观告知玩家何时可以使用技能

## 二、基础冷却计时器

### 2.1 核心数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 技能冷却状态
/// </summary>
[Serializable]
public class CooldownState
{
    public string SkillId;
    public float TotalCooldown;    // 总冷却时间
    public float RemainingCooldown; // 剩余冷却时间
    public int MaxCharges;          // 最大充能数
    public int CurrentCharges;      // 当前充能数
    public float ChargeRechargeTime; // 单次充能恢复时间
    
    public bool IsReady => RemainingCooldown <= 0 && CurrentCharges > 0;
    public float Progress => TotalCooldown > 0 ? 
        1f - RemainingCooldown / TotalCooldown : 1f;
    
    // 冷却组ID（同组技能共享冷却）
    public string CooldownGroup;
}

/// <summary>
/// 冷却管理器
/// </summary>
public class CooldownManager : MonoBehaviour
{
    private Dictionary<string, CooldownState> cooldowns 
        = new Dictionary<string, CooldownState>();
    
    // 全局冷却（GCD）
    private float globalCooldown;
    private float globalCooldownDuration = 1.5f;
    
    // 冷却组
    private Dictionary<string, float> cooldownGroups 
        = new Dictionary<string, float>();
    
    // 冷却修改器（Buff/Debuff影响）
    private float cooldownReductionRate = 0f; // 冷却缩减百分比
    
    // 事件
    public event Action<string> OnCooldownStarted;
    public event Action<string> OnCooldownExpired;
    public event Action<string, int> OnChargeChanged;
    
    void Update()
    {
        float dt = Time.deltaTime;
        
        // 更新全局冷却
        if (globalCooldown > 0)
            globalCooldown = Mathf.Max(0, globalCooldown - dt);
        
        // 更新冷却组
        var groupKeys = new List<string>(cooldownGroups.Keys);
        foreach (var key in groupKeys)
        {
            cooldownGroups[key] = Mathf.Max(0, cooldownGroups[key] - dt);
        }
        
        // 更新各技能冷却
        foreach (var kvp in cooldowns)
        {
            var state = kvp.Value;
            if (state.RemainingCooldown > 0)
            {
                state.RemainingCooldown = Mathf.Max(0, state.RemainingCooldown - dt);
                
                if (state.RemainingCooldown <= 0)
                {
                    // 充能恢复
                    if (state.CurrentCharges < state.MaxCharges)
                    {
                        state.CurrentCharges++;
                        OnChargeChanged?.Invoke(kvp.Key, state.CurrentCharges);
                        
                        // 如果充能未满，继续下一次充能冷却
                        if (state.CurrentCharges < state.MaxCharges)
                        {
                            state.RemainingCooldown = state.ChargeRechargeTime;
                        }
                    }
                    
                    OnCooldownExpired?.Invoke(kvp.Key);
                }
            }
        }
    }
    
    /// <summary>
    /// 注册技能冷却
    /// </summary>
    public void RegisterSkill(string skillId, float cooldown, 
        int maxCharges = 1, string group = null)
    {
        cooldowns[skillId] = new CooldownState
        {
            SkillId = skillId,
            TotalCooldown = cooldown,
            RemainingCooldown = 0,
            MaxCharges = maxCharges,
            CurrentCharges = maxCharges,
            ChargeRechargeTime = cooldown,
            CooldownGroup = group
        };
    }
    
    /// <summary>
    /// 触发冷却（使用技能时调用）
    /// </summary>
    public bool TriggerCooldown(string skillId, bool triggerGCD = true)
    {
        if (!CanUse(skillId)) return false;
        
        var state = cooldowns[skillId];
        state.CurrentCharges--;
        
        // 应用冷却缩减
        float actualCooldown = state.TotalCooldown * (1f - cooldownReductionRate);
        actualCooldown = Mathf.Max(actualCooldown, state.TotalCooldown * 0.2f); // 最多缩减80%
        
        if (state.CurrentCharges == 0)
        {
            state.RemainingCooldown = actualCooldown;
        }
        
        // 触发冷却组
        if (!string.IsNullOrEmpty(state.CooldownGroup))
        {
            float groupCooldown = GetGroupCooldown(state.CooldownGroup);
            cooldownGroups[state.CooldownGroup] = groupCooldown;
        }
        
        // 触发全局冷却
        if (triggerGCD)
        {
            float gcd = globalCooldownDuration * (1f - cooldownReductionRate);
            gcd = Mathf.Max(gcd, 0.75f); // GCD最小0.75秒
            globalCooldown = gcd;
        }
        
        OnCooldownStarted?.Invoke(skillId);
        return true;
    }
    
    /// <summary>
    /// 检查技能是否可用
    /// </summary>
    public bool CanUse(string skillId)
    {
        if (!cooldowns.TryGetValue(skillId, out var state))
            return false;
        
        // 检查自身冷却
        if (state.CurrentCharges <= 0 || state.RemainingCooldown > 0)
            return false;
        
        // 检查全局冷却
        if (globalCooldown > 0 && IsAffectedByGCD(skillId))
            return false;
        
        // 检查冷却组
        if (!string.IsNullOrEmpty(state.CooldownGroup))
        {
            if (cooldownGroups.TryGetValue(state.CooldownGroup, out float groupCD) && groupCD > 0)
                return false;
        }
        
        return true;
    }
    
    /// <summary>
    /// 获取剩余冷却时间
    /// </summary>
    public float GetRemainingCooldown(string skillId)
    {
        if (!cooldowns.TryGetValue(skillId, out var state))
            return 0;
        
        float remaining = state.RemainingCooldown;
        
        // 考虑冷却组
        if (!string.IsNullOrEmpty(state.CooldownGroup))
        {
            if (cooldownGroups.TryGetValue(state.CooldownGroup, out float groupCD))
                remaining = Mathf.Max(remaining, groupCD);
        }
        
        // 考虑GCD
        if (IsAffectedByGCD(skillId))
            remaining = Mathf.Max(remaining, globalCooldown);
        
        return remaining;
    }
    
    /// <summary>
    /// 立即重置冷却（道具/技能效果）
    /// </summary>
    public void ResetCooldown(string skillId)
    {
        if (cooldowns.TryGetValue(skillId, out var state))
        {
            state.RemainingCooldown = 0;
            state.CurrentCharges = state.MaxCharges;
            OnCooldownExpired?.Invoke(skillId);
        }
    }
    
    /// <summary>
    /// 设置冷却缩减（来自Buff）
    /// </summary>
    public void SetCooldownReduction(float rate)
    {
        cooldownReductionRate = Mathf.Clamp01(rate);
    }
    
    private bool IsAffectedByGCD(string skillId) => true; // 可按需配置
    
    private float GetGroupCooldown(string group)
    {
        // 从配置获取组冷却时长，这里简化返回固定值
        return 1.0f;
    }
    
    public float GlobalCooldownRemaining => globalCooldown;
    public CooldownState GetState(string skillId) => 
        cooldowns.TryGetValue(skillId, out var s) ? s : null;
}
```

## 三、全局冷却（GCD）系统

全局冷却（Global Cooldown）是MMORPG中的核心机制，确保技能使用节奏：

```csharp
/// <summary>
/// GCD管理器（独立组件）
/// </summary>
public class GlobalCooldownController
{
    private float gcdDuration;
    private float gcdRemaining;
    private float hasteMultiplier = 1f; // 急速系数（降低GCD）
    
    // 不受GCD影响的技能分类
    private HashSet<string> gcdExemptCategories = new HashSet<string>
    {
        "instant_off_gcd",   // 即时非GCD技能
        "passive",           // 被动技能
        "item_use",          // 道具使用
        "movement"           // 移动技能（冲刺等）
    };
    
    public bool IsOnGCD => gcdRemaining > 0;
    public float GCDProgress => gcdDuration > 0 ? 1f - gcdRemaining / gcdDuration : 1f;
    
    public GlobalCooldownController(float baseDuration = 1.5f)
    {
        gcdDuration = baseDuration;
    }
    
    public void Update(float deltaTime)
    {
        gcdRemaining = Mathf.Max(0, gcdRemaining - deltaTime);
    }
    
    public void Trigger(string skillCategory = "normal")
    {
        if (gcdExemptCategories.Contains(skillCategory)) return;
        
        float actualGCD = gcdDuration / hasteMultiplier;
        actualGCD = Mathf.Clamp(actualGCD, 0.75f, gcdDuration);
        gcdRemaining = actualGCD;
    }
    
    public bool IsExempt(string category) => gcdExemptCategories.Contains(category);
    
    public void SetHaste(float multiplier)
    {
        hasteMultiplier = Mathf.Max(0.1f, multiplier);
    }
}
```

## 四、技能充能（Charges）机制

充能系统允许技能有多次使用次数，每次充能独立回复：

```csharp
/// <summary>
/// 充能技能示例（如闪现最多存2层）
/// </summary>
public class ChargedSkill : MonoBehaviour
{
    [Header("充能配置")]
    [SerializeField] private string skillId = "flash";
    [SerializeField] private int maxCharges = 2;
    [SerializeField] private float chargeRechargeTime = 12f;
    [SerializeField] private bool rechargeOneAtATime = true;
    
    private CooldownManager cooldownManager;
    private int currentCharges;
    private float[] chargeTimers; // 每个充能独立计时
    
    void Awake()
    {
        cooldownManager = GetComponent<CooldownManager>();
        currentCharges = maxCharges;
        chargeTimers = new float[maxCharges];
        
        cooldownManager.RegisterSkill(skillId, chargeRechargeTime, maxCharges);
    }
    
    public bool UseSkill()
    {
        if (currentCharges <= 0) return false;
        
        currentCharges--;
        
        // 开始为消耗的那个充能计时
        StartChargeTimer();
        
        return true;
    }
    
    void StartChargeTimer()
    {
        // 找一个空闲的计时槽
        for (int i = 0; i < chargeTimers.Length; i++)
        {
            if (chargeTimers[i] <= 0)
            {
                chargeTimers[i] = chargeRechargeTime;
                break;
            }
        }
    }
    
    void Update()
    {
        for (int i = 0; i < chargeTimers.Length; i++)
        {
            if (chargeTimers[i] > 0)
            {
                chargeTimers[i] -= Time.deltaTime;
                if (chargeTimers[i] <= 0 && currentCharges < maxCharges)
                {
                    currentCharges++;
                    chargeTimers[i] = 0;
                }
            }
        }
    }
    
    public float GetNextChargeProgress()
    {
        float minTimer = float.MaxValue;
        foreach (var t in chargeTimers)
        {
            if (t > 0) minTimer = Mathf.Min(minTimer, t);
        }
        return minTimer == float.MaxValue ? 1f : 1f - minTimer / chargeRechargeTime;
    }
}
```

## 五、冷却UI系统

```csharp
/// <summary>
/// 技能冷却UI组件
/// </summary>
public class SkillCooldownUI : MonoBehaviour
{
    [SerializeField] private string skillId;
    [SerializeField] private Image cooldownOverlay;    // 冷却遮罩
    [SerializeField] private Text cooldownText;        // 冷却时间文字
    [SerializeField] private Image[] chargeIndicators; // 充能点
    [SerializeField] private Image gcdOverlay;         // GCD遮罩
    
    [SerializeField] private Color readyColor = Color.white;
    [SerializeField] private Color cooldownColor = new Color(0.4f, 0.4f, 0.4f);
    
    [Header("动画")]
    [SerializeField] private float popScale = 1.2f;
    [SerializeField] private float popDuration = 0.2f;
    
    private CooldownManager cooldownManager;
    private bool wasReady;
    
    void Start()
    {
        cooldownManager = FindObjectOfType<CooldownManager>();
        cooldownManager.OnCooldownExpired += OnCooldownExpired;
    }
    
    void Update()
    {
        if (cooldownManager == null) return;
        
        var state = cooldownManager.GetState(skillId);
        if (state == null) return;
        
        bool isReady = state.IsReady;
        float remaining = cooldownManager.GetRemainingCooldown(skillId);
        
        // 更新冷却遮罩（径向填充）
        if (cooldownOverlay != null)
        {
            cooldownOverlay.fillAmount = 1f - state.Progress;
            cooldownOverlay.color = isReady ? Color.clear : 
                new Color(0, 0, 0, 0.65f);
        }
        
        // 更新冷却文字
        if (cooldownText != null)
        {
            if (remaining > 0)
            {
                cooldownText.gameObject.SetActive(true);
                cooldownText.text = remaining >= 1f ? 
                    Mathf.CeilToInt(remaining).ToString() : 
                    remaining.ToString("F1");
            }
            else
            {
                cooldownText.gameObject.SetActive(false);
            }
        }
        
        // 更新充能指示器
        if (chargeIndicators != null)
        {
            for (int i = 0; i < chargeIndicators.Length; i++)
            {
                if (chargeIndicators[i] != null)
                {
                    chargeIndicators[i].color = i < state.CurrentCharges ? 
                        readyColor : cooldownColor;
                }
            }
        }
        
        // 更新GCD遮罩
        if (gcdOverlay != null)
        {
            float gcdProgress = 1f - cooldownManager.GlobalCooldownRemaining / 1.5f;
            gcdOverlay.fillAmount = Mathf.Clamp01(1f - gcdProgress);
        }
    }
    
    void OnCooldownExpired(string id)
    {
        if (id != skillId) return;
        
        // 播放就绪动画
        transform.localScale = Vector3.one;
        LeanTween.scale(gameObject, Vector3.one * popScale, popDuration * 0.5f)
            .setEaseOutQuad()
            .setOnComplete(() =>
            {
                LeanTween.scale(gameObject, Vector3.one, popDuration * 0.5f)
                    .setEaseInQuad();
            });
    }
    
    void OnDestroy()
    {
        if (cooldownManager != null)
            cooldownManager.OnCooldownExpired -= OnCooldownExpired;
    }
}
```

## 六、网络环境下的冷却同步

在网络游戏中，冷却需要服务端权威验证：

```csharp
/// <summary>
/// 服务端权威冷却校验（伪代码，需集成到具体网络框架）
/// </summary>
public class NetworkCooldownSync
{
    // 客户端：乐观预测（立即在本地触发冷却，不等待服务端确认）
    public void ClientPredictCooldown(string skillId)
    {
        // 本地立即更新UI和状态，提升响应感
        LocalCooldownManager.TriggerCooldown(skillId);
    }
    
    // 服务端返回确认
    public void OnServerConfirmSkillUse(string skillId, float serverTime, bool approved)
    {
        if (approved)
        {
            // 对齐服务端时间（处理网络延迟导致的时间差）
            float latencyOffset = (LocalTime - serverTime) * 0.001f;
            LocalCooldownManager.AdjustCooldown(skillId, -latencyOffset);
        }
        else
        {
            // 服务端拒绝（被攻击/作弊），回滚本地冷却
            LocalCooldownManager.ResetCooldown(skillId);
            // 提示玩家操作被拒绝
        }
    }
    
    // 服务端定期下发冷却快照（心跳同步）
    public void OnReceiveCooldownSnapshot(Dictionary<string, float> snapshot)
    {
        foreach (var kvp in snapshot)
        {
            float serverRemaining = kvp.Value;
            float localRemaining = LocalCooldownManager.GetRemainingCooldown(kvp.Key);
            
            // 允许小误差（<=200ms），超出则强制对齐
            if (Mathf.Abs(serverRemaining - localRemaining) > 0.2f)
            {
                LocalCooldownManager.ForceSetCooldown(kvp.Key, serverRemaining);
            }
        }
    }
    
    private float LocalTime => Time.realtimeSinceStartup;
    private CooldownManager LocalCooldownManager => null; // 注入实际引用
}
```

## 七、冷却系统设计要点

| 功能 | 实现方式 | 注意点 |
|------|----------|--------|
| 基础冷却 | 计时器倒计时 | 使用 `Time.unscaledDeltaTime` 防止暂停影响 |
| 全局冷却 | 独立GCD计时器 | 部分技能可绕过GCD |
| 充能机制 | 独立计时槽数组 | 每个充能独立计时，消耗后开始回复 |
| 冷却缩减 | 百分比乘数 | 设置下限防止无限冷却缩减 |
| 网络同步 | 客户端预测+服务端校验 | 误差在200ms内不需纠正 |
| UI反馈 | 径向填充+数字倒计时 | 就绪时播放提示动画 |

冷却系统虽小，但其设计质量直接影响游戏手感与竞技公平性。在大型项目中务必将其设计为可配置、可热更新的独立系统。
