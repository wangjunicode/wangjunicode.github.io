## 一、战斗数值系统

### 1.1 属性体系设计

```
典型RPG属性体系：

基础属性（策划设计，技术实现）：
- 攻击力（Attack）
- 防御力（Defense）
- 生命值（HP/MaxHP）
- 速度（Speed）
- 暴击率（CritRate）
- 暴击伤害（CritDamage）

派生属性（通过公式计算）：
- 最终攻击力 = 基础攻击力 × (1 + 攻击加成%) + 固定攻击加成
- 最终防御力 = 基础防御力 × (1 + 防御加成%)
- 有效生命值 = HP × (1 + 防御力/(防御力+200))

养成维度（影响属性的方向）：
- 等级（每级固定成长）
- 装备（附加属性）
- 技能（特殊修正）
- Buff/Debuff（临时修改）
```

```csharp
// 属性系统实现：支持多种修改来源
public class AttributeSystem
{
    // 属性枚举
    public enum AttributeType
    {
        Attack, Defense, HP, MaxHP, Speed, CritRate, CritDamage
    }
    
    // 基础属性
    private Dictionary<AttributeType, float> _baseAttributes = new();
    
    // 属性修改器列表（来自装备/Buff/技能等）
    private Dictionary<AttributeType, List<AttributeModifier>> _modifiers = new();
    
    // 计算最终属性值
    public float GetFinalAttribute(AttributeType type)
    {
        float baseValue = _baseAttributes.GetValueOrDefault(type, 0f);
        
        if (!_modifiers.TryGetValue(type, out var mods) || mods.Count == 0)
            return baseValue;
        
        // 分类计算：加法修正 → 乘法修正
        float addBonus = 0f;
        float multiBonus = 1f;
        float baseMultiBonus = 1f;
        
        foreach (var mod in mods)
        {
            switch (mod.Type)
            {
                case ModifierType.FlatAdd:        // 固定值加法 +100攻击
                    addBonus += mod.Value;
                    break;
                case ModifierType.PercentAdd:     // 百分比加法（叠加）+20%攻击
                    addBonus += baseValue * mod.Value;
                    break;
                case ModifierType.PercentMultiply: // 百分比乘法（相乘）×1.2攻击
                    multiBonus *= (1f + mod.Value);
                    break;
                case ModifierType.BaseMultiply:   // 基础值乘法（最先计算）
                    baseMultiBonus *= (1f + mod.Value);
                    break;
            }
        }
        
        return (baseValue * baseMultiBonus + addBonus) * multiBonus;
    }
    
    // 添加修改器（装备强化/Buff施加时调用）
    public void AddModifier(AttributeType type, AttributeModifier modifier)
    {
        if (!_modifiers.ContainsKey(type))
            _modifiers[type] = new List<AttributeModifier>();
        _modifiers[type].Add(modifier);
    }
    
    // 移除修改器（装备卸下/Buff消除时调用）
    public void RemoveModifier(AttributeType type, AttributeModifier modifier)
    {
        _modifiers.TryGetValue(type, out var mods);
        mods?.Remove(modifier);
    }
}

public class AttributeModifier
{
    public ModifierType Type;
    public float Value;
    public string Source; // 来源标识（装备ID、Buff名等），用于调试
    public int Priority;  // 计算优先级（某些特殊Buff需要最后计算）
}

public enum ModifierType { FlatAdd, PercentAdd, PercentMultiply, BaseMultiply }
```

### 1.2 伤害公式设计

```csharp
// 分层伤害公式（技术实现 + 策划理解）
public class DamageCalculator
{
    // 完整伤害流程
    public DamageResult Calculate(AttackContext ctx)
    {
        var result = new DamageResult();
        
        // Step 1: 基础伤害
        float baseDamage = GetBaseDamage(ctx);
        
        // Step 2: 暴击判定
        bool isCrit = IsCritical(ctx.Attacker.GetAttribute(AttributeType.CritRate));
        float critMultiplier = isCrit ? ctx.Attacker.GetAttribute(AttributeType.CritDamage) : 1f;
        
        // Step 3: 技能倍率
        float skillMultiplier = ctx.Skill?.DamageMultiplier ?? 1f;
        
        // Step 4: 攻防计算
        float atkPower = ctx.Attacker.GetAttribute(AttributeType.Attack);
        float defPower = ctx.Defender.GetAttribute(AttributeType.Defense);
        float defReduction = defPower / (defPower + 200f); // 防御公式（经典设计）
        
        // Step 5: 元素反应加成（如果有）
        float elementBonus = GetElementBonus(ctx.AttackElement, ctx.DefenderElement);
        
        // Step 6: 随机浮动（±5%，避免每次伤害完全一样）
        float randomFactor = UnityEngine.Random.Range(0.95f, 1.05f);
        
        // 最终伤害公式
        float finalDamage = baseDamage 
            * (atkPower / 100f + 1f)  // 攻击力加成
            * (1f - defReduction)      // 防御减免
            * skillMultiplier          // 技能倍率
            * critMultiplier           // 暴击
            * elementBonus             // 元素反应
            * randomFactor;            // 随机浮动
        
        // 最低伤害保底（避免0伤害造成体验问题）
        result.Damage = Mathf.Max(1f, Mathf.Floor(finalDamage));
        result.IsCrit = isCrit;
        result.ElementReaction = GetElementReactionType(ctx.AttackElement, ctx.DefenderElement);
        
        return result;
    }
    
    float GetBaseDamage(AttackContext ctx)
    {
        // 技能有自己的基础伤害，普通攻击用攻击力基础值
        return ctx.Skill?.BaseDamage ?? ctx.Attacker.GetAttribute(AttributeType.Attack) * 0.5f;
    }
    
    bool IsCritical(float critRate)
    {
        return UnityEngine.Random.value < Mathf.Clamp01(critRate);
    }
    
    float GetElementBonus(ElementType attack, ElementType defend)
    {
        // 克制关系：火克草=2.0，草克火=0.5
        // 使用配置表驱动，而不是硬编码
        return ElementConfig.GetBonus(attack, defend);
    }
}
```

```csharp
public class ConfigManager : IGameSystem
{
    private Dictionary<Type, Dictionary<int, ConfigBase>> _configs = new();
    
    // 加载配置（从JSON文件）
    public void LoadConfig<T>(string filePath) where T : ConfigBase
    {
        string json = File.ReadAllText(filePath);
        T[] items = JsonConvert.DeserializeObject<T[]>(json);
        
        var dict = new Dictionary<int, ConfigBase>(items.Length);
        foreach (var item in items) dict[item.Id] = item;
        
        _configs[typeof(T)] = dict;
        Debug.Log($"加载配置 {typeof(T).Name}: {items.Length} 条");
    }
    
    // 获取配置
    public T Get<T>(int id) where T : ConfigBase
    {
        if (!_configs.TryGetValue(typeof(T), out var dict)) return null;
        if (!dict.TryGetValue(id, out var config)) return null;
        return (T)config;
    }
    
    // 获取所有配置
    public IEnumerable<T> GetAll<T>() where T : ConfigBase
    {
        if (!_configs.TryGetValue(typeof(T), out var dict))
            return Enumerable.Empty<T>();
        return dict.Values.Cast<T>();
    }
    
    public void Initialize() { }
    public void Shutdown() { _configs.Clear(); }
}
```

## 三、虚拟经济系统

### 3.1 货币体系设计

```
典型手游货币体系：

免费货币（软货币）：
- 来源：游戏内容奖励（任务、副本、登录）
- 用途：购买普通道具、强化材料
- 设计意图：让玩家保持参与感

付费货币（硬货币）：
- 来源：充值（现实货币购买）
- 用途：购买高价值内容（抽卡、限时道具）
- 设计意图：游戏收益来源

中间货币（绑定货币）：
- 来源：免费获得的付费货币（首充奖励、活动等）
- 用途：同硬货币，但不能交易/退款
- 设计意图：区分真实充值和赠送

货币设计原则：
1. 软货币要"够用但不过剩"（让玩家有目标感）
2. 硬货币要"稀缺但可获得"（保持氪金吸引力）
3. 严格的单向流通（付费货币不能变回真实货币）
```

### 3.3 装备强化系统

```csharp
// 装备强化：技术实现与策划配置的结合

[Serializable]
public class EquipEnhanceConfig
{
    public int EquipLevel;          // 强化等级
    public float SuccessRate;       // 成功率（0-1）
    public float BreakRate;         // 装备损毁率（高风险强化）
    public float DowngradeRate;     // 降级率（+10以上失败可能掉级）
    public int CostGold;            // 金币消耗
    public int CostMaterial;        // 材料消耗
    public float AttackBonus;       // 强化成功后攻击加成
    public float DefenseBonus;      // 防御加成
}

public class EquipEnhanceSystem
{
    private ConfigManager _configManager;
    
    public EnhanceResult TryEnhance(Equipment equip, int materialCount, int goldAmount)
    {
        var config = _configManager.Get<EquipEnhanceConfig>(equip.Level + 1);
        if (config == null) return EnhanceResult.MaxLevel;
        
        // 消耗资源（必须在判定结果前扣除！）
        if (!ConsumeResources(goldAmount, materialCount, config))
            return EnhanceResult.InsufficientResources;
        
        float roll = UnityEngine.Random.value;
        
        // 强化结果判定
        if (roll < config.SuccessRate)
        {
            // 成功
            equip.Level++;
            ApplyEnhanceBonuses(equip, config);
            return EnhanceResult.Success;
        }
        else if (roll < config.SuccessRate + config.BreakRate)
        {
            // 装备损毁（仅高危强化才有）
            equip.IsBroken = true;
            return EnhanceResult.Break;
        }
        else if (roll < config.SuccessRate + config.BreakRate + config.DowngradeRate)
        {
            // 失败降级
            equip.Level = Mathf.Max(0, equip.Level - 1);
            return EnhanceResult.Downgrade;
        }
        else
        {
            // 失败保级
            return EnhanceResult.Fail;
        }
    }
    
    bool ConsumeResources(int gold, int material, EquipEnhanceConfig config)
    {
        // 服务端：原子性扣除（所有扣除成功才继续）
        bool canAfford = gold >= config.CostGold && material >= config.CostMaterial;
        if (!canAfford) return false;
        
        // 实际扣除
        DeductGold(config.CostGold);
        DeductMaterial(config.CostMaterial);
        return true;
    }
    
    void ApplyEnhanceBonuses(Equipment equip, EquipEnhanceConfig config)
    {
        equip.Attributes.AddModifier(AttributeType.Attack, 
            new AttributeModifier { Type = ModifierType.FlatAdd, Value = config.AttackBonus, Source = $"Enhance+{equip.Level}" });
    }
    
    void DeductGold(int amount) { }
    void DeductMaterial(int amount) { }
    
    public enum EnhanceResult { Success, Fail, Break, Downgrade, MaxLevel, InsufficientResources }
}
```

## 四、程序技术负责人与数值策划的协作

### 4.1 高效协作模式

```
合作原则：

技术负责人提供：
→ 配置表结构设计（什么字段，什么类型）
→ 数值范围的技术约束（不能超过什么值会溢出）
→ 公式实现文档（让策划能验证计算结果）
→ 数值调试工具（可视化当前公式效果）
→ 热更新支持（策划修改配置无需重新发包）

数值策划提供：
→ 具体的公式参数（攻防系数、爆率等）
→ 养成曲线（1-100级的成长节奏）
→ 经济循环设计（玩家获取和消耗的平衡）

避免的合作反模式：
❌ 程序直接决定数值公式（没有策划验证）
❌ 策划不懂技术约束（提需求时不考虑实现）
❌ 数值更改需要改代码（热更新必要性）
```

## 总结

数值系统的技术核心：

```
架构设计原则：
✅ 数据驱动（配置表控制，不硬编码数值）
✅ 分层计算（基础值 → 加法修正 → 乘法修正）
✅ 可审计（每次计算都能追溯来源）
✅ 可热更新（策划调整不需要发版）
✅ 服务端权威（重要数值计算在服务端）

技术负责人的数值责任：
→ 建立属性修改器系统（装备/Buff/技能统一管理）
→ 实现数值溢出保护（整数溢出可能导致负伤害等BUG）
→ 提供策划友好的调试工具
→ 建立数值变更的回归测试（改了公式不影响已有内容）
→ 识别经济系统风险（防刷机制是技术+策划共同责任）
```
