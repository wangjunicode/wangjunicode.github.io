## 一、类型安全泛型事件总线

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 强类型事件总线（不依赖字符串Key，编译期检查）
/// 用法：EventBus<PlayerDeathEvent>.Publish(new PlayerDeathEvent { PlayerId = id });
/// </summary>
public static class EventBus<TEvent> where TEvent : struct
{
    private static List<(int Priority, Action<TEvent> Handler)> handlers
        = new List<(int, Action<TEvent>)>();
    
    private static bool isPublishing; // 防止在事件回调中修改监听者列表
    private static List<(int Priority, Action<TEvent> Handler)> pendingAdd
        = new List<(int, Action<TEvent>)>();
    private static List<Action<TEvent>> pendingRemove
        = new List<Action<TEvent>>();
    
    // 统计（调试用）
    public static int ListenerCount => handlers.Count;

    /// <summary>
    /// 订阅事件
    /// </summary>
    /// <param name="handler">处理函数</param>
    /// <param name="priority">优先级（数值越大越先执行）</param>
    public static void Subscribe(Action<TEvent> handler, int priority = 0)
    {
        if (handler == null) return;
        
        if (isPublishing)
        {
            pendingAdd.Add((priority, handler));
            return;
        }
        
        // 检查重复订阅
        for (int i = 0; i < handlers.Count; i++)
        {
            if (handlers[i].Handler == handler) return;
        }
        
        handlers.Add((priority, handler));
        // 按优先级降序排序
        handlers.Sort((a, b) => b.Priority.CompareTo(a.Priority));
    }

    /// <summary>
    /// 取消订阅
    /// </summary>
    public static void Unsubscribe(Action<TEvent> handler)
    {
        if (handler == null) return;
        
        if (isPublishing)
        {
            pendingRemove.Add(handler);
            return;
        }
        
        handlers.RemoveAll(h => h.Handler == handler);
    }

    /// <summary>
    /// 发布事件
    /// </summary>
    public static void Publish(TEvent evt)
    {
        isPublishing = true;
        
        try
        {
            for (int i = 0; i < handlers.Count; i++)
            {
                try
                {
                    handlers[i].Handler(evt);
                }
                catch (Exception e)
                {
                    Debug.LogError($"[EventBus<{typeof(TEvent).Name}>] Handler error: {e.Message}\n{e.StackTrace}");
                }
            }
        }
        finally
        {
            isPublishing = false;
            
            // 处理延迟的订阅/取消
            foreach (var add in pendingAdd)
                Subscribe(add.Handler, add.Priority);
            pendingAdd.Clear();
            
            foreach (var remove in pendingRemove)
                Unsubscribe(remove);
            pendingRemove.Clear();
        }
    }

    /// <summary>
    /// 清除所有监听者（场景切换时调用）
    /// </summary>
    public static void ClearAll()
    {
        handlers.Clear();
        pendingAdd.Clear();
        pendingRemove.Clear();
    }
}

// ============ 事件定义（用结构体确保栈分配，无GC）============

public struct PlayerDeathEvent
{
    public string PlayerId;
    public Vector3 DeathPosition;
    public string KillerId;
}

public struct LevelUpEvent
{
    public int OldLevel;
    public int NewLevel;
    public int StatPointsGained;
}

public struct ItemPickupEvent
{
    public string ItemId;
    public int Count;
    public ItemRarity Rarity;
}

public struct SceneLoadedEvent
{
    public string SceneName;
    public float LoadTime;
}

public struct UIOpenEvent
{
    public string PanelName;
    public bool IsPush; // 是Push还是直接Open
}
```

## 二、可取消事件（Interceptable Event）

```csharp
/// <summary>
/// 可拦截事件（处理函数可以阻止后续处理）
/// </summary>
public struct InterceptableEvent<TEvent> where TEvent : struct
{
    public TEvent Data;
    public bool IsCancelled;
    
    public void Cancel() => IsCancelled = true;
}

/// <summary>
/// 可拦截事件总线
/// </summary>
public static class InterceptableEventBus<TEvent> where TEvent : struct
{
    private static List<(int Priority, Func<InterceptableEvent<TEvent>, InterceptableEvent<TEvent>> Handler)>
        handlers = new List<(int, Func<InterceptableEvent<TEvent>, InterceptableEvent<TEvent>>)>();

    public static void Subscribe(
        Func<InterceptableEvent<TEvent>, InterceptableEvent<TEvent>> handler, 
        int priority = 0)
    {
        handlers.Add((priority, handler));
        handlers.Sort((a, b) => b.Priority.CompareTo(a.Priority));
    }

    public static InterceptableEvent<TEvent> Publish(TEvent evt)
    {
        var wrapper = new InterceptableEvent<TEvent> { Data = evt };
        
        foreach (var h in handlers)
        {
            wrapper = h.Handler(wrapper);
            if (wrapper.IsCancelled)
                break; // 被拦截，停止传播
        }
        
        return wrapper;
    }
}

// 使用示例：伤害可以被护盾拦截
public struct DamageEvent { public float Amount; public string TargetId; }

void ExampleInterceptUsage()
{
    // 注册高优先级护盾处理（先执行）
    InterceptableEventBus<DamageEvent>.Subscribe(e =>
    {
        if (HasShield)
        {
            e.Data = new DamageEvent { Amount = 0, TargetId = e.Data.TargetId };
            e.Cancel(); // 护盾吸收，阻止后续伤害处理
        }
        return e;
    }, priority: 100);

    // 注册普通伤害处理（低优先级）
    InterceptableEventBus<DamageEvent>.Subscribe(e =>
    {
        ApplyDamage(e.Data.Amount);
        return e;
    }, priority: 0);
}

bool HasShield = false;
void ApplyDamage(float amount) { }
```

## 三、延迟事件队列

```csharp
/// <summary>
/// 延迟事件队列（下帧执行 / 帧末执行）
/// 避免在物理回调、协程中直接发布引发的时序问题
/// </summary>
public class DeferredEventQueue : MonoBehaviour
{
    private static DeferredEventQueue instance;
    public static DeferredEventQueue Instance => instance;

    private Queue<Action> frameEndQueue = new Queue<Action>();
    private Queue<(Action Action, float Delay)> delayedQueue 
        = new Queue<(Action, float)>();
    private List<(Action Action, float ExecuteTime)> timedQueue 
        = new List<(Action, float)>();

    void Awake() { instance = this; }

    void LateUpdate()
    {
        // 执行帧末队列
        while (frameEndQueue.Count > 0)
        {
            try { frameEndQueue.Dequeue()(); }
            catch (Exception e) { Debug.LogError(e.Message); }
        }
        
        // 执行到时的延迟事件
        for (int i = timedQueue.Count - 1; i >= 0; i--)
        {
            if (Time.time >= timedQueue[i].ExecuteTime)
            {
                try { timedQueue[i].Action(); }
                catch (Exception e) { Debug.LogError(e.Message); }
                timedQueue.RemoveAt(i);
            }
        }
    }

    /// <summary>
    /// 帧末执行（当前帧 LateUpdate）
    /// </summary>
    public void DeferToFrameEnd(Action action)
    {
        frameEndQueue.Enqueue(action);
    }

    /// <summary>
    /// 延迟 seconds 秒后执行
    /// </summary>
    public void DeferBySeconds(Action action, float seconds)
    {
        timedQueue.Add((action, Time.time + seconds));
    }

    /// <summary>
    /// 下一帧执行
    /// </summary>
    public void DeferToNextFrame(Action action)
    {
        StartCoroutine(ExecuteNextFrame(action));
    }

    System.Collections.IEnumerator ExecuteNextFrame(Action action)
    {
        yield return null;
        action?.Invoke();
    }
}
```

## 四、事件调试监控

```csharp
/// <summary>
/// 事件调试器（在 Inspector/编辑器窗口中可视化事件流）
/// </summary>
public class EventDebugger : MonoBehaviour
{
    [SerializeField] private bool enableLogging = true;
    [SerializeField] private int maxLogEntries = 100;
    
    private static EventDebugger instance;
    public static EventDebugger Instance => instance;
    
    private List<EventLogEntry> logs = new List<EventLogEntry>();

    public class EventLogEntry
    {
        public string EventType;
        public string EventData;
        public float Timestamp;
        public int ListenerCount;
    }

    void Awake() { instance = this; }

    public void Log<T>(T evt, int listenerCount)
    {
        if (!enableLogging) return;
        
        logs.Add(new EventLogEntry
        {
            EventType = typeof(T).Name,
            EventData = evt.ToString(),
            Timestamp = Time.time,
            ListenerCount = listenerCount
        });
        
        if (logs.Count > maxLogEntries)
            logs.RemoveAt(0);
    }

    public List<EventLogEntry> GetLogs() => logs;

    public void Clear() => logs.Clear();
}
```

## 五、事件系统分层策略

```
事件分层架构：

全局层（Global）
├── 跨场景事件（玩家数据变化、游戏状态变化）
└── EventBus<GlobalEvent>（常驻内存，不随场景清除）

场景层（Scene）
├── 场景内实体通信
└── EventBus<SceneEvent>（场景卸载时清除）

组件层（Component）
├── 同一 GameObject 上的组件通信
└── 直接方法调用 / C# event（最快）
```

**选型建议：**

| 通信方式 | 适用场景 | 性能 | 耦合度 |
|----------|----------|------|--------|
| 直接调用 | 同组件/已知引用 | 最快 | 高 |
| C# event | 一对多，同类通信 | 快 | 中 |
| EventBus<T> | 跨模块解耦通信 | 中 | 低 |
| UnityEvent | 策划可配置回调 | 慢（反射）| 低 |
| SendMessage | 不推荐 | 极慢 | 低 |
