## 一、3D 音频衰减设计

```csharp
using UnityEngine;
using System.Collections.Generic;

/// <summary>
/// 音效发射器（高级3D音频封装）
/// </summary>
[RequireComponent(typeof(AudioSource))]
public class SpatialAudioEmitter : MonoBehaviour
{
    [Header("衰减配置")]
    [SerializeField] private float minDistance = 1f;       // 最大音量距离
    [SerializeField] private float maxDistance = 30f;      // 最小音量距离
    [SerializeField] private AudioRolloffMode rolloffMode = AudioRolloffMode.Custom;
    
    [Header("遮挡设置")]
    [SerializeField] private bool enableOcclusion = true;  // 是否启用遮挡
    [SerializeField] private LayerMask occlusionMask;      // 遮挡物层
    [SerializeField, Range(0, 1)] private float occludedVolumeFactor = 0.3f; // 被遮挡时的音量比例
    [SerializeField, Range(0, 1)] private float occludedLowpassFactor = 0.3f; // 被遮挡时的高频衰减
    
    [Header("多普勒")]
    [SerializeField, Range(0, 3)] private float dopplerLevel = 1f;

    private AudioSource audioSource;
    private AudioLowPassFilter lowPassFilter;
    private float targetVolume;
    private float targetCutoff;
    private Transform listenerTransform;
    private float occlusionCheckTimer;
    private const float OCCLUSION_CHECK_INTERVAL = 0.05f;

    void Awake()
    {
        audioSource = GetComponent<AudioSource>();
        lowPassFilter = GetOrAddComponent<AudioLowPassFilter>();
        
        // 配置 AudioSource
        audioSource.spatialBlend = 1f;       // 完全3D
        audioSource.rolloffMode = rolloffMode;
        audioSource.minDistance = minDistance;
        audioSource.maxDistance = maxDistance;
        audioSource.dopplerLevel = dopplerLevel;
        
        // 自定义衰减曲线（更自然的对数衰减）
        if (rolloffMode == AudioRolloffMode.Custom)
        {
            var curve = new AnimationCurve();
            curve.AddKey(0, 1f);
            curve.AddKey(0.1f, 0.8f);
            curve.AddKey(0.5f, 0.3f);
            curve.AddKey(1f, 0f);
            audioSource.SetCustomCurve(AudioSourceCurveType.CustomRolloff, curve);
        }
        
        targetVolume = audioSource.volume;
        targetCutoff = 22000f; // 无遮挡时通过全频率
        
        // 找到监听器
        listenerTransform = FindObjectOfType<AudioListener>()?.transform;
    }

    void Update()
    {
        if (!enableOcclusion || listenerTransform == null) return;
        
        // 限频检测遮挡（不需要每帧都算）
        occlusionCheckTimer += Time.deltaTime;
        if (occlusionCheckTimer >= OCCLUSION_CHECK_INTERVAL)
        {
            occlusionCheckTimer = 0;
            UpdateOcclusion();
        }
        
        // 平滑应用遮挡效果
        audioSource.volume = Mathf.Lerp(audioSource.volume, targetVolume, 15f * Time.deltaTime);
        lowPassFilter.cutoffFrequency = Mathf.Lerp(
            lowPassFilter.cutoffFrequency, targetCutoff, 10f * Time.deltaTime);
    }

    void UpdateOcclusion()
    {
        Vector3 listenerPos = listenerTransform.position;
        Vector3 emitterPos = transform.position;
        Vector3 direction = listenerPos - emitterPos;
        float distance = direction.magnitude;
        
        // 发射多条射线检测遮挡（单条射线不稳定）
        int hitCount = 0;
        int rayCount = 3;
        
        for (int i = 0; i < rayCount; i++)
        {
            Vector3 spread = Random.insideUnitSphere * 0.3f;
            Vector3 target = listenerPos + spread;
            
            if (Physics.Raycast(emitterPos, (target - emitterPos).normalized, 
                distance, occlusionMask))
            {
                hitCount++;
            }
        }
        
        float occlusionFactor = (float)hitCount / rayCount;
        
        // 更新目标音量和低通滤波
        float baseVolume = 1f; // 可以从原始音量获取
        targetVolume = Mathf.Lerp(baseVolume, baseVolume * occludedVolumeFactor, occlusionFactor);
        targetCutoff = Mathf.Lerp(22000f, 800f * occludedLowpassFactor + 800f, occlusionFactor);
    }

    public void PlayOneShot(AudioClip clip, float volumeScale = 1f)
    {
        audioSource.PlayOneShot(clip, volumeScale);
    }

    T GetOrAddComponent<T>() where T : Component 
        => GetComponent<T>() ?? gameObject.AddComponent<T>();
}
```

## 二、音效表面识别（脚步声）

```csharp
/// <summary>
/// 脚步声表面识别系统
/// </summary>
public class FootstepSoundSystem : MonoBehaviour
{
    [System.Serializable]
    public class SurfaceSound
    {
        public string SurfaceTag;           // 地面标签（如 "Grass", "Stone", "Water"）
        public PhysicMaterial PhysicMat;    // 物理材质
        public AudioClip[] WalkSounds;      // 走路声音池
        public AudioClip[] RunSounds;       // 奔跑声音池
        [Range(0.5f, 2f)] public float VolumeMultiplier = 1f;
    }

    [SerializeField] private SurfaceSound[] surfaces;
    [SerializeField] private AudioSource audioSource;
    [SerializeField] private float stepInterval = 0.4f;     // 走路时步频
    [SerializeField] private float runStepInterval = 0.25f; // 跑步时步频
    [SerializeField] private LayerMask groundMask;

    private float stepTimer;
    private bool isRunning;

    /// <summary>
    /// 由动画事件调用
    /// </summary>
    public void OnFootstep(bool isLeftFoot)
    {
        // 检测脚下地面类型
        if (!Physics.Raycast(transform.position + Vector3.up * 0.1f, 
            Vector3.down, out RaycastHit hit, 0.5f, groundMask))
            return;
        
        SurfaceSound surface = DetectSurface(hit);
        PlayFootstepSound(surface);
    }

    SurfaceSound DetectSurface(RaycastHit hit)
    {
        // 优先按物理材质匹配
        PhysicMaterial mat = hit.collider.sharedMaterial;
        if (mat != null)
        {
            var surface = System.Array.Find(surfaces, s => s.PhysicMat == mat);
            if (surface != null) return surface;
        }
        
        // 按Tag匹配
        string tag = hit.collider.tag;
        var tagSurface = System.Array.Find(surfaces, s => s.SurfaceTag == tag);
        if (tagSurface != null) return tagSurface;
        
        // 默认（第一个配置）
        return surfaces.Length > 0 ? surfaces[0] : null;
    }

    void PlayFootstepSound(SurfaceSound surface)
    {
        if (surface == null) return;
        
        var clips = isRunning ? surface.RunSounds : surface.WalkSounds;
        if (clips == null || clips.Length == 0) return;
        
        var clip = clips[Random.Range(0, clips.Length)];
        audioSource.PlayOneShot(clip, surface.VolumeMultiplier);
    }
}
```

## 三、混响区域（室内/洞穴效果）

```csharp
/// <summary>
/// 混响区域触发器（进入触发音频效果变化）
/// </summary>
[RequireComponent(typeof(Collider))]
public class ReverbZoneTrigger : MonoBehaviour
{
    [SerializeField] private AudioReverbPreset reverbPreset = AudioReverbPreset.Cave;
    [SerializeField] private float transitionTime = 1f;     // 过渡时间
    
    private AudioReverbZone unityReverbZone;

    void Awake()
    {
        GetComponent<Collider>().isTrigger = true;
        unityReverbZone = GetOrAdd<AudioReverbZone>();
        unityReverbZone.reverbPreset = reverbPreset;
        unityReverbZone.minDistance = 0;
        unityReverbZone.maxDistance = 1000f;
        unityReverbZone.gameObject.SetActive(false);
    }

    void OnTriggerEnter(Collider other)
    {
        if (!other.CompareTag("Player")) return;
        unityReverbZone.gameObject.SetActive(true);
        AudioManager.Instance?.SetReverb(reverbPreset, transitionTime);
    }

    void OnTriggerExit(Collider other)
    {
        if (!other.CompareTag("Player")) return;
        AudioManager.Instance?.SetReverb(AudioReverbPreset.Off, transitionTime);
    }

    T GetOrAdd<T>() where T : Component => GetComponent<T>() ?? gameObject.AddComponent<T>();
}
```

## 四、动态音乐系统

```csharp
/// <summary>
/// 动态音乐系统（根据战斗状态切换音乐）
/// </summary>
public class DynamicMusicSystem : MonoBehaviour
{
    [System.Serializable]
    public class MusicState
    {
        public string StateName;
        public AudioClip IntroClip;     // 进入时播放一次
        public AudioClip LoopClip;      // 循环播放
        public float Volume = 1f;
        public float FadeTime = 2f;
    }

    [SerializeField] private MusicState[] musicStates;
    [SerializeField] private AudioSource musicSource1;  // 双缓冲交叉淡入淡出
    [SerializeField] private AudioSource musicSource2;

    private int currentStateIndex = -1;
    private bool useSource1 = true;

    public void TransitionToState(string stateName)
    {
        int index = System.Array.FindIndex(musicStates, s => s.StateName == stateName);
        if (index < 0 || index == currentStateIndex) return;
        
        currentStateIndex = index;
        var state = musicStates[index];
        
        AudioSource active = useSource1 ? musicSource1 : musicSource2;
        AudioSource inactive = useSource1 ? musicSource2 : musicSource1;
        useSource1 = !useSource1;
        
        // 交叉淡入淡出
        inactive.clip = state.LoopClip;
        inactive.loop = true;
        inactive.volume = 0;
        inactive.Play();
        
        StartCoroutine(CrossFade(active, inactive, state.Volume, state.FadeTime));
    }

    System.Collections.IEnumerator CrossFade(AudioSource from, AudioSource to, 
        float targetVolume, float duration)
    {
        float elapsed = 0;
        float startVolume = from.volume;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            from.volume = Mathf.Lerp(startVolume, 0, t);
            to.volume = Mathf.Lerp(0, targetVolume, t);
            yield return null;
        }
        
        from.Stop();
        to.volume = targetVolume;
    }
}
```

## 五、音频中间件选型

| 方案 | 优点 | 缺点 | 适用 |
|------|------|------|------|
| Unity Audio | 内置，零成本 | 功能有限 | 小项目/独立游戏 |
| FMOD Studio | 功能强大，音频师工具友好 | 学习曲线，授权费 | 中大型项目 |
| Wwise | 企业级，最强功能 | 昂贵，复杂 | AAA项目 |
| Fabric | 轻量，介于FMOD和Unity之间 | 社区小 | 中型项目 |
