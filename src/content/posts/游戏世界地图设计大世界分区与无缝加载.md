## 一、大世界分区设计

```
大世界分区架构：

World Grid (世界网格)
每格大小: 500x500 单位

[0,3][1,3][2,3][3,3]
[0,2][1,2][2,2][3,2]  ← 玩家所在 [2,2]

[0,0][1,0][2,0][3,0]

加载区域（玩家周围3x3格）：
[1,1][2,1][3,1]
[1,2][2,2][3,2]  ← 玩家
[1,3][2,3][3,3]

卸载区域（超出5x5范围则卸载）
```

## 二、流式加载管理器

```csharp
using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.SceneManagement;

public class WorldStreamingManager : MonoBehaviour
{
    [Header("分区配置")]
    [SerializeField] private float chunkSize = 500f;     // 每格大小
    [SerializeField] private int loadRadius = 1;         // 加载半径（格）
    [SerializeField] private int unloadRadius = 3;       // 卸载半径（格）
    [SerializeField] private Transform playerTransform;
    
    private Vector2Int lastPlayerChunk = Vector2Int.one * int.MinValue;
    private Dictionary<Vector2Int, ChunkState> chunkStates 
        = new Dictionary<Vector2Int, ChunkState>();
    private Queue<Vector2Int> loadQueue = new Queue<Vector2Int>();
    private HashSet<Vector2Int> unloadQueue = new HashSet<Vector2Int>();

    enum ChunkState { Unloaded, Loading, Loaded, Unloading }

    void Update()
    {
        Vector2Int currentChunk = WorldToChunk(playerTransform.position);
        if (currentChunk != lastPlayerChunk)
        {
            lastPlayerChunk = currentChunk;
            UpdateLoadedChunks(currentChunk);
        }
    }

    void UpdateLoadedChunks(Vector2Int center)
    {
        var shouldLoad = new HashSet<Vector2Int>();
        
        for (int x = -loadRadius; x <= loadRadius; x++)
            for (int y = -loadRadius; y <= loadRadius; y++)
                shouldLoad.Add(new Vector2Int(center.x + x, center.y + y));
        
        // 加载新区域
        foreach (var chunk in shouldLoad)
        {
            if (!chunkStates.ContainsKey(chunk) || 
                chunkStates[chunk] == ChunkState.Unloaded)
            {
                chunkStates[chunk] = ChunkState.Loading;
                loadQueue.Enqueue(chunk);
            }
        }
        
        // 卸载超出范围区域
        foreach (var kv in chunkStates)
        {
            if (kv.Value != ChunkState.Loaded) continue;
            float dist = Vector2Int.Distance(kv.Key, center);
            if (dist > unloadRadius)
                unloadQueue.Add(kv.Key);
        }
        
        StartCoroutine(ProcessQueues());
    }

    IEnumerator ProcessQueues()
    {
        // 每帧最多加载1个，避免卡顿
        if (loadQueue.Count > 0)
        {
            var chunk = loadQueue.Dequeue();
            yield return StartCoroutine(LoadChunk(chunk));
        }
        
        // 卸载
        if (unloadQueue.Count > 0)
        {
            var chunks = new List<Vector2Int>(unloadQueue);
            unloadQueue.Clear();
            foreach (var chunk in chunks)
                yield return StartCoroutine(UnloadChunk(chunk));
        }
    }

    IEnumerator LoadChunk(Vector2Int chunk)
    {
        string sceneName = $"World_{chunk.x}_{chunk.y}";
        
        if (IsSceneLoaded(sceneName))
        {
            chunkStates[chunk] = ChunkState.Loaded;
            yield break;
        }
        
        var op = SceneManagement.SceneManager.LoadSceneAsync(sceneName, LoadSceneMode.Additive);
        if (op == null) { chunkStates[chunk] = ChunkState.Unloaded; yield break; }
        
        op.allowSceneActivation = true;
        yield return op;
        
        chunkStates[chunk] = ChunkState.Loaded;
        Debug.Log($"[World] 已加载区块: {chunk}");
    }

    IEnumerator UnloadChunk(Vector2Int chunk)
    {
        string sceneName = $"World_{chunk.x}_{chunk.y}";
        if (!IsSceneLoaded(sceneName)) 
        {
            chunkStates[chunk] = ChunkState.Unloaded;
            yield break;
        }
        
        chunkStates[chunk] = ChunkState.Unloading;
        var op = SceneManagement.SceneManager.UnloadSceneAsync(sceneName);
        yield return op;
        
        chunkStates[chunk] = ChunkState.Unloaded;
        Debug.Log($"[World] 已卸载区块: {chunk}");
    }

    Vector2Int WorldToChunk(Vector3 worldPos)
    {
        return new Vector2Int(
            Mathf.FloorToInt(worldPos.x / chunkSize),
            Mathf.FloorToInt(worldPos.z / chunkSize));
    }

    bool IsSceneLoaded(string sceneName)
    {
        var scene = SceneManagement.SceneManager.GetSceneByName(sceneName);
        return scene.isLoaded;
    }
}
```

## 三、浮点精度问题解决

```csharp
/// <summary>
/// 大世界浮点精度修正（当玩家远离原点时，浮点数精度降低）
/// 解决方案：世界原点跟随玩家（Origin Shift）
/// </summary>
public class OriginShiftManager : MonoBehaviour
{
    [SerializeField] private float shiftThreshold = 2000f;
    [SerializeField] private Transform playerTransform;

    void Update()
    {
        if (playerTransform.position.magnitude > shiftThreshold)
            ShiftOrigin();
    }

    void ShiftOrigin()
    {
        Vector3 shift = playerTransform.position;
        
        // 将所有场景对象平移
        foreach (var root in UnityEngine.SceneManagement.SceneManager.GetActiveScene().GetRootGameObjects())
            root.transform.position -= shift;
        
        // 更新内部世界坐标记录
        WorldPositionTracker.Offset += shift;
        
        Debug.Log($"[OriginShift] 坐标系偏移: {shift}");
    }
}

public static class WorldPositionTracker
{
    public static Vector3 Offset;
    
    public static Vector3 LocalToWorld(Vector3 local) => local + Offset;
    public static Vector3 WorldToLocal(Vector3 world) => world - Offset;
}
```

## 四、快速旅行系统

```csharp
public class FastTravelSystem : MonoBehaviour
{
    [SerializeField] private FastTravelPoint[] travelPoints;
    
    public async System.Threading.Tasks.Task TravelTo(string pointId)
    {
        var point = System.Array.Find(travelPoints, p => p.PointId == pointId);
        if (point == null) return;
        
        // 加载目标区域
        UIManager.Instance?.ShowLoadingScreen();
        await WorldStreamingManager.Instance?.LoadAreaAroundPoint(point.WorldPosition);
        
        // 传送玩家
        PlayerController.Instance?.Teleport(point.WorldPosition);
        UIManager.Instance?.HideLoadingScreen();
    }
}

[Serializable]
public class FastTravelPoint
{
    public string PointId;
    public string DisplayName;
    public Vector3 WorldPosition;
    public bool IsUnlocked;
    public Sprite Icon;
}
```
