## 一、活动数据模型

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

public enum ActivityStatus { NotStarted, Active, Ended, ClaimAvailable }
public enum ActivityType { Login, Collection, Exchange, Challenge, Gacha, Ranking }

[Serializable]
public class ActivityData
{
    public string ActivityId;
    public string Title;
    public string Description;
    public ActivityType Type;
    public long StartTime;          // Unix时间戳
    public long EndTime;
    public long ClaimEndTime;       // 领取截止时间（可能比活动结束晚）
    public Sprite Banner;
    public int Priority;            // 展示优先级（越大越靠前）
    public bool IsNew;              // 新活动标记
    
    public ActivityStatus GetStatus()
    {
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        if (now < StartTime) return ActivityStatus.NotStarted;
        if (now < EndTime)   return ActivityStatus.Active;
        if (ClaimEndTime > 0 && now < ClaimEndTime) return ActivityStatus.ClaimAvailable;
        return ActivityStatus.Ended;
    }
    
    public float GetProgress()
    {
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        if (now < StartTime) return 0;
        if (now >= EndTime) return 1;
        return (float)(now - StartTime) / (EndTime - StartTime);
    }
    
    public TimeSpan GetTimeRemaining()
    {
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        long target = GetStatus() == ActivityStatus.Active ? EndTime : ClaimEndTime;
        return TimeSpan.FromSeconds(Math.Max(0, target - now));
    }
}
```

## 二、活动管理器

```csharp
public class ActivityManager : MonoBehaviour
{
    private static ActivityManager instance;
    public static ActivityManager Instance => instance;
    
    private List<ActivityData> activities = new List<ActivityData>();
    
    public event Action OnActivityListUpdated;
    public event Action<int> OnRedDotCountChanged;

    void Awake() { instance = this; }

    public async System.Threading.Tasks.Task LoadActivities()
    {
        activities = await NetworkService.GetActivityList();
        
        // 过滤已完全结束的活动（超过ClaimEndTime）
        activities.RemoveAll(a => a.GetStatus() == ActivityStatus.Ended);
        
        // 排序：Active > ClaimAvailable > NotStarted，同状态按Priority降序
        activities.Sort((a, b) =>
        {
            var statusOrder = GetStatusOrder(a.GetStatus());
            var bStatusOrder = GetStatusOrder(b.GetStatus());
            if (statusOrder != bStatusOrder) return statusOrder.CompareTo(bStatusOrder);
            return b.Priority.CompareTo(a.Priority);
        });
        
        UpdateRedDot();
        OnActivityListUpdated?.Invoke();
    }

    int GetStatusOrder(ActivityStatus s) => s switch
    {
        ActivityStatus.ClaimAvailable => 0, // 最优先（有东西可领）
        ActivityStatus.Active => 1,
        ActivityStatus.NotStarted => 2,
        _ => 3
    };

    void UpdateRedDot()
    {
        int count = activities.FindAll(a => 
            a.GetStatus() == ActivityStatus.ClaimAvailable || a.IsNew).Count;
        OnRedDotCountChanged?.Invoke(count);
    }

    public List<ActivityData> GetActiveActivities() =>
        activities.FindAll(a => a.GetStatus() == ActivityStatus.Active);
    
    public ActivityData GetActivity(string id) =>
        activities.Find(a => a.ActivityId == id);
}
```

## 三、活动页面基类

```csharp
/// <summary>
/// 活动UI基类（所有活动类型的公共框架）
/// </summary>
public abstract class ActivityPanelBase : MonoBehaviour
{
    protected ActivityData activityData;
    
    [SerializeField] protected UnityEngine.UI.Text titleText;
    [SerializeField] protected UnityEngine.UI.Text countdownText;
    [SerializeField] protected UnityEngine.UI.Button closeButton;

    public void Initialize(ActivityData data)
    {
        activityData = data;
        titleText.text = data.Title;
        closeButton.onClick.AddListener(() => gameObject.SetActive(false));
        
        OnInitialize(data);
        StartCoroutine(UpdateCountdown());
    }

    protected abstract void OnInitialize(ActivityData data);

    System.Collections.IEnumerator UpdateCountdown()
    {
        while (gameObject.activeSelf)
        {
            var remaining = activityData.GetTimeRemaining();
            countdownText.text = FormatRemaining(remaining);
            yield return new WaitForSeconds(1f);
        }
    }

    string FormatRemaining(TimeSpan ts)
    {
        if (ts.TotalDays >= 1) return $"{(int)ts.TotalDays}天{ts.Hours:D2}:{ts.Minutes:D2}:{ts.Seconds:D2}";
        return $"{ts.Hours:D2}:{ts.Minutes:D2}:{ts.Seconds:D2}";
    }
}
```

## 四、活动系统运营规范

| 要素 | 规范 |
|------|------|
| 活动密度 | 同时进行不超过5个活动 |
| 时区统一 | 全服统一服务器时间，区分大区时区 |
| 预告期 | 活动开始前24小时在列表出现 |
| 领取期 | 活动结束后保留3-7天领取期 |
| 配置热更 | 活动参数可热更无需更新包 |
| 灰度发布 | 新活动类型先小范围测试 |
