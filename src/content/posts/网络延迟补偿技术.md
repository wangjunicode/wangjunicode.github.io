## 网络延迟的本质

```
玩家A在北京，服务器在上海：
  A按下射击键 → 消息传到服务器：~30ms
  服务器计算结果 → 消息传回A：~30ms
  总延迟：60ms

玩家B在广州：
  B的延迟：~50ms 往返
  
现实：
  A看到的B的位置，实际上是50ms前的位置
  B射击时，实际瞄准的是50ms前A的位置

问题：
  高延迟下，玩家明明瞄准了却打不中（因为对方已经移走）
  玩家感受："这游戏打不中人，垃圾网络！"
```

## 一、客户端预测（Client-Side Prediction）

### 1.1 预测原理

```
无预测的体验（纯服务端权威）：
  玩家按下 W 键
  ↓ 发送到服务器（30ms）
  服务器计算新位置
  ↓ 返回给客户端（30ms）
  客户端更新位置
  
  总延迟：60ms → 玩家感受到明显的输入延迟

客户端预测的方案：
  玩家按下 W 键
  → 本地立即移动（不等服务器）
  → 同时发送输入给服务器
  
  服务器收到输入 → 计算权威位置 → 返回
  
  客户端收到权威位置：
    如果与本地预测接近 → 正常
    如果偏差较大 → 平滑修正（不是突变，是平滑插值到正确位置）
```

### 1.2 客户端预测实现

```csharp
/// <summary>
/// 客户端预测系统
/// 基于输入历史进行本地模拟，收到服务端校正时回滚重播
/// </summary>
public class ClientPrediction : MonoBehaviour
{
    // 输入历史（用于服务端校正后的回滚重播）
    private readonly Queue<PlayerInput> _inputHistory = new();
    private uint _currentInputSequence = 0;
    
    // 服务端确认的最后一次位置
    private Vector3 _serverConfirmedPosition;
    private uint _serverConfirmedSequence = 0;
    
    [System.Serializable]
    public struct PlayerInput
    {
        public uint Sequence;        // 输入序列号
        public Vector3 Movement;     // 移动方向
        public float Timestamp;      // 本地时间戳
    }
    
    [System.Serializable]
    public struct ServerState
    {
        public uint LastProcessedSequence; // 服务端处理的最后一个输入序号
        public Vector3 AuthoritativePosition; // 服务端权威位置
    }
    
    void Update()
    {
        // 1. 读取输入
        var input = new PlayerInput
        {
            Sequence = ++_currentInputSequence,
            Movement = GetMovementInput(),
            Timestamp = Time.time
        };
        
        // 2. 本地立即应用（不等服务器）
        ApplyInput(input);
        
        // 3. 保存输入历史
        _inputHistory.Enqueue(input);
        
        // 4. 发送到服务器
        SendInputToServer(input);
        
        // 5. 清理过旧的历史（只保留最近2秒）
        while (_inputHistory.Count > 0 && 
               Time.time - _inputHistory.Peek().Timestamp > 2f)
        {
            _inputHistory.Dequeue();
        }
    }
    
    /// <summary>
    /// 收到服务端校正时调用
    /// </summary>
    public void OnServerStateReceived(ServerState serverState)
    {
        // 如果服务端位置与本地预测差异太大，进行回滚重播
        float deviation = Vector3.Distance(
            transform.position, serverState.AuthoritativePosition);
        
        if (deviation < 0.05f)
        {
            // 差异极小，忽略（网络抖动导致的正常误差）
            return;
        }
        
        if (deviation < 0.5f)
        {
            // 小偏差：平滑插值修正（不突变，玩家几乎感知不到）
            StartCoroutine(SmoothCorrect(serverState.AuthoritativePosition, 0.1f));
            return;
        }
        
        // 大偏差：回滚到服务端状态，然后重新播放未确认的输入
        Debug.LogWarning($"[Prediction] Large deviation: {deviation:F2}m, rolling back");
        RollbackAndReplay(serverState);
    }
    
    private void RollbackAndReplay(ServerState serverState)
    {
        // 1. 将位置设置为服务端确认的位置
        transform.position = serverState.AuthoritativePosition;
        
        // 2. 找到服务端未处理的输入（比 LastProcessedSequence 新的）
        var unprocessed = new List<PlayerInput>();
        foreach (var input in _inputHistory)
        {
            if (input.Sequence > serverState.LastProcessedSequence)
                unprocessed.Add(input);
        }
        
        // 3. 重新应用这些输入（相当于重新模拟）
        foreach (var input in unprocessed)
        {
            ApplyInput(input);
        }
        
        Debug.Log($"[Prediction] Rolled back and replayed {unprocessed.Count} inputs");
    }
    
    private Vector3 GetMovementInput()
    {
        return new Vector3(
            Input.GetAxisRaw("Horizontal"),
            0,
            Input.GetAxisRaw("Vertical")
        ).normalized;
    }
    
    private void ApplyInput(PlayerInput input)
    {
        float moveSpeed = 5f;
        Vector3 delta = input.Movement * moveSpeed * Time.fixedDeltaTime;
        transform.position += delta;
    }
    
    private void SendInputToServer(PlayerInput input)
    {
        // 通过网络发送到服务器
        // NetworkManager.Instance.Send(input);
    }
    
    private IEnumerator SmoothCorrect(Vector3 targetPos, float duration)
    {
        Vector3 start = transform.position;
        float elapsed = 0f;
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            transform.position = Vector3.Lerp(start, targetPos, elapsed / duration);
            yield return null;
        }
        transform.position = targetPos;
    }
}
```

## 二、插值（其他玩家的平滑显示）

### 2.1 为什么需要插值

```
问题：
  服务端每秒发送 20 个状态更新（50ms 一个）
  客户端渲染 60fps
  如果直接跳转到新位置：每秒抖动 20 次
  
插值方案：
  在两个已知状态之间平滑插值
  代价：显示的是"过去"的状态（通常延迟 100~150ms）
  收益：其他玩家的移动看起来非常流畅
```

### 2.2 实体插值实现

```csharp
/// <summary>
/// 实体插值：平滑显示其他玩家的移动
/// 核心思路：维护一个状态历史，始终渲染100~150ms前的状态
/// </summary>
public class EntityInterpolation : MonoBehaviour
{
    private const float INTERPOLATION_DELAY = 0.1f; // 100ms 插值延迟
    
    [System.Serializable]
    private struct PositionSample
    {
        public float Timestamp;
        public Vector3 Position;
        public Quaternion Rotation;
    }
    
    private readonly List<PositionSample> _history = new(20);
    
    // 收到服务端状态时调用（在接收网络包时）
    public void AddSample(Vector3 position, Quaternion rotation)
    {
        _history.Add(new PositionSample
        {
            Timestamp = Time.time,
            Position = position,
            Rotation = rotation
        });
        
        // 清理太旧的历史（只保留最近1秒）
        while (_history.Count > 0 && Time.time - _history[0].Timestamp > 1f)
            _history.RemoveAt(0);
    }
    
    void Update()
    {
        // 要显示的目标时间 = 当前时间 - 插值延迟
        float renderTime = Time.time - INTERPOLATION_DELAY;
        
        // 在历史中找到 renderTime 前后的两个状态
        PositionSample? before = null;
        PositionSample? after = null;
        
        for (int i = _history.Count - 1; i >= 0; i--)
        {
            if (_history[i].Timestamp <= renderTime)
            {
                before = _history[i];
                if (i + 1 < _history.Count)
                    after = _history[i + 1];
                break;
            }
        }
        
        if (before == null) return; // 历史不足，等待
        
        if (after == null)
        {
            // 只有过去的状态，使用最新的
            transform.position = before.Value.Position;
            transform.rotation = before.Value.Rotation;
            return;
        }
        
        // 在 before 和 after 之间插值
        float duration = after.Value.Timestamp - before.Value.Timestamp;
        float t = duration > 0 
            ? (renderTime - before.Value.Timestamp) / duration
            : 1f;
        
        t = Mathf.Clamp01(t);
        
        transform.position = Vector3.Lerp(before.Value.Position, after.Value.Position, t);
        transform.rotation = Quaternion.Slerp(before.Value.Rotation, after.Value.Rotation, t);
    }
}
```

## 三、Lag Compensation（延迟补偿）

### 3.1 命中判定的时间回溯

```
问题描述：
  玩家A（延迟 80ms）在 T=1000ms 时射击，瞄准的是屏幕上B的位置
  
  但是！A看到的B的位置是 T=920ms 时的位置（插值延迟 80ms）
  服务器在 T=1080ms 收到射击包时，B已经移动到了 T=1080ms 的位置
  
  服务器按 T=1080ms 的B位置判断 → B已经不在瞄准点上 → Miss
  玩家A感受："我明明打中了！！"

Lag Compensation 解决方案：
  服务器保存过去一段时间内所有玩家的位置历史
  当收到射击包时：
    1. 根据玩家A的延迟，回溯到 A 射击时看到的世界状态
    2. 用那个时间点的B的位置做碰撞检测
    3. 判断是否命中
```

### 3.2 服务端实现（概念代码）

```csharp
/// <summary>
/// 服务端延迟补偿系统
/// </summary>
public class LagCompensationSystem
{
    private const float HISTORY_DURATION = 1f; // 保存1秒历史
    
    // 每个玩家的位置历史
    private Dictionary<int, Queue<PlayerSnapshot>> _playerHistories = new();
    
    [System.Serializable]
    public struct PlayerSnapshot
    {
        public float ServerTimestamp;
        public Vector3 Position;
        public float Radius; // 碰撞体半径
    }
    
    /// <summary>
    /// 每帧保存所有玩家的快照
    /// </summary>
    public void RecordSnapshot(int playerId, Vector3 position, float radius)
    {
        if (!_playerHistories.TryGetValue(playerId, out var history))
        {
            history = new Queue<PlayerSnapshot>(64);
            _playerHistories[playerId] = history;
        }
        
        history.Enqueue(new PlayerSnapshot
        {
            ServerTimestamp = Time.time,
            Position = position,
            Radius = radius
        });
        
        // 清理过旧的快照
        while (history.Count > 0 && 
               Time.time - history.Peek().ServerTimestamp > HISTORY_DURATION)
        {
            history.Dequeue();
        }
    }
    
    /// <summary>
    /// 处理射击请求（带延迟补偿）
    /// </summary>
    public bool ProcessShot(
        int shooterId,          // 射击者 ID
        Vector3 rayOrigin,      // 射线起点
        Vector3 rayDirection,   // 射线方向
        float clientTimestamp,  // 客户端射击时的时间戳
        float clientLatency)    // 客户端延迟
    {
        // 回溯到客户端看到的时间点
        float targetTime = clientTimestamp - clientLatency;
        
        // 对所有其他玩家进行射线检测
        foreach (var (playerId, history) in _playerHistories)
        {
            if (playerId == shooterId) continue;
            
            // 找到 targetTime 时该玩家的位置
            Vector3? historicPosition = GetPositionAtTime(history, targetTime);
            
            if (historicPosition == null) continue;
            
            // 使用历史位置做球体射线检测
            float radius = 0.5f; // 简化：固定半径
            if (RaySphereIntersect(rayOrigin, rayDirection, historicPosition.Value, radius))
            {
                // 命中！
                Debug.Log($"[LagComp] Player {shooterId} hit Player {playerId} (compensated)");
                ApplyDamage(shooterId, playerId, 100);
                return true;
            }
        }
        
        return false; // 未命中
    }
    
    private Vector3? GetPositionAtTime(Queue<PlayerSnapshot> history, float targetTime)
    {
        var samples = history.ToArray(); // 注意：实际中用 List 更高效
        
        for (int i = 0; i < samples.Length - 1; i++)
        {
            if (samples[i].ServerTimestamp <= targetTime && 
                samples[i + 1].ServerTimestamp >= targetTime)
            {
                // 在两个快照之间插值
                float t = (targetTime - samples[i].ServerTimestamp) / 
                          (samples[i + 1].ServerTimestamp - samples[i].ServerTimestamp);
                return Vector3.Lerp(samples[i].Position, samples[i + 1].Position, t);
            }
        }
        
        return null; // 历史不足
    }
    
    private static bool RaySphereIntersect(Vector3 origin, Vector3 dir, Vector3 center, float radius)
    {
        Vector3 oc = origin - center;
        float b = Vector3.Dot(oc, dir);
        float c = Vector3.Dot(oc, oc) - radius * radius;
        return b * b - c >= 0;
    }
    
    private void ApplyDamage(int shooterId, int targetId, int damage)
    {

}
```

## 四、抖动缓冲（Jitter Buffer）

### 4.1 处理网络抖动

```csharp
/// <summary>
/// 抖动缓冲：处理网络包乱序和抖动
/// </summary>
public class JitterBuffer<T>
{
    private const float BUFFER_SIZE = 0.15f; // 150ms 缓冲
    
    private readonly SortedList<float, T> _buffer = new();
    
    public void Insert(float timestamp, T data)
    {
        _buffer.TryAdd(timestamp, data); // 按时间戳有序插入
    }
    
    public bool TryGet(float renderTime, out T data)
    {
        data = default;
        
        // 找到 renderTime 之前的最新数据
        foreach (var kv in _buffer)
        {
            if (kv.Key <= renderTime - BUFFER_SIZE)
            {
                data = kv.Value;
                // 清理更旧的数据
                while (_buffer.Count > 0 && _buffer.Keys[0] < kv.Key)
                    _buffer.RemoveAt(0);
                return true;
            }
        }
        
        return false;
    }
}
```

## 总结

网络延迟补偿技术栈：

| 技术 | 解决什么问题 | 适用对象 |
|------|------------|---------|
| 客户端预测 | 本地玩家的输入延迟感知 | 玩家自己的角色 |
| 实体插值 | 其他玩家的移动抖动 | 其他玩家/NPC |
| Lag Compensation | 命中判定的准确性 | 射击类游戏的服务端 |
| 抖动缓冲 | 网络包乱序/抖动 | 所有网络实体 |

**实践建议**：
- 100ms 以内延迟：客户端预测 + 实体插值即可
- 100~200ms 延迟：需要 Lag Compensation
- 200ms+ 延迟：体验明显下降，需要服务区匹配优化
