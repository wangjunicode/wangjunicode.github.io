# 游戏建造系统设计与实现：网格建造、放置预览与碰撞检测完全指南

## 1. 建造系统概述

建造系统是沙盒游戏、生存游戏、城市建设类游戏的核心玩法。玩家通过放置各种建筑物、道具、地形元素来构建世界。一个高质量的建造系统需要解决以下核心问题：

- **坐标对齐**：建筑物精确吸附到网格
- **放置预览**：实时显示建筑物放置位置和合法性
- **碰撞检测**：防止建筑物重叠或放置在非法区域
- **撤销/重做**：完善的操作历史管理
- **存档序列化**：高效保存和恢复建造数据
- **性能优化**：大量建筑物的渲染与碰撞优化

## 3. 网格管理器

```csharp
public class GridManager : MonoBehaviour
{
    [Header("网格配置")]
    [SerializeField] private float cellSize = 1.0f;
    [SerializeField] private int gridWidth = 100;
    [SerializeField] private int gridHeight = 100;
    [SerializeField] private bool showDebugGrid = false;
    
    // 已占用的格子 (网格坐标 -> 建筑实例ID)
    private Dictionary<Vector2Int, string> occupiedCells = new();
    
    // 网格原点（世界坐标）
    private Vector3 gridOrigin;
    
    // 单例
    public static GridManager Instance { get; private set; }
    
    private void Awake()
    {
        Instance = this;
        gridOrigin = transform.position;
    }
    
    /// <summary>
    /// 世界坐标转网格坐标
    /// </summary>
    public Vector2Int WorldToGrid(Vector3 worldPos)
    {
        Vector3 localPos = worldPos - gridOrigin;
        int x = Mathf.FloorToInt(localPos.x / cellSize);
        int z = Mathf.FloorToInt(localPos.z / cellSize);
        return new Vector2Int(x, z);
    }
    
    /// <summary>
    /// 网格坐标转世界坐标（返回格子中心）
    /// </summary>
    public Vector3 GridToWorld(Vector2Int gridPos)
    {
        return new Vector3(
            gridOrigin.x + (gridPos.x + 0.5f) * cellSize,
            gridOrigin.y,
            gridOrigin.z + (gridPos.y + 0.5f) * cellSize
        );
    }
    
    /// <summary>
    /// 将世界坐标吸附到最近的网格点
    /// </summary>
    public Vector3 SnapToGrid(Vector3 worldPos)
    {
        Vector2Int gridPos = WorldToGrid(worldPos);
        return GridToWorld(gridPos);
    }
    
    /// <summary>
    /// 获取建筑物占用的所有格子坐标
    /// </summary>
    public List<Vector2Int> GetOccupiedCells(Vector2Int origin, Vector2Int size, int rotation)
    {
        var cells = new List<Vector2Int>();
        
        // 根据旋转角度调整尺寸
        Vector2Int actualSize = rotation % 2 == 0 ? size : new Vector2Int(size.y, size.x);
        
        for (int x = 0; x < actualSize.x; x++)
        {
            for (int y = 0; y < actualSize.y; y++)
            {
                cells.Add(new Vector2Int(origin.x + x, origin.y + y));
            }
        }
        
        return cells;
    }
    
    /// <summary>
    /// 检查格子是否可用
    /// </summary>
    public bool AreCellsAvailable(List<Vector2Int> cells)
    {
        foreach (var cell in cells)
        {
            // 越界检查
            if (cell.x < 0 || cell.x >= gridWidth || cell.y < 0 || cell.y >= gridHeight)
                return false;
            
            // 占用检查
            if (occupiedCells.ContainsKey(cell))
                return false;
        }
        return true;
    }
    
    /// <summary>
    /// 占用格子
    /// </summary>
    public void OccupyCells(List<Vector2Int> cells, string instanceId)
    {
        foreach (var cell in cells)
        {
            occupiedCells[cell] = instanceId;
        }
    }
    
    /// <summary>
    /// 释放格子
    /// </summary>
    public void ReleaseCells(string instanceId)
    {
        var cellsToRemove = new List<Vector2Int>();
        foreach (var kvp in occupiedCells)
        {
            if (kvp.Value == instanceId)
                cellsToRemove.Add(kvp.Key);
        }
        foreach (var cell in cellsToRemove)
            occupiedCells.Remove(cell);
    }
    
    /// <summary>
    /// 获取占用格子的建筑实例ID
    /// </summary>
    public string GetBuildingAtCell(Vector2Int cell)
    {
        occupiedCells.TryGetValue(cell, out string id);
        return id;
    }
    
#if UNITY_EDITOR
    private void OnDrawGizmos()
    {
        if (!showDebugGrid) return;
        
        Gizmos.color = new Color(0, 1, 0, 0.2f);
        for (int x = 0; x < gridWidth; x++)
        {
            for (int z = 0; z < gridHeight; z++)
            {
                var cell = new Vector2Int(x, z);
                Vector3 center = GridToWorld(cell);
                center.y = gridOrigin.y + 0.01f;
                
                if (occupiedCells.ContainsKey(cell))
                    Gizmos.color = new Color(1, 0, 0, 0.3f);
                else
                    Gizmos.color = new Color(0, 1, 0, 0.1f);
                
                Gizmos.DrawCube(center, new Vector3(cellSize * 0.95f, 0.01f, cellSize * 0.95f));
            }
        }
    }
#endif
}
```

```csharp
public class PreviewRenderer : MonoBehaviour
{
    [Header("预览材质")]
    [SerializeField] private Material validPlacementMaterial;    // 合法位置（绿色半透明）
    [SerializeField] private Material invalidPlacementMaterial;  // 非法位置（红色半透明）
    
    private GameObject previewObject;
    private Renderer[] previewRenderers;
    private bool isValid;
    
    // 预览材质属性块（避免创建新材质球实例）
    private static readonly int ColorProperty = Shader.PropertyToID("_BaseColor");
    
    /// <summary>
    /// 创建预览对象
    /// </summary>
    public void CreatePreview(GameObject prefab)
    {
        DestroyPreview();
        
        previewObject = Instantiate(prefab);
        previewObject.name = "[BuildPreview]";
        
        // 禁用碰撞体（预览时不参与物理）
        foreach (var col in previewObject.GetComponentsInChildren<Collider>())
            col.enabled = false;
        
        // 禁用脚本组件（避免预览时执行逻辑）
        foreach (var mono in previewObject.GetComponentsInChildren<MonoBehaviour>())
        {
            if (mono != null && !(mono is Renderer) && !(mono is MeshFilter))
                mono.enabled = false;
        }
        
        previewRenderers = previewObject.GetComponentsInChildren<Renderer>();
        ApplyPreviewMaterial(true);
    }
    
    /// <summary>
    /// 更新预览位置和旋转
    /// </summary>
    public void UpdatePreview(Vector3 position, Quaternion rotation, bool valid)
    {
        if (previewObject == null) return;
        
        previewObject.transform.position = position;
        previewObject.transform.rotation = rotation;
        
        if (valid != isValid)
        {
            isValid = valid;
            ApplyPreviewMaterial(valid);
        }
    }
    
    private void ApplyPreviewMaterial(bool valid)
    {
        if (previewRenderers == null) return;
        
        Material mat = valid ? validPlacementMaterial : invalidPlacementMaterial;
        foreach (var r in previewRenderers)
        {
            var mats = new Material[r.sharedMaterials.Length];
            for (int i = 0; i < mats.Length; i++)
                mats[i] = mat;
            r.materials = mats;
        }
    }
    
    /// <summary>
    /// 销毁预览对象
    /// </summary>
    public void DestroyPreview()
    {
        if (previewObject != null)
        {
            Destroy(previewObject);
            previewObject = null;
            previewRenderers = null;
        }
    }
    
    public bool HasPreview => previewObject != null;
}
```

## 5. 碰撞验证器

```csharp
public class CollisionValidator : MonoBehaviour
{
    [Header("检测配置")]
    [SerializeField] private LayerMask buildingLayer;
    [SerializeField] private LayerMask terrainLayer;
    
    /// <summary>
    /// 综合验证放置合法性
    /// </summary>
    public PlacementValidationResult Validate(
        BuildingConfig config, 
        Vector3 position, 
        Quaternion rotation,
        string playerId = null)
    {
        var result = new PlacementValidationResult();
        
        // 1. 射线检测地面
        if (!CheckGroundContact(config, position, rotation, out Vector3 groundPos, out float slope))
        {
            result.isValid = false;
            result.invalidReason = "无法放置在此表面";
            return result;
        }
        
        // 2. 坡度检测
        if (slope > config.maxSlopeAngle)
        {
            result.isValid = false;
            result.invalidReason = $"坡度过大（最大 {config.maxSlopeAngle}°）";
            return result;
        }
        
        // 3. 网格占用检测
        var gridPos = GridManager.Instance.WorldToGrid(position);
        var cells = GridManager.Instance.GetOccupiedCells(gridPos, config.gridSize, GetRotationIndex(rotation));
        if (!GridManager.Instance.AreCellsAvailable(cells))
        {
            result.isValid = false;
            result.invalidReason = "该位置已被占用";
            return result;
        }
        
        // 4. 物理碰撞体积检测（精确检测）
        if (!CheckPhysicsOverlap(config, position, rotation))
        {
            result.isValid = false;
            result.invalidReason = "与其他物体重叠";
            return result;
        }
        
        // 5. 放置规则检测
        if (!CheckPlacementRule(config, position, rotation))
        {
            result.isValid = false;
            result.invalidReason = "不符合放置规则";
            return result;
        }
        
        result.isValid = true;
        result.snappedPosition = groundPos;
        result.occupiedCells = cells;
        return result;
    }
    
    private bool CheckGroundContact(BuildingConfig config, Vector3 position, 
        Quaternion rotation, out Vector3 groundPos, out float slope)
    {
        groundPos = position;
        slope = 0f;
        
        // 从建筑中心向下射线检测
        Ray ray = new Ray(position + Vector3.up * 2f, Vector3.down);
        if (Physics.Raycast(ray, out RaycastHit hit, 10f, terrainLayer))
        {
            groundPos = hit.point;
            slope = Vector3.Angle(hit.normal, Vector3.up);
            return true;
        }
        
        return false;
    }
    
    private bool CheckPhysicsOverlap(BuildingConfig config, Vector3 position, Quaternion rotation)
    {
        // 使用OverlapBox检测建筑物体积内是否有其他碰撞体
        var halfExtents = config.colliderSize * 0.5f - Vector3.one * 0.05f; // 略微缩小避免边缘误判
        var center = position + rotation * config.colliderOffset;
        
        var colliders = Physics.OverlapBox(center, halfExtents, rotation, buildingLayer);
        return colliders.Length == 0;
    }
    
    private bool CheckPlacementRule(BuildingConfig config, Vector3 position, Quaternion rotation)
    {
        switch (config.placementRule)
        {
            case PlacementRule.OnGround:
                return IsOnGround(position);
            case PlacementRule.OnWall:
                return IsOnWall(position, rotation);
            case PlacementRule.OnOtherBuilding:
                return IsOnOtherBuilding(position);
            case PlacementRule.Anywhere:
            case PlacementRule.Floating:
                return true;
            default:
                return true;
        }
    }
    
    private bool IsOnGround(Vector3 pos)
    {
        return Physics.Raycast(pos + Vector3.up * 0.1f, Vector3.down, 0.5f, terrainLayer);
    }
    
    private bool IsOnWall(Vector3 pos, Quaternion rotation)
    {
        Vector3 wallCheckDir = rotation * Vector3.back;
        return Physics.Raycast(pos, wallCheckDir, 0.6f, buildingLayer);
    }
    
    private bool IsOnOtherBuilding(Vector3 pos)
    {
        return Physics.Raycast(pos + Vector3.up * 0.1f, Vector3.down, 1f, buildingLayer);
    }
    
    private int GetRotationIndex(Quaternion rotation)
    {
        float angle = rotation.eulerAngles.y;
        return Mathf.RoundToInt(angle / 90f) % 4;
    }
}

public class PlacementValidationResult
{
    public bool isValid;
    public string invalidReason;
    public Vector3 snappedPosition;
    public List<Vector2Int> occupiedCells;
}
```

## 6. 放置控制器（核心逻辑）

```csharp
public class PlacementController : MonoBehaviour
{
    [Header("组件引用")]
    [SerializeField] private PreviewRenderer previewRenderer;
    [SerializeField] private CollisionValidator validator;
    [SerializeField] private Camera buildCamera;
    
    [Header("建造配置")]
    [SerializeField] private float maxBuildDistance = 10f;
    [SerializeField] private LayerMask raycastLayers;
    
    // 当前选中的建筑配置
    private BuildingConfig currentConfig;
    private int currentRotation;    // 0=0°, 1=90°, 2=180°, 3=270°
    private bool isBuilding;
    
    // 上一次验证结果（用于优化，避免每帧重复验证）
    private PlacementValidationResult lastValidation;
    private Vector3 lastCheckPos;
    private float validationInterval = 0.05f;
    private float lastValidationTime;
    
    // 事件
    public event Action<BuildingConfig, Vector3, Quaternion> OnBuildingPlaced;
    public event Action OnBuildingCanceled;
    
    private void Update()
    {
        if (!isBuilding) return;
        
        HandleInput();
        UpdatePreview();
    }
    
    /// <summary>
    /// 开始建造模式
    /// </summary>
    public void StartBuilding(BuildingConfig config)
    {
        currentConfig = config;
        currentRotation = 0;
        isBuilding = true;
        previewRenderer.CreatePreview(config.prefab);
    }
    
    /// <summary>
    /// 取消建造模式
    /// </summary>
    public void CancelBuilding()
    {
        isBuilding = false;
        currentConfig = null;
        previewRenderer.DestroyPreview();
        OnBuildingCanceled?.Invoke();
    }
    
    private void HandleInput()
    {
        // 旋转
        if (Input.GetKeyDown(KeyCode.R))
        {
            currentRotation = (currentRotation + 1) % 4;
        }
        
        // 确认放置
        if (Input.GetMouseButtonDown(0) && lastValidation != null && lastValidation.isValid)
        {
            ConfirmPlacement();
        }
        
        // 取消
        if (Input.GetMouseButtonDown(1) || Input.GetKeyDown(KeyCode.Escape))
        {
            CancelBuilding();
        }
    }
    
    private void UpdatePreview()
    {
        Ray ray = buildCamera.ScreenPointToRay(Input.mousePosition);
        if (!Physics.Raycast(ray, out RaycastHit hit, maxBuildDistance, raycastLayers))
        {
            previewRenderer.UpdatePreview(Vector3.zero, Quaternion.identity, false);
            return;
        }
        
        // 吸附到网格
        Vector3 snappedPos = GridManager.Instance.SnapToGrid(hit.point);
        snappedPos.y = hit.point.y;
        
        Quaternion rotation = Quaternion.Euler(0, currentRotation * 90f, 0);
        
        // 节流验证（避免每帧都做物理检测）
        if (Vector3.Distance(snappedPos, lastCheckPos) > 0.01f || 
            Time.time - lastValidationTime > validationInterval)
        {
            lastValidation = validator.Validate(currentConfig, snappedPos, rotation);
            lastCheckPos = snappedPos;
            lastValidationTime = Time.time;
        }
        
        bool valid = lastValidation?.isValid ?? false;
        Vector3 displayPos = lastValidation?.snappedPosition ?? snappedPos;
        
        previewRenderer.UpdatePreview(displayPos, rotation, valid);
    }
    
    private void ConfirmPlacement()
    {
        if (lastValidation == null || !lastValidation.isValid) return;
        
        Quaternion rotation = Quaternion.Euler(0, currentRotation * 90f, 0);
        
        // 触发建造事件（由BuildingSystem处理实际生成逻辑）
        OnBuildingPlaced?.Invoke(currentConfig, lastValidation.snappedPosition, rotation);
        
        // 保持建造模式（连续放置）
        lastValidation = null;
    }
}
```

```csharp
// 命令模式实现撤销/重做
public interface IBuildCommand
{
    void Execute();
    void Undo();
    string Description { get; }
}

public class PlaceBuildingCommand : IBuildCommand
{
    private readonly BuildingConfig config;
    private readonly Vector3 position;
    private readonly Quaternion rotation;
    private readonly BuildingSystem buildingSystem;
    private string instanceId;
    
    public string Description => $"放置 {config.displayName}";
    
    public PlaceBuildingCommand(BuildingConfig config, Vector3 pos, Quaternion rot, BuildingSystem system)
    {
        this.config = config;
        this.position = pos;
        this.rotation = rot;
        this.buildingSystem = system;
    }
    
    public void Execute()
    {
        instanceId = buildingSystem.SpawnBuilding(config, position, rotation);
    }
    
    public void Undo()
    {
        if (!string.IsNullOrEmpty(instanceId))
            buildingSystem.DestroyBuilding(instanceId);
    }
}

public class DestroyBuildingCommand : IBuildCommand
{
    private readonly string instanceId;
    private readonly BuildingSystem buildingSystem;
    private BuildingInstanceData savedData;
    
    public string Description => $"拆除建筑";
    
    public DestroyBuildingCommand(string instanceId, BuildingSystem system)
    {
        this.instanceId = instanceId;
        this.buildingSystem = system;
    }
    
    public void Execute()
    {
        savedData = buildingSystem.GetBuildingData(instanceId);
        buildingSystem.DestroyBuilding(instanceId);
    }
    
    public void Undo()
    {
        if (savedData != null)
            buildingSystem.RestoreBuilding(savedData);
    }
}

public class BuildHistory
{
    private readonly Stack<IBuildCommand> undoStack = new();
    private readonly Stack<IBuildCommand> redoStack = new();
    private const int MaxHistorySize = 50;
    
    public void Execute(IBuildCommand command)
    {
        command.Execute();
        undoStack.Push(command);
        redoStack.Clear();
        
        // 限制历史数量
        if (undoStack.Count > MaxHistorySize)
        {
            // Stack不支持底部移除，用List重建
            var list = new List<IBuildCommand>(undoStack);
            list.RemoveAt(list.Count - 1);
            undoStack.Clear();
            list.Reverse();
            foreach (var cmd in list)
                undoStack.Push(cmd);
        }
    }
    
    public void Undo()
    {
        if (undoStack.Count == 0) return;
        var command = undoStack.Pop();
        command.Undo();
        redoStack.Push(command);
    }
    
    public void Redo()
    {
        if (redoStack.Count == 0) return;
        var command = redoStack.Pop();
        command.Execute();
        undoStack.Push(command);
    }
    
    public bool CanUndo => undoStack.Count > 0;
    public bool CanRedo => redoStack.Count > 0;
    public string NextUndoDescription => undoStack.Count > 0 ? undoStack.Peek().Description : null;
}
```

```csharp
[System.Serializable]
public class BuildingSaveData
{
    public string version = "1.0";
    public long saveTimestamp;
    public List<BuildingInstanceData> buildings = new();
}

public class BuildSaveSystem : MonoBehaviour
{
    private const string SaveFileName = "building_save.json";
    
    private string SavePath => Path.Combine(Application.persistentDataPath, SaveFileName);
    
    /// <summary>
    /// 保存所有建筑数据
    /// </summary>
    public void Save(List<BuildingInstanceData> buildings)
    {
        var saveData = new BuildingSaveData
        {
            saveTimestamp = DateTimeOffset.UtcNow.ToUnixTimeSeconds(),
            buildings = buildings
        };
        
        string json = JsonUtility.ToJson(saveData, prettyPrint: true);
        
        // 先写临时文件再替换，防止写入中途崩溃导致存档损坏
        string tempPath = SavePath + ".tmp";
        File.WriteAllText(tempPath, json, System.Text.Encoding.UTF8);
        
        if (File.Exists(SavePath))
            File.Delete(SavePath);
        
        File.Move(tempPath, SavePath);
        Debug.Log($"[BuildSave] 保存成功，共 {buildings.Count} 个建筑");
    }
    
    /// <summary>
    /// 加载建筑数据
    /// </summary>
    public BuildingSaveData Load()
    {
        if (!File.Exists(SavePath))
            return new BuildingSaveData();
        
        try
        {
            string json = File.ReadAllText(SavePath, System.Text.Encoding.UTF8);
            return JsonUtility.FromJson<BuildingSaveData>(json);
        }
        catch (Exception e)
        {
            Debug.LogError($"[BuildSave] 加载失败: {e.Message}");
            return new BuildingSaveData();
        }
    }
    
    /// <summary>
    /// 导出为共享码（分享建筑设计）
    /// </summary>
    public string ExportShareCode(List<BuildingInstanceData> buildings)
    {
        // 使用紧凑格式序列化
        var saveData = new BuildingSaveData { buildings = buildings };
        string json = JsonUtility.ToJson(saveData, prettyPrint: false);
        return System.Convert.ToBase64String(System.Text.Encoding.UTF8.GetBytes(json));
    }
    
    /// <summary>
    /// 从共享码导入
    /// </summary>
    public BuildingSaveData ImportFromShareCode(string code)
    {
        try
        {
            byte[] bytes = System.Convert.FromBase64String(code);
            string json = System.Text.Encoding.UTF8.GetString(bytes);
            return JsonUtility.FromJson<BuildingSaveData>(json);
        }
        catch
        {
            Debug.LogError("[BuildSave] 无效的分享码");
```

```csharp
public class BuildingSystem : MonoBehaviour

            instanceId = instanceId,
            configId = config.buildingId,
            position = position,
            rotation = rotation,
            health = config.durability,
            placedTimestamp = DateTimeOffset.UtcNow.ToUnixTimeSeconds()
        };

        buildingDataMap[instanceId] = data;
        
        return instanceId;
    }

    /// 销毁建筑物
    /// </summary>
    public void DestroyBuilding(string instanceId)
    {
        if (activeBuildings.TryGetValue(instanceId, out var go))
        {
            GridManager.Instance.ReleaseCells(instanceId);
            Destroy(go);
            activeBuildings.Remove(instanceId);
            buildingDataMap.Remove(instanceId);
        }
    }

    /// 恢复建筑物（撤销拆除时使用）
    /// </summary>

    {
        if (!configRegistry.TryGetValue(data.configId, out var config)) return;
        
        var go = Instantiate(config.prefab, data.position, data.rotation, buildingRoot);
        go.name = $"{config.buildingId}_{data.instanceId[..8]}";

        int rotIdx = Mathf.RoundToInt(data.rotation.eulerAngles.y / 90f) % 4;
        var cells = GridManager.Instance.GetOccupiedCells(gridPos, config.gridSize, rotIdx);
        GridManager.Instance.OccupyCells(cells, data.instanceId);
        
        activeBuildings[data.instanceId] = go;
        buildingDataMap[data.instanceId] = data;
    }

        buildingDataMap.TryGetValue(instanceId, out var data);
        return data;
    }

            RestoreBuilding(data);
        }
        Debug.Log($"[BuildingSystem] 加载了 {saveData.buildings.Count} 个建筑");
    }
    
    private void OnApplicationQuit()
    {
        Save();
    }
}
```

### 10.1 大规模建筑的 GPU Instancing 优化

```csharp
public class BatchedBuildingRenderer : MonoBehaviour
{
    [System.Serializable]
    private class InstanceGroup
    {
        public BuildingConfig config;
        public List<Matrix4x4> matrices = new();
        public ComputeBuffer argsBuffer;
        public bool isDirty;
    }
    
    private Dictionary<string, InstanceGroup> groups = new();
    
    /// <summary>
    /// 添加建筑到批量渲染组
    /// </summary>
    public void AddBuilding(BuildingConfig config, Vector3 pos, Quaternion rot, Vector3 scale)
    {
        string key = config.buildingId;
        if (!groups.ContainsKey(key))
            groups[key] = new InstanceGroup { config = config };
        
        groups[key].matrices.Add(Matrix4x4.TRS(pos, rot, scale));
        groups[key].isDirty = true;
    }
    
    private void Update()
    {
        foreach (var kvp in groups)
        {
            var group = kvp.Value;
            if (group.matrices.Count == 0) continue;
            
            // 使用 Graphics.DrawMeshInstanced 批量渲染
            // 注意：单次最多渲染 1023 个实例
            var mesh = group.config.prefab.GetComponentInChildren<MeshFilter>()?.sharedMesh;
            var material = group.config.prefab.GetComponentInChildren<MeshRenderer>()?.sharedMaterial;
            
            if (mesh == null || material == null) continue;
            
            // 分批处理（每批 1023 个）
            const int batchSize = 1023;
            for (int i = 0; i < group.matrices.Count; i += batchSize)
            {
                int count = Mathf.Min(batchSize, group.matrices.Count - i);
                var batch = group.matrices.GetRange(i, count).ToArray();
                Graphics.DrawMeshInstanced(mesh, 0, material, batch);
            }
        }
    }
}
```

```csharp
public class BuildingLODController : MonoBehaviour
{
    [SerializeField] private float lodDistance1 = 30f;  // 切换到 LOD1 的距离

    private Camera mainCamera;
    private Dictionary<string, GameObject> buildings;

    {
        if (mainCamera == null) return;
        
        Vector3 camPos = mainCamera.transform.position;
        
        foreach (var kvp in buildings)
        {
            var go = kvp.Value;
            if (go == null) continue;
            
            float dist = Vector3.Distance(camPos, go.transform.position);
            var lodGroup = go.GetComponent<LODGroup>();
            
            if (dist > cullDistance)
            {
                go.SetActive(false);
            }
            else
            {
                if (!go.activeSelf) go.SetActive(true);
                // LODGroup 会自动处理 LOD 切换
            }
        }
    }
}
```

## 11. 最佳实践总结

### 架构层面
1. **命令模式**：所有建造操作封装为命令对象，天然支持撤销/重做，方便调试回放
2. **数据驱动**：建筑配置全部用 ScriptableObject，策划可在编辑器自由配置，不需要改代码
3. **分层验证**：先做快速的网格检查，再做慢速的物理碰撞检测，节省性能

### 性能层面
4. **验证节流**：放置预览验证不需要每帧执行，设置0.05秒间隔或位置变化阈值触发
5. **GPU Instancing**：同类型建筑使用批量渲染，减少 DrawCall
6. **LOD 剔除**：远距离建筑自动降级渲染或剔除

### 用户体验层面
7. **即时反馈**：合法/非法位置用不同颜色材质实时提示
8. **连续放置**：放置后保持建造模式，提升批量放置效率
9. **吸附提示**：放置时用虚线或网格高亮显示吸附目标格子
10. **存档安全**：先写临时文件再替换，防止存档损坏

### 扩展性层面
11. **插件化放置规则**：通过 `PlacementRule` 枚举方便添加新的放置约束
12. **共享码功能**：支持导出/导入建筑设计，增加社区互动性
13. **多层网格**：支持不同楼层/高度的独立网格，实现多层建筑
