## 一、抽卡系统实现

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class GachaPool
{
    public string PoolId;
    public string Name;
    public GachaItem[] Items;           // 奖池物品
    public int SoftPityStart;           // 软保底起始抽数（如74抽）
    public int HardPityCount;           // 硬保底（如90抽必出SSR）
    public float SoftPityIncrement;     // 每抽概率递增量（如6%）
    public bool HasFatePoint;           // 是否有定轨（50/50）
}

[Serializable]
public class GachaItem
{
    public string ItemId;
    public string Name;
    public GachaRarity Rarity;
    public float BaseRate;              // 基础概率
    public bool IsUpItem;               // 是否是UP物品
}

public enum GachaRarity { R, SR, SSR }

[Serializable]
public class GachaPlayerState
{
    public string PoolId;
    public int TotalPulls;              // 总抽数
    public int PullsSinceLastSSR;       // 距上次SSR的抽数
    public bool FatePointActive;        // 定轨激活（下次SSR必定是UP）
    public List<GachaRecord> History;
}

[Serializable]
public class GachaRecord
{
    public string ItemId;
    public GachaRarity Rarity;
    public long Timestamp;
    public int PullNumber;
}
```

## 二、抽卡逻辑（服务端执行）

```csharp
// 注意：实际抽卡逻辑必须在服务端执行！
// 以下为模拟/预览逻辑

public class GachaSimulator
{
    private System.Random rng;
    
    public GachaSimulator(int seed = -1)
    {
        rng = seed >= 0 ? new System.Random(seed) : new System.Random();
    }

    public GachaItem Pull(GachaPool pool, GachaPlayerState state)
    {
        state.PullsSinceLastSSR++;
        
        // 计算实际SSR概率（加入软保底）
        float ssrRate = GetSSRRate(pool, state.PullsSinceLastSSR);
        
        // 抽卡
        float roll = (float)rng.NextDouble();
        GachaRarity resultRarity;
        
        if (roll < ssrRate || state.PullsSinceLastSSR >= pool.HardPityCount)
        {
            resultRarity = GachaRarity.SSR;
            state.PullsSinceLastSSR = 0;
        }
        else if (roll < ssrRate + 0.051f)  // SR基础概率5.1%
        {
            resultRarity = GachaRarity.SR;
        }
        else
        {
            resultRarity = GachaRarity.R;
        }
        
        // 选择具体物品
        return SelectItem(pool, state, resultRarity);
    }

    float GetSSRRate(GachaPool pool, int pullsSinceLastSSR)
    {
        float baseRate = 0.006f; // 0.6%基础SSR概率
        
        if (pullsSinceLastSSR >= pool.SoftPityStart)
        {
            int softPityPulls = pullsSinceLastSSR - pool.SoftPityStart;
            return baseRate + softPityPulls * pool.SoftPityIncrement;
        }
        
        return baseRate;
    }

    GachaItem SelectItem(GachaPool pool, GachaPlayerState state, GachaRarity rarity)
    {
        var candidates = System.Array.FindAll(pool.Items, i => i.Rarity == rarity);
        if (candidates.Length == 0) return null;
        
        if (rarity == GachaRarity.SSR)
        {
            // 定轨机制：50/50，激活定轨后必出UP
            var upItems = System.Array.FindAll(candidates, i => i.IsUpItem);
            var normalItems = System.Array.FindAll(candidates, i => !i.IsUpItem);
            
            bool getUp = state.FatePointActive || rng.NextDouble() < 0.5;
            
            if (getUp)
            {
                state.FatePointActive = false;
                return RandomFrom(upItems);
            }
            else
            {
                state.FatePointActive = true; // 下次必出UP
                return normalItems.Length > 0 ? RandomFrom(normalItems) : RandomFrom(upItems);
            }
        }
        
        return RandomFrom(candidates);
    }

    T RandomFrom<T>(T[] array)
    {
        if (array.Length == 0) return default;
        return array[rng.Next(array.Length)];
    }

    /// <summary>
    /// 预期抽数统计（蒙特卡罗模拟）
    /// </summary>
    public (float avgPulls, float stdDev) SimulateExpectedPulls(
        GachaPool pool, int iterations = 10000)
    {
        float sum = 0, sumSq = 0;
        
        for (int i = 0; i < iterations; i++)
        {
            var state = new GachaPlayerState { PoolId = pool.PoolId };
            int pulls = 0;
            
            while (true)
            {
                pulls++;
                var item = Pull(pool, state);
                if (item?.Rarity == GachaRarity.SSR && item.IsUpItem) break;
                if (pulls > 500) break; // 防无限循环
            }
            
            sum += pulls;
            sumSq += pulls * pulls;
        }
        
        float avg = sum / iterations;
        float variance = sumSq / iterations - avg * avg;
        return (avg, Mathf.Sqrt(variance));
    }
}
```

## 三、抽卡合规要求

| 合规点 | 要求 |
|--------|------|
| 概率公示 | 必须在抽卡界面显示所有物品概率 |
| 保底公示 | 必须说明保底规则 |
| 未成年保护 | 实名认证，限制消费额度 |
| 随机数公平 | 服务端使用密码学安全随机数 |
| 历史记录 | 玩家可查询抽卡历史 |
| 退款保护 | 某些地区需要提供退款选项 |

**注意：** 多国家/地区对抽卡有不同法规（如比利时/荷兰认定为赌博，中国要求概率公示），发布前必须做合规审查。
