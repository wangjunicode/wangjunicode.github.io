# 游戏任务系统设计：从简单到复杂的完整架构

任务系统是绝大多数游戏的核心功能之一，负责引导玩家、推动剧情、提供目标感。一个好的任务系统需要兼顾**灵活性**（支持各种任务类型）、**可扩展性**（策划随时添加新任务）和**可维护性**（出现问题易于排查）。

---

## 一、任务系统需求分析

### 1.1 任务的分类

| 类型 | 特点 | 示例 |
|------|------|------|
| 主线任务 | 串联剧情，有严格前置依赖 | 第一章：寻找遗迹 |
| 支线任务 | 可选，丰富世界观 | 帮助村民打跑野怪 |
| 日常任务 | 每日/每周刷新 | 每日击杀10个怪物 |
| 成就任务 | 累计达成，永久记录 | 累计走路1000公里 |
| 活动任务 | 限时，与活动系统联动 | 春节活动：收集100个红包 |

### 1.2 任务的生命周期

```
未解锁 → 可接取 → 进行中 → 已完成 → 已领奖
              ↓（放弃）
           已放弃（可重新接取）
```

### 1.3 核心功能点

- **前置条件检查**：等级、任务、物品、时间等
- **进度追踪**：多种条件类型、进度持久化
- **自动完成检测**：事件驱动，无需轮询
- **奖励发放**：经验、金币、物品、解锁内容
- **任务链**：任务完成后自动开启下一个

### 2.2 任务运行时数据

```csharp
// 任务运行时状态（需要持久化）
[Serializable]
public class QuestRuntimeData
{
    public string questId;
    public QuestStatus status;
    public DateTime acceptTime;
    public DateTime completeTime;
    public Dictionary<string, int> conditionProgress;  // 条件ID -> 当前进度
    public float remainingTime;    // 剩余时间（限时任务）
    
    public QuestRuntimeData(string id)
    {
        questId = id;
        status = QuestStatus.InProgress;
        acceptTime = DateTime.Now;
        conditionProgress = new Dictionary<string, int>();
    }
    
    // 检查指定条件是否完成
    public bool IsConditionComplete(QuestCondition condition)
    {
        if (!conditionProgress.TryGetValue(condition.conditionId, out int progress))
            return false;
        return progress >= condition.requiredCount;
    }
    
    // 检查所有条件是否完成
    public bool IsAllObjectivesComplete(QuestConfig config)
    {
        return config.objectives.TrueForAll(obj => IsConditionComplete(obj));
    }
}

public enum QuestStatus
{
    Locked,         // 未解锁
    Available,      // 可接取
    InProgress,     // 进行中
    Completed,      // 已完成（未领奖）
    Rewarded,       // 已领奖
    Failed,         // 已失败
    Abandoned,      // 已放弃
}
```

## 三、任务系统核心实现

### 3.1 QuestManager 主控

```csharp
public class QuestManager : MonoSingleton<QuestManager>
{
    // 所有任务配置（从配置中心加载）
    private Dictionary<string, QuestConfig> _allConfigs = new();
    
    // 运行时数据
    private Dictionary<string, QuestRuntimeData> _runtimeData = new();
    
    // 事件
    public event Action<string> OnQuestAccepted;
    public event Action<string> OnQuestCompleted;
    public event Action<string> OnQuestRewarded;
    public event Action<string, string, int, int> OnQuestProgressUpdated; // questId, conditionId, current, required
    
    protected override void Awake()
    {
        base.Awake();
        LoadConfigs();
        LoadSaveData();
        RegisterEvents();
    }
    
    // ============ 初始化 ============
    
    void LoadConfigs()
    {
        var configs = Resources.LoadAll<QuestConfig>("Quests");
        foreach (var config in configs)
            _allConfigs[config.questId] = config;
        
        Debug.Log($"[QuestManager] 加载了 {_allConfigs.Count} 个任务配置");
    }
    
    void LoadSaveData()
    {
        var saveData = SaveManager.Instance.Load<QuestSaveData>("quest_data");
        if (saveData != null)
            _runtimeData = saveData.runtimeData;
        
        // 初始化未解锁任务的状态
        foreach (var config in _allConfigs.Values)
        {
            if (!_runtimeData.ContainsKey(config.questId))
                _runtimeData[config.questId] = new QuestRuntimeData(config.questId) { status = QuestStatus.Locked };
        }
    }
    
    // 注册游戏事件监听
    void RegisterEvents()
    {
        GameEventSystem.AddListener<EnemyKilledEvent>(OnEnemyKilled);
        GameEventSystem.AddListener<ItemCollectedEvent>(OnItemCollected);
        GameEventSystem.AddListener<NPCTalkedEvent>(OnNPCTalked);
        GameEventSystem.AddListener<LocationReachedEvent>(OnLocationReached);
        GameEventSystem.AddListener<SkillUsedEvent>(OnSkillUsed);
        GameEventSystem.AddListener<PlayerLevelUpEvent>(OnPlayerLevelUp);
    }
    
    // ============ 任务接取 ============
    
    public bool CanAcceptQuest(string questId)
    {
        if (!_allConfigs.TryGetValue(questId, out var config)) return false;
        if (!_runtimeData.TryGetValue(questId, out var data)) return false;
        
        // 检查状态
        if (data.status != QuestStatus.Available && data.status != QuestStatus.Locked)
        {
            if (data.status == QuestStatus.Rewarded && config.isRepeatable)
                return true; // 可重复任务
            return false;
        }
        
        return CheckPrerequisite(config.prerequisite);
    }
    
    bool CheckPrerequisite(QuestPrerequisite pre)
    {
        // 等级检查
        if (PlayerManager.Instance.Level < pre.minLevel) return false;
        
        // 前置任务检查
        foreach (var requiredQuestId in pre.requiredCompletedQuests)
        {
            if (!IsQuestRewarded(requiredQuestId)) return false;
        }
        
        // 物品检查
        foreach (var itemId in pre.requiredItems)
        {
            if (!InventoryManager.Instance.HasItem(itemId)) return false;
        }
        
        // 时间检查（日常任务）
        if (pre.availableDays != null && pre.availableDays.Length > 0)
        {
            var today = DateTime.Now.DayOfWeek;
            if (!System.Array.Contains(pre.availableDays, today)) return false;
        }
        
        return true;
    }
    
    public bool AcceptQuest(string questId)
    {
        if (!CanAcceptQuest(questId)) return false;
        
        var data = _runtimeData[questId];
        var config = _allConfigs[questId];
        
        data.status = QuestStatus.InProgress;
        data.acceptTime = DateTime.Now;
        data.conditionProgress.Clear();
        
        // 初始化进度
        foreach (var obj in config.objectives)
            data.conditionProgress[obj.conditionId] = 0;
        
        // 启动计时器（限时任务）
        if (config.timeLimit > 0)
        {
            data.remainingTime = config.timeLimit;
            StartCoroutine(QuestTimerCoroutine(questId));
        }
        
        // 某些条件可能已经满足（如"拥有物品"）
        RefreshQuestConditions(questId);
        
        SaveQuestData();
        OnQuestAccepted?.Invoke(questId);
        
        Debug.Log($"[QuestManager] 接取任务: {config.questName}");
        return true;
    }
    
    IEnumerator QuestTimerCoroutine(string questId)
    {
        while (_runtimeData.TryGetValue(questId, out var data) && 
               data.status == QuestStatus.InProgress && 
               data.remainingTime > 0)
        {
            yield return new WaitForSeconds(1f);
            data.remainingTime -= 1f;
        }
        
        if (_runtimeData.TryGetValue(questId, out var d) && d.status == QuestStatus.InProgress)
            FailQuest(questId);
    }
    
    // ============ 进度更新 ============
    
    void UpdateQuestProgress(QuestConditionType type, string targetId, int count = 1)
    {
        foreach (var (questId, data) in _runtimeData)
        {
            if (data.status != QuestStatus.InProgress) continue;
            if (!_allConfigs.TryGetValue(questId, out var config)) continue;
            
            bool anyUpdated = false;
            foreach (var obj in config.objectives)
            {
                if (obj.type != type) continue;
                if (!string.IsNullOrEmpty(obj.targetId) && obj.targetId != targetId) continue;
                if (data.IsConditionComplete(obj)) continue;
                
                int oldProgress = data.conditionProgress.GetValueOrDefault(obj.conditionId, 0);
                int newProgress = Mathf.Min(oldProgress + count, obj.requiredCount);
                data.conditionProgress[obj.conditionId] = newProgress;
                
                OnQuestProgressUpdated?.Invoke(questId, obj.conditionId, newProgress, obj.requiredCount);
                anyUpdated = true;
                
                Debug.Log($"[QuestManager] 任务进度: {config.questName} - {obj.description} ({newProgress}/{obj.requiredCount})");
            }
            
            if (anyUpdated && data.IsAllObjectivesComplete(config))
            {
                if (config.isAutoComplete)
                    CompleteQuest(questId);
                else
                    data.status = QuestStatus.Completed; // 等待玩家手动交任务
            }
        }
        
        SaveQuestData();
    }
    
    // 刷新当前所有进行中任务的条件（用于接任务时检查已满足条件）
    void RefreshQuestConditions(string questId)
    {
        if (!_allConfigs.TryGetValue(questId, out var config)) return;
        if (!_runtimeData.TryGetValue(questId, out var data)) return;
        
        foreach (var obj in config.objectives)
        {
            switch (obj.type)
            {
                case QuestConditionType.OwnItem:
                    int itemCount = InventoryManager.Instance.GetItemCount(obj.targetId);
                    data.conditionProgress[obj.conditionId] = Mathf.Min(itemCount, obj.requiredCount);
                    break;
                case QuestConditionType.PlayerLevel:
                    data.conditionProgress[obj.conditionId] = PlayerManager.Instance.Level;
                    break;
                case QuestConditionType.CompleteQuest:
                    data.conditionProgress[obj.conditionId] = IsQuestRewarded(obj.targetId) ? 1 : 0;
                    break;
            }
        }
    }
    
    // ============ 完成与领奖 ============
    
    void CompleteQuest(string questId)
    {
        if (!_runtimeData.TryGetValue(questId, out var data)) return;
        if (data.status != QuestStatus.InProgress && data.status != QuestStatus.Completed) return;
        
        data.status = QuestStatus.Completed;
        data.completeTime = DateTime.Now;
        
        SaveQuestData();
        OnQuestCompleted?.Invoke(questId);
        
        var config = _allConfigs[questId];
        Debug.Log($"[QuestManager] 任务完成: {config.questName}");
    }
    
    public bool ClaimReward(string questId)
    {
        if (!_runtimeData.TryGetValue(questId, out var data)) return false;
        if (data.status != QuestStatus.Completed) return false;
        if (!_allConfigs.TryGetValue(questId, out var config)) return false;
        
        // 发放奖励
        GrantReward(config.reward);
        
        data.status = QuestStatus.Rewarded;
        
        // 解锁后续任务
        foreach (var nextId in config.nextQuestIds)
            UnlockQuest(nextId);
        
        // 处理任务链中的奖励解锁任务
        foreach (var unlockQuestId in config.reward.unlockQuests)
            UnlockQuest(unlockQuestId);
        
        SaveQuestData();
        OnQuestRewarded?.Invoke(questId);
        
        // 检查是否有自动接取的后续任务
        foreach (var nextId in config.nextQuestIds)
        {
            if (_allConfigs.TryGetValue(nextId, out var nextConfig) && nextConfig.isAutoAccept)
                AcceptQuest(nextId);
        }
        
        return true;
    }
    
    void GrantReward(QuestReward reward)
    {
        PlayerManager.Instance.AddExp(reward.exp);
        EconomyManager.Instance.AddGold(reward.gold);
        
        foreach (var item in reward.items)
            InventoryManager.Instance.AddItem(item.itemId, item.count);
        
        foreach (var content in reward.unlockContent)
            ContentUnlockManager.Instance.Unlock(content);
    }
    
    void UnlockQuest(string questId)
    {
        if (!_runtimeData.TryGetValue(questId, out var data)) return;
        if (data.status != QuestStatus.Locked) return;
        
        data.status = QuestStatus.Available;
        
        // 如果配置了自动接取
        if (_allConfigs.TryGetValue(questId, out var config) && config.isAutoAccept)
            AcceptQuest(questId);
    }
    
    void FailQuest(string questId)
    {
        if (!_runtimeData.TryGetValue(questId, out var data)) return;
        data.status = QuestStatus.Failed;
        SaveQuestData();
        
        Debug.Log($"[QuestManager] 任务失败: {questId}");
    }
    
    // ============ 事件监听 ============
    
    void OnEnemyKilled(EnemyKilledEvent e) => UpdateQuestProgress(QuestConditionType.KillEnemy, e.enemyId);
    void OnItemCollected(ItemCollectedEvent e) => UpdateQuestProgress(QuestConditionType.CollectItem, e.itemId, e.count);
    void OnNPCTalked(NPCTalkedEvent e) => UpdateQuestProgress(QuestConditionType.TalkToNPC, e.npcId);
    void OnLocationReached(LocationReachedEvent e) => UpdateQuestProgress(QuestConditionType.ReachLocation, e.locationId);
    void OnSkillUsed(SkillUsedEvent e) => UpdateQuestProgress(QuestConditionType.UseSkill, e.skillId);
    void OnPlayerLevelUp(PlayerLevelUpEvent e)
    {
        // 刷新等级相关条件
        foreach (var questId in GetInProgressQuests())
            RefreshQuestConditions(questId);
    }
    
    // ============ 查询接口 ============
    
    public QuestStatus GetQuestStatus(string questId) =>
        _runtimeData.TryGetValue(questId, out var d) ? d.status : QuestStatus.Locked;
    
    public bool IsQuestRewarded(string questId) =>
        GetQuestStatus(questId) == QuestStatus.Rewarded;
    
    public bool IsQuestInProgress(string questId) =>
        GetQuestStatus(questId) == QuestStatus.InProgress;
    
    public float GetConditionProgress(string questId, string conditionId)
    {
        if (!_runtimeData.TryGetValue(questId, out var data)) return 0;
        if (!_allConfigs.TryGetValue(questId, out var config)) return 0;
        
        var obj = config.objectives.Find(o => o.conditionId == conditionId);
        if (obj == null) return 0;
        
        int current = data.conditionProgress.GetValueOrDefault(conditionId, 0);
        return (float)current / obj.requiredCount;
    }
    
    public List<string> GetInProgressQuests() =>
        _runtimeData.Where(kv => kv.Value.status == QuestStatus.InProgress)
                    .Select(kv => kv.Key).ToList();
    
    public List<string> GetAvailableQuests() =>
        _allConfigs.Keys.Where(id => CanAcceptQuest(id)).ToList();
    
    void SaveQuestData()
    {
        SaveManager.Instance.Save("quest_data", new QuestSaveData { runtimeData = _runtimeData });
    }
}
```

### 4.1 任务追踪器

```csharp
public class QuestTrackerUI : MonoBehaviour
{
    [SerializeField] private Transform container;
    [SerializeField] private QuestTrackerItem itemPrefab;
    [SerializeField] private int maxTrackedQuests = 3;
    
    private List<string> _trackedQuestIds = new();
    private Dictionary<string, QuestTrackerItem> _items = new();
    
    void Start()
    {
        QuestManager.Instance.OnQuestAccepted += OnQuestAccepted;
        QuestManager.Instance.OnQuestCompleted += OnQuestCompleted;
        QuestManager.Instance.OnQuestRewarded += OnQuestRewarded;
        QuestManager.Instance.OnQuestProgressUpdated += OnProgressUpdated;
    }
    
    void OnQuestAccepted(string questId)
    {
        if (_trackedQuestIds.Count < maxTrackedQuests)
            TrackQuest(questId);
    }
    
    void TrackQuest(string questId)
    {
        if (_trackedQuestIds.Contains(questId)) return;
        _trackedQuestIds.Add(questId);
        
        var item = Instantiate(itemPrefab, container);
        item.Setup(questId);
        _items[questId] = item;
    }
    
    void OnProgressUpdated(string questId, string conditionId, int current, int required)
    {
        if (_items.TryGetValue(questId, out var item))
            item.UpdateProgress(conditionId, current, required);
    }
    
    void OnQuestCompleted(string questId)
    {
        if (_items.TryGetValue(questId, out var item))
            item.ShowCompleted();
    }
    
    void OnQuestRewarded(string questId)
    {
        if (_items.TryGetValue(questId, out var item))
        {
            _items.Remove(questId);
            _trackedQuestIds.Remove(questId);
            Destroy(item.gameObject);
        }
    }
}
```

## 五、日常任务与周期刷新

```csharp
public class DailyQuestManager : MonoBehaviour
{
    private const string LAST_REFRESH_KEY = "daily_quest_last_refresh";
    
    void Start()
    {
        CheckAndRefreshDailyQuests();
    }
    
    void CheckAndRefreshDailyQuests()
    {
        string lastRefresh = PlayerPrefs.GetString(LAST_REFRESH_KEY, "");
        string today = DateTime.Now.ToString("yyyy-MM-dd");
        
        if (lastRefresh != today)
        {
            RefreshDailyQuests();
            PlayerPrefs.SetString(LAST_REFRESH_KEY, today);
        }
    }
    
    void RefreshDailyQuests()
    {
        var dailyQuests = QuestManager.Instance.GetQuestsByType(QuestType.Daily);
        foreach (var questId in dailyQuests)
        {
            var status = QuestManager.Instance.GetQuestStatus(questId);
            if (status == QuestStatus.Rewarded || status == QuestStatus.Failed)
            {
                // 重置为可接取状态
                QuestManager.Instance.ResetQuest(questId);
            }
        }
        
        Debug.Log("[DailyQuestManager] 日常任务已刷新");
    }
}
```

```csharp
[CreateAssetMenu(menuName = "Game/Quest/Scripts/OrderedDialogScript")]
public class OrderedDialogQuestScript : QuestScript
{
    public string[] npcDialogOrder; // 必须按顺序对话的NPC
    private int _currentStep = 0;
    
    public override void OnQuestAccepted(string questId)
    {
        _currentStep = 0;
        GameEventSystem.AddListener<NPCTalkedEvent>(OnNPCTalked);
    }
    
    void OnNPCTalked(NPCTalkedEvent e)
    {
        if (_currentStep < npcDialogOrder.Length && 
            e.npcId == npcDialogOrder[_currentStep])
        {
            _currentStep++;
        }
    }
    
    public override bool CheckCustomCondition(string questId)
    {
        return _currentStep >= npcDialogOrder.Length;
    }
    
    public override void OnQuestCompleted(string questId)
    {
        GameEventSystem.RemoveListener<NPCTalkedEvent>(OnNPCTalked);
    }
}
```

## 七、性能与工程实践

### 7.1 事件驱动 vs 轮询

```csharp
// ❌ 错误：每帧轮询检查
void Update()
{
    foreach (var questId in inProgressQuests)
        CheckAllConditions(questId); // 性能杀手！
}

// ✅ 正确：事件驱动
// 只在相关游戏事件发生时才检查对应任务条件
// 大幅减少不必要的计算
```

### 7.2 任务存档优化

```csharp
// 只保存"有变化"的任务状态，减少存档体积
public class QuestSaveData
{
    // 只存储不是默认值（Locked）的任务
    public Dictionary<string, QuestRuntimeData> runtimeData = new();
    
    // 序列化时过滤掉锁定且无进度的任务
    public Dictionary<string, QuestRuntimeData> GetSaveableData()
    {
        return runtimeData.Where(kv => 
            kv.Value.status != QuestStatus.Locked || 
            kv.Value.conditionProgress.Count > 0)
            .ToDictionary(kv => kv.Key, kv => kv.Value);
    }
}
```

## 总结

| 设计要点 | 推荐方案 |
|---------|---------|
| 数据配置 | ScriptableObject + 配置中心 |
| 进度追踪 | 事件驱动，拒绝轮询 |
| 条件扩展 | 枚举 + 策略模式 |
| 复杂逻辑 | QuestScript 附加脚本 |
| 持久化 | 只存变化数据，减少体积 |
| 日常刷新 | 记录上次刷新日期，启动时检查 |

一个良好设计的任务系统，不仅是策划的内容工具，也是连接游戏各个系统的核心枢纽——它依赖战斗、道具、NPC、地图等系统的事件，同时向玩家传递目标与进度，是提升留存率的核心功能。
