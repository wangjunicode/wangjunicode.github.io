# 游戏水体渲染系统深度实现指南

## 一、水体渲染的挑战与分层设计

水是游戏中视觉表现最为复杂的自然元素之一。它的独特之处在于：
- **光学复杂性**：同时需要处理折射、反射、散射、吸收
- **动态形变**：波浪形状随时间变化，表面法线不断更新
- **视角依赖**：菲涅尔效应使水面在不同角度呈现截然不同的外观
- **边界交互**：岸边浪花、物体入水激起水花、泡沫等

### 1.1 水体渲染的质量分级

| 质量级别 | 方案 | DrawCall开销 | 适用平台 |
|---------|------|-------------|---------|
| 低 | 静态贴图 + 简单扰动 | 1 | 超低端移动 |
| 中 | 滚动法线贴图 + Gerstner波 | 1~2 | 主流移动端 |
| 高 | Gerstner多波叠加 + 屏幕空间反射 | 3~5 | PC/主机/高端移动 |
| 超高 | FFT海洋 + 物理次表面散射 | 10+ | PC/主机大作 |

            "Queue" = "Transparent"
            "RenderPipeline" = "UniversalPipeline"

            {
                float4 positionCS   : SV_POSITION;

                float3 normalWS     : TEXCOORD1;
                float3 tangentWS    : TEXCOORD2;

                UNITY_INITIALIZE_VERTEX_OUTPUT_STEREO(OUT);

                VertexPositionInputs posInputs = GetVertexPositionInputs(IN.positionOS.xyz);
                VertexNormalInputs normInputs  = GetVertexNormalInputs(IN.normalOS, IN.tangentOS);

                OUT.positionCS  = posInputs.positionCS;
                OUT.positionWS  = posInputs.positionWS;
                OUT.normalWS    = normInputs.normalWS;
                OUT.tangentWS   = normInputs.tangentWS;
                OUT.bitangentWS = normInputs.bitangentWS;
                OUT.uv          = IN.uv;
                OUT.grabPos     = ComputeGrabScreenPos(OUT.positionCS);
                OUT.screenPos   = ComputeScreenPos(OUT.positionCS);

                return OUT;
            }

            half4 WaterFrag(Varyings IN) : SV_Target
            {
                // ---- 1. 法线采样（双层滚动叠加）----
                float2 uv1 = IN.uv * _NormalMap1_ST.xy + _NormalScrollSpeed1.xy * _Time.y;
                float2 uv2 = IN.uv * _NormalMap2_ST.xy + _NormalScrollSpeed2.xy * _Time.y;
                
                float3 n1 = UnpackNormalScale(SAMPLE_TEXTURE2D(_NormalMap1, sampler_NormalMap1, uv1), _NormalStrength);
                float3 n2 = UnpackNormalScale(SAMPLE_TEXTURE2D(_NormalMap2, sampler_NormalMap2, uv2), _NormalStrength);
                float3 blendedNormal = normalize(n1 + n2);
                
                // 将切线空间法线转换到世界空间
                float3x3 TBN = float3x3(IN.tangentWS, IN.bitangentWS, IN.normalWS);
                float3 worldNormal = normalize(mul(blendedNormal, TBN));

                // ---- 2. 水深度计算（用于颜色渐变和泡沫）----
                float2 screenUV = IN.screenPos.xy / IN.screenPos.w;
                float rawDepth = SampleSceneDepth(screenUV);
                float sceneDepth = LinearEyeDepth(rawDepth, _ZBufferParams);
                float waterDepth = sceneDepth - IN.screenPos.w;
                float depthFade = saturate(waterDepth / _DepthFadeDistance);

                // ---- 3. 颜色混合（浅水→深水）----
                float4 waterColor = lerp(_ShallowColor, _DeepColor, depthFade);

                // ---- 4. 折射（扰动屏幕UV）----
                float2 refractOffset = blendedNormal.xy * _RefractionStrength * (1.0 - depthFade);
                float2 grabUV = (IN.grabPos.xy + refractOffset) / IN.grabPos.w;
                float4 refractColor = SAMPLE_TEXTURE2D(_GrabTexture, sampler_GrabTexture, grabUV);

                // ---- 5. 菲涅尔系数 ----
                float3 viewDir = normalize(GetCameraPositionWS() - IN.positionWS);
                float VdotN = saturate(dot(viewDir, worldNormal));
                float fresnel = _FresnelBias + _FresnelScale * pow(1.0 - VdotN, _FresnelPower);
                fresnel = saturate(fresnel);

                // ---- 6. 高光 ----
                Light mainLight = GetMainLight();
                float3 halfDir = normalize(viewDir + mainLight.direction);
                float HdotN = saturate(dot(halfDir, worldNormal));
                float specular = pow(HdotN, _SpecularPower) * _SpecularIntensity;
                float4 specularColor = _SpecularColor * specular;

                // ---- 7. 岸边泡沫 ----
                float foamMask = 1.0 - saturate(waterDepth / _FoamEdgeWidth);
                float2 foamUV = IN.uv * _FoamTexture_ST.xy + _Time.y * 0.1;
                float foamTex = SAMPLE_TEXTURE2D(_FoamTexture, sampler_FoamTexture, foamUV).r;
                float foam = foamMask * foamTex * _FoamIntensity;

                // ---- 8. 最终合成 ----
                // 折射占比随菲涅尔和深度变化
                float refractionBlend = (1.0 - fresnel) * _WaterClarity;
                float4 finalColor = lerp(refractColor, waterColor, refractionBlend);
                finalColor += specularColor;
                finalColor.rgb = lerp(finalColor.rgb, float3(1, 1, 1), foam);
                finalColor.a = saturate(waterColor.a + foam);

                return finalColor;
            }
            ENDHLSL
        }
    }
}

## 三、Gerstner波浪顶点动画

Gerstner波是游戏中最常用的波浪物理模拟，能产生真实的尖锐浪峰：

### 3.1 Gerstner波数学原理

标准Gerstner波公式（XZ平面上的波浪）：

```
x(u,v,t) = u - Σ( Qi * Ai * Di.x * sin(Wi·Di·P + φi·t) )
z(u,v,t) = v - Σ( Qi * Ai * Di.z * sin(Wi·Di·P + φi·t) )
y(u,v,t) =     Σ( Ai * cos(Wi·Di·P + φi·t) )
```

其中：
- `Ai` = 振幅，`Wi` = 频率，`Qi` = 陡度系数
- `Di` = 波行进方向，`φi` = 相位速度

### 3.2 多波Gerstner实现

```hlsl
// GerstnerWave.hlsl - 包含文件
#ifndef GERSTNER_WAVE_INCLUDED
#define GERSTNER_WAVE_INCLUDED

struct WaveParams
{
    float2 direction;   // 波行进方向（归一化）
    float amplitude;    // 振幅（米）
    float wavelength;   // 波长（米）
    float speed;        // 传播速度（米/秒）
    float steepness;    // 陡度 Q（0=正弦波, 1=最尖锐Gerstner波）
};

// 计算单个Gerstner波对顶点位置和法线的贡献
void GerstnerWave(WaveParams wave, float3 position, float time,
                  inout float3 offset, inout float3 tangent, inout float3 binormal)
{
    float k = 2.0 * 3.14159 / wave.wavelength;       // 波数
    float c = sqrt(9.81 / k);                         // 波速（深水色散关系）
    float d = wave.speed / c;                          // 速度比例
    float2 dir = normalize(wave.direction);
    
    float f = k * (dot(dir, position.xz) - wave.speed * time);
    float a = wave.steepness / k;                     // 陡度转换为振幅
    
    float cosF = cos(f);
    float sinF = sin(f);
    
    // 位置偏移
    offset.x += a * dir.x * cosF;
    offset.z += a * dir.y * cosF;
    offset.y += wave.amplitude * sinF;
    
    // 切线向量（X方向）
    tangent.x -= wave.steepness * dir.x * dir.x * sinF;
    tangent.y += dir.x * wave.amplitude * cosF;
    tangent.z -= wave.steepness * dir.x * dir.y * sinF;
    
    // 副切线向量（Z方向）
    binormal.x -= wave.steepness * dir.x * dir.y * sinF;
    binormal.y += dir.y * wave.amplitude * cosF;
    binormal.z -= wave.steepness * dir.y * dir.y * sinF;
}

// 叠加4个Gerstner波（可根据性能扩展更多）
float3 ComputeGerstnerDisplacement(float3 worldPos, float time,
                                    out float3 worldNormal)
{
    // 预设4个方向和大小不同的波
    WaveParams waves[4];
    waves[0] = (WaveParams){ float2(1, 0),    0.6, 10.0, 1.2, 0.8 };
    waves[1] = (WaveParams){ float2(0.7, 0.3),0.4,  7.0, 1.0, 0.7 };
    waves[2] = (WaveParams){ float2(0.2, 0.9),0.3,  5.0, 0.9, 0.6 };
    waves[3] = (WaveParams){ float2(-0.5,0.8),0.2,  4.0, 0.8, 0.5 };
    
    float3 offset   = float3(0, 0, 0);
    float3 tangent  = float3(1, 0, 0);
    float3 binormal = float3(0, 0, 1);
    
    for (int i = 0; i < 4; i++)
    {
        GerstnerWave(waves[i], worldPos, time, offset, tangent, binormal);
    }
    
    worldNormal = normalize(cross(binormal, tangent));
    return offset;
}

#endif
```

    };

    private static readonly WavePreset StormyPreset = new WavePreset
    {
        amplitude = 2.0f, frequency = 1.5f, speed = 2.5f,
        steepness = 0.9f, normalStrength = 2.0f, foamIntensity = 1.5f
    };

    [System.Serializable]
    struct WavePreset
    {
        public float amplitude, frequency, speed, steepness, normalStrength, foamIntensity;
    }

    void Start()
    {
        var renderer = GetComponent<MeshRenderer>();
        waterMaterial = renderer.material;
    }

    void Update()
    {
        targetIntensity = currentState switch
        {
            SeaState.Calm   => 0.0f,
            SeaState.Choppy => 0.5f,
            SeaState.Stormy => 1.0f,
            _ => 0.5f
        };

        currentIntensity = Mathf.Lerp(currentIntensity, targetIntensity, 
            Time.deltaTime * transitionSpeed);

        ApplyWaveParams(currentIntensity);

    {

        WavePreset from = t < 0.5f ? CalmPreset : ChoppyPreset;
        WavePreset to   = t < 0.5f ? ChoppyPreset : StormyPreset;

        waterMaterial.SetFloat(WaveFrequencyID,  freq);
        waterMaterial.SetFloat(WaveSpeedID,      spd);
        waterMaterial.SetFloat(WaveSteepnessID,  steep);
        waterMaterial.SetFloat(NormalStrengthID, normalS);
        waterMaterial.SetFloat(FoamIntensityID,  foam);
    }

    /// 外部触发海况切换（例如：暴风雨剧情）
    /// </summary>
    public void SetSeaState(SeaState newState, float transSpeed = 1.0f)
    {
        currentState = newState;
        transitionSpeed = transSpeed;
    }
}

## 四、FFT海洋频谱（高级）

### 4.1 FFT海洋原理

FFT（Fast Fourier Transform）海洋基于菲利普斯频谱，将统计上真实的海浪分布转换到空域：

```
H(k, t) = H0(k) * e^(iωt) + H0(-k)* * e^(-iωt)
```

通过 GPU 端的 2D FFT，每帧从频域转换到空域，得到逼真的大洋波浪。

### 4.2 菲利普斯频谱 Compute Shader

```hlsl
// OceanSpectrum.compute
#pragma kernel GenerateSpectrum
#pragma kernel UpdateSpectrum

#define PI 3.14159265358979

// 输出纹理（复数形式：R=实部, G=虚部）
RWTexture2D<float4> _InitialSpectrum;  // H0(k)
RWTexture2D<float4> _CurrentSpectrum; // H(k,t)

// 参数
float _WindSpeed;           // 风速（m/s）
float2 _WindDirection;      // 风向（归一化）
float _Amplitude;           // 总振幅系数
int   _Resolution;          // 频谱分辨率（256或512）
float _PatchSize;           // 海面patch大小（米）
float _Time;

// Gauss随机数生成（Box-Muller变换）
float2 GaussRandom(uint2 id)
{
    // 使用Hash函数生成伪随机数
    float2 uv = float2(id) / float(_Resolution);
    float u1 = frac(sin(dot(uv, float2(127.1, 311.7))) * 43758.5453);
    float u2 = frac(sin(dot(uv, float2(269.5, 183.3))) * 73858.5453);
    
    u1 = max(1e-6, u1);
    float mag = sqrt(-2.0 * log(u1));
    return float2(mag * cos(2.0 * PI * u2), mag * sin(2.0 * PI * u2));
}

// 菲利普斯频谱
float PhillipsSpectrum(float2 k)
{
    float kLen = length(k);
    if (kLen < 0.0001) return 0;
    
    float kLen2 = kLen * kLen;
    float kLen4 = kLen2 * kLen2;
    
    // 最大波长（由风速决定）
    float L = _WindSpeed * _WindSpeed / 9.81;
    float L2 = L * L;
    
    // 抑制非常小的波浪
    float l = L / 1000.0;
    
    float2 kNorm = normalize(k);
    float cosTheta = dot(kNorm, _WindDirection);
    
    // Phillips公式核心
    float Ph = _Amplitude * (exp(-1.0 / (kLen2 * L2))) / kLen4;
    Ph *= cosTheta * cosTheta;           // 方向性：只有顺风方向的波
    Ph *= exp(-kLen2 * l * l);           // 小波浪抑制
    
    return max(0, Ph);
}

// 生成初始频谱 H0(k)
[numthreads(8, 8, 1)]
void GenerateSpectrum(uint3 id : SV_DispatchThreadID)
{
    float2 k = PI * 2.0 * (float2(id.xy) - float(_Resolution) * 0.5) / _PatchSize;
    
    float2 gauss = GaussRandom(id.xy);
    float ph = sqrt(PhillipsSpectrum(k) * 0.5);
    
    _InitialSpectrum[id.xy] = float4(gauss.x * ph, gauss.y * ph, 0, 0);
}

// 每帧更新频谱（加入时间演化）
[numthreads(8, 8, 1)]
void UpdateSpectrum(uint3 id : SV_DispatchThreadID)
{
    float2 k = PI * 2.0 * (float2(id.xy) - float(_Resolution) * 0.5) / _PatchSize;
    float kLen = length(k);
    
    // 深水色散关系: ω = sqrt(g*|k|)
    float omega = sqrt(9.81 * kLen);
    
    float cosOmega = cos(omega * _Time);
    float sinOmega = sin(omega * _Time);
    
    float4 h0  = _InitialSpectrum[id.xy];
    
    // 镜像点 H0(-k)*
    int2 mirrorId = (_Resolution - id.xy) % _Resolution;
    float4 h0Conj = _InitialSpectrum[mirrorId];
    h0Conj.y = -h0Conj.y;  // 取复共轭
    
    // H(k,t) = H0(k)*e^(iωt) + H0(-k)*·e^(-iωt)
    float realPart = h0.x * cosOmega - h0.y * sinOmega
                   + h0Conj.x * cosOmega + h0Conj.y * sinOmega;
    float imagPart = h0.x * sinOmega + h0.y * cosOmega
                   - h0Conj.x * sinOmega + h0Conj.y * cosOmega;
    
    _CurrentSpectrum[id.xy] = float4(realPart, imagPart, 0, 0);
}
```

### 4.3 FFT海洋 C# 控制器

    private RenderTexture spectrumHt;       // 当前时刻频谱
    private RenderTexture displacementMap;  // 位移贴图（输出给Shader）
    private RenderTexture normalMap;        // 法线贴图（输出给Shader）
    
    private int generateKernel;
    private int updateKernel;
    
    // 参数ID缓存
    private static readonly int WindSpeedID     = Shader.PropertyToID("_WindSpeed");
    private static readonly int WindDirID       = Shader.PropertyToID("_WindDirection");
    private static readonly int AmplitudeID     = Shader.PropertyToID("_Amplitude");
    private static readonly int ResolutionID    = Shader.PropertyToID("_Resolution");
    private static readonly int PatchSizeID     = Shader.PropertyToID("_PatchSize");
    private static readonly int TimeID          = Shader.PropertyToID("_Time");
    private static readonly int DisplacementID  = Shader.PropertyToID("_DisplacementMap");
    private static readonly int OceanNormalID   = Shader.PropertyToID("_OceanNormalMap");

    void Start()
    {
        InitializeTextures();
        GetKernels();
        GenerateInitialSpectrum();
    }

    void InitializeTextures()
    {
        var desc = new RenderTextureDescriptor(resolution, resolution, 
            RenderTextureFormat.ARGBFloat, 0)
        {
            enableRandomWrite = true,
            useMipMap = false
        };

        spectrumH0      = CreateRT(desc, "SpectrumH0");
        spectrumHt      = CreateRT(desc, "SpectrumHt");
        displacementMap = CreateRT(desc, "DisplacementMap");
        normalMap       = CreateRT(desc, "NormalMap");
    }

    RenderTexture CreateRT(RenderTextureDescriptor desc, string name)
    {
        var rt = new RenderTexture(desc) { name = name };
        rt.Create();
        return rt;
    }

    void GetKernels()
    {
        generateKernel = oceanSpectrumShader.FindKernel("GenerateSpectrum");
        updateKernel   = oceanSpectrumShader.FindKernel("UpdateSpectrum");
    }

    void GenerateInitialSpectrum()
    {
        oceanSpectrumShader.SetFloat(WindSpeedID,  windSpeed);
        oceanSpectrumShader.SetVector(WindDirID,   windDirection.normalized);
        oceanSpectrumShader.SetFloat(AmplitudeID,  amplitude);
        oceanSpectrumShader.SetInt(ResolutionID,   resolution);
        oceanSpectrumShader.SetFloat(PatchSizeID,  patchSize);
        
        oceanSpectrumShader.SetTexture(generateKernel, "_InitialSpectrum", spectrumH0);
        
        int groups = resolution / 8;
        oceanSpectrumShader.Dispatch(generateKernel, groups, groups, 1);
        
        Debug.Log($"[FFTOcean] 初始频谱生成完成 ({resolution}x{resolution})");
    }

    void Update()
    {
        // 每帧更新时序频谱
        UpdateTimeSpectrum();
        
        // 将位移贴图传给渲染材质
        if (oceanMaterial != null)
        {
            oceanMaterial.SetTexture(DisplacementID, displacementMap);
            oceanMaterial.SetTexture(OceanNormalID, normalMap);
        }
    }

    void UpdateTimeSpectrum()
    {
        oceanSpectrumShader.SetFloat(TimeID, Time.time);
        oceanSpectrumShader.SetTexture(updateKernel, "_InitialSpectrum", spectrumH0);
        oceanSpectrumShader.SetTexture(updateKernel, "_CurrentSpectrum", spectrumHt);
        
        int groups = resolution / 8;
        oceanSpectrumShader.Dispatch(updateKernel, groups, groups, 1);
        
        // TODO: 对 spectrumHt 执行 2D IFFT（逆快速傅里叶变换）
        // 得到 displacementMap（XYZ三个分量分别对应水平位移和垂直位移）
        // Unity项目中可使用 GPU FFT库或自行实现蝶形运算
    }

    void OnDestroy()
    {
        spectrumH0?.Release();
        spectrumHt?.Release();
        displacementMap?.Release();
        normalMap?.Release();
    }

    void OnGUI()
    {
        GUILayout.Label($"[FFT Ocean] 分辨率: {resolution}x{resolution}");
        GUILayout.Label($"  Patch大小: {patchSize}m, 风速: {windSpeed}m/s");
    }
}

## 五、屏幕空间反射（SSR）与平面反射

### 5.1 实时平面反射

```csharp
using UnityEngine;

/// <summary>
/// 水面平面反射摄像机
/// 通过镜像翻转摄像机，将倒影渲染到贴图供水面Shader采样
/// </summary>
[ExecuteInEditMode]
public class PlanarWaterReflection : MonoBehaviour
{
    [Header("反射配置")]
    public LayerMask reflectionLayers = -1;     // 参与反射的层级
    public int textureResolution = 512;          // 反射贴图分辨率
    public float clipPlaneOffset = 0.07f;        // 防止穿帮的偏移
    
    [Header("优化")]
    public bool renderOnce = false;              // 是否每帧渲染（性能敏感时设为false）
    public int renderEveryNFrames = 2;           // 每N帧渲染一次

    private Camera reflectionCamera;
    private RenderTexture reflectionRT;
    private Material waterMaterial;
    private int frameCount = 0;

    private static readonly int ReflectionTexID = Shader.PropertyToID("_ReflectionTexture");

    void Start()
    {
        waterMaterial = GetComponent<Renderer>().material;
        CreateReflectionCamera();
    }

    void CreateReflectionCamera()
    {
        var camGO = new GameObject("__WaterReflectionCamera");
        camGO.hideFlags = HideFlags.HideAndDontSave;
        
        reflectionCamera = camGO.AddComponent<Camera>();
        reflectionCamera.enabled = false;  // 手动控制渲染时机
        
        reflectionRT = new RenderTexture(textureResolution, textureResolution, 16);
        reflectionRT.name = "WaterReflectionRT";
        reflectionCamera.targetTexture = reflectionRT;
        
        if (waterMaterial != null)
            waterMaterial.SetTexture(ReflectionTexID, reflectionRT);
    }

    void OnWillRenderObject()
    {
        frameCount++;
        if (!renderOnce && frameCount % renderEveryNFrames != 0) return;

        Camera mainCam = Camera.current;
        if (mainCam == null) return;

        // 复制主相机设置
        CopyCameraSettings(mainCam, reflectionCamera);
        reflectionCamera.cullingMask = reflectionLayers;

        // 计算反射矩阵
        Vector3 planeNormal = transform.up;
        Vector3 planePoint  = transform.position;
        float planeD = -Vector3.Dot(planeNormal, planePoint) - clipPlaneOffset;
        Vector4 reflectPlane = new Vector4(planeNormal.x, planeNormal.y, planeNormal.z, planeD);

        Matrix4x4 reflectMatrix = CalculateReflectionMatrix(reflectPlane);
        reflectionCamera.worldToCameraMatrix = mainCam.worldToCameraMatrix * reflectMatrix;

        // 斜截面裁剪（避免水面下的物体出现在反射中）
        Vector4 clipPlaneCS = CameraSpacePlane(reflectionCamera, planePoint, planeNormal, 1.0f);
        reflectionCamera.projectionMatrix = mainCam.CalculateObliqueMatrix(clipPlaneCS);

        // 翻转面剔除方向
        GL.invertCulling = true;
        reflectionCamera.Render();
        GL.invertCulling = false;
    }

    static Matrix4x4 CalculateReflectionMatrix(Vector4 plane)
    {
        return new Matrix4x4(
            new Vector4(1 - 2*plane[0]*plane[0], -2*plane[0]*plane[1], -2*plane[0]*plane[2], -2*plane[3]*plane[0]),
            new Vector4(-2*plane[0]*plane[1], 1 - 2*plane[1]*plane[1], -2*plane[1]*plane[2], -2*plane[3]*plane[1]),
            new Vector4(-2*plane[0]*plane[2], -2*plane[1]*plane[2], 1 - 2*plane[2]*plane[2], -2*plane[3]*plane[2]),
            new Vector4(0, 0, 0, 1)
        );
    }

    static Vector4 CameraSpacePlane(Camera cam, Vector3 pos, Vector3 normal, float sideSign)
    {
        Vector3 camSpacePos = cam.worldToCameraMatrix.MultiplyPoint(pos);
        Vector3 camSpaceNorm = cam.worldToCameraMatrix.MultiplyVector(normal).normalized * sideSign;
        return new Vector4(camSpaceNorm.x, camSpaceNorm.y, camSpaceNorm.z, -Vector3.Dot(camSpacePos, camSpaceNorm));
    }

    void CopyCameraSettings(Camera src, Camera dst)
    {
        dst.clearFlags      = src.clearFlags;
        dst.backgroundColor = src.backgroundColor;
        dst.fieldOfView     = src.fieldOfView;
        dst.nearClipPlane   = src.nearClipPlane;
        dst.farClipPlane    = src.farClipPlane;
        dst.aspect          = src.aspect;
    }

    void OnDestroy()
    {
        if (reflectionRT != null) reflectionRT.Release();
        if (reflectionCamera != null) DestroyImmediate(reflectionCamera.gameObject);
    }
}
```

## 六、最佳实践总结

### 性能优化策略

| 优化手段 | 效果 | 实施成本 |
|---------|------|---------|
| 降低法线贴图分辨率（1024→512） | 节省内存30% | 低 |
| 每2帧更新反射 | 省50%反射开销 | 低 |
| 使用Cubemap代替平面反射（开放水域） | 省80%反射开销 | 中 |
| 关闭GrabPass，改用_CameraOpaqueTexture | 省一次全屏拷贝 | 中 |
| 移动端禁用SSR，改用简单菲涅尔 | 节省30% GPU | 低 |

### 技术方案选型建议

```
移动端（中低端）:
  → 滚动双层法线贴图 + Gerstner 2波 + 简化菲涅尔
  → 禁用折射，用颜色深度替代

移动端（高端）:
  → Gerstner 4~6波 + 屏幕空间折射 + Cubemap反射
  
PC/主机:
  → Gerstner 8+波 / FFT海洋 + 平面反射 + SSR
  → 添加海浪粒子特效、体积散射
```

### 常见问题与解决

1. **水面锯齿严重**：法线扰动幅度过大，降低 `_NormalStrength`；或增加 TAA
2. **反射与场景不匹配**：检查 `clipPlaneOffset`，通常 0.05~0.1 合适
3. **移动端帧率骤降**：首先检查 GrabPass 是否每帧触发，改用 `_CameraOpaqueTexture`
4. **岸边泡沫闪烁**：深度缓冲精度不足，增加近裁剪面距离或使用 Logarithmic Depth Buffer
5. **Gerstner波顶点穿透地面**：在顶点Shader中钳制 Y 最小值为地形高度

水体渲染是游戏图形学中综合难度最高的领域之一，建议从基础法线扰动出发，按游戏品质需求逐步升级到Gerstner甚至FFT方案，切勿一开始就追求顶级效果而忽视性能预算。
