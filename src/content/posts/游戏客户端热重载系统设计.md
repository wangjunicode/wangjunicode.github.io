## 一、热重载的应用场景

| 场景 | 热重载类型 | 使用时机 |
|------|-----------|----------|
| 调整美术资源 | Texture/Sprite热替换 | 开发期，美术实时预览 |
| 修改游戏配置 | 配置表热更新 | 开发期+线上调参 |
| 修复逻辑Bug | 代码热更新 | 线上紧急修复 |
| 调整UI布局 | UI热重载 | 开发期+A/B测试 |
| 平衡性调整 | 数值配置热更新 | 线上运营调参 |

## 二、资源热替换系统

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using UnityEngine;
using UnityEngine.AddressableAssets;

/// <summary>
/// 资源热替换管理器
/// 支持在运行时替换 Texture、Audio、Prefab 等资源
/// </summary>
public class AssetHotReloader : MonoBehaviour
{
    private static AssetHotReloader instance;
    public static AssetHotReloader Instance => instance;

    [Header("监控配置")]
    [SerializeField] private bool enableFileWatcher = true;
    [SerializeField] private string watchDirectory = "HotAssets";  // 相对于 StreamingAssets
    [SerializeField] private float pollInterval = 0.5f;            // 轮询间隔（秒）

    // 文件修改时间缓存（用于轮询检测）
    private Dictionary<string, DateTime> fileModifyTimes 
        = new Dictionary<string, DateTime>();
    
    // 已替换资源注册表
    private Dictionary<string, UnityEngine.Object> hotAssets 
        = new Dictionary<string, UnityEngine.Object>();
    
    // 资源引用订阅者（资源更新时通知）
    private Dictionary<string, List<Action<UnityEngine.Object>>> subscribers 
        = new Dictionary<string, List<Action<UnityEngine.Object>>>();

    private float pollTimer;
    private string watchPath;

    void Awake()
    {
        instance = this;
        watchPath = Path.Combine(Application.streamingAssetsPath, watchDirectory);
        
        if (!Directory.Exists(watchPath))
            Directory.CreateDirectory(watchPath);
        
        // 初始化文件时间戳
        ScanDirectory();
    }

    void Update()
    {
        if (!enableFileWatcher) return;
        
        pollTimer += Time.deltaTime;
        if (pollTimer >= pollInterval)
        {
            pollTimer = 0;
            CheckForChanges();
        }
    }

    void ScanDirectory()
    {
        foreach (var file in Directory.GetFiles(watchPath, "*.*", SearchOption.AllDirectories))
        {
            fileModifyTimes[file] = File.GetLastWriteTime(file);
        }
    }

    void CheckForChanges()
    {
        try
        {
            foreach (var file in Directory.GetFiles(watchPath, "*.*", SearchOption.AllDirectories))
            {
                DateTime currentTime = File.GetLastWriteTime(file);
                
                if (fileModifyTimes.TryGetValue(file, out var lastTime))
                {
                    if (currentTime > lastTime)
                    {
                        fileModifyTimes[file] = currentTime;
                        OnFileChanged(file);
                    }
                }
                else
                {
                    fileModifyTimes[file] = currentTime;
                    OnFileChanged(file); // 新文件
                }
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"[HotReload] File watch error: {e.Message}");
        }
    }

    void OnFileChanged(string filePath)
    {
        string ext = Path.GetExtension(filePath).ToLower();
        string assetKey = Path.GetFileNameWithoutExtension(filePath);
        
        Debug.Log($"[HotReload] File changed: {filePath}");
        
        switch (ext)
        {
            case ".png":
            case ".jpg":
            case ".jpeg":
                StartCoroutine(ReloadTexture(filePath, assetKey));
                break;
            case ".wav":
            case ".mp3":
            case ".ogg":
                StartCoroutine(ReloadAudio(filePath, assetKey));
                break;
            case ".json":
                ReloadConfig(filePath, assetKey);
                break;
        }
    }

    System.Collections.IEnumerator ReloadTexture(string filePath, string key)
    {
        yield return new WaitForEndOfFrame();
        
        // 使用 UnityWebRequest 加载本地文件
        using var req = UnityEngine.Networking.UnityWebRequestTexture.GetTexture(
            "file://" + filePath);
        yield return req.SendWebRequest();
        
        if (req.result == UnityEngine.Networking.UnityWebRequest.Result.Success)
        {
            var texture = UnityEngine.Networking.DownloadHandlerTexture.GetContent(req);
            texture.name = key;
            
            hotAssets[key] = texture;
            NotifySubscribers(key, texture);
            
            Debug.Log($"[HotReload] Texture reloaded: {key}");
        }
        else
        {
            Debug.LogError($"[HotReload] Failed to load texture {filePath}: {req.error}");
        }
    }

    System.Collections.IEnumerator ReloadAudio(string filePath, string key)
    {
        yield return new WaitForEndOfFrame();
        
        UnityEngine.Networking.AudioType audioType = 
            Path.GetExtension(filePath).ToLower() switch
            {
                ".wav" => UnityEngine.Networking.AudioType.WAV,
                ".mp3" => UnityEngine.Networking.AudioType.MPEG,
                ".ogg" => UnityEngine.Networking.AudioType.OGGVORBIS,
                _ => UnityEngine.Networking.AudioType.UNKNOWN
            };
        
        using var req = UnityEngine.Networking.UnityWebRequestMultimedia.GetAudioClip(
            "file://" + filePath, audioType);
        yield return req.SendWebRequest();
        
        if (req.result == UnityEngine.Networking.UnityWebRequest.Result.Success)
        {
            var clip = UnityEngine.Networking.DownloadHandlerAudioClip.GetContent(req);
            clip.name = key;
            
            hotAssets[key] = clip;
            NotifySubscribers(key, clip);
        }
    }

    void ReloadConfig(string filePath, string key)
    {
        try
        {
            string json = File.ReadAllText(filePath);
            ConfigHotUpdater.Instance?.OnConfigUpdated(key, json);
            NotifySubscribers(key, null);
        }
        catch (Exception e)
        {
            Debug.LogError($"[HotReload] Config reload failed: {e.Message}");
        }
    }

    /// <summary>
    /// 订阅资源更新
    /// </summary>
    public void Subscribe(string assetKey, Action<UnityEngine.Object> callback)
    {
        if (!subscribers.ContainsKey(assetKey))
            subscribers[assetKey] = new List<Action<UnityEngine.Object>>();
        subscribers[assetKey].Add(callback);
    }

    public void Unsubscribe(string assetKey, Action<UnityEngine.Object> callback)
    {
        if (subscribers.TryGetValue(assetKey, out var list))
            list.Remove(callback);
    }

    void NotifySubscribers(string key, UnityEngine.Object asset)
    {
        if (subscribers.TryGetValue(key, out var list))
        {
            foreach (var cb in new List<Action<UnityEngine.Object>>(list))
                cb?.Invoke(asset);
        }
    }

    public T GetHotAsset<T>(string key) where T : UnityEngine.Object
    {
        return hotAssets.TryGetValue(key, out var asset) ? asset as T : null;
    }
}
```

## 三、配置表热更新

```csharp
/// <summary>
/// 配置表热更新管理器
/// </summary>
public class ConfigHotUpdater : MonoBehaviour
{
    private static ConfigHotUpdater instance;
    public static ConfigHotUpdater Instance => instance;

    private Dictionary<string, string> configCache = new Dictionary<string, string>();
    private Dictionary<string, List<Action<string>>> configSubscribers 
        = new Dictionary<string, List<Action<string>>>();

    void Awake() { instance = this; }

    public void OnConfigUpdated(string configId, string json)
    {
        configCache[configId] = json;
        
        // 通知使用该配置的系统
        if (configSubscribers.TryGetValue(configId, out var subs))
        {
            foreach (var sub in new List<Action<string>>(subs))
                sub?.Invoke(json);
        }
        
        Debug.Log($"[ConfigHot] Config updated: {configId}");
    }

    public void WatchConfig(string configId, Action<string> onUpdate)
    {
        if (!configSubscribers.ContainsKey(configId))
            configSubscribers[configId] = new List<Action<string>>();
        configSubscribers[configId].Add(onUpdate);
        
        // 如果已有缓存，立即回调
        if (configCache.TryGetValue(configId, out var cached))
            onUpdate?.Invoke(cached);
    }
}

/// <summary>
/// 技能配置热更新示例
/// </summary>
public class SkillConfigManager : MonoBehaviour
{
    private Dictionary<string, SkillRuntimeConfig> configs 
        = new Dictionary<string, SkillRuntimeConfig>();

    void Start()
    {
        ConfigHotUpdater.Instance?.WatchConfig("skill_config", OnSkillConfigUpdated);
    }

    void OnSkillConfigUpdated(string json)
    {
        try
        {
            var newConfigs = JsonUtility.FromJson<SkillConfigList>(json);
            
            foreach (var config in newConfigs.Skills)
            {
                configs[config.SkillId] = config;
                Debug.Log($"[SkillConfig] Updated: {config.SkillId}, damage={config.Damage}");
            }
        }
        catch (Exception e)
        {
            Debug.LogError($"[SkillConfig] Parse failed: {e.Message}");
        }
    }

    public SkillRuntimeConfig GetSkill(string id) =>
        configs.TryGetValue(id, out var c) ? c : null;

    [Serializable]
    class SkillConfigList { public SkillRuntimeConfig[] Skills; }
}

[Serializable]
public class SkillRuntimeConfig
{
    public string SkillId;
    public float Damage;
    public float CooldownSeconds;
    public float MpCost;
    public float Range;
}
```

## 四、代码热更新（HybridCLR）

```csharp
/// <summary>
/// HybridCLR 热更新代码加载管理
/// </summary>
public class HybridCLRHotUpdate : MonoBehaviour
{
    [SerializeField] private string hotUpdateDllName = "HotUpdate.dll";
    [SerializeField] private string remoteHotUpdateUrl = "https://cdn.example.com/hotfix/";

    private System.Reflection.Assembly hotUpdateAssembly;

    async System.Threading.Tasks.Task LoadHotUpdateAsync()
    {
        // 1. 检查服务器是否有新版本
        string version = await FetchLatestVersion();
        string localVersion = PlayerPrefs.GetString("HotUpdateVersion", "");
        
        if (version != localVersion)
        {
            // 2. 下载新的 DLL
            await DownloadHotUpdateDll(version);
            PlayerPrefs.SetString("HotUpdateVersion", version);
        }
        
        // 3. 加载 DLL 到内存
        string dllPath = Path.Combine(Application.persistentDataPath, hotUpdateDllName);
        if (File.Exists(dllPath))
        {
            byte[] dllBytes = File.ReadAllBytes(dllPath);
            
            #if ENABLE_HYBRIDCLR
            // HybridCLR 专用加载（支持运行时编译的C#）
            hotUpdateAssembly = System.Reflection.Assembly.Load(dllBytes);
            #else
            // 编辑器中直接使用编译好的程序集
            hotUpdateAssembly = System.AppDomain.CurrentDomain.GetAssemblies()
                .FirstOrDefault(a => a.GetName().Name == "HotUpdate");
            #endif
            
            if (hotUpdateAssembly != null)
            {
                Debug.Log($"[HotUpdate] Assembly loaded: {hotUpdateAssembly.FullName}");
                InitHotUpdateEntry();
            }
        }
    }

    void InitHotUpdateEntry()
    {
        var entryType = hotUpdateAssembly.GetType("HotUpdate.GameEntry");
        if (entryType != null)
        {
            var method = entryType.GetMethod("Initialize");
            method?.Invoke(null, null);
        }
    }

    async System.Threading.Tasks.Task<string> FetchLatestVersion()
    {
        // 从服务器获取最新版本号
        return "1.0.1"; // 简化
    }

    async System.Threading.Tasks.Task DownloadHotUpdateDll(string version)
    {
        string url = $"{remoteHotUpdateUrl}{version}/{hotUpdateDllName}";
        using var req = UnityEngine.Networking.UnityWebRequest.Get(url);
        
        var op = req.SendWebRequest();
        while (!op.isDone) 
            await System.Threading.Tasks.Task.Yield();
        
        if (req.result == UnityEngine.Networking.UnityWebRequest.Result.Success)
        {
            string savePath = Path.Combine(Application.persistentDataPath, hotUpdateDllName);
            File.WriteAllBytes(savePath, req.downloadHandler.data);
        }
    }
}
```

## 五、UI 布局热重载

```csharp
/// <summary>
/// UI 热重载（运行时重新加载 UI Prefab）
/// </summary>
public class UIHotReloader : MonoBehaviour
{
    [SerializeField] private string uiHotReloadPath = "HotUI";

    public void ReloadPanel(string panelName, Transform parent)
    {
        string path = Path.Combine(Application.streamingAssetsPath, uiHotReloadPath, panelName + ".prefab");
        
        // 注意：StreamingAssets 中的 Prefab 需要用 AssetBundle 打包
        // 这里演示 Addressables 重新加载的方式
        StartCoroutine(ReloadFromAddressables(panelName, parent));
    }

    System.Collections.IEnumerator ReloadFromAddressables(string panelName, Transform parent)
    {
        // 先卸载旧版本
        Addressables.Release(panelName);
        
        // 重新加载
        var handle = Addressables.InstantiateAsync(panelName, parent);
        yield return handle;
        
        if (handle.Status == UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded)
        {
            Debug.Log($"[UIHotReload] Panel reloaded: {panelName}");
        }
    }
}
```

## 六、热重载系统设计要点

| 类型 | 开发期 | 线上 | 安全性 |
|------|--------|------|--------|
| 资源替换 | 文件监控，实时生效 | Addressables 版本更新 | 低风险 |
| 配置表 | 文件监控，实时生效 | CDN 下发 + 版本号 | 需要校验格式 |
| 代码热更新 | HybridCLR + 编辑器模拟 | 严格审核 + 签名验证 | 高风险，需要签名 |
| UI布局 | Prefab 重新加载 | AssetBundle/Addressables | 中等风险 |

**关键原则：**
1. 线上代码热更新必须有签名验证和权限控制
2. 配置热更新需要 Schema 校验，防止格式错误导致崩溃
3. 提供热更新失败的降级方案（保留旧版本）
4. 热更新变更需要有完整的灰度发布机制
