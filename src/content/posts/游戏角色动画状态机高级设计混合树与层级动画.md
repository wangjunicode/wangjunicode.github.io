## 一、Blend Tree 混合树设计

### 1.1 移动混合树（2D Freeform Directional）

```
Movement Blend Tree（2D）
├── Idle        (0, 0)
├── Walk_F      (0, 0.5)
├── Walk_B      (0, -0.5)
├── Walk_L      (-0.5, 0)
├── Walk_R      (0.5, 0)
├── Run_F       (0, 1)
├── Run_B       (0, -1)
├── Run_L       (-1, 0)
└── Run_R       (1, 0)

参数：BlendX = InputX * Speed, BlendY = InputZ * Speed
```

```csharp
using UnityEngine;

/// <summary>
/// 角色动画控制器
/// </summary>
[RequireComponent(typeof(Animator))]
public class CharacterAnimationController : MonoBehaviour
{
    [Header("动画参数")]
    private static readonly int BlendX = Animator.StringToHash("BlendX");
    private static readonly int BlendY = Animator.StringToHash("BlendY");
    private static readonly int Speed   = Animator.StringToHash("Speed");
    private static readonly int IsGrounded = Animator.StringToHash("IsGrounded");
    private static readonly int IsAiming = Animator.StringToHash("IsAiming");
    private static readonly int AttackTrigger = Animator.StringToHash("Attack");
    private static readonly int JumpTrigger = Animator.StringToHash("Jump");
    private static readonly int HitTrigger = Animator.StringToHash("Hit");
    private static readonly int DeathTrigger = Animator.StringToHash("Death");
    private static readonly int WeaponType = Animator.StringToHash("WeaponType");

    [Header("平滑配置")]
    [SerializeField] private float blendDampTime = 0.1f; // 混合树平滑时间
    [SerializeField] private float speedDampTime = 0.05f;

    private Animator anim;
    private CharacterController controller;
    
    private Vector2 currentBlend;
    private Vector2 targetBlend;
    private float currentSpeed;

    void Awake()
    {
        anim = GetComponent<Animator>();
        controller = GetComponent<CharacterController>();
    }

    void Update()
    {
        UpdateMovementBlend();
        UpdateGroundState();
    }

    void UpdateMovementBlend()
    {
        // 获取角色速度（相对于角色朝向）
        Vector3 velocity = controller.velocity;
        Vector3 localVelocity = transform.InverseTransformDirection(velocity);
        
        targetBlend = new Vector2(localVelocity.x, localVelocity.z);
        float maxSpeed = 6f; // 最大速度参考值
        targetBlend /= maxSpeed;
        targetBlend = Vector2.ClampMagnitude(targetBlend, 1f);
        
        // 平滑过渡（防止动画突变）
        currentBlend = Vector2.Lerp(currentBlend, targetBlend, 
            blendDampTime > 0 ? Time.deltaTime / blendDampTime : 1f);
        
        anim.SetFloat(BlendX, currentBlend.x);
        anim.SetFloat(BlendY, currentBlend.y);
        
        // 整体速度参数（用于控制是走路还是跑步）
        float speed = velocity.magnitude / maxSpeed;
        currentSpeed = Mathf.Lerp(currentSpeed, speed, 
            speedDampTime > 0 ? Time.deltaTime / speedDampTime : 1f);
        anim.SetFloat(Speed, currentSpeed);
    }

    void UpdateGroundState()
    {
        anim.SetBool(IsGrounded, controller.isGrounded);
    }

    public void TriggerAttack(int comboIndex = 0)
    {
        anim.SetTrigger(AttackTrigger);
        anim.SetInteger("ComboIndex", comboIndex);
    }

    public void TriggerJump()   => anim.SetTrigger(JumpTrigger);
    public void TriggerHit()    => anim.SetTrigger(HitTrigger);
    public void TriggerDeath()  => anim.SetTrigger(DeathTrigger);

    public void SetAiming(bool isAiming) => anim.SetBool(IsAiming, isAiming);

    /// <summary>
    /// 切换武器类型（影响动画层）
    /// 0=Unarmed, 1=Sword, 2=Bow, 3=Staff
    /// </summary>
    public void SetWeaponType(int type)
    {
        anim.SetInteger(WeaponType, type);
        
        // 动态切换动画层权重
        anim.SetLayerWeight(1, type > 0 ? 1f : 0f); // 武器攻击层
        anim.SetLayerWeight(2, type == 2 ? 1f : 0f); // 弓箭瞄准层
    }
}
```

## 二、层级动画（Animation Layer）

```csharp
/// <summary>
/// 动画层管理（高级层叠加）
/// </summary>
public class AnimationLayerManager : MonoBehaviour
{
    private Animator anim;
    
    // 层级定义（与 Animator 中的层顺序对应）
    private const int BaseLayer      = 0;  // 移动、跳跃等基础动画
    private const int UpperBodyLayer = 1;  // 上半身（武器/攻击）
    private const int AdditiveLayer  = 2;  // 叠加层（喘气、受伤倾斜）
    private const int FaceLayer      = 3;  // 面部表情
    
    [SerializeField] private float layerTransitionSpeed = 5f;
    
    // 目标权重
    private float[] targetWeights = new float[4];

    void Awake()
    {
        anim = GetComponent<Animator>();
        // 初始化所有层权重
        for (int i = 1; i < anim.layerCount; i++)
            anim.SetLayerWeight(i, 0f);
    }

    void Update()
    {
        // 平滑过渡层权重
        for (int i = 1; i < Mathf.Min(targetWeights.Length, anim.layerCount); i++)
        {
            float current = anim.GetLayerWeight(i);
            float target = targetWeights[i];
            if (Mathf.Abs(current - target) > 0.001f)
            {
                anim.SetLayerWeight(i, Mathf.MoveTowards(current, target, 
                    layerTransitionSpeed * Time.deltaTime));
            }
        }
    }

    public void SetUpperBodyWeight(float weight)
    {
        targetWeights[UpperBodyLayer] = Mathf.Clamp01(weight);
    }

    public void SetAdditiveLayerWeight(float weight)
    {
        targetWeights[AdditiveLayer] = Mathf.Clamp01(weight);
    }

    public void SetFaceLayerWeight(float weight)
    {
        targetWeights[FaceLayer] = Mathf.Clamp01(weight);
    }
}
```

## 三、动画事件系统

```csharp
/// <summary>
/// 动画事件接收器
/// 在 Animation Clip 中打点调用，通过 UnityMessage 分发
/// </summary>
public class AnimationEventReceiver : MonoBehaviour
{
    [SerializeField] private WeaponController weapon;
    [SerializeField] private FootstepAudioPlayer footstep;
    [SerializeField] private ParticleSystem slashEffect;

    // ============ 攻击事件 ============
    
    /// <summary>
    /// 攻击开始击中判定（动画帧调用）
    /// </summary>
    void OnAttackStart()
    {
        weapon?.EnableHitbox(true);
    }

    /// <summary>
    /// 攻击结束击中判定
    /// </summary>
    void OnAttackEnd()
    {
        weapon?.EnableHitbox(false);
    }

    /// <summary>
    /// 释放抛射物（箭矢/法球）
    /// </summary>
    void OnSpawnProjectile()
    {
        weapon?.SpawnProjectile();
    }

    // ============ 音效事件 ============

    /// <summary>
    /// 脚步声（动画帧调用，左脚/右脚）
    /// </summary>
    void OnFootstepLeft()  => footstep?.PlayLeft();
    void OnFootstepRight() => footstep?.PlayRight();

    // ============ 特效事件 ============

    void OnSlashEffect() => slashEffect?.Play();

    // ============ 自定义字符串事件 ============

    /// <summary>
    /// 通用字符串事件（灵活扩展）
    /// </summary>
    void OnAnimEvent(string eventName)
    {
        switch (eventName)
        {
            case "draw_weapon":
                // 拔剑动作完成
                break;
            case "sheathe_weapon":
                // 收剑动作完成
                break;
            case "skill_cast_start":
                // 技能施法开始
                break;
        }
    }
}
```

## 四、IK 反向运动学

```csharp
/// <summary>
/// 角色 IK 控制器
/// 实现：脚部地面贴合、手持武器看向目标、头部朝向
/// </summary>
[RequireComponent(typeof(Animator))]
public class CharacterIKController : MonoBehaviour
{
    [Header("头部/眼部朝向")]
    [SerializeField] private Transform lookAtTarget;
    [SerializeField, Range(0, 1)] private float headWeight = 0.7f;
    [SerializeField, Range(0, 1)] private float bodyWeight = 0.3f;
    [SerializeField, Range(0, 1)] private float eyesWeight = 0.0f;
    [SerializeField, Range(0, 1)] private float clampWeight = 0.5f;
    
    [Header("手持武器 IK")]
    [SerializeField] private Transform rightHandTarget;
    [SerializeField] private Transform leftHandTarget;
    [SerializeField, Range(0, 1)] private float rightHandWeight = 1f;
    [SerializeField, Range(0, 1)] private float leftHandWeight = 1f;
    
    [Header("脚部地面贴合")]
    [SerializeField] private bool enableFootIK = true;
    [SerializeField] private float footOffset = 0.1f;   // 脚底偏移
    [SerializeField] private LayerMask groundMask;
    
    private Animator anim;
    
    // 脚部 IK 缓存
    private Vector3 rightFootPos, leftFootPos;
    private Quaternion rightFootRot, leftFootRot;
    private float rightFootWeight, leftFootWeight;

    void Awake() => anim = GetComponent<Animator>();

    void Update()
    {
        if (enableFootIK)
            CalculateFootIK();
    }

    void OnAnimatorIK(int layerIndex)
    {
        if (anim == null) return;

        // 头部朝向 LookAt
        if (lookAtTarget != null)
        {
            anim.SetLookAtWeight(headWeight, bodyWeight, headWeight, eyesWeight, clampWeight);
            anim.SetLookAtPosition(lookAtTarget.position);
        }
        
        // 右手 IK（持武器瞄准）
        if (rightHandTarget != null)
        {
            anim.SetIKPositionWeight(AvatarIKGoal.RightHand, rightHandWeight);
            anim.SetIKRotationWeight(AvatarIKGoal.RightHand, rightHandWeight);
            anim.SetIKPosition(AvatarIKGoal.RightHand, rightHandTarget.position);
            anim.SetIKRotation(AvatarIKGoal.RightHand, rightHandTarget.rotation);
        }
        
        // 左手 IK（扶住武器）
        if (leftHandTarget != null)
        {
            anim.SetIKPositionWeight(AvatarIKGoal.LeftHand, leftHandWeight);
            anim.SetIKRotationWeight(AvatarIKGoal.LeftHand, leftHandWeight);
            anim.SetIKPosition(AvatarIKGoal.LeftHand, leftHandTarget.position);
            anim.SetIKRotation(AvatarIKGoal.LeftHand, leftHandTarget.rotation);
        }
        
        // 脚部 IK
        if (enableFootIK)
            ApplyFootIK();
    }

    void CalculateFootIK()
    {
        // 右脚地面贴合
        Vector3 rightFoot = anim.GetBoneTransform(HumanBodyBones.RightFoot).position;
        rightFootWeight = anim.GetFloat("RightFootWeight"); // Animator 中设置的权重参数
        
        if (Physics.Raycast(rightFoot + Vector3.up, Vector3.down, 
            out RaycastHit rightHit, 2f, groundMask))
        {
            rightFootPos = rightHit.point + Vector3.up * footOffset;
            rightFootRot = Quaternion.FromToRotation(Vector3.up, rightHit.normal) * transform.rotation;
        }
        
        // 左脚地面贴合
        Vector3 leftFoot = anim.GetBoneTransform(HumanBodyBones.LeftFoot).position;
        leftFootWeight = anim.GetFloat("LeftFootWeight");
        
        if (Physics.Raycast(leftFoot + Vector3.up, Vector3.down, 
            out RaycastHit leftHit, 2f, groundMask))
        {
            leftFootPos = leftHit.point + Vector3.up * footOffset;
            leftFootRot = Quaternion.FromToRotation(Vector3.up, leftHit.normal) * transform.rotation;
        }
    }

    void ApplyFootIK()
    {
        anim.SetIKPositionWeight(AvatarIKGoal.RightFoot, rightFootWeight);
        anim.SetIKRotationWeight(AvatarIKGoal.RightFoot, rightFootWeight);
        anim.SetIKPosition(AvatarIKGoal.RightFoot, rightFootPos);
        anim.SetIKRotation(AvatarIKGoal.RightFoot, rightFootRot);
        
        anim.SetIKPositionWeight(AvatarIKGoal.LeftFoot, leftFootWeight);
        anim.SetIKRotationWeight(AvatarIKGoal.LeftFoot, leftFootWeight);
        anim.SetIKPosition(AvatarIKGoal.LeftFoot, leftFootPos);
        anim.SetIKRotation(AvatarIKGoal.LeftFoot, leftFootRot);
    }

    /// <summary>
    /// 临时覆盖 LookAt 目标（用于对话等场景）
    /// </summary>
    public void SetLookAt(Transform target, float weight = 1f)
    {
        lookAtTarget = target;
        headWeight = weight;
    }

    public void ClearLookAt()
    {
        headWeight = 0f;
        lookAtTarget = null;
    }
}
```

## 五、动画状态机架构建议

| 层级 | 内容 | Avatar Mask |
|------|------|-------------|
| Base Layer | 移动、跳跃、死亡、Idle | 全身 |
| Upper Body Layer | 攻击、技能、持武器姿势 | 上半身 |
| Additive Layer | 受伤倾斜、喘气起伏 | 全身（叠加模式） |
| Face Layer | 表情变化（说话、喜怒哀乐）| 头部 |
| Hand Layer | 手持物品调整 | 手部 |

**核心原则：**
1. 混合树平滑过渡，消除动画切换时的抖动
2. 动画事件精确控制击中判定时机，而非用延迟
3. IK 在 OnAnimatorIK 中应用，不要在 Update 中设置
4. 所有动画参数使用 `Animator.StringToHash` 缓存哈希值，避免字符串查找开销
