## 一、好友数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 好友关系类型
/// </summary>
public enum FriendshipType
{
    Friend,     // 互为好友
    Following,  // 单向关注（我关注对方）
    Follower,   // 被关注（对方关注我）
    Blocked     // 已屏蔽
}

/// <summary>
/// 玩家在线状态
/// </summary>
public enum OnlineStatus
{
    Offline,    // 离线
    Online,     // 在线（主菜单等）
    InGame,     // 游戏中
    InBattle,   // 战斗中
    InDungeon,  // 副本中
    Away,       // 暂离（超时未操作）
    Invisible   // 隐身（显示为离线）
}

/// <summary>
/// 好友信息
/// </summary>
[Serializable]
public class FriendInfo
{
    public string PlayerId;
    public string Nickname;
    public int Level;
    public string Avatar;
    public FriendshipType Relationship;
    public OnlineStatus Status;
    public long LastOnlineTime;     // 最近在线时间
    public string StatusMessage;    // 个性签名/当前状态文字
    public string CurrentGameMode; // 当前游戏模式（如"5V5排位"）
    
    public bool IsOnline => Status != OnlineStatus.Offline && Status != OnlineStatus.Invisible;
    
    public string GetStatusText()
    {
        return Status switch
        {
            OnlineStatus.Online    => "在线",
            OnlineStatus.InGame    => $"游戏中 · {CurrentGameMode}",
            OnlineStatus.InBattle  => "战斗中",
            OnlineStatus.Away      => "暂离",
            OnlineStatus.Offline   => GetLastOnlineText(),
            _ => "离线"
        };
    }
    
    string GetLastOnlineText()
    {
        var elapsed = DateTimeOffset.UtcNow.ToUnixTimeSeconds() - LastOnlineTime;
        if (elapsed < 3600) return $"{elapsed / 60}分钟前在线";
        if (elapsed < 86400) return $"{elapsed / 3600}小时前在线";
        return $"{elapsed / 86400}天前在线";
    }
}

/// <summary>
/// 好友申请
/// </summary>
[Serializable]
public class FriendRequest
{
    public string RequestId;
    public string FromPlayerId;
    public string FromPlayerName;
    public string FromPlayerAvatar;
    public int FromPlayerLevel;
    public string Message;          // 申请留言
    public long CreateTime;
    public FriendRequestStatus Status;
}

public enum FriendRequestStatus { Pending, Accepted, Rejected, Expired }
```

## 二、好友管理器

```csharp
/// <summary>
/// 好友管理器（客户端）
/// </summary>
public class FriendManager : MonoBehaviour
{
    private static FriendManager instance;
    public static FriendManager Instance => instance;

    private List<FriendInfo> friends = new List<FriendInfo>();
    private List<FriendRequest> pendingRequests = new List<FriendRequest>();
    private int maxFriends = 200;
    
    public IReadOnlyList<FriendInfo> Friends => friends.AsReadOnly();
    
    // 事件
    public event Action<FriendInfo> OnFriendAdded;
    public event Action<string> OnFriendRemoved;           // playerId
    public event Action<FriendInfo> OnFriendStatusChanged;
    public event Action<FriendRequest> OnFriendRequestReceived;
    
    public event Action OnFriendListUpdated;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    public void SetFriendList(List<FriendInfo> friendList, List<FriendRequest> requests)
    {
        friends = new List<FriendInfo>(friendList);
        pendingRequests = new List<FriendRequest>(requests);
        
        // 排序：在线好友在前
        SortFriends();
        
        OnFriendListUpdated?.Invoke();
    }

    void SortFriends()
    {
        friends.Sort((a, b) =>
        {
            // 在线优先
            if (a.IsOnline != b.IsOnline) return b.IsOnline.CompareTo(a.IsOnline);
            // 同状态按名字排序
            return string.Compare(a.Nickname, b.Nickname, StringComparison.CurrentCulture);
        });
    }

    public async System.Threading.Tasks.Task<bool> SendFriendRequest(string targetPlayerId, 
        string message = "")
    {
        if (friends.Count >= maxFriends)
        {
            UIManager.Instance?.ShowMessage($"好友数量已达上限（{maxFriends}）");
            return false;
        }
        
        try
        {
            await NetworkService.SendFriendRequest(targetPlayerId, message);
            UIManager.Instance?.ShowMessage("好友申请已发送");
            return true;
        }
        catch (Exception e)
        {
            UIManager.Instance?.ShowMessage($"发送失败: {e.Message}");
            return false;
        }
    }

    public async System.Threading.Tasks.Task AcceptRequest(string requestId)
    {
        var request = pendingRequests.Find(r => r.RequestId == requestId);
        if (request == null) return;
        
        await NetworkService.AcceptFriendRequest(requestId);
        pendingRequests.Remove(request);
        
        // 本地添加好友
        // 服务器会推送新好友信息
    }

    public async System.Threading.Tasks.Task RejectRequest(string requestId)
    {
        var request = pendingRequests.Find(r => r.RequestId == requestId);
        if (request == null) return;
        
        await NetworkService.RejectFriendRequest(requestId);
        pendingRequests.Remove(request);
    }

    public async System.Threading.Tasks.Task RemoveFriend(string playerId)
    {
        bool confirmed = await UIManager.Instance.ShowConfirmDialog("确定删除好友？");
        if (!confirmed) return;
        
        await NetworkService.RemoveFriend(playerId);
        
        friends.RemoveAll(f => f.PlayerId == playerId);
        OnFriendRemoved?.Invoke(playerId);
        OnFriendListUpdated?.Invoke();
    }

    public async System.Threading.Tasks.Task BlockPlayer(string playerId)
    {
        await NetworkService.BlockPlayer(playerId);
        
        // 屏蔽后从好友列表移除
        var friend = friends.Find(f => f.PlayerId == playerId);
        if (friend != null)
        {
            friends.Remove(friend);
            OnFriendRemoved?.Invoke(playerId);
        }
        
        UIManager.Instance?.ShowMessage("已屏蔽该玩家");
    }

    /// <summary>
    /// 收到好友状态变化推送（服务器实时推送）
    /// </summary>
    public void OnServerStatusUpdate(string playerId, OnlineStatus status, 
        string gameMode = "")
    {
        var friend = friends.Find(f => f.PlayerId == playerId);
        if (friend == null) return;
        
        friend.Status = status;
        friend.CurrentGameMode = gameMode;
        
        if (status == OnlineStatus.Offline)
            friend.LastOnlineTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        
        OnFriendStatusChanged?.Invoke(friend);
        
        // 重新排序（状态变化影响排序）
        SortFriends();
        OnFriendListUpdated?.Invoke();
    }

    public List<FriendInfo> GetOnlineFriends() => 
        friends.FindAll(f => f.IsOnline);

    public List<FriendInfo> SearchFriends(string keyword) =>
        friends.FindAll(f => f.Nickname.Contains(keyword, 
            StringComparison.OrdinalIgnoreCase));
    
    public int GetPendingRequestCount() => pendingRequests.Count;
    public FriendInfo GetFriend(string playerId) => 
        friends.Find(f => f.PlayerId == playerId);
}
```

## 三、好友推荐系统

```csharp
/// <summary>
/// 好友推荐算法（客户端侧展示）
/// </summary>
public class FriendRecommendation
{
    public FriendInfo Player;
    public float Score;           // 推荐分数（越高越相关）
    public string Reason;         // 推荐原因文字

    public static List<FriendRecommendation> FromServerData(
        List<RecommendationData> serverData)
    {
        var result = new List<FriendRecommendation>();
        
        foreach (var data in serverData)
        {
            string reason = data.ReasonType switch
            {
                "mutual_friends" => $"你们有 {data.Value} 个共同好友",
                "same_server"    => "与你同服务器",
                "same_guild"     => "与你同公会",
                "recent_game"    => "最近与你对局过",
                _                => "推荐好友"
            };
            
            result.Add(new FriendRecommendation
            {
                Player = data.PlayerInfo,
                Score = data.Score,
                Reason = reason
            });
        }
        
        return result;
    }
}

[Serializable]
public class RecommendationData
{
    public FriendInfo PlayerInfo;
    public float Score;
    public string ReasonType;
    public int Value;
}
```

## 四、好友系统设计要点

| 要点 | 方案 |
|------|------|
| 在线状态隐私 | 支持隐身模式，对好友显示为离线 |
| 好友上限 | 推荐200人上限，防止数据库查询压力 |
| 状态推送 | 用WebSocket实时推送状态变化 |
| 拉黑处理 | 拉黑后双方互相看不到对方（服务端过滤）|
| 推荐好友 | 通过社交图谱计算，服务端推荐 |
| 离线消息 | 好友不在线时消息存储，上线后推送 |
