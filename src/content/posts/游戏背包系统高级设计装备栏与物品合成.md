## 一、背包系统核心

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

[Serializable]
public class InventorySlot
{
    public ItemData Item;
    public int Count;
    public bool IsEmpty => Item == null || Count <= 0;
    
    public void Clear() { Item = null; Count = 0; }
}

public class InventorySystem : MonoBehaviour
{
    [SerializeField] private int initialSlotCount = 30;
    private List<InventorySlot> slots;
    
    public event Action OnInventoryChanged;
    public event Action<ItemData, int> OnItemAdded;
    public event Action<ItemData, int> OnItemRemoved;

    void Awake()
    {
        slots = new List<InventorySlot>(initialSlotCount);
        for (int i = 0; i < initialSlotCount; i++)
            slots.Add(new InventorySlot());
    }

    public bool HasSpace(ItemData item, int count = 1)
    {
        if (item.IsStackable)
        {
            // 检查是否有同类物品可堆叠
            int remaining = count;
            foreach (var slot in slots)
            {
                if (!slot.IsEmpty && slot.Item == item)
                    remaining -= (item.MaxStack - slot.Count);
                if (remaining <= 0) return true;
            }
        }
        // 检查空格子
        int emptyCount = slots.FindAll(s => s.IsEmpty).Count;
        return emptyCount > 0;
    }

    public bool AddItem(ItemData item, int count = 1)
    {
        if (!HasSpace(item, count)) return false;
        
        int remaining = count;
        
        if (item.IsStackable)
        {
            // 先填满现有堆叠
            foreach (var slot in slots)
            {
                if (!slot.IsEmpty && slot.Item == item && slot.Count < item.MaxStack)
                {
                    int canAdd = Mathf.Min(remaining, item.MaxStack - slot.Count);
                    slot.Count += canAdd;
                    remaining -= canAdd;
                    if (remaining <= 0) break;
                }
            }
        }
        
        // 放入空格子
        while (remaining > 0)
        {
            var emptySlot = slots.Find(s => s.IsEmpty);
            if (emptySlot == null) break;
            
            int toAdd = item.IsStackable ? Mathf.Min(remaining, item.MaxStack) : 1;
            emptySlot.Item = item;
            emptySlot.Count = toAdd;
            remaining -= toAdd;
        }
        
        OnItemAdded?.Invoke(item, count - remaining);
        OnInventoryChanged?.Invoke();
        return remaining == 0;
    }

    public bool RemoveItem(string itemId, int count = 1)
    {
        int remaining = count;
        
        foreach (var slot in slots)
        {
            if (slot.IsEmpty || slot.Item.ItemId != itemId) continue;
            
            int toRemove = Mathf.Min(remaining, slot.Count);
            slot.Count -= toRemove;
            remaining -= toRemove;
            
            if (slot.Count <= 0) slot.Clear();
            if (remaining <= 0) break;
        }
        
        if (remaining < count)
        {
            OnItemRemoved?.Invoke(GetItemById(itemId), count - remaining);
            OnInventoryChanged?.Invoke();
        }
        
        return remaining == 0;
    }

    public bool HasItem(string itemId, int count = 1)
    {
        int total = 0;
        foreach (var slot in slots)
        {
            if (!slot.IsEmpty && slot.Item.ItemId == itemId)
                total += slot.Count;
        }
        return total >= count;
    }

    public int GetItemCount(string itemId)
    {
        int total = 0;
        foreach (var slot in slots)
            if (!slot.IsEmpty && slot.Item.ItemId == itemId) total += slot.Count;
        return total;
    }

    public void ExpandSlots(int additionalSlots)
    {
        for (int i = 0; i < additionalSlots; i++)
            slots.Add(new InventorySlot());
        OnInventoryChanged?.Invoke();
    }

    public List<InventorySlot> GetSlots() => slots;
    
    ItemData GetItemById(string id)
    {
        foreach (var slot in slots)
            if (!slot.IsEmpty && slot.Item.ItemId == id) return slot.Item;
        return null;
    }
}
```

## 二、物品合成系统

```csharp
[CreateAssetMenu(menuName = "Game/Crafting Recipe")]
public class CraftingRecipe : ScriptableObject
{
    [Serializable] public class Ingredient { public ItemData Item; public int Count; }
    
    public string RecipeId;
    public Ingredient[] Ingredients;
    public ItemData ResultItem;
    public int ResultCount = 1;
    public float CraftingTime = 0;  // 秒，0=即时
}

public class CraftingSystem : MonoBehaviour
{
    [SerializeField] private CraftingRecipe[] allRecipes;
    [SerializeField] private InventorySystem inventory;

    public bool CanCraft(CraftingRecipe recipe)
    {
        foreach (var ing in recipe.Ingredients)
            if (!inventory.HasItem(ing.Item.ItemId, ing.Count)) return false;
        return true;
    }

    public bool Craft(CraftingRecipe recipe)
    {
        if (!CanCraft(recipe)) return false;
        if (!inventory.HasSpace(recipe.ResultItem, recipe.ResultCount)) return false;
        
        // 消耗材料
        foreach (var ing in recipe.Ingredients)
            inventory.RemoveItem(ing.Item.ItemId, ing.Count);
        
        // 添加产品
        inventory.AddItem(recipe.ResultItem, recipe.ResultCount);
        
        Debug.Log($"[Craft] 合成成功: {recipe.ResultItem.DisplayName} x{recipe.ResultCount}");
        return true;
    }

    public List<CraftingRecipe> GetAvailableRecipes()
        => new List<CraftingRecipe>(allRecipes).FindAll(r => CanCraft(r));
}
```
