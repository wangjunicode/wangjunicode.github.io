## 一、游戏时间管理器

```csharp
using System;
using UnityEngine;

/// <summary>
/// 游戏时间管理器（独立于 Unity 渲染时间）
/// </summary>
public class GameTimeManager : MonoBehaviour
{
    private static GameTimeManager instance;
    public static GameTimeManager Instance => instance;

    [Header("时间配置")]
    [SerializeField] private float realSecondsPerGameDay = 1800f;   // 现实30分钟 = 游戏1天
    [SerializeField] private int gameStartHour = 6;                  // 游戏初始时间（6:00 AM）
    
    // 游戏时间缩放（不影响 Unity 物理/动画）
    private float gameTimeScale = 1f;
    
    // 当前游戏内时间（以游戏内秒计算）
    private float currentGameTime;
    
    // 游戏内 "秒" 对应现实时间
    private float gameSecondInReal => realSecondsPerGameDay / (24f * 60f * 60f);
    
    // 事件
    public event Action<int> OnHourChanged;   // 游戏内每过一小时
    public event Action OnDawnBegins;          // 游戏内黎明
    public event Action OnNightBegins;         // 游戏内夜晚
    
    private int lastHour = -1;

    void Awake()
    {
        instance = this;
        currentGameTime = gameStartHour * 3600f; // 初始时间（秒）
        LoadSavedTime();
    }

    void Update()
    {
        // 游戏内时间推进（受游戏时间缩放影响）
        currentGameTime += Time.deltaTime * gameTimeScale / gameSecondInReal;
        
        // 处理一天循环（86400秒 = 一天）
        if (currentGameTime >= 86400f)
            currentGameTime -= 86400f;
        
        // 触发小时变化事件
        int currentHour = GetGameHour();
        if (currentHour != lastHour)
        {
            lastHour = currentHour;
            OnHourChanged?.Invoke(currentHour);
            
            if (currentHour == 6) OnDawnBegins?.Invoke();
            if (currentHour == 20) OnNightBegins?.Invoke();
        }
    }

    // ============ 时间查询接口 ============

    public float GameTimeScale
    {
        get => gameTimeScale;
        set => gameTimeScale = Mathf.Max(0, value);
    }

    public int GetGameHour() => (int)(currentGameTime / 3600f);
    public int GetGameMinute() => (int)(currentGameTime % 3600f / 60f);
    public float GetDayProgress() => currentGameTime / 86400f;  // 0=午夜, 0.25=6:00, 0.5=正午
    
    public bool IsNight() => GetGameHour() >= 20 || GetGameHour() < 6;
    public bool IsDaytime() => !IsNight();
    public bool IsDawn() => GetGameHour() >= 6 && GetGameHour() < 9;
    public bool IsDusk() => GetGameHour() >= 17 && GetGameHour() < 20;
    
    /// <summary>
    /// 获取太阳角度（用于天空盒/光照）
    /// 0 = 日出, 90 = 正午, 180 = 日落, 270 = 午夜
    /// </summary>
    public float GetSunAngle()
    {
        float hour = currentGameTime / 3600f;
        return (hour / 24f) * 360f;
    }

    public string GetTimeDisplay()
    {
        int h = GetGameHour();
        int m = GetGameMinute();
        return $"{h:00}:{m:00}";
    }

    void LoadSavedTime()
    {
        if (PlayerPrefs.HasKey("game_time"))
            currentGameTime = PlayerPrefs.GetFloat("game_time");
    }

    void OnApplicationPause(bool paused)
    {
        if (!paused)
        {
            // App 恢复时计算离线时间
            CalculateOfflineTime();
        }
        else
        {
            // App 进入后台时保存时间戳
            PlayerPrefs.SetString("pause_timestamp", 
                DateTimeOffset.UtcNow.ToUnixTimeSeconds().ToString());
            PlayerPrefs.SetFloat("game_time", currentGameTime);
        }
    }

    void CalculateOfflineTime()
    {
        if (!long.TryParse(PlayerPrefs.GetString("pause_timestamp", "0"), out long pauseTime))
            return;
        
        long currentTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        long offlineSeconds = currentTime - pauseTime;
        
        if (offlineSeconds <= 0 || offlineSeconds > 7 * 24 * 3600) return;
        
        // 推进游戏内时间
        float gameTimeAdvance = offlineSeconds / gameSecondInReal;
        currentGameTime = (currentGameTime + gameTimeAdvance) % 86400f;
        
        // 通知离线收益系统
        OfflineRewardManager.Instance?.CalculateRewards(offlineSeconds);
    }
}
```

## 二、子弹时间效果

```csharp
/// <summary>
/// 子弹时间效果（局部时间缩放）
/// 可以对特定角色减速，而其他物体正常运行
/// </summary>
public class BulletTimeController : MonoBehaviour
{
    [Header("子弹时间配置")]
    [SerializeField] private float bulletTimeScale = 0.1f;   // 慢放倍率
    [SerializeField] private float enterDuration = 0.1f;     // 进入过渡时间
    [SerializeField] private float exitDuration = 0.3f;      // 退出过渡时间
    [SerializeField] private float maxDuration = 5f;         // 最大持续时间（秒，现实时间）
    
    [Header("音效配置")]
    [SerializeField] private AudioClip enterSound;
    [SerializeField] private AudioClip exitSound;
    
    private bool isInBulletTime;
    private float currentDuration;
    private Coroutine bulletTimeCoroutine;

    public bool IsInBulletTime => isInBulletTime;

    public void EnterBulletTime()
    {
        if (isInBulletTime) return;
        
        if (bulletTimeCoroutine != null)
            StopCoroutine(bulletTimeCoroutine);
        
        bulletTimeCoroutine = StartCoroutine(BulletTimeRoutine());
    }

    public void ExitBulletTime()
    {
        if (!isInBulletTime) return;
        isInBulletTime = false;
        
        if (bulletTimeCoroutine != null)
        {
            StopCoroutine(bulletTimeCoroutine);
            bulletTimeCoroutine = null;
        }
        
        StartCoroutine(RestoreTimeScale());
    }

    System.Collections.IEnumerator BulletTimeRoutine()
    {
        isInBulletTime = true;
        currentDuration = 0f;
        
        if (enterSound != null)
        {
            // 音效也需要调整（子弹时间中音效应该变慢/变低沉）
            AudioManager.Instance?.PlaySFX(enterSound);
        }
        
        // 进入：缓慢降速
        float timer = 0;
        while (timer < enterDuration)
        {
            timer += Time.unscaledDeltaTime;
            float t = timer / enterDuration;
            Time.timeScale = Mathf.Lerp(1f, bulletTimeScale, t);
            Time.fixedDeltaTime = Time.timeScale * 0.02f; // 保持物理稳定
            yield return null;
        }
        
        Time.timeScale = bulletTimeScale;
        
        // 持续
        while (isInBulletTime && currentDuration < maxDuration)
        {
            currentDuration += Time.unscaledDeltaTime;
            yield return null;
        }
        
        // 退出
        isInBulletTime = false;
        yield return RestoreTimeScale();
    }

    System.Collections.IEnumerator RestoreTimeScale()
    {
        if (exitSound != null)
            AudioManager.Instance?.PlaySFX(exitSound);
        
        float startScale = Time.timeScale;
        float timer = 0;
        
        while (timer < exitDuration)
        {
            timer += Time.unscaledDeltaTime;
            float t = timer / exitDuration;
            Time.timeScale = Mathf.Lerp(startScale, 1f, t);
            Time.fixedDeltaTime = Time.timeScale * 0.02f;
            yield return null;
        }
        
        Time.timeScale = 1f;
        Time.fixedDeltaTime = 0.02f;
    }

    void OnDisable()
    {
        // 确保时间缩放恢复
        Time.timeScale = 1f;
        Time.fixedDeltaTime = 0.02f;
    }
}
```

## 三、离线收益计算器

```csharp
/// <summary>
/// 离线收益管理器（玩家离线期间的资源积累）
/// </summary>
public class OfflineRewardManager : MonoBehaviour
{
    private static OfflineRewardManager instance;
    public static OfflineRewardManager Instance => instance;

    [Header("离线收益配置")]
    [SerializeField] private float maxOfflineHours = 12f;     // 最大离线收益时长（小时）
    [SerializeField] private float goldPerMinute = 10f;       // 每分钟离线金币
    [SerializeField] private float expPerMinute = 5f;         // 每分钟离线经验
    
    void Awake() { instance = this; }

    /// <summary>
    /// 计算离线收益
    /// </summary>
    public OfflineRewards CalculateRewards(long offlineSeconds)
    {
        float offlineMinutes = Mathf.Min(
            offlineSeconds / 60f, 
            maxOfflineHours * 60f);
        
        if (offlineMinutes < 1f) return null; // 离线不到1分钟不计算
        
        var rewards = new OfflineRewards
        {
            OfflineMinutes = (int)offlineMinutes,
            Gold = (long)(goldPerMinute * offlineMinutes),
            Exp = (long)(expPerMinute * offlineMinutes)
        };
        
        Debug.Log($"[OfflineReward] 离线 {offlineMinutes:F0} 分钟，" +
                  $"获得 {rewards.Gold} 金币 / {rewards.Exp} 经验");
        
        // 展示离线收益弹窗
        ShowOfflineRewardDialog(rewards);
        
        return rewards;
    }

    void ShowOfflineRewardDialog(OfflineRewards rewards)
    {
        UIManager.Instance?.ShowOfflineRewardDialog(rewards);
    }
}

public class OfflineRewards
{
    public int OfflineMinutes;
    public long Gold;
    public long Exp;
    public List<string> BonusItems = new List<string>(); // 额外掉落物品
}
```

## 四、定时事件系统

```csharp
/// <summary>
/// 游戏内定时事件（Boss刷新/商店刷新/活动开始）
/// </summary>
public class GameTimedEventManager : MonoBehaviour
{
    [System.Serializable]
    public class TimedEvent
    {
        public string EventId;
        public string DisplayName;
        public int TriggerHour;       // 游戏内几点触发（0-23）
        public bool Repeating = true; // 每天重复
        public System.Action OnTrigger;
        public bool HasTriggeredToday;
    }

    [SerializeField] private List<TimedEvent> events = new List<TimedEvent>();

    void OnEnable()
    {
        GameTimeManager.Instance.OnHourChanged += OnGameHourChanged;
    }

    void OnDisable()
    {
        if (GameTimeManager.Instance != null)
            GameTimeManager.Instance.OnHourChanged -= OnGameHourChanged;
    }

    void OnGameHourChanged(int hour)
    {
        foreach (var evt in events)
        {
            if (evt.TriggerHour == hour && !evt.HasTriggeredToday)
            {
                evt.HasTriggeredToday = true;
                evt.OnTrigger?.Invoke();
                Debug.Log($"[TimedEvent] Triggered: {evt.DisplayName}");
            }
        }
    }

    // 在游戏内新的一天开始时重置
    public void OnNewGameDay()
    {
        foreach (var evt in events)
            if (evt.Repeating)
                evt.HasTriggeredToday = false;
    }
}
```

## 五、时间系统设计要点

| 场景 | 时间处理方案 |
|------|------------|
| UI动画/非游戏暂停 | Time.unscaledDeltaTime |
| 物理/游戏逻辑 | Time.deltaTime（受timeScale影响）|
| 子弹时间 | 修改Time.timeScale，同步fixedDeltaTime |
| 离线收益 | Unix时间戳差值计算 |
| 游戏内时钟 | 独立计算，不依赖Unity时间 |
| 跨时区服务器时间 | UTC时间统一，客户端本地显示 |
