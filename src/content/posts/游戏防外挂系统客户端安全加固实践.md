## 一、内存数值加密

```csharp
/// <summary>
/// 加密整数（防止 Cheat Engine 直接搜索内存数值）
/// 使用 XOR 混淆，实际数值 = 存储值 XOR 密钥
/// </summary>
public struct EncryptedInt
{
    private int _encrypted;
    private int _key;

    public static EncryptedInt Create(int value)
    {
        var e = new EncryptedInt();
        e._key = UnityEngine.Random.Range(int.MinValue, int.MaxValue);
        e._encrypted = value ^ e._key;
        return e;
    }

    public int Value
    {
        get => _encrypted ^ _key;
        set => _encrypted = value ^ _key;
    }

    public static implicit operator int(EncryptedInt e) => e.Value;
    public static EncryptedInt operator +(EncryptedInt a, int b)
    {
        return Create(a.Value + b);
    }

    public override string ToString() => Value.ToString();
}

/// <summary>
/// 加密浮点数
/// </summary>
public struct EncryptedFloat
{
    private float _encrypted;
    private int _key;

    public static EncryptedFloat Create(float value)
    {
        var e = new EncryptedFloat();
        e._key = UnityEngine.Random.Range(int.MinValue, int.MaxValue);
        
        // 将 float 和 key 的字节进行 XOR
        var bytes = BitConverter.GetBytes(value);
        var keyBytes = BitConverter.GetBytes(e._key);
        for (int i = 0; i < bytes.Length; i++)
            bytes[i] ^= keyBytes[i % keyBytes.Length];
        
        e._encrypted = BitConverter.ToSingle(bytes, 0);
        return e;
    }

    public float Value
    {
        get
        {
            var bytes = BitConverter.GetBytes(_encrypted);
            var keyBytes = BitConverter.GetBytes(_key);
            for (int i = 0; i < bytes.Length; i++)
                bytes[i] ^= keyBytes[i % keyBytes.Length];
            return BitConverter.ToSingle(bytes, 0);
        }
        set
        {
            var bytes = BitConverter.GetBytes(value);
            var keyBytes = BitConverter.GetBytes(_key);
            for (int i = 0; i < bytes.Length; i++)
                bytes[i] ^= keyBytes[i % keyBytes.Length];
            _encrypted = BitConverter.ToSingle(bytes, 0);
        }
    }

    public static implicit operator float(EncryptedFloat e) => e.Value;
}

/// <summary>
/// 使用示例
/// </summary>
public class PlayerStatsSecure : MonoBehaviour
{
    // 替代普通变量，外挂扫内存只能看到加密后的乱码
    private EncryptedInt _gold = EncryptedInt.Create(1000);
    private EncryptedInt _level = EncryptedInt.Create(1);
    private EncryptedFloat _hp = EncryptedFloat.Create(100f);

    public int Gold
    {
        get => _gold.Value;
        set => _gold.Value = value;
    }
    
    public float HP
    {
        get => _hp.Value;
        set => _hp.Value = value;
    }
}
```

## 二、速度外挂检测

```csharp
/// <summary>
/// 速度外挂检测（检测 Time.timeScale 是否被篡改）
/// </summary>
public class SpeedHackDetector : MonoBehaviour
{
    [SerializeField] private float checkInterval = 1f;       // 检测间隔
    [SerializeField] private float allowedDeviation = 0.2f;  // 允许误差（20%）
    [SerializeField] private int violationThreshold = 3;      // 违规次数阈值
    
    private float lastSystemTime;
    private float lastGameTime;
    private int violationCount;

    void OnEnable()
    {
        lastSystemTime = GetSystemTime();
        lastGameTime = Time.realtimeSinceStartup;
        InvokeRepeating(nameof(CheckTimeScale), checkInterval, checkInterval);
    }

    void OnDisable()
    {
        CancelInvoke(nameof(CheckTimeScale));
    }

    void CheckTimeScale()
    {
        float currentSystemTime = GetSystemTime();
        float currentGameTime = Time.realtimeSinceStartup;
        
        float systemDelta = currentSystemTime - lastSystemTime;
        float gameDelta = currentGameTime - lastGameTime;
        
        lastSystemTime = currentSystemTime;
        lastGameTime = currentGameTime;
        
        if (systemDelta <= 0 || gameDelta <= 0) return;
        
        // 比较系统时间和游戏时间的比率
        float ratio = gameDelta / systemDelta;
        
        // 正常比率应接近1.0，速度外挂会使游戏时间走更快
        if (ratio > 1f + allowedDeviation || ratio < 1f - allowedDeviation)
        {
            violationCount++;
            Debug.LogWarning($"[Security] Speed hack detected! Ratio: {ratio:F2}, Violations: {violationCount}");
            
            if (violationCount >= violationThreshold)
            {
                OnSpeedHackConfirmed();
            }
        }
        else
        {
            // 逐渐减少违规计数
            violationCount = Mathf.Max(0, violationCount - 1);
        }
    }

    float GetSystemTime()
    {
        // 使用系统时间（不受游戏timeScale影响）
        return (float)(DateTime.UtcNow - DateTime.UnixEpoch).TotalSeconds;
    }

    void OnSpeedHackConfirmed()
    {
        Debug.LogError("[Security] Speed hack confirmed! Taking action...");
        
        // 策略1：立即通知服务端
        // NetworkService.ReportCheat(CheatType.SpeedHack);
        
        // 策略2：软封锁（悄悄限制游戏体验）
        // 比如让怪物血量无限，或者装作正常但记录账号
        
        // 策略3：强制退出
        // Application.Quit();
    }
}
```

## 三、请求签名防重放

```csharp
/// <summary>
/// API 请求签名（防止重放攻击）
/// </summary>
public class RequestSigner
{
    private static readonly string SECRET_KEY = "your_app_secret_key_32chars";

    /// <summary>
    /// 生成请求签名
    /// </summary>
    public static string Sign(Dictionary<string, string> parameters)
    {
        // 1. 添加时间戳（防止重放：5分钟内有效）
        parameters["timestamp"] = DateTimeOffset.UtcNow.ToUnixTimeSeconds().ToString();
        
        // 2. 添加随机Nonce（确保相同参数也会产生不同签名）
        parameters["nonce"] = Guid.NewGuid().ToString("N")[..16];
        
        // 3. 参数按Key排序
        var sortedKeys = new List<string>(parameters.Keys);
        sortedKeys.Sort(StringComparer.Ordinal);
        
        // 4. 拼接字符串
        var sb = new System.Text.StringBuilder();
        foreach (var key in sortedKeys)
        {
            if (key != "sign") // 排除sign字段本身
            {
                sb.Append(key);
                sb.Append('=');
                sb.Append(parameters[key]);
                sb.Append('&');
            }
        }
        sb.Append("key=");
        sb.Append(SECRET_KEY);
        
        // 5. HMAC-SHA256 签名
        using var hmac = new System.Security.Cryptography.HMACSHA256(
            System.Text.Encoding.UTF8.GetBytes(SECRET_KEY));
        byte[] hash = hmac.ComputeHash(
            System.Text.Encoding.UTF8.GetBytes(sb.ToString()));
        
        return BitConverter.ToString(hash).Replace("-", "").ToLower();
    }

    /// <summary>
    /// 发送签名请求
    /// </summary>
    public static async System.Threading.Tasks.Task<string> SignedPost(
        string url, Dictionary<string, string> data)
    {
        string sign = Sign(data);
        data["sign"] = sign;
        
        var form = new UnityEngine.Networking.WWWForm();
        foreach (var kv in data)
            form.AddField(kv.Key, kv.Value);
        
        using var req = UnityEngine.Networking.UnityWebRequest.Post(url, form);
        var op = req.SendWebRequest();
        while (!op.isDone)
            await System.Threading.Tasks.Task.Yield();
        
        if (req.result != UnityEngine.Networking.UnityWebRequest.Result.Success)
        {
            Debug.LogError($"[Security] Request failed: {req.error}");
            return null;
        }
        
        return req.downloadHandler.text;
    }
}
```

## 五、安全分层防护

| 层级 | 防护手段 | 效果 |
|------|----------|------|
| 客户端 | 内存加密/混淆 | 增加破解成本 |
| 传输层 | HTTPS + 请求签名 | 防抓包/重放 |
| 服务端 | 数值合法性验证 | 核心防线 |
| 运营层 | 异常行为检测 | 封号处置 |

**核心原则：客户端无法100%防护，服务端才是防外挂的核心。客户端安全的目的是提高作弊门槛，而非完全杜绝。**
