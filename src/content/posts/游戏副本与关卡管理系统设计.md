## 一、副本系统的核心职责

副本（Dungeon/Instance）系统的主要功能：

1. **场景隔离**：每个队伍独立一份副本实例，互不干扰
2. **生命周期管理**：创建、进行中、完成、失败、超时清理
3. **难度分级**：普通/英雄/史诗/神话等难度配置
4. **进度追踪**：Boss击败、机关激活、收集任务进度
5. **奖励结算**：通关后根据表现发放奖励

## 三、副本实例管理器

```csharp
/// <summary>
/// 副本实例数据（运行时状态）
/// </summary>
public class DungeonInstance
{
    public string InstanceId;           // 唯一实例ID
    public DungeonConfig Config;
    public DifficultyLevel Difficulty;
    public DungeonState State;
    
    public List<string> PlayerIds;      // 参与玩家ID
    public float StartTime;
    public float ElapsedTime;
    
    // 目标进度
    public Dictionary<string, int> ObjectiveProgress 
        = new Dictionary<string, int>();
    
    // 检查点（用于断线重连）
    public string LastCheckpoint;
    public Vector3 CheckpointPosition;
    
    public bool IsCompleted => State == DungeonState.Completed;
    public bool IsFailed => State == DungeonState.Failed;
    public float RemainingTime => Config.HasTimeLimit ? 
        Config.TimeLimitSeconds - ElapsedTime : float.MaxValue;
}

public enum DungeonState
{
    Preparing,      // 准备中（加载场景）
    InProgress,     // 进行中
    BossEncounter,  // Boss战中（可能有特殊规则）
    Completed,      // 已通关
    Failed,         // 失败
    Timeout,        // 超时
    Abandoned       // 玩家主动放弃
}

/// <summary>
/// 副本管理器
/// </summary>
public class DungeonManager : MonoBehaviour
{
    private static DungeonManager instance;
    public static DungeonManager Instance => instance;

    [SerializeField] private DungeonConfig[] allDungeons;

    private DungeonInstance currentInstance;
    private Dictionary<string, DungeonConfig> dungeonIndex 
        = new Dictionary<string, DungeonConfig>();

    // 副本事件
    public event Action<DungeonInstance> OnDungeonStarted;
    public event Action<DungeonInstance> OnDungeonCompleted;
    public event Action<DungeonInstance> OnDungeonFailed;
    public event Action<string, int, int> OnObjectiveUpdated; // id, current, required
    public event Action<float> OnTimerUpdated;

    void Awake()
    {
        instance = this;
        foreach (var d in allDungeons)
            dungeonIndex[d.DungeonId] = d;
    }

    void Update()
    {
        if (currentInstance == null || 
            currentInstance.State != DungeonState.InProgress) return;
        
        currentInstance.ElapsedTime += Time.deltaTime;
        
        // 计时器更新
        OnTimerUpdated?.Invoke(currentInstance.RemainingTime);
        
        // 超时检测
        if (currentInstance.Config.HasTimeLimit && 
            currentInstance.ElapsedTime >= currentInstance.Config.TimeLimitSeconds)
        {
            TimeoutDungeon();
        }
    }

    /// <summary>
    /// 进入副本
    /// </summary>
    public async System.Threading.Tasks.Task<bool> EnterDungeon(
        string dungeonId, DifficultyLevel difficulty, List<string> playerIds)
    {
        if (!dungeonIndex.TryGetValue(dungeonId, out var config))
        {
            Debug.LogError($"Dungeon not found: {dungeonId}");
            return false;
        }
        
        // 检查进入条件
        if (!CheckEntryRequirements(config, playerIds))
            return false;
        
        // 消耗进入门票
        ConsumeEntryItems(config);
        
        // 创建副本实例
        currentInstance = new DungeonInstance
        {
            InstanceId = Guid.NewGuid().ToString(),
            Config = config,
            Difficulty = difficulty,
            PlayerIds = playerIds,
            StartTime = Time.realtimeSinceStartup,
            State = DungeonState.Preparing
        };
        
        // 初始化目标进度
        foreach (var obj in config.Objectives)
            currentInstance.ObjectiveProgress[obj.ObjectiveId] = 0;
        
        // 加载副本场景
        bool loaded = await LoadDungeonScene(config.SceneAddress);
        if (!loaded)
        {
            currentInstance.State = DungeonState.Failed;
            return false;
        }
        
        currentInstance.State = DungeonState.InProgress;
        OnDungeonStarted?.Invoke(currentInstance);
        
        // 初始化敌人（应用难度乘数）
        InitializeEnemies(difficulty, config);
        
        return true;
    }

    async System.Threading.Tasks.Task<bool> LoadDungeonScene(string sceneAddress)
    {
        var handle = UnityEngine.AddressableAssets.Addressables
            .LoadSceneAsync(sceneAddress, 
                UnityEngine.SceneManagement.LoadSceneMode.Additive);
        await handle.Task;
        return handle.Status == 
            UnityEngine.ResourceManagement.AsyncOperations.AsyncOperationStatus.Succeeded;
    }

    void InitializeEnemies(DifficultyLevel difficulty, DungeonConfig config)
    {
        var diffConfig = Array.Find(config.Difficulties, d => d.Level == difficulty);
        if (diffConfig == null) return;
        
        // 通知所有敌人应用难度倍率
        var enemies = FindObjectsOfType<EnemyController>();
        foreach (var enemy in enemies)
        {
            enemy.SetDifficultyMultiplier(
                diffConfig.EnemyHpMultiplier, 
                diffConfig.EnemyDamageMultiplier);
        }
    }

    /// <summary>
    /// 更新目标进度
    /// </summary>
    public void UpdateObjectiveProgress(string objectiveId, int delta = 1)
    {
        if (currentInstance == null) return;
        
        var objective = Array.Find(currentInstance.Config.Objectives, 
            o => o.ObjectiveId == objectiveId);
        if (objective == null) return;
        
        if (!currentInstance.ObjectiveProgress.ContainsKey(objectiveId))
            currentInstance.ObjectiveProgress[objectiveId] = 0;
        
        currentInstance.ObjectiveProgress[objectiveId] += delta;
        int current = currentInstance.ObjectiveProgress[objectiveId];
        int required = objective.RequiredCount;
        
        OnObjectiveUpdated?.Invoke(objectiveId, current, required);
        
        // 检查是否所有必要目标都完成
        if (CheckAllObjectivesComplete())
        {
            CompleteDungeon();
        }
    }

    bool CheckAllObjectivesComplete()
    {
        foreach (var obj in currentInstance.Config.Objectives)
        {
            if (obj.IsOptional) continue;
            
            int progress = currentInstance.ObjectiveProgress.TryGetValue(
                obj.ObjectiveId, out var p) ? p : 0;
            
            if (progress < obj.RequiredCount) return false;
        }
        return true;
    }

    void CompleteDungeon()
    {
        if (currentInstance.State != DungeonState.InProgress) return;
        
        currentInstance.State = DungeonState.Completed;
        
        // 计算奖励
        GrantRewards();
        
        // 记录通关
        RecordClearHistory();
        
        // 触发任务/成就检查
        QuestSystem.Instance?.OnDungeonCleared(currentInstance.Config.DungeonId);
        AchievementSystem.Instance?.CheckDungeonAchievements(currentInstance);
        
        OnDungeonCompleted?.Invoke(currentInstance);
    }

    void TimeoutDungeon()
    {
        currentInstance.State = DungeonState.Timeout;
        OnDungeonFailed?.Invoke(currentInstance);
    }

    public void AbandonDungeon()
    {
        if (currentInstance == null) return;
        currentInstance.State = DungeonState.Abandoned;
        OnDungeonFailed?.Invoke(currentInstance);
        
        // 卸载副本场景
        UnloadDungeonScene();
    }

    void GrantRewards()
    {
        var diffConfig = Array.Find(currentInstance.Config.Difficulties, 
            d => d.Level == currentInstance.Difficulty);
        float rewardMul = diffConfig?.RewardMultiplier ?? 1f;
        
        bool isFirstClear = !HasClearedBefore(
            currentInstance.Config.DungeonId, currentInstance.Difficulty);
        
        var rewardsToGrant = new List<DungeonReward>(currentInstance.Config.Rewards);
        
        if (isFirstClear && currentInstance.Config.FirstClearBonus != null)
            rewardsToGrant.AddRange(currentInstance.Config.FirstClearBonus);
        
        foreach (var reward in rewardsToGrant)
        {
            if (UnityEngine.Random.value <= reward.DropChance)
            {
                int count = Mathf.RoundToInt(
                    UnityEngine.Random.Range(reward.CountMin, reward.CountMax) * rewardMul);
                InventorySystem.Instance?.AddItem(reward.ItemId, count);
            }
        }
    }

    /// <summary>
    /// 设置检查点（用于断线重连定位玩家）
    /// </summary>
    public void SetCheckpoint(string checkpointId, Vector3 position)
    {
        if (currentInstance == null) return;
        currentInstance.LastCheckpoint = checkpointId;
        currentInstance.CheckpointPosition = position;
        
        // 持久化检查点（防止客户端崩溃丢失进度）
        SaveCheckpointData();
    }

    bool CheckEntryRequirements(DungeonConfig config, List<string> playerIds)
    {
        // 检查必要条件...
        return true;
    }

    void ConsumeEntryItems(DungeonConfig config)
    {
        foreach (var item in config.RequiredItems ?? Array.Empty<string>())
            InventorySystem.Instance?.RemoveItem(item, 1);
    }

    bool HasClearedBefore(string dungeonId, DifficultyLevel diff) => false;
    void RecordClearHistory() { }
    void SaveCheckpointData() { }
    void UnloadDungeonScene() { }
    
    public DungeonInstance CurrentInstance => currentInstance;
}
```

## 四、副本状态机 UI

```csharp
/// <summary>
/// 副本UI控制器
/// </summary>
public class DungeonHUD : MonoBehaviour
{
    [SerializeField] private GameObject hudRoot;
    [SerializeField] private Text timerText;
    [SerializeField] private Transform objectiveContainer;
    [SerializeField] private GameObject objectiveItemPrefab;
    [SerializeField] private GameObject completionPanel;
    [SerializeField] private GameObject failurePanel;
    [SerializeField] private Text difficultyText;

    void Start()
    {
        var dm = DungeonManager.Instance;
        dm.OnDungeonStarted += OnDungeonStarted;
        dm.OnDungeonCompleted += OnDungeonCompleted;
        dm.OnDungeonFailed += OnDungeonFailed;
        dm.OnObjectiveUpdated += OnObjectiveUpdated;
        dm.OnTimerUpdated += OnTimerUpdated;
    }

    void OnDungeonStarted(DungeonInstance instance)
    {
        hudRoot.SetActive(true);
        completionPanel.SetActive(false);
        failurePanel.SetActive(false);
        
        difficultyText.text = instance.Difficulty.ToString();
        
        // 初始化目标列表
        foreach (Transform child in objectiveContainer)
            Destroy(child.gameObject);
        
        foreach (var obj in instance.Config.Objectives)
        {
            var item = Instantiate(objectiveItemPrefab, objectiveContainer);
            var text = item.GetComponentInChildren<Text>();
            text.text = $"[ ] {obj.Description} (0/{obj.RequiredCount})";
            item.name = obj.ObjectiveId;
        }
    }

    void OnObjectiveUpdated(string id, int current, int required)
    {
        var item = objectiveContainer.Find(id);
        if (item == null) return;
        
        var text = item.GetComponentInChildren<Text>();
        bool complete = current >= required;
        text.text = $"{(complete ? "[✓]" : "[ ]")} {text.text.Split(']')[1].Trim().Split('(')[0].Trim()} ({current}/{required})";
        text.color = complete ? Color.green : Color.white;
    }

    void OnTimerUpdated(float remaining)
    {
        if (timerText == null) return;
        int minutes = Mathf.FloorToInt(remaining / 60);
        int seconds = Mathf.FloorToInt(remaining % 60);
        timerText.text = $"{minutes:D2}:{seconds:D2}";
        timerText.color = remaining < 60f ? Color.red : Color.white;
    }

    void OnDungeonCompleted(DungeonInstance instance)
    {
        completionPanel.SetActive(true);
    }

    void OnDungeonFailed(DungeonInstance instance)
    {
        failurePanel.SetActive(true);
    }
}
```

## 五、副本系统最佳实践

| 功能 | 设计建议 |
|------|----------|
| 场景隔离 | 使用 Additive 场景加载，每个副本独立场景 |
| 多实例 | 服务器架构下一个服务端进程管理多个副本实例 |
| 断线重连 | 检查点 + 服务端状态保存，重连后恢复到检查点 |
| 难度动态 | 通过乘数系数配置，避免hardcode不同难度数值 |
| 奖励结算 | 服务端执行，防止客户端篡改奖励结果 |
| 副本超时 | 达到时限强制结束并给部分奖励，避免玩家长挂 |
| 重试限制 | 每日重置计数，防止无限刷本 |
