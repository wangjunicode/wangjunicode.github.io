## 一、输入抽象层设计

```csharp
using UnityEngine;
using UnityEngine.InputSystem;
using System;
using System.Collections.Generic;

/// <summary>
/// 游戏动作枚举（与具体按键解耦）
/// </summary>
public enum GameAction
{
    // 移动
    MoveHorizontal,
    MoveVertical,
    Sprint,
    Jump,
    Crouch,
    
    // 战斗
    PrimaryAttack,
    SecondaryAttack,
    Skill1,
    Skill2,
    Skill3,
    Skill4,
    
    // UI
    Confirm,
    Cancel,
    Inventory,
    Map,
    Pause,
    
    // 视角
    LookHorizontal,
    LookVertical,
    ZoomIn,
    ZoomOut,
}

/// <summary>
/// 输入状态（按下/持续/抬起）
/// </summary>
public class InputState
{
    public bool IsPressed;      // 当前帧按下
    public bool IsHeld;         // 持续按住
    public bool IsReleased;     // 当前帧抬起
    public float AxisValue;     // 轴值（-1到1）
    public Vector2 Vector2Value; // 二维轴值
}

/// <summary>
/// 游戏输入管理器（统一抽象层）
/// </summary>
public class GameInputManager : MonoBehaviour
{
    private static GameInputManager instance;
    public static GameInputManager Instance => instance;

    // New Input System 生成的类（通过 Input Action Asset 生成）
    private PlayerInputActions inputActions;
    
    private Dictionary<GameAction, InputState> actionStates 
        = new Dictionary<GameAction, InputState>();
    
    public enum InputDevice { KeyboardMouse, Gamepad, Touch }
    public InputDevice CurrentDevice { get; private set; } = InputDevice.KeyboardMouse;
    
    public event Action<InputDevice> OnDeviceChanged;

    void Awake()
    {
        instance = this;
        inputActions = new PlayerInputActions();
        
        // 初始化所有动作状态
        foreach (GameAction action in Enum.GetValues(typeof(GameAction)))
            actionStates[action] = new InputState();
        
        // 设备变化检测
        InputSystem.onActionChange += OnInputActionChange;
    }

    void OnEnable() => inputActions.Enable();
    void OnDisable() => inputActions.Disable();

    void OnDestroy()
    {
        InputSystem.onActionChange -= OnInputActionChange;
        inputActions.Dispose();
    }

    void Update()
    {
        UpdateAllStates();
        DetectActiveDevice();
    }

    void UpdateAllStates()
    {
        // 移动轴
        var moveInput = inputActions.Player.Move.ReadValue<Vector2>();
        actionStates[GameAction.MoveHorizontal].AxisValue = moveInput.x;
        actionStates[GameAction.MoveVertical].AxisValue = moveInput.y;
        actionStates[GameAction.MoveHorizontal].Vector2Value = moveInput;
        
        // 视角轴
        var lookInput = inputActions.Player.Look.ReadValue<Vector2>();
        actionStates[GameAction.LookHorizontal].AxisValue = lookInput.x;
        actionStates[GameAction.LookVertical].AxisValue = lookInput.y;
        
        // 按钮动作
        UpdateButtonState(GameAction.Jump, inputActions.Player.Jump);
        UpdateButtonState(GameAction.Sprint, inputActions.Player.Sprint);
        UpdateButtonState(GameAction.Crouch, inputActions.Player.Crouch);
        UpdateButtonState(GameAction.PrimaryAttack, inputActions.Player.PrimaryAttack);
        UpdateButtonState(GameAction.Skill1, inputActions.Player.Skill1);
        UpdateButtonState(GameAction.Inventory, inputActions.Player.Inventory);
        UpdateButtonState(GameAction.Pause, inputActions.Player.Pause);
    }

    void UpdateButtonState(GameAction action, InputAction inputAction)
    {
        var state = actionStates[action];
        state.IsPressed = inputAction.WasPressedThisFrame();
        state.IsHeld = inputAction.IsPressed();
        state.IsReleased = inputAction.WasReleasedThisFrame();
        state.AxisValue = state.IsHeld ? 1f : 0f;
    }

    void DetectActiveDevice()
    {
        // 自动检测当前使用的输入设备
        if (Gamepad.current != null && Gamepad.current.wasUpdatedThisFrame)
        {
            SetDevice(InputDevice.Gamepad);
        }
        else if (Keyboard.current != null && 
                 (Keyboard.current.anyKey.wasPressedThisFrame || 
                  Mouse.current.delta.ReadValue() != Vector2.zero))
        {
            SetDevice(InputDevice.KeyboardMouse);
        }
        
        #if UNITY_IOS || UNITY_ANDROID
        if (Input.touchCount > 0)
            SetDevice(InputDevice.Touch);
        #endif
    }

    void SetDevice(InputDevice newDevice)
    {
        if (newDevice == CurrentDevice) return;
        CurrentDevice = newDevice;
        OnDeviceChanged?.Invoke(newDevice);
        Debug.Log($"[Input] Device changed to: {newDevice}");
        
        // 更新UI提示图标（键鼠图标 vs 手柄图标）
        UIHintManager.Instance?.UpdateHints(newDevice);
    }

    void OnInputActionChange(object obj, InputActionChange change) { }

    // ============ 公共查询接口 ============

    public bool IsPressed(GameAction action) => 
        actionStates.TryGetValue(action, out var s) && s.IsPressed;
    
    public bool IsHeld(GameAction action) => 
        actionStates.TryGetValue(action, out var s) && s.IsHeld;
    
    public bool IsReleased(GameAction action) => 
        actionStates.TryGetValue(action, out var s) && s.IsReleased;
    
    public float GetAxis(GameAction action) => 
        actionStates.TryGetValue(action, out var s) ? s.AxisValue : 0f;
    
    public Vector2 GetVector2(GameAction action) => 
        actionStates.TryGetValue(action, out var s) ? s.Vector2Value : Vector2.zero;
    
    public Vector2 GetMoveInput() => GetVector2(GameAction.MoveHorizontal);
    public Vector2 GetLookInput() => GetVector2(GameAction.LookHorizontal);
}
```

## 二、虚拟摇杆实现

```csharp
/// <summary>
/// 移动端虚拟摇杆
/// </summary>
public class VirtualJoystick : MonoBehaviour, 
    UnityEngine.EventSystems.IPointerDownHandler,
    UnityEngine.EventSystems.IDragHandler,
    UnityEngine.EventSystems.IPointerUpHandler
{
    [Header("摇杆配置")]
    [SerializeField] private RectTransform background;   // 摇杆背景
    [SerializeField] private RectTransform handle;       // 摇杆手柄
    [SerializeField] private float maxRadius = 80f;       // 最大拖动半径
    [SerializeField] private float deadZone = 0.1f;       // 死区
    [SerializeField] private bool dynamicPosition = true; // 按下时移动到手指位置

    public Vector2 InputValue { get; private set; }
    public bool IsActive { get; private set; }
    
    private Vector2 startPosition;
    private Canvas canvas;
    private int touchId = -1;

    void Awake()
    {
        canvas = GetComponentInParent<Canvas>();
        startPosition = background.anchoredPosition;
    }

    public void OnPointerDown(UnityEngine.EventSystems.PointerEventData eventData)
    {
        touchId = eventData.pointerId;
        IsActive = true;
        
        if (dynamicPosition)
        {
            // 将摇杆移动到按下位置
            RectTransformUtility.ScreenPointToLocalPointInRectangle(
                canvas.GetComponent<RectTransform>(),
                eventData.position,
                canvas.worldCamera,
                out Vector2 localPos);
            
            background.anchoredPosition = localPos;
        }
        
        handle.anchoredPosition = Vector2.zero;
        OnDrag(eventData);
    }

    public void OnDrag(UnityEngine.EventSystems.PointerEventData eventData)
    {
        if (eventData.pointerId != touchId) return;
        
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            background,
            eventData.position,
            canvas.worldCamera,
            out Vector2 localPoint);
        
        // 限制在最大半径内
        Vector2 clampedPos = Vector2.ClampMagnitude(localPoint, maxRadius);
        handle.anchoredPosition = clampedPos;
        
        // 计算输入值（-1到1）
        InputValue = clampedPos / maxRadius;
        
        // 应用死区
        if (InputValue.magnitude < deadZone)
            InputValue = Vector2.zero;
    }

    public void OnPointerUp(UnityEngine.EventSystems.PointerEventData eventData)
    {
        if (eventData.pointerId != touchId) return;
        
        touchId = -1;
        IsActive = false;
        InputValue = Vector2.zero;
        handle.anchoredPosition = Vector2.zero;
        
        if (dynamicPosition)
            background.anchoredPosition = startPosition;
    }
}
```

## 三、按键重映射系统

```csharp
/// <summary>
/// 按键重映射管理器
/// </summary>
public class KeyRemappingManager : MonoBehaviour
{
    private PlayerInputActions inputActions;
    private const string BINDINGS_KEY = "input_bindings";
    
    void Awake()
    {
        inputActions = new PlayerInputActions();
        LoadBindings();
    }

    /// <summary>
    /// 开始监听新按键（重映射流程）
    /// </summary>
    public async System.Threading.Tasks.Task<bool> StartRebind(
        string actionName, int bindingIndex = 0)
    {
        var action = inputActions.FindAction(actionName);
        if (action == null) return false;
        
        action.Disable();
        
        bool success = false;
        var rebindOperation = action.PerformInteractiveRebinding(bindingIndex)
            .WithControlsExcluding("<Mouse>/position")
            .WithControlsExcluding("<Mouse>/delta")
            .OnMatchWaitForAnother(0.1f)
            .OnComplete(op =>
            {
                op.Dispose();
                success = true;
                action.Enable();
                SaveBindings();
            })
            .OnCancel(op =>
            {
                op.Dispose();
                action.Enable();
            });
        
        rebindOperation.Start();
        
        // 等待重映射完成（最多10秒超时）
        float timeout = 10f;
        while (rebindOperation != null && timeout > 0)
        {
            await System.Threading.Tasks.Task.Delay(100);
            timeout -= 0.1f;
        }
        
        return success;
    }

    public void ResetToDefault(string actionName)
    {
        var action = inputActions.FindAction(actionName);
        action?.RemoveAllBindingOverrides();
        SaveBindings();
    }

    void SaveBindings()
    {
        string json = inputActions.SaveBindingOverridesAsJson();
        PlayerPrefs.SetString(BINDINGS_KEY, json);
        PlayerPrefs.Save();
    }

    void LoadBindings()
    {
        string json = PlayerPrefs.GetString(BINDINGS_KEY, "");
        if (!string.IsNullOrEmpty(json))
            inputActions.LoadBindingOverridesFromJson(json);
    }

    public string GetBindingDisplayString(string actionName, int bindingIndex = 0)
    {
        var action = inputActions.FindAction(actionName);
        if (action == null) return "";
        return action.GetBindingDisplayString(bindingIndex);
    }
}
```

## 四、手柄振动

```csharp
/// <summary>
/// 手柄振动/触觉反馈管理
/// </summary>
public class HapticFeedbackManager : MonoBehaviour
{
    private static HapticFeedbackManager instance;
    public static HapticFeedbackManager Instance => instance;

    [SerializeField] private bool hapticEnabled = true;

    void Awake() { instance = this; }

    /// <summary>
    /// 触发振动
    /// </summary>
    /// <param name="lowFrequency">低频马达（大振动，0-1）</param>
    /// <param name="highFrequency">高频马达（细振动，0-1）</param>
    /// <param name="duration">持续时间（秒）</param>
    public void Vibrate(float lowFrequency, float highFrequency, float duration)
    {
        if (!hapticEnabled) return;
        
        var gamepad = Gamepad.current;
        if (gamepad == null) return;
        
        gamepad.SetMotorSpeeds(lowFrequency, highFrequency);
        
        // 定时停止
        Invoke(nameof(StopVibration), duration);
    }

    // 预设振动模式
    public void ShootRecoil()   => Vibrate(0.3f, 0.7f, 0.1f);
    public void TakeDamage()    => Vibrate(0.8f, 0.4f, 0.3f);
    public void LandGround()    => Vibrate(0.6f, 0.2f, 0.2f);
    public void UI_Click()      => Vibrate(0f, 0.3f, 0.05f);
    public void LevelUp()       => Vibrate(0.5f, 0.5f, 0.5f);
    public void Death()         => Vibrate(1f, 1f, 0.8f);

    void StopVibration() => Gamepad.current?.ResetHaptics();
    
    public void SetHapticEnabled(bool enabled)
    {
        hapticEnabled = enabled;
        if (!enabled) Gamepad.current?.ResetHaptics();
        PlayerPrefs.SetInt("haptic_enabled", enabled ? 1 : 0);
    }
}
```

## 五、输入提示图标动态切换

```csharp
/// <summary>
/// 输入提示UI（自动切换键鼠/手柄图标）
/// </summary>
public class InputHintUI : MonoBehaviour
{
    [System.Serializable]
    public class HintEntry
    {
        public GameAction Action;
        public Sprite KeyboardIcon;
        public Sprite GamepadIcon;
        public string ActionLabel;
    }

    [SerializeField] private HintEntry[] hints;
    [SerializeField] private UnityEngine.UI.Image iconImage;
    [SerializeField] private UnityEngine.UI.Text labelText;
    [SerializeField] private GameAction displayAction;

    void Start()
    {
        GameInputManager.Instance.OnDeviceChanged += UpdateDisplay;
        UpdateDisplay(GameInputManager.Instance.CurrentDevice);
    }

    void UpdateDisplay(GameInputManager.InputDevice device)
    {
        var hint = System.Array.Find(hints, h => h.Action == displayAction);
        if (hint == null) return;
        
        iconImage.sprite = device == GameInputManager.InputDevice.Gamepad ?
            hint.GamepadIcon : hint.KeyboardIcon;
        
        // 根据设备显示对应提示文本
        string binding = GameInputManager.Instance.CurrentDevice == 
            GameInputManager.InputDevice.Gamepad ? "[A]" : "[Space]";
        labelText.text = $"{binding} {hint.ActionLabel}";
    }
}
```

## 六、各平台输入策略

| 平台 | 主要输入 | 特殊处理 |
|------|----------|----------|
| PC | 键鼠（精准） | 支持完全重映射 |
| Console | 手柄 | 振动反馈、按钮图标差异（PS/Xbox）|
| Mobile | 触屏 | 虚拟摇杆、手势识别 |
| PC+手柄 | 自动切换 | 运行时检测设备变化，切换UI提示 |
