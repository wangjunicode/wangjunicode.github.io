## 音频系统的重要性

> "游戏中 50% 的体验来自音频"——游戏音频设计师的共识

音频是玩家情感共鸣的重要媒介：
- 背景音乐营造气氛
- 音效提供即时反馈
- 3D 音效增强空间感
- 动态音乐与游戏状态同步

技术负责人需要设计一套能支撑这些需求的音频系统。

## 一、Unity 内置音频系统

### 1.1 AudioSource / AudioClip / AudioMixer 关系

```
AudioClip：音频数据容器（wav、mp3、ogg 文件的加载结果）
AudioSource：音频播放器（挂在 GameObject 上，控制播放行为）
AudioMixer：音频混音器（多通道混音、效果器、音量控制）
AudioListener：音频接收器（通常挂在摄像机上）

数据流：
AudioClip → AudioSource（播放） → AudioMixer（混音） → AudioListener（输出到扬声器）
```

### 1.2 基础音频管理器

```csharp
/// <summary>
/// 游戏音频管理器：统一管理所有音频播放
/// </summary>
public class AudioManager : Singleton<AudioManager>
{
    [Header("Audio Mixer")]
    [SerializeField] private AudioMixer _masterMixer;
    
    [Header("Audio Sources")]
    [SerializeField] private AudioSource _bgmSource;     // 背景音乐
    [SerializeField] private AudioSource _uiSource;      // UI 音效
    [SerializeField] private int _sfxPoolSize = 20;      // 音效对象池大小
    
    private readonly Queue<AudioSource> _sfxPool = new();
    
    // 音量设置（保存到 PlayerPrefs）
    private float _masterVolume = 1f;
    private float _bgmVolume = 0.7f;
    private float _sfxVolume = 1f;
    
    protected override void Init()
    {
        // 创建 SFX 对象池
        for (int i = 0; i < _sfxPoolSize; i++)
        {
            var go = new GameObject($"SFX_{i}");
            go.transform.SetParent(transform);
            var source = go.AddComponent<AudioSource>();
            source.outputAudioMixerGroup = _masterMixer.FindMatchingGroups("SFX")[0];
            _sfxPool.Enqueue(source);
        }
        
        LoadVolumeSettings();
    }
    
    /// <summary>
    /// 播放背景音乐（带淡入淡出）
    /// </summary>
    public async UniTask PlayBGM(AudioClip clip, float fadeDuration = 1f, 
                                  CancellationToken ct = default)
    {
        if (_bgmSource.clip == clip) return; // 已经在播放
        
        // 淡出当前 BGM
        if (_bgmSource.isPlaying)
            await FadeOut(_bgmSource, fadeDuration, ct);
        
        _bgmSource.clip = clip;
        _bgmSource.loop = true;
        _bgmSource.Play();
        
        // 淡入新 BGM
        await FadeIn(_bgmSource, fadeDuration, ct);
    }
    
    /// <summary>
    /// 播放音效（使用对象池）
    /// </summary>
    public void PlaySFX(AudioClip clip, float volume = 1f)
    {
        if (clip == null) return;
        
        // 从池中获取 AudioSource
        var source = GetSFXSource();
        source.clip = clip;
        source.volume = volume * _sfxVolume;
        source.Play();
        
        // 播放完后归还
        ReturnSFXSourceAfterPlay(source, clip.length).Forget();
    }
    
    /// <summary>
    /// 在世界坐标播放 3D 音效
    /// </summary>
    public void PlaySFX3D(AudioClip clip, Vector3 worldPosition, float volume = 1f)
    {
        var source = GetSFXSource();
        source.transform.position = worldPosition;
        source.spatialBlend = 1f; // 完全 3D
        source.clip = clip;
        source.volume = volume * _sfxVolume;
        source.Play();
        
        ReturnSFXSourceAfterPlay(source, clip.length).Forget();
    }
    
    private AudioSource GetSFXSource()
    {
        if (_sfxPool.Count > 0)
            return _sfxPool.Dequeue();
        
        // 池耗尽时临时创建（带警告）
        Debug.LogWarning("[Audio] SFX pool exhausted, creating new source");
        var go = new GameObject("SFX_Extra");
        go.transform.SetParent(transform);
        return go.AddComponent<AudioSource>();
    }
    
    private async UniTaskVoid ReturnSFXSourceAfterPlay(AudioSource source, float duration)
    {
        await UniTask.Delay(TimeSpan.FromSeconds(duration + 0.1f));
        
        source.Stop();
        source.clip = null;
        source.spatialBlend = 0f; // 重置为 2D
        _sfxPool.Enqueue(source);
    }
    
    // 音量控制
    public void SetMasterVolume(float volume)
    {
        _masterVolume = volume;
        _masterMixer.SetFloat("MasterVolume", VolumeToDecibel(volume));
        PlayerPrefs.SetFloat("MasterVolume", volume);
    }
    
    public void SetBGMVolume(float volume)
    {
        _bgmVolume = volume;
        _masterMixer.SetFloat("BGMVolume", VolumeToDecibel(volume));
        PlayerPrefs.SetFloat("BGMVolume", volume);
    }
    
    private float VolumeToDecibel(float volume)
    {
        // 将 [0,1] 线性音量转换为分贝
        return volume > 0.0001f ? Mathf.Log10(volume) * 20f : -80f;
    }
    
    private void LoadVolumeSettings()
    {
        SetMasterVolume(PlayerPrefs.GetFloat("MasterVolume", 1f));
        SetBGMVolume(PlayerPrefs.GetFloat("BGMVolume", 0.7f));
    }
    
    private async UniTask FadeOut(AudioSource source, float duration, CancellationToken ct)
    {
        float startVolume = source.volume;
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            source.volume = Mathf.Lerp(startVolume, 0f, elapsed / duration);
            await UniTask.NextFrame(ct);
        }
        
        source.Stop();
        source.volume = startVolume;
    }
    
    private async UniTask FadeIn(AudioSource source, float duration, CancellationToken ct)
    {
        source.volume = 0f;
        float targetVolume = _bgmVolume;
        float elapsed = 0f;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            source.volume = Mathf.Lerp(0f, targetVolume, elapsed / duration);
            await UniTask.NextFrame(ct);
        }
        
        source.volume = targetVolume;
    }
}
```

## 二、音频 ScriptableObject 设计

### 2.1 音频数据配置化

```csharp
/// <summary>
/// 音效配置（ScriptableObject，可以在 Inspector 中配置）
/// </summary>
[CreateAssetMenu(fileName = "NewSound", menuName = "Audio/Sound")]
public class SoundConfig : ScriptableObject
{
    [Header("音频剪辑")]
    [SerializeField] private AudioClip[] _clips; // 随机从多个音效中选择一个
    
    [Header("音量")]
    [SerializeField, Range(0f, 1f)] private float _volume = 1f;
    [SerializeField, Range(0.8f, 1.2f)] private float _pitchMin = 0.9f; // 随机音调范围
    [SerializeField, Range(0.8f, 1.2f)] private float _pitchMax = 1.1f;
    
    [Header("3D音效")]
    [SerializeField] private bool _is3D = false;
    [SerializeField, Range(0.1f, 50f)] private float _minDistance = 1f;
    [SerializeField, Range(1f, 500f)] private float _maxDistance = 20f;
    
    [Header("冷却")]
    [SerializeField] private float _minInterval = 0f; // 防止同一音效连续播放
    
    private float _lastPlayTime = float.MinValue;
    
    public AudioClip GetRandomClip()
    {
        if (_clips == null || _clips.Length == 0) return null;
        return _clips[UnityEngine.Random.Range(0, _clips.Length)];
    }
    
    public float Volume => _volume;
    public float RandomPitch => UnityEngine.Random.Range(_pitchMin, _pitchMax);
    public bool Is3D => _is3D;
    public float MinDistance => _minDistance;
    public float MaxDistance => _maxDistance;
    
    public bool CanPlay()
    {
        if (Time.time - _lastPlayTime < _minInterval) return false;
        _lastPlayTime = Time.time;
        return true;
    }
}

// 使用 SoundConfig 的扩展方法
public static class AudioManagerExtensions
{
    public static void Play(this AudioManager audio, SoundConfig config, 
                             Vector3? position = null)
    {
        if (config == null || !config.CanPlay()) return;
        
        var clip = config.GetRandomClip();
        if (clip == null) return;
        
        if (config.Is3D && position.HasValue)
            audio.PlaySFX3D(clip, position.Value, config.Volume);
        else
            audio.PlaySFX(clip, config.Volume);
    }
}

// 示例：武器打击音效配置
// WeaponHitSound.asset:
//   Clips: [hit_sword_1.wav, hit_sword_2.wav, hit_sword_3.wav]
//   Volume: 0.8
//   PitchMin: 0.9, PitchMax: 1.1
//   Is3D: true
//   MinDistance: 1, MaxDistance: 15
//   MinInterval: 0.05 (防止快速攻击时音效重叠太密)
```

## 三、动态音乐系统

### 3.1 基于游戏状态的动态音乐

```csharp
/// <summary>
/// 动态音乐管理器：根据游戏状态自动切换音乐
/// </summary>
public class DynamicMusicManager : MonoBehaviour
{
    [System.Serializable]
    public class MusicState
    {
        public string StateName;
        public AudioClip Track;
        public float Volume = 1f;
        [Range(0f, 5f)] public float TransitionDuration = 1f;
    }
    
    [SerializeField] private MusicState[] _states;
    
    private string _currentStateName;
    private AudioSource _source1;
    private AudioSource _source2;
    private bool _using1 = true; // 交叉淡入淡出用两个 AudioSource
    
    void Awake()
    {
        _source1 = gameObject.AddComponent<AudioSource>();
        _source2 = gameObject.AddComponent<AudioSource>();
        _source1.loop = _source2.loop = true;
        
        // 订阅游戏状态变化
        EventBus.Subscribe<BattleStartedEvent>(_ => TransitionTo("Battle"));
        EventBus.Subscribe<BattleEndedEvent>(_ => TransitionTo("Exploration"));
        EventBus.Subscribe<BossEncounterEvent>(_ => TransitionTo("BossBattle"));
    }
    
    public void TransitionTo(string stateName)
    {
        if (_currentStateName == stateName) return;
        
        var state = System.Array.Find(_states, s => s.StateName == stateName);
        if (state == null) return;
        
        _currentStateName = stateName;
        CrossFade(state).Forget();
    }
    
    private async UniTaskVoid CrossFade(MusicState newState)
    {
        var current = _using1 ? _source1 : _source2;
        var next = _using1 ? _source2 : _source1;
        
        next.clip = newState.Track;
        next.volume = 0f;
        next.Play();
        // 可以选择在同一位置继续（无缝切换）
        // next.time = current.time % newState.Track.length;
        
        float elapsed = 0f;
        float duration = newState.TransitionDuration;
        float startVolume = current.volume;
        
        while (elapsed < duration)
        {
            elapsed += Time.deltaTime;
            float t = elapsed / duration;
            
            current.volume = Mathf.Lerp(startVolume, 0f, t);
            next.volume = Mathf.Lerp(0f, newState.Volume, t);
            
            await UniTask.NextFrame();
        }
        
        current.Stop();
        _using1 = !_using1;
    }
    
    void OnDestroy()
    {
        EventBus.Unsubscribe<BattleStartedEvent>(_ => TransitionTo("Battle"));
        // ... 取消其他订阅
    }
}
```

## 四、FMOD 集成

### 4.1 为什么需要 FMOD

```
Unity 内置音频系统的局限：
  - 音效随机化支持有限
  - 复杂的音频状态机需要大量代码
  - 混音调节不够灵活
  - 不支持自适应音频（根据游戏状态实时变化）

FMOD 的优势：
  ✅ Event 系统：每个音效是一个独立的"事件"，包含随机化、循环等设置
  ✅ Snapshot：快速切换混音配置（如：进入水下后低频化）
  ✅ 设计师工具：音频设计师可以直接在 FMOD Studio 中调整，不需要程序
  ✅ 参数系统：根据参数（如：血量、速度）实时改变音效
  ✅ 空间音效：强大的 3D 音效支持
```

### 4.2 FMOD Unity 集成

```csharp
using FMODUnity;
using FMOD.Studio;

/// <summary>
/// FMOD 音频管理器
/// </summary>
public class FMODAudioManager : MonoBehaviour
{
    // FMOD Event 路径（在 FMOD Studio 中定义）
    private const string MUSIC_EVENT = "event:/Music/Battle";
    private const string FOOTSTEP_EVENT = "event:/SFX/Character/Footstep";
    
    // 持续播放的 BGM 实例
    private EventInstance _musicInstance;
    
    void Start()
    {
        PlayBattleMusic();
    }
    
    public void PlayBattleMusic()
    {
        _musicInstance = RuntimeManager.CreateInstance(MUSIC_EVENT);
        _musicInstance.start();
    }
    
    /// <summary>
    /// 修改 FMOD 参数（实时影响音效）
    /// 例：随着玩家血量降低，音乐变得更紧张
    /// </summary>
    public void SetMusicIntensity(float intensity) // 0~1
    {
        _musicInstance.setParameterByName("Intensity", intensity);
    }
    
    /// <summary>
    /// 播放一次性音效（不需要管理生命周期）
    /// </summary>
    public void PlayFootstep(Vector3 position, float surface)
    {
        var instance = RuntimeManager.CreateInstance(FOOTSTEP_EVENT);
        instance.set3DAttributes(RuntimeUtils.To3DAttributes(position));
        instance.setParameterByName("Surface", surface); // 0=草地,1=石地,2=木地
        instance.start();
        instance.release(); // 立即释放，FMOD 会在播放完后自动清理
    }
    
    /// <summary>
    /// 使用 FMOD Snapshot：进入水下时改变混音
    /// </summary>
    private EventInstance _underwaterSnapshot;
    
    public void EnterWater()
    {
        _underwaterSnapshot = RuntimeManager.CreateInstance("snapshot:/Underwater");
        _underwaterSnapshot.start();
    }
    
    public void ExitWater()
    {
        _underwaterSnapshot.stop(FMOD.Studio.STOP_MODE.ALLOWFADEOUT);
        _underwaterSnapshot.release();
    }
    
    void OnDestroy()
    {
        _musicInstance.stop(FMOD.Studio.STOP_MODE.ALLOWFADEOUT);
        _musicInstance.release();
    }
}

/// <summary>
/// FMOD 参数驱动的战斗音乐
/// </summary>
public class BattleMusicController : MonoBehaviour
{
    private FMODAudioManager _audio;
    private Health _playerHealth;
    
    void Start()
    {
        _audio = FMODAudioManager.Instance;
        _playerHealth = Player.Instance.Health;
    }
    
    void Update()
    {
        // 根据血量实时调整音乐紧张度
        float intensity = 1f - _playerHealth.Percentage;
        _audio.SetMusicIntensity(intensity);
        
        // 根据敌人数量增加音乐强度
        int enemyCount = EnemyManager.NearbyEnemyCount;
        if (enemyCount > 5) _audio.SetMusicIntensity(Mathf.Min(1f, intensity + 0.3f));
    }
}
```

## 五、音频性能优化

### 5.1 AudioClip 加载策略

```
根据使用频率选择加载策略：

Decompress On Load（内存解压）：
  - 短音效（< 200KB），高频使用的（UI点击声、脚步声）
  - 牺牲内存，获得零延迟播放
  - 适合：几乎每帧可能播放的音效

Compressed In Memory（内存压缩）：
  - 中等大小音效（200KB ~ 2MB）
  - 平衡内存和解压延迟
  - 适合：技能音效、环境音

Streaming（流式播放）：
  - 长音轨（> 2MB），如背景音乐
  - 不占用大量内存，但 I/O 有开销
  - 适合：BGM、环境音乐

在 AudioClip Import Settings 中设置 Load Type
```

### 5.2 音频优先级管理

```csharp
/// <summary>
/// 音频优先级管理：当同时播放的音效过多时，自动停止低优先级音效
/// </summary>
public class AudioPriorityManager : MonoBehaviour
{
    private const int MAX_CONCURRENT_SOUNDS = 32;
    
    // Unity AudioSource.priority: 0=最高优先级, 256=最低优先级
    public enum SoundPriority
    {
        Critical = 0,    // 永远不会被打断（角色死亡音效等）
        High = 64,       // 玩家音效
        Medium = 128,    // 敌人音效
        Low = 192,       // 环境音效
        Ambient = 255    // 背景环境音（最容易被打断）
    }
    
    public static int PriorityToUnity(SoundPriority priority) => (int)priority;
}
```

## 总结

游戏音频系统设计的核心原则：

1. **对象池**：所有音效通过对象池管理 AudioSource，避免频繁 Instantiate
2. **配置化**：音效参数（音量、音调、间隔）用 ScriptableObject 配置，设计师可调整
3. **分层混音**：通过 AudioMixer 分层管理音量（Master/BGM/SFX/UI），支持独立控制
4. **动态音乐**：根据游戏状态（战斗/探索/Boss）自动切换音乐，提升沉浸感
5. **性能意识**：合理设置 AudioClip 加载策略，控制同时播放的音效数量

**技术选型建议**：
- 小型项目：Unity 内置音频系统足够
- 中大型项目：引入 FMOD，音频设计师可以独立迭代，减少程序介入
