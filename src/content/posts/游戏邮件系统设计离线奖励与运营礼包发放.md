## 一、邮件数据模型

```csharp
using System;
using System.Collections.Generic;

public enum MailType { System, Personal, OperationEvent, Compensation }

[Serializable]
public class MailData
{
    public string MailId;
    public MailType Type;
    public string Title;
    public string Body;
    public string SenderName;  // "系统" / 玩家名
    public long SendTime;
    public long ExpireTime;    // 过期时间（Unix时间戳），0=永不过期
    public bool IsRead;
    public bool IsCollected;   // 附件是否已领取
    public MailAttachment[] Attachments;
}

[Serializable]
public class MailAttachment
{
    public string ItemId;
    public int Count;
    public AttachmentType Type;
}

public enum AttachmentType { Item, Currency, Premium }
```

## 二、邮件管理器

```csharp
public class MailManager : MonoBehaviour
{
    private static MailManager instance;
    public static MailManager Instance => instance;
    
    private List<MailData> mails = new List<MailData>();
    private int unreadCount;
    
    public event Action<int> OnUnreadCountChanged;  // 未读数量变化（更新红点）
    public event Action OnMailListUpdated;

    void Awake() { instance = this; }

    public async System.Threading.Tasks.Task LoadMails()
    {
        mails = await NetworkService.GetMailList();
        
        // 过滤过期邮件（客户端展示层过滤）
        long now = DateTimeOffset.UtcNow.ToUnixTimeSeconds();
        mails.RemoveAll(m => m.ExpireTime > 0 && m.ExpireTime < now);
        
        // 排序：未读>已读，时间降序
        mails.Sort((a, b) =>
        {
            if (a.IsRead != b.IsRead) return a.IsRead.CompareTo(b.IsRead);
            return b.SendTime.CompareTo(a.SendTime);
        });
        
        UpdateUnreadCount();
        OnMailListUpdated?.Invoke();
    }

    public async System.Threading.Tasks.Task<bool> MarkAsRead(string mailId)
    {
        var mail = mails.Find(m => m.MailId == mailId);
        if (mail == null || mail.IsRead) return false;
        
        bool success = await NetworkService.MarkMailRead(mailId);
        if (success)
        {
            mail.IsRead = true;
            UpdateUnreadCount();
        }
        return success;
    }

    public async System.Threading.Tasks.Task<bool> CollectAttachment(string mailId)
    {
        var mail = mails.Find(m => m.MailId == mailId);
        if (mail == null || mail.IsCollected || mail.Attachments == null) return false;
        
        var result = await NetworkService.CollectMailAttachment(mailId);
        if (result != null)
        {
            mail.IsCollected = true;
            mail.IsRead = true;
            
            // 添加到背包
            foreach (var att in mail.Attachments)
                Debug.Log($"[Mail] 领取: {att.ItemId} x{att.Count}");
            
            UpdateUnreadCount();
            OnMailListUpdated?.Invoke();
            
            // 显示领取动画
            UIManager.Instance?.ShowRewardAnimation(mail.Attachments);
        }
        return result != null;
    }

    public async System.Threading.Tasks.Task CollectAllAttachments()
    {
        var uncollected = mails.FindAll(m => !m.IsCollected && 
            m.Attachments != null && m.Attachments.Length > 0);
        
        if (uncollected.Count == 0) return;
        
        var mailIds = uncollected.ConvertAll(m => m.MailId);
        await NetworkService.CollectMailAttachmentsBatch(mailIds);
        
        foreach (var mail in uncollected)
        {
            mail.IsCollected = true;
            mail.IsRead = true;
        }
        
        UpdateUnreadCount();
        OnMailListUpdated?.Invoke();
    }

    public async System.Threading.Tasks.Task DeleteMail(string mailId)
    {
        var mail = mails.Find(m => m.MailId == mailId);
        if (mail == null) return;
        
        // 有未领取附件时不允许删除
        if (!mail.IsCollected && mail.Attachments?.Length > 0)
        {
            UIManager.Instance?.ShowMessage("请先领取附件再删除");
            return;
        }
        
        await NetworkService.DeleteMail(mailId);
        mails.Remove(mail);
        UpdateUnreadCount();
        OnMailListUpdated?.Invoke();
    }

    void UpdateUnreadCount()
    {
        unreadCount = mails.FindAll(m => !m.IsRead).Count;
        OnUnreadCountChanged?.Invoke(unreadCount);
    }

    public List<MailData> GetMails() => mails;
    public int GetUnreadCount() => unreadCount;
    public bool HasUnclaimedAttachments() => mails.Exists(m => 
        !m.IsCollected && m.Attachments?.Length > 0);
}
```

## 三、邮件红点提示

```csharp
public class MailRedDot : MonoBehaviour
{
    [SerializeField] private GameObject redDot;

    void Start()
    {
        MailManager.Instance.OnUnreadCountChanged += UpdateRedDot;
        UpdateRedDot(MailManager.Instance.GetUnreadCount());
    }

    void UpdateRedDot(int count)
    {
        redDot.SetActive(count > 0 || MailManager.Instance.HasUnclaimedAttachments());
    }
}
```
