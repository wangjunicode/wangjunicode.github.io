## 一、经验曲线设计

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 等级经验配置（策划可调整）
/// </summary>
[CreateAssetMenu(fileName = "LevelConfig", menuName = "Game/Level Config")]
public class LevelConfig : ScriptableObject
{
    public int MaxLevel = 100;
    public AnimationCurve ExpCurve;    // 经验需求曲线（归一化，0-1 → 实际exp倍率）
    public long BaseExpRequired = 100; // 1→2级所需基础经验
    public float MaxLevelMultiplier = 200f; // 满级经验是基础的N倍
    
    private long[] expTable;

    public void BuildTable()
    {
        expTable = new long[MaxLevel];
        expTable[0] = 0; // 1级不需要经验
        
        for (int level = 1; level < MaxLevel; level++)
        {
            float t = (float)level / (MaxLevel - 1);
            float multiplier = ExpCurve.Evaluate(t) * MaxLevelMultiplier;
            expTable[level] = (long)(BaseExpRequired * multiplier);
        }
    }
    
    /// <summary>
    /// 获取从当前等级升到下一级所需经验
    /// </summary>
    public long GetExpRequired(int level)
    {
        if (expTable == null) BuildTable();
        if (level < 1 || level >= MaxLevel) return long.MaxValue;
        return expTable[level - 1];
    }

    /// <summary>
    /// 根据总经验计算当前等级
    /// </summary>
    public int CalculateLevel(long totalExp)
    {
        if (expTable == null) BuildTable();
        
        long accumulated = 0;
        for (int i = 1; i < MaxLevel; i++)
        {
            accumulated += expTable[i - 1];
            if (totalExp < accumulated) return i;
        }
        return MaxLevel;
    }
}

/// <summary>
/// 经验来源类型
/// </summary>
public enum ExpSourceType
{
    Combat,       // 战斗击杀
    QuestReward,  // 任务奖励
    DungeonClear, // 副本通关
    Exploration,  // 探索
    Social,       // 社交活动
    Login,        // 登录奖励
    Event,        // 活动奖励
}
```

## 二、经验管理器

```csharp
/// <summary>
/// 经验与等级管理器
/// </summary>
public class ExperienceManager : MonoBehaviour
{
    private static ExperienceManager instance;
    public static ExperienceManager Instance => instance;

    [SerializeField] private LevelConfig levelConfig;
    
    private long totalExp;
    private int currentLevel;
    private long currentLevelExp;    // 当前等级已获得的经验
    private long currentLevelNeeded; // 升到下一级需要的经验
    
    // 经验加成（活动/VIP等）
    private float expMultiplier = 1f;
    private List<ExpBonus> activeExpBonuses = new List<ExpBonus>();
    
    public int Level => currentLevel;
    public float LevelProgress => currentLevelNeeded > 0 ? 
        (float)currentLevelExp / currentLevelNeeded : 1f;
    
    public event Action<int, int> OnLevelUp; // oldLevel, newLevel
    public event Action<long> OnExpGained;   // amount

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    public void Initialize(long exp, int level)
    {
        totalExp = exp;
        currentLevel = level;
        RefreshLevelState();
    }

    public void AddExp(long amount, ExpSourceType source = ExpSourceType.Combat)
    {
        if (currentLevel >= levelConfig.MaxLevel)
        {
            Debug.Log("[Exp] 已满级，经验不再增加");
            return;
        }
        
        // 应用加成
        float bonus = GetCurrentExpMultiplier(source);
        long finalAmount = (long)(amount * bonus);
        
        totalExp += finalAmount;
        OnExpGained?.Invoke(finalAmount);
        
        // 检查升级
        CheckLevelUp();
        
        // 显示经验获得UI
        UIManager.Instance?.ShowExpGain(finalAmount, source, bonus > 1f);
    }

    void CheckLevelUp()
    {
        int oldLevel = currentLevel;
        
        while (currentLevel < levelConfig.MaxLevel)
        {
            long needed = levelConfig.GetExpRequired(currentLevel);
            
            if (currentLevelExp >= needed)
            {
                currentLevel++;
                currentLevelExp -= needed;
                Debug.Log($"[Exp] 升级！当前等级: {currentLevel}");
            }
            else break;
        }
        
        if (currentLevel != oldLevel)
        {
            RefreshLevelState();
            OnLevelUp?.Invoke(oldLevel, currentLevel);
            
            // 触发升级演出
            StartCoroutine(LevelUpSequence(oldLevel, currentLevel));
        }
    }

    void RefreshLevelState()
    {
        currentLevelNeeded = levelConfig.GetExpRequired(currentLevel);
        
        // 重新计算当前等级已获得的经验
        long accumulated = 0;
        for (int i = 1; i < currentLevel; i++)
            accumulated += levelConfig.GetExpRequired(i);
        
        currentLevelExp = totalExp - accumulated;
    }

    /// <summary>
    /// 升级演出
    /// </summary>
    System.Collections.IEnumerator LevelUpSequence(int oldLevel, int newLevel)
    {
        // 播放升级特效
        VFXPool.Instance?.Play("VFX_LevelUp", 
            GameObject.FindGameObjectWithTag("Player")?.transform.position ?? Vector3.zero);
        
        // 播放升级音效
        AudioManager.Instance?.PlaySFX(null);
        
        // 等待一帧
        yield return null;
        
        // 显示升级UI
        UIManager.Instance?.ShowLevelUpPanel(oldLevel, newLevel, GetLevelUpRewards(newLevel));
    }

    List<QuestReward> GetLevelUpRewards(int newLevel)
    {
        // 返回升级奖励列表
        var rewards = new List<QuestReward>();
        
        if (newLevel % 10 == 0) // 每10级给稀有奖励
        {
            rewards.Add(new QuestReward { Type = RewardType.Item, ItemId = "elite_chest", Amount = 1 });
        }
        
        rewards.Add(new QuestReward { Type = RewardType.Currency, Amount = newLevel * 100 });
        
        return rewards;
    }

    float GetCurrentExpMultiplier(ExpSourceType source)
    {
        float multiplier = expMultiplier;
        
        // 检查活跃加成
        foreach (var bonus in activeExpBonuses)
        {
            if (bonus.ApplicableSources.Contains(source))
                multiplier *= bonus.Multiplier;
        }
        
        return multiplier;
    }

    public void AddExpBonus(ExpBonus bonus)
    {
        activeExpBonuses.Add(bonus);
        RecalculateMultiplier();
    }

    public void RemoveExpBonus(string bonusId)
    {
        activeExpBonuses.RemoveAll(b => b.BonusId == bonusId);
        RecalculateMultiplier();
    }

    void RecalculateMultiplier()
    {
        expMultiplier = 1f;
        // 全局加成（不区分来源）
        foreach (var bonus in activeExpBonuses)
            if (bonus.IsGlobal) expMultiplier *= bonus.Multiplier;
    }
}

[Serializable]
public class ExpBonus
{
    public string BonusId;
    public float Multiplier;         // 倍率（1.5=150%经验）
    public bool IsGlobal;            // true=所有来源都加，false=特定来源
    public List<ExpSourceType> ApplicableSources;
    public long ExpireTime;          // 过期时间，0=永久
}
```

## 三、经验分配（战斗）

```csharp
/// <summary>
/// 战斗经验分配器（多人击杀时的经验分配）
/// </summary>
public static class CombatExpDistributor
{
    /// <summary>
    /// 按伤害贡献分配经验
    /// </summary>
    public static Dictionary<string, long> DistributeExp(
        long totalExp, 
        Dictionary<string, float> damageContributions)
    {
        float totalDamage = 0;
        foreach (var v in damageContributions.Values) totalDamage += v;
        
        var result = new Dictionary<string, long>();
        
        if (totalDamage <= 0)
        {
            // 无伤害记录，均分
            long share = totalExp / damageContributions.Count;
            foreach (var key in damageContributions.Keys)
                result[key] = share;
            return result;
        }
        
        // 按贡献比例分配
        long distributed = 0;
        string maxContributor = null;
        float maxContrib = 0;
        
        foreach (var kv in damageContributions)
        {
            float ratio = kv.Value / totalDamage;
            long share = (long)(totalExp * ratio);
            result[kv.Key] = share;
            distributed += share;
            
            if (kv.Value > maxContrib)
            {
                maxContrib = kv.Value;
                maxContributor = kv.Key;
            }
        }
        
        // 余数给贡献最大的玩家
        if (maxContributor != null)
            result[maxContributor] += totalExp - distributed;
        
        return result;
    }
}
```

## 四、经验系统设计要点

| 要点 | 方案 |
|------|------|
| 经验曲线 | S型曲线：前期快（成就感），后期慢（留存）|
| 满级处理 | 满级后经验转换为其他资源（材料/货币）|
| 多倍经验 | 活动/VIP/首充，短期激励玩家活跃 |
| 升级奖励 | 每N级给稀有奖励，形成里程碑期待感 |
| 经验分配 | 多人战斗按贡献分配，激励积极参与 |
| 服务端校验 | 最终等级/经验以服务端为准 |
