## 一、网络状态监测

```csharp
using System;
using System.Collections;
using UnityEngine;

/// <summary>
/// 网络连接状态
/// </summary>
public enum NetworkState
{
    Unknown,
    Connected_WiFi,
    Connected_Mobile,
    Connected_Unknown,
    Disconnected
}

/// <summary>
/// 网络状态管理器
/// </summary>
public class NetworkStateManager : MonoBehaviour
{
    private static NetworkStateManager instance;
    public static NetworkStateManager Instance => instance;

    [SerializeField] private float checkInterval = 3f;
    
    private NetworkState currentState;
    private NetworkState previousState;
    
    public NetworkState CurrentState => currentState;
    public bool IsConnected => currentState != NetworkState.Disconnected;
    public bool IsWeakSignal { get; private set; }
    
    // 事件
    public event Action<NetworkState, NetworkState> OnNetworkStateChanged; // old, new
    public event Action OnConnected;
    public event Action OnDisconnected;
    public event Action OnWeakSignalDetected;

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
    }

    void Start()
    {
        UpdateNetworkState();
        StartCoroutine(NetworkCheckLoop());
    }

    IEnumerator NetworkCheckLoop()
    {
        while (true)
        {
            yield return new WaitForSeconds(checkInterval);
            UpdateNetworkState();
        }
    }

    void UpdateNetworkState()
    {
        previousState = currentState;
        
        var reachability = Application.internetReachability;
        
        currentState = reachability switch
        {
            NetworkReachability.ReachableViaLocalAreaNetwork => NetworkState.Connected_WiFi,
            NetworkReachability.ReachableViaCarrierDataNetwork => NetworkState.Connected_Mobile,
            NetworkReachability.NotReachable => NetworkState.Disconnected,
            _ => NetworkState.Unknown
        };
        
        // 移动数据网络认为是弱网
        IsWeakSignal = currentState == NetworkState.Connected_Mobile;
        
        if (currentState != previousState)
        {
            OnNetworkStateChanged?.Invoke(previousState, currentState);
            
            if (currentState == NetworkState.Disconnected && 
                previousState != NetworkState.Disconnected)
                OnDisconnected?.Invoke();
            
            if (currentState != NetworkState.Disconnected && 
                previousState == NetworkState.Disconnected)
                OnConnected?.Invoke();
            
            if (IsWeakSignal)
                OnWeakSignalDetected?.Invoke();
            
            Debug.Log($"[Network] 状态变化: {previousState} → {currentState}");
        }
    }
}
```

## 二、请求重试（指数退避）

```csharp
/// <summary>
/// 带重试的HTTP请求（指数退避算法）
/// </summary>
public class RetryableHttpClient
{
    private int maxRetries;
    private float baseDelay;    // 基础延迟（秒）
    private float maxDelay;     // 最大延迟上限
    private float jitter;       // 抖动（防止同时重试雪崩）

    public RetryableHttpClient(int maxRetries = 3, float baseDelay = 1f, 
        float maxDelay = 30f, float jitter = 0.5f)
    {
        this.maxRetries = maxRetries;
        this.baseDelay = baseDelay;
        this.maxDelay = maxDelay;
        this.jitter = jitter;
    }

    public async System.Threading.Tasks.Task<T> Execute<T>(
        System.Func<System.Threading.Tasks.Task<T>> operation,
        System.Func<Exception, bool> shouldRetry = null)
    {
        Exception lastException = null;
        
        for (int attempt = 0; attempt <= maxRetries; attempt++)
        {
            try
            {
                return await operation();
            }
            catch (Exception ex)
            {
                lastException = ex;
                
                bool canRetry = attempt < maxRetries;
                bool willRetry = canRetry && (shouldRetry == null || shouldRetry(ex));
                
                if (!willRetry)
                    throw;
                
                // 指数退避 + 抖动
                float delay = Mathf.Min(
                    baseDelay * Mathf.Pow(2f, attempt) + 
                    UnityEngine.Random.Range(-jitter, jitter),
                    maxDelay);
                
                Debug.Log($"[Retry] 第{attempt + 1}次重试，等待 {delay:F1}s，原因: {ex.Message}");
                
                await System.Threading.Tasks.Task.Delay(
                    TimeSpan.FromSeconds(delay));
            }
        }
        
        throw lastException;
    }
}
```

## 三、WebSocket 断线重连

```csharp
/// <summary>
/// WebSocket管理器（自动重连 + 状态恢复）
/// </summary>
public class WebSocketManager : MonoBehaviour
{
    private static WebSocketManager instance;
    public static WebSocketManager Instance => instance;

    [SerializeField] private string serverUrl;
    [SerializeField] private float reconnectBaseDelay = 1f;
    [SerializeField] private int maxReconnectAttempts = 10;
    [SerializeField] private float heartbeatInterval = 30f;

    private NativeWebSocket.WebSocket webSocket;
    private int reconnectAttempts;
    private bool isConnecting;
    private float heartbeatTimer;
    
    public bool IsConnected => webSocket?.State == NativeWebSocket.WebSocketState.Open;
    
    public event Action OnConnected;
    public event Action<string> OnDisconnected;
    public event Action<byte[]> OnMessageReceived;

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    public async System.Threading.Tasks.Task Connect()
    {
        if (isConnecting || IsConnected) return;
        isConnecting = true;
        
        try
        {
            webSocket = new NativeWebSocket.WebSocket(serverUrl);
            
            webSocket.OnOpen += () =>
            {
                Debug.Log("[WS] 连接成功");
                reconnectAttempts = 0;
                isConnecting = false;
                OnConnected?.Invoke();
                
                // 重新发送未确认的消息（断线重连恢复）
                ResendPendingMessages();
            };
            
            webSocket.OnClose += (code) =>
            {
                Debug.Log($"[WS] 连接断开: {code}");
                isConnecting = false;
                OnDisconnected?.Invoke(code.ToString());
                
                // 自动重连
                if (reconnectAttempts < maxReconnectAttempts)
                    StartCoroutine(ReconnectCoroutine());
            };
            
            webSocket.OnMessage += bytes =>
            {
                OnMessageReceived?.Invoke(bytes);
            };
            
            webSocket.OnError += error =>
            {
                Debug.LogError($"[WS] 错误: {error}");
                isConnecting = false;
            };
            
            await webSocket.Connect();
        }
        catch (Exception e)
        {
            Debug.LogError($"[WS] 连接失败: {e.Message}");
            isConnecting = false;
        }
    }

    IEnumerator ReconnectCoroutine()
    {
        reconnectAttempts++;
        
        float delay = Mathf.Min(
            reconnectBaseDelay * Mathf.Pow(2f, reconnectAttempts - 1),
            60f);
        
        Debug.Log($"[WS] {delay:F1}s 后重连 (第{reconnectAttempts}次)");
        yield return new WaitForSeconds(delay);
        
        Connect().GetAwaiter();
    }

    void Update()
    {
        if (!IsConnected) return;
        
        // 处理消息队列（NativeWebSocket需要）
        webSocket?.DispatchMessageQueue();
        
        // 心跳
        heartbeatTimer += Time.deltaTime;
        if (heartbeatTimer >= heartbeatInterval)
        {
            heartbeatTimer = 0;
            SendHeartbeat();
        }
    }

    void SendHeartbeat()
    {
        SendMessage(new { type = "ping", timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds() });
    }

    public async void SendMessage(object data)
    {
        if (!IsConnected)
        {
            Debug.LogWarning("[WS] 消息发送失败：未连接");
            return;
        }
        
        string json = Newtonsoft.Json.JsonConvert.SerializeObject(data);
        await webSocket.SendText(json);
    }

    void ResendPendingMessages()
    {
        // 重发断线期间未确认的消息
        Debug.Log("[WS] 重发断线期间的消息...");
    }

    async void OnApplicationQuit()
    {
        if (webSocket != null)
            await webSocket.Close();
    }
}
```

## 四、弱网降级策略

| 场景 | 正常网络 | 弱网降级 |
|------|----------|----------|
| 排行榜刷新 | 30秒间隔 | 120秒间隔，优先用缓存 |
| 头像加载 | 实时下载 | 显示默认头像 |
| 聊天消息 | 实时推送 | 仅显示最新N条 |
| 商店价格 | 实时价格 | 使用本地缓存价格 |
| 游戏内容更新 | 自动下载 | 提示用户主动触发 |
| 帧同步 | 完整数据 | 压缩包大小，降低频率 |
