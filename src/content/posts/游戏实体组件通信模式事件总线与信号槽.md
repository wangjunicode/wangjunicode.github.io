## 一、通信模式的核心问题

游戏中实体间通信面临的典型挑战：

```
技能系统触发事件 → 需要通知：UI、音效、特效、成就、统计
玩家死亡 → 需要通知：UI、存档、任务、AI、摄像机、成就
```

如果每个系统都直接引用其他系统，依赖关系将变成一张网，难以维护。

## 二、五种通信模式对比

### 方案一：直接引用调用

```csharp
// ❌ 紧耦合的反模式
public class SkillSystem : MonoBehaviour
{
    [SerializeField] private UIManager uiManager;
    [SerializeField] private AudioManager audioManager;
    [SerializeField] private AchievementManager achievementManager;
    [SerializeField] private AnalyticsManager analyticsManager;
    
    public void UseSkill(string skillId)
    {
        // 技能系统需要知道所有依赖方
        uiManager.UpdateSkillCooldown(skillId);
        audioManager.PlaySkillSound(skillId);
        achievementManager.ReportProgress("use_skill_100");
        analyticsManager.TrackEvent("skill_used", skillId);
    }
}
```

**问题**：技能系统与所有接收方产生硬依赖，新增接收方需要修改技能系统。

---

### 方案二：UnityEvent（Unity内置）

```csharp
using UnityEngine;
using UnityEngine.Events;

public class PlayerHealth : MonoBehaviour
{
    [SerializeField] private UnityEvent<float> onHealthChanged;
    [SerializeField] private UnityEvent onPlayerDied;
    
    private float health = 100f;
    
    public void TakeDamage(float damage)
    {
        health -= damage;
        onHealthChanged.Invoke(health);
        
        if (health <= 0)
            onPlayerDied.Invoke();
    }
}
```

**优点**：Inspector 中可视化配置，零代码连接  
**缺点**：跨场景通信困难，无法传递复杂数据，性能较差（委托调用）

### 方案三：静态事件总线（Event Bus）

```csharp
/// <summary>
/// 类型安全的事件总线
/// </summary>
public static class EventBus<T> where T : struct
{
    private static event Action<T> onEvent;
    private static readonly object lockObj = new object();
    
    public static void Subscribe(Action<T> handler)
    {
        lock (lockObj) { onEvent += handler; }
    }
    
    public static void Unsubscribe(Action<T> handler)
    {
        lock (lockObj) { onEvent -= handler; }
    }
    
    public static void Publish(T eventData)
    {
        onEvent?.Invoke(eventData);
    }
    
    // 清理（场景切换时调用）
    public static void Clear()
    {
        lock (lockObj) { onEvent = null; }
    }
}

// 事件数据结构（使用 struct，避免 GC）
public struct PlayerDiedEvent
{
    public Vector3 Position;
    public string KilledBy;
    public int FinalLevel;
}

public struct SkillUsedEvent
{
    public string SkillId;
    public int UserId;
    public Vector3 Position;
    public float Damage;
}

public struct EnemyKilledEvent
{
    public string EnemyId;
    public EnemyType Type;
    public Vector3 Position;
    public string KilledBy;
}

// ============ 发布者 ============
public class Player : MonoBehaviour
{
    private float health = 100f;

    public void TakeDamage(float dmg, string source)
    {
        health -= dmg;
        
        if (health <= 0)
        {
            EventBus<PlayerDiedEvent>.Publish(new PlayerDiedEvent
            {
                Position = transform.position,
                KilledBy = source,
                FinalLevel = GetComponent<PlayerStats>()?.Level ?? 1
            });
        }
    }
}

// ============ 订阅者 ============
public class DeathUIHandler : MonoBehaviour
{
    void OnEnable() => EventBus<PlayerDiedEvent>.Subscribe(OnPlayerDied);
    void OnDisable() => EventBus<PlayerDiedEvent>.Unsubscribe(OnPlayerDied);
    
    void OnPlayerDied(PlayerDiedEvent e)
    {
        // 显示死亡UI
        Debug.Log($"Player died at {e.Position}, killed by {e.KilledBy}");
    }
}

public class AchievementTracker : MonoBehaviour
{
    void OnEnable()
    {
        EventBus<EnemyKilledEvent>.Subscribe(OnEnemyKilled);
        EventBus<SkillUsedEvent>.Subscribe(OnSkillUsed);
    }
    
    void OnDisable()
    {
        EventBus<EnemyKilledEvent>.Unsubscribe(OnEnemyKilled);
        EventBus<SkillUsedEvent>.Unsubscribe(OnSkillUsed);
    }
    
    void OnEnemyKilled(EnemyKilledEvent e)
    {
        AchievementManager.Instance?.ReportProgress("kill_100_enemies");
    }
    
    void OnSkillUsed(SkillUsedEvent e)
    {
        AchievementManager.Instance?.ReportProgress("use_skill_100_times");
    }
}
```

### 方案四：字符串事件总线（灵活但不类型安全）

```csharp
/// <summary>
/// 基于字符串键的通用事件总线（动态脚本友好）
/// </summary>
public class MessageBus : MonoBehaviour
{
    private static MessageBus instance;
    public static MessageBus Instance => instance;

    private Dictionary<string, List<Action<object>>> handlers 
        = new Dictionary<string, List<Action<object>>>();

    void Awake() { instance = this; DontDestroyOnLoad(gameObject); }

    public static void On(string eventName, Action<object> handler)
    {
        if (!Instance.handlers.ContainsKey(eventName))
            Instance.handlers[eventName] = new List<Action<object>>();
        Instance.handlers[eventName].Add(handler);
    }

    public static void Off(string eventName, Action<object> handler)
    {
        if (Instance.handlers.TryGetValue(eventName, out var list))
            list.Remove(handler);
    }

    public static void Emit(string eventName, object data = null)
    {
        if (!Instance.handlers.TryGetValue(eventName, out var list)) return;
        
        // 复制列表防止回调内修改
        var copy = new List<Action<object>>(list);
        foreach (var h in copy) h?.Invoke(data);
    }
}

// 使用示例
void Start()
{
    MessageBus.On("player.died", OnPlayerDied);
    MessageBus.On("boss.killed", OnBossKilled);
}

void OnPlayerDied(object data)
{
    if (data is PlayerDiedEvent e)
        Debug.Log($"Player died: {e.KilledBy}");
}

// 发布
MessageBus.Emit("player.died", new PlayerDiedEvent { KilledBy = "Dragon" });
```

### 方案五：响应式信号槽（ScriptableObject Signal）

```csharp
/// <summary>
/// 基于 ScriptableObject 的信号系统
/// 优点：Inspector 可视化连接，支持跨场景通信
/// </summary>
[CreateAssetMenu(fileName = "GameEvent", menuName = "Events/Game Event")]
public class GameEvent : ScriptableObject
{
    private List<GameEventListener> listeners = new List<GameEventListener>();
    
    public void Raise()
    {
        // 反向遍历，防止回调内Remove
        for (int i = listeners.Count - 1; i >= 0; i--)
            listeners[i].OnEventRaised();
    }
    
    public void RegisterListener(GameEventListener listener) 
        => listeners.Add(listener);
    
    public void UnregisterListener(GameEventListener listener) 
        => listeners.Remove(listener);
}

/// <summary>
/// 带参数的信号
/// </summary>
[CreateAssetMenu(fileName = "GameEventFloat", menuName = "Events/Float Event")]
public class GameEventFloat : ScriptableObject
{
    private List<Action<float>> listeners = new List<Action<float>>();
    
    public void Raise(float value)
    {
        for (int i = listeners.Count - 1; i >= 0; i--)
            listeners[i]?.Invoke(value);
    }
    
    public void AddListener(Action<float> cb) => listeners.Add(cb);
    public void RemoveListener(Action<float> cb) => listeners.Remove(cb);
}

/// <summary>
/// 事件监听组件
/// </summary>
public class GameEventListener : MonoBehaviour
{
    [SerializeField] private GameEvent gameEvent;
    [SerializeField] private UnityEvent response;
    
    void OnEnable() => gameEvent.RegisterListener(this);
    void OnDisable() => gameEvent.UnregisterListener(this);
    
    public void OnEventRaised() => response.Invoke();
}
```

## 三、延迟/优先级消息队列

```csharp
/// <summary>
/// 支持延迟发送和优先级的消息队列
/// </summary>
public class PriorityMessageQueue : MonoBehaviour
{
    private static PriorityMessageQueue instance;
    public static PriorityMessageQueue Instance => instance;
    
    private struct QueuedMessage
    {
        public string EventName;
        public object Data;
        public float TriggerTime;
        public int Priority; // 越高越先处理
    }
    
    private SortedList<float, Queue<QueuedMessage>> messageQueue 
        = new SortedList<float, Queue<QueuedMessage>>();
    
    void Awake() => instance = this;

    void Update()
    {
        float now = Time.time;
        while (messageQueue.Count > 0 && messageQueue.Keys[0] <= now)
        {
            var batch = messageQueue.Values[0];
            while (batch.Count > 0)
            {
                var msg = batch.Dequeue();
                MessageBus.Emit(msg.EventName, msg.Data);
            }
            messageQueue.RemoveAt(0);
        }
    }
    
    /// <summary>
    /// 延迟发送消息
    /// </summary>
    public void EmitDeferred(string eventName, object data, float delaySeconds = 0)
    {
        float triggerTime = Time.time + delaySeconds;
        
        if (!messageQueue.TryGetValue(triggerTime, out var queue))
        {
            queue = new Queue<QueuedMessage>();
            messageQueue[triggerTime] = queue;
        }
        
        queue.Enqueue(new QueuedMessage
        {
            EventName = eventName,
            Data = data,
            TriggerTime = triggerTime
        });
    }
}
```

## 四、选型指南

| 场景 | 推荐方案 | 理由 |
|------|----------|------|
| 简单UI更新 | UnityEvent | Inspector配置，直观 |
| 跨系统核心事件 | 泛型EventBus<T> | 类型安全，GC友好（struct事件） |
| 跨场景持久事件 | ScriptableObject Signal | 资产引用，不依赖单例 |
| Lua/热更脚本通信 | 字符串MessageBus | 动态键值，脚本友好 |
| 调试/测试解耦 | 接口+DI | 单测友好，可Mock |

**最佳实践原则：**
1. 事件数据用 `struct`，避免GC压力
2. 订阅和取消订阅成对出现，防止内存泄漏
3. 场景切换时清理静态事件总线
4. 不要在事件回调内修改订阅列表（需要复制遍历）
5. 高频事件（每帧数百次）不适合事件总线，直接调用更高效
