## 二、好友系统

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 好友信息
/// </summary>
[Serializable]
public class FriendInfo
{
    public string PlayerId;
    public string Nickname;
    public string AvatarUrl;
    public OnlineStatus Status;
    public string StatusMessage;    // 个性签名/当前状态
    public int Level;
    public long LastOnlineTime;     // 最后在线时间
    public string CurrentScene;     // 当前在玩的场景/关卡

    public bool IsOnline => Status == OnlineStatus.Online || Status == OnlineStatus.InGame;
    
    public string GetLastOnlineDisplay()
    {
        if (IsOnline) return "在线";
        TimeSpan ts = DateTime.UtcNow - DateTimeOffset.FromUnixTimeSeconds(LastOnlineTime).UtcDateTime;
        if (ts.TotalMinutes < 60) return $"{(int)ts.TotalMinutes}分钟前";
        if (ts.TotalHours < 24) return $"{(int)ts.TotalHours}小时前";
        return $"{(int)ts.TotalDays}天前";
    }
}

public enum OnlineStatus { Offline, Online, InGame, Busy, Away }

/// <summary>
/// 好友申请
/// </summary>
public class FriendRequest
{
    public string RequestId;
    public string FromPlayerId;
    public string FromNickname;
    public string Message;          // 申请留言
    public long CreateTime;
    public FriendRequestStatus Status;
}

public enum FriendRequestStatus { Pending, Accepted, Rejected, Expired }

/// <summary>
/// 好友管理器
/// </summary>
public class FriendManager : MonoBehaviour
{
    private static FriendManager instance;
    public static FriendManager Instance => instance;

    private Dictionary<string, FriendInfo> friends = new Dictionary<string, FriendInfo>();
    private List<FriendRequest> pendingRequests = new List<FriendRequest>();
    private HashSet<string> blockedPlayers = new HashSet<string>();
    
    public IReadOnlyDictionary<string, FriendInfo> Friends => friends;
    
    // 事件
    public event Action<FriendInfo> OnFriendOnline;
    public event Action<string> OnFriendOffline;
    public event Action<FriendRequest> OnFriendRequestReceived;
    public event Action<FriendInfo> OnFriendAdded;
    public event Action<string> OnFriendRemoved;

    void Awake() { instance = this; }

    /// <summary>
    /// 发送好友申请
    /// </summary>
    public async System.Threading.Tasks.Task<bool> SendFriendRequest(
        string targetPlayerId, string message = "")
    {
        if (friends.ContainsKey(targetPlayerId))
        {
            UIManager.Instance?.ShowMessage("已经是好友了");
            return false;
        }
        
        if (blockedPlayers.Contains(targetPlayerId))
        {
            UIManager.Instance?.ShowMessage("该玩家已被屏蔽");
            return false;
        }
        
        if (friends.Count >= 200) // 好友上限
        {
            UIManager.Instance?.ShowMessage("好友数量已达上限（200人）");
            return false;
        }
        
        bool success = await NetworkService.SendFriendRequest(targetPlayerId, message);
        if (success)
            UIManager.Instance?.ShowMessage("好友申请已发送");
        
        return success;
    }

    /// <summary>
    /// 处理好友申请
    /// </summary>
    public async System.Threading.Tasks.Task AcceptFriendRequest(string requestId)
    {
        var req = pendingRequests.Find(r => r.RequestId == requestId);
        if (req == null) return;
        
        bool success = await NetworkService.AcceptFriendRequest(requestId);
        if (success)
        {
            pendingRequests.Remove(req);
            // 服务端会推送新好友信息过来
        }
    }

    public async System.Threading.Tasks.Task RejectFriendRequest(string requestId)
    {
        var req = pendingRequests.Find(r => r.RequestId == requestId);
        if (req == null) return;
        
        await NetworkService.RejectFriendRequest(requestId);
        pendingRequests.Remove(req);
    }

    /// <summary>
    /// 删除好友
    /// </summary>
    public async System.Threading.Tasks.Task RemoveFriend(string playerId)
    {
        bool confirmed = await UIManager.Instance.ShowConfirmDialog(
            "确认删除好友？", $"确认要删除好友 '{friends[playerId]?.Nickname}' 吗？");
        
        if (!confirmed) return;
        
        await NetworkService.RemoveFriend(playerId);
        friends.Remove(playerId);
        OnFriendRemoved?.Invoke(playerId);
    }

    /// <summary>
    /// 屏蔽玩家
    /// </summary>
    public async System.Threading.Tasks.Task BlockPlayer(string playerId)
    {
        await NetworkService.BlockPlayer(playerId);
        blockedPlayers.Add(playerId);
        
        // 如果是好友，同时删除好友关系
        if (friends.ContainsKey(playerId))
        {
            friends.Remove(playerId);
            OnFriendRemoved?.Invoke(playerId);
        }
    }

    // ============ 服务端推送处理 ============

    public void OnFriendStatusChanged(string playerId, OnlineStatus newStatus)
    {
        if (friends.TryGetValue(playerId, out var friend))
        {
            var oldStatus = friend.Status;
            friend.Status = newStatus;
            
            if (!friend.IsOnline && newStatus == OnlineStatus.Online)
                OnFriendOnline?.Invoke(friend);
            else if (friend.IsOnline && newStatus == OnlineStatus.Offline)
                OnFriendOffline?.Invoke(playerId);
        }
    }

    public void OnFriendRequestPush(FriendRequest request)
    {
        if (!blockedPlayers.Contains(request.FromPlayerId))
        {
            pendingRequests.Add(request);
            OnFriendRequestReceived?.Invoke(request);
        }
    }

    public List<FriendInfo> GetOnlineFriends()
    {
        var list = new List<FriendInfo>();
        foreach (var f in friends.Values)
            if (f.IsOnline) list.Add(f);
        return list;
    }

    public List<FriendInfo> GetOfflineFriends()
    {
        var list = new List<FriendInfo>();
        foreach (var f in friends.Values)
            if (!f.IsOnline) list.Add(f);
        list.Sort((a, b) => b.LastOnlineTime.CompareTo(a.LastOnlineTime));
        return list;
    }
}
```

## 三、聊天系统

```csharp
/// <summary>
/// 聊天消息
/// </summary>
[Serializable]
public class ChatMessage
{
    public string MessageId;
    public string SenderId;
    public string SenderName;
    public string SenderAvatarUrl;
    public ChatChannel Channel;
    public string Content;           // 纯文本或富文本
    public long Timestamp;
    public List<ChatItem> Items;     // 嵌入的游戏物品链接
    
    public bool IsFromLocalPlayer => SenderId == 
        PlayerDataService.GetLocalPlayerData()?.PlayerId;
}

public enum ChatChannel
{
    World,      // 世界频道
    Guild,      // 公会频道
    Private,    // 私聊
    Nearby,     // 附近（范围广播）
    System,     // 系统公告
    Trade,      // 交易频道
}

/// <summary>
/// 聊天内嵌物品（可点击查看详情）
/// </summary>
public class ChatItem
{
    public string ItemId;
    public string ItemName;
    public ItemRarity Rarity;
    public int InsertPosition; // 在消息文本中的插入位置
}

/// <summary>
/// 聊天管理器
/// </summary>
public class ChatManager : MonoBehaviour
{
    private static ChatManager instance;
    public static ChatManager Instance => instance;

    [Header("频道配置")]
    [SerializeField] private int maxMessagesPerChannel = 200;
    [SerializeField] private float worldChannelCooldown = 10f;  // 世界频道发言冷却
    
    // 每个频道的消息历史
    private Dictionary<ChatChannel, List<ChatMessage>> channelMessages 
        = new Dictionary<ChatChannel, List<ChatMessage>>();
    
    // 私聊消息（按对象 PlayerId 分组）
    private Dictionary<string, List<ChatMessage>> privateMessages 
        = new Dictionary<string, List<ChatMessage>>();
    
    // 未读消息数
    private Dictionary<ChatChannel, int> unreadCounts 
        = new Dictionary<ChatChannel, int>();
    
    private float worldCooldownTimer;
    
    public event Action<ChatMessage> OnMessageReceived;
    public event Action<string, ChatMessage> OnPrivateMessageReceived; // playerId, message

    void Awake()
    {
        instance = this;
        
        foreach (ChatChannel channel in Enum.GetValues(typeof(ChatChannel)))
        {
            channelMessages[channel] = new List<ChatMessage>();
            unreadCounts[channel] = 0;
        }
    }

    void Update()
    {
        if (worldCooldownTimer > 0)
            worldCooldownTimer -= Time.deltaTime;
    }

    /// <summary>
    /// 发送消息
    /// </summary>
    public bool SendMessage(ChatChannel channel, string content, 
        string targetPlayerId = null, List<ChatItem> items = null)
    {
        // 频率检测
        if (channel == ChatChannel.World && worldCooldownTimer > 0)
        {
            UIManager.Instance?.ShowMessage($"世界频道冷却中，请{worldCooldownTimer:F0}秒后发言");
            return false;
        }
        
        // 内容过滤
        content = FilterContent(content);
        if (string.IsNullOrEmpty(content)) return false;
        
        // 长度限制
        if (content.Length > 200)
        {
            UIManager.Instance?.ShowMessage("消息过长（最多200字）");
            return false;
        }
        
        var msg = new ChatMessage
        {
            MessageId = Guid.NewGuid().ToString("N"),
            SenderId = PlayerDataService.GetLocalPlayerData().PlayerId,
            SenderName = PlayerDataService.GetLocalPlayerData().Nickname,
            Channel = channel,
            Content = content,
            Timestamp = DateTimeOffset.UtcNow.ToUnixTimeMilliseconds(),
            Items = items
        };
        
        if (channel == ChatChannel.World)
            worldCooldownTimer = worldChannelCooldown;
        
        // 发送到服务器
        NetworkService.SendChatMessage(channel, content, targetPlayerId, items);
        
        // 本地立即显示（乐观发送）
        AddMessageToChannel(channel, msg, targetPlayerId);
        
        return true;
    }

    /// <summary>
    /// 接收来自服务器的消息
    /// </summary>
    public void OnReceiveMessage(ChatMessage msg)
    {
        if (IsBlocked(msg.SenderId)) return; // 过滤被屏蔽的玩家
        
        if (msg.Channel == ChatChannel.Private)
        {
            if (!privateMessages.ContainsKey(msg.SenderId))
                privateMessages[msg.SenderId] = new List<ChatMessage>();
            privateMessages[msg.SenderId].Add(msg);
            
            OnPrivateMessageReceived?.Invoke(msg.SenderId, msg);
        }
        else
        {
            AddMessageToChannel(msg.Channel, msg, null);
        }
        
        OnMessageReceived?.Invoke(msg);
    }

    void AddMessageToChannel(ChatChannel channel, ChatMessage msg, string targetId)
    {
        var list = channelMessages[channel];
        list.Add(msg);
        
        // 超出上限时移除最旧的消息
        if (list.Count > maxMessagesPerChannel)
            list.RemoveAt(0);
        
        // 增加未读计数（当前不在该频道时）
        unreadCounts[channel]++;
    }

    public List<ChatMessage> GetChannelMessages(ChatChannel channel)
    {
        return channelMessages.TryGetValue(channel, out var list) ? 
            new List<ChatMessage>(list) : new List<ChatMessage>();
    }

    public int GetUnreadCount(ChatChannel channel)
    {
        return unreadCounts.TryGetValue(channel, out int count) ? count : 0;
    }

    public void MarkChannelRead(ChatChannel channel)
    {
        unreadCounts[channel] = 0;
    }

    string FilterContent(string content)
    {
        if (string.IsNullOrWhiteSpace(content)) return "";
        // 敏感词过滤
        // 实际项目使用服务端过滤或本地词库
        return content.Trim();
    }

    bool IsBlocked(string playerId)
    {
        return FriendManager.Instance?.blockedPlayers?.Contains(playerId) ?? false;
    }
}
```

## 四、公会系统架构

```csharp
/// <summary>
/// 公会成员信息
/// </summary>
public class GuildMember
{
    public string PlayerId;
    public string Nickname;
    public GuildPosition Position;  // 职位
    public int Contribution;        // 贡献值
    public long JoinTime;
    public bool IsOnline;
}

public enum GuildPosition
{
    Leader   = 0, // 会长（唯一）
    Officer  = 1, // 副会长/长老
    Elite    = 2, // 精英成员
    Member   = 3, // 普通成员
    Initiate = 4  // 见习成员
}

/// <summary>
/// 公会数据
/// </summary>
public class GuildData
{
    public string GuildId;
    public string Name;
    public string Emblem;           // 公会徽章（图片ID）
    public string Announcement;    // 公会公告
    public int Level;
    public int MemberCount;
    public int MaxMembers;
    public List<GuildMember> Members;
    
    public bool CanInvite(GuildPosition position) => 
        position <= GuildPosition.Officer;
    
    public bool CanKick(GuildPosition position) =>
        position <= GuildPosition.Officer;
    
    public bool CanManage(GuildPosition position) =>
        position <= GuildPosition.Leader;
}
```

## 五、社交系统设计要点

| 模块 | 核心设计 | 技术难点 |
|------|----------|----------|
| 好友在线状态 | 服务端推送（WebSocket长连接）| 连接断开后的状态维护 |
| 聊天消息 | 本地乐观显示 + 服务端确认 | 消息顺序保证 |
| 敏感词过滤 | 服务端过滤为主，客户端为辅 | 多语言敏感词库 |
| 反骚扰 | 屏蔽系统 + 举报系统 + 频率限制 | 防刷屏 |
| 物品链接 | 富文本+序列化嵌入 | 恶意物品链接防刷 |
