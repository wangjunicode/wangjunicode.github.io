## 一、角色控制器的核心职责

角色控制器是玩家与游戏世界交互的核心枢纽，其主要职责包括：

1. **输入响应**：接收并处理来自输入系统的移动、跳跃、交互指令
2. **碰撞检测**：与环境正确发生碰撞，防止穿透
3. **运动解算**：计算最终位移向量，处理坡度、台阶、斜面
4. **状态管理**：维护站立、跳跃、冲刺、游泳等状态机
5. **动画驱动**：将运动参数传递给动画系统

## 二、三种主流方案对比

### 2.1 Unity CharacterController

Unity 内置的 `CharacterController` 组件是最常用的入门方案：

```csharp
public class BasicCharacterController : MonoBehaviour
{
    [Header("移动参数")]
    public float moveSpeed = 5f;
    public float jumpHeight = 2f;
    public float gravity = -9.81f;
    
    [Header("地面检测")]
    public Transform groundCheck;
    public float groundDistance = 0.4f;
    public LayerMask groundMask;

    private CharacterController controller;
    private Vector3 velocity;
    private bool isGrounded;

    void Awake()
    {
        controller = GetComponent<CharacterController>();
    }

    void Update()
    {
        // 地面检测
        isGrounded = Physics.CheckSphere(groundCheck.position, groundDistance, groundMask);
        
        if (isGrounded && velocity.y < 0)
        {
            velocity.y = -2f; // 轻微下压，保持贴地
        }

        // 读取输入
        float x = Input.GetAxis("Horizontal");
        float z = Input.GetAxis("Vertical");

        // 计算移动向量（相对相机朝向）
        Vector3 move = transform.right * x + transform.forward * z;
        controller.Move(move * moveSpeed * Time.deltaTime);

        // 跳跃
        if (Input.GetButtonDown("Jump") && isGrounded)
        {
            velocity.y = Mathf.Sqrt(jumpHeight * -2f * gravity);
        }

        // 重力
        velocity.y += gravity * Time.deltaTime;
        controller.Move(velocity * Time.deltaTime);
    }
}
```

**优点：** 简单稳定，自动处理台阶和坡度  
**缺点：** 不参与物理模拟，无法被力推动，穿透大物体时有问题

### 2.2 Rigidbody 物理驱动

基于 `Rigidbody` 的方案能与物理世界完全交互：

```csharp
public class PhysicsCharacterController : MonoBehaviour
{
    [Header("移动参数")]
    public float maxSpeed = 8f;
    public float acceleration = 50f;
    public float deceleration = 40f;
    public float jumpForce = 7f;
    
    [Header("地面参数")]
    public float groundCheckRadius = 0.3f;
    public LayerMask groundLayer;
    public float maxSlopeAngle = 45f;

    private Rigidbody rb;
    private bool isGrounded;
    private Vector3 groundNormal;
    private CapsuleCollider capsule;

    void Awake()
    {
        rb = GetComponent<Rigidbody>();
        capsule = GetComponent<CapsuleCollider>();
        
        // 防止物理旋转
        rb.freezeRotation = true;
    }

    void FixedUpdate()
    {
        CheckGround();
        HandleMovement();
    }

    void CheckGround()
    {
        Vector3 bottom = transform.position + capsule.center 
            - Vector3.up * (capsule.height / 2f - capsule.radius);
            
        isGrounded = Physics.SphereCast(
            transform.position + capsule.center,
            groundCheckRadius,
            Vector3.down,
            out RaycastHit hit,
            capsule.height / 2f - capsule.radius + 0.1f,
            groundLayer
        );
        
        if (isGrounded)
        {
            groundNormal = hit.normal;
        }
    }

    void HandleMovement()
    {
        float x = Input.GetAxisRaw("Horizontal");
        float z = Input.GetAxisRaw("Vertical");
        
        Vector3 inputDir = new Vector3(x, 0, z).normalized;
        
        // 将输入转换到坡面方向
        Vector3 moveDir = Vector3.ProjectOnPlane(
            transform.TransformDirection(inputDir), 
            groundNormal
        ).normalized;
        
        Vector3 targetVelocity = moveDir * maxSpeed;
        Vector3 currentVelocity = new Vector3(rb.velocity.x, 0, rb.velocity.z);
        
        float accel = inputDir.magnitude > 0 ? acceleration : deceleration;
        Vector3 velocityDelta = (targetVelocity - currentVelocity) * accel * Time.fixedDeltaTime;
        
        rb.AddForce(velocityDelta, ForceMode.VelocityChange);
        
        // 限制最大速度
        Vector3 horizontalVel = new Vector3(rb.velocity.x, 0, rb.velocity.z);
        if (horizontalVel.magnitude > maxSpeed)
        {
            horizontalVel = horizontalVel.normalized * maxSpeed;
            rb.velocity = new Vector3(horizontalVel.x, rb.velocity.y, horizontalVel.z);
        }
    }

    public void Jump()
    {
        if (!isGrounded) return;
        rb.velocity = new Vector3(rb.velocity.x, 0, rb.velocity.z);
        rb.AddForce(Vector3.up * jumpForce, ForceMode.VelocityChange);
    }
}
```

### 2.3 自定义 Kinematic 控制器

对于需要完全掌控的项目，可实现自定义 Kinematic 控制器：

```csharp
public class KinematicCharacterController : MonoBehaviour
{
    [Header("基础参数")]
    public float moveSpeed = 6f;
    public float runSpeed = 10f;
    public float jumpHeight = 3f;
    public float gravity = 20f;
    
    [Header("碰撞参数")]
    public float skinWidth = 0.01f;
    public int maxBounces = 5;
    public float maxSlopeAngle = 55f;
    
    private CapsuleCollider capsule;
    private Vector3 velocity;
    private bool isGrounded;
    private int layerMask;

    void Awake()
    {
        capsule = GetComponent<CapsuleCollider>();
        layerMask = ~LayerMask.GetMask("Player", "Trigger");
    }

    void Update()
    {
        float x = Input.GetAxisRaw("Horizontal");
        float z = Input.GetAxisRaw("Vertical");
        bool run = Input.GetKey(KeyCode.LeftShift);
        
        float speed = run ? runSpeed : moveSpeed;
        Vector3 inputVelocity = (transform.right * x + transform.forward * z).normalized * speed;
        
        velocity.x = inputVelocity.x;
        velocity.z = inputVelocity.z;
        
        if (isGrounded && Input.GetButtonDown("Jump"))
        {
            velocity.y = Mathf.Sqrt(2f * gravity * jumpHeight);
        }
        
        velocity.y -= gravity * Time.deltaTime;
        
        // 执行碰撞解算移动
        Vector3 displacement = SweepAndSlide(velocity * Time.deltaTime);
        transform.position += displacement;
    }

    /// <summary>
    /// Sweep & Slide 碰撞响应算法
    /// </summary>
    Vector3 SweepAndSlide(Vector3 motion)
    {
        Vector3 totalMovement = Vector3.zero;
        
        for (int i = 0; i < maxBounces; i++)
        {
            if (motion.magnitude < 0.001f) break;
            
            // CapsuleCast 检测碰撞
            Vector3 point1, point2;
            GetCapsulePoints(out point1, out point2);
            
            if (Physics.CapsuleCast(
                point1, point2,
                capsule.radius - skinWidth,
                motion.normalized,
                out RaycastHit hit,
                motion.magnitude + skinWidth,
                layerMask))
            {
                // 移动到碰撞点（留出skin width）
                float safeDistance = Mathf.Max(0, hit.distance - skinWidth);
                totalMovement += motion.normalized * safeDistance;
                
                float slopeAngle = Vector3.Angle(hit.normal, Vector3.up);
                
                if (slopeAngle < maxSlopeAngle)
                {
                    // 可行走坡面：投影到坡面
                    isGrounded = true;
                    velocity.y = Mathf.Max(0, velocity.y);
                    motion = Vector3.ProjectOnPlane(motion, hit.normal);
                }
                else
                {
                    // 不可行走面：滑动
                    Vector3 slideDir = Vector3.ProjectOnPlane(motion, hit.normal);
                    motion = slideDir * (1f - hit.fraction);
                }
            }
            else
            {
                // 无碰撞，完整移动
                totalMovement += motion;
                break;
            }
        }
        
        // 地面检测
        if (!CheckGround()) isGrounded = false;
        
        return totalMovement;
    }

    void GetCapsulePoints(out Vector3 point1, out Vector3 point2)
    {
        float offset = capsule.height / 2f - capsule.radius;
        point1 = transform.position + capsule.center + Vector3.up * offset;
        point2 = transform.position + capsule.center - Vector3.up * offset;
    }

    bool CheckGround()
    {
        Vector3 point1, point2;
        GetCapsulePoints(out point1, out point2);
        return Physics.CapsuleCast(point1, point2, capsule.radius - skinWidth,
            Vector3.down, 0.05f + skinWidth, layerMask);
    }
}
```

## 三、完整状态机驱动的角色控制器

以下是一个生产级的角色控制器，包含完整状态机：

```csharp
/// <summary>
/// 角色移动状态枚举
/// </summary>
public enum CharacterState
{
    Idle,        // 站立
    Walking,     // 行走
    Running,     // 奔跑
    Jumping,     // 跳跃
    Falling,     // 下落
    Crouching,   // 蹲伏
    Sprinting,   // 冲刺
    Climbing,    // 攀爬
    Swimming,    // 游泳
    Dead         // 死亡
}

public class AdvancedCharacterController : MonoBehaviour
{
    [Header("移动配置")]
    [SerializeField] private float walkSpeed = 4f;
    [SerializeField] private float runSpeed = 7f;
    [SerializeField] private float sprintSpeed = 12f;
    [SerializeField] private float crouchSpeed = 2.5f;
    [SerializeField] private float accelerationRate = 12f;
    [SerializeField] private float decelerationRate = 16f;
    [SerializeField] private float rotationSpeed = 720f;

    [Header("跳跃配置")]
    [SerializeField] private float jumpHeight = 2.5f;
    [SerializeField] private float gravity = -20f;
    [SerializeField] private float coyoteTime = 0.15f;      // 土狼时间（离地后仍可跳跃）
    [SerializeField] private float jumpBufferTime = 0.1f;   // 跳跃预输入缓冲
    [SerializeField] private int maxAirJumps = 1;           // 最大空中跳跃次数

    [Header("地面检测")]
    [SerializeField] private float groundCheckDistance = 0.1f;
    [SerializeField] private LayerMask groundLayer;

    [Header("蹲伏配置")]
    [SerializeField] private float crouchHeight = 1.2f;
    [SerializeField] private float standHeight = 2f;
    [SerializeField] private float crouchTransitionSpeed = 8f;

    // 组件引用
    private CharacterController controller;
    private Animator animator;
    private Transform cameraTransform;

    // 运动状态
    private CharacterState currentState;
    private Vector3 moveVelocity;
    private float verticalVelocity;
    
    // 跳跃状态
    private float coyoteTimeCounter;
    private float jumpBufferCounter;
    private int remainingAirJumps;
    private bool isJumpHeld;

    // 地面状态
    private bool isGrounded;
    private bool wasGrounded;
    private Vector3 groundNormal;
    private float groundAngle;

    // 状态机回调
    private Dictionary<CharacterState, Action> stateUpdateCallbacks;
    private Dictionary<CharacterState, Action<CharacterState>> stateEnterCallbacks;

    // 动画参数哈希（性能优化）
    private static readonly int SpeedHash = Animator.StringToHash("Speed");
    private static readonly int IsGroundedHash = Animator.StringToHash("IsGrounded");
    private static readonly int IsJumpingHash = Animator.StringToHash("IsJumping");
    private static readonly int IsCrouchingHash = Animator.StringToHash("IsCrouching");
    private static readonly int VerticalVelocityHash = Animator.StringToHash("VerticalVelocity");

    void Awake()
    {
        controller = GetComponent<CharacterController>();
        animator = GetComponentInChildren<Animator>();
        cameraTransform = Camera.main.transform;

        InitializeStateMachine();
        TransitionTo(CharacterState.Idle);
    }

    void InitializeStateMachine()
    {
        stateUpdateCallbacks = new Dictionary<CharacterState, Action>
        {
            { CharacterState.Idle, UpdateIdle },
            { CharacterState.Walking, UpdateWalking },
            { CharacterState.Running, UpdateRunning },
            { CharacterState.Jumping, UpdateJumping },
            { CharacterState.Falling, UpdateFalling },
            { CharacterState.Crouching, UpdateCrouching },
            { CharacterState.Sprinting, UpdateSprinting },
        };
    }

    void Update()
    {
        wasGrounded = isGrounded;
        CheckGround();
        UpdateTimers();
        UpdateInput();

        // 执行当前状态
        if (stateUpdateCallbacks.TryGetValue(currentState, out var callback))
            callback.Invoke();

        // 应用重力
        ApplyGravity();

        // 执行移动
        Vector3 finalVelocity = moveVelocity + Vector3.up * verticalVelocity;
        controller.Move(finalVelocity * Time.deltaTime);

        // 更新动画
        UpdateAnimator();
    }

    void CheckGround()
    {
        Vector3 origin = transform.position + Vector3.up * controller.radius;
        isGrounded = Physics.SphereCast(origin, controller.radius - 0.01f,
            Vector3.down, out RaycastHit hit,
            controller.radius + groundCheckDistance, groundLayer);

        if (isGrounded)
        {
            groundNormal = hit.normal;
            groundAngle = Vector3.Angle(Vector3.up, groundNormal);
        }
        else
        {
            groundNormal = Vector3.up;
            groundAngle = 0f;
        }
    }

    void UpdateTimers()
    {
        // 土狼时间：离地后短时间内仍可跳跃
        if (isGrounded)
            coyoteTimeCounter = coyoteTime;
        else
            coyoteTimeCounter -= Time.deltaTime;

        // 跳跃预输入缓冲
        if (Input.GetButtonDown("Jump"))
            jumpBufferCounter = jumpBufferTime;
        else
            jumpBufferCounter -= Time.deltaTime;
    }

    void UpdateInput()
    {
        isJumpHeld = Input.GetButton("Jump");
    }

    void ApplyGravity()
    {
        if (isGrounded && verticalVelocity < 0)
        {
            verticalVelocity = -2f;
        }
        else
        {
            // 松开跳跃键时增加下落速度（可变跳跃高度）
            float gravityMultiplier = (!isJumpHeld && verticalVelocity > 0) ? 2f : 1f;
            verticalVelocity += gravity * gravityMultiplier * Time.deltaTime;
        }
        
        verticalVelocity = Mathf.Clamp(verticalVelocity, -40f, 40f);
    }

    Vector3 GetInputDirection()
    {
        float h = Input.GetAxisRaw("Horizontal");
        float v = Input.GetAxisRaw("Vertical");

        // 相对于摄像机方向计算移动方向
        Vector3 camForward = Vector3.ProjectOnPlane(cameraTransform.forward, Vector3.up).normalized;
        Vector3 camRight = Vector3.ProjectOnPlane(cameraTransform.right, Vector3.up).normalized;

        return (camForward * v + camRight * h).normalized;
    }

    void MoveWithAcceleration(Vector3 inputDir, float targetSpeed)
    {
        Vector3 targetVelocity = inputDir * targetSpeed;
        float accel = inputDir.magnitude > 0 ? accelerationRate : decelerationRate;
        moveVelocity = Vector3.MoveTowards(moveVelocity, targetVelocity, accel * Time.deltaTime);

        // 角色朝向跟随移动方向
        if (moveVelocity.magnitude > 0.1f)
        {
            Quaternion targetRot = Quaternion.LookRotation(moveVelocity.normalized);
            transform.rotation = Quaternion.RotateTowards(
                transform.rotation, targetRot, rotationSpeed * Time.deltaTime);
        }
    }

    bool TryJump()
    {
        bool canJump = (coyoteTimeCounter > 0) || (remainingAirJumps > 0);
        bool wantsJump = jumpBufferCounter > 0;

        if (!canJump || !wantsJump) return false;

        if (!isGrounded && coyoteTimeCounter <= 0)
        {
            remainingAirJumps--;
        }

        verticalVelocity = Mathf.Sqrt(-2f * gravity * jumpHeight);
        jumpBufferCounter = 0;
        coyoteTimeCounter = 0;
        return true;
    }

    // ============ 状态更新方法 ============

    void UpdateIdle()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, 0);

        if (TryJump()) TransitionTo(CharacterState.Jumping);
        else if (!isGrounded) TransitionTo(CharacterState.Falling);
        else if (input.magnitude > 0.1f)
        {
            bool run = Input.GetKey(KeyCode.LeftShift);
            TransitionTo(run ? CharacterState.Running : CharacterState.Walking);
        }
        else if (Input.GetKeyDown(KeyCode.LeftControl)) TransitionTo(CharacterState.Crouching);
    }

    void UpdateWalking()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, walkSpeed);

        if (TryJump()) TransitionTo(CharacterState.Jumping);
        else if (!isGrounded) TransitionTo(CharacterState.Falling);
        else if (input.magnitude < 0.1f) TransitionTo(CharacterState.Idle);
        else if (Input.GetKey(KeyCode.LeftShift)) TransitionTo(CharacterState.Running);
    }

    void UpdateRunning()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, runSpeed);

        if (TryJump()) TransitionTo(CharacterState.Jumping);
        else if (!isGrounded) TransitionTo(CharacterState.Falling);
        else if (input.magnitude < 0.1f) TransitionTo(CharacterState.Idle);
        else if (!Input.GetKey(KeyCode.LeftShift)) TransitionTo(CharacterState.Walking);
        else if (Input.GetKey(KeyCode.LeftControl)) TransitionTo(CharacterState.Sprinting);
    }

    void UpdateSprinting()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, sprintSpeed);

        if (TryJump()) TransitionTo(CharacterState.Jumping);
        else if (!isGrounded) TransitionTo(CharacterState.Falling);
        else if (input.magnitude < 0.1f || !Input.GetKey(KeyCode.LeftControl))
            TransitionTo(CharacterState.Running);
    }

    void UpdateJumping()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, runSpeed * 0.85f); // 空中移动轻微限制

        TryJump(); // 允许二段跳

        if (verticalVelocity < 0) TransitionTo(CharacterState.Falling);
        else if (isGrounded && !wasGrounded) TransitionTo(CharacterState.Idle);
    }

    void UpdateFalling()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, runSpeed * 0.7f);

        TryJump(); // 允许土狼时间 / 二段跳

        if (isGrounded) TransitionTo(CharacterState.Idle);
    }

    void UpdateCrouching()
    {
        Vector3 input = GetInputDirection();
        MoveWithAcceleration(input, crouchSpeed);

        // 尝试站起（检测头顶是否有碰撞）
        if (!Input.GetKey(KeyCode.LeftControl))
        {
            if (!CheckCeilingBlocked())
                TransitionTo(CharacterState.Idle);
        }
    }

    bool CheckCeilingBlocked()
    {
        return Physics.SphereCast(transform.position, controller.radius,
            Vector3.up, out _, standHeight - controller.radius, groundLayer);
    }

    void TransitionTo(CharacterState newState)
    {
        if (currentState == newState) return;

        // 状态退出处理
        OnStateExit(currentState);
        currentState = newState;
        // 状态进入处理
        OnStateEnter(newState);

        if (newState == CharacterState.Jumping || newState == CharacterState.Falling)
        {
            if (isGrounded) remainingAirJumps = maxAirJumps;
        }
    }

    void OnStateEnter(CharacterState state)
    {
        switch (state)
        {
            case CharacterState.Crouching:
                StartCoroutine(TransitionHeight(crouchHeight));
                break;
            case CharacterState.Idle:
            case CharacterState.Walking:
            case CharacterState.Running:
                StartCoroutine(TransitionHeight(standHeight));
                break;
            case CharacterState.Jumping:
                remainingAirJumps = maxAirJumps;
                break;
        }
    }

    void OnStateExit(CharacterState state) { }

    System.Collections.IEnumerator TransitionHeight(float targetHeight)
    {
        while (Mathf.Abs(controller.height - targetHeight) > 0.01f)
        {
            controller.height = Mathf.MoveTowards(
                controller.height, targetHeight, crouchTransitionSpeed * Time.deltaTime);
            controller.center = new Vector3(0, controller.height / 2f, 0);
            yield return null;
        }
        controller.height = targetHeight;
    }

    void UpdateAnimator()
    {
        if (animator == null) return;

        float speed = moveVelocity.magnitude;
        animator.SetFloat(SpeedHash, speed, 0.1f, Time.deltaTime);
        animator.SetBool(IsGroundedHash, isGrounded);
        animator.SetBool(IsJumpingHash, currentState == CharacterState.Jumping);
        animator.SetBool(IsCrouchingHash, currentState == CharacterState.Crouching);
        animator.SetFloat(VerticalVelocityHash, verticalVelocity, 0.05f, Time.deltaTime);
    }

    void OnDrawGizmosSelected()
    {
        if (controller == null) return;
        Gizmos.color = isGrounded ? Color.green : Color.red;
        Gizmos.DrawWireSphere(transform.position + Vector3.up * controller.radius,
            controller.radius);
    }
}
```

## 四、坡面处理与台阶检测

### 4.1 坡面滑动

```csharp
/// <summary>
/// 在超过最大坡度时自动滑落
/// </summary>
void HandleSlopeSliding()
{
    if (!isGrounded) return;
    
    if (groundAngle > controller.slopeLimit)
    {
        // 计算下滑方向
        Vector3 slideDirection = Vector3.ProjectOnPlane(
            Vector3.down, groundNormal).normalized;
            
        float slideSpeed = (groundAngle - controller.slopeLimit) / 
            (90f - controller.slopeLimit) * 15f;
            
        moveVelocity += slideDirection * slideSpeed * Time.deltaTime;
    }
}
```

### 4.2 台阶自动攀爬

```csharp
/// <summary>
/// 自动爬上小台阶
/// </summary>
bool TryStepUp(Vector3 moveDir, float stepHeight = 0.3f)
{
    if (!isGrounded || moveDir.magnitude < 0.1f) return false;
    
    // 在脚踝高度检测前方碰撞
    Vector3 ankleOrigin = transform.position + Vector3.up * 0.05f;
    if (!Physics.Raycast(ankleOrigin, moveDir, controller.radius + 0.15f, groundLayer))
        return false;
    
    // 在台阶高度检测前方是否通畅
    Vector3 kneeOrigin = transform.position + Vector3.up * (stepHeight + 0.05f);
    if (Physics.Raycast(kneeOrigin, moveDir, controller.radius + 0.15f, groundLayer))
        return false;
    
    // 确认台阶顶部可以站立
    Vector3 stepTopOrigin = transform.position + moveDir * (controller.radius + 0.1f) 
        + Vector3.up * (stepHeight + 0.1f);
    if (!Physics.Raycast(stepTopOrigin, Vector3.down, stepHeight + 0.2f, groundLayer))
        return false;
    
    // 平滑上移
    transform.position += Vector3.up * stepHeight;
    return true;
}
```

## 五、性能优化建议

| 优化点 | 方案 | 收益 |
|--------|------|------|
| 减少 Physics 调用 | 合并地面检测为单次球形检测 | 减少 CPU 消耗 |
| 动画参数哈希化 | 使用 Animator.StringToHash 缓存 | 避免字符串查找 |
| 输入去噪 | Raw 轴 + 手动加速度曲线 | 更平滑的手感 |
| 状态机精简 | 只在状态变化时更新动画 | 减少动画系统开销 |
| 层级分离 | 逻辑层与表现层分离 | 便于网络同步 |

| Rigidbody | 物理驱动、需要力推动角色 | 中 | 强 |
| 自定义 Kinematic | 竞技/格斗/平台跳跃 | 高 | 完全自控 |

选择合适的角色控制器方案需要结合项目类型、网络同步需求及物理交互程度综合考量。对于大型商业项目，推荐基于自定义 Kinematic + 状态机的方案，兼顾手感与可扩展性。
