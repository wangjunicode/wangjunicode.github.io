## 一、程序化地形生成

```csharp
using UnityEngine;

/// <summary>
/// 程序化地形生成器（基于 Perlin 噪声）
/// </summary>
[RequireComponent(typeof(Terrain))]
public class ProceduralTerrainGenerator : MonoBehaviour
{
    [Header("地形尺寸")]
    [SerializeField] private int width = 512;
    [SerializeField] private int height = 512;
    [SerializeField] private float heightMultiplier = 50f;

    [Header("噪声参数")]
    [SerializeField] private float baseScale = 0.003f;      // 基础频率（地形整体形状）
    [SerializeField] private float detailScale = 0.02f;     // 细节频率（小山丘）
    [SerializeField] private float roughnessScale = 0.1f;   // 粗糙度（地形质感）
    
    [Header("FBM 叠加（分形布朗运动）")]
    [SerializeField] private int octaves = 6;               // 叠加层数
    [SerializeField] private float lacunarity = 2f;         // 频率增长
    [SerializeField] private float persistence = 0.5f;      // 振幅衰减
    
    [Header("种子")]
    [SerializeField] private int seed = 42;
    [SerializeField] private bool randomSeedOnStart;

    private Terrain terrain;
    private TerrainData terrainData;

    void Awake()
    {
        terrain = GetComponent<Terrain>();
        terrainData = terrain.terrainData;
    }

    [ContextMenu("Generate Terrain")]
    public void Generate()
    {
        if (randomSeedOnStart)
            seed = Random.Range(0, 100000);
        
        terrainData.heightmapResolution = width + 1;
        terrainData.size = new Vector3(width, heightMultiplier, height);
        
        float[,] heights = GenerateHeightMap();
        terrainData.SetHeights(0, 0, heights);
        
        PaintTextures(heights);
        PlaceVegetation(heights);
        
        Debug.Log($"[Terrain] Generated with seed: {seed}");
    }

    float[,] GenerateHeightMap()
    {
        float[,] heights = new float[width + 1, height + 1];
        
        Vector2 offset = new Vector2(seed * 1234.567f, seed * 7654.321f);
        
        for (int x = 0; x <= width; x++)
        {
            for (int z = 0; z <= height; z++)
            {
                float h = FractionalBrownianMotion(
                    (x + offset.x) * baseScale, 
                    (z + offset.y) * baseScale);
                
                // 应用高度曲线（让低地平整，高山陡峭）
                h = ApplyHeightCurve(h);
                
                heights[x, z] = h;
            }
        }
        
        return heights;
    }

    float FractionalBrownianMotion(float x, float z)
    {
        float value = 0f;
        float amplitude = 1f;
        float frequency = 1f;
        float maxValue = 0f;
        
        for (int i = 0; i < octaves; i++)
        {
            value += Mathf.PerlinNoise(x * frequency, z * frequency) * amplitude;
            maxValue += amplitude;
            amplitude *= persistence;
            frequency *= lacunarity;
        }
        
        return value / maxValue; // 归一化到 0-1
    }

    float ApplyHeightCurve(float h)
    {
        // 低处平缓（海岸/平原），高处陡峭（山峰）
        // 使用 SmoothStep 处理过渡
        if (h < 0.3f) return h * 0.5f;           // 低洼地区
        if (h < 0.6f) return h;                    // 中等高度
        return 0.6f + (h - 0.6f) * 2f;            // 高山地区（放大）
    }

    void PaintTextures(float[,] heights)
    {
        if (terrainData.terrainLayers == null || terrainData.terrainLayers.Length < 3)
            return;
        
        int alphaWidth = terrainData.alphamapWidth;
        int alphaHeight = terrainData.alphamapHeight;
        int layerCount = terrainData.terrainLayers.Length;
        
        float[,,] alphaMap = new float[alphaWidth, alphaHeight, layerCount];
        
        for (int x = 0; x < alphaWidth; x++)
        {
            for (int z = 0; z < alphaHeight; z++)
            {
                // 高度映射到纹理（坐标系注意转换）
                float h = heights[
                    Mathf.RoundToInt((float)x / alphaWidth * width),
                    Mathf.RoundToInt((float)z / alphaHeight * height)];
                
                // 坡度计算
                float steepness = terrainData.GetSteepness(
                    (float)x / alphaWidth, (float)z / alphaHeight);
                
                // 根据高度和坡度分配纹理权重
                float grassWeight, rockWeight, snowWeight;
                
                if (h < 0.25f) // 低处：草地
                {
                    grassWeight = 1f; rockWeight = 0f; snowWeight = 0f;
                }
                else if (h < 0.6f) // 中等：草地+岩石混合
                {
                    float t = (h - 0.25f) / 0.35f;
                    grassWeight = 1f - t;
                    rockWeight = t;
                    snowWeight = 0f;
                }
                else // 高处：雪地
                {
                    grassWeight = 0f;
                    rockWeight = steepness > 30 ? 1f : 0.3f; // 陡峭处更多岩石
                    snowWeight = steepness > 30 ? 0f : 0.7f;
                }
                
                if (layerCount > 0) alphaMap[x, z, 0] = grassWeight;
                if (layerCount > 1) alphaMap[x, z, 1] = rockWeight;
                if (layerCount > 2) alphaMap[x, z, 2] = snowWeight;
            }
        }
        
        terrainData.SetAlphamaps(0, 0, alphaMap);
    }

    void PlaceVegetation(float[,] heights)
    {
        var treeInstances = new System.Collections.Generic.List<TreeInstance>();
        
        Random.InitState(seed);
        int treePrototypeCount = terrainData.treePrototypes.Length;
        
        if (treePrototypeCount == 0) return;
        
        for (int x = 0; x < width; x += 8) // 每8格检查一次
        {
            for (int z = 0; z < height; z += 8)
            {
                float h = heights[x, z];
                
                // 只在草地区域种树
                if (h < 0.2f || h > 0.5f) continue;
                
                // 随机概率
                if (Random.value > 0.15f) continue;
                
                // 添加一棵树（随机偏移）
                var tree = new TreeInstance
                {
                    position = new Vector3(
                        ((float)x + Random.Range(-3, 3)) / width,
                        0,
                        ((float)z + Random.Range(-3, 3)) / height),
                    prototypeIndex = Random.Range(0, treePrototypeCount),
                    widthScale = Random.Range(0.7f, 1.3f),
                    heightScale = Random.Range(0.8f, 1.2f),
                    color = Color.white,
                    lightmapColor = Color.white
                };
                
                treeInstances.Add(tree);
            }
        }
        
        terrainData.treeInstances = treeInstances.ToArray();
        Debug.Log($"[Terrain] Placed {treeInstances.Count} trees");
    }
}
```

## 二、运行时地形破坏

```csharp
/// <summary>
/// 运行时地形破坏（爆炸坑）
/// </summary>
public class TerrainDeformer : MonoBehaviour
{
    [SerializeField] private Terrain terrain;
    [SerializeField] private float craterDepth = 5f;
    [SerializeField] private float craterRadius = 3f;

    /// <summary>
    /// 在指定世界位置创建爆炸坑
    /// </summary>
    public void CreateCrater(Vector3 worldPos)
    {
        if (terrain == null) return;
        
        TerrainData td = terrain.terrainData;
        
        // 世界坐标转换为地形坐标（0-1）
        Vector3 terrainPos = worldPos - terrain.transform.position;
        float normalizedX = terrainPos.x / td.size.x;
        float normalizedZ = terrainPos.z / td.size.z;
        
        // 地形高度图坐标
        int hmWidth = td.heightmapResolution;
        int hmHeight = td.heightmapResolution;
        
        int centerX = Mathf.RoundToInt(normalizedX * (hmWidth - 1));
        int centerZ = Mathf.RoundToInt(normalizedZ * (hmHeight - 1));
        
        // 影响半径（转换为高度图像素数）
        int radiusInPixels = Mathf.RoundToInt(craterRadius / td.size.x * hmWidth);
        
        int xStart = Mathf.Max(0, centerX - radiusInPixels);
        int zStart = Mathf.Max(0, centerZ - radiusInPixels);
        int xEnd = Mathf.Min(hmWidth - 1, centerX + radiusInPixels);
        int zEnd = Mathf.Min(hmHeight - 1, centerZ + radiusInPixels);
        
        int sampleWidth = xEnd - xStart + 1;
        int sampleHeight = zEnd - zStart + 1;
        
        float[,] heights = td.GetHeights(xStart, zStart, sampleWidth, sampleHeight);
        float depthNormalized = craterDepth / td.size.y;
        
        for (int x = 0; x < sampleWidth; x++)
        {
            for (int z = 0; z < sampleHeight; z++)
            {
                float dx = (x + xStart - centerX) / (float)radiusInPixels;
                float dz = (z + zStart - centerZ) / (float)radiusInPixels;
                float dist = Mathf.Sqrt(dx * dx + dz * dz);
                
                if (dist <= 1f)
                {
                    // 用余弦曲线创建弧形坑
                    float deformAmount = (1f - dist) * depthNormalized;
                    deformAmount *= Mathf.Cos(dist * Mathf.PI * 0.5f);
                    heights[z, x] = Mathf.Max(0f, heights[z, x] - deformAmount);
                }
            }
        }
        
        td.SetHeights(xStart, zStart, heights);
    }

    /// <summary>
    /// 脚印效果（角色走过留下痕迹）
    /// </summary>
    public void CreateFootprint(Vector3 worldPos, float radius = 0.3f, float depth = 0.05f)
    {
        // 与 CreateCrater 类似但参数更小
        var savedCraterDepth = craterDepth;
        var savedCraterRadius = craterRadius;
        craterDepth = depth;
        craterRadius = radius;
        CreateCrater(worldPos);
        craterDepth = savedCraterDepth;
        craterRadius = savedCraterRadius;
    }
}
```

## 三、大型地形分块管理

```csharp
/// <summary>
/// 开放世界地形分块（Terrain Tile）管理
/// </summary>
public class TerrainTileManager : MonoBehaviour
{
    [SerializeField] private int tileSize = 500;        // 每块地形大小（米）
    [SerializeField] private int loadRadius = 2;         // 加载半径（格）
    [SerializeField] private Transform player;

    private Dictionary<Vector2Int, Terrain> loadedTiles 
        = new Dictionary<Vector2Int, Terrain>();
    private Vector2Int lastPlayerTile = new Vector2Int(int.MinValue, int.MinValue);

    void Update()
    {
        if (player == null) return;
        
        Vector2Int currentTile = WorldToTile(player.position);
        
        if (currentTile != lastPlayerTile)
        {
            lastPlayerTile = currentTile;
            UpdateLoadedTiles(currentTile);
        }
    }

    Vector2Int WorldToTile(Vector3 worldPos)
    {
        return new Vector2Int(
            Mathf.FloorToInt(worldPos.x / tileSize),
            Mathf.FloorToInt(worldPos.z / tileSize));
    }

    void UpdateLoadedTiles(Vector2Int centerTile)
    {
        var toLoad = new HashSet<Vector2Int>();
        
        for (int x = -loadRadius; x <= loadRadius; x++)
        {
            for (int z = -loadRadius; z <= loadRadius; z++)
            {
                toLoad.Add(centerTile + new Vector2Int(x, z));
            }
        }
        
        // 卸载超出范围的地形
        var toUnload = new List<Vector2Int>();
        foreach (var tile in loadedTiles.Keys)
        {
            if (!toLoad.Contains(tile))
                toUnload.Add(tile);
        }
        
        foreach (var tile in toUnload)
            UnloadTile(tile);
        
        // 加载新地形
        foreach (var tile in toLoad)
        {
            if (!loadedTiles.ContainsKey(tile))
                LoadTile(tile);
        }
    }

    async void LoadTile(Vector2Int tileCoord)
    {
        string address = $"Terrain_{tileCoord.x}_{tileCoord.y}";
        var terrainPrefab = await AssetLoader.Instance.LoadAsync<GameObject>(address);
        
        if (terrainPrefab != null)
        {
            Vector3 worldPos = new Vector3(tileCoord.x * tileSize, 0, tileCoord.y * tileSize);
            var go = Instantiate(terrainPrefab, worldPos, Quaternion.identity);
            var terrain = go.GetComponent<Terrain>();
            
            if (terrain != null)
                loadedTiles[tileCoord] = terrain;
        }
    }

    void UnloadTile(Vector2Int tileCoord)
    {
        if (loadedTiles.TryGetValue(tileCoord, out var terrain))
        {
            Destroy(terrain.gameObject);
            loadedTiles.Remove(tileCoord);
            AssetLoader.Instance.Release($"Terrain_{tileCoord.x}_{tileCoord.y}");
        }
    }
}
```

## 四、地形性能优化

| 优化项 | 设置 | 效果 |
|--------|------|------|
| Pixel Error | 增大（15-25）| 减少三角形数量 |
| Base Map Distance | 减小 | 减少细节纹理范围 |
| Detail Distance | 减小 | 草地/细节渲染距离 |
| Tree Billboard Start | 减小 | 远处树用广告牌 |
| Cast Shadows | 关闭或仅Self | 降低Shadow Pass开销 |
| Draw Instanced | 开启 | GPU Instancing加速 |
