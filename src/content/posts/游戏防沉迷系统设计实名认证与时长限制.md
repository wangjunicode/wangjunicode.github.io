## 一、防沉迷法规要求（中国）

```
国家新闻出版总署 2021年8月 最新规定：

未成年人（18岁以下）：
- 工作日：不可玩游戏（0时间）
- 节假日：每天最多1小时（晚上8点到9点）
- 全年未成年游戏时长：每个节假日只有1小时窗口

实名认证要求：
- 所有网络游戏玩家必须实名认证
- 游客模式：每天最多1小时，累计最多3小时
- 认证为未成年：按未成年规则限制
- 认证为成年：无时间限制（但仍有健康提醒）

月度充值限制（未成年）：
- 8-16岁：月充值≤200元
- 16-18岁：月充值≤400元
- 8岁以下：禁止充值
```

## 二、防沉迷管理器

```csharp
using System;
using UnityEngine;

/// <summary>
/// 防沉迷状态
/// </summary>
public enum AntiAddictionStatus
{
    Guest,          // 游客（未实名）
    Adult,          // 成年人
    Minor_Teen,     // 未成年（16-17岁）
    Minor_Child,    // 未成年（8-15岁）
    Minor_Baby,     // 未成年（8岁以下）
    Blocked         // 被封锁（无法游戏）
}

/// <summary>
/// 防沉迷管理器
/// </summary>
public class AntiAddictionManager : MonoBehaviour
{
    private static AntiAddictionManager instance;
    public static AntiAddictionManager Instance => instance;

    private AntiAddictionStatus status = AntiAddictionStatus.Guest;
    private float todayPlaySeconds;         // 今日游戏时长（秒）
    private float sessionPlaySeconds;       // 本次会话时长
    private DateTime lastLoginDate;
    private bool isGameBlocked;
    
    // 时间限制（秒）
    private const float GUEST_DAILY_LIMIT = 3600f;           // 游客每日1小时
    private const float MINOR_HOLIDAY_LIMIT = 3600f;         // 未成年节假日1小时
    private const float ADULT_REMINDER_INTERVAL = 5400f;     // 成年人每1.5小时提醒一次
    
    public event Action OnGameBlocked;                       // 游戏被封锁
    public event Action<float> OnPlayTimeWarning;            // 时间警告（剩余秒数）
    public event Action<TimeSpan> OnNextAvailableTime;       // 下次可玩时间

    void Awake()
    {
        instance = this;
        DontDestroyOnLoad(gameObject);
    }

    void Start()
    {
        LoadPlayTimeData();
        ResetDailyIfNewDay();
        StartCoroutine(PlayTimeUpdateLoop());
    }

    /// <summary>
    /// 设置认证状态（实名认证完成后调用）
    /// </summary>
    public void SetAuthenticationStatus(int age, bool isAuthenticated)
    {
        if (!isAuthenticated)
        {
            status = AntiAddictionStatus.Guest;
            return;
        }

        if (age < 8)
            status = AntiAddictionStatus.Minor_Baby;
        else if (age < 16)
            status = AntiAddictionStatus.Minor_Child;
        else if (age < 18)
            status = AntiAddictionStatus.Minor_Teen;
        else
            status = AntiAddictionStatus.Adult;
        
        Debug.Log($"[AntiAddiction] Status set to: {status} (age: {age})");
        CheckAndEnforceRestrictions();
    }

    void CheckAndEnforceRestrictions()
    {
        switch (status)
        {
            case AntiAddictionStatus.Minor_Baby:
            case AntiAddictionStatus.Minor_Child:
            case AntiAddictionStatus.Minor_Teen:
                CheckMinorRestrictions();
                break;
                
            case AntiAddictionStatus.Guest:
                CheckGuestRestrictions();
                break;
        }
    }

    void CheckMinorRestrictions()
    {
        DateTime now = DateTime.Now;
        
        // 检查时间是否在允许的窗口内（节假日20:00-21:00）
        bool isHoliday = IsHoliday(now);
        bool isInTimeWindow = now.Hour == 20; // 20:00-21:00
        
        if (!isHoliday || !isInTimeWindow)
        {
            BlockGame($"未成年人只能在节假日20:00-21:00游戏");
            
            // 计算下次可玩时间
            DateTime nextPlayTime = GetNextMinorPlayTime(now);
            OnNextAvailableTime?.Invoke(nextPlayTime - now);
            return;
        }
        
        // 检查今日已游戏时长
        if (todayPlaySeconds >= MINOR_HOLIDAY_LIMIT)
        {
            BlockGame("今日游戏时间已用完（1小时）");
            return;
        }
        
        // 发出警告
        float remaining = MINOR_HOLIDAY_LIMIT - todayPlaySeconds;
        if (remaining <= 300) // 剩余5分钟警告
            OnPlayTimeWarning?.Invoke(remaining);
    }

    void CheckGuestRestrictions()
    {
        if (todayPlaySeconds >= GUEST_DAILY_LIMIT)
        {
            BlockGame("游客每日游戏时间已用完，请实名认证继续游戏");
        }
    }

    void BlockGame(string reason)
    {
        if (isGameBlocked) return;
        
        isGameBlocked = true;
        Debug.LogWarning($"[AntiAddiction] Game blocked: {reason}");
        OnGameBlocked?.Invoke();
        
        // 显示封锁界面
        UIManager.Instance?.ShowAntiAddictionBlock(reason);
    }

    System.Collections.IEnumerator PlayTimeUpdateLoop()
    {
        while (true)
        {
            yield return new WaitForSecondsRealtime(1f);
            
            if (!isGameBlocked)
            {
                todayPlaySeconds += 1f;
                sessionPlaySeconds += 1f;
                
                // 成年人健康提醒
                if (status == AntiAddictionStatus.Adult && 
                    sessionPlaySeconds % ADULT_REMINDER_INTERVAL < 1f)
                {
                    UIManager.Instance?.ShowHealthReminder();
                }
                
                CheckAndEnforceRestrictions();
                SavePlayTimeData();
            }
        }
    }

    bool IsHoliday(DateTime date)
    {
        // 检查是否是节假日（周末或法定假日）
        if (date.DayOfWeek == DayOfWeek.Saturday || date.DayOfWeek == DayOfWeek.Sunday)
            return true;
        
        // 法定节假日列表（需从服务端更新）
        var holidays = GetHolidayList();
        return holidays.Contains(date.Date);
    }

    HashSet<DateTime> GetHolidayList()
    {
        // 从缓存的节假日列表获取
        return new HashSet<DateTime>(); // 实际从服务端获取
    }

    DateTime GetNextMinorPlayTime(DateTime now)
    {
        // 找下一个节假日的20:00
        DateTime check = now.Date.AddHours(20);
        while (!IsHoliday(check) || check < now)
            check = check.AddDays(1).Date.AddHours(20);
        return check;
    }

    /// <summary>
    /// 充值限额检查（未成年人月充值限制）
    /// </summary>
    public bool CanRecharge(float amount)
    {
        if (status == AntiAddictionStatus.Minor_Baby)
        {
            UIManager.Instance?.ShowMessage("8岁以下未成年人不可充值");
            return false;
        }
        
        float monthlyLimit = status switch
        {
            AntiAddictionStatus.Minor_Child => 200f,
            AntiAddictionStatus.Minor_Teen  => 400f,
            _ => float.MaxValue
        };
        
        float monthlySpent = GetMonthlySpent();
        
        if (monthlySpent + amount > monthlyLimit)
        {
            UIManager.Instance?.ShowMessage($"本月充值已达上限（{monthlyLimit}元），无法继续充值");
            return false;
        }
        
        return true;
    }

    float GetMonthlySpent() => PlayerPrefs.GetFloat("monthly_spent", 0f);

    void LoadPlayTimeData()
    {
        string savedDate = PlayerPrefs.GetString("play_date", "");
        if (DateTime.TryParse(savedDate, out lastLoginDate))
        {
            if (lastLoginDate.Date == DateTime.Today)
                todayPlaySeconds = PlayerPrefs.GetFloat("today_play_seconds", 0f);
        }
    }

    void ResetDailyIfNewDay()
    {
        if (DateTime.Today > lastLoginDate.Date)
        {
            todayPlaySeconds = 0f;
            lastLoginDate = DateTime.Today;
            SavePlayTimeData();
        }
    }

    void SavePlayTimeData()
    {
        PlayerPrefs.SetFloat("today_play_seconds", todayPlaySeconds);
        PlayerPrefs.SetString("play_date", DateTime.Today.ToString("yyyy-MM-dd"));
    }

    public float GetTodayPlayMinutes() => todayPlaySeconds / 60f;
    public AntiAddictionStatus CurrentStatus => status;
}
```

## 三、实名认证接入

```csharp
/// <summary>
/// 实名认证接入（腾讯健康系统 / 公安网络游戏防沉迷实名认证系统）
/// </summary>
public class RealNameAuthService
{
    private const string AUTH_URL = "https://wlc.nppa.gov.cn/test/authentication/check";
    private const string APP_ID = "your_app_id";
    private const string BUSINESS_ID = "your_biz_id";
    
    [Serializable]
    private class AuthRequest
    {
        public string ai;       // 用户唯一标识（游戏内UID的Hash）
        public string name;     // 姓名
        public string idNum;    // 身份证号
    }

    [Serializable]
    private class AuthResponse
    {
        public int errcode;
        public string errmsg;
        public AuthResultData data;
    }

    [Serializable]
    private class AuthResultData
    {
        public int result;      // 0=未成年, 1=成年, 2=未知
        public string pi;       // 公安平台分配的用户ID（用于后续查询）
    }

    public async System.Threading.Tasks.Task<(bool success, bool isAdult, int age)> 
        Authenticate(string userId, string realName, string idCard)
    {
        // 注意：实际项目中实名认证必须在服务端完成，客户端只负责UI
        // 身份证号不能明文传输到服务端以外的地方
        
        // 向自家服务端发送认证请求，服务端再调用公安API
        var result = await CallServerAuth(userId, realName, idCard);
        
        if (result.success)
        {
            // 根据身份证提取年龄（取生日字段）
            int age = ExtractAgeFromIdCard(idCard);
            AntiAddictionManager.Instance?.SetAuthenticationStatus(age, true);
            return (true, age >= 18, age);
        }
        
        return (false, false, 0);
    }

    async System.Threading.Tasks.Task<(bool success, int result)> 
        CallServerAuth(string userId, string name, string idCard)
    {
        // 发送到自家服务端（服务端调用公安API）
        // 不在客户端直接调用公安API（安全要求）
        return (true, 1); // 简化
    }

    int ExtractAgeFromIdCard(string idCard)
    {
        if (idCard.Length != 18) return -1;
        
        string birthStr = idCard.Substring(6, 8);
        if (DateTime.TryParseExact(birthStr, "yyyyMMdd", null, 
            System.Globalization.DateTimeStyles.None, out DateTime birth))
        {
            int age = DateTime.Today.Year - birth.Year;
            if (DateTime.Today < birth.AddYears(age)) age--;
            return age;
        }
        return -1;
    }
}
```
