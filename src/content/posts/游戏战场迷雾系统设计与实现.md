## 一、战场迷雾的核心需求

战场迷雾（Fog of War）系统的主要功能：

1. **视野管理**：每个单位有自己的视野范围
2. **迷雾渲染**：未探索区域完全黑暗，已探索但当前不在视野的区域灰暗
3. **灰雾区域**（Previously Seen）：曾经探索过的区域保持灰色地形显示
4. **视野障碍**：山丘、建筑物遮挡视野
5. **多人同步**：每个玩家看到本方队伍的合并视野

---

## 二、迷雾纹理方案

最常用的实现方案是基于 **渲染纹理（Render Texture）** 的迷雾遮罩：

```
FogTexture（R通道 = 当前可见, G通道 = 历史已探索）
↓ 投影到世界地图
↓ 采样颜色混合（黑色/灰色/正常颜色）
```

### 2.1 迷雾纹理管理器

```csharp
using Unity.Collections;
using Unity.Jobs;
using Unity.Burst;
using UnityEngine;

/// <summary>
/// 战场迷雾系统核心
/// </summary>
public class FogOfWarSystem : MonoBehaviour
{
    private static FogOfWarSystem instance;
    public static FogOfWarSystem Instance => instance;

    [Header("地图配置")]
    [SerializeField] private Vector2 mapSize = new Vector2(200, 200);     // 地图尺寸（米）
    [SerializeField] private int fogTextureResolution = 512;               // 迷雾纹理分辨率
    [SerializeField] private float updateInterval = 0.1f;                  // 更新间隔（秒）
    
    [Header("渲染")]
    [SerializeField] private Material fogMaterial;                         // 迷雾覆盖材质
    [SerializeField] private Color unexploredColor = new Color(0, 0, 0, 1f);      // 未探索
    [SerializeField] private Color exploredColor = new Color(0, 0, 0, 0.65f);     // 已探索未可见
    [SerializeField] private float fogBlurRadius = 3f;                     // 迷雾边缘模糊

    // 迷雾纹理数据
    private Texture2D fogTexture;
    private NativeArray<byte> fogBuffer;     // R=当前可见, G=历史已探索
    
    // 视野单位列表
    private List<FogVisionUnit> visionUnits = new List<FogVisionUnit>();
    
    // 地图射线遮挡（使用静态遮挡图）
    private NativeArray<byte> occlusionMap;
    
    private float updateTimer;
    private int texWidth, texHeight;

    void Awake()
    {
        instance = this;
        
        texWidth = fogTextureResolution;
        texHeight = fogTextureResolution;
        
        // 初始化迷雾纹理（全黑）
        fogTexture = new Texture2D(texWidth, texHeight, TextureFormat.RG16, false);
        fogTexture.filterMode = FilterMode.Bilinear;
        fogTexture.wrapMode = TextureWrapMode.Clamp;
        
        // 初始化Buffer（2字节：R=当前可见 G=已探索）
        fogBuffer = new NativeArray<byte>(texWidth * texHeight * 2, Allocator.Persistent);
        occlusionMap = new NativeArray<byte>(texWidth * texHeight, Allocator.Persistent);
        
        // 全部初始化为未探索
        for (int i = 0; i < fogBuffer.Length; i++)
            fogBuffer[i] = 0;
        
        // 传递纹理给迷雾材质
        if (fogMaterial != null)
        {
            fogMaterial.SetTexture("_FogTexture", fogTexture);
            fogMaterial.SetVector("_MapBounds", 
                new Vector4(-mapSize.x/2, -mapSize.y/2, mapSize.x/2, mapSize.y/2));
        }
        
        // 烘焙遮挡图
        BakeOcclusionMap();
    }

    void Update()
    {
        updateTimer -= Time.deltaTime;
        if (updateTimer > 0) return;
        updateTimer = updateInterval;
        
        UpdateFog();
    }

    void UpdateFog()
    {
        // 1. 清除当前帧的可见性（R通道）
        ClearCurrentVisibility();
        
        // 2. 为每个视野单位计算可见区域
        foreach (var unit in visionUnits)
        {
            if (unit == null || !unit.IsActive) continue;
            CalculateVision(unit);
        }
        
        // 3. 应用模糊（平滑迷雾边缘）
        // ApplyBlur(); // 可选，性能开销较大
        
        // 4. 上传纹理到GPU
        UploadFogTexture();
    }

    void ClearCurrentVisibility()
    {
        // 仅清除R通道（当前可见），保留G通道（历史探索）
        for (int i = 0; i < texWidth * texHeight; i++)
        {
            fogBuffer[i * 2] = 0; // R = 当前可见 = 0
            // fogBuffer[i * 2 + 1] 不清除（历史探索保留）
        }
    }

    void CalculateVision(FogVisionUnit unit)
    {
        Vector2Int center = WorldToFogTexCoord(unit.transform.position);
        int radius = WorldDistanceToTexels(unit.VisionRange);
        
        // 简单圆形视野
        int radiusSq = radius * radius;
        
        for (int dy = -radius; dy <= radius; dy++)
        for (int dx = -radius; dx <= radius; dx++)
        {
            if (dx * dx + dy * dy > radiusSq) continue;
            
            int tx = center.x + dx;
            int ty = center.y + dy;
            
            if (tx < 0 || tx >= texWidth || ty < 0 || ty >= texHeight) continue;
            
            // 视线检测（检查遮挡图）
            if (!HasLineOfSight(center, new Vector2Int(tx, ty))) continue;
            
            int idx = ty * texWidth + tx;
            fogBuffer[idx * 2] = 255;       // R = 当前可见
            fogBuffer[idx * 2 + 1] = 255;   // G = 已探索
        }
    }

    /// <summary>
    /// Bresenham 直线视线检测
    /// </summary>
    bool HasLineOfSight(Vector2Int from, Vector2Int to)
    {
        int x0 = from.x, y0 = from.y;
        int x1 = to.x, y1 = to.y;
        
        int dx = Mathf.Abs(x1 - x0);
        int dy = Mathf.Abs(y1 - y0);
        int sx = x0 < x1 ? 1 : -1;
        int sy = y0 < y1 ? 1 : -1;
        int err = dx - dy;
        
        while (x0 != x1 || y0 != y1)
        {
            int idx = y0 * texWidth + x0;
            if (idx >= 0 && idx < occlusionMap.Length && occlusionMap[idx] > 0)
                return false; // 被遮挡
            
            int e2 = 2 * err;
            if (e2 > -dy) { err -= dy; x0 += sx; }
            if (e2 < dx) { err += dx; y0 += sy; }
        }
        
        return true;
    }

    void UploadFogTexture()
    {
        // NativeArray 数据直接写入Texture2D
        var data = fogTexture.GetRawTextureData<byte>();
        NativeArray<byte>.Copy(fogBuffer, data);
        fogTexture.Apply(updateMipmaps: false, makeNoLongerReadable: false);
    }

    /// <summary>
    /// 烘焙遮挡图（静态障碍物）
    /// </summary>
    void BakeOcclusionMap()
    {
        // 从场景中的遮挡物建立遮挡图
        var obstacles = FindObjectsOfType<FogOccluder>();
        
        foreach (var obs in obstacles)
        {
            // 将障碍物投影到迷雾纹理坐标
            var bounds = obs.GetComponent<Collider>().bounds;
            Vector2Int min = WorldToFogTexCoord(bounds.min);
            Vector2Int max = WorldToFogTexCoord(bounds.max);
            
            for (int y = min.y; y <= max.y; y++)
            for (int x = min.x; x <= max.x; x++)
            {
                if (x < 0 || x >= texWidth || y < 0 || y >= texHeight) continue;
                occlusionMap[y * texWidth + x] = 255;
            }
        }
    }

    // ============ 辅助方法 ============

    public Vector2Int WorldToFogTexCoord(Vector3 worldPos)
    {
        float nx = (worldPos.x + mapSize.x / 2f) / mapSize.x;
        float ny = (worldPos.z + mapSize.y / 2f) / mapSize.y;
        return new Vector2Int(
            Mathf.Clamp(Mathf.FloorToInt(nx * texWidth), 0, texWidth - 1),
            Mathf.Clamp(Mathf.FloorToInt(ny * texHeight), 0, texHeight - 1)
        );
    }

    int WorldDistanceToTexels(float worldDist)
    {
        return Mathf.RoundToInt(worldDist / mapSize.x * texWidth);
    }

    /// <summary>
    /// 检查世界坐标点当前是否可见
    /// </summary>
    public bool IsVisible(Vector3 worldPos)
    {
        var coord = WorldToFogTexCoord(worldPos);
        int idx = coord.y * texWidth + coord.x;
        return idx >= 0 && idx < fogBuffer.Length / 2 && fogBuffer[idx * 2] > 128;
    }

    /// <summary>
    /// 检查是否已被探索（灰雾区域）
    /// </summary>
    public bool IsExplored(Vector3 worldPos)
    {
        var coord = WorldToFogTexCoord(worldPos);
        int idx = coord.y * texWidth + coord.x;
        return idx >= 0 && idx < fogBuffer.Length / 2 && fogBuffer[idx * 2 + 1] > 128;
    }

    public void RegisterVisionUnit(FogVisionUnit unit) => visionUnits.Add(unit);
    public void UnregisterVisionUnit(FogVisionUnit unit) => visionUnits.Remove(unit);

    void OnDestroy()
    {
        fogBuffer.Dispose();
        occlusionMap.Dispose();
    }
}
```

## 三、迷雾 Shader（URP）

```hlsl
// FogOfWar.shader - URP 迷雾覆盖着色器
Shader "Custom/FogOfWar"
{
    Properties
    {
        _FogTexture ("Fog Texture", 2D) = "black" {}
        _MapBounds ("Map Bounds (xmin, zmin, xmax, zmax)", Vector) = (-100, -100, 100, 100)
        _UnexploredColor ("Unexplored Color", Color) = (0,0,0,1)
        _ExploredAlpha ("Explored Alpha (Fog Density)", Range(0,1)) = 0.65
        _EdgeBlur ("Edge Blur", Range(0, 0.05)) = 0.01
    }
    
    SubShader
    {
        Tags { "RenderType"="Transparent" "Queue"="Transparent+100" }
        Blend SrcAlpha OneMinusSrcAlpha
        ZWrite Off
        ZTest Always
        
        Pass
        {
            HLSLPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Core.hlsl"
            
            struct Attributes
            {
                float4 positionOS : POSITION;
                float2 uv : TEXCOORD0;
            };
            
            struct Varyings
            {
                float4 positionHCS : SV_POSITION;
                float3 worldPos : TEXCOORD0;
            };
            
            TEXTURE2D(_FogTexture);
            SAMPLER(sampler_FogTexture);
            
            float4 _MapBounds;
            float4 _UnexploredColor;
            float _ExploredAlpha;
            float _EdgeBlur;
            
            Varyings vert(Attributes IN)
            {
                Varyings OUT;
                VertexPositionInputs posInputs = GetVertexPositionInputs(IN.positionOS.xyz);
                OUT.positionHCS = posInputs.positionCS;
                OUT.worldPos = posInputs.positionWS;
                return OUT;
            }
            
            half4 frag(Varyings IN) : SV_Target
            {
                // 世界坐标转迷雾纹理UV
                float2 fogUV = float2(
                    (IN.worldPos.x - _MapBounds.x) / (_MapBounds.z - _MapBounds.x),
                    (IN.worldPos.z - _MapBounds.y) / (_MapBounds.w - _MapBounds.y)
                );
                
                // 超出地图范围显示全黑
                if (any(fogUV < 0) || any(fogUV > 1))
                    return _UnexploredColor;
                
                float2 fogData = SAMPLE_TEXTURE2D(_FogTexture, sampler_FogTexture, fogUV).rg;
                float currentVisible = fogData.r;  // 当前可见
                float explored = fogData.g;        // 历史探索
                
                // 边缘平滑
                currentVisible = smoothstep(0.3 - _EdgeBlur, 0.3 + _EdgeBlur, currentVisible);
                explored = smoothstep(0.3 - _EdgeBlur, 0.3 + _EdgeBlur, explored);
                
                half4 color = _UnexploredColor;
                
                if (currentVisible > 0.5)
                {
                    // 当前可见：完全透明
                    color.a = 0;
                }
                else if (explored > 0.5)
                {
                    // 已探索未可见：半透明灰雾
                    color = half4(0, 0, 0, _ExploredAlpha);
                }
                else
                {
                    // 未探索：完全黑色
                    color = _UnexploredColor;
                }
                
                return color;
            }
            ENDHLSL
        }
    }
}
```

## 四、灌木丛隐藏机制

```csharp
/// <summary>
/// 灌木丛隐藏系统（MOBA特有）
/// 在灌木丛中的单位只有敌方也在灌木丛中才能看到
/// </summary>
public class BushSystem : MonoBehaviour
{
    private List<BushZone> bushZones = new List<BushZone>();

    public bool IsUnitVisible(GameObject viewer, GameObject target)
    {
        // 目标是否在灌木丛中
        var targetBush = GetBushContaining(target);
        if (targetBush == null) return true; // 不在灌木丛，直接可见
        
        // 观察者是否在同一灌木丛
        var viewerBush = GetBushContaining(viewer);
        if (viewerBush == targetBush) return true; // 同一灌木丛可见
        
        // 目标在灌木丛，观察者不在同一灌木丛 -> 不可见
        return false;
    }

    BushZone GetBushContaining(GameObject obj)
    {
        foreach (var bush in bushZones)
        {
            if (bush.ContainsObject(obj)) return bush;
        }
        return null;
    }
}

public class BushZone : MonoBehaviour
{
    private Collider zoneCollider;
    private HashSet<GameObject> objectsInBush = new HashSet<GameObject>();

    void Awake() => zoneCollider = GetComponent<Collider>();

    void OnTriggerEnter(Collider other) => objectsInBush.Add(other.gameObject);
    void OnTriggerExit(Collider other) => objectsInBush.Remove(other.gameObject);

    public bool ContainsObject(GameObject obj) => objectsInBush.Contains(obj);
}
```

## 五、性能优化策略

| 优化方向 | 方案 | 收益 |
|----------|------|------|
| 更新频率 | 固定间隔（0.1s）而非每帧 | 节省90%计算 |
| Job System | NativeArray + Burst并行计算 | 多核利用，提升5-10倍 |
| 粗粒度遮挡 | 先AABB测试，再精确射线 | 减少Bresenham调用次数 |
| 脏标记 | 单位未移动时跳过计算 | 静止单位不重算 |
| 分层迷雾 | 将地图分为多个区块，只更新有单位的区块 | 大地图的关键优化 |
| 纹理压缩 | 使用 RG16 纹理格式 | 内存减半 |

战场迷雾是 RTS/MOBA 游戏的核心玩法机制，其实现需要在精度、性能和视觉效果之间取得平衡。建议在立项初期就确定迷雾方案，避免后期重构代价高昂。
