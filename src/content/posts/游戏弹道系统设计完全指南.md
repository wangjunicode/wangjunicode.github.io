## 一、弹道系统的两种主流方案

| 方案 | 原理 | 适用场景 | 优缺点 |
|------|------|----------|--------|
| **Hitscan（射线检测）** | 射出时立即判定命中 | 现代枪战、即时命中武器 | 简单高效，无飞行时间感 |
| **Projectile（物理子弹）** | 子弹对象在空中飞行 | 弓箭、火炮、RPG | 真实感强，有飞行时间 |
| **混合方案** | 快速子弹用Hitscan，慢速用Projectile | 多样化武器系统 | 最佳体验，实现复杂 |

## 二、Hitscan 弹道系统

```csharp
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// Hitscan 武器系统（即时命中）
/// </summary>
public class HitscanWeapon : MonoBehaviour
{
    [Header("武器参数")]
    [SerializeField] private float damage = 30f;
    [SerializeField] private float range = 200f;
    [SerializeField] private float fireRate = 10f;        // 射速（发/秒）
    [SerializeField] private int bulletsPerShot = 1;      // 每发子弹数（霰弹枪 > 1）
    [SerializeField] private LayerMask hitMask;
    
    [Header("精度与散布")]
    [SerializeField] private float baseSpread = 0.02f;    // 基础散布（弧度）
    [SerializeField] private float maxSpread = 0.15f;     // 最大散布
    [SerializeField] private float spreadIncrement = 0.03f; // 每发增加的散布
    [SerializeField] private float spreadRecoveryRate = 0.1f; // 散布恢复速度
    
    [Header("视觉效果")]
    [SerializeField] private ParticleSystem muzzleFlash;  // 枪口火焰
    [SerializeField] private GameObject bulletTrailPrefab; // 弹道特效预制体
    [SerializeField] private GameObject bulletHolePrefab;  // 弹孔预制体
    [SerializeField] private GameObject impactEffectPrefab; // 命中特效
    [SerializeField] private float bulletTrailDuration = 0.05f;
    
    [Header("穿透")]
    [SerializeField] private bool enablePenetration = false;
    [SerializeField] private int maxPenetrationCount = 3;  // 最大穿透层数
    [SerializeField] private float penetrationDamageFalloff = 0.7f; // 穿透伤害衰减

    private float currentSpread;
    private float lastFireTime;
    private Camera weaponCamera;
    
    // 弹孔对象池
    private ObjectPool<GameObject> bulletHolePool;

    void Awake()
    {
        weaponCamera = Camera.main;
        
        if (bulletHolePrefab != null)
        {
            bulletHolePool = new ObjectPool<GameObject>(
                () => Instantiate(bulletHolePrefab),
                go => go.SetActive(true),
                go => go.SetActive(false),
                maxSize: 50
            );
        }
    }

    void Update()
    {
        // 散布自动恢复
        currentSpread = Mathf.Max(baseSpread, 
            currentSpread - spreadRecoveryRate * Time.deltaTime);
    }

    public bool TryFire()
    {
        if (Time.time - lastFireTime < 1f / fireRate) return false;
        lastFireTime = Time.time;
        
        Fire();
        return true;
    }

    void Fire()
    {
        // 播放枪口火焰
        muzzleFlash?.Play();
        
        for (int i = 0; i < bulletsPerShot; i++)
        {
            Vector3 shootDir = CalculateShootDirection();
            PerformHitscan(shootDir, damage);
        }
        
        // 每次射击增加散布
        currentSpread = Mathf.Min(maxSpread, currentSpread + spreadIncrement);
    }

    Vector3 CalculateShootDirection()
    {
        Vector3 screenCenter = new Vector3(Screen.width / 2f, Screen.height / 2f, 0);
        Ray aimRay = weaponCamera.ScreenPointToRay(screenCenter);
        
        // 在散布圆内随机方向
        float spreadRadius = Random.Range(0, currentSpread);
        float spreadAngle = Random.Range(0, Mathf.PI * 2);
        
        Vector3 right = weaponCamera.transform.right;
        Vector3 up = weaponCamera.transform.up;
        
        Vector3 spreadOffset = right * Mathf.Cos(spreadAngle) * spreadRadius
            + up * Mathf.Sin(spreadAngle) * spreadRadius;
        
        return (aimRay.direction + spreadOffset).normalized;
    }

    void PerformHitscan(Vector3 direction, float currentDamage)
    {
        Vector3 origin = weaponCamera.transform.position;
        
        if (!enablePenetration)
        {
            // 单次射线检测
            if (Physics.Raycast(origin, direction, out RaycastHit hit, range, hitMask))
            {
                ProcessHit(hit, origin, currentDamage);
                SpawnBulletTrail(origin, hit.point);
            }
            else
            {
                // 未命中，弹道飞向最远点
                SpawnBulletTrail(origin, origin + direction * range);
            }
        }
        else
        {
            // 穿透射线检测
            PerformPenetratingShot(origin, direction, currentDamage);
        }
    }

    void PerformPenetratingShot(Vector3 origin, Vector3 direction, float damage)
    {
        // RaycastAll 返回所有命中，按距离排序
        RaycastHit[] hits = Physics.RaycastAll(origin, direction, range, hitMask);
        System.Array.Sort(hits, (a, b) => a.distance.CompareTo(b.distance));
        
        Vector3 lastPoint = origin + direction * range;
        int penetrationCount = 0;
        float currentDmg = damage;
        
        foreach (var hit in hits)
        {
            if (penetrationCount >= maxPenetrationCount) break;
            
            ProcessHit(hit, origin, currentDmg);
            lastPoint = hit.point;
            
            currentDmg *= penetrationDamageFalloff;
            penetrationCount++;
        }
        
        SpawnBulletTrail(origin, lastPoint);
    }

    void ProcessHit(RaycastHit hit, Vector3 origin, float damage)
    {
        // 应用伤害
        var damageable = hit.collider.GetComponentInParent<IDamageable>();
        if (damageable != null)
        {
            float distanceFactor = 1f - (hit.distance / range) * 0.3f; // 远距离轻微减伤
            damageable.TakeDamage(new DamageInfo
            {
                Amount = damage * distanceFactor,
                Source = gameObject,
                HitPoint = hit.point,
                HitNormal = hit.normal,
                DamageType = DamageType.Bullet
            });
            
            // 命中生物不显示弹孔
            SpawnImpactEffect(hit.point, hit.normal);
        }
        else
        {
            // 命中环境，显示弹孔
            SpawnBulletHole(hit);
            SpawnImpactEffect(hit.point, hit.normal);
        }
    }

    void SpawnBulletHole(RaycastHit hit)
    {
        if (bulletHolePool == null) return;
        
        var hole = bulletHolePool.Get();
        hole.transform.position = hit.point + hit.normal * 0.001f;
        hole.transform.rotation = Quaternion.LookRotation(-hit.normal);
        hole.transform.SetParent(hit.collider.transform);
        
        // 一段时间后回收
        StartCoroutine(ReturnToPool(hole, 30f));
    }

    void SpawnBulletTrail(Vector3 start, Vector3 end)
    {
        if (bulletTrailPrefab == null) return;
        
        var trail = Instantiate(bulletTrailPrefab, start, Quaternion.identity);
        var lr = trail.GetComponent<LineRenderer>();
        if (lr != null)
        {
            lr.SetPosition(0, start);
            lr.SetPosition(1, end);
        }
        Destroy(trail, bulletTrailDuration);
    }

    void SpawnImpactEffect(Vector3 pos, Vector3 normal)
    {
        if (impactEffectPrefab == null) return;
        var effect = Instantiate(impactEffectPrefab, pos, Quaternion.LookRotation(normal));
        Destroy(effect, 2f);
    }

    System.Collections.IEnumerator ReturnToPool(GameObject obj, float delay)
    {
        yield return new WaitForSeconds(delay);
        bulletHolePool.Release(obj);
    }
}
```

## 三、物理子弹系统（Projectile）

```csharp
/// <summary>
/// 物理子弹（支持重力、空气阻力、弹跳）
/// </summary>
public class Projectile : MonoBehaviour
{
    [Header("飞行参数")]
    public float Speed = 50f;
    public float Gravity = 9.8f;          // 重力系数（0 = 无重力，如激光）
    public float AirResistance = 0.01f;    // 空气阻力
    public float MaxLifetime = 10f;        // 最长存活时间
    public float Damage = 50f;
    public LayerMask HitMask;
    
    [Header("弹跳")]
    public bool EnableBounce = false;
    public int MaxBounces = 3;
    public float BounceDampening = 0.6f;  // 每次弹跳速度衰减
    
    [Header("特效")]
    public GameObject HitEffectPrefab;
    public TrailRenderer Trail;
    
    private Vector3 velocity;
    private Vector3 previousPosition;
    private int bounceCount;
    private float lifetime;
    private GameObject owner;

    public void Initialize(Vector3 direction, float speed, GameObject owner, float damage)
    {
        this.owner = owner;
        this.Damage = damage;
        velocity = direction.normalized * speed;
        previousPosition = transform.position;
    }

    void Update()
    {
        float dt = Time.deltaTime;
        lifetime += dt;
        
        if (lifetime > MaxLifetime)
        {
            DestroyProjectile();
            return;
        }
        
        // 应用重力
        velocity.y -= Gravity * dt;
        
        // 应用空气阻力
        velocity *= (1f - AirResistance * dt);
        
        // 计算移动量
        Vector3 movement = velocity * dt;
        float moveDist = movement.magnitude;
        
        // 用 SphereCast 而非 Raycast，防止子弹过快穿透薄物体
        if (Physics.SphereCast(
            previousPosition,
            0.05f,
            velocity.normalized,
            out RaycastHit hit,
            moveDist,
            HitMask,
            QueryTriggerInteraction.Ignore))
        {
            OnHit(hit);
        }
        else
        {
            transform.position += movement;
            transform.rotation = Quaternion.LookRotation(velocity);
        }
        
        previousPosition = transform.position;
    }

    void OnHit(RaycastHit hit)
    {
        // 应用伤害
        var damageable = hit.collider.GetComponentInParent<IDamageable>();
        damageable?.TakeDamage(new DamageInfo
        {
            Amount = Damage,
            Source = owner,
            HitPoint = hit.point,
            HitNormal = hit.normal,
            DamageType = DamageType.Projectile
        });
        
        // 弹跳处理
        if (EnableBounce && bounceCount < MaxBounces && damageable == null)
        {
            bounceCount++;
            velocity = Vector3.Reflect(velocity, hit.normal) * BounceDampening;
            transform.position = hit.point + hit.normal * 0.01f;
            return;
        }
        
        // 命中特效
        if (HitEffectPrefab != null)
        {
            var effect = Instantiate(HitEffectPrefab, hit.point, 
                Quaternion.LookRotation(hit.normal));
            Destroy(effect, 3f);
        }
        
        DestroyProjectile();
    }

    void DestroyProjectile()
    {
        // 如果使用对象池，则归还而非销毁
        ProjectilePool.Instance?.ReturnProjectile(this);
        // 或者：Destroy(gameObject);
    }
}

/// <summary>
/// 抛物线预测器（用于弓箭瞄准辅助）
/// </summary>
public class ProjectileTrajectoryPredictor : MonoBehaviour
{
    [SerializeField] private LineRenderer trajectoryLine;
    [SerializeField] private int samplePoints = 30;
    [SerializeField] private float timeStep = 0.1f;

    public void ShowTrajectory(Vector3 startPos, Vector3 initialVelocity, 
        float gravity, float maxTime)
    {
        var points = new Vector3[samplePoints];
        
        for (int i = 0; i < samplePoints; i++)
        {
            float t = i * (maxTime / samplePoints);
            float x = startPos.x + initialVelocity.x * t;
            float y = startPos.y + initialVelocity.y * t - 0.5f * gravity * t * t;
            float z = startPos.z + initialVelocity.z * t;
            
            points[i] = new Vector3(x, y, z);
            
            // 检测碰撞，提前终止
            if (i > 0 && Physics.Linecast(points[i-1], points[i]))
            {
                // 截断到碰撞点
                System.Array.Resize(ref points, i + 1);
                break;
            }
        }
        
        trajectoryLine.positionCount = points.Length;
        trajectoryLine.SetPositions(points);
    }

    public void HideTrajectory()
    {
        trajectoryLine.positionCount = 0;
    }
}
```

## 四、子弹对象池

```csharp
/// <summary>
/// 子弹对象池管理器
/// </summary>
public class ProjectilePool : MonoBehaviour
{
    private static ProjectilePool instance;
    public static ProjectilePool Instance => instance;

    [SerializeField] private Projectile projectilePrefab;
    [SerializeField] private int initialPoolSize = 50;
    [SerializeField] private int maxPoolSize = 200;

    private Queue<Projectile> pool = new Queue<Projectile>();

    void Awake()
    {
        instance = this;
        
        // 预创建
        for (int i = 0; i < initialPoolSize; i++)
        {
            var proj = CreateProjectile();
            proj.gameObject.SetActive(false);
            pool.Enqueue(proj);
        }
    }

    public Projectile GetProjectile(Vector3 pos, Vector3 dir, float speed, 
        GameObject owner, float damage)
    {
        Projectile proj;
        
        if (pool.Count > 0)
        {
            proj = pool.Dequeue();
        }
        else if (pool.Count + ActiveCount < maxPoolSize)
        {
            proj = CreateProjectile();
        }
        else
        {
            return null; // 达到上限
        }
        
        proj.transform.position = pos;
        proj.transform.rotation = Quaternion.LookRotation(dir);
        proj.gameObject.SetActive(true);
        proj.Initialize(dir, speed, owner, damage);
        ActiveCount++;
        
        return proj;
    }

    public void ReturnProjectile(Projectile proj)
    {
        proj.gameObject.SetActive(false);
        pool.Enqueue(proj);
        ActiveCount--;
    }

    Projectile CreateProjectile()
    {
        var go = Instantiate(projectilePrefab, transform);
        go.gameObject.SetActive(false);
        return go;
    }

    public int ActiveCount { get; private set; }
}
```

## 五、网络同步方案

```csharp
/// <summary>
/// 网络子弹同步策略
/// </summary>
public class NetworkBulletSync
{
    // 策略一：服务端权威射线检测
    // 客户端发送：射击方向、开火时间
    // 服务端执行：Hitscan 并广播命中结果
    // 优点：防作弊；缺点：高延迟感

    // 策略二：客户端预测 + 服务端验证
    // 客户端：立即显示子弹飞行效果
    // 服务端：验证命中，拒绝时通知客户端回滚
    // 优点：响应流畅；缺点：实现复杂

    // 策略三：服务端广播，客户端重播
    // 服务端：执行检测并广播子弹生成事件（含初始位置/速度/时间戳）
    // 客户端：收到后从时间戳开始重播子弹飞行
    // 适用于：慢速子弹（箭矢、炮弹）
}
```

## 六、精度与武器手感调优

```csharp
/// <summary>
/// 武器精度模型（影响散布的因素）
/// </summary>
[System.Serializable]
public class WeaponAccuracyModel
{
    [Header("基础精度")]
    public float BaseSpread = 0.02f;        // 静止精准射击
    
    [Header("状态惩罚")]
    public float MovingSpreadAdd = 0.05f;   // 移动时增加散布
    public float JumpingSpreadAdd = 0.15f;  // 跳跃时增加散布
    public float CrouchSpreadMul = 0.6f;    // 蹲伏时散布乘以此值（减小）
    
    [Header("连发惩罚")]
    public float SpreadPerShot = 0.02f;     // 每发后散布增加量
    public float MaxSpread = 0.12f;         // 散布上限
    public float SpreadRecovery = 0.08f;    // 每秒散布恢复量
    
    [Header("瞄准状态")]
    public float AimSpreadMul = 0.3f;       // 腰瞄时散布乘以此值（ADS效果）
    
    public float CalculateCurrentSpread(bool isMoving, bool isJumping, 
        bool isCrouching, bool isAiming, float continuousSpread)
    {
        float spread = BaseSpread;
        
        if (isMoving) spread += MovingSpreadAdd;
        if (isJumping) spread += JumpingSpreadAdd;
        if (isCrouching) spread *= CrouchSpreadMul;
        if (isAiming) spread *= AimSpreadMul;
        
        spread += continuousSpread;
        spread = Mathf.Min(spread, MaxSpread);
        
        return spread;
    }
}
```

## 七、设计要点汇总

| 功能 | 推荐方案 | 注意事项 |
|------|----------|----------|
| 近现代枪 | Hitscan + 弹道特效 | SphereCast防穿透，弹孔用对象池 |
| 弓箭/炮弹 | Projectile + 轨迹预测 | SphereCast防穿透，重力系数可配置 |
| 霰弹枪 | 多发Hitscan 聚合 | 散布角度配置，随距离衰减 |
| 狙击枪 | Hitscan + 远距离衰减 | 距离超过一定值减伤 |
| 穿透子弹 | RaycastAll排序处理 | 每层穿透伤害衰减 |
| 网络同步 | 客户端预测+服务端校验 | 高速子弹推荐Hitscan，慢速用广播重播 |
