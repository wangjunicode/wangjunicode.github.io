## 一、排行榜数据结构

```csharp
using System;
using System.Collections.Generic;
using UnityEngine;

/// <summary>
/// 排行榜类型
/// </summary>
public enum LeaderboardType
{
    Global,     // 全球总榜
    Friends,    // 好友榜
    Weekly,     // 每周榜
    Season,     // 赛季榜
    GuildRank,  // 公会排名
}

/// <summary>
/// 排行榜条目
/// </summary>
[Serializable]
public class LeaderboardEntry
{
    public int Rank;            // 名次
    public string PlayerId;
    public string PlayerName;
    public string AvatarUrl;
    public long Score;          // 分数
    public string ExtraData;    // 附加数据（角色名/战力等）
    public bool IsCurrentPlayer; // 是否是当前玩家
    public long UpdateTime;
}

/// <summary>
/// 排行榜数据
/// </summary>
[Serializable]
public class LeaderboardData
{
    public LeaderboardType Type;
    public List<LeaderboardEntry> TopEntries;    // 前N名
    public LeaderboardEntry CurrentPlayerEntry; // 当前玩家排名（可能在前N名之外）
    public List<LeaderboardEntry> NearbyEntries; // 玩家附近排名（上下各2名）
    public long LastUpdateTime;
    public long NextResetTime;   // 下次重置时间（周期榜）
    public int TotalPlayers;
}
```

## 二、排行榜管理器

```csharp
/// <summary>
/// 排行榜管理器（客户端）
/// </summary>
public class LeaderboardManager : MonoBehaviour
{
    private static LeaderboardManager instance;
    public static LeaderboardManager Instance => instance;

    [Header("缓存配置")]
    [SerializeField] private float cacheLifetime = 300f; // 缓存有效期（秒）
    
    private Dictionary<LeaderboardType, LeaderboardData> cache 
        = new Dictionary<LeaderboardType, LeaderboardData>();
    private Dictionary<LeaderboardType, float> cacheTimestamps 
        = new Dictionary<LeaderboardType, float>();
    
    public event Action<LeaderboardType, LeaderboardData> OnLeaderboardUpdated;

    void Awake() { instance = this; }

    /// <summary>
    /// 获取排行榜数据
    /// </summary>
    public async System.Threading.Tasks.Task<LeaderboardData> GetLeaderboard(
        LeaderboardType type, bool forceRefresh = false)
    {
        // 检查缓存
        if (!forceRefresh && IsCacheValid(type))
        {
            return cache[type];
        }
        
        // 从服务端拉取
        try
        {
            var data = await FetchFromServer(type);
            if (data != null)
            {
                cache[type] = data;
                cacheTimestamps[type] = Time.time;
                OnLeaderboardUpdated?.Invoke(type, data);
            }
            return data;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Leaderboard] 拉取失败: {e.Message}");
            
            // 返回缓存数据（即使过期）
            return cache.TryGetValue(type, out var cached) ? cached : null;
        }
    }

    /// <summary>
    /// 提交分数
    /// </summary>
    public async System.Threading.Tasks.Task<SubmitScoreResult> SubmitScore(
        LeaderboardType type, long score, string extraData = "")
    {
        try
        {
            var result = await PostScoreToServer(type, score, extraData);
            
            if (result.Success)
            {
                // 本地更新当前玩家排名
                if (cache.TryGetValue(type, out var data) && data.CurrentPlayerEntry != null)
                {
                    data.CurrentPlayerEntry.Score = score;
                    data.CurrentPlayerEntry.Rank = result.NewRank;
                }
                
                // 使缓存失效，下次刷新获取最新
                cacheTimestamps.Remove(type);
            }
            
            return result;
        }
        catch (Exception e)
        {
            Debug.LogError($"[Leaderboard] 提交分数失败: {e.Message}");
            return new SubmitScoreResult { Success = false };
        }
    }

    bool IsCacheValid(LeaderboardType type)
    {
        if (!cacheTimestamps.TryGetValue(type, out float timestamp)) return false;
        return Time.time - timestamp < cacheLifetime;
    }

    async System.Threading.Tasks.Task<LeaderboardData> FetchFromServer(LeaderboardType type)
    {
        await System.Threading.Tasks.Task.Delay(100); // 模拟网络
        return GenerateMockData(type); // 示例数据
    }

    async System.Threading.Tasks.Task<SubmitScoreResult> PostScoreToServer(
        LeaderboardType type, long score, string extra)
    {
        await System.Threading.Tasks.Task.Delay(50);
        return new SubmitScoreResult { Success = true, NewRank = 42 };
    }

    LeaderboardData GenerateMockData(LeaderboardType type)
    {
        var data = new LeaderboardData
        {
            Type = type,
            TopEntries = new List<LeaderboardEntry>(),
            TotalPlayers = 10000,
            LastUpdateTime = DateTimeOffset.UtcNow.ToUnixTimeSeconds()
        };
        
        for (int i = 0; i < 100; i++)
        {
            data.TopEntries.Add(new LeaderboardEntry
            {
                Rank = i + 1,
                PlayerName = $"Player_{i + 1}",
                Score = 100000 - i * 100,
                IsCurrentPlayer = (i == 41) // 模拟当前玩家第42名
            });
        }
        
        return data;
    }
}

[Serializable]
public class SubmitScoreResult
{
    public bool Success;
    public int NewRank;
    public int PreviousRank;
    public bool IsNewHighScore;
}
```

## 三、排行榜UI（无限滚动列表）

```csharp
/// <summary>
/// 排行榜UI（虚拟列表，支持大量数据高性能滚动）
/// </summary>
public class LeaderboardUI : MonoBehaviour
{
    [SerializeField] private UnityEngine.UI.ScrollRect scrollRect;
    [SerializeField] private GameObject entryPrefab;
    [SerializeField] private int visibleItemCount = 15;  // 同时显示的条目数
    [SerializeField] private float itemHeight = 80f;
    
    [Header("特殊样式")]
    [SerializeField] private Color topThreeColor = new Color(1f, 0.84f, 0f);
    [SerializeField] private Color currentPlayerColor = new Color(0.3f, 0.7f, 1f);

    private List<LeaderboardEntry> allEntries = new List<LeaderboardEntry>();
    private List<GameObject> pooledItems = new List<GameObject>();
    private int firstVisibleIndex;

    public async void LoadLeaderboard(LeaderboardType type)
    {
        var data = await LeaderboardManager.Instance.GetLeaderboard(type);
        if (data == null) return;
        
        allEntries = data.TopEntries;
        
        // 初始化列表高度
        var content = scrollRect.content;
        content.sizeDelta = new Vector2(content.sizeDelta.x, 
            allEntries.Count * itemHeight);
        
        // 创建可见条目池
        CreateItemPool();
        UpdateVisibleItems();
    }

    void CreateItemPool()
    {
        foreach (var item in pooledItems)
            Destroy(item);
        pooledItems.Clear();
        
        for (int i = 0; i < visibleItemCount + 2; i++)
        {
            var item = Instantiate(entryPrefab, scrollRect.content);
            pooledItems.Add(item);
        }
    }

    void Update()
    {
        // 检测滚动，更新可见条目
        int newFirstIndex = Mathf.Max(0, 
            Mathf.FloorToInt(scrollRect.content.anchoredPosition.y / itemHeight));
        
        if (newFirstIndex != firstVisibleIndex)
        {
            firstVisibleIndex = newFirstIndex;
            UpdateVisibleItems();
        }
    }

    void UpdateVisibleItems()
    {
        for (int i = 0; i < pooledItems.Count; i++)
        {
            int dataIndex = firstVisibleIndex + i;
            
            if (dataIndex >= allEntries.Count)
            {
                pooledItems[i].SetActive(false);
                continue;
            }
            
            pooledItems[i].SetActive(true);
            
            // 设置位置
            var rt = pooledItems[i].GetComponent<RectTransform>();
            rt.anchoredPosition = new Vector2(0, -dataIndex * itemHeight);
            
            // 填充数据
            FillEntryUI(pooledItems[i], allEntries[dataIndex]);
        }
    }

    void FillEntryUI(GameObject item, LeaderboardEntry entry)
    {
        var texts = item.GetComponentsInChildren<UnityEngine.UI.Text>();
        if (texts.Length >= 3)
        {
            texts[0].text = $"#{entry.Rank}";
            texts[1].text = entry.PlayerName;
            texts[2].text = FormatScore(entry.Score);
        }
        
        // 样式
        var bg = item.GetComponent<UnityEngine.UI.Image>();
        if (bg != null)
        {
            if (entry.IsCurrentPlayer)
                bg.color = currentPlayerColor;
            else if (entry.Rank <= 3)
                bg.color = topThreeColor;
            else
                bg.color = Color.white;
        }
    }

    string FormatScore(long score)
    {
        if (score >= 1000000) return $"{score / 1000000.0:F1}M";
        if (score >= 1000) return $"{score / 1000.0:F1}K";
        return score.ToString();
    }
}
```

## 四、排行榜系统设计要点

| 要点 | 方案 |
|------|------|
| 服务端权威 | 分数只在服务端计算，客户端无法伪造 |
| 缓存策略 | 前100名本地缓存，减少请求频率 |
| 周期重置 | 服务端定时任务重置+归档历史数据 |
| 附近排名 | 查询玩家上下各2名，保持竞争感 |
| 防作弊 | 分数突变检测，异常分数人工审核 |
| 虚拟列表 | 只渲染可见条目，支持数万条目无卡顿 |
